  65 static void
  66 setupvariables(int verbose)
  67 {
  68     const char *p;
  69 
  70     if ((p = getenv("GTAGSOBJDIRPREFIX")) != NULL ||
  71         (p = getenv("MAKEOBJDIRPREFIX")) != NULL)
  72     {
  73         gtagsobjdirprefix = p;
  74         if (verbose)
  75             fprintf(stderr, "GTAGSOBJDIRPREFIX is set to '%s'.\n", p);
  76     } else {
  77         gtagsobjdirprefix = "/usr/obj";
  78     }
  79     if ((p = getenv("GTAGSOBJDIR")) != NULL ||
  80         (p = getenv("MAKEOBJDIR")) != NULL)
  81     {
  82         gtagsobjdir = p;
  83         if (verbose)
  84             fprintf(stderr, "GTAGSOBJDIR is set to '%s'.\n", p);
  85     } else {
  86         gtagsobjdir = "obj";
  87     }
  88 }
  89 /**
  90  * getobjdir: get objdir if it exists.
  91  *
  92  *  @param[in]  candidate candidate root directory
  93  *  @param[in]  verbose verbose mode 1: on, 0: off
  94  *  @return     objdir(NULL: not found)
  95  */
  96 char *
  97 getobjdir(const char *candidate, int verbose)
  98 {
  99     static char path[MAXPATHLEN];
 100 
 101     /*
 102      * setup gtagsobjdir and gtagsobjdirprefix (only first time).
 103      */
 104     if (gtagsobjdir == NULL)
 105         setupvariables(0);
 106     snprintf(path, sizeof(path), "%s/%s", candidate, gtagsobjdir);
 107     if (test("d", path)) {
 108         if (!test("drw", path))
 109             die("Found objdir '%s', but you don't have read/write permission for it.", path);
 110         if (verbose)
 111             fprintf(stderr, "Using objdir '%s'.\n", path);
 112         return path;
 113     }
 114 #if !defined(_WIN32) && !defined(__DJGPP__)
 115     if (test("d", gtagsobjdirprefix)) {
 116         snprintf(path, sizeof(path), "%s%s", gtagsobjdirprefix, candidate);
 117         if (test("d", path)) {
 118             if (!test("drw", path))
 119                 die("Found objdir '%s', but you don't have read/write permission for it.", path);
 120             if (verbose)
 121                 fprintf(stderr, "Using objdir '%s'.\n", path);
 122             return path;
 123         }
 124         if (makedirectories(gtagsobjdirprefix, candidate + 1, verbose) < 0)
 125             die("Found the base for objdir '%s', but you cannot create new directory in it.", path);
 126         if (verbose)
 127             fprintf(stderr, "Using objdir '%s'.\n", path);
 128         return path;
 129     }
 130 #endif
 131     return NULL;
 132 }
 133 /**
 134  * gtagsexist: test whether GTAGS's existence.
 135  *
 136  *  @param[in]  candidate candidate root directory
 137  *  @param[out] dbpath  directory which "GTAGS" exist
 138  *  @param[in]  size    size of dbpath buffer
 139  *  @param[in]  verbose verbose mode 1: on, 0: off
 140  *  @return     0: not found, 1: found
 141  *
 142  * Gtagsexist locate "GTAGS" file in "$candidate/", "$candidate/obj/" and
 143  * "/usr/obj/$candidate/" in this order by default.
 144  * This behavior is same with BSD make(1)'s one.
 145  */
 146 int
 147 gtagsexist(const char *candidate, char *dbpath, int size, int verbose)
 148 {
 149     char path[MAXPATHLEN];
 150     const char *candidate_without_slash;
 151 
 152     /*
 153      * setup gtagsobjdir and gtagsobjdirprefix (only first time).
 154      */
 155     if (gtagsobjdir == NULL)
 156         setupvariables(verbose);
 157 
 158     if (strcmp(candidate, "/") == 0)
 159         candidate_without_slash = "";
 160     else
 161         candidate_without_slash = candidate;
 162     snprintf(path, sizeof(path), "%s/%s", candidate_without_slash, dbname(GTAGS));
 163     if (verbose)
 164         fprintf(stderr, "checking %s\n", path);
 165     if (test("fr", path)) {
 166         if (verbose)
 167             fprintf(stderr, "GTAGS found at '%s'.\n", path);
 168         snprintf(dbpath, size, "%s", candidate);
 169         return 1;
 170     }
 171     snprintf(path, sizeof(path),
 172         "%s/%s/%s", candidate_without_slash, gtagsobjdir, dbname(GTAGS));
 173     if (verbose)
 174         fprintf(stderr, "checking %s\n", path);
 175     if (test("fr", path)) {
 176         if (verbose)
 177             fprintf(stderr, "GTAGS found at '%s'.\n", path);
 178         snprintf(dbpath, size, "%s/%s", candidate_without_slash, gtagsobjdir);
 179         return 1;
 180     }
 181 #if !defined(_WIN32) && !defined(__DJGPP__)
 182     snprintf(path, sizeof(path),
 183         "%s%s/%s", gtagsobjdirprefix, candidate_without_slash, dbname(GTAGS));
 184     if (verbose)
 185         fprintf(stderr, "checking %s\n", path);
 186     if (test("fr", path)) {
 187         if (verbose)
 188             fprintf(stderr, "GTAGS found at '%s'.\n", path);
 189         snprintf(dbpath, size, "%s%s", gtagsobjdirprefix, candidate_without_slash);
 190         return 1;
 191     }
 192 #endif
 193     return 0;
 194 }

 200 /*
 201  * setupdbpath: setup dbpath directory
 202  *
 203  *  @param[in]  verbose verbose mode 1: on, 0: off
 204  *  @return 0: normal, 0<: error
 205  *  Globals used (output):
 206  *      cwd:    current directory,
 207  *      root:   root of source tree,
 208  *      dbpath: directory which "GTAGS" exist,
 209  *      gtags_dbpath_error: set if status (return value) < 0
 210  *
 211  */
 212 int
 213 setupdbpath(int verbose)
 214 {
 215     struct stat sb;
 216     char *p;
 217     static char msg[1024];
 218 
 219     if (!vgetcwd(cwd, MAXPATHLEN)) {
 220         gtags_dbpath_error = "cannot get current directory.";
 221         return -1;
 222     }
 223     canonpath(cwd);
 224 
 225     if ((p = getenv("GTAGSROOT")) != NULL) {
 226         if (verbose)
 227             fprintf(stderr, "GTAGSROOT is set to '%s'.\n", p);
 228         if (!isabspath(p)) {
 229             gtags_dbpath_error = "GTAGSROOT must be an absolute path.";
 230             return -1;
 231         }
 232         if (stat(p, &sb) || !S_ISDIR(sb.st_mode)) {
 233             snprintf(msg, sizeof(msg), "directory '%s' not found.", p);
 234             gtags_dbpath_error = msg;
 235             return -1;
 236         }
 237         if (getenv("GTAGSLOGICALPATH")) {
 238             strlimcpy(root, p, sizeof(root));
 239         } else {
 240             if (realpath(p, root) == NULL) {
 241                 snprintf(msg, sizeof(msg), "cannot get real path of '%s'.", p);
 242                 gtags_dbpath_error = msg;
 243                 return -1;
 244             }
 245         }
 246         /*
 247          * GTAGSDBPATH is meaningful only when GTAGSROOT exist.
 248          */
 249         if ((p = getenv("GTAGSDBPATH")) != NULL) {
 250             if (verbose)
 251                 fprintf(stderr, "GTAGSDBPATH is set to '%s'.\n", p);
 252             if (!isabspath(p)) {
 253                 gtags_dbpath_error = "GTAGSDBPATH must be an absolute path.";
 254                 return -1;
 255             }
 256             if (stat(p, &sb) || !S_ISDIR(sb.st_mode)) {
 257                 snprintf(msg, sizeof(msg), "directory '%s' not found.", p);
 258                 gtags_dbpath_error = msg;
 259                 return -1;
 260             }
 261             strlimcpy(dbpath, getenv("GTAGSDBPATH"), MAXPATHLEN);
 262         } else {
 263             if (!gtagsexist(root, dbpath, MAXPATHLEN, verbose)) {
 264                 gtags_dbpath_error = "GTAGS not found.";
 265                 return -3;
 266             }
 267         }
 268     } else {
 269         if (verbose && getenv("GTAGSDBPATH"))
 270             fprintf(stderr, "warning: GTAGSDBPATH is ignored because GTAGSROOT is not set.\n");
 271         /*
 272          * start from current directory to '/' directory.
 273          */
 274         strlimcpy(root, cwd, MAXPATHLEN);
 275         p = root + strlen(root);
 276         while (!gtagsexist(root, dbpath, MAXPATHLEN, verbose)) {
 277             if (!strcmp(root+ROOT, "/")) {  /* reached the system's root directory */
 278                 *(root+ROOT) = '\0';
 279                 break;
 280             }
 281             while (*--p != '/' && p > (root+ROOT))
 282                 ;
 283             if (p == (root+ROOT))
 284                 p++;
 285             *p = 0;
 286         }
 287         if (*(root+ROOT) == 0) {
 288             gtags_dbpath_error = "GTAGS not found.";
 289             return -3;
 290         }
 291         /*
 292          * If file 'GTAGSROOT' found without environment variable
 293          * GTAGSDBPATH, use the value of it as GTAGSROOT.
 294          */
 295         do {
 296             FILE *fp;
 297             STRBUF *sb;
 298             const char *s, *path;
 299 
 300             path = makepath(root, "GTAGSROOT", NULL);
 301             if (!test("fr", path)) {
 302                 break;
 303             }
 304             fp = fopen(path, "r");
 305             if (fp == NULL) {
 306                 if (verbose)
 307                     fprintf(stderr, "'%s' ignored because it cannot be opened.\n", path);
 308                 break;
 309             }
 310             sb = strbuf_open(0);
 311             s = strbuf_fgets(sb, fp, STRBUF_NOCRLF);
 312             if (s == NULL) {
 313                 /* Empty file? */
 314                 if (verbose)
 315                     fprintf(stderr, "'%s' ignored; strbuf_fgets() returned NULL.\n", path);
 316                 goto ignore_lab;
 317             }
 318             if (!test("d", s)) {    /* XXX: if s == NULL, test() will use previous path used. */
 319                 if (verbose)
 320                     fprintf(stderr, "'%s' ignored because it doesn't include existent directory name.\n", path);
 321             } else {
 322                 char buf[MAXPATHLEN];
 323 
 324                 if (verbose)
 325                     fprintf(stderr, "GTAGSROOT found at '%s'.\n", path);
 326                 if (!isabspath(s))
 327                     s = realpath(makepath(root, s, NULL), buf);
 328                 strlimcpy(root, s, MAXPATHLEN);
 329             }
 330 ignore_lab:;
 331             fclose(fp);
 332             strbuf_close(sb);
 333             break;
 334         } while (0);
 335     }
 336     if (!strcmp(root+ROOT, "/"))
 337         strlimcpy(root_with_slash, root, sizeof(root_with_slash));
 338     else
 339         snprintf(root_with_slash, sizeof(root_with_slash), "%s/", root);
 340     /*
 341      * relative from the project root directory.
 342      */
 343     snprintf(relative_cwd_with_slash,
 344         sizeof(relative_cwd_with_slash), "./%s/", cwd + strlen(root) + 1);
 345     return 0;
 346 }
 347 /**
 348  * in_the_project: test whether path is in the project.
 349  *
 350  *  @param[in]  path    target file or directory
 351  *  @return     0: out of the project, 1: in the project
 352  *
 353  * [Note] Please pass an absolute path name which does not include "." (current directory) or
 354  *       ".." (parent directory).
 355  */
 356 int
 357 in_the_project(const char *path)
 358 {
 359     if (!strcmp(path, root) || locatestring(path, root_with_slash, MATCH_AT_FIRST))
 360         return 1;
 361     return 0;
 362 }
