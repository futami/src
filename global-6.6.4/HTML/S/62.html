 123 /**
 124  * printconf: print configuration data.
 125  *
 126  *  @param[in]  name    label of config data
 127  *  @return     exit code
 128  */
 129 int
 130 printconf(const char *name)
 131 {
 132     int num;
 133     int exist = 1;
 134 
 135     if (getconfn(name, &num))
 136         fprintf(stdout, "%d\n", num);
 137     else if (getconfb(name))
 138         fprintf(stdout, "1\n");
 139     else {
 140         STRBUF *sb = strbuf_open(0);
 141         if (getconfs(name, sb))
 142             fprintf(stdout, "%s\n", strbuf_value(sb));
 143         else
 144             exist = 0;
 145         strbuf_close(sb);
 146     }
 147     return exist;
 148 }

 224 int
 225 main(int argc, char **argv)
 226 {
 227     char dbpath[MAXPATHLEN];
 228     char cwd[MAXPATHLEN];
 229     STRBUF *sb = strbuf_open(0);
 230     int optchar;
 231     int option_index = 0;
 232     STATISTICS_TIME *tim;
 233 
 234     /*
 235      * Setup GTAGSCONF and GTAGSLABEL environment variable
 236      * according to the --gtagsconf and --gtagslabel option.
 237      */
 238     preparse_options(argc, argv);
 239     /*
 240      * Get the project root directory.
 241      */
 242     if (!vgetcwd(cwd, MAXPATHLEN))
 243         die("cannot get current directory.");
 244     canonpath(cwd);
 245     /*
 246      * Load configuration file.
 247      */
 248     openconf(cwd);
 249     configuration();
 250     setenv_from_config();
 251     {
 252         char *env = getenv("GTAGS_OPTIONS");
 253         if (env && *env)
 254             argv = prepend_options(&argc, argv, env);
 255     }
 256     /*
 257      * Execute gtags_hook before the jobs.
 258      */
 259     if (getconfs("gtags_hook", sb)) {
 260         char *p = serialize_options(argc, argv);
 261         set_env("GTAGS_COMMANDLINE", p);
 262         free(p);
 263         if (system(strbuf_value(sb)))
 264             fprintf(stderr, "gtags-hook failed: %s\n", strbuf_value(sb));
 265     }
 266     logging_arguments(argc, argv);
 267     while ((optchar = getopt_long(argc, argv, short_options, long_options, &option_index)) != EOF) {
 268         switch (optchar) {
 269         case 0:
 270             /* already flags set */
 271             break;
 272         case OPT_CONFIG:
 273             show_config = 1;
 274             if (optarg)
 275                 config_name = optarg;
 276             break;
 277         case OPT_GTAGSCONF:
 278         case OPT_GTAGSLABEL:
 279             /* These options are already parsed in preparse_options() */
 280             break;
 281         case OPT_SINGLE_UPDATE:
 282             iflag++;
 283             single_update = optarg;
 284             break;
 285         case OPT_ACCEPT_DOTFILES:
 286             accept_dotfiles = 1;
 287             break;
 288         case OPT_SKIP_UNREADABLE:
 289             skip_unreadable = 1;
 290             break;
 291         case OPT_GTAGSSKIP_SYMLINK:
 292             skip_symlink = SKIP_SYMLINK_FOR_ALL;
 293             if (optarg) {
 294                 if (!strcmp(optarg, "f"))
 295                     skip_symlink = SKIP_SYMLINK_FOR_FILE;
 296                 else if (!strcmp(optarg, "d"))
 297                     skip_symlink = SKIP_SYMLINK_FOR_DIR;
 298                 else if (!strcmp(optarg, "a"))
 299                     skip_symlink = SKIP_SYMLINK_FOR_ALL;
 300                 else
 301                     die("--skip-symlink: %s: unknown type.", optarg);
 302             }
 303             break;
 304         case 'c':
 305             cflag++;
 306             break;
 307         case 'd':
 308             dump_target = optarg;
 309             break;
 310         case 'f':
 311             file_list = optarg;
 312             break;
 313         case 'i':
 314             iflag++;
 315             break;
 316         case 'I':
 317             Iflag++;
 318             break;
 319         case 'o':
 320             /*
 321              * Though the -o(--omit-gsyms) was removed, this code
 322              * is left for compatibility.
 323              */
 324             break;
 325         case 'O':
 326             Oflag++;
 327             break;
 328         case 'q':
 329             qflag++;
 330             break;
 331         case 'w':
 332             wflag++;
 333             break;
 334         case 'v':
 335             vflag++;
 336             break;
 337         default:
 338             usage();
 339             break;
 340         }
 341     }
 342     if (skip_symlink)
 343         set_skip_symlink(skip_symlink);
 344     if (qflag) {
 345         vflag = 0;
 346         setquiet();
 347     }
 348     if (vflag)
 349         setverbose();
 350     if (wflag)
 351         set_langmap_wflag();
 352     if (show_version)
 353         version(NULL, vflag);
 354     if (show_help)
 355         help();
 356 
 357     argc -= optind;
 358         argv += optind;
 359 
 360     /* If dbpath is specified, -O(--objdir) option is ignored. */
 361     if (argc > 0)
 362         Oflag = 0;
 363     if (show_config) {
 364         openconf(setupdbpath(0) == 0 ? get_root() : NULL);
 365         if (config_name)
 366             printconf(config_name);
 367         else
 368             fprintf(stdout, "%s\n", getconfline());
 369         exit(0);
 370     } else if (dump_target) {
 371         /*
 372          * Dump a tag file.
 373          */
 374         DBOP *dbop = NULL;
 375         const char *dat = 0;
 376         int is_gpath = 0;
 377 
 378         if (!test("f", dump_target))
 379             die("file '%s' not found.", dump_target);
 380         if ((dbop = dbop_open(dump_target, 0, 0, DBOP_RAW)) == NULL)
 381             die("file '%s' is not a tag file.", dump_target);
 382         /*
 383          * The file which has a NEXTKEY record is GPATH.
 384          */
 385         if (dbop_get(dbop, NEXTKEY))
 386             is_gpath = 1;
 387         for (dat = dbop_first(dbop, NULL, NULL, 0); dat != NULL; dat = dbop_next(dbop)) {
 388             const char *flag = is_gpath ? dbop_getflag(dbop) : "";
 389 
 390             if (*flag)
 391                 printf("%s\t%s\t%s\n", dbop->lastkey, dat, flag);
 392             else
 393                 printf("%s\t%s\n", dbop->lastkey, dat);
 394         }
 395         dbop_close(dbop);
 396         exit(0);
 397     } else if (Iflag) {
 398 #define REQUIRED_MKID_VERSION "4.5"
 399         char *p;
 400 
 401         if (!usable("mkid"))
 402             die("mkid not found.");
 403         if (read_first_line("mkid --version", sb))
 404             die("mkid cannot executed.");
 405         p = strrchr(strbuf_value(sb), ' ');
 406         if (p == NULL)
 407             die("invalid version string of mkid: %s", strbuf_value(sb));
 408         switch (check_version(p + 1, REQUIRED_MKID_VERSION)
 409 #ifdef _WIN32
 410             | strcmp(p + 1, "3.2.99") == 0
 411 #endif
 412             )  {
 413         case 1:     break;  /* OK */
 414         case 0:     die("mkid version %s or later is required.", REQUIRED_MKID_VERSION);
 415         default:    die("invalid version string of mkid: %s", strbuf_value(sb));
 416         }
 417     }
 418 
 419     /*
 420      * If 'gtags.files' exists, use it as a file list.
 421      * If the file_list other than "-" is given, it must be readable file.
 422      */
 423     if (file_list == NULL && test("f", GTAGSFILES))
 424         file_list = GTAGSFILES;
 425     if (file_list && strcmp(file_list, "-")) {
 426         if (test("d", file_list))
 427             die("'%s' is a directory.", file_list);
 428         else if (!test("f", file_list))
 429             die("'%s' not found.", file_list);
 430         else if (!test("r", file_list))
 431             die("'%s' is not readable.", file_list);
 432     }
 433     /*
 434      * Regularize the path name for single updating (--single-update).
 435      */
 436     if (single_update) {
 437         static char regular_path_name[MAXPATHLEN];
 438         char *p = single_update;
 439         
 440 #if _WIN32 || __DJGPP__
 441         for (; *p; p++)
 442             if (*p == '\\')
 443                 *p = '/';
 444         p = single_update;
 445 #define LOCATEFLAG MATCH_AT_FIRST|IGNORE_CASE
 446 #else
 447 #define LOCATEFLAG MATCH_AT_FIRST
 448 #endif
 449         if (isabspath(p)) {
 450             char *q = locatestring(p, cwd, LOCATEFLAG);
 451 
 452             if (q && *q == '/')
 453                 snprintf(regular_path_name, MAXPATHLEN, "./%s", q + 1);
 454             else
 455                 die("path '%s' is out of the project.", p);
 456 
 457         } else {
 458             if (p[0] == '.' && p[1] == '/')
 459                 snprintf(regular_path_name, MAXPATHLEN, "%s", p);
 460             else
 461                 snprintf(regular_path_name, MAXPATHLEN, "./%s", p);
 462         }
 463         single_update = regular_path_name;
 464     }
 465     /*
 466      * Decide directory (dbpath) in which gtags make tag files.
 467      *
 468      * Gtags create tag files at current directory by default.
 469      * If dbpath is specified as an argument then use it.
 470      * If the -i option specified and both GTAGS and GRTAGS exists
 471      * at one of the candidate directories then gtags use existing
 472      * tag files.
 473      */
 474     if (iflag) {
 475         if (argc > 0)
 476             realpath(*argv, dbpath);
 477         else if (!gtagsexist(cwd, dbpath, MAXPATHLEN, vflag))
 478             strlimcpy(dbpath, cwd, sizeof(dbpath));
 479     } else {
 480         if (argc > 0)
 481             realpath(*argv, dbpath);
 482         else if (Oflag) {
 483             char *objdir = getobjdir(cwd, vflag);
 484 
 485             if (objdir == NULL)
 486                 die("Objdir not found.");
 487             strlimcpy(dbpath, objdir, sizeof(dbpath));
 488         } else
 489             strlimcpy(dbpath, cwd, sizeof(dbpath));
 490     }
 491     if (iflag && (!test("f", makepath(dbpath, dbname(GTAGS), NULL)) ||
 492         !test("f", makepath(dbpath, dbname(GRTAGS), NULL)) ||
 493         !test("f", makepath(dbpath, dbname(GPATH), NULL)))) {
 494         if (wflag)
 495             warning("GTAGS, GRTAGS or GPATH not found. -i option ignored.");
 496         iflag = 0;
 497     }
 498     if (!test("d", dbpath))
 499         die("directory '%s' not found.", dbpath);
 500     /*
 501      * Start processing.
 502      */
 503     if (accept_dotfiles)
 504         set_accept_dotfiles();
 505     if (skip_unreadable)
 506         set_skip_unreadable();
 507     if (vflag) {
 508         const char *config_path = getconfigpath();
 509         const char *config_label = getconfiglabel();
 510 
 511         fprintf(stderr, "[%s] Gtags started.\n", now());
 512         if (config_path)
 513             fprintf(stderr, " Using configuration file '%s'.\n", config_path);
 514         else {
 515             fprintf(stderr, " Using default configuration.\n");
 516             if (getenv("GTAGSLABEL"))
 517                 fprintf(stderr, " GTAGSLABEL(--gtagslabel) ignored since configuration file not found.\n");
 518         }
 519         if (config_label)
 520             fprintf(stderr, " Using configuration label '%s'.\n", config_label);
 521         if (file_list)
 522             fprintf(stderr, " Using '%s' as a file list.\n", file_list);
 523     }
 524     /*
 525      * initialize parser.
 526      */
 527     if (vflag && gtags_parser)
 528         fprintf(stderr, " Using plug-in parser.\n");
 529     parser_init(langmap, gtags_parser);
 530     /*
 531      * Start statistics.
 532      */
 533     init_statistics();
 534     /*
 535      * incremental update.
 536      */
 537     if (iflag) {
 538         /*
 539          * Version check. If existing tag files are old enough
 540          * gtagsopen() abort with error message.
 541          */
 542         GTOP *gtop = gtags_open(dbpath, cwd, GTAGS, GTAGS_MODIFY, 0);
 543         gtags_close(gtop);
 544         /*
 545          * GPATH is needed for incremental updating.
 546          * Gtags check whether or not GPATH exist, since it may be
 547          * removed by mistake.
 548          */
 549         if (!test("f", makepath(dbpath, dbname(GPATH), NULL)))
 550             die("Old version tag file found. Please remake it.");
 551         (void)incremental(dbpath, cwd);
 552         print_statistics(statistics);
 553         exit(0);
 554     }
 555     /*
 556      * create GTAGS and GRTAGS
 557      */
 558     createtags(dbpath, cwd);
 559     /*
 560      * create idutils index.
 561      */
 562     if (Iflag) {
 563         FILE *op;
 564         GFIND *gp;
 565         const char *path;
 566 
 567         tim = statistics_time_start("Time of creating ID");
 568         if (vflag)
 569             fprintf(stderr, "[%s] Creating indexes for idutils.\n", now());
 570         strbuf_reset(sb);
 571         /*
 572          * Since idutils stores the value of PWD in ID file, we need to
 573          * force idutils to follow our style.
 574          */
 575 #if _WIN32 || __DJGPP__
 576         strbuf_puts(sb, "mkid --files0-from=-");
 577 #else
 578         strbuf_sprintf(sb, "PWD=%s mkid --files0-from=-", quote_shell(cwd));
 579 #endif
 580         if (vflag)
 581             strbuf_puts(sb, " -v");
 582         strbuf_sprintf(sb, " --file=%s/ID", quote_shell(dbpath));
 583         if (vflag) {
 584 #ifdef __DJGPP__
 585             if (is_unixy()) /* test for 4DOS as well? */
 586 #endif
 587             strbuf_puts(sb, " 1>&2");
 588         } else {
 589             strbuf_puts(sb, " >" NULL_DEVICE);
 590 #ifdef __DJGPP__
 591             if (is_unixy()) /* test for 4DOS as well? */
 592 #endif
 593             strbuf_puts(sb, " 2>&1");
 594         }
 595         if (debug)
 596             fprintf(stderr, "executing mkid like: %s\n", strbuf_value(sb));
 597         op = popen(strbuf_value(sb), "w");
 598         if (op == NULL)
 599             die("cannot execute '%s'.", strbuf_value(sb));
 600         gp = gfind_open(dbpath, NULL, GPATH_BOTH, 0);
 601         while ((path = gfind_read(gp)) != NULL) {
 602             fputs(path, op);
 603             fputc('\0', op);
 604         }
 605         gfind_close(gp);
 606         if (pclose(op) != 0)
 607             die("terminated abnormally '%s' (errno = %d).", strbuf_value(sb), errno);
 608         if (test("f", makepath(dbpath, "ID", NULL)))
 609             if (chmod(makepath(dbpath, "ID", NULL), 0644) < 0)
 610                 die("cannot chmod ID file.");
 611         statistics_time_end(tim);
 612     }
 613     if (vflag)
 614         fprintf(stderr, "[%s] Done.\n", now());
 615     closeconf();
 616     strbuf_close(sb);
 617     print_statistics(statistics);
 618     return 0;
 619 }
 620 /**
 621  * incremental: incremental update
 622  *
 623  *  @param[in]  dbpath  dbpath directory
 624  *  @param[in]  root    root directory of source tree
 625  *  @return     0: not updated, 1: updated
 626  */
 627 int
 628 incremental(const char *dbpath, const char *root)
 629 {
 630     STATISTICS_TIME *tim;
 631     struct stat statp;
 632     time_t gtags_mtime;
 633     STRBUF *addlist = strbuf_open(0);
 634     STRBUF *deletelist = strbuf_open(0);
 635     STRBUF *addlist_other = strbuf_open(0);
 636     IDSET *deleteset, *findset;
 637     int updated = 0;
 638     const char *path;
 639     unsigned int id, limit;
 640 
 641     tim = statistics_time_start("Time of inspecting %s and %s.", dbname(GTAGS), dbname(GRTAGS));
 642     if (vflag) {
 643         fprintf(stderr, " Tag found in '%s'.\n", dbpath);
 644         fprintf(stderr, " Incremental updating.\n");
 645     }
 646     /*
 647      * get modified time of GTAGS.
 648      */
 649     path = makepath(dbpath, dbname(GTAGS), NULL);
 650     if (stat(path, &statp) < 0)
 651         die("stat failed '%s'.", path);
 652     gtags_mtime = statp.st_mtime;
 653 
 654     if (gpath_open(dbpath, 2) < 0)
 655         die("GPATH not found.");
 656     /*
 657      * deleteset:
 658      *  The list of the path name which should be deleted from GPATH.
 659      * findset:
 660      *  The list of the path name which exists in the current project.
 661      *  A project is limited by the --file option.
 662      */
 663     deleteset = idset_open(gpath_nextkey());
 664     findset = idset_open(gpath_nextkey());
 665     total = 0;
 666     /*
 667      * Make add list and delete list for update.
 668      */
 669     if (single_update) {
 670         int type;
 671         const char *fid;
 672 
 673         if (skipthisfile(single_update))
 674             goto exit;
 675         if (test("b", single_update))
 676             goto exit;
 677         fid = gpath_path2fid(single_update, &type);
 678         if (fid == NULL) {
 679             /* new file */
 680             if (!test("f", single_update))
 681                 die("'%s' not found.", single_update);
 682             type = issourcefile(single_update) ? GPATH_SOURCE : GPATH_OTHER;
 683             if (type == GPATH_OTHER)
 684                 strbuf_puts0(addlist_other, single_update);
 685             else {
 686                 strbuf_puts0(addlist, single_update);
 687                 total++;
 688             }
 689         } else if (!test("f", single_update)) {
 690             /* delete */
 691             if (type != GPATH_OTHER) {
 692                 idset_add(deleteset, atoi(fid));
 693                 total++;
 694             }
 695             strbuf_puts0(deletelist, single_update);
 696         } else {
 697             /* update */
 698             if (type == GPATH_OTHER)
 699                 goto exit;
 700             idset_add(deleteset, atoi(fid));
 701             strbuf_puts0(addlist, single_update);
 702             total++;
 703         }
 704     } else {
 705         if (file_list)
 706             find_open_filelist(file_list, root, explain);
 707         else
 708             find_open(NULL, explain);
 709         while ((path = find_read()) != NULL) {
 710             const char *fid;
 711             int n_fid = 0;
 712             int other = 0;
 713 
 714             /* a blank at the head of path means 'NOT SOURCE'. */
 715             if (*path == ' ') {
 716                 if (test("b", ++path))
 717                     continue;
 718                 other = 1;
 719             }
 720             if (stat(path, &statp) < 0)
 721                 die("stat failed '%s'.", path);
 722             fid = gpath_path2fid(path, NULL);
 723             if (fid) { 
 724                 n_fid = atoi(fid);
 725                 idset_add(findset, n_fid);
 726             }
 727             if (other) {
 728                 if (fid == NULL)
 729                     strbuf_puts0(addlist_other, path);
 730             } else {
 731                 if (fid == NULL) {
 732                     strbuf_puts0(addlist, path);
 733                     total++;
 734                 } else if (gtags_mtime < statp.st_mtime) {
 735                     strbuf_puts0(addlist, path);
 736                     total++;
 737                     idset_add(deleteset, n_fid);
 738                 }
 739             }
 740         }
 741         find_close();
 742         /*
 743          * make delete list.
 744          */
 745         limit = gpath_nextkey();
 746         for (id = 1; id < limit; id++) {
 747             char fid[MAXFIDLEN];
 748             int type;
 749 
 750             snprintf(fid, sizeof(fid), "%d", id);
 751             /*
 752              * This is a hole of GPATH. The hole increases if the deletion
 753              * and the addition are repeated.
 754              */
 755             if ((path = gpath_fid2path(fid, &type)) == NULL)
 756                 continue;
 757             /*
 758              * The file which does not exist in the findset is treated
 759              * assuming that it does not exist in the file system.
 760              */
 761             if (type == GPATH_OTHER) {
 762                 if (!idset_contains(findset, id) || !test("f", path) || test("b", path))
 763                     strbuf_puts0(deletelist, path);
 764             } else {
 765                 if (!idset_contains(findset, id) || !test("f", path)) {
 766                     strbuf_puts0(deletelist, path);
 767                     idset_add(deleteset, id);
 768                 }
 769             }
 770         }
 771     }
 772     statistics_time_end(tim);
 773     /*
 774      * execute updating.
 775      */
 776     if ((!idset_empty(deleteset) || strbuf_getlen(addlist) > 0) ||
 777         (strbuf_getlen(deletelist) + strbuf_getlen(addlist_other) > 0))
 778     {
 779         int db;
 780         updated = 1;
 781         tim = statistics_time_start("Time of updating %s and %s.", dbname(GTAGS), dbname(GRTAGS));
 782         if (!idset_empty(deleteset) || strbuf_getlen(addlist) > 0)
 783             updatetags(dbpath, root, deleteset, addlist);
 784         if (strbuf_getlen(deletelist) + strbuf_getlen(addlist_other) > 0) {
 785             const char *start, *end, *p;
 786 
 787             if (vflag)
 788                 fprintf(stderr, "[%s] Updating '%s'.\n", now(), dbname(GPATH));
 789             /* gpath_open(dbpath, 2); */
 790             if (strbuf_getlen(deletelist) > 0) {
 791                 start = strbuf_value(deletelist);
 792                 end = start + strbuf_getlen(deletelist);
 793 
 794                 for (p = start; p < end; p += strlen(p) + 1)
 795                     gpath_delete(p);
 796             }
 797             if (strbuf_getlen(addlist_other) > 0) {
 798                 start = strbuf_value(addlist_other);
 799                 end = start + strbuf_getlen(addlist_other);
 800 
 801                 for (p = start; p < end; p += strlen(p) + 1) {
 802                     gpath_put(p, GPATH_OTHER);
 803                 }
 804             }
 805             /* gpath_close(); */
 806         }
 807         /*
 808          * Update modification time of tag files
 809          * because they may have no definitions.
 810          */
 811         for (db = GTAGS; db < GTAGLIM; db++)
 812             utime(makepath(dbpath, dbname(db), NULL), NULL);
 813         statistics_time_end(tim);
 814     }
 815 exit:
 816     if (vflag) {
 817         if (updated)
 818             fprintf(stderr, " Global databases have been modified.\n");
 819         else
 820             fprintf(stderr, " Global databases are up to date.\n");
 821         fprintf(stderr, "[%s] Done.\n", now());
 822     }
 823     strbuf_close(addlist);
 824     strbuf_close(deletelist);
 825     strbuf_close(addlist_other);
 826     gpath_close();
 827     idset_close(deleteset);
 828     idset_close(findset);
 829 
 830     return updated;
 831 }
 832 /**
 833  * static void put_syms(int type, const char *tag, int lno, const char *path, const char *line_image, void *arg)
 834  *
 835  * callback functions for built-in parser
 836  */
 837 struct put_func_data {
 838     GTOP *gtop[GTAGLIM];
 839     const char *fid;
 840 };
 841 static void
 842 put_syms(int type, const char *tag, int lno, const char *path, const char *line_image, void *arg)
 843 {
 844     const struct put_func_data *data = arg;
 845     GTOP *gtop;
 846     const char *p;
 847 
 848     /*
 849      * sanity checks
 850      * These checks are required, because there is no telling what kind of string
 851      * comes as 'a symbol' from external plug-in parsers.
 852      */
 853     for (p = tag; *p; p++) {
 854         if (isspace(*p)) {
 855             if (wflag)
 856                 warning("symbol name includs a space character. (Ignored) [+%d %s]", lno, path);
 857             return;
 858         }
 859     }
 860     if (p == tag) {
 861         if (wflag)
 862             warning("symbol name is null. (Ignored) [+%d %s]", lno, path);
 863         return;
 864     }
 865     if (p - tag >= IDENTLEN) {
 866         if (wflag)
 867             warning("symbol name is too long. (Ignored) [+%d %s]", lno, path);
 868         return;
 869     }
 870     switch (type) {
 871     case PARSER_DEF:
 872         gtop = data->gtop[GTAGS];
 873         break;
 874     case PARSER_REF_SYM:
 875         gtop = data->gtop[GRTAGS];
 876         if (gtop == NULL)
 877             return;
 878         break;
 879     default:
 880         return;
 881     }
 882     gtags_put_using(gtop, tag, lno, data->fid, line_image);
 883 }
 884 /**
 885  * updatetags: update tag file.
 886  *
 887  *  @param[in]  dbpath      directory in which tag file exist
 888  *  @param[in]  root        root directory of source tree
 889  *  @param[in]  deleteset   bit array of fid of deleted or modified files 
 890  *  @param[in]  addlist     '\0' separated list of added or modified files
 891  */
 892 void
 893 updatetags(const char *dbpath, const char *root, IDSET *deleteset, STRBUF *addlist)
 894 {
 895     struct put_func_data data;
 896     int seqno, flags;
 897     const char *path, *start, *end;
 898 
 899     if (vflag)
 900         fprintf(stderr, "[%s] Updating '%s' and '%s'.\n", now(), dbname(GTAGS), dbname(GRTAGS));
 901     /*
 902      * Open tag files.
 903      */
 904     data.gtop[GTAGS] = gtags_open(dbpath, root, GTAGS, GTAGS_MODIFY, 0);
 905     if (test("f", makepath(dbpath, dbname(GRTAGS), NULL))) {
 906         data.gtop[GRTAGS] = gtags_open(dbpath, root, GRTAGS, GTAGS_MODIFY, 0);
 907     } else {
 908         /*
 909          * If you set NULL to data.gtop[GRTAGS], parse_file() doesn't write to
 910          * GRTAGS. See put_syms().
 911          */
 912         data.gtop[GRTAGS] = NULL;
 913     }
 914     /*
 915      * Delete tags from GTAGS.
 916      */
 917     if (!idset_empty(deleteset)) {
 918         if (vflag) {
 919             char fid[MAXFIDLEN];
 920             int total = idset_count(deleteset);
 921             unsigned int id;
 922 
 923             seqno = 1;
 924             for (id = idset_first(deleteset); id != END_OF_ID; id = idset_next(deleteset)) {
 925                 snprintf(fid, sizeof(fid), "%d", id);
 926                 path = gpath_fid2path(fid, NULL);
 927                 if (path == NULL)
 928                     die("GPATH is corrupted.");
 929                 fprintf(stderr, " [%d/%d] deleting tags of %s\n", seqno++, total, path + 2);
 930             }
 931         }
 932         gtags_delete(data.gtop[GTAGS], deleteset);
 933         if (data.gtop[GRTAGS] != NULL)
 934             gtags_delete(data.gtop[GRTAGS], deleteset);
 935     }
 936     /*
 937      * Set flags.
 938      */
 939     data.gtop[GTAGS]->flags = 0;
 940     if (extractmethod)
 941         data.gtop[GTAGS]->flags |= GTAGS_EXTRACTMETHOD;
 942     data.gtop[GRTAGS]->flags = data.gtop[GTAGS]->flags;
 943     flags = 0;
 944     if (vflag)
 945         flags |= PARSER_VERBOSE;
 946     if (debug)
 947         flags |= PARSER_DEBUG;
 948     if (wflag)
 949         flags |= PARSER_WARNING;
 950     if (explain)
 951         flags |= PARSER_EXPLAIN;
 952     if (getenv("GTAGSFORCEENDBLOCK"))
 953         flags |= PARSER_END_BLOCK;
 954     /*
 955      * Add tags to GTAGS and GRTAGS.
 956      */
 957     start = strbuf_value(addlist);
 958     end = start + strbuf_getlen(addlist);
 959     seqno = 0;
 960     for (path = start; path < end; path += strlen(path) + 1) {
 961         gpath_put(path, GPATH_SOURCE);
 962         data.fid = gpath_path2fid(path, NULL);
 963         if (data.fid == NULL)
 964             die("GPATH is corrupted.('%s' not found)", path);
 965         if (vflag)
 966             fprintf(stderr, " [%d/%d] extracting tags of %s\n", ++seqno, total, path + 2);
 967         parse_file(path, flags, put_syms, &data);
 968         gtags_flush(data.gtop[GTAGS], data.fid);
 969         if (data.gtop[GRTAGS] != NULL)
 970             gtags_flush(data.gtop[GRTAGS], data.fid);
 971     }
 972     parser_exit();
 973     gtags_close(data.gtop[GTAGS]);
 974     if (data.gtop[GRTAGS] != NULL)
 975         gtags_close(data.gtop[GRTAGS]);
 976 }
 977 /**
 978  * createtags: create tags file
 979  *
 980  *  @param[in]  dbpath  dbpath directory
 981  *  @param[in]  root    root directory of source tree
 982  */
 983 void
 984 createtags(const char *dbpath, const char *root)
 985 {
 986     STATISTICS_TIME *tim;
 987     STRBUF *sb = strbuf_open(0);
 988     struct put_func_data data;
 989     int openflags, flags, seqno;
 990     const char *path;
 991 
 992     tim = statistics_time_start("Time of creating %s and %s.", dbname(GTAGS), dbname(GRTAGS));
 993     if (vflag)
 994         fprintf(stderr, "[%s] Creating '%s' and '%s'.\n", now(), dbname(GTAGS), dbname(GRTAGS));
 995     openflags = cflag ? GTAGS_COMPACT : 0;
 996 #ifdef USE_SQLITE3
 997     if (use_sqlite3)
 998         openflags |= GTAGS_SQLITE3;
 999 #endif
1000     data.gtop[GTAGS] = gtags_open(dbpath, root, GTAGS, GTAGS_CREATE, openflags);
1001     data.gtop[GTAGS]->flags = 0;
1002     if (extractmethod)
1003         data.gtop[GTAGS]->flags |= GTAGS_EXTRACTMETHOD;
1004     data.gtop[GRTAGS] = gtags_open(dbpath, root, GRTAGS, GTAGS_CREATE, openflags);
1005     data.gtop[GRTAGS]->flags = data.gtop[GTAGS]->flags;
1006     flags = 0;
1007     if (vflag)
1008         flags |= PARSER_VERBOSE;
1009     if (debug)
1010         flags |= PARSER_DEBUG;
1011     if (wflag)
1012         flags |= PARSER_WARNING;
1013     if (explain)
1014         flags |= PARSER_EXPLAIN;
1015     if (getenv("GTAGSFORCEENDBLOCK"))
1016         flags |= PARSER_END_BLOCK;
1017     /*
1018      * Add tags to GTAGS and GRTAGS.
1019      */
1020     if (file_list)
1021         find_open_filelist(file_list, root, explain);
1022     else
1023         find_open(NULL, explain);
1024     seqno = 0;
1025     while ((path = find_read()) != NULL) {
1026         if (*path == ' ') {
1027             path++;
1028             if (!test("b", path))
1029                 gpath_put(path, GPATH_OTHER);
1030             continue;
1031         }
1032         gpath_put(path, GPATH_SOURCE);
1033         data.fid = gpath_path2fid(path, NULL);
1034         if (data.fid == NULL)
1035             die("GPATH is corrupted.('%s' not found)", path);
1036         seqno++;
1037         if (vflag)
1038             fprintf(stderr, " [%d] extracting tags of %s\n", seqno, path + 2);
1039         parse_file(path, flags, put_syms, &data);
1040         gtags_flush(data.gtop[GTAGS], data.fid);
1041         gtags_flush(data.gtop[GRTAGS], data.fid);
1042     }
1043     total = seqno;
1044     parser_exit();
1045     find_close();
1046     statistics_time_end(tim);
1047     tim = statistics_time_start("Time of flushing B-tree cache");
1048     gtags_close(data.gtop[GTAGS]);
1049     gtags_close(data.gtop[GRTAGS]);
1050     statistics_time_end(tim);
1051     strbuf_reset(sb);
1052     if (getconfs("GTAGS_extra", sb)) {
1053         tim = statistics_time_start("Time of executing GTAGS_extra command");
1054         if (system(strbuf_value(sb)))
1055             fprintf(stderr, "GTAGS_extra command failed: %s\n", strbuf_value(sb));
1056         statistics_time_end(tim);
1057     }
1058     strbuf_reset(sb);
1059     if (getconfs("GRTAGS_extra", sb)) {
1060         tim = statistics_time_start("Time of executing GRTAGS_extra command");
1061         if (system(strbuf_value(sb)))
1062             fprintf(stderr, "GRTAGS_extra command failed: %s\n", strbuf_value(sb));
1063         statistics_time_end(tim);
1064     }
1065     strbuf_close(sb);
1066 }
