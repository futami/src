  27 /*
  28  * Substring manager like perl's split.
  29  *
  30  * Initial status.
  31  *              +------------------------------------------------------
  32  *         line |main         100    ./main.c        main(argc, argv)\n
  33  *
  34  * The result of split(line, 4, &list):
  35  *
  36  *              +------------------------------------------------------
  37  * list    line |main\0       100\0  ./main.c\0      main(argc, argv)\n
  38  * +---------+   ^   ^        ^  ^   ^       ^       ^
  39  * |npart=4  |   |   |        |  |   |       |       |
  40  * +---------+   |   |        |  |   |       |       |
  41  * | start  *----+   |        |  |   |       |       |
  42  * | end    *--------+        |  |   |       |       |
  43  * | save ' '|                |  |   |       |       |
  44  * +---------+                |  |   |       |       |
  45  * | start  *-----------------+  |   |       |       |
  46  * | end    *--------------------+   |       |       |
  47  * | save ' '|                       |       |       |
  48  * +---------+                       |       |       |
  49  * | start  *------------------------+       |       |
  50  * | end    *--------------------------------+       |
  51  * | save ' '|                                       |
  52  * +---------+                                       |
  53  * | start  *----------------------------------------+
  54  * | end    *--+
  55  * | save    | |
  56  * +---------+ =
  57  *
  58  * The result of split(line, 2, &list):
  59  *
  60  *              +------------------------------------------------------
  61  * list    line |main\0       100    ./main.c        main(argc, argv)\n
  62  * +---------+   ^   ^        ^
  63  * |npart=2  |   |   |        |
  64  * +---------+   |   |        |
  65  * | start  *----+   |        |
  66  * | end    *--------+        |
  67  * | save ' '|                |
  68  * +---------+                |
  69  * | start  *-----------------+
  70  * | end    *--+
  71  * | save    | |
  72  * +---------+ =
  73  *
  74  * The result of recover().
  75  *
  76  *              +------------------------------------------------------
  77  *         line |main         100    ./main.c        main(argc, argv)\n
  78  *
  79  * Recover() recover initial status of line with saved char in savec.
  80  */

  82 #define isblank(c)  ((c) == ' ' || (c) == '\t')

  84 /**
  85  * split: split a string into pieces
  86  *
  87  *  @param[in]  line    string
  88  *  @param[in]  npart   parts number
  89  *  @param[in,out]  list    split table
  90  *  @return     part count
  91  */
  92 int
  93 split(const char *line, int npart, SPLIT *list)     /* virtually const */
  94 {
  95     char *s = (char *)line;
  96     struct part *part = &list->part[0];
  97     int count;
  98 
  99     if (npart > NPART)
 100         npart = NPART;
 101     npart--;
 102     for (count = 0; *s && count < npart; count++) {
 103         while (*s && isblank(*s))
 104             s++;
 105         if (*s == '\0')
 106             break;
 107         part->start = s;
 108         while (*s && !isblank(*s))
 109             s++;
 110         part->end = s;
 111         part->savec = *s;
 112         part++;
 113     }
 114     if (*s) {
 115         while (*s && isblank(*s))
 116             s++;
 117         part->start = s;
 118         part->end = (char *)0;
 119         part->savec = 0;
 120         count++;
 121         part++;
 122     }
 123     while (part-- > &list->part[0]) {
 124         if (part->savec != '\0')
 125             *part->end = '\0';
 126     }
 127     return list->npart = count;
 128 }
 129 /**
 130  * recover: recover initial status of line.
 131  *
 132  *  @param[in,out]  list    split table
 133  */
 134 void
 135 recover(SPLIT *list)
 136 {
 137     int i, c;
 138     for (i = 0; i < list->npart; i++) {
 139         if ((c = list->part[i].savec) != '\0')
 140             *(list->part[i].end) = c;
 141     }
 142 }
 143 /**
 144  * split_dump: dump split structure.
 145  */
 146 void
 147 split_dump(SPLIT *list)
 148 {
 149     int i;
 150     struct part *part;
 151 
 152     fprintf(stderr, "npart: %d\n", list->npart);
 153     
 154     for (i = 0; i < list->npart; i++) {
 155         part = &list->part[i];
 156         fprintf(stderr, "string[%d]: |%s|\n", i, part->start);
 157         fprintf(stderr, "savec[%d] : |%c|\n", i, part->savec);
 158     }
 159 }
 160 /**
 161  * parse_xid: extract fid from ctags_xid format record.
 162  *
 163  *  @param[in]  ctags_xid   ctags-xid record
 164  *  @param[out] s_fid       file id(string) if not NULL
 165  *  @param[out] n_fid       file id(integer) if not NULL
 166  *  @return         pointer to the ctags_x part
 167  */
 168 const char *
 169 parse_xid(const char *ctags_xid, char *s_fid, int *n_fid)
 170 {
 171     const char *p;
 172     int i, n;
 173 
 174     i = n = 0;
 175     for (p = ctags_xid; *p && *p >= '0' && *p <= '9'; p++) {
 176         n = n * 10 + (*p - '0');
 177         if (s_fid)
 178             s_fid[i++] = *p;
 179     }
 180     if (*p++ != ' ')
 181         die("invalid ctags-xid format record. '%s'", ctags_xid);
 182     if (s_fid)
 183         s_fid[i] = '\0';
 184     if (n_fid != NULL)
 185         *n_fid = n;
 186     return p;
 187 }
 188 /**
 189  * nextstring: seek to the next string.
 190  *
 191  *      @param[in]      s       original string
 192  *      @return              next string
 193  *
 194  *  s       v
 195  * "aaaaaa\0bbbbb\0"
 196  */
 197 const char *
 198 nextstring(const char *s)
 199 {
 200     while (*s)
 201         s++;
 202     return s + 1;
 203 }
 204 /**
 205  * nextelement: seek to the next element
 206  *
 207  *  @param[in]  s   point the current element or the following blanks
 208  *  @return     next element
 209  *
 210  *  s     s   return value
 211  *  v     v   v
 212  *  xxxxx     yyyyy    zzzzzz
 213  *  (current) (next)
 214  */
 215 const char *
 216 nextelement(const char *s)
 217 {
 218     while (*s && !isblank(*s))
 219         s++;
 220     if (!*s)
 221         die("nextelement: unexpected end of string(1).");
 222     while (*s && isblank(*s))
 223         s++;
 224     if (!*s)
 225         die("nextelement: unexpected end of string(2).");
 226     return s;
 227 }
