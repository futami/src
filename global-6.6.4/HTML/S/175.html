  49 #define SIG_BLOCK   1
  50 #define SIG_UNBLOCK 2
  51 #define SIG_SETMASK 3
  52 
  53 static int __sigtemp;       /**< For the use of sigprocmask */
  54 
  55 /*
  56  *  Repeated test of oset != NULL is to avoid "*0".
  57  */
  58 #define sigprocmask(how, set, oset)                 \
  59     ((__sigtemp =                           \
  60     (((how) == SIG_BLOCK) ?                     \
  61         sigblock(0) | *(set) :                  \
  62     (((how) == SIG_UNBLOCK) ?                   \
  63         sigblock(0) & ~(*(set)) :               \
  64     ((how) == SIG_SETMASK ?                     \
  65         *(set) : sigblock(0))))),               \
  66     ((oset) ? (*(oset ? oset : set) = sigsetmask(__sigtemp)) :  \
  67         sigsetmask(__sigtemp)), 0)
  68 #endif

  86 /* Compatibility macros */
  87 /*
  88  * Old definitions were rewritten using 'HAVE_XXX' macros.
  89  *
  90  * #if defined(SYSV) || defined(SYSTEM5)
  91  * #define  index(a, b)     strchr(a, b)
  92  * #define  rindex(a, b)        strrchr(a, b)
  93  * #define  bzero(a, b)     memset(a, 0, b)
  94  * #define  bcmp(a, b, n)       memcmp(a, b, n)
  95  * #define  bcopy(a, b, n)      memmove(b, a, n)
  96  * #endif
  97  * 
  98  * #if defined(BSD) || defined(BSD4_3)
  99  * #define  strchr(a, b)        index(a, b)
 100  * #define  strrchr(a, b)       rindex(a, b)
 101  * #define  memcmp(a, b, n)     bcmp(a, b, n)
 102  * #define  memmove(a, b, n)    bcopy(b, a, n)
 103  * #endif
 104 */
 105 #if !defined (HAVE_INDEX) && defined (HAVE_STRCHR)
 106 #define index(a, b)     strchr(a, b)
 107 #endif
 108 #if !defined (HAVE_RINDEX) && defined (HAVE_STRRCHR)
 109 #define rindex(a, b)        strrchr(a, b)
 110 #endif
 111 #if !defined (HAVE_BZERO) && defined (HAVE_MEMSET)
 112 #define bzero(a, b)     memset(a, 0, b)
 113 #endif
 114 #if !defined (HAVE_BCMP) && defined (HAVE_MEMCMP)
 115 #define bcmp(a, b, n)       memcmp(a, b, n)
 116 #endif
 117 #if !defined (HAVE_BCOPY) && defined (HAVE_MEMMOVE)
 118 #define bcopy(a, b, n)      memmove(b, a, n)
 119 #endif
 120 
 121 #if !defined (HAVE_STRCHR) && defined (HAVE_INDEX)
 122 #define strchr(a, b)        index(a, b)
 123 #endif
 124 #if !defined (HAVE_STRRCHR) && defined (HAVE_RINDEX)
 125 #define strrchr(a, b)       rindex(a, b)
 126 #endif
 127 #if !defined (HAVE_MEMCMP) && defined (HAVE_BCMP)
 128 #define memcmp(a, b, n)     bcmp(a, b, n)
 129 #endif
 130 #if !defined (HAVE_MEMMOVE) && defined (HAVE_BCOPY)
 131 #define memmove(a, b, n)    bcopy(b, a, n)
 132 #endif
