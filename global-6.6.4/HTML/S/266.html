  59  * args_open_filelist: args_open like interface for handling output of find(1).
  60  *
  61  *  @param[in]  filename    file including list of file names.
  62  *              When "-" is specified, read from standard input.
  63  */
  64 void
  65 args_open_filelist(const char *filename)
  66 {
  67     type = ARGS_FILELIST;
  68     if (!strcmp(filename, "-")) {
  69         ip = stdin;
  70     } else {
  71         ip = fopen(filename, "r");
  72         if (ip == NULL)
  73             die("cannot open '%s'.", filename);
  74     }

 112 /**
 113  * args_read: read path From args.
 114  *
 115  *  @return     path (NULL: end of argument)
 116  */
 117 const char *
 118 args_read(void)
 119 {
 120     const char *p;
 121     STATIC_STRBUF(sb);
 122 
 123     strbuf_clear(sb);
 124     switch (type) {
 125     case ARGS_NOP:
 126         p = NULL;
 127         break;
 128     case ARGS_ARGS:
 129         p = *argslist++;
 130         break;
 131     case ARGS_FILELIST:
 132         p = strbuf_fgets(sb, ip, STRBUF_NOCRLF);
 133         break;
 134     case ARGS_GFIND:
 135         p = gfind_read(gp);
 136         break;
 137     case ARGS_BOTH:
 138         if (*argslist != NULL)
 139             p = *argslist++;
 140         else
 141             p = strbuf_fgets(sb, ip, STRBUF_NOCRLF);
 142         break;
 143     default:
 144         die("args_read: invalid type.");
 145     }
 146     return p;
 147 }
 148 /**
 149  * args_close: close args.
 150  */
 151 void
 152 args_close(void)
 153 {
 154     switch (type) {
 155     case ARGS_NOP:
 156     case ARGS_ARGS:
 157         break;
 158     case ARGS_FILELIST:
 159     case ARGS_BOTH:
 160         if (ip != NULL && ip != stdin)
 161             fclose(ip);
 162         ip = NULL;
 163         break;
 164     case ARGS_GFIND:
 165         if (gp != NULL)
 166             gfind_close(gp);
 167         gp = NULL;
 168         break;
 169     default:
 170         die("something wrong.");
 171     }
 172 }
 173 /**
 174  * preparse_options
 175  *
 176  *  @param[in]  argc    main()'s argc integer
 177  *  @param[in]  argv    main()'s argv string array
 178  *
 179  * Setup the "GTAGSCONF" and the "GTAGSLABEL" environment variables
 180  * according to the --gtagsconf and --gtagslabel options.
 181  */
 182 void
 183 preparse_options(int argc, char *const *argv)
 184 {
 185     int i;
 186     char *p;
 187     char *confpath = NULL;
 188     char *label = NULL;
 189     const char *opt_gtagsconf = "--gtagsconf";
 190     const char *opt_gtagslabel = "--gtagslabel";
 191 
 192     for (i = 1; i < argc; i++) {
 193         if ((p = locatestring(argv[i], opt_gtagsconf, MATCH_AT_FIRST))) {
 194             if (*p == '\0') {
 195                 if (++i >= argc)
 196                     die("%s needs an argument.", opt_gtagsconf);
 197                 confpath = argv[i];
 198             } else {
 199                 if (*p++ == '=' && *p)
 200                     confpath = p;
 201             }
 202         } else if ((p = locatestring(argv[i], opt_gtagslabel, MATCH_AT_FIRST))) {
 203             if (*p == '\0') {
 204                 if (++i >= argc)
 205                     die("%s needs an argument.", opt_gtagslabel);
 206                 label = argv[i];
 207             } else {
 208                 if (*p++ == '=' && *p)
 209                     label = p;
 210             }
 211         }
 212     }
 213     if (confpath) {
 214         char real[MAXPATHLEN];
 215 
 216         if (!test("f", confpath))
 217             die("%s file not found.", opt_gtagsconf);
 218         if (!realpath(confpath, real))
 219             die("cannot get absolute path of %s file.", opt_gtagsconf);
 220         set_env("GTAGSCONF", real);
 221     }
 222     if (label)
 223         set_env("GTAGSLABEL", label);
 224 }
 225 /**
 226  * prepend_options: creates a new argv main() array, by prepending (space separated)
 227  *      options and arguments from the string argument options.
 228  *
 229  *  @param[in,out]  argc    pointer to main()'s argc integer
 230  *  @param[in]  argv    main()'s argv string array
 231  *  @param[in]  options string
 232  *  @return The new argv array.
 233  *
 234  *  The program's name is copied back into: returned[0] (argv[0]).
 235  */
 236 char **
 237 prepend_options(int *argc, char *const *argv, const char *options)
 238 {
 239     STRBUF *sb = strbuf_open(0);
 240     const char *p, *opt = check_strdup(options);
 241     int count = 1;
 242     int quote = 0;
 243     const char **newargv;
 244     int i = 0, j = 1;
 245 
 246     for (p = opt; *p && isspace(*p); p++)
 247         ;
 248     for (; *p; p++) {
 249         int c = *p;
 250 
 251         if (quote) {
 252             if (quote == c)
 253                 quote = 0;
 254             else
 255                 strbuf_putc(sb, c);
 256         } else if (c == '\\') {
 257             if (*(p + 1))
 258                 strbuf_putc(sb, *++p);
 259         } else if (c == '\'' || c == '"') {
 260             quote = c;
 261         } else if (isspace(c)) {
 262             strbuf_putc(sb, '\0');
 263             count++;
 264             while (*p && isspace(*p))
 265                 p++;
 266             p--;
 267         } else {
 268             strbuf_putc(sb, *p);
 269         }
 270     }
 271     newargv = (const char **)check_malloc(sizeof(char *) * (*argc + count + 1));
 272     newargv[i++] = argv[0];
 273     p = strbuf_value(sb);
 274     while (count--) {
 275         newargv[i++] = p;
 276         p += strlen(p) + 1;
 277     }
 278     while (j < *argc)
 279         newargv[i++] = argv[j++];
 280     newargv[i] = NULL;
 281     *argc = i;
 282 #ifdef DEBUG
 283     for (i = 0; i < *argc; i++)
 284         fprintf(stderr, "newargv[%d] = '%s'\n", i, newargv[i]);
 285 #endif
 286     /* doesn't close string buffer. */
 287 
 288     return (char **)newargv;
 289 }
 290 /**
 291  * serialize_options
 292  */
 293 char *
 294 serialize_options(int argc, char *const *argv)
 295 {
 296     STRBUF *sb = strbuf_open(0);
 297     char *string = NULL;
 298     char *p = NULL;
 299     int i;
 300     for (i = 0; i < argc; i++) {
 301         if (i > 0)
 302             strbuf_putc(sb, ' ');
 303         for (p = argv[i]; *p; p++) {
 304             /* quote spaces using url encoding */
 305             if (*p == ' ')
 306                 strbuf_puts(sb, "%20");
 307             else
 308                 strbuf_putc(sb, *p);
 309         }
 310     }
 311     string = check_strdup(strbuf_value(sb));
 312     strbuf_close(sb);
 313     return string;
 314 }
