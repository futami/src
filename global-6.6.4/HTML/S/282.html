 126 static void
 127 start_sort_process(DBOP *dbop) {
 128     int opipe[2], ipipe[2];
 129 
 130     if (!strcmp(POSIX_SORT, "no"))
 131         return;
 132     if (!test("fx", POSIX_SORT)) {
 133         static int informed;
 134 
 135         if (!informed) {
 136             warning("%s", sortnotfound);
 137             informed = 1;
 138         }
 139         return;
 140     }
 141     /*
 142      * Setup pipe for two way communication
 143      *
 144      *  Parent(gtags)               Child(sort)
 145      *  ---------------------------------------------------
 146      *  (dbop->sortout) opipe[1] =====> opipe[0] (stdin)
 147      *  (dbop->sortin)  ipipe[0] <===== ipipe[1] (stdout)
 148      */
 149     if (pipe(opipe) < 0 || pipe(ipipe) < 0)
 150         die("pipe(2) failed.");
 151     dbop->pid = fork();
 152     if (dbop->pid == 0) {
 153         /* child process */
 154         close(opipe[1]);
 155         close(ipipe[0]);
 156         if (dup2(opipe[0], 0) < 0 || dup2(ipipe[1], 1) < 0)
 157             die("dup2(2) failed.");
 158         close(opipe[0]);
 159         close(ipipe[1]);
 160         /*
 161          * Use C locale in order to avoid the degradation of performance     
 162          * by internationalized sort command.    
 163          */
 164         set_env("LC_ALL", "C");
 165         execvp(POSIX_SORT, argv);
 166         die("cannot execute '%s'. (execvp failed)", POSIX_SORT);
 167     } else if (dbop->pid < 0)
 168         die("fork(2) failed.");
 169     /* parent process */
 170     close(opipe[0]);
 171     close(ipipe[1]);
 172     fcntl(ipipe[0], F_SETFD, FD_CLOEXEC);
 173     fcntl(opipe[1], F_SETFD, FD_CLOEXEC);
 174     dbop->sortout = fdopen(opipe[1], "w");
 175     dbop->sortin = fdopen(ipipe[0], "r");
 176     if (dbop->sortout == NULL || dbop->sortin == NULL)
 177         die("fdopen(3) failed.");
 178 }
 179 static void
 180 terminate_sort_process(DBOP *dbop) {
 181     int ret, status;
 182 
 183     while ((ret = waitpid(dbop->pid, &status, 0)) < 0 && errno == EINTR)
 184         ;
 185     if (ret < 0)
 186         die("waitpid(2) failed.");
 187     if (!WIFEXITED(status) || WEXITSTATUS(status) != 0)
 188         die("terminated abnormally. '%s'", POSIX_SORT);
 189 }

 210 /**
 211  * dbop_open: open db database.
 212  *
 213  *  @param[in]  path    database name
 214  *  @param[in]  mode    0: read only, 1: create, 2: modify
 215  *  @param[in]  perm    file permission
 216  *  @param[in]  flags
 217  *          DBOP_DUP: allow duplicate records.
 218  *          DBOP_SORTED_WRITE: use sorted writing. This requires POSIX sort.
 219  *  @return     descripter for dbop_xxx() or NULL
 220  *
 221  * Sorted wirting is fast because all writing is done by not insertion but addition.
 222  */
 223 DBOP *
 224 dbop_open(const char *path, int mode, int perm, int flags)
 225 {
 226     DB *db;
 227     int rw = 0;
 228     DBOP *dbop;
 229     BTREEINFO info;
 230 
 231 #ifdef USE_SQLITE3
 232     if (mode != 1 && is_sqlite3(path))
 233         flags |= DBOP_SQLITE3;
 234     if (flags & DBOP_SQLITE3) {
 235         dbop = dbop3_open(path, mode, perm, flags);
 236         goto finish;
 237     }
 238 #endif
 239     /*
 240      * setup arguments.
 241      */
 242     switch (mode) {
 243     case 0:
 244         rw = O_RDONLY;
 245         break;
 246     case 1:
 247         rw = O_RDWR|O_CREAT|O_TRUNC;
 248         break;
 249     case 2:
 250         rw = O_RDWR;
 251         break;
 252     default:
 253         assert(0);
 254     }
 255     memset(&info, 0, sizeof(info));
 256     if (flags & DBOP_DUP)
 257         info.flags |= R_DUP;
 258     info.psize = DBOP_PAGESIZE;
 259     /*
 260      * Decide cache size. The default value is 5MB.
 261      * See libutil/gparam.h for the details.
 262      */
 263     info.cachesize = GTAGSCACHE;
 264     if (getenv("GTAGSCACHE") != NULL)
 265         info.cachesize = atoi(getenv("GTAGSCACHE"));
 266     if (info.cachesize < GTAGSMINCACHE)
 267         info.cachesize = GTAGSMINCACHE;
 268 
 269     /*
 270      * if unlink do job normally, those who already open tag file can use
 271      * it until closing.
 272      */
 273     if (path != NULL && mode == 1 && test("f", path))
 274         (void)unlink(path);
 275     db = dbopen(path, rw, 0600, DB_BTREE, &info);
 276     if (!db)
 277         return NULL;
 278     /*
 279      * if file size == 0 then it should be corrupted.
 280      */
 281     if (mode != 1) {
 282         int fd = (*db->fd)(db);
 283         struct stat stat;
 284 
 285         if (fstat(fd, &stat) < 0)
 286             die("fstat failed.");
 287         if (stat.st_size == 0) {
 288             errno = EFTYPE;
 289             return NULL;
 290         }
 291     }
 292     dbop = (DBOP *)check_calloc(sizeof(DBOP), 1);
 293     if (path == NULL)
 294         dbop->dbname[0] = '\0';
 295     else
 296         strlimcpy(dbop->dbname, path, sizeof(dbop->dbname));
 297     dbop->db    = db;
 298     dbop->openflags = flags;
 299     dbop->perm  = (mode == 1) ? perm : 0;
 300     dbop->lastdat   = NULL;
 301     dbop->lastsize  = 0;
 302     dbop->sortout   = NULL;
 303     dbop->sortin    = NULL;
 304     /*
 305      * Setup sorted writing.
 306      */
 307     if (mode != 0 && dbop->openflags & DBOP_SORTED_WRITE)
 308         start_sort_process(dbop);
 309 #ifdef USE_SQLITE3
 310 finish:
 311 #endif
 312     return dbop;
 313 }
 314 /**
 315  * dbop_get: get data by a key.
 316  *
 317  *  @param[in]  dbop    descripter
 318  *  @param[in]  name    name
 319  *  @return     pointer to data or NULL
 320  */
 321 const char *
 322 dbop_get(DBOP *dbop, const char *name)
 323 {
 324     DB *db = dbop->db;
 325     DBT key, dat;
 326     int status;
 327 
 328 #ifdef USE_SQLITE3
 329     if (dbop->openflags & DBOP_SQLITE3)
 330         return dbop3_get(dbop, name);
 331 #endif
 332     key.data = (char *)name;
 333     key.size = strlen(name)+1;
 334 
 335     status = (*db->get)(db, &key, &dat, 0);
 336     dbop->lastdat = (char *)dat.data;
 337     dbop->lastsize = dat.size;
 338     switch (status) {
 339     case RET_SUCCESS:
 340         break;
 341     case RET_ERROR:
 342         die("dbop_get failed.");
 343     case RET_SPECIAL:
 344         return (NULL);
 345     }
 346     return (dat.data);
 347 }
 348 /**
 349  * dbop_put: put data by a key.
 350  *
 351  *  @param[in]  dbop    descripter
 352  *  @param[in]  name    key
 353  *  @param[in]  data    data
 354  */
 355 void
 356 dbop_put(DBOP *dbop, const char *name, const char *data)
 357 {
 358     DB *db = dbop->db;
 359     DBT key, dat;
 360     int status;
 361     int len;
 362 
 363 #ifdef USE_SQLITE3
 364     if (dbop->openflags & DBOP_SQLITE3) {
 365         dbop3_put(dbop, name, data, NULL);
 366         return;
 367     }
 368 #endif
 369     if (!(len = strlen(name)))
 370         die("primary key size == 0.");
 371     if (len > MAXKEYLEN)
 372         die("primary key too long.");
 373     /* sorted writing */
 374     if (dbop->sortout != NULL) {
 375         fputs(name, dbop->sortout);
 376         putc(SORT_SEP, dbop->sortout);
 377         fputs(data, dbop->sortout);
 378         putc('\n', dbop->sortout);
 379         return;
 380     }
 381     key.data = (char *)name;
 382     key.size = len+1;
 383     dat.data = (char *)data;
 384     dat.size = strlen(data)+1;
 385 
 386     status = (*db->put)(db, &key, &dat, 0);
 387     switch (status) {
 388     case RET_SUCCESS:
 389         break;
 390     case RET_ERROR:
 391     case RET_SPECIAL:
 392         die("%s", dbop->put_errmsg ? dbop->put_errmsg : "dbop_put failed.");
 393     }
 394 }
 395 /**
 396  * dbop_put_tag: put a tag
 397  *
 398  *  @param[in]  dbop    descripter
 399  *  @param[in]  name    key
 400  *  @param[in]  data    data
 401  */
 402 void
 403 dbop_put_tag(DBOP *dbop, const char *name, const char *data)
 404 {
 405 #ifdef USE_SQLITE3
 406     if (dbop->openflags & DBOP_SQLITE3) {
 407         int len;
 408         char fid[MAXFIDLEN], *q = fid;
 409         const char *p = data;
 410 
 411         /* extract fid */
 412         while (*p && isdigit(*p))
 413             *q++ = *p++;
 414         *q = '\0';
 415         /* trim line */
 416         len = strlen(data);
 417         if (data[len-1] == '\n')
 418             len--;
 419         if (data[len-1] == '\r')
 420             len--;
 421         if (data[len] == '\r' || data[len] == '\n') {
 422             STATIC_STRBUF(sb);
 423 
 424             strbuf_clear(sb);
 425             strbuf_nputs(sb, data, len);
 426             data = strbuf_value(sb);
 427         }
 428         dbop3_put(dbop, name, data, fid);
 429         return;
 430     }
 431 #endif
 432     dbop_put(dbop, name, data);
 433     return;
 434 }

 526 /**
 527  * dbop_first: get first record. 
 528  * 
 529  *  @param[in]  dbop    dbop descripter
 530  *  @param[in]  name    key value or prefix,
 531  *          !=NULL: indexed read by key,
 532  *          ==NULL: sequential read
 533  *  @param[in]  preg    compiled regular expression if any.
 534  *  @param[in]  flags   following dbop_next call take over this.
 535  *          DBOP_KEY:   read key part,
 536  *          DBOP_PREFIX:    prefix read; only valid when sequential read
 537  *  @return     data or NULL
 538  */
 539 const char *
 540 dbop_first(DBOP *dbop, const char *name, regex_t *preg, int flags)
 541 {
 542     DB *db = dbop->db;
 543     DBT key, dat;
 544     int status;
 545     int len;
 546 
 547     dbop->preg = preg;
 548     dbop->ioflags = flags;
 549     if (flags & DBOP_PREFIX && !name)
 550         flags &= ~DBOP_PREFIX;
 551 #ifdef USE_SQLITE3
 552     if (dbop->openflags & DBOP_SQLITE3)
 553         return dbop3_first(dbop, name, preg, flags);
 554 #endif
 555     if (name) {
 556         if ((len = strlen(name)) > MAXKEYLEN)
 557             die("primary key too long.");
 558         strlimcpy(dbop->key, name, sizeof(dbop->key));
 559         key.data = (char *)name;
 560         key.size = len;
 561         /*
 562          * includes NULL character unless prefix read.
 563          */
 564         if (!(flags & DBOP_PREFIX))
 565             key.size++;
 566         dbop->keylen = key.size;
 567         for (status = (*db->seq)(db, &key, &dat, R_CURSOR);
 568             status == RET_SUCCESS;
 569             status = (*db->seq)(db, &key, &dat, R_NEXT)) {
 570             dbop->readcount++;
 571             if (flags & DBOP_PREFIX) {
 572                 if (strncmp((char *)key.data, dbop->key, dbop->keylen))
 573                     return NULL;
 574             } else {
 575                 if (strcmp((char *)key.data, dbop->key))
 576                     return NULL;
 577             }
 578             if (preg && regexec(preg, (char *)key.data, 0, 0, 0) != 0)
 579                 continue;
 580             break;
 581         }
 582     } else {
 583         dbop->keylen = dbop->key[0] = 0;
 584         for (status = (*db->seq)(db, &key, &dat, R_FIRST);
 585             status == RET_SUCCESS;
 586             status = (*db->seq)(db, &key, &dat, R_NEXT)) {
 587             dbop->readcount++;
 588             /* skip meta records */
 589             if (ismeta(key.data) && !(dbop->openflags & DBOP_RAW))
 590                 continue;
 591             if (preg && regexec(preg, (char *)key.data, 0, 0, 0) != 0)
 592                 continue;
 593             break;
 594         }
 595     }
 596     dbop->lastdat = (char *)dat.data;
 597     dbop->lastsize = dat.size;
 598     dbop->lastkey = (char *)key.data;
 599     dbop->lastkeysize = key.size;
 600     switch (status) {
 601     case RET_SUCCESS:
 602         break;
 603     case RET_ERROR:
 604         die("dbop_first failed.");
 605     case RET_SPECIAL:
 606         return (NULL);
 607     }
 608     if (flags & DBOP_KEY) {
 609         strlimcpy(dbop->prev, (char *)key.data, sizeof(dbop->prev));
 610         return (char *)key.data;
 611     }
 612     return ((char *)dat.data);
 613 }
 614 /**
 615  * dbop_next: get next record. 
 616  * 
 617  *  @param[in]  dbop    dbop descripter
 618  *  @return     data or NULL
 619  *
 620  * [Note] dbop_next() always skip meta records.
 621  */
 622 const char *
 623 dbop_next(DBOP *dbop)
 624 {
 625     DB *db = dbop->db;
 626     int flags = dbop->ioflags;
 627     DBT key, dat;
 628     int status;
 629 
 630     if (dbop->unread) {
 631         dbop->unread = 0;
 632         return dbop->lastdat;
 633     }
 634 #ifdef USE_SQLITE3
 635     if (dbop->openflags & DBOP_SQLITE3)
 636         return dbop3_next(dbop);
 637 #endif
 638     while ((status = (*db->seq)(db, &key, &dat, R_NEXT)) == RET_SUCCESS) {
 639         dbop->readcount++;
 640         assert(dat.data != NULL);
 641         /* skip meta records */
 642         if (!(dbop->openflags & DBOP_RAW)) {
 643             if (flags & DBOP_KEY && ismeta(key.data))
 644                 continue;
 645             else if (ismeta(dat.data))
 646                 continue;
 647         }
 648         if (flags & DBOP_KEY) {
 649             if (!strcmp(dbop->prev, (char *)key.data))
 650                 continue;
 651             if (strlen((char *)key.data) > MAXKEYLEN)
 652                 die("primary key too long.");
 653             strlimcpy(dbop->prev, (char *)key.data, sizeof(dbop->prev));
 654         }
 655         dbop->lastdat   = (char *)dat.data;
 656         dbop->lastsize  = dat.size;
 657         dbop->lastkey = (char *)key.data;
 658         dbop->lastkeysize = key.size;
 659         if (flags & DBOP_PREFIX) {
 660             if (strncmp((char *)key.data, dbop->key, dbop->keylen))
 661                 return NULL;
 662         } else if (dbop->keylen) {
 663             if (strcmp((char *)key.data, dbop->key))
 664                 return NULL;
 665         }
 666         if (dbop->preg && regexec(dbop->preg, (char *)key.data, 0, 0, 0) != 0)

 669     }
 670     if (status == RET_ERROR)
 671         die("dbop_next failed.");
 672     return NULL;
 673 }

 788 /**
 789  * dbop_close: close db
 790  * 
 791  *  @param[in]  dbop    dbop descripter
 792  */
 793 void
 794 dbop_close(DBOP *dbop)
 795 {
 796     DB *db = dbop->db;
 797 
 798     /*
 799      * Load sorted tag records and write them to the tag file.
 800      */
 801     if (dbop->sortout != NULL) {
 802         STRBUF *sb = strbuf_open(256);
 803         char *p;
 804 
 805         /*
 806          * End of the former stage of sorted writing.
 807          * fclose() and sortout = NULL is important.
 808          *
 809          * fclose(): enables reading from sortin descriptor.
 810          * sortout = NULL: makes the following dbop_put write to the tag file directly.
 811          */
 812         fclose(dbop->sortout);
 813         dbop->sortout = NULL;
 814         /*
 815          * The last stage of sorted writing.
 816          */
 817         while (strbuf_fgets(sb, dbop->sortin, STRBUF_NOCRLF)) {
 818             for (p = strbuf_value(sb); *p && *p != SORT_SEP; p++)
 819                 ;
 820             if (!*p)
 821                 die("unexpected end of record.");
 822             *p++ = '\0';
 823             dbop_put(dbop, strbuf_value(sb), p);
 824         }
 825         fclose(dbop->sortin);
 826         strbuf_close(sb);
 827         terminate_sort_process(dbop);
 828     }
 829 #ifdef USE_SQLITE3
 830     if (dbop->openflags & DBOP_SQLITE3) {
 831         dbop3_close(dbop);
 832         return;
 833     }
 834 #endif
 835 #ifdef USE_DB185_COMPAT
 836     (void)db->close(db);
 837 #else
 838     /*
 839      * If dbname = NULL, omit writing to the disk in __bt_close().
 840      */
 841     (void)db->close(db, dbop->dbname[0] == '\0' ? 1 : 0);
 842 #endif
 843     if (dbop->dbname[0] != '\0') {
 844         if (dbop->perm && chmod(dbop->dbname, dbop->perm) < 0)
 845             die("chmod(2) failed.");
 846     }
 847     (void)free(dbop);
 848 }

 850 DBOP *
 851 dbop3_open(const char *path, int mode, int perm, int flags) {
 852     int rc, rw = 0;
 853     char *errmsg = 0;
 854     DBOP *dbop;
 855     sqlite3 *db3;
 856     const char *tblname;
 857     int cache_size = 0;
 858     STRBUF *sql = strbuf_open_tempbuf();
 859     char buf[1024];
 860 
 861     /*
 862      * When the path is NULL string and private, temporary file is used.
 863      * The database will be removed when the session is closed.
 864      */
 865     if (path == NULL) {
 866         path = "";
 867         tblname = "temp";
 868     } else {
 869         /*
 870          * In case of creation.
 871          */
 872         if (mode == 1) {
 873 #ifndef _WIN32
 874             (void)truncate(path, 0);
 875 #else
 876             FILE* f = fopen(path, "w");
 877             if (f)
 878                 fclose(f);
 879 #endif
 880         }
 881         tblname = "db";
 882     }
 883     /*
 884      * setup arguments.
 885      */
 886     switch (mode) {
 887     case 0:
 888         rw = SQLITE_OPEN_READONLY;
 889         break;
 890     case 1:
 891         rw = SQLITE_OPEN_CREATE|SQLITE_OPEN_READWRITE;
 892         break;
 893     case 2:
 894         rw = SQLITE_OPEN_READWRITE;
 895         break;
 896     default:
 897         assert(0);
 898     }
 899     /*
 900      * When the fourth argument is NULL, sqlite3_vfs is used.
 901      */
 902     rc = sqlite3_open_v2(path, &db3, rw, NULL);
 903     if (rc != SQLITE_OK)
 904         die("sqlite3_open_v2 failed. (rc = %d)", rc);
 905     dbop = (DBOP *)check_calloc(sizeof(DBOP), 1);
 906     strlimcpy(dbop->dbname, path, sizeof(dbop->dbname));
 907     dbop->sb        = strbuf_open(0);
 908     dbop->db3       = db3;
 909     dbop->openflags = flags;
 910     dbop->perm  = (mode == 1) ? perm : 0;
 911     dbop->mode      = mode;
 912     dbop->lastdat   = NULL;
 913     dbop->lastflag  = NULL;
 914     dbop->lastsize  = 0;
 915     dbop->sortout   = NULL;
 916     dbop->sortin    = NULL;
 917     dbop->stmt      = NULL;
 918     dbop->tblname   = check_strdup(tblname);
 919     /*
 920      * Maximum file size is DBOP_PAGESIZE * 2147483646.
 921      * if DBOP_PAGESIZE == 8192 then maximum file size is 17592186028032 (17T).
 922      */
 923     snprintf(buf, sizeof(buf), "pragma page_size=%d", DBOP_PAGESIZE);
 924     rc = sqlite3_exec(dbop->db3, buf,  NULL, NULL, &errmsg);
 925     if (rc != SQLITE_OK)
 926         die("pragma page_size error: %s", errmsg);
 927     /*
 928      * create table (GTAGS, GRTAGS, GSYMS, GPATH).
 929      */
 930     if (mode == 1) {
 931         /* drop table */
 932         strbuf_clear(sql);
 933         strbuf_puts(sql, "drop table ");
 934         strbuf_puts(sql, dbop->tblname);
 935         rc = sqlite3_exec(dbop->db3, strbuf_value(sql), NULL, NULL, &errmsg);
 936             if (rc != SQLITE_OK) {
 937             /* ignore */
 938         }
 939         /* create table */
 940         strbuf_clear(sql);
 941         strbuf_puts(sql, "create table ");
 942         strbuf_puts(sql, dbop->tblname);
 943         strbuf_puts(sql, " (key text, dat text, extra text");
 944         if (!(flags & DBOP_DUP))
 945             strbuf_puts(sql, ", primary key(key)");
 946         strbuf_putc(sql, ')');
 947         rc = sqlite3_exec(dbop->db3, strbuf_value(sql), NULL, NULL, &errmsg);
 948             if (rc != SQLITE_OK)
 949             die("create table error: %s", errmsg);
 950     }
 951     /*
 952     rc = sqlite3_exec(dbop->db3, "pragma synchronous=off", NULL, NULL, &errmsg);
 953         if (rc != SQLITE_OK)
 954         die("pragma synchronous=off error: %s", errmsg);
 955     */
 956     /*
 957          * Decide cache size.
 958          * See libutil/gparam.h for the details.
 959          */
 960     cache_size = GTAGSCACHE;
 961     if (getenv("GTAGSCACHE") != NULL)
 962         cache_size = atoi(getenv("GTAGSCACHE"));
 963     if (cache_size < GTAGSMINCACHE)
 964         cache_size = GTAGSMINCACHE;
 965     cache_size = (cache_size + DBOP_PAGESIZE - 1) / DBOP_PAGESIZE;
 966     snprintf(buf, sizeof(buf), "pragma cache_size=%d", cache_size);
 967     rc = sqlite3_exec(dbop->db3, buf,  NULL, NULL, &errmsg);
 968         if (rc != SQLITE_OK)
 969         die("pragma cache_size error: %s", errmsg);
 970     sqlite3_exec(dbop->db3, "pragma journal_mode=memory", NULL, NULL, &errmsg);
 971         if (rc != SQLITE_OK)
 972         die("pragma journal_mode=memory error: %s", errmsg);
 973     sqlite3_exec(dbop->db3, "pragma synchronous=off", NULL, NULL, &errmsg);
 974         if (rc != SQLITE_OK)
 975         die("pragma synchronous=off error: %s", errmsg);
 976     rc = sqlite3_exec(dbop->db3, "begin transaction", NULL, NULL, &errmsg);
 977         if (rc != SQLITE_OK)
 978         die("begin transaction error: %s", errmsg);
 979     strbuf_release_tempbuf(sql);
 980     return dbop;
 981 }

1041 void
1042 dbop3_put(DBOP *dbop, const char *p1, const char *p2, const char *p3) {
1043     int rc, len;
1044     char *errmsg = 0;
1045     STRBUF *sql = strbuf_open_tempbuf();
1046 
1047     if (!(len = strlen(p1)))
1048         die("primary key size == 0.");
1049     if (len > MAXKEYLEN)
1050         die("primary key too long.");
1051     if (dbop->stmt_put3 == NULL) {
1052         strbuf_sprintf(sql, "insert into %s values (?, ?, ?)", dbop->tblname);
1053         rc = sqlite3_prepare_v2(dbop->db3, strbuf_value(sql), -1, &dbop->stmt_put3, NULL);
1054         if (rc != SQLITE_OK) {
1055             die("dbop3_put prepare failed. (rc = %d, sql = %s)", rc, strbuf_value(sql));
1056         }
1057     }
1058     rc = sqlite3_bind_text(dbop->stmt_put3, 1, p1, -1, SQLITE_STATIC);
1059         if (rc != SQLITE_OK) {
1060         die("dbop3_put 1 failed. (rc = %d)", rc);
1061     }
1062     rc = sqlite3_bind_text(dbop->stmt_put3, 2, p2, -1, SQLITE_STATIC);
1063         if (rc != SQLITE_OK) {
1064         die("dbop3_put 2 failed. (rc = %d)", rc);
1065     }
1066     rc = sqlite3_bind_text(dbop->stmt_put3, 3, p3, -1, SQLITE_STATIC);
1067         if (rc != SQLITE_OK) {
1068         die("dbop3_put 3 failed. (rc = %d)", rc);
1069     }
1070     rc = sqlite3_step(dbop->stmt_put3);
1071         if (rc != SQLITE_DONE) {
1072         die("dbop3_put failed. (rc = %d)", rc);
1073     }
1074     rc = sqlite3_reset(dbop->stmt_put3);
1075         if (rc != SQLITE_OK) {
1076         die("dbop3_put reset failed. (rc = %d)", rc);
1077     }
1078     if (dbop->writecount++ > DBOP_COMMIT_THRESHOLD) {
1079         dbop->writecount = 0;
1080         rc = sqlite3_exec(dbop->db3, "end transaction", NULL, NULL, &errmsg);
1081         if (rc != SQLITE_OK)
1082             die("end transaction error: %s", errmsg);
1083         rc = sqlite3_exec(dbop->db3, "begin transaction", NULL, NULL, &errmsg);
1084         if (rc != SQLITE_OK)
1085             die("begin transaction error: %s", errmsg);
1086     }
1087     strbuf_release_tempbuf(sql);
1088 }

1143 const char *
1144 dbop3_first(DBOP *dbop, const char *name, regex_t *preg, int flags) {
1145     int rc;
1146     char *key;
1147     STRBUF *sql = strbuf_open_tempbuf();
1148 
1149     dbop->done = 0;     /* This is turned on when it receives SQLITE_DONE. */
1150     strbuf_puts(sql, "select rowid, * from ");
1151     strbuf_puts(sql, dbop->tblname);
1152     if (name) {
1153         strbuf_puts(sql, " where key ");
1154         if (dbop->ioflags & DBOP_PREFIX) {
1155             /*
1156              * In sqlite3, 'like' ignores case. 'glob' does not ignore case.
1157              */
1158             strbuf_puts(sql, "glob '");
1159             strbuf_puts(sql, name);
1160             strbuf_puts(sql, "*'");
1161         } else {
1162             strbuf_puts(sql, "= '");
1163             strbuf_puts(sql, name);
1164             strbuf_puts(sql, "'");
1165         }
1166         strlimcpy(dbop->key, name, sizeof(dbop->key));
1167         dbop->keylen = strlen(name);
1168     }
1169     strbuf_puts(sql, " order by key");
1170     if (dbop->stmt) {
1171         rc = sqlite3_finalize(dbop->stmt);
1172         if (rc != SQLITE_OK)
1173             die("dbop3_finalize failed. (rc = %d)", rc);
1174         dbop->stmt = NULL;
1175     }
1176     rc = sqlite3_prepare_v2(dbop->db3, strbuf_value(sql), -1, &dbop->stmt, NULL);
1177     if (rc != SQLITE_OK)
1178         die("dbop3_first: sqlite3_prepare_v2 failed. (rc = %d)", rc);
1179     /*
1180      *  0: rowid
1181      *  1: key
1182      *  2: dat
1183      *  3: flags
1184      */
1185     for (;;) {
1186         /* Once it receives SQLITE_DONE, do not never return value */
1187         if (dbop->done)
1188             return NULL;
1189         rc = sqlite3_step(dbop->stmt);
1190         if (rc == SQLITE_DONE)
1191             goto finish;
1192         else if (rc == SQLITE_ROW) {
1193             dbop->readcount++;
1194             dbop->lastrowid = sqlite3_column_int64(dbop->stmt, 0);
1195             key = (char *)sqlite3_column_text(dbop->stmt, 1);
1196             if (name) {
1197                 if (dbop->ioflags & DBOP_PREFIX) {
1198                     if (strncmp(key, dbop->key, dbop->keylen))
1199                         goto finish;
1200                 } else {
1201                     if (strcmp(key, dbop->key)) 
1202                         goto finish;
1203                 }
1204                 if (dbop->preg && regexec(dbop->preg, key, 0, 0, 0) != 0)
1205                     continue;
1206             } else {
1207                 /* skip meta records */
1208                 if (ismeta(key) && !(dbop->openflags & DBOP_RAW))
1209                     continue;
1210                 if (dbop->preg && regexec(dbop->preg, key, 0, 0, 0) != 0)
1211                     continue;
1212             }
1213             break;
1214         } else {
1215             die("dbop3_first: something is wrong (rc = %d).", rc);
1216         }
1217     }
1218     strbuf_clear(dbop->sb);
1219     strbuf_puts0(dbop->sb, (char *)sqlite3_column_text(dbop->stmt, 2));
1220     dbop->lastsize = strbuf_getlen(dbop->sb) - 1;
1221     dbop->lastflag = (char *)sqlite3_column_text(dbop->stmt, 3);
1222     if (dbop->lastflag)
1223         strbuf_puts(dbop->sb, dbop->lastflag);
1224     dbop->lastdat = strbuf_value(dbop->sb);
1225     if (dbop->lastflag)
1226         dbop->lastflag = dbop->lastdat + dbop->lastsize + 1;
1227     dbop->lastkey = key;
1228     dbop->lastkeysize = strlen(dbop->lastkey);
1229     strbuf_release_tempbuf(sql);
1230     if (flags & DBOP_KEY) {
1231         strlimcpy(dbop->prev, key, sizeof(dbop->prev));
1232         return key;
1233     }
1234     return dbop->lastdat;
1235 finish:
1236     strbuf_release_tempbuf(sql);
1237     dbop->done = 1;
1238     dbop->lastdat = NULL;
1239     dbop->lastsize = 0;
1240     dbop->lastflag = NULL;
1241     return dbop->lastdat;
1242 }
1243 const char *
1244 dbop3_next(DBOP *dbop) {
1245     int rc;
1246     char *key, *dat;
1247 
1248     /*
1249      *  0: rowid
1250      *  1: key
1251      *  2: dat
1252      *  3: flags
1253      */
1254     for (;;) {
1255         /* Once it receives SQLITE_DONE, do not never return value */
1256         if (dbop->done)
1257             return NULL;
1258         rc = sqlite3_step(dbop->stmt);
1259         if (rc == SQLITE_DONE)
1260             goto finish;
1261         else if (rc == SQLITE_ROW) {
1262             dbop->readcount++;
1263             dbop->lastrowid = sqlite3_column_int64(dbop->stmt, 0);
1264             key = (char *)sqlite3_column_text(dbop->stmt, 1);
1265             dat = (char *)sqlite3_column_text(dbop->stmt, 2);
1266             /* skip meta records */
1267             if (!(dbop->openflags & DBOP_RAW)) {
1268                 if (dbop->ioflags & DBOP_KEY && ismeta(key))
1269                     continue;
1270                 else if (ismeta(dat))
1271                     continue;
1272             }
1273             if (dbop->ioflags & DBOP_KEY) {

1275                     continue;

1277                     die("primary key too long.");

1279             }
1280             if (dbop->ioflags & DBOP_PREFIX) {
1281                 if (strncmp(key, dbop->key, dbop->keylen))
1282                     goto finish;
1283             } else if (dbop->keylen) {
1284                 if (strcmp(key, dbop->key)) 
1285                     goto finish;
1286             }
1287             if (dbop->preg && regexec(dbop->preg, key, 0, 0, 0) != 0)
1288                 continue;
1289             break;
1290         } else {
1291             die("dbop3_next: something is wrong (rc = %d).", rc);
1292         }
1293     }
1294     strbuf_clear(dbop->sb);
1295     strbuf_puts0(dbop->sb, (char *)sqlite3_column_text(dbop->stmt, 2));
1296     dbop->lastsize = strbuf_getlen(dbop->sb) - 1;
1297     dbop->lastflag = (char *)sqlite3_column_text(dbop->stmt, 3);
1298     if (dbop->lastflag)
1299         strbuf_puts(dbop->sb, dbop->lastflag);
1300     dbop->lastdat = strbuf_value(dbop->sb);
1301     if (dbop->lastflag)
1302         dbop->lastflag = dbop->lastdat + dbop->lastsize + 1;
1303     dbop->lastkey = key;
1304     dbop->lastkeysize = strlen(dbop->lastkey);
1305     if (dbop->ioflags & DBOP_KEY) {
1306         strlimcpy(dbop->prev, key, sizeof(dbop->prev));
1307         return key;
1308     }
1309     return dbop->lastdat;
1310 finish:
1311     dbop->done = 1;
1312     dbop->lastdat = NULL;
1313     dbop->lastsize = 0;
1314     return dbop->lastdat;
1315 }
