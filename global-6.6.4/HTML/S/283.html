  46 typedef struct {
  47     /*
  48      * (1) COMMON PART
  49      */
  50     int mode;           /**< 0:read, 1:create, 2:modify */
  51     int openflags;          /**< flags of xxxx_open() */
  52     int ioflags;            /**< flags of xxxx_first() */
  53     char *lastdat;          /**< the data of last located record */
  54     int lastsize;           /**< the size of the lastdat */
  55     char *lastkey;          /**< the key of last located record */
  56     int lastkeysize;        /**< the size of the key */
  57     regex_t *preg;          /**< compiled regular expression */
  58     int unread;         /**< leave record to read again */
  59     const char *put_errmsg;     /**< error message for put_xxx() */
  60 
  61     /*
  62      * (2) DB185 PART
  63      */
  64     DB *db;             /**< descripter of DB */
  65     char dbname[MAXPATHLEN];    /**< dbname */
  66     char key[MAXKEYLEN];        /**< key */
  67     int keylen;         /**< key length */
  68     char prev[MAXKEYLEN];       /**< previous key value */
  69     int perm;           /**< file permission */
  70 
  71     /*
  72      * (3) sorted write
  73      */
  74     FILE *sortout;          /**< write to sort command */
  75     FILE *sortin;           /**< read from sort command */
  76 #if defined(_WIN32) && !defined(__CYGWIN__)
  77     HANDLE pid;
  78 #else
  79     int pid;            /**< sort process id */
  80 #endif
  81 #ifdef USE_SQLITE3
  82     /*
  83      * (4) sqlite3 part
  84      */
  85     sqlite3 *db3;
  86     STRBUF *sb;
  87     int done;
  88     const char *tblname;
  89     sqlite3_stmt *stmt;
  90     sqlite3_stmt *stmt_put3;
  91     sqlite3_int64 lastrowid;
  92     char *lastflag;
  93 #endif
  94     /** statistics */
  95     int readcount;
  96 #ifdef USE_SQLITE3
  97     /** for commit */
  98     int writecount;
  99 #endif
 100 } DBOP;
