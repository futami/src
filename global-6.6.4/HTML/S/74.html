 140     mdisprefs = lastdispline - REFLINE + 1;

 157 void
 158 display(void)
 159 {
 160     char    *subsystem;             /* OGS subsystem name */
 161     char    *book;                  /* OGS book name */
 162     char    file[PATHLEN + 1];      /* file name */
 163     char    function[PATLEN + 1];   /* function name */
 164     char    linenum[NUMLEN + 1];    /* line number */
 165     int     screenline;             /* screen line number */
 166     int     width;                  /* source line display width */
 167     int     i;
 168     char    *s;
 169 
 170     /* see if this is the initial display */
 171     erase();
 172     if (refsfound == NULL) {
 173 #if CCS
 174     if (displayversion == YES) {
 175         printw("cscope %s", ESG_REL);
 176     }
 177     else {
 178         printw("cscope");
 179     }
 180 #else
 181     printw("Gtags-cscope (based on cscope version %d%s)", FILEVERSION, FIXVERSION);
 182 #endif
 183     move(0, COLS - (int) sizeof(helpstring));
 184     addstr(helpstring);
 185     } else if (totallines == 0) {
 186     /* if no references were found */
 187     /* redisplay the last message */
 188     addstr(lastmsg);
 189     } else {
 190     /* display the pattern */
 191     if (changing == YES) {
 192         printw("Change \"%s\" to \"%s\"", Pattern, newpat);
 193     } else {
 194         printw("%c%s: %s", toupper((unsigned char)fields[field].text2[0]),
 195            fields[field].text2 + 1, Pattern);
 196     }
 197     /* display the column headings */
 198     move(2, 2);
 199     if (ogs == YES && field != FILENAME) {
 200         printw("%-*s ", subsystemlen, "Subsystem");
 201         printw("%-*s ", booklen, "Book");
 202     }
 203     if (dispcomponents > 0)
 204         printw("%-*s ", filelen, "File");
 205 
 206     if (field == SYMBOL || field == CALLEDBY || field == CALLING) {
 207         printw("%-*s ", fcnlen, "Function");
 208     }
 209     if (field != FILENAME) {
 210         addstr("Line");
 211     }
 212     addch('\n');
 213 
 214     /* if at end of file go back to beginning */
 215     if (nextline > totallines) {
 216         seekline(1);
 217     }
 218     /* calculate the source text column */
 219 
 220     width = COLS - numlen - 3;
 221 
 222     if (ogs == YES) {
 223         width -= subsystemlen + booklen + 2;
 224     }
 225     if (dispcomponents > 0) {
 226         width -= filelen + 1;
 227     }
 228     if (field == SYMBOL || field == CALLEDBY || field == CALLING) {
 229         width -= fcnlen + 1;
 230     }
 231 
 232     /* until the max references have been displayed or 
 233        there is no more room */
 234     topline = nextline;
 235     for (disprefs = 0, screenline = REFLINE;
 236          disprefs < mdisprefs && screenline <= lastdispline;
 237          ++disprefs, ++screenline) {
 238         /* read the reference line */
 239         if (fscanf(refsfound, "%" PATHLEN_STR "s%" PATHLEN_STR "s%" NUMLEN_STR "s %" TEMPSTRING_LEN_STR "[^\n]", file, function, 
 240                linenum, tempstring) < 4) {
 241         break;
 242         }
 243         ++nextline;
 244         displine[disprefs] = screenline;
 245             
 246         /* if no mouse, display the selection number */
 247         if (mouse == YES) {
 248         addch(' ');
 249         } else {
 250         printw("%c", dispchars[disprefs]);
 251         }
 252 
 253         /* display any change mark */
 254         if (changing == YES && 
 255         change[topline + disprefs - 1] == YES) {
 256         addch('>');
 257         } else {
 258         addch(' ');
 259         }
 260 
 261         /* display the file name */
 262         if (field == FILENAME) {
 263         printw("%-*s ", filelen, file);
 264         } else {
 265         /* if OGS, display the subsystem and book names */
 266         if (ogs == YES) {
 267             ogsnames(file, &subsystem, &book);
 268             printw("%-*.*s ", subsystemlen, subsystemlen, subsystem);
 269             printw("%-*.*s ", booklen, booklen, book);
 270         }
 271         /* display the requested path components */
 272         if (dispcomponents > 0) {
 273             printw("%-*.*s ", filelen, filelen,
 274                pathcomponents(file, dispcomponents));
 275         }
 276         } /* else(field == FILENAME) */
 277 
 278         /* display the function name */
 279         if (field == SYMBOL || field == CALLEDBY || field == CALLING) {
 280         printw("%-*.*s ", fcnlen, fcnlen, function);
 281         }
 282         if (field == FILENAME) {
 283         addch('\n');    /* go to next line */
 284         continue;
 285         }
 286 
 287         /* display the line number */
 288         printw("%*s ", numlen, linenum);
 289         /* there may be tabs in egrep output */
 290         while ((s = strchr(tempstring, '\t')) != NULL) {
 291         *s = ' ';
 292         }
 293 
 294         /* display the source line */
 295         s = tempstring;
 296         for (;;) {
 297         /* see if the source line will fit */
 298         if ((i = strlen(s)) > width) {
 299                     
 300             /* find the nearest blank */
 301             for (i = width; s[i] != ' ' && i > 0; --i) {
 302             ;
 303             }
 304             if (i == 0) {
 305             i = width;  /* no blank */
 306             }
 307         }
 308         /* print up to this point */
 309         printw("%.*s", i, s);
 310         s += i;
 311                 
 312         /* if line didn't wrap around */
 313         if (i < width) {
 314             addch('\n');    /* go to next line */
 315         }
 316         /* skip blanks */
 317         while (*s == ' ') {
 318             ++s;
 319         }
 320         /* see if there is more text */
 321         if (*s == '\0') {
 322             break;
 323         }
 324         /* if the source line is too long */
 325         if (++screenline > lastdispline) {
 326 
 327             /* if this is the first displayed line,
 328                display what will fit on the screen */
 329             if (topline == nextline -1) {
 330             disprefs++;
 331             /* break out of two loops */
 332             goto endrefs;
 333             }
 334                     
 335             /* erase the reference */
 336             while (--screenline >= displine[disprefs]) {
 337             move(screenline, 0);
 338             clrtoeol();
 339             }
 340             ++screenline;
 341                      
 342             /* go back to the beginning of this reference */
 343             --nextline;
 344             seekline(nextline);
 345             goto endrefs;
 346         }
 347         /* indent the continued source line */
 348         move(screenline, COLS - width);
 349         } /* for(ever) */
 350     } /* for(reference output lines) */
 351     endrefs:
 352     /* position the cursor for the message */
 353     i = FLDLINE - 1;
 354     if (screenline < i) {
 355         addch('\n');
 356     }
 357     else {
 358         move(i, 0);
 359     }
 360     /* check for more references */
 361     i = totallines - nextline + 1;
 362     bottomline = nextline;
 363     if (i > 0) {
 364         s = "s";
 365         if (i == 1) {
 366         s = "";
 367         }
 368         printw("* %d more line%s - press the space bar to display more *", i, s);
 369     }
 370     /* if this is the last page of references */
 371     else if (topline > 1 && nextline > totallines) {
 372         addstr("* Press the space bar to display the first lines again *");
 373     }
 374     }
 375     /* display the input fields */
 376     move(FLDLINE, 0);
 377     for (i = 0; i < FIELDS; ++i) {
 378     printw("%s %s:\n", fields[i].text1, fields[i].text2);
 379     }
 380     /* display any prompt */
 381     if (changing == YES) {
 382     move(PRLINE, 0);
 383     addstr(selprompt);
 384     }
 385     drawscrollbar(topline, nextline);   /* display the scrollbar */
 386     refresh();
 387 }

 427 BOOL
 428 search(void)
 429 {
 430     char    *findresult = NULL; /* find function output */
 431     sighandler_t savesig;       /* old value of signal */
 432     FP  f;          /* searching function */
 433     int c;
 434 #ifdef _WIN32
 435     DWORD   savemode;
 436 #endif
 437     
 438     /* open the references found file for writing */
 439     if (writerefsfound() == NO) {
 440         return(NO);
 441     }
 442 #ifdef _WIN32
 443     /* close it again, otherwise the redirection won't work */
 444     (void) fclose(refsfound);
 445     /* remember the current console mode, since the system */
 446     /* call resets it, thus preventing ctrl(c) from working */
 447     GetConsoleMode(GetStdHandle(STD_INPUT_HANDLE), &savemode);
 448 #endif
 449     /* find the pattern - stop on an interrupt */
 450     if (linemode == NO) {
 451         postmsg("Searching ...");
 452     }
 453     searchcount = 0;
 454     savesig = signal(SIGINT, jumpback);
 455     if (sigsetjmp(env, 1) == 0) {
 456         f = fields[field].findfcn;
 457         findresult = (*f)(Pattern);
 458     }
 459     signal(SIGINT, savesig);
 460 
 461 #ifdef _WIN32
 462     SetConsoleMode(GetStdHandle(STD_INPUT_HANDLE), savemode);
 463 #else
 464     /* reopen the references found file for reading */
 465     (void) fclose(refsfound);
 466 #endif
 467     if ((refsfound = myfopen(temp1, "r")) == NULL) {
 468         cannotopen(temp1);
 469         return(NO);
 470     }
 471     nextline = 1;
 472     totallines = 0;
 473     disprefs = 0;
 474     
 475     /* see if it is empty */
 476     if ((c = getc(refsfound)) == EOF) {
 477         if (findresult != NULL) {
 478             (void) snprintf(lastmsg, sizeof(lastmsg), "Could not find the %s: %s [%s]",
 479                        fields[field].text2, Pattern, findresult);
 480         } else {
 481             (void) snprintf(lastmsg, sizeof(lastmsg), "Could not find the %s: %s", 
 482                        fields[field].text2, Pattern);
 483         }
 484         return(NO);
 485     }
 486     /* put back the character read */
 487     (void) ungetc(c, refsfound);
 488 
 489     /* HBB 20041027: this used to hold a copy of the code of 
 490      * countrefs(), but with the crucial display width adjustments
 491      * missing.  Just call the real thing instead! */
 492     countrefs();
 493     return(YES);
 494 }

 498 void
 499 progress(char *what, long current, long max)
 500 {
 501     static  long    start;
 502     long    now;
 503     char    msg[MSGLEN + 1];
 504     int i;
 505 
 506     /* save the start time */
 507     if (searchcount == 0) {
 508         start = time(NULL);
 509     }
 510     if ((now = time(NULL)) - start >= 1)
 511     {
 512         if (linemode == NO)
 513         {
 514             move(MSGLINE, 0);
 515             clrtoeol();
 516             addstr(what);
 517             snprintf(msg, sizeof(msg), "%ld", current);
 518             move(MSGLINE, (COLS / 2) - (strlen(msg) / 2));
 519             addstr(msg);
 520             snprintf(msg, sizeof(msg), "%ld", max);
 521             move(MSGLINE, COLS - strlen(msg));
 522             addstr(msg);
 523             refresh();
 524         }
 525         else if (verbosemode == YES)
 526         {
 527             snprintf(msg, sizeof(msg), "> %s %ld of %ld", what, current, max);
 528         }
 529 
 530         start = now;
 531         if ((linemode == NO) && (incurses == YES))
 532         {
 533             move(MSGLINE, 0);
 534             i = (float)COLS * (float)current / (float)max;
 535 
 536             standout();
 537             for (; i > 0; i--)
 538                 addch(inch());
 539             standend();
 540             refresh();
 541         }
 542         else
 543             if (linemode == NO || verbosemode == YES)
 544                 postmsg(msg);
 545     }
 546     ++searchcount;
 547 }

 663 void
 664 seekline(unsigned int line) 
 665 {
 666     int c;
 667 
 668     /* verify that there is a references found file */
 669     if (refsfound == NULL) {
 670         return;
 671     }
 672     /* go to the beginning of the file */
 673     rewind(refsfound);
 674     
 675     /* find the requested line */
 676     nextline = 1;
 677     while (nextline < line && (c = getc(refsfound)) != EOF) {
 678         if (c == '\n') {
 679             nextline++;
 680         }
 681     }
 682 }

 684 /** get the OGS subsystem and book names */
 685 
 686 void
 687 ogsnames(char *file, char **subsystem, char **book)
 688 {
 689     static  char    buf[PATHLEN + 1];
 690     char    *s, *slash;
 691 
 692     *subsystem = *book = "";
 693     (void) strcpy(buf,file);
 694     s = buf;
 695     if (*s == '/') {
 696         ++s;
 697     }
 698     while ((slash = strchr(s, '/')) != NULL) {
 699         *slash = '\0';
 700         if ((int)strlen(s) >= 3 && strncmp(slash - 3, ".ss", 3) == 0) {
 701             *subsystem = s;
 702             s = slash + 1;
 703             if ((slash = strchr(s, '/')) != NULL) {
 704                 *book = s;
 705                 *slash = '\0';
 706             }
 707             break;
 708         }
 709         s = slash + 1;
 710     }
 711 }

 713 /** get the requested path components */
 714 
 715 char *
 716 pathcomponents(char *path, int components)
 717 {
 718     int i;
 719     char    *s;
 720     
 721     s = path + strlen(path) - 1;
 722     for (i = 0; i < components; ++i) {
 723         while (s > path && *--s != '/') {
 724             ;
 725         }
 726     }
 727     if (s > path && *s == '/') {
 728         ++s;
 729     }
 730     return(s);
 731 }

 735 BOOL
 736 writerefsfound(void)
 737 {
 738     if (refsfound == NULL) {
 739         if ((refsfound = myfopen(temp1, "wb")) == NULL) {
 740             cannotopen(temp1);
 741             return(NO);
