 147 static const char *
 148 seekto(const char *string, int n)
 149 {
 150     const char *p = string;
 151     while (n--) {
 152         p = strchr(p, ' ');
 153         if (p == NULL)
 154             return NULL;
 155         p++;
 156     }
 157     return p;

 159 /*
 160  * Tag format
 161  *
 162  * [Specification of format version 6]
 163  * 
 164  * Standard format:
 165  *
 166  *  This format is the default format of GTAGS.
 167  * 
 168  *         <file id> <tag name> <line number> <line image>
 169  * 
 170  *                 * Separator is single blank.
 171  * 
 172  *         [example]
 173  *         +------------------------------------
 174  *         |110 func 10 int func(int a)
 175  *         |110 func 30 func(int a1, int a2)
 176  * 
 177  *         Line image might be compressed (GTAGS_COMPRESS).
 178  *         Tag name might be compressed (GTAGS_COMPNAME).
 179  *
 180  * Compact format:
 181  * 
 182  *  This format is the default format of GRTAGS.
 183  *  It is used for GTAGS with the -c option.
 184  *
 185  *         <file id> <tag name> <line number>,...
 186  * 
 187  *                 * Separator is single blank.
 188  * 
 189  *         [example]
 190  *         +------------------------------------
 191  *         |110 func 10,30
 192  * 
 193  *         Line numbers are sorted in a line.
 194  *     Each line number might be expressed as difference from the previous
 195  *     line number except for the head (GTAGS_COMPLINE).
 196  *           ex: 10,3,2 means '10 13 15'.
 197  *     In addition,successive line numbers are expressed as a range.
 198  *           ex: 10-3 means '10 11 12 13'.
 199  *
 200  * [Description]
 201  * 
 202  * - Standard format is applied to GTAGS, and compact format is applied
 203  *     to GRTAGS by default.
 204  * - GSYMS is not used any longer. It is virtually included by GRTAGS.
 205  * - Above two formats are same to the first line number. So, we can use
 206  *     common function to sort them.
 207  * - Separator is single blank.
 208  *     This decrease disk space used a little, and make it easy to parse
 209  *     tag record.
 210  * - Use file id instead of path name.
 211  *     This allows blanks in path name at least in tag files.
 212  * - Put file id at the head of tag record.
 213  *     We can access file id without string processing.
 214  *     This is advantageous for deleting tag record when incremental updating.
 215  * 
 216  * [Concept of format version]
 217  *
 218  * Since GLOBAL's tag files are machine independent, they can be distributed
 219  * apart from GLOBAL itself. For example, if some network file system available,
 220  * client may execute global using server's tag files. In this case, both
 221  * GLOBAL are not necessarily the same version. So, we should assume that
 222  * older version of GLOBAL might access the tag files which generated
 223  * by new GLOBAL. To deal in such case, we decided to buried a version number
 224  * to both global(1) and tag files. The conclete procedure is like follows:
 225  *
 226  *  1. gtags(1) bury the version number in tag files.
 227  *  2. global(1) pick up the version number from a tag file. If the number
 228  *      is larger than its acceptable version number then global give up work
 229  *      any more and display error message.
 230  *  3. If version number is not found then it assumes version 1.
 231  *
 232  * [History of format version]
 233  *
 234   GLOBAL-1.0 - 1.8     no idea about format version.
 235   GLOBAL-1.9 - 2.24    understand format version.
 236                        support format version 1 (default).
 237                        if (format > 1) then print error message.
 238   GLOBAL-3.0 - 4.5     support format version 1 and 2.
 239                        if (format > 2) then print error message.
 240   GLOBAL-4.5.1 - 4.8.7 support format version 1, 2 and 3.
 241                        if (format > 3) then print error message.
 242   GLOBAL-5.0 -  5.3 support format version only 4.
 243                        if (format !=  4) then print error message.
 244   GLOBAL-5.4 - 5.8.2    support format version 4 and 5
 245                        if (format > 5 || format < 4) then print error message.
 246   GLOBAL-5.9 -      support only format version 6
 247                        if (format > 6 || format < 6) then print error message.
 248  *
 249  * In GLOBAL-5.0, we threw away the compatibility with the past formats.
 250  * Though we could continue the support for older formats, it seemed
 251  * not to be worthy. Because keeping maintaining the all formats hinders
 252  * new optimization and the function addition in the future.
 253  * Instead, the following error messages are displayed in a wrong usage.
 254  *
 255  *       [older global and new tag file]
 256  *       $ global -x main
 257  *       GTAGS seems new format. Please install the latest GLOBAL.
 258  *       [new global and older tag file]
 259  *       $ global -x main
 260  *       GTAGS seems older format. Please remake tag files.
 261  */
 262 static int new_format_version = 6;  /**< new format version */
 263 static int upper_bound_version = 6; /**< acceptable format version (upper bound) */
 264 static int lower_bound_version = 6; /**< acceptable format version (lower bound) */
 265 static const char *const tagslist[] = {"GPATH", "GTAGS", "GRTAGS", "GSYMS"};
 266 /**
 267  * Virtual GRTAGS, GSYMS processing:
 268  *
 269  * We use a real GRTAGS as virtual GRTAGS and GSYMS.
 270  * In fact, GSYMS tag file doesn't exist.
 271  *
 272  * Real tag file    virtual tag file
 273  * --------------------------------------
 274  * GTAGS =============> GTAGS
 275  *
 276  * GRTAGS ============> GRTAGS + GSYMS
 277  *            +=======> GRTAGS  tags which is defined in GTAGS
 278  *            +=======> GSYMS   tags which is not defined in GTAGS
 279  */
 280 #define VIRTUAL_GRTAGS_GSYMS_PROCESSING(gtop)                       \
 281     if (gtop->db == GRTAGS || gtop->db == GSYMS) {                  \
 282         int defined = is_defined_in_GTAGS(gtop, gtop->dbop->lastkey);       \
 283         if ((gtop->db == GRTAGS && !defined) || (gtop->db == GSYMS && defined)) \
 284             continue;                           \
 285     }
 286 /**
 287  * is_defined_in_GTAGS: whether or not the name is defined in GTAGS.
 288  *
 289  *  @param[in]  gtop
 290  *  @param[in]  name    tag name
 291  *  @return     0: not defined, 1: defined
 292  *
 293  * [Note] It is assumed that the input stream is sorted by the tag name.
 294  */
 295 static int
 296 is_defined_in_GTAGS(GTOP *gtop, const char *name)
 297 {
 298     static char prev_name[MAXTOKEN+1];
 299     static int prev_result;
 300 
 301     if (!strcmp(name, prev_name))
 302         return prev_result;
 303     strlimcpy(prev_name, name, sizeof(prev_name));
 304     return prev_result = dbop_get(gtop->gtags, prev_name) ? 1 : 0;
 305 }

 320 /**
 321  * gtags_open: open global tag.
 322  *
 323  *  @param[in]  dbpath  dbpath directory
 324  *  @param[in]  root    root directory (needed when compact format)
 325  *  @param[in]  db  GTAGS, GRTAGS, GSYMS
 326  *  @param[in]  mode    GTAGS_READ: read only,
 327  *          GTAGS_CREATE: create tag,
 328  *          GTAGS_MODIFY: modify tag
 329  *  @param[in]  flags   GTAGS_COMPACT: compact format
 330  *  @return     GTOP structure
 331  *
 332  * [Note] when error occurred, gtags_open() doesn't return.
 333  */
 334 GTOP *
 335 gtags_open(const char *dbpath, const char *root, int db, int mode, int flags)
 336 {
 337     GTOP *gtop;
 338     char tagfile[MAXPATHLEN];
 339     int dbmode;
 340     int dbop_flags = DBOP_DUP;
 341 
 342     gtop = (GTOP *)check_calloc(sizeof(GTOP), 1);
 343     gtop->db = db;
 344     gtop->mode = mode;
 345     gtop->openflags = flags;
 346     /*
 347      * Open tag file allowing duplicate records.
 348      */
 349     switch (gtop->mode) {
 350     case GTAGS_READ:
 351         dbmode = 0;
 352         break;
 353     case GTAGS_CREATE:
 354         dbmode = 1;
 355         break;
 356     case GTAGS_MODIFY:
 357         dbmode = 2;
 358         break;
 359     default:
 360         assert(0);
 361     }
 362 #ifdef USE_SQLITE3
 363     /*
 364      * Don't use sorted writing with sqlite, because it is slow on the contrary.
 365      */
 366     if (flags & GTAGS_SQLITE3) {
 367         dbop_flags |= DBOP_SQLITE3;
 368         set_gpath_flags(DBOP_SQLITE3);
 369     } else
 370 #endif
 371         dbop_flags |= DBOP_SORTED_WRITE;
 372     /*
 373      * GRTAGS and GSYMS are virtual tag file. They are included in a real GRTAGS file.
 374      * In fact, GSYMS doesn't exist now.
 375      *
 376      * GRTAGS:  tags which belongs to GRTAGS, and are defined in GTAGS.
 377      * GSYMS:   tags which belongs to GRTAGS, and is not defined in GTAGS.
 378      */
 379     strlimcpy(tagfile, makepath(dbpath, dbname(db == GSYMS ? GRTAGS : db), NULL), sizeof(tagfile));
 380     gtop->dbop = dbop_open(tagfile, dbmode, 0644, dbop_flags);
 381     if (gtop->dbop == NULL) {
 382         if (dbmode == 1)
 383             die("cannot make %s.", dbname(db));
 384         else if (errno == EFTYPE)
 385             die("%s seems corrupted.", tagfile);
 386         else
 387             die("%s not found.", dbname(db));
 388     }
 389     if (gtop->mode == GTAGS_READ && db != GTAGS) {
 390         const char *gtags = makepath(dbpath, dbname(GTAGS), NULL);
 391         int format_version;
 392 
 393         gtop->gtags = dbop_open(gtags, 0, 0, 0);
 394         if (gtop->gtags == NULL)
 395             die("GTAGS not found.");
 396         format_version = dbop_getversion(gtop->dbop);
 397         if (format_version > upper_bound_version)
 398             die("%s seems new format. Please install the latest GLOBAL.", gtags);
 399         else if (format_version < lower_bound_version)
 400             die("%s seems older format. Please remake tag files.", gtags);
 401     }
 402     if (gtop->mode == GTAGS_CREATE) {
 403         /*
 404          * Decide format.
 405          */
 406         gtop->format = 0;
 407         gtop->format_version = new_format_version;
 408         /*
 409          * GRTAGS and GSYSM always use compact format.
 410          * GTAGS uses compact format only when the -c option specified.
 411          */
 412         if (gtop->db == GRTAGS || gtop->db == GSYMS || gtop->openflags & GTAGS_COMPACT) {
 413             gtop->format |= GTAGS_COMPACT;
 414             gtop->format |= GTAGS_COMPLINE;
 415         } else {
 416             /* standard format */
 417             gtop->format |= GTAGS_COMPRESS;
 418         }
 419         gtop->format |= GTAGS_COMPNAME;
 420         if (gtop->format & GTAGS_COMPACT)
 421             dbop_putoption(gtop->dbop, COMPACTKEY, NULL);
 422         if (gtop->format & GTAGS_COMPRESS) {
 423             dbop_putoption(gtop->dbop, COMPRESSKEY, DEFAULT_ABBREVIATION);
 424             abbrev_open(DEFAULT_ABBREVIATION);
 425         }
 426         if (gtop->format & GTAGS_COMPLINE)
 427             dbop_putoption(gtop->dbop, COMPLINEKEY, NULL);
 428         if (gtop->format & GTAGS_COMPNAME)
 429             dbop_putoption(gtop->dbop, COMPNAMEKEY, NULL);
 430         dbop_putversion(gtop->dbop, gtop->format_version); 
 431     } else {
 432         /*
 433          * recognize format version of GTAGS. 'format version record'
 434          * is saved as a META record in GTAGS and GRTAGS.
 435          * if 'format version record' is not found, it's assumed
 436          * version 1.
 437          */
 438         const char *p;
 439         /*
 440          * check format version.
 441          */
 442         gtop->format_version = dbop_getversion(gtop->dbop);
 443         if (gtop->format_version > upper_bound_version)
 444             die("%s seems new format. Please install the latest GLOBAL.", tagfile);
 445         else if (gtop->format_version < lower_bound_version)
 446             die("%s seems older format. Please remake tag files.", tagfile);
 447         gtop->format = 0;
 448         if (dbop_getoption(gtop->dbop, COMPACTKEY) != NULL)
 449             gtop->format |= GTAGS_COMPACT;
 450         if ((p = dbop_getoption(gtop->dbop, COMPRESSKEY)) != NULL) {
 451             abbrev_open(p);
 452             gtop->format |= GTAGS_COMPRESS;
 453         }
 454         if (dbop_getoption(gtop->dbop, COMPLINEKEY) != NULL)
 455             gtop->format |= GTAGS_COMPLINE;
 456         if (dbop_getoption(gtop->dbop, COMPNAMEKEY) != NULL)
 457             gtop->format |= GTAGS_COMPNAME;
 458     }
 459     if (gpath_open(dbpath, dbmode) < 0) {
 460         if (dbmode == 1)
 461             die("cannot create GPATH.");
 462         else
 463             die("GPATH not found.");
 464     }
 465     if (gtop->mode != GTAGS_READ)
 466         gtop->sb = strbuf_open(0);  /* This buffer is used for working area. */
 467     /*
 468      * Stuff for compact format.
 469      */
 470     if (gtop->format & GTAGS_COMPACT) {
 471         assert(root != NULL);
 472         strlimcpy(gtop->root, root, sizeof(gtop->root));
 473         if (gtop->mode != GTAGS_READ)
 474             gtop->path_hash = strhash_open(HASHBUCKETS);
 475     }
 476     gtop->sb_compress = strbuf_open(0);
 477     return gtop;
 478 }
 479 /**
 480  * gtags_put_using: put tag record with packing.
 481  *
 482  *  @param[in]  gtop    descripter of GTOP
 483  *  @param[in]  tag tag name
 484  *  @param[in]  lno line number
 485  *  @param[in]  fid file id
 486  *  @param[in]  img line image
 487  */
 488 void
 489 gtags_put_using(GTOP *gtop, const char *tag, int lno, const char *fid, const char *img)
 490 {
 491     const char *key;
 492 
 493     if (gtop->format & GTAGS_COMPACT) {
 494         struct sh_entry *entry;
 495 
 496         /*
 497          * Register each record into the pool.
 498          *
 499          * Pool image:
 500          *
 501          * tagname   lno
 502          * ------------------------------
 503          * "funcA"   | 1| 3| 7|23|11| 2|...
 504          * "funcB"   |34| 2| 5|66| 3|...
 505          * ...
 506          */
 507         entry = strhash_assign(gtop->path_hash, tag, 1);
 508         if (entry->value == NULL)
 509             entry->value = varray_open(sizeof(int), 100);
 510         *(int *)varray_append((VARRAY *)entry->value) = lno;
 511         return;
 512     }
 513     /*
 514      * extract method when class method definition.
 515      *
 516      * Ex: Class::method(...)
 517      *
 518      * key  = 'method'
 519      * data = 'Class::method  103 ./class.cpp ...'
 520      */
 521     if (gtop->flags & GTAGS_EXTRACTMETHOD) {
 522         if ((key = locatestring(tag, ".", MATCH_LAST)) != NULL)
 523             key++;
 524         else if ((key = locatestring(tag, "::", MATCH_LAST)) != NULL)
 525             key += 2;
 526         else
 527             key = tag;
 528     } else {
 529         key = tag;
 530     }
 531     strbuf_reset(gtop->sb);
 532     strbuf_puts(gtop->sb, fid);
 533     strbuf_putc(gtop->sb, ' ');
 534     strbuf_puts(gtop->sb, (gtop->format & GTAGS_COMPNAME) ? compress(tag, key, gtop->sb_compress) : tag);
 535     strbuf_putc(gtop->sb, ' ');
 536     strbuf_putn(gtop->sb, lno);
 537     strbuf_putc(gtop->sb, ' ');
 538     strbuf_puts(gtop->sb, (gtop->format & GTAGS_COMPRESS) ? compress(img, key, gtop->sb_compress) : img);
 539     dbop_put_tag(gtop->dbop, key, strbuf_value(gtop->sb));
 540 }
 541 /**

 555 /**
 556  * gtags_delete: delete records belong to set of fid.
 557  *
 558  *  @param[in]  gtop    GTOP structure
 559  *  @param[in]  deleteset bit array of fid
 560  */
 561 void
 562 gtags_delete(GTOP *gtop, IDSET *deleteset)
 563 {
 564     const char *tagline;
 565     int fid;
 566 
 567 #ifdef USE_SQLITE3
 568     if (gtop->dbop->openflags & DBOP_SQLITE3) {
 569         STRBUF *where = strbuf_open(0);
 570         long id;
 571         strbuf_puts(where, "(");
 572         for (id = idset_first(deleteset); id != END_OF_ID; id = idset_next(deleteset)) {
 573             strbuf_puts(where, "'");
 574             strbuf_putn(where, id);
 575             strbuf_puts(where, "',");
 576         }
 577         strbuf_unputc(where, ',');
 578         strbuf_puts(where, ")");
 579         dbop_delete(gtop->dbop, strbuf_value(where));
 580         strbuf_close(where);
 581     } else
 582 #endif
 583     for (tagline = dbop_first(gtop->dbop, NULL, NULL, 0); tagline; tagline = dbop_next(gtop->dbop)) {
 584         /*
 585          * Extract path from the tag line.
 586          */
 587         fid = atoi(tagline);
 588         /*
 589          * If the file id exists in the deleteset, delete the tagline.
 590          */
 591         if (idset_contains(deleteset, fid))
 592             dbop_delete(gtop->dbop, NULL);
 593     }
 594 }
 595 /**
 596  * get_prefix: get as long prefix of the pattern as possible.
 597  *
 598  *  @param[in]  pattern
 599  *  @param[in]  flags for gtags_first()
 600  *  @return     prefix for search
 601  */
 602 static char *
 603 get_prefix(const char *pattern, int flags)
 604 {
 605     static char buffer[IDENTLEN];
 606     char *prefix = buffer;
 607 
 608     if (pattern == NULL || pattern[0] == 0) {
 609         prefix = NULL;
 610     } else if (flags & GTOP_NOREGEX || !isregex(pattern)) {
 611         if (flags & GTOP_IGNORECASE) {
 612             buffer[0] = toupper(*pattern);
 613             buffer[1] = 0;
 614         } else {
 615             prefix = NULL;
 616         }
 617     } else if (*pattern == '^') {
 618         int save = 0;
 619         char *p = (char *)(pattern + 1);
 620         char *q = locatestring(p, ".*$", MATCH_AT_LAST);
 621 
 622         if (!q)
 623             q = locatestring(pattern, "$", MATCH_AT_LAST);
 624         if (!q)
 625             q = locatestring(pattern, ".*", MATCH_AT_LAST);
 626         if (q) {
 627             save = *q;
 628             *q = 0;
 629         }
 630         if (*p == 0 || isregex(p)) {
 631             prefix = NULL;
 632         } else {
 633             if (flags & GTOP_IGNORECASE) {
 634                 prefix[0] = toupper(*p);
 635                 prefix[1] = 0;
 636             } else
 637                 strlimcpy(buffer, p, sizeof(buffer));
 638         }
 639         if (save)
 640             *q = save;
 641     }
 642     return prefix;
 643 }
 644 /**
 645  * gtags_restart: restart dbop iterator using lower case prefix.
 646  *
 647  *  @param[in]  gtop    GTOP structure
 648  *  @return     prepared or not
 649  *          0:  cannot continue
 650  *          1:  can continue
 651  */
 652 static int
 653 gtags_restart(GTOP *gtop)
 654 {
 655     int upper, lower;
 656 
 657     if (gtop->prefix == NULL)
 658         die("gtags_restart: impossible.");
 659     upper = gtop->prefix[0];
 660     lower = tolower(upper);
 661     if (upper < lower) {
 662         gtop->prefix[0] = lower;
 663         gtop->key = gtop->prefix;
 664         gtop->prefix = NULL;
 665         if (gtop->openflags & GTAGS_DEBUG)
 666             fprintf(stderr, "Using prefix: %s\n", gtop->key);
 667         return 1;
 668     }
 669     if (gtop->openflags & GTAGS_DEBUG)
 670         fprintf(stderr, "gtags_restart: not prepared.\n");
 671     return 0;
 672 }
 673 /**
 674  * gtags_first: return first record
 675  *
 676  *  @param[in]  gtop    GTOP structure
 677  *  @param[in]  pattern tag name,
 678  *      - may be regular expression
 679  *      - may be NULL
 680  *  @param[in]  flags   GTOP_PREFIX:    prefix read,
 681  *          GTOP_KEY:   read key only,
 682  *          GTOP_PATH:  read path only,
 683  *          GTOP_NOREGEX:   don't use regular expression.
 684  *          GTOP_IGNORECASE:    ignore case distinction.
 685  *          GTOP_BASICREGEX:    use basic regular expression.
 686  *          GTOP_NEARSORT:      use 'Nearness sort'.
 687  *          GTOP_NOSORT:        don't sort
 688  *
 689  *          By default, sort is done by alphabetical order.
 690  *  @return     record
 691  */
 692 GTP *
 693 gtags_first(GTOP *gtop, const char *pattern, int flags)
 694 {
 695     int regflags = 0;
 696     static regex_t reg;
 697     const char *tagline;
 698     STATIC_STRBUF(regex);
 699 
 700     strbuf_clear(regex);
 701     gtop->preg = &reg;
 702     gtop->key = NULL;
 703     gtop->prefix = NULL;
 704     gtop->flags = flags;
 705     gtop->dbflags = 0;
 706     gtop->readcount = 1;
 707 
 708     /* Settlement for last time if any */
 709     if (gtop->path_hash) {
 710         strhash_close(gtop->path_hash);
 711         gtop->path_hash = NULL;
 712     }
 713     if (gtop->path_array) {
 714         free(gtop->path_array);
 715         gtop->path_array = NULL;
 716     }
 717 
 718     if (flags & GTOP_KEY)
 719         gtop->dbflags |= DBOP_KEY;
 720     if (!(flags & GTOP_BASICREGEX))
 721         regflags |= REG_EXTENDED;
 722     if (flags & GTOP_NEARSORT)
 723         if ((nearbase = get_nearbase_path()) == NULL)
 724             die("cannot get nearbase path.");
 725     /*
 726      * decide a read method
 727      */
 728     if (pattern == NULL)
 729         gtop->preg = NULL;
 730     else if (pattern[0] == 0)
 731         return NULL;
 732     else if (!strcmp(pattern, ".*") || !strcmp(pattern, "^.*$") ||
 733         !strcmp(pattern, "^") || !strcmp(pattern, "$") ||
 734         !strcmp(pattern, "^.*") || !strcmp(pattern, ".*$")) {
 735         /*
 736          * Since these regular expressions match to any record,
 737          * we take sequential read method.
 738          */
 739         gtop->preg = NULL;
 740     } else if (flags & GTOP_IGNORECASE) {
 741         regflags |= REG_ICASE;
 742         if (flags & GTOP_NOREGEX || !isregex(pattern)) {
 743             gtop->prefix = get_prefix(pattern, flags);
 744             if (gtop->openflags & GTAGS_DEBUG)
 745                 if (gtop->prefix != NULL)
 746                     fprintf(stderr, "Using prefix: %s\n", gtop->prefix);
 747             if (gtop->prefix == NULL)
 748                 die("gtags_first: impossible (1).");
 749             strbuf_putc(regex, '^');
 750             strbuf_puts(regex, quote_string(pattern));
 751             if (!(flags & GTOP_PREFIX))
 752                 strbuf_putc(regex, '$');
 753         } else if (*pattern == '^' && (gtop->prefix = get_prefix(pattern, flags)) != NULL) {
 754             if (gtop->openflags & GTAGS_DEBUG)
 755                 fprintf(stderr, "Using prefix: %s\n", gtop->prefix);
 756             strbuf_puts(regex, pattern);
 757         } else {
 758             strbuf_puts(regex, pattern);
 759         }
 760     } else {
 761         if (flags & GTOP_NOREGEX || !isregex(pattern)) {
 762             if (flags & GTOP_PREFIX)
 763                 gtop->dbflags |= DBOP_PREFIX;
 764             gtop->key = pattern;
 765             gtop->preg = NULL;
 766         } else if (*pattern == '^' && (gtop->key = get_prefix(pattern, flags)) != NULL) {
 767             if (gtop->openflags & GTAGS_DEBUG)
 768                 fprintf(stderr, "Using prefix: %s\n", gtop->key);
 769             gtop->dbflags |= DBOP_PREFIX;
 770             gtop->preg = NULL;
 771         } else {
 772             strbuf_puts(regex, pattern);
 773         }
 774     }
 775     if (gtop->prefix) {
 776         if (gtop->key)
 777             die("gtags_first: impossible (2).");
 778         gtop->key = gtop->prefix;
 779         gtop->dbflags |= DBOP_PREFIX;
 780     }
 781     if (strbuf_getlen(regex) > 0) {
 782         if (gtop->preg == NULL)
 783             die("gtags_first: impossible (3).");
 784         if (regcomp(gtop->preg, strbuf_value(regex), regflags) != 0)
 785             die("invalid regular expression.");
 786     }
 787     /*
 788      * If GTOP_PATH is set, at first, we collect all path names in a pool and
 789      * sort them. gtags_first() and gtags_next() returns one of the pool.
 790      */
 791     if (gtop->flags & GTOP_PATH) {
 792         struct sh_entry *entry;
 793         char *p;
 794         const char *cp;
 795         unsigned long i;
 796 
 797         gtop->path_hash = strhash_open(HASHBUCKETS);
 798         /*
 799          * Pool path names.
 800          *
 801          * fid      path name
 802          * +--------------------------
 803          * |100     ./aaa/a.c
 804          * |105     ./aaa/b.c
 805          *  ...
 806          */
 807 again0:
 808         for (tagline = dbop_first(gtop->dbop, gtop->key, gtop->preg, gtop->dbflags);
 809              tagline != NULL;
 810              tagline = dbop_next(gtop->dbop))
 811         {
 812             VIRTUAL_GRTAGS_GSYMS_PROCESSING(gtop);
 813             /* extract file id */
 814             p = locatestring(tagline, " ", MATCH_FIRST);
 815             if (p == NULL)
 816                 die("Invalid tag record. '%s'\n", tagline);
 817             *p = '\0';
 818             entry = strhash_assign(gtop->path_hash, tagline, 1);
 819             /* new entry: get path name and set. */
 820             if (entry->value == NULL) {
 821                 cp = gpath_fid2path(tagline, NULL);
 822                 if (cp == NULL)
 823                     die("GPATH is corrupted.(file id '%s' not found)", tagline);
 824                 entry->value = strhash_strdup(gtop->path_hash, cp, 0);
 825             }
 826         }
 827         if (gtop->prefix && gtags_restart(gtop))
 828             goto again0;
 829         /*
 830          * Sort path names.
 831          *
 832          * fid      path name   path_array (sort)
 833          * +--------------------------  +---+
 834          * |100     ./aaa/a.c <-------* |
 835          * |105     ./aaa/b.c <-------* |
 836          *  ...             ...
 837          */
 838         gtop->path_array = (char **)check_malloc(gtop->path_hash->entries * sizeof(char *));
 839         i = 0;
 840         for (entry = strhash_first(gtop->path_hash); entry != NULL; entry = strhash_next(gtop->path_hash))
 841             gtop->path_array[i++] = entry->value;
 842         if (i != gtop->path_hash->entries)
 843             die("Something is wrong. 'i = %lu, entries = %lu'" , i, gtop->path_hash->entries);
 844         if (!(gtop->flags & GTOP_NOSORT))
 845             qsort(gtop->path_array, gtop->path_hash->entries, sizeof(char *),
 846                 gtop->flags & GTOP_NEARSORT ? compare_nearpath : compare_path);
 847         gtop->path_count = gtop->path_hash->entries;
 848         gtop->path_index = 0;
 849 
 850         if (gtop->path_index >= gtop->path_count)
 851             return NULL;
 852         gtop->gtp.path = gtop->path_array[gtop->path_index++];
 853         return &gtop->gtp;
 854     } else if (gtop->flags & GTOP_KEY) {
 855 again1:
 856         for (gtop->gtp.tag = dbop_first(gtop->dbop, gtop->key, gtop->preg, gtop->dbflags);
 857              gtop->gtp.tag != NULL;
 858              gtop->gtp.tag = dbop_next(gtop->dbop))
 859         {
 860             VIRTUAL_GRTAGS_GSYMS_PROCESSING(gtop);
 861             break;
 862         }
 863         if (gtop->gtp.tag == NULL) {
 864             if (gtop->prefix && gtags_restart(gtop))
 865                 goto again1;
 866         }
 867         return gtop->gtp.tag ? &gtop->gtp : NULL;
 868     } else {
 869         if (gtop->vb == NULL)
 870             gtop->vb = varray_open(sizeof(GTP), 200);
 871         else
 872             varray_reset(gtop->vb);
 873         if (gtop->segment_pool == NULL)
 874             gtop->segment_pool = pool_open();
 875         else
 876             pool_reset(gtop->segment_pool);
 877         if (gtop->path_hash == NULL)
 878             gtop->path_hash = strhash_open(HASHBUCKETS);
 879         else
 880             strhash_reset(gtop->path_hash);
 881 again2:
 882         for (tagline = dbop_first(gtop->dbop, gtop->key, gtop->preg, gtop->dbflags);
 883             tagline != NULL;
 884             tagline = dbop_next(gtop->dbop))
 885         {
 886             VIRTUAL_GRTAGS_GSYMS_PROCESSING(gtop);
 887             break;
 888         }
 889         if (tagline == NULL) {
 890             if (gtop->prefix && gtags_restart(gtop))
 891                 goto again2;
 892             return NULL;
 893         }
 894         /*
 895          * Dbop_next() will read the same record again.
 896          */
 897         dbop_unread(gtop->dbop);
 898         /*
 899          * Read a tag segment with sorting.
 900          */
 901         segment_read(gtop);
 902         return  &gtop->gtp_array[gtop->gtp_index++];
 903     }
 904 }
 905 /**
 906  * gtags_next: return next record.
 907  *
 908  *  @param[in]  gtop    GTOP structure
 909  *  @return     record
 910  *          NULL end of tag
 911  */
 912 GTP *
 913 gtags_next(GTOP *gtop)
 914 {
 915     gtop->readcount++;
 916     if (gtop->flags & GTOP_PATH) {
 917         if (gtop->path_index >= gtop->path_count)
 918             return NULL;
 919         gtop->gtp.path = gtop->path_array[gtop->path_index++];
 920         return &gtop->gtp;
 921     } else if (gtop->flags & GTOP_KEY) {
 922         gtop->gtp.tag = dbop_next(gtop->dbop);
 923 again3:
 924         for (; gtop->gtp.tag != NULL; gtop->gtp.tag = dbop_next(gtop->dbop))
 925         {
 926             VIRTUAL_GRTAGS_GSYMS_PROCESSING(gtop);
 927             break;
 928         }
 929         if (gtop->gtp.tag == NULL) {
 930             if (gtop->prefix && gtags_restart(gtop)) {
 931                 gtop->gtp.tag = dbop_first(gtop->dbop, gtop->key, gtop->preg, gtop->dbflags);
 932                 goto again3;
 933             }
 934         }
 935         return gtop->gtp.tag ? &gtop->gtp : NULL;
 936     } else {
 937         /*
 938          * End of segment.
 939          * Reset resources and read new segment again.
 940          */
 941         if (gtop->gtp_index >= gtop->gtp_count) {
 942             varray_reset(gtop->vb);
 943             pool_reset(gtop->segment_pool);
 944             /* strhash_reset(gtop->path_hash); */
 945             segment_read(gtop);
 946         }
 947         if (gtop->gtp_index >= gtop->gtp_count) {
 948             if (gtop->prefix && gtags_restart(gtop)) {
 949                 gtop->gtp.tag = dbop_first(gtop->dbop, gtop->key, gtop->preg, gtop->dbflags);
 950                 if (gtop->gtp.tag == NULL)
 951                     return NULL;
 952                 dbop_unread(gtop->dbop);
 953                 segment_read(gtop);
 954             } else
 955                 return NULL;
 956         }
 957         return &gtop->gtp_array[gtop->gtp_index++];
 958     }
 959 }

 994 /**
 995  * flush_pool: flush and write the pool as compact format.
 996  *
 997  *  @param[in]  gtop    descripter of GTOP
 998  *  @param[in]  s_fid
 999  */
1000 static void
1001 flush_pool(GTOP *gtop, const char *s_fid)
1002 {
1003     struct sh_entry *entry;
1004     int header_offset;
1005     int i, last;
1006 
1007     if (s_fid == NULL)
1008         die("flush_pool: impossible");
1009     /*
1010      * Write records as compact format and free line number table
1011      * for each entry in the pool.
1012      */
1013     for (entry = strhash_first(gtop->path_hash); entry; entry = strhash_next(gtop->path_hash)) {
1014         VARRAY *vb = (VARRAY *)entry->value;
1015         int *lno_array = varray_assign(vb, 0, 0);
1016         const char *key = entry->name;
1017 
1018         /*
1019          * extract method when class method definition.
1020          *
1021          * Ex: Class::method(...)
1022          *
1023          * key  = 'method'
1024          * data = 'Class::method  103 ./class.cpp ...'
1025          */
1026         if (gtop->flags & GTAGS_EXTRACTMETHOD) {
1027             if ((key = locatestring(entry->name, ".", MATCH_LAST)) != NULL)
1028                 key++;
1029             else if ((key = locatestring(entry->name, "::", MATCH_LAST)) != NULL)
1030                 key += 2;
1031             else
1032                 key = entry->name;
1033         }
1034         /* Sort line number table */
1035         qsort(lno_array, vb->length, sizeof(int), compare_lineno); 
1036 
1037         strbuf_reset(gtop->sb);
1038         strbuf_puts(gtop->sb, s_fid);
1039         strbuf_putc(gtop->sb, ' ');
1040         if (gtop->format & GTAGS_COMPNAME) {
1041             strbuf_puts(gtop->sb, compress(entry->name, key, gtop->sb_compress));
1042         } else {
1043             strbuf_puts(gtop->sb, entry->name);
1044         }
1045         strbuf_putc(gtop->sb, ' ');
1046         header_offset = strbuf_getlen(gtop->sb);
1047         /*
1048          * If GTAGS_COMPLINE flag is set, each line number is expressed as the
1049          * difference from the previous line number except for the head.
1050          * GTAGS_COMPLINE is set by default in format version 5.
1051          */
1052         if (gtop->format & GTAGS_COMPLINE) {
1053             int cont = 0;
1054 
1055             last = 0;           /* line 0 doesn't exist */
1056             for (i = 0; i < vb->length; i++) {
1057                 int n = lno_array[i];
1058 
1059                 if (n == last)
1060                     continue;
1061                 if (last > 0 && n == last + 1) {
1062                     if (!cont) {
1063                         /*
1064                          * Don't use range expression at the head.
1065                          */
1066                         if (strbuf_getlen(gtop->sb) == header_offset)
1067                             strbuf_putn(gtop->sb, n);
1068                         else
1069                             cont = last;
1070                     }
1071                 } else {
1072                     /*
1073                      * Range expression. ex: 10-2 means 10 11 12
1074                      */
1075                     if (cont) {
1076                         strbuf_putc(gtop->sb, '-');
1077                         strbuf_putn(gtop->sb, last - cont);
1078                         cont = 0;
1079                     }
1080                     if (strbuf_getlen(gtop->sb) > header_offset) {
1081                         strbuf_putc(gtop->sb, ',');
1082                         strbuf_putn(gtop->sb, n - last);
1083                     } else {
1084                         strbuf_putn(gtop->sb, n);
1085                     }
1086                     if (strbuf_getlen(gtop->sb) > DBOP_PAGESIZE / 4) {
1087                         dbop_put_tag(gtop->dbop, key, strbuf_value(gtop->sb));
1088                         strbuf_setlen(gtop->sb, header_offset);
1089                     }
1090                 }
1091                 last = n;
1092             }
1093             if (cont) {
1094                 strbuf_putc(gtop->sb, '-');
1095                 strbuf_putn(gtop->sb, last - cont);
1096             }
1097         } else {
1098             /*
1099              * This code is to support older format (version 4).
1100              */
1101             last = 0;           /* line 0 doesn't exist */
1102             for (i = 0; i < vb->length; i++) {
1103                 int n = lno_array[i];
1104 
1105                 if (n == last)
1106                     continue;
1107                 if (strbuf_getlen(gtop->sb) > header_offset)
1108                     strbuf_putc(gtop->sb, ',');
1109                 strbuf_putn(gtop->sb, n);
1110                 if (strbuf_getlen(gtop->sb) > DBOP_PAGESIZE / 4) {
1111                     dbop_put_tag(gtop->dbop, key, strbuf_value(gtop->sb));
1112                     strbuf_setlen(gtop->sb, header_offset);
1113                 }
1114                 last = n;
1115             }
1116         }
1117         if (strbuf_getlen(gtop->sb) > header_offset) {
1118             dbop_put_tag(gtop->dbop, key, strbuf_value(gtop->sb));
1119         }
1120         /* Free line number table */
1121         varray_close(vb);
1122     }
1123 }
1124 /**
1125  * Read a tag segment with sorting.
1126  *
1127  *  @param[in]  gtop    GTOP structure
1128  *      Output: gtop->gtp_array     segment table
1129  *      Output: gtop->gtp_count     segment table size
1130  *      Output: gtop->gtp_index     segment table index (initial value = 0)
1131  *      Output: gtop->cur_tagname   current tag name
1132  *
1133  * A segment is a set of tag records which have same tag name.
1134  * This function read a segment from tag file, sort it and put it on segment table.
1135  * This function can treat both of standard format and compact format.
1136  *
1137  * Sorting is done by three keys.
1138  *  - 1st key: tag name
1139  *  - 2nd key: file name
1140  *  - 3rd key: line number
1141  *
1142  * Since all records in a segment have same tag name, you need not think about 1st key.
1143  */
1144 void
1145 segment_read(GTOP *gtop)
1146 {
1147     const char *tagline, *fid, *path, *lineno;
1148     GTP *gtp;
1149     struct sh_entry *sh;
1150 
1151     /*
1152      * Save tag lines.
1153      */
1154     gtop->cur_tagname[0] = '\0';
1155     while ((tagline = dbop_next(gtop->dbop)) != NULL) {
1156         VIRTUAL_GRTAGS_GSYMS_PROCESSING(gtop);
1157         /*
1158          * get tag name and line number.
1159          *
1160          * tagline = <file id> <tag name> <line number>
1161          */
1162         if (gtop->cur_tagname[0] == '\0') {
1163             strlimcpy(gtop->cur_tagname, gtop->dbop->lastkey, sizeof(gtop->cur_tagname));
1164         } else if (strcmp(gtop->cur_tagname, gtop->dbop->lastkey) != 0) {
1165             /*
1166              * Dbop_next() wil read the same record again.
1167              */
1168             dbop_unread(gtop->dbop);
1169             break;
1170         }
1171         gtp = varray_append(gtop->vb);
1172         gtp->tagline = pool_strdup(gtop->segment_pool, tagline, 0);
1173         gtp->tag = (const char *)gtop->cur_tagname;
1174         /*
1175          * convert fid into hashed path name to save memory.
1176          */
1177         fid = (const char *)strmake(tagline, " ");
1178         path = gpath_fid2path(fid, NULL);
1179         if (path == NULL)
1180             die("GPATH is corrupted.(file id '%s' not found)", fid);
1181         sh = strhash_assign(gtop->path_hash, path, 1);
1182         gtp->path = sh->name;
1183         lineno = seekto(gtp->tagline, SEEKTO_LINENO);
1184         if (lineno == NULL)
1185             die("invalid tag record.\n%s", tagline);
1186         gtp->lineno = atoi(lineno);
1187     }
1188     /*
1189      * Sort tag lines.
1190      */
1191     gtop->gtp_array = varray_assign(gtop->vb, 0, 0);
1192     gtop->gtp_count = gtop->vb->length;
1193     gtop->gtp_index = 0;
1194     if (!(gtop->flags & GTOP_NOSORT))
