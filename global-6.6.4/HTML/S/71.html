  82 /** execute the command */
  83 BOOL
  84 command(int commandc)
  85 {
  86     STATIC_STRBUF(sb);
  87     char filename[PATHLEN + 1]; /* file path name */
  88     MOUSE *p;           /* mouse data */
  89     int c, i;
  90     FILE *file;
  91     struct cmd *curritem, *item;    /* command history */
  92     char *s;
  93 
  94     switch (commandc) {
  95     case ctrl('C'): /* toggle caseless mode */
  96     if (caseless == NO) {
  97         caseless = YES;
  98         postmsg2("Caseless mode is now ON");
  99     } else {
 100         caseless = NO;
 101         postmsg2("Caseless mode is now OFF");
 102     }
 103     return(NO);
 104 
 105     case ctrl('R'): /* rebuild the cross reference */
 106     if (isuptodate == YES) {
 107         postmsg("The -d option prevents rebuilding the symbol database");
 108         return(NO);
 109     }
 110     postmsg("Rebuilding cross-reference...");
 111     rebuild();
 112     postmsg("");
 113     totallines = 0;
 114     disprefs = 0;   
 115     topline = nextline = 1;
 116     selecting = 0;
 117     break;
 118 
 119 #if UNIXPC
 120     case ESC:   /* possible unixpc mouse selection */
 121 #endif
 122     case ctrl('X'): /* mouse selection */
 123     if ((p = getmouseaction(DUMMYCHAR)) == NULL) {
 124         return(NO); /* unknown control sequence */
 125     }
 126     /* if the button number is a scrollbar tag */
 127     if (p->button == '0') {
 128         scrollbar(p);
 129         break;
 130     } 
 131     /* ignore a sweep */
 132     if (p->x2 >= 0) {
 133         return(NO);
 134     }
 135     /* if this is a line selection */
 136     if (p->y1 < FLDLINE) {
 137 
 138         /* find the selected line */
 139         /* note: the selection is forced into range */
 140         for (i = disprefs - 1; i > 0; --i) {
 141         if (p->y1 >= displine[i]) {
 142             break;
 143         }
 144         }
 145         /* display it in the file with the editor */
 146         editref(i);
 147     } else {    /* this is an input field selection */
 148         field = p->y1 - FLDLINE;
 149         /* force it into range */
 150         if (field >= FIELDS) {
 151         field = FIELDS - 1;
 152         }
 153         setfield();
 154         resetcmd();
 155         return(NO);
 156     }
 157     break;
 158 
 159     case '\t':  /* go to next input field */
 160     if (disprefs) {
 161         selecting = !selecting;
 162         if (selecting) {
 163         move(displine[curdispline], 0);
 164         refresh();
 165         } else {
 166         atfield();
 167         resetcmd();
 168         }
 169     }
 170     return(NO);
 171 
 172 #ifdef KEY_ENTER
 173     case KEY_ENTER:
 174 #endif
 175     case '\r':
 176     case '\n':  /* go to reference */
 177     if (selecting) {
 178         editref(curdispline);
 179         return(YES);
 180     }
 181     /* FALLTHROUGH */
 182 
 183     case ctrl('N'):
 184 #ifdef KEY_DOWN
 185     case KEY_DOWN:
 186 #endif      
 187 #ifdef KEY_RIGHT
 188     case KEY_RIGHT:
 189 #endif
 190     if (selecting) {
 191         if ((curdispline + 1) < disprefs) {
 192         move(displine[++curdispline], 0);
 193         refresh();
 194         }
 195     } else {
 196         field = (field + 1) % FIELDS;
 197         if (field == 2)
 198         field = 3;
 199         setfield();
 200         atfield();
 201         resetcmd();
 202     }
 203     return(NO);
 204 
 205     case ctrl('P'): /* go to previous input field */
 206 #ifdef KEY_UP
 207     case KEY_UP:
 208 #endif
 209 #ifdef KEY_LEFT     
 210     case KEY_LEFT:
 211 #endif
 212     if (selecting) {
 213         if (curdispline) {
 214         move(displine[--curdispline], 0);
 215         refresh();
 216         }
 217     } else {
 218         field = (field + (FIELDS - 1)) % FIELDS;
 219         if (field == 2)
 220         field = 1;
 221         setfield();
 222         atfield();
 223         resetcmd();
 224     }
 225     return(NO);
 226 #ifdef KEY_HOME
 227     case KEY_HOME:  /* go to first input field */
 228     if (selecting) {
 229         curdispline = 0;
 230         move(REFLINE, 0);
 231         refresh();
 232     } else {
 233         field = 0;
 234         setfield();
 235         atfield();
 236         resetcmd();
 237     }
 238     return(NO);
 239 #endif
 240 
 241 #ifdef KEY_LL
 242     case KEY_LL:    /* go to last input field */
 243     if (selecting) {
 244         move(displine[disprefs - 1], 0);
 245         refresh();
 246     } else {
 247         field = FIELDS - 1;
 248         setfield();
 249         atfield();
 250         resetcmd();
 251     }
 252     return(NO);
 253 #endif /* def(KEY_LL) */
 254 
 255     case ' ':   /* display next page */
 256     case '+':
 257     case ctrl('V'):
 258 #ifdef KEY_NPAGE
 259     case KEY_NPAGE:
 260 #endif
 261     /* don't redisplay if there are no lines */
 262     if (totallines == 0) {
 263         return(NO);
 264     }
 265     /* note: seekline() is not used to move to the next 
 266      * page because display() leaves the file pointer at
 267      * the next page to optimize paging forward
 268      */
 269     curdispline = 0;
 270     break;
 271 
 272     case ctrl('H'):
 273     case '-':   /* display previous page */
 274 #ifdef KEY_PPAGE
 275     case KEY_PPAGE:
 276 #endif
 277     /* don't redisplay if there are no lines */
 278     if (totallines == 0) {
 279         return(NO);
 280     }
 281 
 282     curdispline = 0;
 283 
 284     /* if there are only two pages, just go to the other one */
 285     if (totallines <= 2 * mdisprefs) {
 286         break;
 287     }
 288     /* if on first page but not at beginning, go to beginning */
 289     nextline -= mdisprefs;  /* already at next page */
 290     if (nextline > 1 && nextline <= mdisprefs) {
 291         nextline = 1;
 292     } else {
 293         nextline -= mdisprefs;
 294         if (nextline < 1) {
 295         nextline = totallines - mdisprefs + 1;
 296         if (nextline < 1) {
 297             nextline = 1;
 298         }
 299         }
 300     }
 301     seekline(nextline);
 302     break;
 303 
 304     case '>':   /* write or append the lines to a file */
 305     if (totallines == 0) {
 306         postmsg("There are no lines to write to a file");
 307     } else {    /* get the file name */
 308         move(PRLINE, 0);
 309         addstr("Write to file: ");
 310         s = "w";
 311         if ((c = mygetch()) == '>') {
 312         move(PRLINE, 0);
 313         addstr(appendprompt);
 314         c = '\0';
 315         s = "a";
 316         }
 317         if (c != '\r' && 
 318         mygetline("", newpat,
 319               COLS - sizeof(appendprompt), c, NO) > 0
 320         ) {
 321         shellpath(filename, sizeof(filename), newpat);
 322         if ((file = myfopen(filename, s)) == NULL) {
 323             cannotopen(filename);
 324         } else {
 325             seekline(1);
 326             while ((c = getc(refsfound)) != EOF) {
 327             putc(c, file);
 328             }
 329             seekline(topline);
 330             fclose(file);
 331         }
 332         }
 333         clearprompt();
 334     }
 335     return(NO); /* return to the previous field */
 336 
 337     case '<':   /* read lines from a file */
 338     move(PRLINE, 0);
 339     addstr(readprompt);
 340     if (mygetline("", newpat, COLS - sizeof(readprompt),
 341               '\0', NO) > 0) {
 342         clearprompt();
 343         shellpath(filename, sizeof(filename), newpat);
 344         if (readrefs(filename) == NO) {
 345         postmsg2("Ignoring an empty file");
 346         return(NO);
 347         }
 348         return(YES);
 349     }
 350     clearprompt();
 351     return(NO);
 352 
 353     case '^':   /* pipe the lines through a shell command */
 354     case '|':   /* pipe the lines to a shell command */
 355     if (totallines == 0) {
 356         postmsg("There are no lines to pipe to a shell command");
 357         return(NO);
 358     }
 359     /* get the shell command */
 360     move(PRLINE, 0);
 361     addstr(pipeprompt);
 362     if (mygetline("", newpat, COLS - sizeof(pipeprompt), '\0', NO)
 363         == 0) {
 364         clearprompt();
 365         return(NO);
 366     }
 367     /* if the ^ command, redirect output to a temp file */
 368     if (commandc == '^') {
 369         strcat(strcat(newpat, " >"), temp2);
 370         /* HBB 20020708: somebody might have even
 371          * their non-interactive default shells
 372          * complain about clobbering
 373          * redirections... --> delete before
 374          * overwriting */
 375         remove(temp2);
 376     }
 377     exitcurses();
 378     if ((file = mypopen(newpat, "w")) == NULL) {
 379         fprintf(stderr, "\
 380 cscope: cannot open pipe to shell command: %s\n", newpat);
 381     } else {
 382         seekline(1);
 383         while ((c = getc(refsfound)) != EOF) {
 384         putc(c, file);
 385         }
 386         seekline(topline);
 387         mypclose(file);
 388     }
 389     if (commandc == '^') {
 390         if (readrefs(temp2) == NO) {
 391         postmsg("Ignoring empty output of ^ command");
 392         }
 393     }
 394     askforreturn();
 395     entercurses();
 396     break;
 397     case ctrl('G'): /* read from the result of global command */
 398     /* get the shell command */
 399     move(PRLINE, 0);
 400     addstr(globalprompt);
 401     /* addstr(globaloption);*/
 402     if (mygetline("", newpat, COLS - sizeof(globalprompt), '\0', NO) == 0) {
 403         clearprompt();
 404         return(NO);
 405     }
 406     strbuf_clear(sb);
 407 #if defined(_WIN32) && !defined(__CYGWIN__)
 408     /*
 409      * Get around CMD.EXE's weird quoting rules by sticking another
 410      * perceived whitespace in front (also works with Take Command).
 411      */
 412     strbuf_putc(sb, ';');
 413 #endif
 414     strbuf_puts(sb, quote_shell(global_command));
 415     strbuf_sprintf(sb, " %s %s > %s", globaloption, quote_shell(newpat), temp2);
 416     remove(temp2);
 417     postmsg("Searching ...");
 418     if (system(strbuf_value(sb)) != 0) {
 419         postmsg("Ignoring failed output of global command");
 420         clearprompt();
 421         return(NO);
 422     }
 423     if (readrefs(temp2) == NO) {
 424         postmsg("Ignoring empty output of global command");
 425         clearprompt();
 426         return(NO);
 427     }
 428     curdispline = 0;
 429     return(YES);
 430     break;
 431 #if defined(KEY_RESIZE) && !defined(__DJGPP__)
 432     case KEY_RESIZE:
 433     exitcurses();
 434     initscr();
 435     entercurses();
 436 #if TERMINFO
 437     keypad(stdscr, TRUE);   /* enable the keypad */
 438 #ifdef HAVE_FIXKEYPAD
 439     fixkeypad();    /* fix for getch() intermittently returning garbage */
 440 #endif
 441 #endif
 442 #if UNIXPC
 443     standend(); /* turn off reverse video */
 444 #endif
 445     dispinit(); /* initialize display parameters */
 446     setfield(); /* set the initial cursor position */
 447     postmsg("");    /* clear any build progress message */
 448     display();  /* display the version number and input fields */
 449     break;
 450 #endif
 451     case ctrl('L'): /* redraw screen */
 452 #ifdef KEY_CLEAR
 453     case KEY_CLEAR:
 454 #endif
 455     clearmsg2();
 456     clearok(curscr, TRUE);
 457     wrefresh(curscr);
 458     drawscrollbar(topline, bottomline);
 459     return(NO);
 460 
 461     case '!':   /* shell escape */
 462     execute(shell, shell, NULL);
 463     seekline(topline);
 464     break;
 465 
 466     case '?':   /* help */
 467     clear();
 468     help();
 469     clear();
 470     seekline(topline);
 471     break;
 472 
 473     case ctrl('E'): /* edit all lines */
 474     editall();
 475     break;
 476 
 477     case ctrl('A'):     /* HBB 20050428: added alt. keymapping */
 478     case ctrl('Y'): /* repeat last pattern */
 479     if (*Pattern != '\0') {
 480         addstr(Pattern);
 481         goto repeat;
 482     }
 483     break;
 484 
 485     case ctrl('B'):     /* cmd history back */
 486     case ctrl('F'):     /* cmd history fwd */
 487     if (selecting)
 488         return(NO);
 489 
 490     curritem = currentcmd();
 491     item = (commandc == ctrl('F')) ? nextcmd() : prevcmd();
 492     clearmsg2();
 493     if (curritem == item) { /* inform user that we're at history end */
 494         postmsg2("End of input field and search pattern history");
 495     }
 496     if (item) {
 497         field = item->field;
 498         setfield();
 499         atfield();
 500         addstr(item->text);
 501         strcpy(Pattern, item->text);
 502         switch (c = mygetch()) {
 503         case '\r':
 504         case '\n':
 505         goto repeat;
 506         case ctrl('F'):
 507         case ctrl('B'):
 508         myungetch(c);
 509         atfield();
 510         clrtoeol(); /* clear current field */
 511         break;
 512         default:
 513         myungetch(c);
 514         if (mygetline(Pattern, newpat, COLS - fldcolumn - 1, '\0', caseless )) {
 515             strcpy (Pattern, newpat);
 516             resetcmd();
 517         }
 518         goto repeat;
 519         break;
 520         }
 521     }
 522     return(NO);
 523 
 524     case '\\':  /* next character is not a command */
 525     addch('\\');    /* display the quote character */
 526 
 527     /* get a character from the terminal */
 528     if ((commandc = mygetch()) == EOF) {
 529         return(NO); /* quit */
 530     }
 531     addstr("\b \b");    /* erase the quote character */
 532     goto ispat;
 533 
 534     case '.':
 535     postmsg("The . command has been replaced by ^Y");
 536     atfield();  /* move back to the input field */
 537     /* FALLTHROUGH */
 538     default:
 539     if (selecting && !mouse) {
 540         char        *c;
 541 
 542         if ((c = strchr(dispchars, commandc)))
 543         editref(c - dispchars);
 544 
 545         /* if this is the start of a pattern */
 546     } else if (isprint(commandc)) {
 547     ispat:
 548         if (mygetline("", newpat, COLS - fldcolumn - 1,
 549               commandc, caseless) > 0) {
 550         strcpy(Pattern, newpat);
 551         resetcmd(); /* reset command history */
 552         repeat:
 553         addcmd(field, Pattern); /* add to command history */
 554         if (field == CHANGE) {
 555             /* prompt for the new text */
 556             move(PRLINE, 0);
 557             addstr(toprompt);
 558             mygetline("", newpat,
 559                   COLS - sizeof(toprompt),
 560                   '\0', NO);
 561         }
 562         /* search for the pattern */
 563         if (search() == YES) {
 564             curdispline = 0;
 565             ++selecting;
 566 
 567             switch (field) {
 568             case DEFINITION:
 569             case FILENAME:
 570             if (totallines > 1) {
 571                 break;
 572             }
 573             topline = 1;
 574             editref(0);
 575             break;
 576             case CHANGE:
 577             return(changestring());
 578             }
 579 
 580         } else if (field == FILENAME && 
 581                access(newpat, READ) == 0) {
 582             /* try to edit the file anyway */
 583             edit(newpat, "1");
 584         }
 585         } else {    /* no pattern--the input was erased */
 586         return(NO);
 587         }
 588     } else {    /* control character */
 589         return(NO);
 590     }
 591     } /* switch(commandc) */
 592     return(YES);
 593 }

 604 /** read references from a file */
 605 
 606 BOOL
 607 readrefs(char *filename)
 608 {
 609     FILE    *file;
 610     int c;
 611 
 612     if ((file = myfopen(filename, "rb")) == NULL) {
 613         cannotopen(filename);
 614         return(NO);
 615     }
 616     if ((c = getc(file)) == EOF) {  /* if file is empty */
 617         return(NO);
 618     }
 619     totallines = 0;
 620     disprefs = 0;
 621     nextline = 1;
 622     if (writerefsfound() == YES) {
 623         putc(c, refsfound);
 624         while ((c = getc(file)) != EOF) {
 625             putc(c, refsfound);
 626         }
 627         fclose(file);
 628         fclose(refsfound);
 629         if ( (refsfound = myfopen(temp1, "rb")) == NULL) {
 630             cannotopen(temp1);
 631             return(NO);
 632         }
 633         countrefs();
 634     }
 635     return(YES);
 636 }

 638 /** change one text string to another */
 639 
 640 static BOOL
 641 changestring(void)
 642 {
 643     char    newfile[PATHLEN + 1];   /* new file name */
 644     char    oldfile[PATHLEN + 1];   /* old file name */
 645     char    linenum[NUMLEN + 1];    /* file line number */
 646     char    msg[MSGLEN + 1];        /* message */
 647     FILE    *script;                /* shell script file */
 648     BOOL    anymarked = NO;         /* any line marked */
 649     MOUSE *p;                       /* mouse data */
 650     int     c;
 651     unsigned int i;
 652     char    *s;
 653 
 654     /* open the temporary file */
 655     if ((script = myfopen(temp2, "w")) == NULL) {
 656     cannotopen(temp2);
 657     return(NO);
 658     }
 659     /* create the line change indicators */
 660     change = mycalloc(totallines, sizeof(BOOL));
 661     changing = YES;
 662     mousemenu();
 663 
 664     /* until the quit command is entered */
 665     for (;;) {
 666     /* display the current page of lines */
 667     display();
 668     same:
 669     atchange();
 670         
 671     /* get a character from the terminal */
 672     if ((c = mygetch()) == EOF
 673         || c == ctrl('D') 
 674         || c == ctrl('Z')) {
 675         break;  /* change lines */
 676     }
 677     /* see if the input character is a command */
 678     switch (c) {
 679     case ' ':   /* display next page */
 680     case '+':
 681     case ctrl('V'):
 682 #ifdef KEY_NPAGE
 683     case KEY_NPAGE:
 684 #endif
 685     case '-':   /* display previous page */
 686 #ifdef KEY_PPAGE
 687     case KEY_PPAGE:
 688 #endif
 689     case '!':   /* shell escape */
 690     case '?':   /* help */
 691         command(c);
 692         break;
 693 
 694     case ctrl('L'): /* redraw screen */
 695 #ifdef KEY_CLEAR
 696     case KEY_CLEAR:
 697 #endif
 698         command(c);
 699         goto same;
 700 
 701     case ESC:   /* don't change lines */
 702 #if UNIXPC
 703         if((p = getmouseaction(DUMMYCHAR)) == NULL) {
 704         goto nochange;  /* unknown escape sequence */
 705         }
 706         break;
 707 #endif
 708     case ctrl('G'):
 709         goto nochange;
 710 
 711     case '*':   /* mark/unmark all displayed lines */
 712         for (i = 0; topline + i < nextline; ++i) {
 713         mark(i);
 714         }
 715         goto same;
 716 
 717     case ctrl('A'): /* mark/unmark all lines */
 718         for (i = 0; i < totallines; ++i) {
 719         if (change[i] == NO) {
 720             change[i] = YES;
 721         } else {
 722             change[i] = NO;
 723         }
 724         }
 725         /* show that all have been marked */
 726         seekline(totallines);
 727         break;
 728 
 729     case ctrl('X'): /* mouse selection */
 730         if ((p = getmouseaction(DUMMYCHAR)) == NULL) {
 731         goto same;  /* unknown control sequence */
 732         }
 733         /* if the button number is a scrollbar tag */
 734         if (p->button == '0') {
 735         scrollbar(p);
 736         break;
 737         }
 738         /* find the selected line */
 739         /* note: the selection is forced into range */
 740         for (i = disprefs - 1; i > 0; --i) {
 741         if (p->y1 >= displine[i]) {
 742             break;
 743         }
 744         }
 745         mark(i);
 746         goto same;
 747 
 748     default:
 749         {
 750         /* if a line was selected */
 751         char        *cc;
 752 
 753         if ((cc = strchr(dispchars, c)))
 754             mark(cc - dispchars);
 755 
 756         goto same;
 757         } /* default case */
 758     } /* switch(change code character) */
 759     } /* for(ever) */
 760 
 761     /* for each line containing the old text */
 762     fprintf(script, "ed - <<\\!\n");
 763     *oldfile = '\0';
 764     seekline(1);
 765     for (i = 0; 
 766      fscanf(refsfound, "%" PATHLEN_STR "s%*s%" NUMLEN_STR "s%*[^\n]", newfile, linenum) == 2;
 767      ++i) {
 768     strcpy(newfile, decode_path(newfile));
 769     /* see if the line is to be changed */
 770     if (change[i] == YES) {
 771         anymarked = YES;
 772         
 773         /* if this is a new file */
 774         if (strcmp(newfile, oldfile) != 0) {
 775                 
 776         /* make sure it can be changed */
 777         if (access(newfile, WRITE) != 0) {
 778             snprintf(msg, sizeof(msg), "Cannot write to file %s", newfile);
 779             postmsg(msg);
 780             anymarked = NO;
 781             break;
 782         }
 783         /* if there was an old file */
 784         if (*oldfile != '\0') {
 785             fprintf(script, "w\n"); /* save it */
 786         }
 787         /* edit the new file */
 788         strcpy(oldfile, newfile);
 789         fprintf(script, "e %s\n", oldfile);
 790         }
 791         /* output substitute command */
 792         fprintf(script, "%ss/", linenum);   /* change */
 793         for (s = Pattern; *s != '\0'; ++s) {
 794         /* old text */
 795         if (strchr("/\\[.^*", *s) != NULL) {
 796             putc('\\', script);
 797         }
 798         if (caseless == YES && isalpha((unsigned char)*s)) {
 799             putc('[', script);
 800             if(islower((unsigned char)*s)) {
 801             putc(toupper((unsigned char)*s), script);
 802             putc(*s, script);
 803             } else {
 804             putc(*s, script);
 805             putc(tolower((unsigned char)*s), script);
 806             }
 807             putc(']', script);
 808         } else  
 809             putc(*s, script);
 810         }
 811         putc('/', script);          /* to */
 812         for (s = newpat; *s != '\0'; ++s) { /* new text */
 813         if (strchr("/\\&", *s) != NULL) {
 814             putc('\\', script);
 815         }
 816         putc(*s, script);
 817         }
 818         fprintf(script, "/gp\n");   /* and print */
 819     }
 820     }
 821     fprintf(script, "w\nq\n!\n");   /* write and quit */
 822     fclose(script);
 823 
 824     /* if any line was marked */
 825     if (anymarked == YES) {
 826         
 827     /* edit the files */
 828     clearprompt();
 829     refresh();
 830     fprintf(stderr, "Changed lines:\n\r");
 831     execute("sh", "sh", temp2, NULL);
 832     askforreturn();
 833     seekline(1);
 834     } else {
 835     nochange:
 836     clearprompt();
 837     }
 838     changing = NO;
 839     mousemenu();
 840     free(change);
 841     return(anymarked);
 842 }

 866 /** scrollbar actions */
 867 static void
 868 scrollbar(MOUSE *p)
 869 {
 870     /* reposition list if it makes sense */
 871     if (totallines == 0) {
 872     return;
 873     }
 874     switch (p->percent) {
 875         
 876     case 101: /* scroll down one page */
 877     if (nextline + mdisprefs > totallines) {
 878         nextline = totallines - mdisprefs + 1;
 879     }
 880     break;
 881         
 882     case 102: /* scroll up one page */
 883     nextline = topline - mdisprefs;
 884     if (nextline < 1) {
 885         nextline = 1;
 886     }
 887     break;
 888 
 889     case 103: /* scroll down one line */
 890     nextline = topline + 1;
 891     break;
 892         
 893     case 104: /* scroll up one line */
 894     if (topline > 1) {
 895         nextline = topline - 1;
 896     }
 897     break;
 898     default:
 899     nextline = p->percent * totallines / 100;
 900     }
 901     seekline(nextline);
 902 }

 905 /** count the references found */
 906 void
 907 countrefs(void)
 908 {
 909     char    *subsystem;             /* OGS subsystem name */
 910     char    *book;                  /* OGS book name */
 911     char    file[PATHLEN + 1];      /* file name */
 912     char    function[PATLEN + 1];   /* function name */
 913     char    linenum[NUMLEN + 1];    /* line number */
 914     int     i;
 915 
 916     /* count the references found and find the length of the file,
 917        function, and line number display fields */
 918     subsystemlen = 9;   /* strlen("Subsystem") */
 919     booklen = 4;        /* strlen("Book") */
 920     filelen = 4;        /* strlen("File") */
 921     fcnlen = 8;     /* strlen("Function") */
 922     numlen = 0;
 923     while ((i = fscanf(refsfound, "%250s%250s%5s %5000[^\n]", file,
 924                function, linenum, tempstring)) != EOF) {
 925     if (i != 4 ||
 926         !isgraph((unsigned char) *file) ||
 927         !isgraph((unsigned char) *function) ||
 928         !isdigit((unsigned char) *linenum)) {
 929         postmsg("File does not have expected format");
 930         totallines = 0;
 931         disprefs = 0;
 932         return;
 933     }
 934     if ((i = strlen(pathcomponents(file, dispcomponents))) > filelen) {
 935         filelen = i;
 936     }
 937     if (ogs == YES) {
 938         ogsnames(file, &subsystem, &book);
 939         if ((i = strlen(subsystem)) > subsystemlen) {
 940         subsystemlen = i;
 941         }
 942         if ((i = strlen(book)) > booklen) {
 943         booklen = i;
 944         }
 945     }
 946     if ((i = strlen(function)) > fcnlen) {
 947         fcnlen = i;
 948     }
 949     if ((i = strlen(linenum)) > numlen) {
 950         numlen = i;
 951     }
 952     ++totallines;
 953     }
 954     rewind(refsfound);
 955 
 956     /* restrict the width of displayed columns */
 957     /* HBB FIXME 20060419: magic number alert! */ 
 958     i = (COLS - 5) / 3;
 959     if (ogs == YES) {
 960     i = (COLS - 7) / 5;
 961     }
 962     if (filelen > i && i > 4) {
 963     filelen = i;
 964     }
 965     if (subsystemlen > i && i > 9) {
 966     subsystemlen = i;
 967     }
 968     if (booklen > i && i > 4) {
 969     booklen = i;
 970     }
 971     if (fcnlen > i && i > 8) {
 972     fcnlen = i;
 973     }
 974 }
