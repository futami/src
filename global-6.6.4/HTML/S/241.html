  71     int savelevel;

  73     char classname[MAXTOKEN];

  75     char *completename_limit = &completename[sizeof(completename)];

  85     *classname = *completename = 0;

  87     stack[0].terminate = completename;

  91     startclass = startthrow = startmacro = startsharp = startequal = 0;
  92 
  93     if (!opentoken(param->file))
  94         die("'%s' cannot open.", param->file);

  96     crflag = 1;         /* require '\n' as a token */
  97     cppmode = 1;            /* treat '::' as a token */

  99     while ((cc = nexttoken(interested, cpp_reserved_word)) != EOF) {
 100         if (cc == '~' && level == stack[classlevel].level)
 101             continue;
 102         switch (cc) {
 103         case SYMBOL:        /* symbol   */
 104             if (startclass || startthrow) {
 105                 PUT(PARSER_REF_SYM, token, lineno, sp);
 106             } else if (peekc(0) == '('/* ) */) {
 107                 if (param->isnotfunction(token)) {
 108                     PUT(PARSER_REF_SYM, token, lineno, sp);
 109                 } else if (level > stack[classlevel].level || startequal || startmacro) {
 110                     PUT(PARSER_REF_SYM, token, lineno, sp);
 111                 } else if (level == stack[classlevel].level && !startmacro && !startsharp && !startequal) {
 112                     char savetok[MAXTOKEN], *saveline;
 113                     int savelineno = lineno;
 114 
 115                     strlimcpy(savetok, token, sizeof(savetok));
 116                     strbuf_reset(sb);
 117                     strbuf_puts(sb, sp);
 118                     saveline = strbuf_value(sb);
 119                     if (function_definition(param)) {
 120                         /* ignore constructor */
 121                         if (strcmp(stack[classlevel].classname, savetok))
 122                             PUT(PARSER_DEF, savetok, savelineno, saveline);
 123                     } else {
 124                         PUT(PARSER_REF_SYM, savetok, savelineno, saveline);
 125                     }
 126                 }
 127             } else {
 128                 PUT(PARSER_REF_SYM, token, lineno, sp);
 129             }
 130             break;
 131         case CPP_USING:
 132             crflag = 0;
 133             /*
 134              * using namespace name;
 135              * using ...;
 136              */
 137             if ((c = nexttoken(interested, cpp_reserved_word)) == CPP_NAMESPACE) {
 138                 if ((c = nexttoken(interested, cpp_reserved_word)) == SYMBOL) {
 139                     PUT(PARSER_REF_SYM, token, lineno, sp);
 140                 } else {
 141                     if (param->flags & PARSER_WARNING)
 142                         warning("missing namespace name. [+%d %s].", lineno, curfile);
 143                     pushbacktoken();
 144                 }
 145             } else if (c  == SYMBOL) {
 146                 char savetok[MAXTOKEN], *saveline;
 147                 int savelineno = lineno;
 148 
 149                 strlimcpy(savetok, token, sizeof(savetok));
 150                 strbuf_reset(sb);
 151                 strbuf_puts(sb, sp);
 152                 saveline = strbuf_value(sb);
 153                 if ((c = nexttoken(interested, cpp_reserved_word)) == '=') {
 154                     PUT(PARSER_DEF, savetok, savelineno, saveline);
 155                 } else {
 156                     PUT(PARSER_REF_SYM, savetok, savelineno, saveline);
 157                     while (c == SYMBOL) {
 158                         PUT(PARSER_REF_SYM, token, lineno, sp);
 159                         c = nexttoken(interested, cpp_reserved_word);
 160                     }
 161                 }
 162             } else {
 163                 pushbacktoken();
 164             }
 165             crflag = 1;
 166             break;
 167         case CPP_NAMESPACE:
 168             crflag = 0;
 169             /*
 170              * namespace name = ...;
 171              * namespace [name] { ... }
 172              */
 173             if ((c = nexttoken(interested, cpp_reserved_word)) == SYMBOL) {
 174                 PUT(PARSER_DEF, token, lineno, sp);
 175                 if ((c = nexttoken(interested, cpp_reserved_word)) == '=') {
 176                     crflag = 1;
 177                     break;
 178                 }
 179             }
 180             /*
 181              * Namespace block doesn't have any influence on level.
 182              */
 183             if (c == '{') /* } */ {
 184                 namespacelevel++;
 185             } else {
 186                 if (param->flags & PARSER_WARNING)
 187                     warning("missing namespace block. [+%d %s](0x%x).", lineno, curfile, c);
 188             }
 189             crflag = 1;
 190             break;
 191         case CPP_EXTERN: /* for 'extern "C"/"C++"' */
 192             if (peekc(0) != '"') /* " */
 193                 continue; /* If does not start with '"', continue. */
 194             while ((c = nexttoken(interested, cpp_reserved_word)) == '\n')
 195                 ;
 196             /*
 197              * 'extern "C"/"C++"' block is a kind of namespace block.
 198              * (It doesn't have any influence on level.)
 199              */
 200             if (c == '{') /* } */
 201                 namespacelevel++;
 202             else
 203                 pushbacktoken();
 204             break;
 205         case CPP_STRUCT:
 206         case CPP_CLASS:
 207             DBG_PRINT(level, cc == CPP_CLASS ? "class" : "struct");
 208             while ((c = nexttoken(NULL, cpp_reserved_word)) == CPP___ATTRIBUTE__ || c == '\n')
 209                 if (c == CPP___ATTRIBUTE__)
 210                     process_attribute(param);
 211             if (c == SYMBOL) {
 212                 char *saveline;
 213                 int savelineno;
 214                 do {
 215                     if (c == SYMBOL) {
 216                         savelineno = lineno;
 217                         strbuf_reset(sb);
 218                         strbuf_puts(sb, sp);
 219                         saveline = strbuf_value(sb);
 220                         strlimcpy(classname, token, sizeof(classname));
 221                     }
 222                     c = nexttoken(NULL, cpp_reserved_word);
 223                     if (c == SYMBOL)
 224                         PUT(PARSER_REF_SYM, classname, savelineno, saveline);
 225                     else if (c == '<') {
 226                         int templates = 1;
 227                         for (;;) {
 228                             c = nexttoken(NULL, cpp_reserved_word);
 229                             if (c == SYMBOL)
 230                                 PUT(PARSER_REF_SYM, token, lineno, sp);
 231                             if (c == '<') {
 232                                 if (peekc(1) == '<')
 233                                     throwaway_nextchar();
 234                                 else
 235                                     ++templates;
 236                             } else if (c == '>') {
 237                                 if (--templates == 0)
 238                                     break;
 239                             } else if (c == EOF) {
 240                                 if (param->flags & PARSER_WARNING) 
 241                                     warning("failed to parse template [+%d %s].", savelineno, curfile);
 242                                 goto finish;
 243                             }
 244                         }
 245                         c = nexttoken(NULL, cpp_reserved_word);
 246                     }
 247                 } while (c == SYMBOL || c == '\n');
 248                 if (c == ':' || c == '{') /* } */ {
 249                     startclass = 1;
 250                     PUT(PARSER_DEF, classname, savelineno, saveline);
 251                 } else
 252                     PUT(PARSER_REF_SYM, classname, savelineno, saveline);
 253             }
 254             pushbacktoken();
 255             break;
 256         case '{':  /* } */
 257             DBG_PRINT(level, "{"); /* } */
 258             ++level;
 259             if ((param->flags & PARSER_BEGIN_BLOCK) && atfirst) {
 260                 if ((param->flags & PARSER_WARNING) && level != 1)
 261                     warning("forced level 1 block start by '{' at column 0 [+%d %s].", lineno, curfile); /* } */
 262                 level = 1;
 263             }
 264             if (startclass) {
 265                 char *p = stack[classlevel].terminate;
 266                 char *q = classname;
 267 
 268                 if (++classlevel >= MAXCLASSSTACK)
 269                     die("class stack over flow.[%s]", curfile);
 270                 if (classlevel > 1 && p < completename_limit)
 271                     *p++ = '.';
 272                 stack[classlevel].classname = p;
 273                 while (*q && p < completename_limit)
 274                     *p++ = *q++;
 275                 stack[classlevel].terminate = p;
 276                 stack[classlevel].level = level;
 277                 *p++ = 0;
 278             }
 279             startclass = startthrow = 0;
 280             break;
 281             /* { */
 282         case '}':
 283             if (--level < 0) {
 284                 if (namespacelevel > 0)
 285                     namespacelevel--;
 286                 else if (param->flags & PARSER_WARNING)
 287                     warning("missing left '{' [+%d %s].", lineno, curfile); /* } */
 288                 level = 0;
 289             }
 290             if ((param->flags & PARSER_END_BLOCK) && atfirst) {
 291                 if ((param->flags & PARSER_WARNING) && level != 0)
 292                     /* { */
 293                     warning("forced level 0 block end by '}' at column 0 [+%d %s].", lineno, curfile);
 294                 level = 0;
 295             }
 296             if (level < stack[classlevel].level)
 297                 *(stack[--classlevel].terminate) = 0;
 298             /* { */
 299             DBG_PRINT(level, "}");
 300             break;
 301         case '=':
 302             /* dirty hack. Don't mimic this. */
 303             if (peekc(0) == '=') {
 304                 throwaway_nextchar();
 305             } else {
 306                 startequal = 1;
 307             }
 308             break;
 309         case ';':
 310             startthrow = startequal = 0;
 311             break;
 312         case '\n':
 313             if (startmacro && level != savelevel) {
 314                 if (param->flags & PARSER_WARNING)
 315                     warning("different level before and after #define macro. reseted. [+%d %s].", lineno, curfile);
 316                 level = savelevel;
 317             }
 318             startmacro = startsharp = 0;
 319             break;
 320         /*
 321          * #xxx
 322          */
 323         case SHARP_DEFINE:
 324         case SHARP_UNDEF:
 325             startmacro = 1;
 326             savelevel = level;
 327             if ((c = nexttoken(interested, cpp_reserved_word)) != SYMBOL) {
 328                 pushbacktoken();
 329                 break;
 330             }
 331             if (peekc(1) == '('/* ) */) {
 332                 PUT(PARSER_DEF, token, lineno, sp);
 333                 while ((c = nexttoken("()", cpp_reserved_word)) != EOF && c != '\n' && c != /* ( */ ')')
 334                     if (c == SYMBOL)
 335                         PUT(PARSER_REF_SYM, token, lineno, sp);
 336                 if (c == '\n')
 337                     pushbacktoken();
 338             }  else {
 339                 PUT(PARSER_DEF, token, lineno, sp);
 340             }
 341             break;
 342         case SHARP_IMPORT:
 343         case SHARP_INCLUDE:
 344         case SHARP_INCLUDE_NEXT:
 345         case SHARP_ERROR:
 346         case SHARP_LINE:
 347         case SHARP_PRAGMA:
 348         case SHARP_WARNING:
 349         case SHARP_IDENT:
 350         case SHARP_SCCS:
 351             while ((c = nexttoken(interested, cpp_reserved_word)) != EOF && c != '\n')
 352                 ;
 353             break;
 354         case SHARP_IFDEF:
 355         case SHARP_IFNDEF:
 356         case SHARP_IF:
 357         case SHARP_ELIF:
 358         case SHARP_ELSE:
 359         case SHARP_ENDIF:
 360             condition_macro(param, cc);
 361             break;
 362         case SHARP_SHARP:       /* ## */
 363             (void)nexttoken(interested, cpp_reserved_word);
 364             break;
 365         case CPP_NEW:
 366             if ((c = nexttoken(interested, cpp_reserved_word)) == SYMBOL)
 367                 PUT(PARSER_REF_SYM, token, lineno, sp);
 368             break;
 369         case CPP_ENUM:
 370         case CPP_UNION:
 371             while ((c = nexttoken(interested, cpp_reserved_word)) == CPP___ATTRIBUTE__)
 372                 process_attribute(param);
 373             while (c == '\n')
 374                 c = nexttoken(interested, cpp_reserved_word);
 375             if (c == SYMBOL) {
 376                 if (peekc(0) == '{') /* } */ {
 377                     PUT(PARSER_DEF, token, lineno, sp);
 378                 } else {
 379                     PUT(PARSER_REF_SYM, token, lineno, sp);
 380                 }
 381                 c = nexttoken(interested, cpp_reserved_word);
 382             }
 383             while (c == '\n')
 384                 c = nexttoken(interested, cpp_reserved_word);
 385             if (c == '{' /* } */ && cc == CPP_ENUM) {
 386                 enumerator_list(param);
 387             } else {
 388                 pushbacktoken();
 389             }
 390             break;
 391         case CPP_TEMPLATE:
 392             {
 393                 int level = 0;
 394 
 395                 while ((c = nexttoken("<>", cpp_reserved_word)) != EOF) {
 396                     if (c == '<')
 397                         ++level;
 398                     else if (c == '>') {
 399                         if (--level == 0)
 400                             break;
 401                     } else if (c == SYMBOL) {
 402                         PUT(PARSER_REF_SYM, token, lineno, sp);
 403                     }
 404                 }
 405                 if (c == EOF && (param->flags & PARSER_WARNING))
 406                     warning("template <...> isn't closed. [+%d %s].", lineno, curfile);
 407             }
 408             break;
 409         case CPP_OPERATOR:
 410             while ((c = nexttoken(";{", /* } */ cpp_reserved_word)) != EOF) {
 411                 if (c == '{') /* } */ {
 412                     pushbacktoken();
 413                     break;
 414                 } else if (c == ';') {
 415                     break;
 416                 } else if (c == SYMBOL) {
 417                     PUT(PARSER_REF_SYM, token, lineno, sp);
 418                 }
 419             }
 420             if (c == EOF && (param->flags & PARSER_WARNING))
 421                 warning("'{' doesn't exist after 'operator'. [+%d %s].", lineno, curfile); /* } */
 422             break;
 423         /* control statement check */
 424         case CPP_THROW:
 425             startthrow = 1;
 426         case CPP_BREAK:
 427         case CPP_CASE:
 428         case CPP_CATCH:
 429         case CPP_CONTINUE:
 430         case CPP_DEFAULT:
 431         case CPP_DELETE:
 432         case CPP_DO:
 433         case CPP_ELSE:
 434         case CPP_FOR:
 435         case CPP_GOTO:
 436         case CPP_IF:
 437         case CPP_RETURN:
 438         case CPP_SWITCH:
 439         case CPP_TRY:
 440         case CPP_WHILE:
 441             if ((param->flags & PARSER_WARNING) && !startmacro && level == 0)
 442                 warning("Out of function. %8s [+%d %s]", token, lineno, curfile);
 443             break;
 444         case CPP_TYPEDEF:
 445             {
 446                 /*
 447                  * This parser is too complex to maintain.
 448                  * We should rewrite the whole.
 449                  */
 450                 char savetok[MAXTOKEN];
 451                 int savelineno = 0;
 452                 int typedef_savelevel = level;
 453                 int templates = 0;
 454 
 455                 savetok[0] = 0;
 456 
 457                 /* skip CV qualifiers */
 458                 do {
 459                     c = nexttoken("{}(),;", cpp_reserved_word);
 460                 } while (IS_CV_QUALIFIER(c) || c == '\n');
 461 
 462                 if ((param->flags & PARSER_WARNING) && c == EOF) {
 463                     warning("unexpected eof. [+%d %s]", lineno, curfile);
 464                     break;
 465                 } else if (c == CPP_ENUM || c == CPP_STRUCT || c == CPP_UNION) {
 466                     char *interest_enum = "{},;";
 467                     int c_ = c;
 468 
 469                     while ((c = nexttoken(interest_enum, cpp_reserved_word)) == CPP___ATTRIBUTE__)
 470                         process_attribute(param);
 471                     while (c == '\n')
 472                         c = nexttoken(interest_enum, cpp_reserved_word);
 473                     /* read tag name if exist */
 474                     if (c == SYMBOL) {
 475                         if (peekc(0) == '{') /* } */ {
 476                             PUT(PARSER_DEF, token, lineno, sp);
 477                         } else {
 478                             PUT(PARSER_REF_SYM, token, lineno, sp);
 479                         }
 480                         c = nexttoken(interest_enum, cpp_reserved_word);
 481                     }
 482                     while (c == '\n')
 483                         c = nexttoken(interest_enum, cpp_reserved_word);
 484                     if (c_ == CPP_ENUM) {
 485                         if (c == '{') /* } */
 486                             c = enumerator_list(param);
 487                         else
 488                             pushbacktoken();
 489                     } else {
 490                         for (; c != EOF; c = nexttoken(interest_enum, cpp_reserved_word)) {
 491                             switch (c) {
 492                             case SHARP_IFDEF:
 493                             case SHARP_IFNDEF:
 494                             case SHARP_IF:
 495                             case SHARP_ELIF:
 496                             case SHARP_ELSE:
 497                             case SHARP_ENDIF:
 498                                 condition_macro(param, c);
 499                                 continue;
 500                             default:
 501                                 break;
 502                             }
 503                             if (c == ';' && level == typedef_savelevel) {
 504                                 if (savetok[0]) {
 505                                     PUT(PARSER_DEF, savetok, savelineno, sp);
 506                                     savetok[0] = 0;
 507                                 }
 508                                 break;
 509                             } else if (c == '{')
 510                                 level++;
 511                             else if (c == '}') {
 512                                 savetok[0] = 0;
 513                                 if (--level == typedef_savelevel)
 514                                     break;
 515                             } else if (c == SYMBOL) {
 516                                 if (level > typedef_savelevel)
 517                                     PUT(PARSER_REF_SYM, token, lineno, sp);
 518                                 /* save lastest token */
 519                                 strlimcpy(savetok, token, sizeof(savetok));
 520                                 savelineno = lineno;
 521                             }
 522                         }
 523                         if (c == ';')
 524                             break;
 525                     }
 526                     if ((param->flags & PARSER_WARNING) && c == EOF) {
 527                         warning("unexpected eof. [+%d %s]", lineno, curfile);
 528                         break;
 529                     }
 530                 } else if (c == SYMBOL) {
 531                     PUT(PARSER_REF_SYM, token, lineno, sp);
 532                 }
 533                 savetok[0] = 0;
 534                 while ((c = nexttoken("()<>,;", cpp_reserved_word)) != EOF) {
 535                     switch (c) {
 536                     case SHARP_IFDEF:
 537                     case SHARP_IFNDEF:
 538                     case SHARP_IF:
 539                     case SHARP_ELIF:
 540                     case SHARP_ELSE:
 541                     case SHARP_ENDIF:
 542                         condition_macro(param, c);
 543                         continue;
 544                     default:
 545                         break;
 546                     }
 547                     if (c == '(')
 548                         level++;
 549                     else if (c == ')')
 550                         level--;
 551                     else if (c == '<')
 552                         templates++;
 553                     else if (c == '>')
 554                         templates--;
 555                     else if (c == SYMBOL) {
 556                         if (level > typedef_savelevel) {
 557                             PUT(PARSER_REF_SYM, token, lineno, sp);
 558                         } else {
 559                             /* put latest token if any */
 560                             if (savetok[0]) {
 561                                 PUT(PARSER_REF_SYM, savetok, savelineno, sp);
 562                             }
 563                             /* save lastest token */
 564                             strlimcpy(savetok, token, sizeof(savetok));
 565                             savelineno = lineno;
 566                         }
 567                     } else if (c == ',' || c == ';') {
 568                         if (savetok[0]) {
 569                             PUT(templates ? PARSER_REF_SYM : PARSER_DEF, savetok, lineno, sp);
 570                             savetok[0] = 0;
 571                         }
 572                     }
 573                     if (level == typedef_savelevel && c == ';')
 574                         break;
 575                 }
 576                 if (param->flags & PARSER_WARNING) {
 577                     if (c == EOF)
 578                         warning("unexpected eof. [+%d %s]", lineno, curfile);
 579                     else if (level != typedef_savelevel)
 580                         warning("unmatched () block. (last at level %d.)[+%d %s]", level, lineno, curfile);
 581                 }
 582             }
 583             break;
 584         case CPP___ATTRIBUTE__:
 585             process_attribute(param);
 586             break;
 587         default:
 588             break;
 589         }
 590     }
 591 finish:
 592     strbuf_close(sb);
 593     if (param->flags & PARSER_WARNING) {
 594         if (level != 0)
 595             warning("unmatched {} block. (last at level %d.)[+%d %s]", level, lineno, curfile);
 596         if (piflevel != 0)
 597             warning("unmatched #if block. (last at level %d.)[+%d %s]", piflevel, lineno, curfile);
 598     }
 599     closetoken();
 600 }
 601 /**
 602  * process_attribute: skip attributes in '__attribute__((...))'.
 603  */
 604 static void
 605 process_attribute(const struct parser_param *param)
 606 {
 607     int brace = 0;
 608     int c;
 609     /*
 610      * Skip '...' in __attribute__((...))
 611      * but pick up symbols in it.
 612      */
 613     while ((c = nexttoken("()", cpp_reserved_word)) != EOF) {
 614         if (c == '(')
 615             brace++;
 616         else if (c == ')')
 617             brace--;
 618         else if (c == SYMBOL) {
 619             PUT(PARSER_REF_SYM, token, lineno, sp);
 620         }
 621         if (brace == 0)
 622             break;
 623     }
 624 }
 625 /**
 626  * function_definition: return if function definition or not.
 627  *
 628  *  @return target type
 629  */
 630 static int
 631 function_definition(const struct parser_param *param)
 632 {
 633     int c;
 634     int brace_level;
 635 
 636     brace_level = 0;
 637     while ((c = nexttoken("()", cpp_reserved_word)) != EOF) {
 638         switch (c) {
 639         case SHARP_IFDEF:
 640         case SHARP_IFNDEF:
 641         case SHARP_IF:
 642         case SHARP_ELIF:
 643         case SHARP_ELSE:
 644         case SHARP_ENDIF:
 645             condition_macro(param, c);
 646             continue;
 647         default:
 648             break;
 649         }
 650         if (c == '('/* ) */)
 651             brace_level++;
 652         else if (c == /* ( */')') {
 653             if (--brace_level == 0)
 654                 break;
 655         }
 656         /* pick up symbol */
 657         if (c == SYMBOL)
 658             PUT(PARSER_REF_SYM, token, lineno, sp);
 659     }
 660     if (c == EOF)
 661         return 0;
 662     if (peekc(0) == ';') {
 663         (void)nexttoken(";", NULL);
 664         return 0;
 665     }
 666     brace_level = 0;
 667     while ((c = nexttoken(",;[](){}=", cpp_reserved_word)) != EOF) {
 668         switch (c) {
 669         case SHARP_IFDEF:
 670         case SHARP_IFNDEF:
 671         case SHARP_IF:
 672         case SHARP_ELIF:
 673         case SHARP_ELSE:
 674         case SHARP_ENDIF:
 675             condition_macro(param, c);
 676             continue;
 677         case CPP___ATTRIBUTE__:
 678             process_attribute(param);
 679             continue;
 680         case SHARP_DEFINE:
 681             pushbacktoken();
 682             return 0;
 683         default:
 684             break;
 685         }
 686         if (c == '('/* ) */ || c == '[')
 687             brace_level++;
 688         else if (c == /* ( */')' || c == ']')
 689             brace_level--;
 690         else if (brace_level == 0 && (c == ';' || c == ','))
 691             break;
 692         else if (c == '{' /* } */) {
 693             pushbacktoken();
 694             return 1;
 695         } else if (c == /* { */'}') {
 696             pushbacktoken();
 697             break;
 698         } else if (c == '=')
 699             break;
 700         /* pick up symbol */
 701         if (c == SYMBOL)
 702             PUT(PARSER_REF_SYM, token, lineno, sp);
 703     }
 704     return 0;
 705 }

 707 /**
 708  * condition_macro: 
 709  *
 710  *  @param[in]  param
 711  *  @param[in]  cc  token
 712  */
 713 static void
 714 condition_macro(const struct parser_param *param, int cc)
 715 {
 716     cur = &pifstack[piflevel];
 717     if (cc == SHARP_IFDEF || cc == SHARP_IFNDEF || cc == SHARP_IF) {
 718         DBG_PRINT(piflevel, "#if");
 719         if (++piflevel >= MAXPIFSTACK)
 720             die("#if pifstack over flow. [%s]", curfile);
 721         ++cur;
 722         cur->start = level;
 723         cur->end = -1;
 724         cur->if0only = 0;
 725         if (peekc(0) == '0')
 726             cur->if0only = 1;
 727         else if ((cc = nexttoken(NULL, cpp_reserved_word)) == SYMBOL && !strcmp(token, "notdef"))
 728             cur->if0only = 1;
 729         else
 730             pushbacktoken();
 731     } else if (cc == SHARP_ELIF || cc == SHARP_ELSE) {
 732         DBG_PRINT(piflevel - 1, "#else");
 733         if (cur->end == -1)
 734             cur->end = level;
 735         else if (cur->end != level && (param->flags & PARSER_WARNING))
 736             warning("uneven level. [+%d %s]", lineno, curfile);
 737         level = cur->start;
 738         cur->if0only = 0;
 739     } else if (cc == SHARP_ENDIF) {
 740         int minus = 0;
 741 
 742         --piflevel;
 743         if (piflevel < 0) {
 744             minus = 1;
 745             piflevel = 0;
 746         }
 747         DBG_PRINT(piflevel, "#endif");
 748         if (minus) {
 749             if (param->flags & PARSER_WARNING)
 750                 warning("unmatched #if block. reseted. [+%d %s]", lineno, curfile);
 751         } else {
 752             if (cur->if0only)
 753                 level = cur->start;
 754             else if (cur->end != -1) {
 755                 if (cur->end != level && (param->flags & PARSER_WARNING))
 756                     warning("uneven level. [+%d %s]", lineno, curfile);
 757                 level = cur->end;
 758             }
 759         }
 760     }
 761     while ((cc = nexttoken(NULL, cpp_reserved_word)) != EOF && cc != '\n') {
 762                 if (cc == SYMBOL && strcmp(token, "defined") != 0) {
 763             PUT(PARSER_REF_SYM, token, lineno, sp);
 764         }
 765     }
 766 }

 768 /**
 769  * enumerator_list: process "symbol (= expression), ... "}
 770  */
 771 static int
 772 enumerator_list(const struct parser_param *param)
 773 {
 774     int savelevel = level;
 775     int in_expression = 0;
 776     int c = '{';
 777 
 778     for (; c != EOF; c = nexttoken("{}(),=", cpp_reserved_word)) {
 779         switch (c) {
 780         case SHARP_IFDEF:
 781         case SHARP_IFNDEF:
 782         case SHARP_IF:
 783         case SHARP_ELIF:
 784         case SHARP_ELSE:
 785         case SHARP_ENDIF:
 786             condition_macro(param, c);
 787             break;
 788         case SYMBOL:
 789             if (in_expression)
 790                 PUT(PARSER_REF_SYM, token, lineno, sp);
 791             else
 792                 PUT(PARSER_DEF, token, lineno, sp);
 793             break;
 794         case '{':
 795         case '(':
 796             level++;
 797             break;
 798         case '}':
 799         case ')':
 800             if (--level == savelevel)
 801                 return c;
 802             break;
 803         case ',':
 804             if (level == savelevel + 1)
 805                 in_expression = 0;
 806             break;
 807         case '=':
 808             in_expression = 1;
 809             break;
 810         default:
 811             break;
 812         }
 813     }
 814 
 815     return c;
 816 }
