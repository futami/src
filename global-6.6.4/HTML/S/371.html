  76 copy_langmap_converting_cpp(char *dst, const char *src)
  77 {
  78     const char *p;
  79 
  80     if (strncmp(src, "cpp:", 4) == 0) {
  81         memcpy(dst, "c++:", 4);
  82         dst += 4;
  83         src += 4;
  84     }
  85     while ((p = strstr(src, ",cpp:")) != NULL) {
  86         memcpy(dst, src, p - src);
  87         dst += p - src;
  88         memcpy(dst, ",c++:", 5);
  89         dst += 5;
  90         src = p + 5;
  91     }
  92     strcpy(dst, src);
  93 }

 113 start_ctags(const struct parser_param *param)
 114 {
 115     HANDLE opipe[2], ipipe[2];
 116     SECURITY_ATTRIBUTES sa;
 117     STARTUPINFO si;
 118     PROCESS_INFORMATION pi;
 119     char* arg;
 120     char *path = param->getconf("ctagscom");
 121 
 122     if (path && strlen(path) > 0 && strcmp(path, "no") != 0)
 123         ctagscom = path;
 124     if (!ctagscom || !strlen(ctagscom) || !strcmp(ctagscom, "no"))
 125         param->die(ctagsnotfound);
 126 
 127     arg = malloc(1 + strlen(ctagscom) + sizeof(argv) + strlen(param->langmap));
 128     if (arg == NULL)
 129         param->die("short of memory.");
 130     *arg = '"';
 131     strcpy(arg+1, ctagscom);
 132     strcat(arg, argv);
 133     copy_langmap_converting_cpp(arg + strlen(arg), param->langmap);
 134 
 135     sa.nLength = sizeof(sa);
 136     sa.bInheritHandle = TRUE;
 137     sa.lpSecurityDescriptor = NULL;
 138     if (!CreatePipe(&opipe[0], &opipe[1], &sa, 0) ||
 139         !CreatePipe(&ipipe[0], &ipipe[1], &sa, 0))
 140         param->die("CreatePipe failed.");
 141     SetHandleInformation(opipe[1], HANDLE_FLAG_INHERIT, 0);
 142     SetHandleInformation(ipipe[0], HANDLE_FLAG_INHERIT, 0);
 143     ZeroMemory(&si, sizeof(si));
 144     si.cb = sizeof(si);
 145     si.hStdInput = opipe[0];
 146     si.hStdOutput = ipipe[1];
 147     si.hStdError = GetStdHandle(STD_ERROR_HANDLE);
 148     si.dwFlags = STARTF_USESTDHANDLES;
 149     if (!CreateProcess(NULL, arg, NULL, NULL, TRUE, 0, NULL, NULL, &si, &pi))
 150         param->die(ctagsnotfound);
 151     CloseHandle(opipe[0]);
 152     CloseHandle(ipipe[1]);
 153     CloseHandle(pi.hThread);
 154     pid = pi.hProcess;
 155     op = fdopen(_open_osfhandle((long)opipe[1], _O_WRONLY), "w");
 156     ip = fdopen(_open_osfhandle((long)ipipe[0], _O_RDONLY), "r");
 157     if (ip == NULL || op == NULL)
 158         param->die("fdopen failed.");
 159 
 160     bufsize = INITIAL_BUFSIZE;
 161     linebuf = malloc(bufsize);
 162     if (linebuf == NULL)
 163         param->die("short of memory.");
 164 }

 193 static void
 194 start_ctags(const struct parser_param *param)
 195 {
 196     int opipe[2], ipipe[2];
 197     char *path = param->getconf("ctagscom");
 198 
 199     if (path && strlen(path) > 0 && strcmp(path, "no") != 0)
 200         ctagscom = path;
 201     if (!ctagscom || !strlen(ctagscom) || !strcmp(ctagscom, "no"))
 202         param->die(ctagsnotfound);
 203     argv[0] = ctagscom;
 204     argv[1] = malloc(sizeof(LANGMAP_OPTION) + strlen(param->langmap));
 205     if (argv[1] == NULL)
 206         param->die("short of memory.");
 207     memcpy(argv[1], LANGMAP_OPTION, sizeof(LANGMAP_OPTION) - 1);
 208     copy_langmap_converting_cpp(argv[1] + sizeof(LANGMAP_OPTION) - 1, param->langmap);
 209 
 210     if (pipe(opipe) < 0 || pipe(ipipe) < 0)
 211         param->die("cannot create pipe.");
 212     pid = fork();
 213     if (pid == 0) {
 214         /* child process */
 215         close(opipe[1]);
 216         close(ipipe[0]);
 217         if (dup2(opipe[0], STDIN_FILENO) < 0
 218          || dup2(ipipe[1], STDOUT_FILENO) < 0)
 219             param->die("dup2 failed.");
 220         close(opipe[0]);
 221         close(ipipe[1]);
 222         execvp(ctagscom, argv);
 223         param->die("cannot execute '%s'. (execvp failed)", ctagscom);
 224     }
 225     /* parent process */
 226     if (pid < 0)
 227         param->die("fork failed.");
 228     free(path);
 229     free(argv[1]);
 230     close(opipe[0]);
 231     close(ipipe[1]);
 232     ip = fdopen(ipipe[0], "r");
 233     op = fdopen(opipe[1], "w");
 234     if (ip == NULL || op == NULL)
 235         param->die("fdopen failed.");
 236 
 237     bufsize = INITIAL_BUFSIZE;
 238     linebuf = malloc(bufsize);
 239     if (linebuf == NULL)
 240         param->die("short of memory.");
 241 }

 256 static char *
 257 get_line(const struct parser_param *param)
 258 {
 259     size_t linelen = 0;
 260 
 261     for (;;) {
 262         if (fgets(linebuf + linelen, bufsize - linelen, ip) == NULL) {
 263             if (linelen == 0)
 264                 return NULL;
 265             break;
 266         }
 267         linelen += strlen(linebuf + linelen);
 268         if (linelen < bufsize - 1 || linebuf[linelen - 1] == '\n'
 269          || feof(ip))
 270             break;
 271         bufsize *= 2;
 272         linebuf = realloc(linebuf, bufsize);
 273         if (linebuf == NULL)
 274             param->die("short of memory.");
 275     }
 276     while (linelen-- > 0
 277         && (linebuf[linelen] == '\n' || linebuf[linelen] == '\r'))
 278         linebuf[linelen] = '\0';
 279     return linebuf;
 280 }

 282 static void
 283 put_line(char *ctags_x, const struct parser_param *param)
 284 {
 285     int lineno;
 286     int type = PARSER_DEF;
 287     char *p, *tagname, *filename;
 288 
 289 #if defined(USE_EXTRA_FIELDS)
 290     /*
 291      * Output of ctags:
 292      * ctags -x ...
 293      * main              326 global/global.c  main(int argc, char **argv)
 294      *
 295      * ctags -x --_xformat="%R %-16N %4n %-16F %C" --extras=+r --fields=+r
 296      * D main              326 global/global.c  main(int argc, char **argv)
 297      */
 298     switch (*ctags_x) {
 299     case 'D':
 300         type = PARSER_DEF;
 301         break;
 302     case 'R':
 303         type = PARSER_REF_SYM;
 304         break;
 305     default:
 306         param->die("unexpected type string.");
 307     }
 308     /* skip type string */
 309     while (*ctags_x && !isspace((unsigned char)*ctags_x))
 310         ctags_x++;
 311     while (*ctags_x  && isspace((unsigned char)*ctags_x))
 312         ctags_x++;
 313 #endif
 314     filename = strstr(ctags_x, param->file);
 315     if (filename == NULL || filename == ctags_x)
 316         return;
 317     p = filename - 1;
 318     if (!isspace((unsigned char)*p))
 319         return;
 320     while (p >= ctags_x && isspace((unsigned char)*p))
 321         *p-- = '\0';
 322     if (p < ctags_x)
 323         return;
 324     if (!isdigit((unsigned char)*p))
 325         return;
 326     while (p >= ctags_x && isdigit((unsigned char)*p))
 327         p--;
 328     if (p < ctags_x)
 329         return;
 330     lineno = atoi(p + 1);
 331     if (!isspace((unsigned char)*p))
 332         return;
 333     while (p >= ctags_x && isspace((unsigned char)*p))
 334         *p-- = '\0';
 335     if (p < ctags_x)
 336         return;
 337     while (p >= ctags_x && !isspace((unsigned char)*p))
 338         p--;
 339     tagname = p + 1;
 340     p = filename + strlen(param->file);
 341     if (*p != '\0') {
 342         if (!isspace((unsigned char)*p))
 343             return;
 344         *p++ = '\0';
 345         while (isspace((unsigned char)*p))
 346             p++;
 347     }
 348     param->put(type, tagname, lineno, filename, p, param->arg);
 349 }

 351 void
 352 parser(const struct parser_param *param)
 353 {
 354     char *ctags_x;
 355 
 356     assert(param->size >= sizeof(*param));
 357 
 358     if (op == NULL)
 359         start_ctags(param);
 360 
 361     /* Write path of input file to pipe. */
 362     fputs(param->file, op);
 363     putc('\n', op);
 364     fflush(op);
 365 
 366     /* Read output of ctags command. */
 367     for (;;) {
 368         ctags_x = get_line(param);
 369         if (ctags_x == NULL)
 370             param->die("unexpected EOF.");
 371         if (strcmp(ctags_x, TERMINATOR) == 0)
 372             break;
 373         put_line(ctags_x, param);
 374     }
 375 }
