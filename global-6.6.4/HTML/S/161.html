  51 /*
  52  * Page 0 of a btree file contains a copy of the meta-data.  This page is also
  53  * used as an out-of-band page, i.e. page pointers that point to nowhere point
  54  * to page 0.  Page 1 is the root of the btree.
  55  */
  56         /** Invalid tree page number. */
  57 #define P_INVALID    0
  58         /** Tree metadata page number. */
  59 #define P_META       0
  60         /** Tree root page number. */
  61 #define P_ROOT       1

  63 /**
  64  * There are five page layouts in the btree: btree internal pages (BINTERNAL),
  65  * btree leaf pages (BLEAF), recno internal pages (RINTERNAL), recno leaf pages
  66  * (RLEAF) and overflow pages.  All five page types have a page header (PAGE).
  67  * This implementation requires that values within structures NOT be padded.
  68  * (ANSI C permits random padding.)  If your compiler pads randomly you'll have
  69  * to do some work to get this package to run.
  70  */
  71 typedef struct _page {
  72     pgno_t  pgno;           /**< this page's page number */
  73     pgno_t  prevpg;         /**< left sibling */
  74     pgno_t  nextpg;         /**< right sibling */
  75 
  76         /** btree internal page */
  77 #define P_BINTERNAL 0x01
  78         /** leaf page */
  79 #define P_BLEAF     0x02
  80         /** overflow page */
  81 #define P_OVERFLOW  0x04
  82         /** recno internal page */
  83 #define P_RINTERNAL 0x08
  84         /** leaf page */
  85 #define P_RLEAF     0x10
  86         /** type mask */
  87 #define P_TYPE      0x1f
  88         /** never delete this chain of pages */
  89 #define P_PRESERVE  0x20
  90     u_int32_t flags;
  91 
  92     indx_t  lower;          /**< lower bound of free space on page */
  93     indx_t  upper;          /**< upper bound of free space on page */
  94     indx_t  linp[1];        /**< indx_t-aligned VAR. LENGTH DATA */
  95 } PAGE;

  97 /** First and next index. */
  98 #define BTDATAOFF                           \
  99     (sizeof(pgno_t) + sizeof(pgno_t) + sizeof(pgno_t) +     \
 100         sizeof(u_int32_t) + sizeof(indx_t) + sizeof(indx_t))
 101 #define NEXTINDEX(p)    (((p)->lower - BTDATAOFF) / sizeof(indx_t))

 103 /**
 104  * For pages other than overflow pages, there is an array of offsets into the
 105  * rest of the page immediately following the page header.  Each offset is to
 106  * an item which is unique to the type of page.  The h_lower offset is just
 107  * past the last filled-in index.  The h_upper offset is the first item on the
 108  * page.  Offsets are from the beginning of the page.
 109  *
 110  * If an item is too big to store on a single page, a flag is set and the item
 111  * is a {page, size} pair such that the page is the first page of an overflow
 112  * chain with size bytes of item.  Overflow pages are simply bytes without any
 113  * external structure.
 114  *
 115  * The page number and size fields in the items are pgno_t-aligned so they can
 116  * be manipulated without copying.  (This presumes that 32 bit items can be
 117  * manipulated on this system.)
 118  */
 119 #define LALIGN(n)   (((n) + sizeof(pgno_t) - 1) & ~(sizeof(pgno_t) - 1))
 120 #define NOVFLSIZE   (sizeof(pgno_t) + sizeof(u_int32_t))
 121 
 122 /**
 123  * For the btree internal pages, the item is a key.  BINTERNALs are {key, pgno}
 124  * pairs, such that the key compares less than or equal to all of the records
 125  * on that page.  For a tree without duplicate keys, an internal page with two
 126  * consecutive keys, a and b, will have all records greater than or equal to a
 127  * and less than b stored on the page associated with a.  Duplicate keys are
 128  * somewhat special and can cause duplicate internal and leaf page records and
 129  * some minor modifications of the above rule.
 130  */
 131 typedef struct _binternal {
 132     u_int32_t ksize;        /**< key size */
 133     pgno_t  pgno;           /**< page number stored on */
 134 
 135         /** overflow data */
 136 #define P_BIGDATA   0x01
 137         /** overflow key */
 138 #define P_BIGKEY    0x02
 139     u_char  flags;
 140     char    bytes[1];       /**< data */
 141 } BINTERNAL;
 142 
 143 /** Get the page's BINTERNAL structure at index indx. */
 144 #define GETBINTERNAL(pg, indx)                      \
 145     ((BINTERNAL *)((char *)(pg) + (pg)->linp[indx]))
 146 
 147 /** Get the number of bytes in the entry. */
 148 #define NBINTERNAL(len)                         \
 149     LALIGN(sizeof(u_int32_t) + sizeof(pgno_t) + sizeof(u_char) + (len))
 150 
 151 /** Copy a BINTERNAL entry to the page. */
 152 #define WR_BINTERNAL(p, size, pgno, flags) {                \
 153     *(u_int32_t *)p = size;                     \
 154     p += sizeof(u_int32_t);                     \
 155     *(pgno_t *)p = pgno;                        \
 156     p += sizeof(pgno_t);                        \
 157     *(u_char *)p = flags;                       \
 158     p += sizeof(u_char);                        \
 159 }

 185 /** For the btree leaf pages, the item is a key and data pair. */
 186 typedef struct _bleaf {
 187     u_int32_t   ksize;      /**< size of key */
 188     u_int32_t   dsize;      /**< size of data */
 189     u_char  flags;          /**< P_BIGDATA, P_BIGKEY */
 190     char    bytes[1];       /**< data */
 191 } BLEAF;
 192 
 193 /** Get the page's BLEAF structure at index indx. */
 194 #define GETBLEAF(pg, indx)                      \
 195     ((BLEAF *)((char *)(pg) + (pg)->linp[indx]))
 196 
 197 /** Get the number of bytes in the entry. */
 198 #define NBLEAF(p)   NBLEAFDBT((p)->ksize, (p)->dsize)
 199 
 200 /** Get the number of bytes in the user's key/data pair. */
 201 #define NBLEAFDBT(ksize, dsize)                     \
 202     LALIGN(sizeof(u_int32_t) + sizeof(u_int32_t) + sizeof(u_char) + \
 203         (ksize) + (dsize))
 204 
 205 /** Copy a BLEAF entry to the page. */
 206 #define WR_BLEAF(p, key, data, flags) {                 \
 207     *(u_int32_t *)p = key->size;                    \
 208     p += sizeof(u_int32_t);                     \
 209     *(u_int32_t *)p = data->size;                   \
 210     p += sizeof(u_int32_t);                     \
 211     *(u_char *)p = flags;                       \
 212     p += sizeof(u_char);                        \
 213     memmove(p, key->data, key->size);               \
 214     p += key->size;                         \
 215     memmove(p, data->data, data->size);             \
 216 }

 245 /**
 246  * A record in the tree is either a pointer to a page and an index in the page
 247  * or a page number and an index.  These structures are used as a cursor, stack
 248  * entry and search returns as well as to pass records to other routines.
 249  *
 250  * One comment about searches.  Internal page searches must find the largest
 251  * record less than key in the tree so that descents work.  Leaf page searches
 252  * must find the smallest record greater than key so that the returned index
 253  * is the record's correct position for insertion.
 254  */
 255 typedef struct _epgno {
 256     pgno_t  pgno;           /**< the page number */
 257     indx_t  index;          /**< the index on the page */
 258 } EPGNO;
 259 
 260 typedef struct _epg {
 261     PAGE    *page;          /**< the (pinned) page */
 262     indx_t   index;         /**< the index on the page */
 263 } EPG;

 265 /*
 266  * About cursors.  The cursor (and the page that contained the key/data pair
 267  * that it referenced) can be deleted, which makes things a bit tricky.  If
 268  * there are no duplicates of the cursor key in the tree (i.e. B_NODUPS is set
 269  * or there simply aren't any duplicates of the key) we copy the key that it
 270  * referenced when it's deleted, and reacquire a new cursor key if the cursor
 271  * is used again.  If there are duplicates keys, we move to the next/previous
 272  * key, and set a flag so that we know what happened.  NOTE: if duplicate (to
 273  * the cursor) keys are added to the tree during this process, it is undefined
 274  * if they will be returned or not in a cursor scan.
 275  *
 276  * The flags determine the possible states of the cursor:
 277  *
 278  * CURS_INIT
 279  *      The cursor references *something*.
 280  * CURS_ACQUIRE
 281  *      The cursor was deleted, and a key has been saved so that
 282  *      we can reacquire the right position in the tree.
 283  * CURS_AFTER, CURS_BEFORE
 284  *      The cursor was deleted, and now references a key/data pair
 285  *      that has not yet been returned, either before or after the
 286  *      deleted key/data pair.
 287  * XXX
 288  * This structure is broken out so that we can eventually offer multiple
 289  * cursors as part of the DB interface.
 290  */
 291 typedef struct _cursor {
 292     EPGNO    pg;            /**< B: Saved tree reference. */
 293     DBT  key;           /**< B: Saved key, or key.data == NULL. */
 294     recno_t  rcursor;       /**< R: recno cursor (1-based) */
 295 
 296         /**  B: Cursor needs to be reacquired. */
 297 #define CURS_ACQUIRE    0x01
 298         /**  B: Unreturned cursor after key. */
 299 #define CURS_AFTER  0x02
 300         /**  B: Unreturned cursor before key. */
 301 #define CURS_BEFORE 0x04
 302         /** RB: Cursor initialized. */
 303 #define CURS_INIT   0x08
 304     u_int8_t flags;
 305 } CURSOR;
