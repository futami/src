  35 /*
  36  * Compress module
  37  *
  38  * Function compress() reduces the size of GTAGS by about 10-20% average.
  39  *
  40  * PROTOCOL:
  41  *
  42  *  meta record: " __.COMPRESS ddefine ttypedef"
  43  *
  44  *  'ddefine' means d => define
  45  *  'ttypedef' means t => typedef
  46  *
  47  *  source      abbreviation
  48  *  -----------------------------------------
  49  *  @       @@
  50  *  <tag name>  @n
  51  *  "define"    @d
  52  *  "typedef"   @t
  53  *  <spaces>    @<digit> or @{<number>}
  54  *
  55  * EXAMPLE OF COMPRESS:
  56  *
  57  *  100 macro 23 #define macro(c) a;      b;
  58  *               ~~~~~~~~~~~~~~~~~~~~~~~~~~~
  59  *               | [compress]   ^ [uncompress]
  60  *               v              |
  61  *  100 macro 23 #@d @n(c) a;@6b;
  62  *               ~~~~~~~~~~~~~~~~
  63  *
  64  * DATA STRUCTURE
  65  *  - ab2name table is used to convert from abbreviation character
  66  *    to the string value.
  67  *  - name2ab table is used to convert from string value to the
  68  *    abbreviation character.
  69  *    
  70  *  ab2name = ('a' => NULL, ... , 'd' => "define", ... 'z' => NULL)
  71  *  name2ab = ("define" => 'a', "typdef" => 't')
  72  */
  73 struct abbrmap {
  74     int c;
  75     char *name;
  76     int length;
  77 };
  78 static struct abbrmap ab2name[26];
  79 static VARRAY *name2ab;
  80 static char abbrev_string[1024];
  81 /**
  82  * setup two internal tables for abbreviation.
  83  *
  84  *  @param[in]  abbrev  abbreviation string
  85  */
  86 void
  87 abbrev_open(const char *abbrev)
  88 {
  89     int i, limit;
  90     struct abbrmap *ab;
  91     char *p;
  92 
  93     /*
  94      * abbrev string: "ddefine ttypedef"
  95      */
  96     /* copy abbrev string to static area */
  97     strlimcpy(abbrev_string, abbrev, sizeof(abbrev_string));
  98     p = abbrev_string;
  99 
 100     /* initialize ab2name table */
 101     limit = sizeof(ab2name) / sizeof(struct abbrmap);
 102     for (i = 0; i < limit; i++) {
 103         ab2name[i].c = 0;
 104         ab2name[i].name = NULL;
 105     }
 106     name2ab = varray_open(sizeof(struct abbrmap), 5);
 107     while (*p) {
 108         ab = (struct abbrmap *)varray_append(name2ab);
 109         ab->c = *p++;
 110         ab->name = p;
 111         for (; *p && *p != ' '; p++)
 112             ;
 113         if (*p == ' ')
 114             *p++ = '\0';
 115         ab->length = strlen(ab->name);
 116         if (ab->c < 'a' || ab->c > 'z')
 117             die("Abbrev character must be a lower alphabetic character. (%s)", abbrev);
 118         i = ab->c - 'a';
 119         ab2name[i].c = ab->c;
 120         ab2name[i].name = ab->name;
 121         ab2name[i].length = ab->length;
 122     }
 123 }
 124 /**
 125  * free allocated memory.
 126  */
 127 void
 128 abbrev_close(void)
 129 {
 130     if (name2ab)
 131         varray_close(name2ab);
 132     name2ab = NULL;
 133 }

 137 void
 138 abbrev_dump(void)
 139 {
 140     struct abbrmap *ab;
 141     int i, limit = sizeof(ab2name) / sizeof(struct abbrmap);
 142 
 143     if (!name2ab) {
 144         fprintf(stderr, "name2ab is NULL.\n");
 145         return;
 146     }
 147     fprintf(stderr, "ab2name: %d entries\n", limit);
 148     for (i = 0; i < limit; i++) {
 149         if (ab2name[i].c != 0) {
 150             fprintf(stderr, "ab2name[%d].c    = %c\n", i, ab2name[i].c);
 151             fprintf(stderr, "ab2name[%d].name = %s\n", i, ab2name[i].name);
 152         }
 153     }
 154     ab = (struct abbrmap *)varray_assign(name2ab, 0, 0);
 155     limit = name2ab->length;
 156     fprintf(stderr, "name2ab: %d entries\n", limit);
 157     for (i = 0; i < limit; i++) {
 158         if (ab[i].c != 0) {
 159             fprintf(stderr, "name2ab[%d].c    = %c\n", i, ab[i].c);
 160             fprintf(stderr, "name2ab[%d].name = %s\n", i, ab[i].name);
 161         }
 162     }
 163 }
 164 /**
 165  * compress source line.
 166  *
 167  *  @param[in]  in  source line
 168  *  @param[in]  name    replaced string
 169  *  @return     compressed string
 170  */
 171 char *
 172 compress(const char *in, const char *name, STRBUF *sb)
 173 {
 174     const char *p = in;
 175     int length = strlen(name);
 176     int spaces = 0;
 177 
 178     strbuf_reset(sb);
 179     while (*p) {
 180         if (*p == ' ') {
 181             spaces++;
 182             p++;
 183             continue;
 184         }
 185         if (spaces > 0) {
 186             if (spaces >= 10) {
 187                 strbuf_putc(sb, '@');
 188                 strbuf_putc(sb, '{');
 189                 strbuf_putn(sb, spaces);
 190                 strbuf_putc(sb, '}');
 191             } else if (spaces > 3) {
 192                 strbuf_putc(sb, '@');
 193                 strbuf_putn(sb, spaces);
 194             } else {
 195                 strbuf_nputc(sb, ' ', spaces);
 196             }
 197         }
 198         spaces = 0;
 199         if (*p == '@') {
 200             strbuf_puts(sb, "@@");
 201             p++;
 202         } else if (!strncmp(p, name, length)) {
 203             strbuf_puts(sb, "@n");
 204             p += length;
 205         } else if (name2ab) {
 206             int i, limit = name2ab->length;
 207             struct abbrmap *ab = (struct abbrmap *)varray_assign(name2ab, 0, 0);
 208 
 209             for (i = 0; i < limit; i++) {
 210                 if (!strncmp(p, ab[i].name, ab[i].length)) {
 211                     strbuf_putc(sb, '@');
 212                     strbuf_putc(sb, ab[i].c);
 213                     p += ab[i].length;
 214                     break;
 215                 }
 216             }
 217             if (i >= limit) {
 218                 strbuf_putc(sb, *p);
 219                 p++;
 220             }
 221         } else {
 222             strbuf_putc(sb, *p);
 223             p++;
 224         }
 225     }
 226     if (spaces > 0) {
 227         if (spaces < 4) {
 228             strbuf_nputc(sb, ' ', spaces);
 229         } else if (spaces < 10) {
 230             strbuf_putc(sb, '@');
 231             strbuf_putn(sb, spaces);
 232         } else {
 233             strbuf_putc(sb, '@');
 234             strbuf_putc(sb, '{');
 235             strbuf_putn(sb, spaces);
 236             strbuf_putc(sb, '}');
 237         }
 238     }
 239     return strbuf_value(sb);
 240 }

 242 /**
 243  * uncompress source line.
 244  *
 245  *  @param[in]  in  compressed string
 246  *  @param[in]  name    replaced string
 247  *  @return     uncompressed string
 248  */
 249 char *
 250 uncompress(const char *in, const char *name, STRBUF *sb)
 251 {
 252     const char *p;
 253     int i;
 254 
 255     strbuf_reset(sb);
 256     for (p = in;  *p; p++) {
 257         if (*p == '@') {
 258             int spaces = 0;
 259 
 260             switch (*++p) {
 261             case '@':
 262                 strbuf_putc(sb, '@');
 263                 break;
 264             case 'n':
 265                 strbuf_puts(sb, name);
 266                 break;
 267             case '{':   /* } */
 268                 for (p++; *p && isdigit((unsigned char)*p); p++)
 269                     spaces = spaces * 10 + *p - '0';
 270                 break;
 271             case '0':
 272             case '1':
 273             case '2':
 274             case '3':
 275             case '4':
 276             case '5':
 277             case '6':
 278             case '7':
 279             case '8':
 280             case '9':
 281                 spaces = *p - '0';
 282                 break;
 283             default:
 284                 if (*p < 'a' || *p > 'z')
 285                     die("Abbrev character must be a lower alphabetic character. (%c)", *p);
 286                 i = *p - 'a';
 287                 if (ab2name[i].name)
 288                     strbuf_puts(sb, ab2name[i].name);
 289                 break;
 290                         }
 291             strbuf_nputc(sb, ' ', spaces);
 292                 } else {
 293             strbuf_putc(sb, *p);
 294         }
 295     }
 296     return strbuf_value(sb);
 297 }
