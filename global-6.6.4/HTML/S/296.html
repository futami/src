  22 /*
  23  * (1) ctags -x format (FORMAT_CTAGS_X)
  24  *
  25  * PART_TAG     PART_LNO PART_PATH      PART_LINE
  26  * +----------------------------------------------
  27  * |main             227 ./src/main.c   main()
  28  *
  29  * (2) ctags -x + file id format (FORMAT_CTAGS_XID)
  30  *
  31  * 0    PART_TAG+1 PART_LNO+1 PART_PATH+1   PART_LINE+1
  32  * +----------------------------------------------
  33  * |110 main             227 ./src/main.c   main()
  34  *
  35  * (3) ctags format (FORMAT_CTAGS)
  36  *
  37  * +----------------------------------------------
  38  * |main<TAB>./src/main.c<TAB>227
  39  *
  40  * (4) ctags format (FORMAT_CTAGS_MOD)
  41  *
  42  * +----------------------------------------------
  43  * |./src/main.c<TAB>227<TAB>main()
  44  *
  45  * (5) ctags format (FORMAT_CTAGS_PLUS)
  46  *
  47  * +----------------------------------------------
  48  * |main<TAB>./src/main.c<TAB>227<TAB>main()
  49  *
  50  * (6) path name format (FORMAT_PATH)
  51  *
  52  * +----------------------------------------------
  53  * |./src/main.c
  54  *
  55  * (7) grep format (FORMAT_GREP)
  56  *
  57  * +----------------------------------------------
  58  * |./src/main.c:227:main()
  59  *
  60  * (8) cscope line mode format (FORMAT_CSCOPE)
  61  *
  62  * +----------------------------------------------
  63  * |./src/main.c main 227 main()
  64  *
  65  */

  67 #define FORMAT_CTAGS_X      2
  68 #define FORMAT_CTAGS_XID    3
  69 #define FORMAT_CTAGS_MOD    4
  70 #define FORMAT_CTAGS_PLUS   5
  71 #define FORMAT_PATH     6
  72 #define FORMAT_GREP     7
  73 #define FORMAT_CSCOPE       8
