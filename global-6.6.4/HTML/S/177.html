  96 /**
  97  * !!!
  98  * The following flags are included in the dbopen(3) call as part of the
  99  * open(2) flags.  In order to avoid conflicts with the open flags, start
 100  * at the top of the 16 or 32-bit number space and work our way down.  If
 101  * the open flags were significantly expanded in the future, it could be
 102  * a problem.  Wish I'd left another flags word in the dbopen call.
 103  *
 104  * !!!
 105  * None of this stuff is implemented yet.  The only reason that it's here
 106  * is so that the access methods can skip copying the key/data pair when
 107  * the DB_LOCK flag isn't set.
 108  */
 109 #if UINT_MAX > 65535
 110                 /** Do locking. */
 111 #define DB_LOCK     0x20000000
 112                 /** Use shared memory. */
 113 #define DB_SHMEM    0x40000000
 114                 /** Do transactions. */
 115 #define DB_TXN      0x80000000
 116 #else
 117                 /** Do locking. */
 118 #define DB_LOCK         0x2000
 119                 /** Use shared memory. */
 120 #define DB_SHMEM        0x4000
 121                 /** Do transactions. */
 122 #define DB_TXN          0x8000
 123 #endif

 191 /*
 192  * Little endian <==> big endian 32-bit swap macros.
 193  *  M_32_SWAP   swap a memory location
 194  *  P_32_SWAP   swap a referenced memory location
 195  *  P_32_COPY   swap from one location to another
 196  */
 197 #define M_32_SWAP(a) {                          \
 198     u_int32_t _tmp = a;                     \
 199     ((char *)&a)[0] = ((char *)&_tmp)[3];               \
 200     ((char *)&a)[1] = ((char *)&_tmp)[2];               \
 201     ((char *)&a)[2] = ((char *)&_tmp)[1];               \
 202     ((char *)&a)[3] = ((char *)&_tmp)[0];               \
 203 }
 204 #define P_32_SWAP(a) {                          \
 205     u_int32_t _tmp = *(u_int32_t *)a;               \
 206     ((char *)a)[0] = ((char *)&_tmp)[3];                \
 207     ((char *)a)[1] = ((char *)&_tmp)[2];                \
 208     ((char *)a)[2] = ((char *)&_tmp)[1];                \
 209     ((char *)a)[3] = ((char *)&_tmp)[0];                \
 210 }
 211 #define P_32_COPY(a, b) {                       \
 212     ((char *)&(b))[0] = ((char *)&(a))[3];              \
 213     ((char *)&(b))[1] = ((char *)&(a))[2];              \
 214     ((char *)&(b))[2] = ((char *)&(a))[1];              \
 215     ((char *)&(b))[3] = ((char *)&(a))[0];              \
 216 }
 217 
 218 /*
 219  * Little endian <==> big endian 16-bit swap macros.
 220  *  M_16_SWAP   swap a memory location
 221  *  P_16_SWAP   swap a referenced memory location
 222  *  P_16_COPY   swap from one location to another
 223  */
 224 #define M_16_SWAP(a) {                          \
 225     u_int16_t _tmp = a;                     \
 226     ((char *)&a)[0] = ((char *)&_tmp)[1];               \
 227     ((char *)&a)[1] = ((char *)&_tmp)[0];               \
 228 }
 229 #define P_16_SWAP(a) {                          \
 230     u_int16_t _tmp = *(u_int16_t *)a;               \
 231     ((char *)a)[0] = ((char *)&_tmp)[1];                \
 232     ((char *)a)[1] = ((char *)&_tmp)[0];                \
 233 }
 234 #define P_16_COPY(a, b) {                       \
 235     ((char *)&(b))[0] = ((char *)&(a))[1];              \
 236     ((char *)&(b))[1] = ((char *)&(a))[0];              \
 237 }
