 258 /**
 259  * decide_tag_by_context: decide tag type by context
 260  *
 261  *  @param[in]  tag tag name
 262  *  @param[in]  file    context file
 263  *  @param[in]  lineno  context lineno
 264  *  @return     GTAGS, GRTAGS, GSYMS
 265  */
 266 int
 267 decide_tag_by_context(const char *tag, const char *file, int lineno)
 268 {
 269 #define NEXT_NUMBER(p) do {                                                         \
 270     for (n = 0; isdigit(*p); p++)                                               \
 271         n = n * 10 + (*p - '0');                                            \
 272 } while (0)
 273     STRBUF *sb = NULL;
 274     char path[MAXPATHLEN], s_fid[MAXFIDLEN];
 275     const char *p;
 276     GTOP *gtop;
 277     GTP *gtp;
 278     int db = GSYMS;
 279     int flags = GTOP_NOREGEX;
 280 
 281     if (iflag)
 282         flags |= GTOP_IGNORECASE;
 283     if (normalize(file, get_root_with_slash(), cwd, path, sizeof(path)) == NULL)
 284         die("'%s' is out of the source project.", file);
 285     /*
 286      * get file id
 287      */
 288     if (gpath_open(dbpath, 0) < 0)
 289         die("GPATH not found.");
 290     if ((p = gpath_path2fid(path, NULL)) == NULL)
 291         die("path name in the context is not found.");
 292     strlimcpy(s_fid, p, sizeof(s_fid));
 293     gpath_close();
 294 
 295     /*
 296      * Algorithm:
 297      *
 298      * (1) If context <file, lineno> is a definition of <tag> then use GRTAGS
 299      * (2) else if there is at least one definition of <tag> then use GTAGS
 300      * (3) else use GSYMS.
 301      */
 302     gtop = gtags_open(dbpath, root, GTAGS, GTAGS_READ, 0);
 303     gtp = gtags_first(gtop, tag, flags);
 304     if (gtp) {
 305         db = GTAGS;
 306         for (; gtp; gtp = gtags_next(gtop)) {
 307             /*
 308              * Examine whether each definition record includes the context.
 309              */
 310             p = locatestring(gtp->tagline, s_fid, MATCH_AT_FIRST);
 311             if (p != NULL && *p == ' ') {
 312                 for (p++; *p && *p != ' '; p++)
 313                     ;
 314                 if (*p++ != ' ' || !isdigit(*p))
 315                     die("Impossible! decide_tag_by_context(1)");
 316                 /*
 317                  * Standard format  n <blank> <image>$
 318                  * Compact format   d,d,d,d$
 319                  */
 320                 if (!(gtop->format & GTAGS_COMPACT)) {  /* Standard format */
 321                     if (atoi(p) == lineno) {
 322                         db = GRTAGS;
 323                         goto finish;
 324                     }
 325                 } else {                /* Compact format */
 326                     int n, cur, last = 0;
 327 
 328                     do {
 329                         if (!isdigit(*p))
 330                             die("Impossible! decide_tag_by_context(2)");
 331                         NEXT_NUMBER(p);
 332                         cur = last + n;
 333                         if (cur == lineno) {
 334                             db = GRTAGS;
 335                             goto finish;
 336                         }
 337                         last = cur;
 338                         if (*p == '-') {
 339                             if (!isdigit(*++p))
 340                                 die("Impossible! decide_tag_by_context(3)");
 341                             NEXT_NUMBER(p);
 342                             cur = last + n;
 343                             if (lineno >= last && lineno <= cur) {
 344                                 db = GRTAGS;
 345                                 goto finish;
 346                             }
 347                             last = cur;
 348                         }
 349                         if (*p) {
 350                             if (*p == ',')
 351                                 p++;
 352                             else
 353                                 die("Impossible! decide_tag_by_context(4)");
 354                         }
 355                     } while (*p);
 356                 }
 357             }
 358         }
 359     }
 360 finish:
 361     gtags_close(gtop);
 362     if (db == GSYMS && getenv("GTAGSLIBPATH")) {
 363         char libdbpath[MAXPATHLEN];
 364         char *libdir = NULL, *nextp = NULL;
 365 
 366         sb = strbuf_open(0);
 367         strbuf_puts(sb, getenv("GTAGSLIBPATH"));
 368         back2slash(sb);
 369         for (libdir = strbuf_value(sb); db != GTAGS && libdir; libdir = nextp) {
 370             if ((nextp = locatestring(libdir, PATHSEP, MATCH_FIRST)) != NULL)
 371                 *nextp++ = 0;
 372             if (!gtagsexist(libdir, libdbpath, sizeof(libdbpath), 0))
 373                 continue;
 374             if (!STRCMP(dbpath, libdbpath))
 375                 continue;
 376             gtop = gtags_open(libdbpath, root, GTAGS, GTAGS_READ, 0);
 377             if ((gtp = gtags_first(gtop, tag, flags)) != NULL)
 378                 db = GTAGS;
 379             gtags_close(gtop);
 380         }
 381         strbuf_close(sb);
 382     }
 383     return db;
 384 }

 688     if (getenv("GTAGSTHROUGH"))
 689         Tflag++;
 690     if (use_color) {
 691 #if defined(_WIN32) && !defined(__CYGWIN__)
 692         if (!(getenv("ANSICON") || LoadLibrary("ANSI32.dll")) && use_color == 2)
 693             use_color = 0;
 694 #endif
 695         if (use_color == 2 && !isatty(1))
 696             use_color = 0;
 697         if (Vflag)
 698             use_color = 0;
 699     }
 700     argc -= optind;
 701     argv += optind;
 702     /*
 703      * Path filter
 704      */
 705     if (do_path) {
 706         /*
 707          * This code is needed for globash.rc.
 708          * This code extract path name from tag line and
 709          * replace it with the relative or the absolute path name.
 710          *
 711          * By default, if we are in src/ directory, the output
 712          * should be converted like follows:
 713          *
 714          * main      10 ./src/main.c  main(argc, argv)\n
 715          * main      22 ./libc/func.c   main(argc, argv)\n
 716          *      v
 717          * main      10 main.c  main(argc, argv)\n
 718          * main      22 ../libc/func.c   main(argc, argv)\n
 719          *
 720          * Similarly, the --path-convert=absolute option specified, then
 721          *      v
 722          * main      10 /prj/xxx/src/main.c  main(argc, argv)\n
 723          * main      22 /prj/xxx/libc/func.c   main(argc, argv)\n
 724          */
 725         STRBUF *ib = strbuf_open(MAXBUFLEN);
 726         CONVERT *cv;
 727         char *ctags_x;
 728 
 729         if (argc < 3)
 730             die("global --path-convert: 3 arguments needed.");
 731         cv = convert_open(convert_type, FORMAT_CTAGS_X, argv[0], argv[1], argv[2], stdout, NOTAGS);
 732         while ((ctags_x = strbuf_fgets(ib, stdin, STRBUF_NOCRLF)) != NULL)
 733             convert_put(cv, ctags_x);
 734         convert_close(cv);
 735         strbuf_close(ib);
 736         exit(0);
 737     }
 738     /*
 739      * At first, we pickup pattern from -e option. If it is not found
 740      * then use argument which is not option.
 741      */
 742     if (!av) {
 743         av = *argv;
 744         /*
 745          * global -g pattern [files ...]
 746          *           av      argv
 747          */
 748         if (gflag && av)
 749             argv++;
 750     }
 751     if (single_update) {
 752         if (command == 0) {
 753             uflag++;
 754             command = 'u';
 755         } else if (command != 'u') {
 756             ;   /* ignored */
 757         }
 758     }
 759     /*
 760      * only -c, -u, -P and -p allows no argument.
 761      */
 762     if (!av) {
 763         switch (command) {
 764         case 'c':
 765         case 'u':
 766         case 'p':
 767         case 'P':
 768             break;
 769         case 'f':
 770             if (file_list)
 771                 break;
 772         default:
 773             usage();
 774             break;
 775         }
 776     }
 777     /*
 778      * -u and -p cannot have any arguments.
 779      */
 780     if (av) {
 781         switch (command) {
 782         case 'u':
 783         case 'p':
 784             usage();
 785         default:
 786             break;
 787         }
 788     }
 789     if (tflag)
 790         xflag = 0;
 791     if (nflag > 1)
 792         nosource = 1;   /* to keep compatibility */
 793     if (print0)
 794         set_print0();
 795     if (cflag && match_part == 0)
 796         match_part = MATCH_PART_ALL;
 797     /*
 798      * remove leading blanks.
 799      */
 800     if (!Iflag && !gflag && av)
 801         for (; *av == ' ' || *av == '\t'; av++)
 802             ;
 803     if (cflag && !Pflag && av && isregex(av))
 804         die_with_code(2, "only name char is allowed with -c option.");
 805     /*
 806      * print dbpath or rootdir.
 807      */
 808     if (pflag) {
 809         fprintf(stdout, "%s\n", (rflag) ? root : dbpath);
 810         exit(0);
 811     }
 812     /*
 813      * incremental update of tag files.
 814      */
 815     if (uflag) {
 816         STRBUF  *sb = strbuf_open(0);
 817         char *gtags_path = usable("gtags");
 818 
 819         if (!gtags_path)
 820             die("gtags command not found.");
 821         if (chdir(root) < 0)
 822             die("cannot change directory to '%s'.", root);
 823 #if defined(_WIN32) && !defined(__CYGWIN__)
 824         /*
 825          * Get around CMD.EXE's weird quoting rules by sticking another
 826          * perceived whitespace in front (also works with Take Command).
 827          */
 828         strbuf_putc(sb, ';');
 829 #endif
 830         strbuf_puts(sb, quote_shell(gtags_path));
 831         strbuf_puts(sb, " -i");
 832         if (vflag)
 833             strbuf_puts(sb, " -v");
 834         if (single_update) {
 835             if (!isabspath(single_update)) {
 836                 static char regular_path_name[MAXPATHLEN];
 837 
 838                 if (rel2abs(single_update, cwd, regular_path_name, sizeof(regular_path_name)) == NULL)
 839                     die("rel2abs failed.");
 840                 single_update = regular_path_name;
 841             }
 842             strbuf_puts(sb, " --single-update ");
 843             strbuf_puts(sb, quote_shell(single_update));
 844         }
 845         strbuf_putc(sb, ' ');
 846         strbuf_puts(sb, quote_shell(dbpath));
 847         if (system(strbuf_value(sb)))
 848             exit(1);
 849         strbuf_close(sb);
 850         exit(0);
 851     }
 852     /*
 853      * decide tag type.
 854      */
 855     if (context_file) {
 856         if (!literal && isregex(av))
 857             die_with_code(2, "regular expression is not allowed with the --from-here option.");
 858         db = decide_tag_by_context(av, context_file, atoi(context_lineno));
 859     } else {
 860         if (dflag)
 861             db = GTAGS;
 862         else if (rflag && sflag)
 863             db = GRTAGS + GSYMS;
 864         else
 865             db = (rflag) ? GRTAGS : ((sflag) ? GSYMS : GTAGS);
 866     }
 867     /*
 868      * complete function name
 869      */
 870     if (cflag) {
 871         if (Iflag)
 872             completion_idutils(dbpath, root, av);
 873         else if (Pflag)
 874             completion_path(dbpath, av);
 875         else
 876             completion(dbpath, root, av, db);
 877         exit(0);
 878     }
 879     /*
 880      * make local prefix.
 881      * local prefix must starts with './' and ends with '/'.
 882      */
 883     if (Sflag) {
 884         STRBUF *sb = strbuf_open(0);
 885         static char buf[MAXPATHLEN];
 886         const char *path = scope;
 887     
 888         /*
 889          * normalize the path of scope directory.
 890          */
 891         if (!test("d", path))
 892             die("'%s' not found or not a directory.", scope);
 893         if (!isabspath(path))
 894             path = makepath(cwd, path, NULL);
 895         if (realpath(path, buf) == NULL)
 896             die("cannot get real path of '%s'.", scope);
 897         if (!in_the_project(buf))
 898             die("'%s' is out of the source project.", scope);
 899         scope = buf;
 900         /*
 901          * make local prefix.
 902          */
 903         strbuf_putc(sb, '.');
 904         if (strcmp(root, scope) != 0) {
 905             const char *p = scope + strlen(root);
 906             if (*p != '/')
 907                 strbuf_putc(sb, '/');
 908             strbuf_puts(sb, p);
 909         }
 910         strbuf_putc(sb, '/');
 911         localprefix = check_strdup(strbuf_value(sb));
 912         strbuf_close(sb);
 913 #ifdef DEBUG
 914         fprintf(stderr, "root=%s\n", root);
 915         fprintf(stderr, "cwd=%s\n", cwd);
 916         fprintf(stderr, "localprefix=%s\n", localprefix);
 917 #endif
 918     }
 919     /*
 920      * convert the file-list path into an absolute path.
 921      */
 922     if (file_list && strcmp(file_list, "-") && !isabspath(file_list)) {
 923         static char buf[MAXPATHLEN];
 924 
 925         if (realpath(file_list, buf) == NULL)
 926             die("'%s' not found.", file_list);
 927         file_list = buf;
 928     }
 929     /*
 930      * decide path conversion type.
 931      */
 932     if (nofilter & PATH_FILTER)
 933         type = PATH_THROUGH;
 934     else if (aflag)
 935         type = PATH_ABSOLUTE;
 936     else
 937         type = PATH_RELATIVE;
 938     if (path_style) {
 939         if (!strcmp(path_style, "relative"))
 940             type = PATH_RELATIVE;
 941         else if (!strcmp(path_style, "absolute"))
 942             type = PATH_ABSOLUTE;
 943         else if (!strcmp(path_style, "through"))
 944             type = PATH_THROUGH;
 945         else if (!strcmp(path_style, "shorter"))
 946             type = PATH_SHORTER;
 947         else if (!strcmp(path_style, "abslib")) {
 948             type = PATH_RELATIVE;
 949             abslib++;
 950         } else
 951             die("invalid path style.");
 952     }
 953     {
 954         int conv_flags = 0;
 955 
 956         if (use_color)
 957             conv_flags |= CONVERT_COLOR;
 958         if (gflag)
 959             conv_flags |= CONVERT_GREP;
 960         if (Gflag)
 961             conv_flags |= CONVERT_BASIC;
 962         if (iflag)
 963             conv_flags |= CONVERT_ICASE;
 964         if (Iflag)
 965             conv_flags |= CONVERT_IDUTILS;
 966         if (Pflag)
 967             conv_flags |= CONVERT_PATH;
 968         set_convert_flags(conv_flags);
 969     }
 970     /*
 971      * exec lid(idutils).
 972      */
 973     if (Iflag) {
 974         chdir(root);
 975         idutils(av, dbpath);
 976     }
 977     /*
 978      * search pattern (regular expression).
 979      */
 980     else if (gflag) {
 981         chdir(root);
 982         grep(av, argv, dbpath);
 983     }
 984     /*
 985      * locate paths including the pattern.
 986      */
 987     else if (Pflag) {
 988         chdir(root);
 989         pathlist(av, dbpath);
 990     }
 991     /*
 992      * parse source files.
 993      */
 994     else if (fflag) {
 995         chdir(root);
 996         parsefile(argv, cwd, root, dbpath, db);
 997     }
 998     /*
 999      * tag search.
1000      */
1001     else {
1002         tagsearch(av, cwd, root, dbpath, db);
1003     }
1004     return 0;
1005 }

1240     dbop_close(dbop);

1451  *  @param[in]  pattern POSIX regular expression
1452  *  @param  argv
1453  *  @param  dbpath
1454  */
1455 void
1456 grep(const char *pattern, char *const *argv, const char *dbpath)
1457 {
1458     FILE *fp;
1459     CONVERT *cv;
1460     GFIND *gp = NULL;
1461     STRBUF *ib = strbuf_open(MAXBUFLEN);
1462     const char *path;
1463     char encoded_pattern[IDENTLEN];
1464     const char *buffer;
1465     int linenum, count;
1466     int flags = 0;
1467     int target = GPATH_SOURCE;
1468     regex_t preg;
1469     int user_specified = 1;
1470     int gfind_flags = 0;
1471 
1472     /*
1473      * convert spaces into %FF format.
1474      */
1475     encode(encoded_pattern, sizeof(encoded_pattern), pattern);
1476     /*
1477      * literal search available?
1478      */
1479     if (!literal) {
1480         const char *p = pattern;
1481         int normal = 1;
1482 
1483         for (; *p; p++) {
1484             if (!(isalpha(*p) || isdigit(*p) || isblank(*p) || *p == '_')) {
1485                 normal = 0;
1486                 break;
1487             }
1488         }
1489         if (normal)
1490             literal = 1;
1491     }
1492     if (oflag)
1493         target = GPATH_BOTH;
1494     if (Oflag)
1495         target = GPATH_OTHER;
1496     if (Nflag)
1497         gfind_flags |= GPATH_NEARSORT;
1498     if (literal) {
1499         literal_comple(pattern);
1500     } else {
1501         if (!Gflag)
1502             flags |= REG_EXTENDED;
1503         if (iflag)
1504             flags |= REG_ICASE;
1505         if (regcomp(&preg, pattern, flags) != 0)
1506             die("invalid regular expression.");
1507     }
1508     cv = convert_open(type, format, root, cwd, dbpath, stdout, NOTAGS);
1509     cv->tag_for_display = encoded_pattern;
1510     count = 0;
1511 
1512     if (*argv && file_list)
1513         args_open_both(argv, file_list);
1514     else if (*argv)
1515         args_open(argv);
1516     else if (file_list)
1517         args_open_filelist(file_list);
1518     else {
1519         args_open_gfind(gp = gfind_open(dbpath, localprefix, target, gfind_flags));
1520         user_specified = 0;
1521     }
1522     while ((path = args_read()) != NULL) {
1523         if (user_specified) {
1524             static char buf[MAXPATHLEN];
1525 
1526             if (normalize(path, get_root_with_slash(), cwd, buf, sizeof(buf)) == NULL) {
1527                 warning("'%s' is out of the source project.", path);
1528                 continue;
1529             }
1530             if (test("d", buf)) {
1531                 warning("'%s' is a directory. Ignored.", path);
1532                 continue;
1533             }
1534             if (!test("f", buf)) {
1535                 warning("'%s' not found. Ignored.", path);
1536                 continue;
1537             }
1538             path = buf;
1539         }
1540         if (Sflag && !locatestring(path, localprefix, MATCH_AT_FIRST))
1541             continue;
1542         if (literal) {
1543             int n = literal_search(cv, path);
1544             if (n > 0)
1545                 count += n;
1546         } else {
1547             if (!(fp = fopen(path, "r")))
1548                 die("cannot open file '%s'.", path);
1549             linenum = 0;
1550             while ((buffer = strbuf_fgets(ib, fp, STRBUF_NOCRLF)) != NULL) {
1551                 int result = regexec(&preg, buffer, 0, 0, 0);
1552                 linenum++;
1553                 if ((!Vflag && result == 0) || (Vflag && result != 0)) {
1554                     count++;
1555                     if (format == FORMAT_PATH) {
1556                         convert_put_path(cv, NULL, path);
1557                         break;
1558                     } else {
1559                         convert_put_using(cv, pattern, path, linenum, buffer,
1560                             (user_specified) ? NULL : gp->dbop->lastdat);
1561                     }
1562                 }
1563             }
1564             fclose(fp);
1565         }
1566     }
1567     args_close();
1568     convert_close(cv);
1569     strbuf_close(ib);
1570     if (literal == 0)
1571         regfree(&preg);
1572     if (vflag) {
1573         print_count(count);
1574         fprintf(stderr, " (no index used).\n");
1575     }
1576 }

1670  * void parsefile(char *const *argv, const char *cwd, const char *root, const char *dbpath, int db)
1671  *
1672  * parsefile: parse file to pick up tags.
1673  *
1674  *  @param[in]  argv
1675  *  @param[in]  cwd current directory
1676  *  @param[in]  root    root directory of source tree
1677  *  @param[in]  dbpath  dbpath
1678  *  @param[in]  db  type of parse
1679  */
1680 #define TARGET_DEF  (1 << GTAGS)
1681 #define TARGET_REF  (1 << GRTAGS)
1682 #define TARGET_SYM  (1 << GSYMS)
1683 struct parsefile_data {
1684     CONVERT *cv;
1685     DBOP *dbop;
1686     int target;
1687     int extractmethod;
1688     int count;
1689     const char *fid;            /**< fid of the file under processing */
1690 };
1691 static void
1692 put_syms(int type, const char *tag, int lno, const char *path, const char *line_image, void *arg)
1693 {
1694     struct parsefile_data *data = arg;
1695     const char *key;
1696 
1697     if (format == FORMAT_PATH && data->count > 0)
1698         return;
1699     switch (type) {
1700     case PARSER_DEF:
1701         if (!(data->target & TARGET_DEF))
1702             return;
1703         break;
1704     case PARSER_REF_SYM:
1705         if (!(data->target & (TARGET_REF | TARGET_SYM)))
1706             return;
1707         /*
1708          * extract method when class method definition.
1709          *
1710          * Ex: Class::method(...)
1711          *
1712          * key  = 'method'
1713          * data = 'Class::method  103 ./class.cpp ...'
1714          */
1715         if (data->extractmethod) {
1716             if ((key = locatestring(tag, ".", MATCH_LAST)) != NULL)
1717                 key++;
1718             else if ((key = locatestring(tag, "::", MATCH_LAST)) != NULL)
1719                 key += 2;
1720             else
1721                 key = tag;
1722         } else {
1723             key = tag;
1724         }
1725         if (data->target == TARGET_REF || data->target == TARGET_SYM) {
1726             if (dbop_get(data->dbop, key) != NULL) {
1727                 if (!(data->target & TARGET_REF))
1728                     return;
1729             } else {
1730                 if (!(data->target & TARGET_SYM))
1731                     return;
1732             }
1733         }
1734         break;
1735     default:
1736         return;
1737     }
1738     convert_put_using(data->cv, tag, path, lno, line_image, data->fid);
1739     data->count++;
1740 }

1855 /**

1861  *  @param[in]  dbpath      database directory

1867 {

1870     GTOP *gtop;
1871     GTP *gtp;

1873

1876      * open tag file.

1879     cv = convert_open(type, format, root, cwd, dbpath, stdout, db);
1880     /*
1881      * search through tag file.
1882      */
1883     if (nofilter & SORT_FILTER)
1884         flags |= GTOP_NOSORT;
1885     else if (Nflag)
1886         flags |= GTOP_NEARSORT;

1888         flags |= GTOP_NOREGEX;
1889     else if (Gflag)
1890         flags |= GTOP_BASICREGEX;
1891     if (format == FORMAT_PATH)
1892         flags |= GTOP_PATH;
1893     if (iflag)

1895     for (gtp = gtags_first(gtop, pattern, flags); gtp; gtp = gtags_next(gtop)) {
1896         if (Sflag && !locatestring(gtp->path, localprefix, MATCH_AT_FIRST))

1907 /**
1908  * tagsearch: execute tag search
1909  *
1910  *  @param[in]  pattern     search pattern
1911  *  @param[in]  cwd     current directory
1912  *  @param[in]  root        root of source tree
1913  *  @param[in]  dbpath      database directory
1914  *  @param[in]  db      GTAGS,GRTAGS,GSYMS
1915  */
1916 void
1917 tagsearch(const char *pattern, const char *cwd, const char *root, const char *dbpath, int db)
1918 {
1919     int count, total = 0;
1920     char buffer[IDENTLEN], *p = buffer;
1921     char libdbpath[MAXPATHLEN];
1922 
1923     /*
1924      * trim pattern (^<no regex>$ => <no regex>)
1925      */
1926     if (!literal && pattern) {
1927         strlimcpy(p, pattern, sizeof(buffer));
1928         if (*p++ == '^') {
1929             char *q = p + strlen(p);
1930             if (*--q == '$') {
1931                 *q = 0;
1932                 if (*p == 0 || !isregex(p))
1933                     pattern = p;
1934             }
1935         }
1936     }
1937     /*
1938      * search in current source tree.
1939      */
1940     count = search(pattern, root, cwd, dbpath, db);
1941     total += count;
1942     /*
1943      * search in library path.
1944      */
1945     if (abslib)
1946         type = PATH_ABSOLUTE;
1947     if (db == GTAGS && getenv("GTAGSLIBPATH") && (count == 0 || Tflag) && !Sflag) {
1948         STRBUF *sb = strbuf_open(0);
1949         char *libdir, *nextp = NULL;
1950 
1951         strbuf_puts(sb, getenv("GTAGSLIBPATH"));
1952         back2slash(sb);
1953         /*
1954          * search for each tree in the library path.
1955          */
1956         for (libdir = strbuf_value(sb); libdir; libdir = nextp) {
1957             if ((nextp = locatestring(libdir, PATHSEP, MATCH_FIRST)) != NULL)
1958                 *nextp++ = 0;
1959             if (!gtagsexist(libdir, libdbpath, sizeof(libdbpath), 0))
1960                 continue;
1961             if (!STRCMP(dbpath, libdbpath))
1962                 continue;
1963             if (!test("f", makepath(libdbpath, dbname(db), NULL)))
1964                 continue;
1965             /*
1966              * search again
1967              */
1968             count = search(pattern, libdir, cwd, libdbpath, db);
1969             total += count;
1970             if (count > 0 && !Tflag) {
1971                 /* for verbose message */
1972                 dbpath = libdbpath;
1973                 break;
1974             }
1975         }
1976         strbuf_close(sb);
1977     }
1978     if (vflag) {
1979         print_count(total);
1980         if (!Tflag)
1981             fprintf(stderr, " (using '%s')", makepath(dbpath, dbname(db), NULL));
1982         fputs(".\n", stderr);
1983     }
