 115 /**
 116  * get the reason for skipping
 117  *
 118  *  @param[in]  path    path name (must start with "./")
 119  *  @return     <directory or not> | <reason for skipping>
 120  *      directory or not: 1: directory, 0: file
 121  *      reason: 1: dot file, 0: others
 122  */
 123 static int
 124 getreason(const char *path)
 125 {
 126     int db, type = 0, is_directory = 0;
 127     const char *p;
 128 
 129     /* seek to the last character */
 130     p = path + strlen(path) - 1;
 131     if (*p == '/') {
 132         is_directory = 1;
 133         p--;
 134     }
 135     /* seek to the basename */
 136     for (; path < p; p--)
 137         if (*p == '/')
 138             break;
 139     if (*p == '/')
 140         p++;
 141     /* check for dot files */
 142     if (*p == '.') {
 143         type = 1;
 144     } else {
 145         /* check for tag files */
 146         for (db = 0; db < GTAGLIM; db++)
 147             if (!strcmp(dbname(db), p))
 148                 type = 2;
 149     }
 150     return is_directory << 8 | type;
 151 }
 152 /**
 153  * prepare_source: preparing regular expression.
 154  *
 155  *  @return compiled regular expression for source files.
 156  */
 157 static regex_t *
 158 prepare_source(void)
 159 {
 160     static regex_t suff_area;
 161     STRBUF *sb = strbuf_open(0);
 162     char *default_langmap = DEFAULTLANGMAP;
 163     char *langmap = default_langmap;
 164     char *p;
 165     int flags = REG_EXTENDED;
 166 
 167     /*
 168      * load icase_path option.
 169      */
 170     if (getconfb("icase_path"))
 171         flags |= REG_ICASE;
 172 #if defined(_WIN32) || defined(__DJGPP__)
 173     flags |= REG_ICASE;
 174 #endif
 175     /*
 176      * make suffix list.
 177      */
 178     strbuf_reset(sb);
 179     if (getconfs("langmap", sb))
 180         langmap =  check_strdup(strbuf_value(sb));
 181     strbuf_reset(sb);
 182     p = langmap;
 183     strbuf_puts(sb, "/(");
 184     if (debug)
 185         fprintf(stderr, "langmap = %s\n", langmap);
 186     while (*p) {
 187         /* skip language name */
 188         for (; *p && *p != ':'; p++)
 189             ;
 190         if (*p == 0)
 191             die_with_code(2, "syntax error in the langmap '%s'.", langmap);
 192         p++;
 193         /* pick up a suffix or a glob pattern */
 194         while (*p == '.' || *p == '(') {
 195             if (*p == '.') {    /* suffix */
 196                 strbuf_puts(sb, "[^/]+\\.");
 197                 for (p++; *p && *p != '.' && *p != '(' && *p != ','; p++) {
 198                     if (!isalnum(*p))
 199                         strbuf_putc(sb, '\\');
 200                     strbuf_putc(sb, *p);
 201                 }
 202             } else if (*p == '(') { /* glob pattern */
 203                 for (p++; *p && *p != ')'; p++) {
 204                     if (*p == '.')
 205                         strbuf_puts(sb, "\\.");
 206                     else if (*p == '*')
 207                         strbuf_puts(sb, "[^/]*");
 208                     else if (*p == '?')
 209                         strbuf_puts(sb, "[^/]");
 210                     else if (*p == '[') {
 211                         strbuf_putc(sb, '[');
 212                         if (*++p == '!') {
 213                             strbuf_putc(sb, '^');
 214                             p++;
 215                         }
 216                         for (; *p && *p != ']'; p++)
 217                             strbuf_putc(sb, *p);
 218                         if (*p == 0)
 219                             die_with_code(2, "syntax error in the langmap '%s'.", langmap);
 220                         strbuf_putc(sb, ']');
 221                     } else
 222                         strbuf_putc(sb, *p);
 223                 }
 224                 if (*p == 0)
 225                     die_with_code(2, "syntax error in the langmap '%s'.", langmap);
 226                 p++;
 227             }
 228             strbuf_putc(sb, '|');
 229         }
 230         if (*p == ',')
 231             p++;
 232     }
 233     strbuf_unputc(sb, '|');
 234     strbuf_puts(sb, ")$");
 235     if (debug)
 236         fprintf(stderr, "prepare_source: %s\n", strbuf_value(sb));
 237     if (regcomp(&suff_area, strbuf_value(sb), flags) != 0)
 238         die("cannot compile regular expression.");
 239     strbuf_close(sb);
 240     if (langmap != default_langmap)
 241         free(langmap);
 242     return &suff_area;
 243 }
 244 /**
 245  * prepare_skip: prepare skipping files.
 246  *
 247  *  Globals used (output):
 248  *      listarray[]:    skip list.
 249  *      list_count:  count of skip list.
 250  *
 251  *  @return compiled regular expression for skip files.
 252  */
 253 static regex_t *
 254 prepare_skip(void)
 255 {
 256     static regex_t skip_area;
 257     char *skiplist;
 258     STRBUF *reg = strbuf_open(0);
 259     char *p, *q;
 260     int flags = REG_EXTENDED|REG_NEWLINE;
 261 
 262     /*
 263      * load icase_path option.
 264      */
 265     if (getconfb("icase_path"))
 266         flags |= REG_ICASE;
 267 #if defined(_WIN32) || defined(__DJGPP__)
 268     flags |= REG_ICASE;
 269 #endif
 270     /*
 271      * load skip data.
 272      */
 273     if (!getconfs("skip", reg)) {
 274         strbuf_close(reg);
 275         return NULL;
 276     }
 277     skiplist = check_strdup(strbuf_value(reg));
 278     if (debug)
 279         fprintf(stderr, "DBG: Original skip list:\n%s\n", skiplist);
 280     strbuf_reset(reg);
 281     /*
 282      * construct regular expression.
 283      */
 284     strbuf_putc(reg, '(');  /* ) */
 285     /*
 286      * Hard coded skip files:
 287      * (1) files which start with '.'
 288      * (2) tag files
 289      */
 290     /* skip files which start with '.' e.g. .cvsignore */
 291     if (!accept_dotfiles) {
 292         strbuf_puts(reg, "/\\.[^/]+$|");
 293         strbuf_puts(reg, "/\\.[^/]+/|");
 294     }
 295     /* skip tag files */
 296     strbuf_puts(reg, "/GTAGS$|");
 297     strbuf_puts(reg, "/GRTAGS$|");
 298     strbuf_puts(reg, "/GSYMS$|");
 299     strbuf_puts(reg, "/GPATH$|");
 300     for (p = skiplist; *p; ) {
 301         char *skipf;
 302         STATIC_STRBUF(sb);
 303         strbuf_clear(sb);
 304 
 305         while (*p && *p == ',')
 306             p++;
 307         if (*p == '\0')
 308             break;
 309         for (; *p; p++) {
 310             if (*p == ',')
 311                 break;
 312             if (*p == '\\' && *(p + 1) == ',') 
 313                 p++;
 314             strbuf_putc(sb, *p);
 315         }
 316         skipf = strbuf_value(sb);
 317         /* '/' means project root directory */
 318         if (*skipf == '/') {
 319             strbuf_puts(reg, "^\\./");
 320             skipf++;
 321         } else {
 322             strbuf_putc(reg, '/');
 323         }
 324         for (q = skipf; *q; q++) {
 325             /*
 326              * replaces wild cards into regular expressions.
 327              *
 328              * '*' -> '[^/]*'
 329              * '?' -> '[^/]'
 330              * '[...]' -> '[...]'
 331              * '[!...]' -> '[^...]'
 332              */
 333             if (*q == '[') {
 334                 char *c = q;
 335                 STATIC_STRBUF(class);
 336                 int isclass = 1;
 337 
 338                 strbuf_clear(class);
 339                 strbuf_putc(class, *c++);       /* '[' */
 340                 if (*c == '\0')
 341                     isclass = 0;
 342                 else if (*c == ']')
 343                     strbuf_putc(class, *c++);
 344                 else if (*c == '!') {
 345                     strbuf_putc(class, '^');
 346                     c++;
 347                 } else
 348                     strbuf_putc(class, *c++);
 349                 if (isclass) {
 350                     while (*c && *c != ']')
 351                         strbuf_putc(class, *c++);
 352                     if (*c == ']')
 353                         strbuf_putc(class, *c); /* ']' */
 354                     else
 355                         isclass = 0;
 356                 }
 357                 if (isclass) {
 358                     strbuf_puts(reg, strbuf_value(class));
 359                     q = c;
 360                 } else {
 361                     /* 'class' is thrown away */
 362                     strbuf_putc(reg, '\\');
 363                     strbuf_putc(reg, *q);
 364                 }
 365             } else if (*q == '*')
 366                 strbuf_puts(reg, "[^/]*");
 367             else if (*q == '?')
 368                 strbuf_puts(reg, "[^/]");
 369             else if (*q == '\\' && *(q + 1) == ',')
 370                 strbuf_putc(reg, *++q);
 371             else if (isregexchar(*q)) {
 372                 strbuf_putc(reg, '\\');
 373                 strbuf_putc(reg, *q);
 374             } else {
 375                 if (*q == '\\' && *(q + 1) != '\0') {
 376                     strbuf_putc(reg, *q++);
 377                     strbuf_putc(reg, *q);
 378                 } else
 379                     strbuf_putc(reg, *q);
 380             }
 381         }
 382         if (*(q - 1) != '/')
 383             strbuf_putc(reg, '$');
 384         if (*p == ',')
 385             strbuf_putc(reg, '|');
 386     }
 387     strbuf_unputc(reg, '|');
 388     strbuf_putc(reg, ')');
 389     /*
 390      * compile regular expression.
 391      */
 392     if (debug)
 393         fprintf(stderr, "DBG: Regular expression of the skip list:\n%s\n", strbuf_value(reg));
 394     if (regcomp(&skip_area, strbuf_value(reg), flags) != 0)
 395         die("cannot compile regular expression.");
 396     strbuf_close(reg);
 397     free(skiplist);
 398 
 399     return &skip_area;
 400 }

 419 /**
 420  * skipthisfile: check whether or not we accept this file.
 421  *
 422  *  @param[in]  path    path name (must start with "./")
 423  *  @return     1: skip, 0: don't skip
 424  *
 425  * [Note]
 426  * Specification of required path name:
 427  * - Path must start with "./".
 428  * - Directory path name must end with "/".
 429  */
 430 int
 431 skipthisfile(const char *path)
 432 {
 433     regmatch_t m;
 434     int i;
 435 
 436     /*
 437      * unit check.
 438      */
 439     if (skip == NULL) {
 440         skip = prepare_skip();
 441         if (skip == NULL)
 442             die("prepare_skip failed.");
 443     }
 444     if (regexec(skip, path, 1, &m, 0) == 0) {
 445         if (debug) {
 446             int len = strlen(path);
 447             fprintf(stderr, "DBG: ");
 448             for (i = 0; i < len; i++) {
 449                 if (m.rm_so == i)
 450                     fputc('[', stderr);
 451                 if (m.rm_eo == i)
 452                     fputc(']', stderr);
 453                 fputc(path[i], stderr);
 454             }
 455             if (m.rm_eo == len)
 456                 fputc(']', stderr);
 457             fprintf(stderr, " => SKIPPED\n");
 458         }
 459         if (find_explain) {
 460             int type = getreason(path);
 461             const char *kind = (type >> 8) ? "Directory" : "File";
 462 
 463             fprintf(stderr, " - %s '%s' is skipped", kind, trimpath(path));
 464             switch (type & 0xff) {
 465             case 1:
 466                 fprintf(stderr, " because the name begins with a dot.\n");
 467                 break;
 468             case 2:
 469                 fprintf(stderr, " because it is a tag file.\n");
 470                 break;
 471             case 0:     
 472                 fprintf(stderr, " by the skip list.\n");
 473                 break;
 474             }
 475         }
 476         return 1;
 477     } else {
 478         if (debug)
 479             fprintf(stderr, "%s\n", path);
 480     }
 481     return 0;
 482 }

 508 /**
 509  * has_symlinkloop: whether or not dir has a symbolic link loops.
 510  *
 511  *  @param[in]  dir directory (should end by "/")
 512  *  @return     1: has a loop, 0: don't have a loop
 513  */
 514 static int
 515 has_symlinkloop(const char *dir)
 516 {
 517     struct stack_entry *sp;
 518     char *real, *p;
 519     int i, ret = 0;
 520 
 521     if (!strcmp(dir, "./"))
 522         return 0;
 523     if ((real = realpath(dir, NULL)) == NULL)
 524         die("cannot get real path of '%s'.", trimpath(dir));
 525 #ifdef SLOOPDEBUG
 526     fprintf(stderr, "======== has_symlinkloop ======\n");
 527     fprintf(stderr, "dir = '%s', real path = '%s'\n", dir, real);
 528     fprintf(stderr, "TEST-1\n");
 529     fprintf(stderr, "\tcheck '%s' < '%s'\n", real, rootdir);
 530 #endif
 531     p = locatestring(rootdir, real, MATCH_AT_FIRST);
 532     if (p && (*p == '/' || *p == '\0' || !strcmp(real, "/"))) {
 533         ret = 1;
 534         goto out;
 535     }
 536     sp = varray_assign(stack, 0, 0);
 537 #ifdef SLOOPDEBUG
 538     fprintf(stderr, "TEST-2\n");
 539 #endif
 540     for (i = current_entry; i >= 0; i--) {
 541 #ifdef SLOOPDEBUG
 542         fprintf(stderr, "%d:\tcheck '%s' == '%s'\n", i, real, sp[i].real);
 543 #endif
 544         if (!strcmp(sp[i].real, real)) {
 545             ret = 1;
 546             goto out;
 547         }
 548     }
 549 #ifdef SLOOPDEBUG
 550     fprintf(stderr, "===============================\n");
 551 #endif
 552 out:
 553     free(real);
 554     return ret;
 555 }
 556 /**
 557  * skips '.', '..'.
 558  * It also skips '.xxxx' and tag files for performance.
 559  */
 560 static int
 561 ignore(const char *path)
 562 {
 563     int db;
 564 
 565     if (path[0] == '.') {
 566         if (path[1] == '\0' || (path[1] == '.' && path[2] == '\0'))
 567             return 1;
 568         if (!find_explain && !accept_dotfiles)
 569             return 1;
 570     }
 571     if (!find_explain) {
 572         if (path[0] == 'G') {
 573             for (db = 0; db < GTAGLIM; db++)
 574                 if (!strcmp(dbname(db), path))
 575                     return 1;
 576         }
 577     }
 578     return 0;
 579 }
 580 /**
 581  * getdirs: get directory list
 582  *
 583  *  @param[in]  dir directory (should end by "/")
 584  *  @param[out] sb  string buffer
 585  *  @return     -1: error, 0: normal
 586  *
 587  * format of directory list:
 588  * |ddir1\0ffile1\0|
 589  * means directory "dir1", file "file1".
 590  */
 591 static int
 592 getdirs(const char *dir, STRBUF *sb)
 593 {
 594     DIR *dirp;
 595     struct dirent *dp;
 596     struct stat st;
 597 
 598     if (check_looplink && has_symlinkloop(dir)) {
 599         warning("symbolic link loop detected. '%s' is ignored.", trimpath(dir));
 600         return -1;
 601     }
 602     if ((dirp = opendir(dir)) == NULL) {
 603         warning("cannot open directory '%s'. ignored.", trimpath(dir));
 604         return -1;
 605     }
 606     while ((dp = readdir(dirp)) != NULL) {
 607         if (ignore(dp->d_name))
 608             continue;
 609         if (stat(makepath(dir, dp->d_name, NULL), &st) < 0) {
 610             warning("cannot stat '%s'. ignored.", trimpath(dp->d_name));
 611             continue;
 612         }
 613         if (S_ISSOCK(st.st_mode) || S_ISFIFO(st.st_mode) || S_ISCHR(st.st_mode) || S_ISBLK(st.st_mode)) {
 614             warning("file is not regular file '%s'. ignored.", trimpath(dp->d_name));
 615             continue;
 616         }
 617         if (access(makepath(dir, dp->d_name, NULL), R_OK) < 0) {
 618             if (!skip_unreadable)
 619                 die("cannot read file '%s'.", trimpath(dp->d_name));
 620             warning("cannot read '%s'. ignored.", trimpath(dp->d_name));
 621             continue;
 622         }
 623 #ifndef __DJGPP__
 624         if (skip_symlink > 0) {
 625 #if defined(_WIN32) && !defined(__CYGWIN__)
 626             DWORD attr = GetFileAttributes(makepath(dir, dp->d_name, NULL));
 627             if (attr != -1 && (attr & FILE_ATTRIBUTE_REPARSE_POINT))
 628 #else
 629             struct stat st2;
 630 
 631             if (lstat(makepath(dir, dp->d_name, NULL), &st2) < 0) {
 632                 warning("cannot lstat '%s'. ignored.", trimpath(dp->d_name));
 633                 continue;
 634             }
 635             if (S_ISLNK(st2.st_mode))
 636 #endif
 637             {
 638                 if (((skip_symlink & SKIP_SYMLINK_FOR_DIR) && S_ISDIR(st.st_mode)) ||
 639                     ((skip_symlink & SKIP_SYMLINK_FOR_FILE) && S_ISREG(st.st_mode)))
 640                 {
 641                     if (find_explain)
 642                         fprintf(stderr, " - Symbolic link '%s' is skipped.\n",
 643                             trimpath(makepath(dir, dp->d_name, NULL)));
 644                     continue;
 645                 }
 646             }
 647         }
 648 #endif
 649         if (S_ISDIR(st.st_mode))
 650             strbuf_putc(sb, 'd');
 651         else if (S_ISREG(st.st_mode))
 652             strbuf_putc(sb, 'f');
 653         else
 654             strbuf_putc(sb, ' ');
 655         strbuf_puts(sb, dp->d_name);
 656         strbuf_putc(sb, '\0');
 657     }
 658     (void)closedir(dirp);
 659     return 0;
 660 }

 693  *  @param[in]  start   start directory,
 694  *          If NULL, assumed "." (current) directory.
 695  *  @param[in]  explain print verbose message
 696  */
 697 void
 698 find_open(const char *start, int explain)
 699 {
 700     struct stack_entry *curp;
 701     assert(find_mode == 0);
 702     find_mode = FIND_OPEN;
 703     find_explain = explain;
 704 
 705     if (!start)
 706         start = "./";
 707         if ((rootdir = realpath(start, NULL)) == NULL)
 708                 die("cannot get real path of '%s'.", trimpath(dir));
 709     /*
 710      * setup stack.
 711      */
 712     stack = varray_open(sizeof(struct stack_entry), 50);
 713     current_entry = 0;
 714     curp = varray_assign(stack, current_entry, 1);
 715     strlimcpy(dir, start, sizeof(dir));
 716     curp->dirp = dir + strlen(dir);
 717     curp->sb = strbuf_open(0);
 718     curp->real = getrealpath(dir);
 719     if (getdirs(dir, curp->sb) < 0)
 720         die("Work is given up.");
 721     curp->start = curp->p = strbuf_value(curp->sb);
 722     curp->end   = curp->start + strbuf_getlen(curp->sb);
 723     strlimcpy(cwddir, get_root(), sizeof(cwddir));
 724 }
 725 /**
 726  * find_open_filelist: find_open like interface for handling output of find(1).
 727  *
 728  *  @param[in]  filename    file including list of file names.
 729  *              When "-" is specified, read from standard input.
 730  *  @param[in]  root        root directory of source tree
 731  *  @param[in]  explain print verbose message
 732  */
 733 void
 734 find_open_filelist(const char *filename, const char *root, int explain)
 735 {
 736     size_t rootdir_len;
 737     assert(find_mode == 0);
 738     find_mode = FILELIST_OPEN;
 739     find_explain = explain;
 740 
 741     if (!strcmp(filename, "-")) {
 742         /*
 743          * If the filename is '-', copy standard input onto
 744          * temporary file to be able to read repeatedly.
 745          */
 746         if (temp == NULL) {
 747             char buf[MAXPATHLEN];
 748 
 749             temp = tmpfile();
 750             while (fgets(buf, sizeof(buf), stdin) != NULL)
 751                 fputs(buf, temp);
 752         }
 753         rewind(temp);
 754         ip = temp;
 755     } else {
 756         ip = fopen(filename, "r");
 757         if (ip == NULL)
 758             die("cannot open '%s'.", trimpath(filename));
 759     }
 760     /*
 761      * rootdir always ends with '/'.
 762      *
 763      * rootdir_len is two characters longer than root to be able
 764      * to append a "/" if needed and the string terminator
 765      * ofcourse.
 766      */
 767     rootdir_len = strlen(root) + 2;
 768     rootdir = malloc(rootdir_len);
 769     if (!rootdir)
 770         die("short of memory.");
 771 
 772     snprintf(rootdir, rootdir_len, "%s%s", root,
 773          strcmp(root+ROOT, "/") ? "/" : "");
 774 
 775     strlimcpy(cwddir, root, sizeof(cwddir));
 776 }
 777 /**
 778  * find_read_filelist: read path from file
 779  *
 780  *  @return     path
 781  */
 782 static char *
 783 find_read_filelist(void)
 784 {
 785     STATIC_STRBUF(ib);
 786     static char buf[MAXPATHLEN + 1];
 787     static char *path;
 788 
 789     strbuf_clear(ib);
 790     for (;;) {
 791         path = strbuf_fgets(ib, ip, STRBUF_NOCRLF);
 792         if (path == NULL) {
 793             /* EOF */
 794             find_eof = 1;
 795             return NULL;
 796         }
 797         if (*path == '\0') {
 798             /* skip empty line.  */
 799             continue;
 800         }
 801         /*
 802          * Lines which start with ". " are considered to be comments.
 803          */
 804         if (*path == '.' && *(path + 1) == ' ')
 805             continue;
 806         /*
 807          * Skip the following:
 808          * o directory
 809          * o file which does not exist
 810          * o dead symbolic link
 811          */
 812         if (!test("f", path)) {
 813             if (test("d", path))
 814                 warning("'%s' is a directory. ignored.", trimpath(path));
 815             else
 816                 warning("'%s' not found. ignored.", trimpath(path));
 817             continue;
 818         }
 819         /*
 820          * normalize path name.
 821          *
 822          *  rootdir  /a/b/
 823          *  buf      /a/b/c/d.c -> c/d.c -> ./c/d.c
 824          */
 825         if (normalize(path, rootdir, cwddir, buf, sizeof(buf)) == NULL) {
 826             warning("'%s' is out of source tree. ignored.", trimpath(path));
 827             continue;
 828         }
 829         path = buf;
 830         /*
 831          * Now GLOBAL can treat the path which includes blanks.
 832          * This message is obsoleted.
 833          */
 834         if (!allow_blank && locatestring(path, " ", MATCH_LAST)) {
 835             warning("'%s' ignored, because it includes blank.", trimpath(path));
 836             continue;
 837         }
 838         if (skipthisfile(path))
 839             continue;
 840         /*
 841          * A blank at the head of path means
 842          * other than source file.
 843          */
 844         if (!issourcefile(path))
 845             *--path = ' ';
 846         return path;
 847     }
 848 }
 849 /**
 850  * find_read_traverse: read path without GPATH.
 851  *
 852  *  @return     path
 853  */
 854 static char *
 855 find_read_traverse(void)
 856 {
 857     static char val[MAXPATHLEN];
 858     char path[MAXPATHLEN];
 859     struct stack_entry *curp = varray_assign(stack, current_entry, 1);
 860 
 861     for (;;) {
 862         while (curp->p < curp->end) {
 863             char type = *(curp->p);
 864             const char *unit = curp->p + 1;
 865 
 866             curp->p += strlen(curp->p) + 1;
 867 
 868             /*
 869              * Skip files described in the skip list.
 870              */
 871                 /* makepath() returns unsafe module local area. */
 872             strlimcpy(path, makepath(dir, unit, NULL), sizeof(path));
 873             if (type == 'd')
 874                 strcat(path, "/");
 875             if (skipthisfile(path))
 876                 continue;
 877             if (type == 'f') {
 878                 /*
 879                  * Skip the following:
 880                  * o directory
 881                  * o file which does not exist
 882                  * o dead symbolic link
 883                  */
 884                 if (!test("f", path)) {
 885                     if (test("d", path))
 886                         warning("'%s' is a directory. ignored.", trimpath(path));
 887                     else
 888                         warning("'%s' not found. ignored.", trimpath(path));
 889                     continue;
 890                 }
 891                 /*
 892                  * Now GLOBAL can treat the path which includes blanks.
 893                  * This message is obsoleted.
 894                  */
 895                 if (!allow_blank && locatestring(path, " ", MATCH_FIRST)) {
 896                     warning("'%s' ignored, because it includes blank.", trimpath(path));
 897                     continue;
 898                 }
 899                 /*
 900                  * A blank at the head of path means
 901                  * other than source file.
 902                  */
 903                 if (issourcefile(path)) {
 904                     strlimcpy(val, path, sizeof(val));
 905                 } else {
 906                     /* other file like 'Makefile' */
 907                     val[0] = ' ';
 908                     strlimcpy(&val[1], path, sizeof(val) - 1);
 909                 }
 910                 val[sizeof(val) - 1] = '\0';
 911                 return val;
 912             }
 913             if (type == 'd') {
 914                 STRBUF *sb = strbuf_open(0);
 915                 char *dirp = curp->dirp;
 916                 strcat(dirp, unit);
 917                 strcat(dirp, "/");
 918                 if (getdirs(dir, sb) < 0) {
 919                     strbuf_close(sb);
 920                     *(curp->dirp) = 0;
 921                     continue;
 922                 }
 923                 /*
 924                  * Push stack.
 925                  */
 926                 curp = varray_assign(stack, ++current_entry, 1);
 927                 curp->dirp = dirp + strlen(dirp);
 928                 curp->real = getrealpath(dir);
 929                 curp->sb = sb;
 930                 curp->start = curp->p = strbuf_value(sb);
 931                 curp->end   = curp->start + strbuf_getlen(sb);
 932             }
 933         }
 934         strbuf_close(curp->sb);
 935         curp->sb = NULL;
 936         free(curp->real);
 937         curp->real = NULL;
 938         if (current_entry == 0)
 939             break;
 940         /*
 941          * Pop stack.
 942          */
 943         curp = varray_assign(stack, --current_entry, 0);
 944         *(curp->dirp) = 0;
 945     }
 946     find_eof = 1;
 947     return NULL;
 948 }
 949 /**
 950  * find_read: read path without GPATH.
 951  *
 952  *  @return     path
 953  */
 954 char *
 955 find_read(void)
 956 {
 957     static char *path;
 958 
 959     assert(find_mode != 0);
 960     if (find_eof)
 961         path = NULL;
 962     else if (find_mode == FILELIST_OPEN)
 963         path = find_read_filelist();
 964     else if (find_mode == FIND_OPEN)
 965         path = find_read_traverse();
 966     else
 967         die("find_read: internal error.");
 968     return path;
 969 }
 970 /**
 971  * find_close: close iterator.
 972  */
 973 void
 974 find_close(void)
 975 {
 976     assert(find_mode != 0);
 977     if (find_mode == FIND_OPEN) {
 978         if (stack)
 979             varray_close(stack);
 980     } else if (find_mode == FILELIST_OPEN) {
 981         /*
 982          * The --file=- option is specified, we don't close file
 983          * to read it repeatedly.
 984          */
 985         if (ip != temp)
 986             fclose(ip);
 987     } else {
 988         die("find_close: internal error.");
 989     }
 990     if (rootdir)
 991         free(rootdir);
 992     if (suff)
 993         regfree(suff);
 994     if (skip)
 995         regfree(skip);
 996     find_eof = find_mode = 0;
 997 }
