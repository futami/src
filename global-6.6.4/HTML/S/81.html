  33 /*
  34  *  display help
  35  *
  36  *  cscope - interactive C symbol cross-reference
  37  *
  38  */
  39 
  40 #include "global-cscope.h"
  41 #if defined(USE_NCURSES) && !defined(RENAMED_NCURSES)
  42 #include <ncurses.h>
  43 #else
  44 #include <curses.h>
  45 #endif
  46 /**
  47     max num of lines of help screen -
  48     this number needs to be increased if more than n help items are needed
  49 */
  50 #define MAXHELP 50  /* maximum number of help strings */
  51 
  52 void
  53 help(void)
  54 {
  55     char    **ep, *s, **tp, *text[MAXHELP]; 
  56     int ln;
  57 
  58     tp = text;
  59     if (changing == NO) {
  60         if (mouse) {
  61             *tp++ = "Point with the mouse and click button 1 to move to the desired input field,\n";
  62             *tp++ = "type the pattern to search for, and then press the RETURN key.  For the first 4\n";
  63             *tp++ = "and last 2 input fields, the pattern can be a regcomp(3) regular expression.\n";
  64             *tp++ = "If the search is successful, you can edit the file containing a displayed line\n";
  65             *tp++ = "by pointing with the mouse and clicking button 1.\n";
  66             *tp++ = "\nYou can either use the button 2 menu or these single-character commands:\n\n";
  67         } else {
  68             *tp++ = "Press the RETURN key repeatedly to move to the desired input field, type the\n";
  69             *tp++ = "pattern to search for, and then press the RETURN key.  For the first 4 and\n";
  70             *tp++ = "last 2 input fields, the pattern can be a regcomp(3) regular expression.\n";
  71             *tp++ = "If the search is successful, you can use these single-character commands:\n\n";
  72             *tp++ = "0-9a-zA-Z\tEdit the file containing the displayed line.\n";
  73         }
  74         *tp++ = "space bar\tDisplay next set of matching lines.\n";
  75         *tp++ = "+\t\tDisplay next set of matching lines.\n";
  76         *tp++ = "^V\t\tDisplay next set of matching lines.\n";
  77         *tp++ = "-\t\tDisplay previous set of matching lines.\n";
  78         *tp++ = "^E\t\tEdit all lines.\n";
  79         *tp++ = ">\t\tWrite the list of lines being displayed to a file.\n";
  80         *tp++ = ">>\t\tAppend the list of lines being displayed to a file.\n";
  81         *tp++ = "<\t\tRead lines from a file.\n";
  82         *tp++ = "^\t\tFilter all lines through a shell command.\n";
  83         *tp++ = "|\t\tPipe all lines to a shell command.\n";
  84         *tp++ = "^G\t\tRead lines from the result of the execution of global(1).\n";
  85         if (!mouse) {
  86             *tp++ = "\nAt any time you can use these single-character commands:\n\n";
  87             *tp++ = "TAB\t\tSwap positions between input and output areas.\n";
  88             *tp++ = "RETURN\t\tMove to the next input field.\n";
  89             *tp++ = "^N\t\tMove to the next input field.\n";
  90             *tp++ = "^P\t\tMove to the previous input field.\n";
  91         }
  92         *tp++ = "^Y / ^A\t\tSearch with the last pattern typed.\n";
  93         *tp++ = "^B\t\tRecall previous input field and search pattern.\n";
  94         *tp++ = "^F\t\tRecall next input field and search pattern.\n";
  95         if(caseless)
  96             *tp++ = "^C\t\tToggle ignore/use letter case when searching (IGNORE).\n";
  97         else
  98             *tp++ = "^C\t\tToggle ignore/use letter case when searching (USE).\n";
  99         *tp++ = "^R\t\tRebuild the cross-reference.\n";
 100         *tp++ = "!\t\tStart an interactive shell (type ^D to return to cscope).\n";
 101         *tp++ = "^L\t\tRedraw the screen.\n";
 102         *tp++ = "?\t\tDisplay this list of commands.\n";
 103         *tp++ = "^D\t\tExit cscope.\n";
 104         *tp++ = "\nNote: If the first character of the pattern you want to search for matches\n";
 105         *tp++ = "a command, type a \\ character first.\n";
 106         *tp++ = "Note: Some ctrl keys may be occupied by your terminal configuration.\n";
 107     } else {
 108         if (mouse) {
 109             *tp++ = "Point with the mouse and click button 1 to mark or unmark the line to be\n";
 110             *tp++ = "changed.  You can also use the button 2 menu or these single-character\n";
 111             *tp++ = "commands:\n\n";
 112         }
 113         else {
 114             *tp++ = "When changing text, you can use these single-character commands:\n\n";
 115             *tp++ = "0-9a-zA-Z\tMark or unmark the line to be changed.\n";
 116         }
 117         *tp++ = "*\t\tMark or unmark all displayed lines to be changed.\n";
 118         *tp++ = "space bar\tDisplay next set of lines.\n";
 119         *tp++ = "+\t\tDisplay next set of lines.\n";
 120         *tp++ = "-\t\tDisplay previous set of lines.\n";
 121         *tp++ = "^A\t\tMark or unmark all lines to be changed.\n";
 122         *tp++ = "^D\t\tChange the marked lines and exit.\n";
 123         *tp++ = "ESC\t\tExit without changing the marked lines.\n";
 124         *tp++ = "!\t\tStart an interactive shell (type ^D to return to cscope).\n";
 125         *tp++ = "^L\t\tRedraw the screen.\n";
 126         *tp++ = "?\t\tDisplay this list of commands.\n";
 127     }
 128     /* print help, a screen at a time */
 129     ep = tp;
 130     ln = 0;
 131     for (tp = text; tp < ep; ) {
 132         if (ln < LINES - 1) {
 133             for (s = *tp; *s != '\0'; ++s) {
 134                 if (*s == '\n') {
 135                     ++ln;
 136                 }
 137             }
 138             (void) addstr(*tp++);
 139         }
 140         else {
 141             (void) addstr("\n");
 142             askforchar();
 143             (void) clear();
 144             ln = 0;
 145         }
 146     }
 147     if (ln) {
 148         (void) addstr("\n");
 149         askforchar();
 150     }
 151 }
