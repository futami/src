  62 /** get next number and seek to the next character */
  63 #define GET_NEXT_NUMBER(p) do {                                                \
  64                 if (!isdigit(*p))                                              \
  65                         p++;                                                   \
  66                 for (n = 0; isdigit(*p); p++)                                  \
  67                         n = n * 10 + (*p - '0');                               \
  68         } while (0)

  70 void
  71 start_output(int a_format, int a_nosource)
  72 {
  73     format = a_format;
  74     nosource = a_nosource;
  75     curpath[0] = curtag[0] = '\0';
  76     cur_lineno = last_lineno = 0;
  77     fp = NULL;
  78     src = "";
  79     sb_uncompress = strbuf_open(0);
  80 }
  81 void
  82 end_output(void)
  83 {
  84     if (sb_uncompress)
  85         strbuf_close(sb_uncompress);
  86     if (fp)
  87         fclose(fp);
  88 }
  89 /**
  90  * output_with_formatting: pass records to the convert filter.
  91  *
  92  *  @param[in]  cv  convert descripter
  93  *  @param[in]  gtp record descripter
  94  *  @param[in]  root    project root directory
  95  *  @param[in]  flags   format flags
  96  *  @return     outputted number of records
  97  */
  98 int
  99 output_with_formatting(CONVERT *cv, GTP *gtp, const char *root, int flags)
 100 {
 101     int count = 0;
 102 
 103     if (format == FORMAT_PATH) {
 104         convert_put_path(cv, NULL, gtp->path);
 105         count++;
 106     } else if (flags & GTAGS_COMPACT) {
 107         count += put_compact_format(cv, gtp, root, flags);
 108     } else {
 109         put_standard_format(cv, gtp, flags);
 110         count++;
 111     }
 112     return count;
 113 }
 114 /*
 115  * Compact format:
 116  */
 117 static int
 118 put_compact_format(CONVERT *cv, GTP *gtp, const char *root, int flags)
 119 {
 120     STATIC_STRBUF(ib);
 121     int count = 0;
 122     char *p = (char *)gtp->tagline;
 123     const char *fid, *tagname;
 124     int n = 0;
 125 
 126     strbuf_clear(ib);
 127     /*                    a          b
 128      * tagline = <file id> <tag name> <line no>,...
 129      */
 130     fid = p;
 131     while (*p != ' ')
 132         p++;
 133     *p++ = '\0';            /* a */
 134     tagname = p;
 135     while (*p != ' ')
 136         p++;
 137     *p++ = '\0';            /* b */
 138     /*
 139      * Reopen or rewind source file.
 140      */
 141     if (!nosource) {
 142         if (strcmp(gtp->path, curpath) != 0) {
 143             if (curpath[0] != '\0' && fp != NULL)
 144                 fclose(fp);
 145             strlimcpy(curtag, tagname, sizeof(curtag));
 146             strlimcpy(curpath, gtp->path, sizeof(curpath));
 147             /*
 148              * Use absolute path name to support GTAGSROOT
 149              * environment variable.
 150              */
 151             fp = fopen(makepath(root, curpath, NULL), "r");
 152             if (fp == NULL)
 153                 warning("source file '%s' is not available.", curpath);
 154             last_lineno = cur_lineno = 0;
 155         } else if (strcmp(gtp->tag, curtag) != 0) {
 156             strlimcpy(curtag, gtp->tag, sizeof(curtag));
 157             if (atoi(p) < last_lineno && fp != NULL) {
 158                 rewind(fp);
 159                 cur_lineno = 0;
 160             }
 161             last_lineno = 0;
 162         }
 163     }
 164     /*
 165      * Unfold compact format.
 166      */
 167     if (!isdigit(*p))
 168         die("invalid compact format.");
 169     if (flags & GTAGS_COMPNAME)
 170         tagname = (char *)uncompress(tagname, gtp->tag, sb_uncompress);
 171     if (flags & GTAGS_COMPLINE) {
 172         /*
 173          * If GTAGS_COMPLINE flag is set, each line number is expressed as
 174          * the difference from the previous line number except for the head.
 175          * Please see flush_pool() in libutil/gtagsop.c for the details.
 176          */
 177         int last = 0, cont = 0;
 178 
 179         while (*p || cont > 0) {
 180             if (cont > 0) {
 181                 n = last + 1;
 182                 if (n > cont) {
 183                     cont = 0;
 184                     continue;
 185                 }
 186             } else if (isdigit(*p)) {
 187                 GET_NEXT_NUMBER(p);
 188             }  else if (*p == '-') {
 189                 GET_NEXT_NUMBER(p);
 190                 cont = n + last;
 191                 n = last + 1;
 192             } else if (*p == ',') {
 193                 GET_NEXT_NUMBER(p);
 194                 n += last;
 195             }
 196             if (last_lineno != n && fp) {
 197                 while (cur_lineno < n) {
 198                     if (!(src = strbuf_fgets(ib, fp, STRBUF_NOCRLF))) {
 199                         src = "";
 200                         fclose(fp);
 201                         fp = NULL;
 202                         break;
 203                     }
 204                     cur_lineno++;
 205                 }
 206             }
 207             convert_put_using(cv, tagname, gtp->path, n, src, fid);
 208             count++;
 209             last_lineno = last = n;
 210         }
 211     } else {
 212         /*
 213          * In fact, when GTAGS_COMPACT is set, GTAGS_COMPLINE is allways set.
 214          * Therefore, the following code are not actually used.
 215          * However, it is left for some test.
 216          */
 217         while (*p) {
 218             for (n = 0; isdigit(*p); p++)
 219                 n = n * 10 + *p - '0';
 220             if (*p == ',')
 221                 p++;
 222             if (last_lineno == n)
 223                 continue;
 224             if (last_lineno != n && fp) {
 225                 while (cur_lineno < n) {
 226                     if (!(src = strbuf_fgets(ib, fp, STRBUF_NOCRLF))) {
 227                         src = "";
 228                         fclose(fp);
 229                         fp = NULL;
 230                         break;
 231                     }
 232                     cur_lineno++;
 233                 }
 234             }
 235             convert_put_using(cv, tagname, gtp->path, n, src, fid);
 236             count++;
 237             last_lineno = n;
 238         }
 239     }
 240     return count;
 241 }
 242 /*
 243  * Standard format:
 244  */
 245 static void
 246 put_standard_format(CONVERT *cv, GTP *gtp, int flags)
 247 {
 248     char *p = (char *)gtp->tagline;
 249     char namebuf[IDENTLEN];
 250     const char *fid, *tagname, *image;
 251 
 252     /*                    a          b         c
 253      * tagline = <file id> <tag name> <line no> <line image>
 254      */
 255     fid = p;
 256     while (*p != ' ')
 257         p++;
 258     *p++ = '\0';            /* a */
 259     tagname = p;
 260     while (*p != ' ')
 261         p++;
 262     *p++ = '\0';            /* b */
 263     if (flags & GTAGS_COMPNAME) {
 264         strlimcpy(namebuf, (char *)uncompress(tagname, gtp->tag, sb_uncompress), sizeof(namebuf));
 265         tagname = namebuf;
 266     }
 267     if (nosource) {
 268         image = " ";
 269     } else {
 270         while (*p != ' ')
 271             p++;
 272         image = p + 1;      /* c + 1 */
 273         if (flags & GTAGS_COMPRESS)
 274             image = (char *)uncompress(image, gtp->tag, sb_uncompress);
 275     }
 276     convert_put_using(cv, tagname, gtp->path, gtp->lineno, image, fid);
 277 }
