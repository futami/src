  53  *
  54  * usage: a piece of code to achieve the following command line.
  55  *
  56  * [command line]
  57  *
  58  *  find . -type f -print | xargs grep pattern
  59  *
  60  * [code]
  61  *
  62  * char *p;
  63  * FILE *ip = popen("find . -type f -print", "r");
  64  * XARGS *xp = xargs_open_with_file("grep pattern", ip)
  65  * xp->ignore_error = 1;
  66  * while ((p = xargs_read(xp)) != NULL)
  67  *  puts(p);
  68  * xargs_close(xp);
  69  */
  70 /**
  71  * exec_line_limit: upper limit of bytes of exec line.
  72  *
  73  *  @param[in]  length  command line length
  74  *  @return 0: unknown or cannot afford long line.
  75  *      > 0: upper limit of exec line
  76  */
  77 static int
  78 exec_line_limit(int length)
  79 {
  80     long limit = 0;
  81 
  82 #ifdef ARG_MAX
  83     /*
  84      * POSIX.2 limits the exec(2) line length to ARG_MAX - 2048.
  85      */
  86     limit = ARG_MAX - 2048;
  87     /*
  88      * The reason is unknown but the xargs(1) in GNU findutils
  89      * use this limit.
  90      */
  91     if (limit > 20 * 1024)
  92         limit = 20 * 1024;
  93     /*
  94      * Add the command line length.
  95      * We estimates additional 80 bytes for popen(3).
  96      *
  97      * for "/bin/sh -c "                11bytes
  98      * reserve                  69bytes
  99      * ----------------------------------------------------
 100      * Total                    80 bytes
 101      */
 102     limit -= length + 80;
 103 
 104     limit -= env_size();
 105 #endif
 106 #if !defined(ARG_MAX) && defined(_WIN32)
 107     /*
 108      * The limit lenght of the command line of cmd.exe is 2047
 109      * characters on Windows 2000 or 8191 characters on Windows XP
 110      * and later. The 80 below is for safety.
 111      */
 112     limit = 2047 - length - 80;
 113 #endif
 114     if (limit < 0)
 115                die("Negative exec line limit = %ld", limit);
 116 
 117     return limit;
 118 }

 141 #define LT_MAX ((xp->max_args == 0 || count < xp->max_args))

 143 /**
 144  * Common processing for each XARGS_XXXX type.
 145  */
 146 #define APPEND_ARGUMENT(p) {\
 147     char *path = (p);\
 148     length = strlen(path);\
 149     if (*path == ' ') {\
 150         if (xp->put_gpath && !test("b", ++path))\
 151             gpath_put(path, GPATH_OTHER);\
 152         continue;\
 153     }\
 154     if (strbuf_getlen(comline) + length + 2 > limit)\
 155         break;\
 156     xp->seqno++;\
 157     if (xp->put_gpath)\
 158         gpath_put(path, GPATH_SOURCE);\
 159     if (xp->skip_assembly && locatestring(path, ".s", MATCH_AT_LAST|IGNORE_CASE) != NULL) {\
 160         if (xp->verbose)\
 161             xp->verbose(path + 2, xp->seqno, 1);\
 162     } else {\
 163         if (xp->verbose)\
 164             xp->verbose(path + 2, xp->seqno, 0);\
 165         strbuf_putc(comline, ' ');\
 166         strbuf_puts(comline, quote_shell(path));\
 167         count++;\
 168     }\
 169 }

 171 /**
 172  * execute_command
 173  *
 174  *  @param[in]  xp  xargs structure
 175  *  @return     !=NULL: file pointer
 176  *          ==NULL: end of argument
 177  *
 178  * This function constructs command line from the following,
 179  *  command: xp->command
 180  *  argument: each argument provider
 181  * execute it on a pipe line, and return the file pointer.
 182  */
 183 static FILE *
 184 execute_command(XARGS *xp)
 185 {
 186     int limit;
 187     STRBUF *comline = strbuf_open(0);
 188     int count = 0;
 189     int length;
 190     FILE *pipe = NULL;
 191     char *p, *meta_p;
 192 
 193 #if defined(_WIN32) && !defined(__CYGWIN__)
 194     /*
 195      * If the command starts with a quote, CMD.EXE requires the entire
 196      * command line to be quoted.
 197      */
 198     if (*xp->command == '"')
 199         strbuf_putc(comline, '"');
 200 #endif
 201     /*
 202      * Copy the part before '%s' of the command skeleton.
 203      * The '%s' in the skeleton is replaced with given arguments.
 204      */
 205     meta_p = locatestring(xp->command, "%s", MATCH_FIRST);
 206     if (meta_p) {
 207         strbuf_nputs(comline, xp->command, meta_p - xp->command);
 208         limit = exec_line_limit(strlen(meta_p + 2));
 209     } else {
 210         strbuf_puts(comline, xp->command);
 211         limit = exec_line_limit(0);
 212     }
 213     /*
 214      * Append arguments as many as possible.
 215      */
 216     switch (xp->type) {
 217     case XARGS_FILE:
 218         for (
 219             /* initial */
 220             fseek(xp->ip, xp->fptr, SEEK_SET)
 221             ;
 222             /* continuation condition */
 223             (LT_MAX &&
 224                 ((p = (strbuf_getlen(xp->path) > 0 ?
 225                 strbuf_value(xp->path) :
 226                 strbuf_fgets(xp->path, xp->ip, STRBUF_NOCRLF))) != NULL))
 227             ;
 228             /* preparation */
 229             strbuf_reset(xp->path)
 230         )
 231             APPEND_ARGUMENT(p);
 232         xp->fptr = ftell(xp->ip);
 233         break;
 234     case XARGS_ARGV:
 235         for (; LT_MAX && xp->argc > 0; xp->argc--, xp->argv++)
 236             APPEND_ARGUMENT(xp->argv[0])
 237         break;
 238     case XARGS_STRBUF:
 239         for (; LT_MAX && xp->curp < xp->endp; xp->curp += length + 1)
 240             APPEND_ARGUMENT(xp->curp)
 241         break;
 242     case XARGS_FIND:
 243         for (; LT_MAX && (p = repeat_find_read()) != NULL; repeat_find_next())
 244             APPEND_ARGUMENT(p)
 245         break;
 246     }
 247     /*
 248      * Copy the left part of the command skeleton.
 249      */
 250     if (meta_p) {
 251         strbuf_putc(comline, ' ');
 252         strbuf_puts(comline, meta_p + 2);
 253     }
 254 #if defined(_WIN32) && !defined(__CYGWIN__)
 255     if (*xp->command == '"')
 256         strbuf_putc(comline, '"');
 257 #endif
 258     if (count > 0) {
 259         pipe = popen(strbuf_value(comline), "r");
 260         if (pipe == NULL)
 261             die("cannot execute command '%s'.", strbuf_value(comline));
 262     }
 263     strbuf_close(comline);
 264     return pipe;
 265 }
 266 /**
 267  * xargs_open_generic: allocate generic part of xargs structure.
 268  *
 269  *  @param[in]  command command line except for the arguments.
 270  *  @param[in]  max_args 0: no limit, >0: max argument
 271  *  @return     xargs structure 
 272  */
 273 static XARGS *
 274 xargs_open_generic(const char *command, int max_args)
 275 {
 276     XARGS *xp;
 277 
 278     xp  = check_calloc(sizeof(XARGS), 1);
 279     xp->command = check_strdup(command);
 280     xp->type = 0;
 281     xp->pipe = NULL;
 282     xp->result = strbuf_open(0);
 283     xp->end_of_arg = 0;
 284     xp->unread = 0;
 285     xp->max_args = max_args;
 286     xp->seqno = 0;
 287     xp->skip_assembly = 0;
 288     /*
 289      * Procedure to display verbose message.
 290      * proc(path, seqno, skip)
 291      */
 292     xp->verbose = NULL;
 293     /*
 294      * By default, the error status returned by pclose() is not ignored,
 295      * because global(1) doesn't return error code when the target
 296      * was not found.
 297      * Some commands like grep(1) return error code when the target
 298      * was not found. If you would like to use such command, set the
 299      * flag to 1.
 300      */
 301     xp->ignore_error = 0;
 302     /*
 303      * By default, we doesn't put the path to GPATH.
 304      * This option is prepared for createtags() and updatetags().
 305      */
 306     xp->put_gpath = 0;
 307     /*
 308      * By default, we don't cut off the blanks at the tail of the line.
 309      */
 310     xp->trim_line = 0;
 311 
 312     return xp;
 313 }
 314 /*
 315  * xargs_open_with_file: open xargs stream using file
 316  *
 317  *  @param[in]  command command skeleton.
 318  *  @param[in]  max_args 0: no limit, >0: max argument
 319  *  @param[in]  ip  file pointer
 320  *  @return     xargs structure
 321  *
 322  * The '%s' in the command skeleton is replaced with given arguments.
 323  * If '%s' doesn't exist, the arguments is appended to the tail of the
 324  * skeleton.
 325  */
 326 XARGS *
 327 xargs_open_with_file(const char *command, int max_args, FILE *ip)
 328 {
 329     XARGS *xp = xargs_open_generic(command, max_args);
 330 
 331     xp->type = XARGS_FILE;
 332     xp->ip = ip;
 333     xp->path = strbuf_open(0);
 334     xp->fptr = 0;
 335     return xp;
 336 }

 401 /**
 402  * xargs_read: read a record from xargs stream
 403  *
 404  *  @param[in]  xp  xargs structure
 405  *  @return     result line
 406  */
 407 char *
 408 xargs_read(XARGS *xp)
 409 {
 410     assert(xp != NULL);
 411     if (xp->end_of_arg)
 412         return NULL;
 413     if (xp->unread) {
 414         xp->unread = 0;
 415         return strbuf_value(xp->result);
 416     }
 417     if (xp->pipe && strbuf_fgets(xp->result, xp->pipe, STRBUF_NOCRLF) != NULL) {
 418         if (xp->trim_line)
 419             strbuf_trim(xp->result);
 420         return strbuf_value(xp->result);
 421     }
 422     if (xp->pipe)
 423         if (pclose(xp->pipe) != 0 && !xp->ignore_error)
 424             die("command failed in xargs_read().");
 425     /*
 426      * Switch to the next segment.
 427      */
 428     do {
 429         xp->pipe = execute_command(xp);
 430         if (xp->pipe && strbuf_fgets(xp->result, xp->pipe, STRBUF_NOCRLF) != NULL) {
 431             if (xp->trim_line)
 432                 strbuf_trim(xp->result);
 433             return strbuf_value(xp->result);
 434         }
 435         if (xp->pipe) {
 436             if (pclose(xp->pipe) != 0 && !xp->ignore_error)
 437                 die("command failed in xargs_read().");
 438         } else {
 439             xp->end_of_arg = 1;
 440         }
 441     } while (!xp->end_of_arg);
 442 
 443     return NULL;
 444 }

 456 /**
 457  * xargs_close(xp)
 458  *
 459  *  @param[in]  xp  xargs structure
 460  */
 461 int
 462 xargs_close(XARGS *xp)
 463 {
 464     int count;
 465 
 466     assert(xp != NULL);
 467     count = xp->seqno;
 468     assert(xp->pipe == NULL);
 469     free(xp->command);
 470     strbuf_close(xp->result);
 471 
 472     switch (xp->type) {
 473     case XARGS_FILE:
 474         strbuf_close(xp->path);
 475         break;
 476     case XARGS_ARGV:
 477     case XARGS_STRBUF:
 478         /* Nothing to do */
 479         break;
 480     }
 481     free(xp);
 482     return count;
 483 }
