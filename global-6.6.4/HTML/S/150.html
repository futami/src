  73 /**
  74  * load contents of FILEMAP into the memory.
  75  *
  76  *  @param[in]  file    path of FILEMAP
  77  *  @param[out] area    file contents (malloced)
  78  *  @param[out] size    size of area
  79  *  @return 0 succesful
  80  *      -1: out of memory.
  81  *      -2: FILEMAP not found.
  82  *      -3: cannot fstat FILEMAP.
  83  *      -4: cannot read FILEMAP.
  84  */
  85 static int
  86 load_filemap_contents(const char *file, char **area, int *size)
  87 {
  88     struct stat st;
  89     char *p = NULL;
  90     int status = 0;
  91     int fd = -1;
  92 
  93     /* Load FILEMAP contents */
  94     status = -2;
  95     if ((fd = open(file, OPENMODE)) < 0)
  96         goto err;
  97     status = -3;
  98     if (fstat(fd, &st) < 0)
  99         goto err;
 100     status = -1;
 101     if ((p = (char *)malloc(st.st_size)) == NULL)
 102         goto err;
 103     status = -4;
 104     if (read(fd, p, st.st_size) != st.st_size)
 105         goto err;
 106     close(fd);
 107     *area = p;
 108     *size = st.st_size;
 109     return 0;
 110 err:
 111     if (fd != -1)
 112         close(fd);
 113     if (p != NULL)
 114         free(p);
 115     return status;
 116 }

 122 cmp(const void *s1, const void *s2)

 124     return strcmp(((struct map *)s1)->name, ((struct map *)s2)->name);

 126 /**
 127  * creates index for in core FILEMAP.
 128  *
 129  *  @param[in]  area    filemap contents
 130  *  @param[in]  size    size of area
 131  *  @param[out] map map structure
 132  *  @param[out] lines   map lines
 133  *  @return  0: succesful
 134  *      -1: out of memory.
 135  *      -5: invalid format.
 136  */
 137 static int
 138 create_filemap_index(char *area, int size, struct map **map, int *lines)
 139 {
 140     char *p, *endp = area + size;
 141     struct map *m;
 142     int n = 0;
 143     int status;
 144     int i;
 145 
 146     /* Count line number */
 147     for (p = area; p < endp; p++)
 148         if (*p == '\n')
 149             n++; 
 150     status = -1;
 151     if ((m = (struct map *)malloc(sizeof(struct map) * n)) == NULL)
 152         return -1;
 153     /*
 154      * FILEMAP format:
 155      * <NAME>\t<PATH>\n
 156      */
 157     p = area;
 158     for (i = 0; i < n; i++) {
 159         m[i].name = p;
 160         while (*p && *p != '\t')
 161             p++;
 162         if (*p == '\0')
 163             goto ferr;
 164         *p++ = '\0';
 165         m[i].path = p;
 166         while (*p && *p != '\r' && *p != '\n')
 167             p++;
 168         if (*p == '\0')
 169             goto ferr;
 170         if (*p == '\r')
 171             *p++ = '\0';
 172         if (*p == '\n')
 173             *p++ = '\0';
 174     }
 175     qsort(m, n, sizeof(struct map), cmp);
 176     *map = m;
 177     *lines = n;
 178     return 0;
 179 ferr:
 180     free(m);
 181     return -5;
 182 }

 187 static void
 188 unload_filemap(void)
 189 {
 190     (void)free(global_map);
 191     global_map = NULL;
 192     (void)free(global_contents);
 193     global_contents = NULL;
 194 }

 198 /**
 199  * loads FILEMAP.
 200  *
 201  *  @param[in]  filemap FILEMAP file
 202  *  @return      0: succesful
 203  *          -1: out of memory.
 204  *          -2: FILEMAP not found.
 205  *          -3: cannot fstat FILEMAP.
 206  *          -4: cannot read FILEMAP.
 207  *          -5: format error.
 208  *  Globals used (output):
 209  *      global_contents filemap contents (for free)
 210  *      global_map  filemap index.
 211  *      global_maplines lines of filemap index.
 212  */
 213 int
 214 htags_load_filemap(const char *filemap)
 215 {
 216     int status = 0;
 217     char *area;
 218     int size, lines;
 219     struct map *map;
 220 
 221     status = load_filemap_contents(filemap, &area, &size);
 222     if (status < 0)
 223         return status;
 224     status = create_filemap_index(area, size, &map, &lines);
 225     if (status < 0) {
 226         (void)free(area);
 227         return status;
 228     }
 229     global_contents = area;
 230     global_map = map;
 231     global_maplines = lines;
 232     return 0;
 233 }

 243 /**
 244  * convert file path name into the URL in the hypertext.
 245  *
 246  *  @param[in]  path    path name in the filesystem.
 247  *  @param[in]  line    0: ignored, !=0: line number in path.
 248  *  @param[out] url result url
 249  *  @param[in]  size    size of url
 250  *  @return     0: succesful
 251  *          1: path not found
 252  *          -1: filemap not loaded yet
 253  *  Globals used (input):
 254  *      global_contents filemap contents (for free)
 255  *      global_map  filemap index.
 256  *      global_maplines lines of filemap index.
 257  *
 258  * URL: S/[file id].html#[line number]
 259  */
 260 int
 261 htags_path2url(const char *path, int line, char *url, int size)
 262 {
 263     struct map tmp;
 264     struct map *result;
 265 
 266     if (global_contents == NULL || global_map == NULL)
 267         return -1;
 268     tmp.name = path;
 269     result = (struct map *)bsearch(&tmp, global_map, global_maplines, sizeof(struct map), cmp);
 270     if (result == NULL)
 271         return 1;
 272     if (line > 0)
 273         snprintf(url, size, "%s#L%d", result->path, line);
 274     else
 275         snprintf(url, size, "%s", result->path);
 276     return 0;
 277 }
