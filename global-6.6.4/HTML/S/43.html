  80 /**
  81  * literal_comple: compile literal for search.
  82  *
  83  *  @param[in]  pat literal string
  84  *
  85  * Literal string is treated as is.
  86  */
  87 void
  88 literal_comple(const char *pat)
  89 {
  90     /*
  91      * convert spaces into %FF format.
  92      */
  93     encode(encoded_pattern, sizeof(encoded_pattern), pat);
  94     strlimcpy(pattern, pat, sizeof(pattern));
  95     /*
  96      * construct a goto table.
  97      */
  98     cgotofn(pattern);
  99     /*
 100      * construct fail links.
 101      */
 102     cfail();
 103 }
 104 /**
 105  * literal_search: execute literal search
 106  *
 107  *  @param[in]  cv  CONVERT structure
 108  *  @param[in]  file    file to search
 109  *  @return     0: normal, -1: error
 110  */
 111 int
 112 literal_search(CONVERT *cv, const char *file)
 113 {
 114 # define ccomp(a,b) (iflag ? lca(a)==lca(b) : a==b)
 115 # define lca(x) (isupper(x) ? tolower(x) : x)
 116     struct words *c;
 117     int ccount;
 118     char *p;
 119     char *buf;
 120     struct stat stb;
 121     char *linep;
 122     long lineno;
 123     int f;
 124     int count = 0;
 125 
 126     if ((f = open(file, O_BINARY)) < 0) {
 127         warning("cannot open '%s'.", file);
 128         return -1;
 129     }
 130     if (fstat(f, &stb) < 0) {
 131         warning("cannot fstat '%s'.", file);
 132         goto skip_empty_file;
 133     }
 134     if (stb.st_size == 0)
 135         goto skip_empty_file;
 136 #ifdef HAVE_MMAP
 137     buf = mmap(0, stb.st_size, PROT_READ, MAP_SHARED, f, 0);
 138     if (buf == MAP_FAILED)
 139         die("mmap failed (%s).", file);
 140 #elif _WIN32
 141     {
 142     HANDLE hMap = CreateFileMapping((HANDLE)_get_osfhandle(f), NULL, PAGE_READONLY, 0, stb.st_size, NULL);
 143     if (hMap == NULL)
 144         die("CreateFileMapping failed (%s).", file);
 145     buf = MapViewOfFile(hMap, FILE_MAP_READ, 0, 0, 0);
 146     if (buf == NULL)
 147         die("MapViewOfFile failed (%s).", file);
 148 #else
 149 #ifdef HAVE_ALLOCA
 150     buf = (char *)alloca(stb.st_size);
 151 #else
 152     buf = (char *)malloc(stb.st_size);
 153 #endif
 154     if (buf == NULL)
 155         die("short of memory.");
 156     if (read(f, buf, stb.st_size) < stb.st_size)
 157         die("read failed (%s).", file);
 158 #endif
 159     linep = p = buf;
 160     ccount = stb.st_size;
 161     lineno = 1;
 162     c = w;
 163     for (;;) {
 164         if (--ccount < 0)
 165             break;
 166         nstate:
 167             if (ccomp(c->inp, *p)) {
 168                 c = c->nst;
 169             }
 170             else if (c->link != 0) {
 171                 c = c->link;
 172                 goto nstate;
 173             }
 174             else {
 175                 c = c->fail;
 176                 if (c==0) {
 177                     c = w;
 178                     istate:
 179                     if (ccomp(c->inp , *p)) {
 180                         c = c->nst;
 181                     }
 182                     else if (c->link != 0) {
 183                         c = c->link;
 184                         goto istate;
 185                     }
 186                 }
 187                 else goto nstate;
 188             }
 189         if (c->out) {
 190             while (*p++ != '\n') {
 191                 if (--ccount < 0)
 192                     break;
 193             }
 194             if (Vflag)
 195                 goto nomatch;
 196     succeed:    if (cv->format == FORMAT_PATH) {
 197                 convert_put_path(cv, NULL, file);
 198                 count++;
 199                 goto finish;
 200             } else {
 201                 STATIC_STRBUF(sb);
 202 
 203                 strbuf_clear(sb);
 204                 strbuf_nputs(sb, linep, p - linep);
 205                 strbuf_unputc(sb, '\n');
 206                 strbuf_unputc(sb, '\r');
 207                 convert_put_using(cv, pattern, file, lineno, strbuf_value(sb), NULL);
 208                 count++;
 209             }
 210     nomatch:    lineno++;
 211             linep = p;
 212             c = w;
 213             continue;
 214         }
 215         if (*p++ == '\n' || ccount == 0) {
 216             if (Vflag)
 217                 goto succeed;
 218             else {
 219                 lineno++;
 220                 linep = p;
 221                 c = w;
 222             }
 223         }
 224     }
 225 finish:
 226 #ifdef HAVE_MMAP
 227     munmap(buf, stb.st_size);
 228 #elif _WIN32
 229     UnmapViewOfFile(buf);
 230     CloseHandle(hMap);
 231     }
 232 #elif HAVE_ALLOCA
 233 #else
 234     free(buf);
 235 #endif
 236 skip_empty_file:
 237     close(f);
 238     return count;
 239 }

 243 void
 244 cgotofn(const char *pattern) {
 245     int c;
 246     struct words *s;
 247 
 248     s = smax = w;
 249 nword:  for(;;) {
 250         c = *pattern++;
 251         if (c==0)
 252             return;
 253         if (c == '\n') {
 254             s->out = 1;
 255             s = w;
 256         } else {
 257         loop:   if (s->inp == c) {
 258                 s = s->nst;
 259                 continue;
 260             }
 261             if (s->inp == 0) goto enter;
 262             if (s->link == 0) {
 263                 if (smax >= &w[MAXSIZ - 1])
 264                     overflo();
 265                 s->link = ++smax;
 266                 s = smax;
 267                 goto enter;
 268             }
 269             s = s->link;
 270             goto loop;
 271         }
 272     }
 273     enter:
 274     do {
 275         s->inp = c;
 276         if (smax >= &w[MAXSIZ - 1])
 277             overflo();
 278         s->nst = ++smax;
 279         s = smax;
 280     } while ((c = *pattern++) != '\n' && c!=0);
 281     smax->out = 1;
 282     s = w;
 283     if (c != 0)
 284         goto nword;
 285 }

 287 void
 288 cfail(void) {
 289     struct words *queue[QSIZE];
 290     struct words **front, **rear;
 291     struct words *state;
 292     int bstart;
 293     char c;
 294     struct words *s;
 295 
 296     s = w;
 297     front = rear = queue;
 298 init:   if ((s->inp) != 0) {
 299         *rear++ = s->nst;
 300         if (rear >= &queue[QSIZE - 1])
 301             overflo();
 302     }
 303     if ((s = s->link) != 0) {
 304         goto init;
 305     }
 306 
 307     while (rear != front) {
 308         s = *front;
 309         if (front == &queue[QSIZE-1])
 310             front = queue;
 311         else front++;
 312     cloop:  if ((c = s->inp) != 0) {
 313             bstart = 0;
 314             *rear = (q = s->nst);
 315             if (front < rear)
 316                 if (rear >= &queue[QSIZE-1])
 317                     if (front == queue) overflo();
 318                     else rear = queue;
 319                 else rear++;
 320             else
 321                 if (++rear == front) overflo();
 322             state = s->fail;
 323         floop:  if (state == 0) {
 324                 state = w;
 325                 bstart = 1;
 326             }
 327             if (state->inp == c) {
 328             qloop:  q->fail = state->nst;
 329                 if ((state->nst)->out == 1)
 330                     q->out = 1;
 331                 if ((q = q->link) != 0)
 332                     goto qloop;
 333             }
 334             else if ((state = state->link) != 0)
 335                 goto floop;
 336             else if (bstart == 0){
 337                 state = 0;
 338                 goto floop;
 339             }
 340         }
 341         if ((s = s->link) != 0)
 342             goto cloop;
 343     }
 344 }
