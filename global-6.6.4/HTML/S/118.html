  86 /**
  87  * extract_lastname: extract the last name of include line.
  88  *
  89  *  @param[in]  image   source image of include
  90  *  @param[in]  is_php  1: is PHP source
  91  *  @return     last name or NULL on error.
  92  */
  93 static const char *
  94 extract_lastname(const char *image, int is_php)
  95 {
  96     static char buf[MAXBUFLEN];
  97     const char *p;
  98     char *q;
  99     int sep;
 100 
 101     /*
 102      * C:   #include <xxx/yyy/zzz.h>
 103      *  #include "xxx/yyy/zzz.h"
 104      * PHP: include('xxx/yyy/zzz');
 105      */
 106     p = image;
 107     while (*p && isspace((unsigned char)*p))        /* skip space */
 108         p++;
 109     if (!*p)
 110         return NULL;
 111     if (*p == '#') {
 112         if (is_php)
 113             return NULL;
 114         p++;
 115         while (*p && isspace((unsigned char)*p))    /* skip space */
 116             p++;
 117         if (!*p)
 118             return NULL;
 119     }
 120     /*
 121      * If match to one of the include keywords then points
 122      * the following character of the keyword.
 123      *            p
 124      *            v
 125      * ... include ....
 126      */
 127     if (is_php) {
 128         if ((p = locatestring(p, "include", MATCH_AT_FIRST)) == NULL)
 129             return NULL;
 130     } else {
 131         char *q;
 132 
 133         if (((q = locatestring(p, "include_next", MATCH_AT_FIRST)) == NULL) &&
 134             ((q = locatestring(p, "import", MATCH_AT_FIRST)) == NULL) &&
 135             ((q = locatestring(p, "include", MATCH_AT_FIRST)) == NULL))
 136             return NULL;
 137         p = q;
 138     }
 139     while (*p && isspace((unsigned char)*p))        /* skip space */
 140         p++;
 141     if (is_php && *p == '(') {
 142         p++;
 143         while (*p && isspace((unsigned char)*p))    /* skip space */
 144             p++;
 145     }
 146     sep = *p;
 147     if (is_php) {
 148         if (sep != '\'' && sep != '"')
 149             return NULL;
 150     } else {
 151         if (sep != '<' && sep != '"')
 152             return NULL;
 153     }
 154     if (sep == '<')
 155         sep = '>';
 156     p++;
 157     if (!*p)
 158         return NULL;
 159     q = buf; 
 160     while (*p && *p != '\n' && *p != sep)
 161         *q++ = *p++;
 162     *q = '\0';
 163     if (*p == sep) {
 164         q = locatestring(buf, "/", MATCH_LAST);
 165         if (q)
 166             q++;
 167         else
 168             q = buf;
 169         return q;
 170     }
 171     return NULL;
 172 }
 173 /**
 174  * get the last part of the path.
 175  *
 176  *  @param[in]  path    path name
 177  *  @return     last part
 178  *
 179  * Examples:
 180  * lastpart("a/b")  => "b"
 181  * lastpart("a")    => "a"
 182  */
 183 static const char *
 184 lastpart(const char *path)
 185 {
 186     const char *p = strrchr(path, '/');
 187 
 188     return p ? p + 1 : path;
 189 }
 190 /**
 191  * get the directory part of the path.
 192  *
 193  *  @param[in]  path    path name
 194  *  @param[out] result  result buffer
 195  *  @return     directory part
 196  *
 197  * Examples:
 198  * dirpart("a/b/c") => "a/b"
 199  */
 200 static const char *
 201 dirpart(const char *path, char *result)
 202 {
 203     char *p = result;
 204     const char *q = path, *limit = strrchr(path, '/');
 205 
 206     while (q < limit)
 207         *p++ = *q++; 
 208     *p = '\0';
 209     return result;
 210 }
 211 /**
 212  * get the local path name if the path is under the dir.
 213  *
 214  *  @param[in]  path    path name
 215  *  @param[in]  dir directory name
 216  *  @return     local path name or NULL
 217  *
 218  * Examples:
 219  * localpath("a/b/c", "a/b")    => "c"
 220  * localpath("a/b/c/d", "a/b")  => "c/d"
 221  * localpath("a/b/c", "a/d")    => NULL
 222  */
 223 static const char *
 224 localpath(const char *path, char *dir)
 225 {
 226     int length = strlen(dir);
 227 
 228     if (!strncmp(path, dir, length) && *(path + length) == '/')
 229         return path + length + 1;
 230     return NULL;
 231 }

 256  *  @param[in]  path    path name

 260  * removedotslash("./a") => "a"

 262 static const char *

 265     return (*path == '.' && *(path + 1) == '/') ? path + 2 : path;

 267 /**
 268  * insert_comma: insert comma to the number.
 269  *
 270  *  @param[in]  n   number
 271  *  @return     edited string
 272  *
 273  * Examples:
 274  * 10000 => 10,000
 275  */
 276 static const char *
 277 insert_comma(unsigned int n)
 278 {
 279 #define RESULTSIZE 15
 280 #define INTERVAL 3
 281     static char result[RESULTSIZE];
 282     int i = RESULTSIZE;
 283     int count = 0;
 284 
 285     if (n == 0)
 286         return (const char *)"0";
 287     if (n > 1000000000)
 288         goto giveup;
 289     result[--i] = '\0';
 290     for (; n > 0; n = n / 10) {
 291         /*
 292          * Buffer overflow. This is not important even if occurring.
 293          */
 294         if (i <= 0)
 295             goto giveup;
 296         if (count && count % INTERVAL == 0)
 297             result[--i] = ',';
 298         result[--i] = n % 10 + '0';
 299         count++;
 300     }
 301     return (const char *)&result[i];
 302 giveup:
 303     snprintf(result, sizeof(result), "*****");
 304     return (const char *)result;
 305 }

 324  * print contents of one directory and the descendant. (recursively called)

 331 /**
 332  * File list of the top level directory (when level == 0) is not written

 334  * it appears in some places.
 335  */
 336 #define PUT(s) do {                     \
 337         if (level == 0) {               \
 338             if (tree_view == 0)         \
 339                 strbuf_puts(files, s);      \
 340         } else                      \
 341              fputs(s, op);              \
 342 } while (0)

 344 static int
 345 print_directory(int level, char *basedir)
 346 {
 347     const char *path;
 348     FILEOP *fileop = NULL;
 349     FILE *op = NULL;
 350     int flist_items = 0;
 351     int count = 0;
 352 
 353     if (level > 0) {
 354         char name[MAXPATHLEN];
 355 
 356         snprintf(name, sizeof(name), "%s/files/%s.%s", distpath, path2fid(basedir), HTML);
 357         fileop = open_output_file(name, 0);
 358         op = get_descripter(fileop);
 359         print_directory_header(op, level, basedir);
 360         if (tree_view) {
 361             char *target = (Fflag) ? "mains" : "_top";
 362 
 363             strbuf_puts(files, dir_begin);
 364             strbuf_puts(files, gen_href_begin_with_title_target("files", path2fid(basedir), HTML, NULL, NULL, target));
 365             strbuf_puts(files, full_path ? removedotslash(basedir) : lastpart(basedir));
 366             strbuf_puts(files, gen_href_end());
 367             strbuf_puts_nl(files, dir_title_end);
 368             strbuf_puts_nl(files, "<ul>");
 369         }
 370     }
 371     while ((path = getpath()) != NULL) {
 372         const char *p, *local = localpath(path, basedir);
 373 
 374         /*
 375          * Path is outside of basedir.
 376          */
 377         if (local == NULL) {
 378             ungetpath();    /* read again by upper level print_directory(). */
 379             break;
 380         }
 381         /*
 382          * Path is inside of basedir.
 383          */
 384         else {
 385             char *slash = strchr(local, '/');
 386 
 387             if (table_flist && flist_items++ % flist_fields == 0)
 388                 PUT(fline_begin);
 389             /*
 390              * Print directory.
 391              */
 392             if (slash) {
 393                 int baselen = strlen(basedir);
 394                 char *q, *last = basedir + baselen;
 395                 int subcount;
 396 
 397                 if (baselen + 1 + (slash - local)  > MAXPATHLEN) {
 398                     fprintf(stderr, "Too long path name.\n");
 399                     exit(1);
 400                 }
 401                 /*
 402                  * Append new directory to the basedir.
 403                  */
 404                 p = local;
 405                 q = last;
 406                 *q++ = '/';
 407                 while (p < slash)
 408                     *q++ = *p++;
 409                 *q = '\0';
 410                 /*
 411                  * print tree for this directory.
 412                  */
 413                 ungetpath();    /* read again by lower level print_directory(). */
 414                 subcount = print_directory(level + 1, basedir);
 415                 PUT(print_directory_name(level, basedir, subcount));
 416                 count += subcount;
 417                 /*
 418                  * Shrink the basedir.
 419                  */
 420                 *last = '\0';
 421             }
 422             /*
 423              * Print file.
 424              */
 425             else {
 426                 const char *file_name = print_file_name(level, path);
 427 
 428                 if (tree_view) {
 429                     int size = filesize(path);
 430                     char *target = (Fflag) ? "mains" : "_top";
 431                     char tips[80];
 432 
 433                     if (size > 1)
 434                         snprintf(tips, sizeof(tips), "%s bytes", insert_comma(size));
 435                     else
 436                         snprintf(tips, sizeof(tips), "%s byte", insert_comma(size));
 437                     strbuf_sprintf(files, "%s%s%s%s%s\n",
 438                         file_begin,
 439                         gen_href_begin_with_title_target(SRCS, path2fid(path), HTML, NULL, tips, target),
 440                         full_path ? removedotslash(path) : lastpart(path),
 441                         gen_href_end(),
 442                         file_end);
 443                 }
 444                 PUT(file_name);
 445                 if (filemap_file)
 446                     fprintf(FILEMAP, "%s\t%s/%s.%s\n", removedotslash(path), SRCS, path2fid(path), HTML);
 447                 count++;
 448             }
 449             if (table_flist && flist_items % flist_fields == 0)
 450                 PUT(fline_end);
 451         }
 452     }
 453     if (flist_items % flist_fields != 0)
 454         PUT(fline_end);
 455     if (level > 0) {
 456         print_directory_footer(op, level, basedir);
 457         close_file(fileop);
 458         if (tree_view) {
 459             strbuf_puts_nl(files, "</ul>");
 460             strbuf_puts_nl(files, dir_end);
 461         }
 462     }
 463     html_count++;
 464     return count;
 465 }
 466 /**
 467  * print directory header.
 468  *
 469  *  @param[in]  op  file index
 470  *  @param[in]  level   1,2...
 471  *  @param[in]  dir directory name
 472  */
 473 static void
 474 print_directory_header(FILE *op, int level, const char *dir)
 475 {
 476     STATIC_STRBUF(sb);
 477     const char *top = (Fflag && !tree_view) ? "../files" : "../mains";
 478 
 479     if (level == 0)
 480         die("print_directory_header: internal error.");
 481     strbuf_clear(sb);
 482     strbuf_puts(sb, removedotslash(dir));
 483     strbuf_putc(sb, '/');
 484     fputs_nl(gen_page_begin(strbuf_value(sb), SUBDIR), op);
 485     fputs_nl(body_begin, op);
 486 
 487     strbuf_clear(sb);
 488     strbuf_sprintf(sb, "%s%sroot%s/", header_begin, gen_href_begin(NULL, top, normal_suffix, NULL), gen_href_end());
 489     fputs(strbuf_value(sb), op);
 490     {
 491         char path[MAXPATHLEN];
 492         char *p, *q;
 493 
 494         strlimcpy(path, dir, sizeof(path));
 495         for (p = path + 1; p != NULL; p = strchr(p, '/')) {
 496             int save = 0;
 497 
 498             q = ++p;
 499             while (*q && *q != '/')
 500                 q++;
 501             save = *q;
 502             if (*q == '/')
 503                 *q = '\0';
 504             if (save == '/')
 505                 fputs(gen_href_begin(NULL, path2fid(path), HTML, NULL), op);
 506             fputs(p, op);
 507             if (save == '/')
 508                 fputs(gen_href_end(), op);
 509             *q = save;
 510             fputc('/', op);
 511         }
 512     }
 513     fputs_nl(header_end, op);
 514     {
 515         char parentdir[MAXPATHLEN];
 516         const char *suffix, *parent;
 517 
 518         (void)dirpart(dir, parentdir);
 519         if (level == 1) {
 520             parent = top;
 521             suffix = normal_suffix;
 522         } else {
 523             parent = path2fid(parentdir);
 524             suffix = HTML;
 525         }
 526         fputs(gen_href_begin_with_title(NULL, parent, suffix, NULL, "Parent Directory"), op);
 527     }
 528     if (Iflag)
 529         fputs(gen_image(PARENT, back_icon, ".."), op);
 530     else
 531         fputs("[..]", op);
 532     fputs_nl(gen_href_end(), op);
 533     if (table_flist)
 534         fputs_nl(flist_begin, op);
 535     else if (!no_order_list)
 536         fputs_nl(list_begin, op);
 537     else {
 538         fputs(br, op);
 539         fputs_nl(br, op);
 540     }
 541 }

 581 /**
 582  * print file name.
 583  *
 584  *  @param[in]  level   0,1,2...
 585  *  @param[in]  path    path of the file
 586  */
 587 static const char *
 588 print_file_name(int level, const char *path)
 589 {
 590     STATIC_STRBUF(sb);
 591     char *target = (Fflag) ? "mains" : "_top";
 592     int size = filesize(path);
 593     char tips[80];
 594 
 595     message(" [%d] adding %s", ++src_count, removedotslash(path));
 596     /*
 597      * We assume the file which has one of the following suffixes
 598      * as a candidate of include file.
 599      *
 600      * C: .h
 601      * C++: .hxx, .hpp, .H, .hh
 602      * PHP: .inc.php
 603      */
 604     if (regexec(&is_include_file, path, 0, 0, 0) == 0)
 605         put_inc(lastpart(path), path, src_count);
 606     strbuf_clear(sb);
 607     if (table_flist)
 608         strbuf_puts(sb, fitem_begin);
 609     else if (!no_order_list)
 610         strbuf_puts(sb, item_begin);
 611     if (size > 1)
 612         snprintf(tips, sizeof(tips), "%s bytes", insert_comma(size));
 613     else
 614         snprintf(tips, sizeof(tips), "%s byte", insert_comma(size));
 615     strbuf_puts(sb, gen_href_begin_with_title_target(level == 0 ? SRCS: upperdir(SRCS),
 616             path2fid(path), HTML, NULL, tips, target));
 617     if (Iflag) {
 618         const char *lang, *suffix, *text_icon;
 619 
 620         if ((suffix = locatestring(path, ".", MATCH_LAST)) != NULL
 621             && (lang = decide_lang(suffix)) != NULL
 622             && (strcmp(lang, "c") == 0 || strcmp(lang, "cpp") == 0
 623                || strcmp(lang, "yacc") == 0))
 624             text_icon = c_icon;
 625         else
 626             text_icon = file_icon;
 627         strbuf_puts(sb, gen_image(level == 0 ? CURRENT : PARENT, text_icon, removedotslash(path)));
 628         strbuf_puts(sb, quote_space);
 629     }
 630     strbuf_puts(sb, full_path ? removedotslash(path) : lastpart(path));
 631     strbuf_puts(sb, gen_href_end());
 632     if (table_flist)
 633         strbuf_puts(sb, fitem_end);
 634     else if (!no_order_list)
 635         strbuf_puts(sb, item_end);
 636     else
 637         strbuf_puts(sb, br);
 638     strbuf_putc(sb, '\n');
 639     return (const char *)strbuf_value(sb);
 640 }

 687 makefileindex(const char *file, STRBUF *a_files)
 688 {
 689     STATIC_STRBUF(sb);
 690     FILE *filesop;
 691     int flags = REG_EXTENDED;
 692     /*
 693      * Basedir is a directory to which we are paying attention on each
 694      * occasion. It starts with ".", grows and shrink according to the
 695      * progress of processing. It isn't copied each every recursive call
 696      * not to waste the stack.
 697      */
 698     char basedir[MAXPATHLEN];
 699 
 700     /*
 701      * Initialize data.
 702      */
 703     indexlink = (Fflag) ? "../files" : "../mains";
 704     src_count = 0;
 705 
 706     gp = gfind_open(dbpath, NULL, other_files ? GPATH_BOTH : GPATH_SOURCE, 0);
 707     /*
 708      * for collecting include files.
 709      */
 710     if (w32)
 711         flags |= REG_ICASE;
 712     strbuf_clear(sb);
 713     strbuf_puts(sb, "\\.(");
 714     {
 715         const char *p = include_file_suffixes;
 716         int c;
 717 
 718         while ((c = (unsigned char)*p++) != '\0') {
 719             if (isregexchar(c))
 720                 strbuf_putc(sb, '\\');
 721             else if (c == ',')
 722                 c = '|';
 723             strbuf_putc(sb, c);
 724         }
 725     }
 726     strbuf_puts(sb, ")$");
 727     if (regcomp(&is_include_file, strbuf_value(sb), flags) != 0)
 728         die("cannot compile regular expression '%s'.", strbuf_value(sb));
 729 
 730     /*
 731      * Write to files.html.
 732      */
 733     if ((filesop = fopen(makepath(distpath, file, NULL), "w")) == NULL)
 734         die("cannot open file '%s'.", file);
 735     fputs_nl(gen_page_index_begin(title_file_index, jscode), filesop);
 736     fputs_nl(body_begin, filesop);
 737     fputs(header_begin, filesop);
 738     fputs(gen_href_begin(NULL, "files", normal_suffix, NULL), filesop);
 739     fputs(title_file_index, filesop);
 740     fputs(gen_href_end(), filesop);
 741     fputs_nl(header_end, filesop);
 742     if (tree_view) {
 743         fputs_nl(tree_control, filesop);
 744         fputs_nl(tree_loading, filesop);
 745         if (tree_view_type) {
 746             fprintf(filesop, tree_begin_using, tree_view_type);
 747             fputc('\n', filesop);
 748         } else {
 749             fputs_nl(tree_begin, filesop);
 750         }
 751     } else if (table_flist)
 752         fputs_nl(flist_begin, filesop);
 753     else if (!no_order_list)
 754         fputs_nl(list_begin, filesop);
 755     FILEMAP = NULL;
 756     if (filemap_file) {
 757         if (!(FILEMAP = fopen(makepath(distpath, "FILEMAP", NULL), "w")))
 758                         die("cannot open '%s'.", makepath(distpath, "FILEMAP", NULL));
 759     }
 760     /*
 761      * print whole directory tree.
 762      */
 763     files = a_files;
 764     strcpy(basedir, ".");
 765 
 766     (void)print_directory(0, basedir);
 767     if (tree_view)
 768         strbuf_puts(files, tree_end);
 769 
 770     if (filemap_file)
 771         fclose(FILEMAP);
 772     gfind_close(gp);
 773     regfree(&is_include_file);
 774 
 775     fputs(strbuf_value(files), filesop);
 776     if (tree_view)
 777         fputs_nl(tree_end, filesop);
 778     else if (table_flist)
 779         fputs_nl(flist_end, filesop);
 780     else if (!no_order_list)
 781         fputs_nl(list_end, filesop);
 782     fputs_nl(body_end, filesop);
 783     fputs_nl(gen_page_end(), filesop);
 784     fclose(filesop);
 785     html_count++;
 786     return src_count;
 787 }
 788 /*----------------------------------------------------------------------*/
 789 /* Main body of generating include file index               */
 790 /*----------------------------------------------------------------------*/
 791 void
 792 makeincludeindex(void)
 793 {
 794     FILE *PIPE;
 795     STRBUF *input = strbuf_open(0);
 796     char *ctags_x;
 797     struct data *inc;
 798     char *target = (Fflag) ? "mains" : "_top";
 799     char command[MAXFILLEN];
 800 
 801     /*
 802      * Pick up include pattern.
 803      *
 804      * C: #include "xxx.h"
 805      * PHP: include("xxx.inc.php");
 806      */
 807     /*
 808      * Unlike Perl regular expression, POSIX regular expression doesn't support C-style escape sequence.
 809      * Therefore, we can not use "\\t" here.
 810      */
 811     snprintf(command, sizeof(command), PQUOTE "%s -gnx --encode-path=\" \t\" \"^[ \t]*(#[ \t]*(import|include)|include[ \t]*\\()\"" PQUOTE, quote_shell(global_path));
 812     if ((PIPE = popen(command, "r")) == NULL)
 813         die("cannot execute '%s'.", command);
 814     strbuf_reset(input);
 815     while ((ctags_x = strbuf_fgets(input, PIPE, STRBUF_NOCRLF)) != NULL) {
 816         SPLIT ptable;
 817         char buf[MAXBUFLEN];
 818         int is_php = 0;
 819         const char *last, *lang, *suffix;
 820 
 821         if (split(ctags_x, 4, &ptable) < 4) {
 822             recover(&ptable);
 823             die("too small number of parts in makefileindex().");
 824         }
 825         if ((suffix = locatestring(ptable.part[PART_PATH].start, ".", MATCH_LAST)) != NULL
 826             && (lang = decide_lang(suffix)) != NULL
 827             && strcmp(lang, "php") == 0)
 828             is_php = 1;
 829         last = extract_lastname(ptable.part[PART_LINE].start, is_php);
 830         if (last == NULL || (inc = get_inc(last)) == NULL)
 831             continue;
 832         recover(&ptable);
 833         /*
 834          * s/^[^ \t]+/$last/;
 835          */
 836         {
 837             const char *p;
 838             char *q = buf;
 839 
 840             for (p = last; *p; p++)
 841                 *q++ = *p;
 842             for (p = ctags_x; *p && *p != ' ' && *p != '\t'; p++)
 843                 ;
 844             for (; *p; p++)
 845                 *q++ = *p;
 846             *q = '\0';
 847         }
 848         put_included(inc, buf);
 849     }
 850     if (pclose(PIPE) != 0)
 851         die("terminated abnormally '%s' (errno = %d).", command, errno);
 852 
 853     for (inc = first_inc(); inc; inc = next_inc()) {
 854         const char *last = inc->name;
 855         int no = inc->id;
 856         FILEOP *fileop_INCLUDE;
 857         FILE *INCLUDE;
 858 
 859         if (inc->count > 1) {
 860             char path[MAXPATHLEN];
 861 
 862             snprintf(path, sizeof(path), "%s/%s/%d.%s", distpath, INCS, no, HTML);
 863             fileop_INCLUDE = open_output_file(path, 0);
 864             INCLUDE = get_descripter(fileop_INCLUDE);
 865             fputs_nl(gen_page_begin(last, SUBDIR), INCLUDE);
 866             fputs_nl(body_begin, INCLUDE);
 867             fputs_nl(verbatim_begin, INCLUDE);
 868             {
 869                 const char *filename = strbuf_value(inc->contents);
 870                 int count = inc->count;
 871 
 872                 for (; count; filename += strlen(filename) + 1, count--) {
 873                     fputs(gen_href_begin_with_title_target(upperdir(SRCS), path2fid(filename), HTML, NULL, NULL, target), INCLUDE);
 874                     fputs(removedotslash(filename), INCLUDE);
 875                     fputs_nl(gen_href_end(), INCLUDE);
 876                 }
 877             }
 878             fputs_nl(verbatim_end, INCLUDE);
 879             fputs_nl(body_end, INCLUDE);
 880             fputs_nl(gen_page_end(), INCLUDE);
 881             close_file(fileop_INCLUDE);
 882             html_count++;
 883             /*
 884              * inc->contents == NULL means that information already
 885              * written to file.
 886              */
 887             strbuf_close(inc->contents);
 888             inc->contents = NULL;
 889         }
 890         if (!inc->ref_count)
 891             continue;
 892         if (inc->ref_count == 1) {
 893             SPLIT ptable;
 894             char buf[1024];
 895 
 896             if (split(strbuf_value(inc->ref_contents), 4, &ptable) < 4) {
 897                 recover(&ptable);
 898                 die("too small number of parts in makefileindex().");
 899             }
 900             snprintf(buf, sizeof(buf), "%s %s", ptable.part[PART_LNO].start, decode_path(ptable.part[PART_PATH].start));
 901             recover(&ptable);
 902             strbuf_reset(inc->ref_contents);
 903             strbuf_puts(inc->ref_contents, buf);
 904         } else {
 905             char path[MAXPATHLEN];
 906 
 907             snprintf(path, sizeof(path), "%s/%s/%d.%s", distpath, INCREFS, no, HTML);
 908             fileop_INCLUDE = open_output_file(path, 0);
 909             INCLUDE = get_descripter(fileop_INCLUDE);
 910             fputs_nl(gen_page_begin(last, SUBDIR), INCLUDE);
 911             fputs_nl(body_begin, INCLUDE);
 912             fputs_nl(gen_list_begin(), INCLUDE);
 913             {
 914                 const char *line = strbuf_value(inc->ref_contents);
 915                 int count = inc->ref_count;
 916 
 917                 for (; count; line += strlen(line) + 1, count--)
 918                     fputs_nl(gen_list_body(upperdir(SRCS), line, NULL), INCLUDE);
 919             }
 920             fputs_nl(gen_list_end(), INCLUDE);
 921             fputs_nl(body_end, INCLUDE);
 922             fputs_nl(gen_page_end(), INCLUDE);
 923             close_file(fileop_INCLUDE);
 924             html_count++;
 925             /*
 926              * inc->ref_contents == NULL means that information already
 927              * written to file.
 928              */
 929             strbuf_close(inc->ref_contents);
 930             inc->ref_contents = NULL;
 931         }
 932     }
 933     strbuf_close(input);
 934 }
