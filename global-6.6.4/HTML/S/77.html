  66 findsymbol(char *pattern)
  67 {
  68     int status;
  69     STATIC_STRBUF(sb);
  70     strbuf_clear(sb);
  71 
  72     strbuf_puts(sb, common());
  73     strbuf_sprintf(sb, " -d %s > %s", quote_shell(pattern), temp1);
  74     status = mysystem("findsymbol_1", strbuf_value(sb));
  75     if (status != 0)
  76         return FAILED;
  77     strbuf_reset(sb);
  78     strbuf_puts(sb, common());
  79     strbuf_sprintf(sb, " -rs %s >> %s", quote_shell(pattern), temp1);
  80     status = mysystem("findsymbol_2", strbuf_value(sb));
  81     if (status != 0)
  82         return FAILED;
  83     return NULL;
  84 }

 116 findcalledby(char *pattern)
 117 {
 118     int status;
 119     char *p;

 228 findinclude(char *pattern)
 229 {
 230     int status;
 231     STATIC_STRBUF(sb);
 232     strbuf_clear(sb);
 233 #if defined(_WIN32) && !defined(__CYGWIN__)
 234 #define INCLUDE "\"^[ \t]*#[ \t]*include[ \t].*[<\\\"/\\]%s[\\\">]\""
 235 #elif defined(__DJGPP__)
 236 #define INCLUDE "'^[ \t]*#[ \t]*include[ \t].*[\"</\\]%s[\">]'"
 237 #else
 238 #define INCLUDE "'^[ \t]*#[ \t]*include[ \t].*[\"</]%s[\">]'"
 239 #endif
 240     strbuf_puts(sb, common());
 241     strbuf_sprintf(sb, " -g " INCLUDE " | sed \"s/<unknown>/<global>/\" > %s", quote_string(pattern), temp1);
 242     status = mysystem("findinclude", strbuf_value(sb));
 243         if (status != 0)
 244                 return FAILED;
 245         return NULL;

 290 void
 291 writeto(FILE *ip, char *outfile, int append) {
 292     FILE *op = fopen(outfile, append ? "a" : "w");
 293     const char *line;
 294     STATIC_STRBUF(sb);
 295     strbuf_clear(sb);
 296 
 297     if (op == NULL)
 298         return;
 299     while ((line = strbuf_fgets(sb, ip, 0)) != NULL) {
 300         fputs(line, op);
 301     }
 302     fclose(op);
 303 }

 310 findsymbol(char *pattern)
 311 {
 312     char **argv;
 313     FILE *ip;
 314 
 315     secure_open_args();
 316     common();
 317     secure_add_args("-d");
 318     secure_add_args(pattern);
 319     argv = secure_close_args();
 320     if (!(ip = secure_popen(global_command, "r", argv)))
 321         return FAILED;
 322     writeto(ip, temp1, 0);
 323     if (secure_pclose(ip) != 0)
 324         return FAILED;
 325

 328     secure_add_args("-rs");
 329     secure_add_args(pattern);
 330     argv = secure_close_args();
 331     if (!(ip = secure_popen(global_command, "r", argv)))
 332         return FAILED;
 333     writeto(ip, temp1, 1);
 334     if (secure_pclose(ip) != 0)
 335         return FAILED;
 336     return NULL;

 505 /*
 506  * [display.c]
 507  *
 508  * {"Find", "files #including this file",          findinclude},
 509  */
 510 char *
 511 findinclude(char *pattern)
 512 {
 513     char **argv;
 514     FILE *ip, *op;
 515     REWRITE *rw;
 516     const char *line;
 517     STATIC_STRBUF(sb);
 518     strbuf_clear(sb);
 519 
 520     secure_open_args();
 521     common();
 522     secure_add_args("-g");
 523     strbuf_puts(sb, "^[ \t]*#[ \t]*include[ \t].*[\"</]");
 524     strbuf_puts(sb, quote_string(pattern));
 525     strbuf_puts(sb, "[\">]");
 526     secure_add_args(strbuf_value(sb));
 527     argv = secure_close_args();
 528     if (!(ip = secure_popen(global_command, "r", argv)))
 529         return FAILED;
 530     op = fopen(temp1, "w");
 531     if (op == NULL)
 532         return FAILED;
 533     rw = rewrite_open("<unknown>", "<global>", 0);
 534     while ((line = strbuf_fgets(sb, ip, 0)) != NULL) {
 535         line = rewrite_string(rw, line, 0);
 536         fputs(line, op);
 537     }
 538     rewrite_close(rw);
 539     fclose(op);
 540     if (secure_pclose(ip) != 0)
 541         return FAILED;
 542     return NULL;
 543 }
 544 /*
 545  * [display.c]
 546  *
 547  * {"Find", "assignments to this symbol",    findassign},
 548  */
 549 char *
 550 findassign(char *pattern)
 551 {
 552     regex_t reg;
 553     const char *line;
 554     char **argv;
 555     FILE *ip, *op;
 556     int i;
 557     char *opts[] = {"-d", "-rs", NULL};
 558     STATIC_STRBUF(sb);
 559     strbuf_clear(sb);
 560 
 561     /*
 562      * For the time being I will support only C-colleagues.
 563      * Lisp, Cobol and etc are out of support.
 564      */
 565     strbuf_sprintf(sb, "\\b%s\\b[ \t]*=[^=]", pattern);
 566     if (regcomp(&reg, strbuf_value(sb), 0) != 0)
 567         return FAILED;
 568 
 569     op = fopen(temp1, "w");
 570     if (op == NULL)
 571         return FAILED;
 572     for (i = 0; opts[i] != NULL; i++) {
 573         secure_open_args();
 574         common();
 575         secure_add_args(opts[i]);
 576         secure_add_args(pattern);
 577         argv = secure_close_args();
 578         if (!(ip = secure_popen(global_command, "r", argv)))
 579             return FAILED;
 580         while ((line = strbuf_fgets(sb, ip, 0)) != NULL) {
 581             if (regexec(&reg, line, 0, 0, 0) == 0)
 582                 fputs(line, op);
 583         }
 584         if (secure_pclose(ip) != 0)
 585             return FAILED;
 586     }
 587     fclose(op);
 588     return NULL;
 589 }
