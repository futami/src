  84 /** see if there is a mouse interface */
  85 
  86 void
  87 mouseinit(void)
  88 {
  89     char    *term;
  90 
  91     /* see if this is emacsterm or viterm */
  92     term = mygetenv("TERM", "");
  93     if (strcmp(term, "emacsterm") == 0 || 
  94         strcmp(term, "viterm") == 0) {
  95         emacsviterm = YES;
  96         mouse = YES;
  97     }
  98     /* the MOUSE enviroment variable is for 5620 terminal programs that have
  99        mouse support but the TERM environment variable is the same as a
 100        terminal without a mouse, such as myx */
 101     else if (strcmp(mygetenv("MOUSE", ""), "myx") == 0) {
 102         mouse = YES;
 103     }
 104 #if UNIXPC
 105     else if (strcmp(term,"s4") == 0 || 
 106              strcmp(term,"s120") == 0 ||
 107              strcmp(term,"s90") == 0) {
 108         int retval;
 109         struct uwdata uwd;  /* Window data structure */
 110         struct umdata umd;  /* Mouse data structure */
 111 
 112         /* Ask for character size info */
 113             
 114         retval = ioctl(1,WIOCGETD,&uwd);
 115         if(retval || uwd.uw_hs <= 0 || uwd.uw_vs <= 0) {
 116             /**************************************************
 117              * something wrong with the kernel, so fake it... 
 118              **************************************************/
 119             if(!strcmp(term,"s4")) {
 120                 uw_hs = 9;
 121                 uw_vs = 12;
 122             }
 123             else {
 124                 uw_hs = 6;
 125                 uw_vs = 10;
 126             }
 127         }
 128         else {
 129             /* Kernel is working and knows about this font */
 130             uw_hs = uwd.uw_hs;
 131             uw_vs = uwd.uw_vs;
 132         }
 133         
 134         /**************************************************
 135          * Now turn on mouse reporting so we can actually
 136          * make use of all this stuff.
 137          **************************************************/
 138         if((retval = ioctl(1,WIOCGETMOUSE,&umd)) != -1) {
 139             umd.um_flags= MSDOWN+MSUP;
 140             ioctl(1,WIOCSETMOUSE,&umd);
 141         }
 142         unixpcmouse = YES;
 143     }
 144 #endif
 145     if (mouse == YES) {
 146         loadmenu(mainmenu);
 147     }
 148 }

 165 /** download a menu */
 166 
 167 static void
 168 loadmenu(MENU *menu)
 169 {
 170     int i;
 171 
 172     if (emacsviterm == YES) {
 173         mousereinit();
 174         (void) printf("\033V1");    /* display the scrollbar */
 175         (void) printf("\033M0@%s@%s@", menu[0].text, menu[0].value);
 176         for (i = 1; menu[i].text != NULL; ++i) {
 177             (void) printf("\033M@%s@%s@", menu[i].text, menu[i].value);
 178         }
 179     }
 180     else {  /* myx */
 181         int len;
 182         
 183         mousecleanup();
 184         (void) printf("\033[6;1X\033[9;1X");
 185         for (i = 0; menu[i].text != NULL; ++i) {
 186             len = strlen(menu[i].text);
 187             (void) printf("\033[%d;%dx%s%s", len,
 188                       (int) (len + strlen(menu[i].value)), 
 189                       menu[i].text, menu[i].value);
 190         }
 191         loaded = menu;
 192     }
 193     (void) fflush(stdout);
 194 }

 196 /** reinitialize the mouse in case curses changed the attributes */

 198 void
 199 mousereinit(void)
 200 {
 201     if (emacsviterm == YES) {
 202 
 203         /* enable the mouse click and sweep coordinate control sequence */
 204         /* and switch to menu 2 */
 205         (void) printf("\033{2\033#2");
 206         (void) fflush(stdout);
 207     }
 208 }

 210 /** restore the mouse attributes */
 211 
 212 void
 213 mousecleanup(void)
 214 {
 215     int i;
 216 
 217     if (loaded != NULL) {   /* only true for myx */
 218         
 219         /* remove the mouse menu */
 220         (void) printf("\033[6;0X\033[9;0X");
 221         for (i = 0; loaded[i].text != NULL; ++i) {
 222             (void) printf("\033[0;0x");
 223         }
 224         loaded = NULL;
 225     }
 226 }

 228 /** draw the scrollbar */
 229 
 230 void
 231 drawscrollbar(int top, int bot)
 232 {
 233     int p1, p2;
 234 
 235     if (emacsviterm == YES) {
 236         if (bot > top) {
 237             p1 = 16 + (top - 1) * 100 / totallines;
 238             p2 = 16 + (bot - 1) * 100 / totallines;
 239             if (p2 > 116) {
 240                 p2 = 116;
 241             }
 242             if (p1 < 16) {
 243                 p1 = 16;
 244             }
 245             /* don't send ^S or ^Q because it will hang a layer using cu(1) */
 246             if (p1 == ctrl('Q') || p1 == ctrl('S')) {
 247                 ++p1;
 248             }
 249             if (p2 == ctrl('Q') || p2 == ctrl('S')) {
 250                 ++p2;
 251             }
 252         }
 253         else {
 254             p1 = p2 = 16;
 255         }
 256         (void) printf("\033W%c%c", p1, p2);
 257     }
 258 }

 262 MOUSE *
 263 getmouseaction(char leading_char)
 264 {
 265     static  MOUSE   m;
 266 
 267 #if UNIXPC
 268 
 269     if(unixpcmouse == YES && leading_char == ESC) {
 270 
 271         /* Called if cscope received an ESC character.  See if it is
 272          * a mouse report and if so, decipher it.  A mouse report
 273          * looks like: "<ESC>[?xx;yy;b;rM"
 274          */
 275         int x = 0, y = 0, button = 0, reason = 0;
 276         int i;
 277     
 278         /* Get a mouse report.  The form is: XX;YY;B;RM where
 279          * XX is 1,2, or 3 decimal digits with the X pixel position.
 280          * Similarly for YY.  B is a single decimal digit with the
 281          * button number (4 for one, 2 for two, and 1 for three).
 282          * R is the reason for the mouse report.
 283          *
 284          * In general, the input is read until the mouse report has
 285          * been completely read in or we have discovered that this
 286          * escape sequence is NOT a mouse report.  In the latter case
 287          * return the last character read to the input stream with
 288          * myungetch().
 289          */
 290         
 291         /* Check for "[?" being next 2 chars */
 292         if(((i = mygetch()) != '[') || ((i = mygetch()) != '?')) {
 293             myungetch(i);
 294             return(NULL);
 295         }
 296     
 297         /* Grab the X position (in pixels) */
 298         while(isdigit(i = mygetch())) {
 299             x = (x*10) + (i - '0');
 300         }
 301         if(i != ';') {
 302             myungetch(i);
 303             return(NULL);   /* not a mouse report after all */
 304         }
 305     
 306         /* Grab the Y position (in pixels) */
 307         while(isdigit(i = mygetch())) {
 308             y = (y*10) + (i - '0');
 309         }
 310         if(i != ';') {
 311             myungetch(i);
 312             return(NULL);
 313         }
 314     
 315         /* Get which button */
 316         if((button = mygetch()) > '4') {
 317             myungetch(button);
 318             return(NULL);
 319         }
 320         if((i = mygetch()) != ';') {
 321             myungetch(i);
 322             return(NULL);
 323         }
 324         
 325         /* Get the reason for this mouse report */
 326         if((reason = mygetch()) > '8') {
 327             myungetch(reason);
 328             return(NULL);
 329         }
 330         
 331         /* sequence should terminate with an 'M' */
 332         if((i = mygetch()) != 'M') {
 333             myungetch(i);
 334             return(NULL);
 335         }
 336     
 337     
 338         /* OK.  We get a mouse report whenever a button is depressed
 339          * or released.  Let's ignore the report whenever the button
 340          * is depressed until when I am ready to implement sweeping.
 341          */
 342         if(reason != '2') {
 343             return(NULL);   /* '2' means button is released */
 344         }
 345     
 346         /************************************************************
 347          * Always indicate button 1 irregardless of which button was
 348          * really pushed.
 349          ************************************************************/
 350         m.button = 1;
 351     
 352         /************************************************************
 353          * Convert pixel coordinates to line and column coords.
 354          * The height and width are obtained using an ioctl() call
 355          * in mouseinit().  This assumes that variable width chars
 356          * are not being used ('though it would probably work anyway).
 357          ************************************************************/
 358         
 359         m.x1 = x/uw_hs; /* pixel/horizontal_spacing */
 360         m.y1 = y/uw_vs; /* pixel/vertical_spacing   */
 361     
 362         /* "null" out the other fields */
 363         m.percent = m.x2 = m.y2 = -1;
 364     }
 365     else
 366 #endif  /* not UNIXPC */
 367 
 368     if (mouse == YES && leading_char == ctrl('X')) {
 369     
 370         switch (mygetch()) {
 371         case ctrl('_'):     /* click */
 372             if ((m.button = mygetch()) == '0') {    /* if scrollbar */
 373                 m.percent = getpercent();
 374             }
 375             else {
 376                 m.x1 = getcoordinate();
 377                 m.y1 = getcoordinate();
 378                 m.x2 = m.y2 = -1;
 379             }
 380             break;
 381     
 382         case ctrl(']'):     /* sweep */
 383             m.button = mygetch();
 384             m.x1 = getcoordinate();
 385             m.y1 = getcoordinate();
 386             m.x2 = getcoordinate();
 387             m.y2 = getcoordinate();
 388             break;
 389         default:
 390             return(NULL);
 391         }
 392     }
 393     else return(NULL);
 394 
 395     return(&m);
 396 }

 398 /** get a row or column coordinate from a mouse button click or sweep */
 399 
 400 static int
 401 getcoordinate(void)
 402 {
 403     int  c, next;
 404 
 405     c = mygetch();
 406     next = 0;
 407     if (c == ctrl('A')) {
 408         next = 95;
 409         c = mygetch();
 410     }
 411     if (c < ' ') {
 412         return (0);
 413     }
 414     return (next + c - ' ');
 415 }
 416 
 417 /** get a percentage */
 418 
 419 static int
 420 getpercent(void)
 421 {
 422     int c;
 423 
 424     c = mygetch();
 425     if (c < 16) {
 426         return(0);
 427     }
 428     if (c > 120) {
 429         return(100);
 430     }
 431     return(c - 16);
 432 }
