  69 static void
  70 copy_langmap_converting_cpp(char *dst, const char *src)
  71 {
  72     const char *p;
  73 
  74     if (strncmp(src, "cpp:", 4) == 0) {
  75         memcpy(dst, "c++:", 4);
  76         dst += 4;
  77         src += 4;
  78     }
  79     while ((p = strstr(src, ",cpp:")) != NULL) {
  80         memcpy(dst, src, p - src);
  81         dst += p - src;
  82         memcpy(dst, ",c++:", 5);
  83         dst += 5;
  84         src = p + 5;
  85     }
  86     strcpy(dst, src);
  87 }

 163 static void
 164 start_process(const struct parser_param *param)
 165 {
 166     int opipe[2], ipipe[2];
 167 
 168     argv[1] = malloc(sizeof(LANGMAP_OPTION) + strlen(param->langmap));
 169     if (argv[1] == NULL)
 170         param->die("short of memory.");
 171     memcpy(argv[1], LANGMAP_OPTION, sizeof(LANGMAP_OPTION) - 1);
 172     copy_langmap_converting_cpp(argv[1] + sizeof(LANGMAP_OPTION) - 1, param->langmap);
 173 
 174     if (pipe(opipe) < 0 || pipe(ipipe) < 0)
 175         param->die("cannot create pipe.");
 176     pid = fork();
 177     if (pid == 0) {
 178         /* child process */
 179         close(opipe[1]);
 180         close(ipipe[0]);
 181         if (dup2(opipe[0], STDIN_FILENO) < 0
 182          || dup2(ipipe[1], STDOUT_FILENO) < 0)
 183             param->die("dup2 failed.");
 184         close(opipe[0]);
 185         close(ipipe[1]);
 186         execvp(PYGMENTS_PARSER, argv);
 187         param->die("cannot execute '%s'. (execvp failed)", PYGMENTS_PARSER);
 188     }
 189     /* parent process */
 190     if (pid < 0)
 191         param->die("fork failed.");
 192     free(argv[1]);
 193     close(opipe[0]);
 194     close(ipipe[1]);
 195     ip = fdopen(ipipe[0], "r");
 196     op = fdopen(opipe[1], "w");
 197     if (ip == NULL || op == NULL)
 198         param->die("fdopen failed.");
 199 
 200     bufsize = INITIAL_BUFSIZE;
 201     linebuf = malloc(bufsize);
 202     if (linebuf == NULL)
 203         param->die("short of memory.");
 204 }

 219 static char *
 220 get_line(const struct parser_param *param)
 221 {
 222     size_t linelen = 0;
 223 
 224     for (;;) {
 225         if (fgets(linebuf + linelen, bufsize - linelen, ip) == NULL) {
 226             if (linelen == 0)
 227                 return NULL;
 228             break;
 229         }
 230         linelen += strlen(linebuf + linelen);
 231         if (linelen < bufsize - 1 || linebuf[linelen - 1] == '\n'
 232          || feof(ip))
 233             break;
 234         bufsize *= 2;
 235         linebuf = realloc(linebuf, bufsize);
 236         if (linebuf == NULL)
 237             param->die("short of memory.");
 238     }
 239     while (linelen-- > 0
 240         && (linebuf[linelen] == '\n' || linebuf[linelen] == '\r'))
 241         linebuf[linelen] = '\0';
 242     return linebuf;
 243 }

 245 static void
 246 put_line(char *line, const struct parser_param *param)
 247 {
 248     int lineno;
 249     int type = PARSER_DEF;
 250     char *p, *tagname, *filename;
 251 
 252     /*
 253      * Output of ctags:
 254      * ctags -x ...
 255      * main              326 global/global.c  main(int argc, char **argv)
 256      *
 257      * ctags -x --gtags ...
 258      * D main              326 global/global.c  main(int argc, char **argv)
 259      */
 260     switch (*line) {
 261     case 'D':
 262         type = PARSER_DEF;
 263         break;
 264     case 'R':
 265         type = PARSER_REF_SYM;
 266         break;
 267     default:
 268         param->die("unexpected type string: %s", line);
 269     }
 270     /* skip type string */
 271     while (*line && !isspace((unsigned char)*line))
 272         line++;
 273     while (*line  && isspace((unsigned char)*line))
 274         line++;
 275     filename = strstr(line, param->file);
 276     if (filename == NULL || filename == line)
 277         return;
 278     p = filename - 1;
 279     if (!isspace((unsigned char)*p))
 280         return;
 281     while (p >= line && isspace((unsigned char)*p))
 282         *p-- = '\0';
 283     if (p < line)
 284         return;
 285     if (!isdigit((unsigned char)*p))
 286         return;
 287     while (p >= line && isdigit((unsigned char)*p))
 288         p--;
 289     if (p < line)
 290         return;
 291     lineno = atoi(p + 1);
 292     if (!isspace((unsigned char)*p))
 293         return;
 294     while (p >= line && isspace((unsigned char)*p))
 295         *p-- = '\0';
 296     if (p < line)
 297         return;
 298     while (p >= line && !isspace((unsigned char)*p))
 299         p--;
 300     tagname = p + 1;
 301     p = filename + strlen(param->file);
 302     if (*p != '\0') {
 303         if (!isspace((unsigned char)*p))
 304             return;
 305         *p++ = '\0';
 306         while (isspace((unsigned char)*p))
 307             p++;
 308     }
 309     param->put(type, tagname, lineno, filename, p, param->arg);
 310 }

 312 void
 313 parser(const struct parser_param *param)
 314 {
 315     char *line;
 316 
 317     assert(param->size >= sizeof(*param));
 318 
 319     if (op == NULL)
 320         start_process(param);
 321 
 322     /* Write path of input file to pipe. */
 323     fputs(param->file, op);
 324     putc('\n', op);
 325     fflush(op);
 326 
 327     /* Read output of the process. */
 328     for (;;) {
 329         line = get_line(param);
 330         if (line == NULL)
 331             param->die("unexpected EOF.");
 332         if (strcmp(line, TERMINATOR) == 0)
 333             break;
 334         put_line(line, param);
 335     }
 336 }
