  80 /**
  81  * __strbuf_expandbuf: expand buffer so that afford to the length data at least.
  82  *
  83  *  @param[in]  sb  STRBUF structure
  84  *  @param[in]  length  required room
  85  */
  86 void
  87 __strbuf_expandbuf(STRBUF *sb, int length)
  88 {
  89     int count = sb->curp - sb->sbuf;
  90     int newsize = sb->sbufsize + (length > EXPANDSIZE ? length : EXPANDSIZE);
  91     char *newbuf;
  92 
  93     newbuf = (char *)check_realloc(sb->sbuf, newsize + 1);
  94     sb->sbufsize = newsize;
  95     sb->sbuf = newbuf;
  96 
  97     sb->curp = sb->sbuf + count;
  98     sb->endp = sb->sbuf + sb->sbufsize;
  99 }
 100 /**
 101  * strbuf_open: open string buffer.
 102  *
 103  *  @param[in]  init    initial buffer size
 104  *          if 0 (zero) is specified then use default value (INITIALSIZE).
 105  *  @return sb  STRBUF structure
 106  */
 107 STRBUF *
 108 strbuf_open(int init)
 109 {
 110     STRBUF *sb = (STRBUF *)check_calloc(sizeof(STRBUF), 1);
 111 
 112     sb->sbufsize = (init > 0) ? init : INITIALSIZE;
 113     sb->sbuf = (char *)check_malloc(sb->sbufsize + 1);
 114     sb->curp = sb->sbuf;
 115     sb->endp = sb->sbuf + sb->sbufsize;
 116 
 117     return sb;
 118 }

 129 /**
 130  * strbuf_clear: clear static string buffer.
 131  *
 132  *  @param[in]  sb  statically defined string buffer
 133  *
 134  * This function is used for the initializing of static string buffer.
 135  * For the detail, see STATIC_STRBUF(sb) macro in "strbuf.h".
 136  */
 137 void
 138 strbuf_clear(STRBUF *sb)
 139 {
 140     if (sb == NULL)
 141         die("NULL string buffer. (strbuf_clear)");
 142     if (strbuf_empty(sb)) {
 143         sb->sbufsize = INITIALSIZE;
 144         sb->sbuf = (char *)check_malloc(sb->sbufsize + 1);
 145         sb->curp = sb->sbuf;
 146         sb->endp = sb->sbuf + sb->sbufsize;
 147     } else {
 148         strbuf_reset(sb);
 149     }
 150 }
 151 /**
 152  * strbuf_nputs: Put string with length
 153  *
 154  *  @param[in,out]  sb  string buffer
 155  *  @param[in]  s   string
 156  *  @param[in]  len length of string
 157  */
 158 void
 159 strbuf_nputs(STRBUF *sb, const char *s, int len)
 160 {
 161     if (len > 0) {
 162         if (sb->curp + len > sb->endp)
 163             __strbuf_expandbuf(sb, len);
 164         while (len-- > 0)
 165             *sb->curp++ = *s++;
 166     }
 167 }
 168 /**
 169  * strbuf_nputc: Put a character, len (number) times
 170  *
 171  *  @param[in,out]  sb  string buffer
 172  *  @param[in]  c   character
 173  *  @param[in]  len number of times to put c
 174  *
 175  *  See strbuf_putc()
 176  */
 177 void
 178 strbuf_nputc(STRBUF *sb, int c, int len)
 179 {
 180     if (len > 0) {
 181         if (sb->curp + len > sb->endp)
 182             __strbuf_expandbuf(sb, len);
 183         while (len-- > 0)
 184             *sb->curp++ = c;
 185     }
 186 }
 187 /**
 188  * strbuf_puts: Put string

 190  *  @param[in,out]  sb  string buffer
 191  *  @param[in]  s   string
 192  */
 193 void
 194 strbuf_puts(STRBUF *sb, const char *s)
 195 {
 196     while (*s) {
 197         if (sb->curp >= sb->endp)
 198             __strbuf_expandbuf(sb, 0);
 199         *sb->curp++ = *s++;
 200     }
 201 }

 203  * strbuf_puts_withterm: Put string until the terminator
 204  *
 205  *  @param[in,out]  sb  string buffer
 206  *  @param[in]  s   string
 207  *  @param[in]  c   terminator
 208  */
 209 void
 210 strbuf_puts_withterm(STRBUF *sb, const char *s, int c)
 211 {
 212     while (*s && *s != c) {
 213         if (sb->curp >= sb->endp)
 214             __strbuf_expandbuf(sb, 0);
 215         *sb->curp++ = *s++;
 216     }
 217 }
 218 /**
 219  * strbuf_puts_nl: Put string with a new line
 220  *
 221  *  @param[in,out]  sb  string buffer
 222  *  @param[in]  s   string
 223  */
 224 void
 225 strbuf_puts_nl(STRBUF *sb, const char *s)
 226 {
 227     while (*s) {
 228         if (sb->curp >= sb->endp)
 229             __strbuf_expandbuf(sb, 0);
 230         *sb->curp++ = *s++;
 231     }
 232     if (sb->curp >= sb->endp)
 233         __strbuf_expandbuf(sb, 0);
 234     *sb->curp++ = '\n';
 235 }
 236 /**
 237  * strbuf_putn: put digit string at the last of buffer.
 238  *
 239  *  @param[in,out]  sb  STRBUF structure
 240  *  @param[in]  n   number
 241  */
 242 void
 243 strbuf_putn(STRBUF *sb, int n)
 244 {
 245     if (n == 0) {
 246         strbuf_putc(sb, '0');
 247     } else {
 248         char num[128];
 249         int i = 0;
 250 
 251         while (n) {
 252             if (i >= sizeof(num))
 253                 die("Too big integer value.");
 254             num[i++] = n % 10 + '0';
 255             n = n / 10;
 256         }
 257         while (--i >= 0)
 258             strbuf_putc(sb, num[i]);
 259     }
 260 }

 286 /**
 287  * strbuf_unputc: remove specified char from the last of buffer
 288  *
 289  *  @param[in,out]  sb  STRBUF structure
 290  *  @param[in]  c   character
 291  *  @return     0: do nothing, 1: removed
 292  */
 293 int
 294 strbuf_unputc(STRBUF *sb, int c)
 295 {
 296     if (sb->curp > sb->sbuf && *(sb->curp - 1) == c) {
 297         sb->curp--;
 298         return 1;
 299     }
 300     return 0;
 301 }

 314 /**
 315  * strbuf_trim: trim following blanks.
 316  *
 317  *  @param[in,out]  sb  STRBUF structure
 318  */
 319 void
 320 strbuf_trim(STRBUF *sb)
 321 {
 322     char *p = sb->curp;
 323 
 324     while (p > sb->sbuf && isblank(*(p - 1)))
 325         *--p = 0;
 326     sb->curp = p;
 327 }
 328 /*
 329  * strbuf_fgets: read whole record into string buffer
 330  *
 331  *  @param[in,out]  sb  string buffer
 332  *  @param[in]  ip  input stream
 333  *  @param[in]  flags   flags
 334  *          STRBUF_NOCRLF:  remove last '\n' and/or '\r' if exist.
 335  *          STRBUF_APPEND:  append next record to existing data
 336  *          STRBUF_SHARPSKIP: skip lines which start with '#'
 337  *  @return     record buffer (NULL at end of file)
 338  *
 339  * Returned buffer has whole record.
 340  * The buffer end with '\0'. If STRBUF_NOCRLF is set then buffer doesn't
 341  * include '\r' and '\n'.
 342  */
 343 char *
 344 strbuf_fgets(STRBUF *sb, FILE *ip, int flags)
 345 {
 346     if (!(flags & STRBUF_APPEND))
 347         strbuf_reset(sb);
 348 
 349     if (sb->curp >= sb->endp)
 350         __strbuf_expandbuf(sb, EXPANDSIZE); /* expand buffer */
 351 
 352     if (flags & STRBUF_SHARPSKIP) {
 353         int c = 0;
 354 
 355         /* skip comment lines */
 356         while ((c = fgetc(ip)) == '#') {
 357             /* read and thrown away until the last of the line */
 358             while ((c = fgetc(ip)) != EOF && c != '\n')
 359                 ;
 360         }
 361         if (c == EOF)
 362             return NULL;
 363         /* push back the first character of the line */
 364         ungetc(c, ip);
 365     }
 366     for (;;) {
 367         if (!fgets(sb->curp, sb->endp - sb->curp, ip)) {
 368             if (sb->curp == sb->sbuf)
 369                 return NULL;
 370             break;
 371         }
 372         sb->curp += strlen(sb->curp);
 373         if (sb->curp > sb->sbuf && *(sb->curp - 1) == '\n')
 374             break;
 375         else if (feof(ip)) {
 376             return sb->sbuf;
 377         }
 378         __strbuf_expandbuf(sb, EXPANDSIZE); /* expand buffer */
 379     }
 380     if (flags & STRBUF_NOCRLF) {
 381         if (*(sb->curp - 1) == '\n')
 382             *(--sb->curp) = 0;
 383         if (sb->curp > sb->sbuf && *(sb->curp - 1) == '\r')
 384             *(--sb->curp) = 0;
 385     }
 386     return sb->sbuf;
 387 }
 388 /*
 389  * strbuf_sprintf: do sprintf into string buffer.
 390  *
 391  *  @param[in,out]  sb  STRBUF structure
 392  *  @param[in]  s   similar to sprintf()
 393  *          Currently the following format is supported.
 394  *          %s, %d, %<number>d, %<number>s, %-<number>d, %-<number>s
 395  */
 396 void
 397 strbuf_sprintf(STRBUF *sb, const char *s, ...)
 398 {
 399     va_list ap;
 400 
 401     va_start(ap, s);
 402     strbuf_vsprintf(sb, s, ap);
 403     va_end(ap);
 404 }
 405 /*
 406  * strbuf_vsprintf: do vsprintf into string buffer.
 407  *
 408  *  @param[in,out]  sb  STRBUF structure
 409  *  @param[in]  s   similar to vsprintf()
 410  *          Currently the following format is supported.
 411  *          %s, %d, %<number>d, %<number>s, %-<number>d, %-<number>s
 412  *  @param[in]  ap 
 413  */
 414 void
 415 strbuf_vsprintf(STRBUF *sb, const char *s, va_list ap)
 416 {
 417     for (; *s; s++) {
 418         /*
 419          * Put the before part of '%'.
 420          */
 421         {
 422             const char *p;
 423             for (p = s; *p && *p != '%'; p++)
 424                 ;
 425             if (p > s) {
 426                 strbuf_nputs(sb, s, p - s);
 427                 s = p;
 428             }
 429         }
 430         if (*s == '\0')
 431             break;
 432         if (*s == '%') {
 433             int c = (unsigned char)*++s;
 434             /*
 435              * '%%' means '%'.
 436              */
 437             if (c == '%') {
 438                 strbuf_putc(sb, c);
 439             }
 440             /*
 441              * If the optional number is specified then
 442              * we forward the job to snprintf(3).
 443              * o %<number>d
 444              * o %<number>s
 445              * o %-<number>d
 446              * o %-<number>s
 447              */
 448             else if (isdigit(c) || (c == '-' && isdigit((unsigned char)*(s + 1)))) {
 449                 char format[32], buf[1024];
 450                 int i = 0;
 451 
 452                 format[i++] = '%';
 453                 if (c == '-')
 454                     format[i++] = *s++;
 455                 while (isdigit((unsigned char)*s))
 456                     format[i++] = *s++;
 457                 format[i++] = c = *s;
 458                 format[i] = '\0';
 459                 if (c == 'd' || c == 'x')
 460                     snprintf(buf, sizeof(buf), format, va_arg(ap, int));
 461                 else if (c == 's')
 462                     snprintf(buf, sizeof(buf), format, va_arg(ap, char *));
 463                 else
 464                     die("Unsupported control character '%c'.", c);
 465                 strbuf_puts(sb, buf);
 466             } else if (c == 's') {
 467                 strbuf_puts(sb, va_arg(ap, char *));
 468             } else if (c == 'd') {
 469                 strbuf_putn(sb, va_arg(ap, int));
 470             } else {
 471                 die("Unsupported control character '%c'.", c);
 472             }
 473         }
 474     }
 475 }

 489 /**
 490  * STRBUF *strbuf_open_tempbuf(void)
 491  *
 492  * Temporary string buffer for general purpose.
 493  *
 494  * Usage:
 495  *
 496  *  STRBUF *sbt = strbuf_open_tempbuf();
 497  *  ....
 498  *  strbuf_puts(sbtemp, "xxx");
 499  *  ...
 500  *  strbuf_release_tempbuf(sbt);
 501  *
 502  */
 503 static int used = 0;
 504 
 505 STRBUF *
 506 strbuf_open_tempbuf(void)
 507 {
 508     STATIC_STRBUF(sb);
 509     if (used)
 510         die("Internal error: temporary string buffer is already used.");
 511     used = 1;
 512     strbuf_clear(sb);
 513     return sb;
 514 }

 516 /**
 517  * [Note] this function is for use with strbuf_open_tempbuf() only.
 518  */
 519 void
 520 strbuf_release_tempbuf(STRBUF *sb)
 521 {
 522     used = 0;
 523 }
