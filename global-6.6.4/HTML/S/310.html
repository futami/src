  57 /**
  58  * construct language map.
  59  *
  60  * copy string langmap (map) and convert it to language map like this:
  61  *
  62  * langmap (string) "c:.c.h,java:.java,cpp:.C.H"
  63  *  |
  64  *  v
  65  * language map     c\0.c.h\0java\0.java\0cpp\0.C.H\0
  66  */
  67 void
  68 setup_langmap(const char *map)
  69 {
  70     char *p;
  71     int onsuffix = 0;       /* not on suffix string */
  72 
  73     active_map = strbuf_open(0);
  74     strbuf_puts(active_map, map);
  75     for (p = strbuf_value(active_map); *p; p++) {
  76         /*
  77          * "c:.c.h,java:.java,cpp:.C.H"
  78          */
  79         if ((onsuffix == 0 && *p == ',') || (onsuffix == 1 && *p == ':'))
  80             die_with_code(2, "syntax error in langmap '%s'.", map);
  81         if (*p == ':' || *p == ',') {
  82             onsuffix ^= 1;
  83             *p = '\0';
  84         }
  85     }
  86     if (onsuffix == 0)
  87         die_with_code(2, "syntax error in langmap '%s'.", map);
  88     /* strbuf_close(active_map); */
  89 }
  90 /**
  91  * trim suffix list
  92  *
  93  * Remove duplicated suffixs from the suffix list.
  94  */
  95 static void
  96 trim_suffix_list(STRBUF *list, STRHASH *hash) {
  97     STATIC_STRBUF(sb);
  98     const char *p;
  99 
 100     strbuf_clear(sb);
 101     strbuf_puts(sb, strbuf_value(list));
 102     strbuf_reset(list);
 103     p = strbuf_value(sb);
 104     while (*p) {
 105         STATIC_STRBUF(suffix);
 106         struct sh_entry *sh;
 107 
 108         strbuf_clear(suffix);
 109         switch (*p) {
 110         case '.':   
 111             strbuf_putc(suffix, *p); 
 112             for (p++; *p && *p != '.'; p++)
 113                 strbuf_putc(suffix, *p); 
 114             break;
 115         case '(':
 116             strbuf_putc(suffix, *p); 
 117             for (p++; *p && *p != ')'; p++)
 118                 strbuf_putc(suffix, *p); 
 119             if (!*p)
 120                 die_with_code(2, "syntax error in the suffix list '%s'.", strbuf_value(sb));
 121             strbuf_putc(suffix, *p++);
 122             break;
 123         default:
 124             die_with_code(2, "syntax error in the suffix list '%s'.", strbuf_value(sb));
 125             break;
 126         }
 127         if ((sh = strhash_assign(hash, strbuf_value(suffix), 0)) != NULL) {
 128             if (!sh->value && wflag)
 129                 warning("langmap: suffix '%s' is duplicated. all except for the head is ignored.", strbuf_value(suffix));
 130             sh->value = (void *)1;
 131         } else {
 132             strbuf_puts(list, strbuf_value(suffix));
 133             (void)strhash_assign(hash, strbuf_value(suffix), 1);
 134         }
 135     }
 136 }
 137 /*
 138  * merge and trim a langmap
 139  *
 140  * (1) duplicated suffixes except for the first one are ignored.
 141  * (2) one more entries which belong to a language are gathered by one.
 142  *
 143  * Example:
 144  * C++:.cpp.c++,Java:.java.cpp,C++:.inl
 145  *      |                 ----(1)  ----(2)
 146  *  v
 147  * C++:.cpp.c++.inl,Java:.java
 148  */
 149 const char *
 150 trim_langmap(const char *map)
 151 {
 152     typedef struct {
 153         char *name; /* language: C++ */
 154         char *list; /* suffixes: .cpp.c++ */
 155     } SUFFIX;
 156     STATIC_STRBUF(sb);
 157     const char *p = map;
 158     STRBUF *name = strbuf_open(0);
 159     STRBUF *list = strbuf_open(0);
 160     STRHASH *hash = strhash_open(10);
 161     VARRAY *vb = varray_open(sizeof(SUFFIX), 32);
 162     SUFFIX *ent = NULL;
 163     int i;
 164 
 165     strbuf_clear(sb);
 166     while (*p) {
 167         strbuf_reset(name);
 168         strbuf_reset(list);
 169         strbuf_puts(name, strmake(p, ":"));
 170         p += strbuf_getlen(name) + 1;
 171         strbuf_puts(list, strmake(p, ","));
 172         p += strbuf_getlen(list);
 173         if (*p)
 174             p++;
 175         if (strbuf_getlen(name) == 0)
 176             die_with_code(2, "syntax error in langmap '%s'.", map);
 177         if (strchr(strbuf_value(name), ','))
 178             die_with_code(2, "syntax error in langmap '%s'.", map);
 179         /*
 180          * ignores duplicated suffixes.
 181          */
 182         trim_suffix_list(list, hash);
 183         if (strbuf_getlen(list) == 0)
 184             continue;
 185         /*
 186          * examine whether it appeared already.
 187          */
 188         ent = NULL;
 189         for (i = 0; i < vb->length; i++) {
 190             SUFFIX *ent0 = varray_assign(vb, i, 0);
 191             if (!strcmp(ent0->name, strbuf_value(name))) {
 192                 ent = ent0;
 193                 break;
 194             }
 195         }
 196         if (ent == NULL) {
 197             /* set initial values to a new entry */
 198             ent = varray_append(vb);
 199             ent->name = check_strdup(strbuf_value(name));
 200             ent->list = check_strdup(strbuf_value(list));
 201         } else {
 202             /* append values to the entry */
 203             ent->list = check_realloc(ent->list,
 204                 strlen(ent->list) + strbuf_getlen(list) + 1);
 205             strcat(ent->list, strbuf_value(list));
 206         }
 207     }
 208     for (i = 0; i < vb->length; i++) {
 209         ent = varray_assign(vb, i, 0);
 210         if (i > 0)
 211             strbuf_putc(sb, ',');
 212         strbuf_puts(sb, ent->name);
 213         strbuf_putc(sb, ':');
 214         strbuf_puts(sb, ent->list);
 215         free(ent->name);
 216         free(ent->list);
 217     }
 218     strbuf_close(name);
 219     strbuf_close(list);
 220     strhash_close(hash);
 221     varray_close(vb);
 222     return strbuf_value(sb);
 223 }

 225 /**
 226  * decide language of the suffix.
 227  *
 228  *      Though '*.h' files are shared by C and C++, GLOBAL treats them
 229  *      as C source files by default. If you set an environment variable
 230  *      'GTAGSFORCECPP' then C++ parser will be invoked.
 231  */
 232 const char *
 233 decide_lang(const char *suffix)
 234 {
 235     const char *lang, *list, *tail;
 236 
 237     /*
 238      * Though '*.h' files are shared by C and C++, GLOBAL treats them
 239      * as C source files by default. If you set an environment variable
 240      * 'GTAGS_FORCECPP' then C++ parser will be invoked.
 241      */
 242     if (!strcmp(suffix, ".h") && getenv("GTAGSFORCECPP") != NULL)
 243         return "cpp";
 244     lang = strbuf_value(active_map);
 245     tail = lang + strbuf_getlen(active_map);
 246 
 247     /* check whether or not list includes suffix. */
 248     while (lang < tail) {
 249         list = lang + strlen(lang) + 1;
 250         if (match_suffix_list(suffix, NULL, list))
 251             return lang;
 252         lang = list + strlen(list) + 1;
 253     }
 254     return NULL;
 255 }
 256 /**
 257  * decide language of the path.
 258  */
 259 const char *
 260 decide_lang_path(const char *path)
 261 {
 262     const char *lang, *list, *tail;
 263     const char *suffix = locatestring(path, ".", MATCH_LAST);
 264     const char *basename = locatestring(path, "/", MATCH_LAST);
 265 
 266     if (basename)
 267         basename++;
 268     /*
 269      * Though '*.h' files are shared by C and C++, GLOBAL treats them
 270      * as C source files by default. If you set an environment variable
 271      * 'GTAGS_FORCECPP' then C++ parser will be invoked.
 272      */
 273     if (!strcmp(suffix, ".h") && getenv("GTAGSFORCECPP") != NULL)
 274         return "cpp";
 275     lang = strbuf_value(active_map);
 276     tail = lang + strbuf_getlen(active_map);
 277 
 278     /* check whether or not list includes suffix. */
 279     while (lang < tail) {
 280         list = lang + strlen(lang) + 1;
 281         if (match_suffix_list(suffix, basename, list))
 282             return lang;
 283         lang = list + strlen(list) + 1;
 284     }
 285     return NULL;
 286 }

 297 static int
 298 match_suffix_list(const char *suffix, const char *basename, const char *list)
 299 {
 300     const char *p;
 301 
 302     strbuf_clear(lastmatch);
 303     suffix++;   /* skip '.' */
 304     /*
 305      * list includes suffixes and/or patterns.
 306      */
 307     while (*list) {
 308         if (*list == '.') {
 309             p = strmake(++list, ".(");
 310             if (locatestring(p, suffix, MATCH_COMPLETE
 311 #if defined(_WIN32) || defined(__DJGPP__)
 312                                 |IGNORE_CASE
 313 #endif
 314             )) {
 315                 strbuf_putc(lastmatch, '.');
 316                 strbuf_puts(lastmatch, p);
 317                 return 1;
 318             }
 319             list += strlen(p);
 320         } else if (*list == '(') {
 321             p = strmake(++list, ")");
 322             if (basename && fnmatch(p, basename, 0) == 0) {
 323                 strbuf_putc(lastmatch, '(');
 324                 strbuf_puts(lastmatch, p);
 325                 strbuf_putc(lastmatch, ')');
 326                 return 1;
 327             }
 328             list += strlen(p) + 1;
 329         }
 330     }
 331     return 0;
 332 }
