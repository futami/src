  56 /*
  57  * Sequential scan support.
  58  *
  59  * The tree can be scanned sequentially, starting from either end of the
  60  * tree or from any specific key.  A scan request before any scanning is
  61  * done is initialized as starting from the least node.
  62  */
  63 
  64 /**
  65  * __bt_seq --
  66  *  Btree sequential scan interface.
  67  *
  68  *  @param dbp  pointer to access method
  69  *  @param key  key for positioning and return value
  70  *  @param data data return value
  71  *  @param flags    R_CURSOR, R_FIRST, R_LAST, R_NEXT, R_PREV.
  72  *
  73  * @return
  74  *  RET_ERROR, RET_SUCCESS or RET_SPECIAL if there's no next key.
  75  */
  76 int
  77 __bt_seq(dbp, key, data, flags)
  78     const DB *dbp;
  79     DBT *key, *data;
  80     u_int flags;
  81 {
  82     BTREE *t;
  83     EPG e;
  84     int status;
  85 
  86     t = dbp->internal;
  87 
  88     /* Toss any page pinned across calls. */
  89     if (t->bt_pinned != NULL) {
  90         mpool_put(t->bt_mp, t->bt_pinned, 0);
  91         t->bt_pinned = NULL;
  92     }
  93 
  94     /*
  95      * If scan unitialized as yet, or starting at a specific record, set
  96      * the scan to a specific key.  Both __bt_seqset and __bt_seqadv pin
  97      * the page the cursor references if they're successful.
  98      */
  99     switch (flags) {
 100     case R_NEXT:
 101     case R_PREV:
 102         if (F_ISSET(&t->bt_cursor, CURS_INIT)) {
 103             status = __bt_seqadv(t, &e, flags);
 104             break;
 105         }
 106         /* FALLTHROUGH */
 107     case R_FIRST:
 108     case R_LAST:
 109     case R_CURSOR:
 110         status = __bt_seqset(t, &e, key, flags);
 111         break;
 112     default:
 113         errno = EINVAL;
 114         return (RET_ERROR);
 115     }
 116 
 117     if (status == RET_SUCCESS) {
 118         __bt_setcur(t, e.page->pgno, e.index);
 119 
 120         status =
 121             __bt_ret(t, &e, key, &t->bt_rkey, data, &t->bt_rdata, 0);
 122 
 123         /*
 124          * If the user is doing concurrent access, we copied the
 125          * key/data, toss the page.
 126          */
 127         if (F_ISSET(t, B_DB_LOCK))
 128             mpool_put(t->bt_mp, e.page, 0);
 129         else
 130             t->bt_pinned = e.page;
 131     }
 132     return (status);
 133 }

 135 /**
 136  * __bt_seqset --
 137  *  Set the sequential scan to a specific key.
 138  *
 139  *  @param t    tree
 140  *  @param ep   storage for returned key
 141  *  @param key  key for initial scan position
 142  *  @param flags    R_CURSOR, R_FIRST, R_LAST, R_NEXT, R_PREV
 143  *  @return     RET_ERROR, RET_SUCCESS or RET_SPECIAL if there's no next key.
 144  *
 145  * Side effects:
 146  *  Pins the page the cursor references.
 147  *
 148  */
 149 static int
 150 __bt_seqset(t, ep, key, flags)
 151     BTREE *t;
 152     EPG *ep;
 153     DBT *key;
 154     int flags;
 155 {
 156     PAGE *h;
 157     pgno_t pg;
 158     int exact;
 159 
 160     /*
 161      * Find the first, last or specific key in the tree and point the
 162      * cursor at it.  The cursor may not be moved until a new key has
 163      * been found.
 164      */
 165     switch (flags) {
 166     case R_CURSOR:              /* Keyed scan. */
 167         /*
 168          * Find the first instance of the key or the smallest key
 169          * which is greater than or equal to the specified key.
 170          */
 171         if (key->data == NULL || key->size == 0) {
 172             errno = EINVAL;
 173             return (RET_ERROR);
 174         }
 175         return (__bt_first(t, key, ep, &exact));
 176     case R_FIRST:               /* First record. */
 177     case R_NEXT:
 178         /* Walk down the left-hand side of the tree. */
 179         for (pg = P_ROOT;;) {
 180             if ((h = mpool_get(t->bt_mp, pg, 0)) == NULL)
 181                 return (RET_ERROR);
 182 
 183             /* Check for an empty tree. */
 184             if (NEXTINDEX(h) == 0) {
 185                 mpool_put(t->bt_mp, h, 0);
 186                 return (RET_SPECIAL);
 187             }
 188 
 189             if (h->flags & (P_BLEAF | P_RLEAF))
 190                 break;
 191             pg = GETBINTERNAL(h, 0)->pgno;
 192             mpool_put(t->bt_mp, h, 0);
 193         }
 194         ep->page = h;
 195         ep->index = 0;
 196         break;
 197     case R_LAST:                /* Last record. */
 198     case R_PREV:
 199         /* Walk down the right-hand side of the tree. */
 200         for (pg = P_ROOT;;) {
 201             if ((h = mpool_get(t->bt_mp, pg, 0)) == NULL)
 202                 return (RET_ERROR);
 203 
 204             /* Check for an empty tree. */
 205             if (NEXTINDEX(h) == 0) {
 206                 mpool_put(t->bt_mp, h, 0);
 207                 return (RET_SPECIAL);
 208             }
 209 
 210             if (h->flags & (P_BLEAF | P_RLEAF))
 211                 break;
 212             pg = GETBINTERNAL(h, NEXTINDEX(h) - 1)->pgno;
 213             mpool_put(t->bt_mp, h, 0);
 214         }
 215 
 216         ep->page = h;
 217         ep->index = NEXTINDEX(h) - 1;
 218         break;
 219     }
 220     return (RET_SUCCESS);
 221 }

 223 /**
 224  * __bt_seqadvance --
 225  *  Advance the sequential scan.
 226  *
 227  *  @param t    tree
 228  *  @param ep
 229  *  @param flags    R_NEXT, R_PREV
 230  *  @return     RET_ERROR, RET_SUCCESS or RET_SPECIAL if there's no next key.
 231  *
 232  * Side effects:
 233  *  Pins the page the new key/data record is on.
 234  */
 235 static int
 236 __bt_seqadv(t, ep, flags)
 237     BTREE *t;
 238     EPG *ep;
 239     int flags;
 240 {
 241     CURSOR *c;
 242     PAGE *h;
 243     indx_t index = 0;
 244     pgno_t pg;
 245     int exact;
 246 
 247     /*
 248      * There are a couple of states that we can be in.  The cursor has
 249      * been initialized by the time we get here, but that's all we know.
 250      */
 251     c = &t->bt_cursor;
 252 
 253     /*
 254      * The cursor was deleted where there weren't any duplicate records,
 255      * so the key was saved.  Find out where that key would go in the
 256      * current tree.  It doesn't matter if the returned key is an exact
 257      * match or not -- if it's an exact match, the record was added after
 258      * the delete so we can just return it.  If not, as long as there's
 259      * a record there, return it.
 260      */
 261     if (F_ISSET(c, CURS_ACQUIRE))
 262         return (__bt_first(t, &c->key, ep, &exact));
 263 
 264     /* Get the page referenced by the cursor. */
 265     if ((h = mpool_get(t->bt_mp, c->pg.pgno, 0)) == NULL)
 266         return (RET_ERROR);
 267 
 268     /*
 269      * Find the next/previous record in the tree and point the cursor at
 270      * it.  The cursor may not be moved until a new key has been found.
 271      */
 272     switch (flags) {
 273     case R_NEXT:            /* Next record. */
 274         /*
 275          * The cursor was deleted in duplicate records, and moved
 276          * forward to a record that has yet to be returned.  Clear
 277          * that flag, and return the record.
 278          */
 279         if (F_ISSET(c, CURS_AFTER))
 280             goto usecurrent;
 281         index = c->pg.index;
 282         if (++index == NEXTINDEX(h)) {
 283             pg = h->nextpg;
 284             mpool_put(t->bt_mp, h, 0);
 285             if (pg == P_INVALID)
 286                 return (RET_SPECIAL);
 287             if ((h = mpool_get(t->bt_mp, pg, 0)) == NULL)
 288                 return (RET_ERROR);
 289             index = 0;
 290         }
 291         break;
 292     case R_PREV:            /* Previous record. */
 293         /*
 294          * The cursor was deleted in duplicate records, and moved
 295          * backward to a record that has yet to be returned.  Clear
 296          * that flag, and return the record.
 297          */
 298         if (F_ISSET(c, CURS_BEFORE)) {
 299 usecurrent:     F_CLR(c, CURS_AFTER | CURS_BEFORE);
 300             ep->page = h;
 301             ep->index = c->pg.index;
 302             return (RET_SUCCESS);
 303         }
 304         index = c->pg.index;
 305         if (index == 0) {
 306             pg = h->prevpg;
 307             mpool_put(t->bt_mp, h, 0);
 308             if (pg == P_INVALID)
 309                 return (RET_SPECIAL);
 310             if ((h = mpool_get(t->bt_mp, pg, 0)) == NULL)
 311                 return (RET_ERROR);
 312             index = NEXTINDEX(h) - 1;
 313         } else
 314             --index;
 315         break;
 316     }
 317 
 318     ep->page = h;
 319     ep->index = index;
 320     return (RET_SUCCESS);
 321 }

 323 /**
 324  * __bt_first --
 325  *  Find the first entry.
 326  *
 327  *  @param t    the tree
 328  *  @param key  the key
 329  *  @param erval    return EPG
 330  *  @param exactp   pointer to exact match flag
 331  *
 332  * @return
 333  *  The first entry in the tree greater than or equal to key,
 334  *  or RET_SPECIAL if no such key exists.
 335  */
 336 static int
 337 __bt_first(t, key, erval, exactp)
 338     BTREE *t;
 339     const DBT *key;
 340     EPG *erval;
 341     int *exactp;
 342 {
 343     PAGE *h;
 344     EPG *ep, save;
 345     pgno_t pg;
 346 
 347     /*
 348      * Find any matching record; __bt_search pins the page.
 349      *
 350      * If it's an exact match and duplicates are possible, walk backwards
 351      * in the tree until we find the first one.  Otherwise, make sure it's
 352      * a valid key (__bt_search may return an index just past the end of a
 353      * page) and return it.
 354      */
 355     if ((ep = __bt_search(t, key, exactp)) == NULL)
 356         return (0);
 357     if (*exactp) {
 358         if (F_ISSET(t, B_NODUPS)) {
 359             *erval = *ep;
 360             return (RET_SUCCESS);
 361         }
 362             
 363         /*
 364          * Walk backwards, as long as the entry matches and there are
 365          * keys left in the tree.  Save a copy of each match in case
 366          * we go too far.
 367          */
 368         save = *ep;
 369         h = ep->page;
 370         do {
 371             if (save.page->pgno != ep->page->pgno) {
 372                 mpool_put(t->bt_mp, save.page, 0);
 373                 save = *ep;
 374             } else
 375                 save.index = ep->index;
 376 
 377             /*
 378              * Don't unpin the page the last (or original) match
 379              * was on, but make sure it's unpinned if an error
 380              * occurs.
 381              */
 382             if (ep->index == 0) {
 383                 if (h->prevpg == P_INVALID)
 384                     break;
 385                 if (h->pgno != save.page->pgno)
 386                     mpool_put(t->bt_mp, h, 0);
 387                 if ((h = mpool_get(t->bt_mp,
 388                     h->prevpg, 0)) == NULL) {
 389                     if (h->pgno == save.page->pgno)
 390                         mpool_put(t->bt_mp,
 391                             save.page, 0);
 392                     return (RET_ERROR);
 393                 }
 394                 ep->page = h;
 395                 ep->index = NEXTINDEX(h);
 396             }
 397             --ep->index;
 398         } while (__bt_cmp(t, key, ep) == 0);
 399 
 400         /*
 401          * Reach here with the last page that was looked at pinned,
 402          * which may or may not be the same as the last (or original)
 403          * match page.  If it's not useful, release it.
 404          */
 405         if (h->pgno != save.page->pgno)
 406             mpool_put(t->bt_mp, h, 0);
 407 
 408         *erval = save;
 409         return (RET_SUCCESS);
 410     }
 411 
 412     /* If at the end of a page, find the next entry. */
 413     if (ep->index == NEXTINDEX(ep->page)) {
 414         h = ep->page;
 415         pg = h->nextpg;
 416         mpool_put(t->bt_mp, h, 0);
 417         if (pg == P_INVALID)
 418             return (RET_SPECIAL);
 419         if ((h = mpool_get(t->bt_mp, pg, 0)) == NULL)
 420             return (RET_ERROR);
 421         ep->index = 0;
 422         ep->page = h;
 423     }
 424     *erval = *ep;
 425     return (RET_SUCCESS);
 426 }

 428 /**
 429  * __bt_setcur --
 430  *  Set the cursor to an entry in the tree.
 431  *
 432  *  @param t    the tree
 433  *  @param pgno page number
 434  *  @param index    page index
 435  */
 436 void
 437 __bt_setcur(t, pgno, index)
 438     BTREE *t;
 439     pgno_t pgno;
 440     u_int index;
 441 {
 442     /* Lose any already deleted key. */
 443     if (t->bt_cursor.key.data != NULL) {
 444         free(t->bt_cursor.key.data);
 445         t->bt_cursor.key.size = 0;
 446         t->bt_cursor.key.data = NULL;
 447     }
 448     F_CLR(&t->bt_cursor, CURS_ACQUIRE | CURS_AFTER | CURS_BEFORE);
 449 
 450     /* Update the cursor. */
 451     t->bt_cursor.pg.pgno = pgno;
 452     t->bt_cursor.pg.index = index;
 453     F_SET(&t->bt_cursor, CURS_INIT);
 454 }
