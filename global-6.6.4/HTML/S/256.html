  72 static void
  73 load_notfunction(const char *filename)
  74 {
  75     FILE *ip;
  76     STRBUF *sb = strbuf_open(0);
  77     STRBUF *ib = strbuf_open(0);
  78     char *p;
  79     int i;
  80 
  81     if ((ip = fopen(filename, "r")) == NULL)
  82         die("'%s' cannot read.", filename);
  83     for (tablesize = 0; (p = strbuf_fgets(ib, ip, STRBUF_NOCRLF)) != NULL; tablesize++)
  84         strbuf_puts0(sb, p);
  85     fclose(ip);
  86     words = (struct words *)check_malloc(sizeof(struct words) * tablesize);
  87     /*
  88      * Don't free *p.
  89      */
  90     p = (char *)check_malloc(strbuf_getlen(sb) + 1);
  91     memcpy(p, strbuf_value(sb), strbuf_getlen(sb) + 1);
  92     for (i = 0; i < tablesize; i++) {
  93         words[i].name = p;
  94         p += strlen(p) + 1;
  95     }
  96     qsort(words, tablesize, sizeof(struct words), cmp);
  97     strbuf_close(sb);
  98     strbuf_close(ib);
  99 }

 101 static int
 102 isnotfunction(const char *name)
 103 {
 104     struct words tmp;
 105     struct words *result;
 106 
 107     if (words == NULL)
 108         return 0;
 109     tmp.name = name;
 110     result = (struct words *)bsearch(&tmp, words, tablesize, sizeof(struct words), cmp);
 111     return (result != NULL) ? 1 : 0;
 112 }

 139 /**
 140  * load_plugin_parser: Load plug-in parsers.
 141  *
 142  *  @param[in]  pluginspec  described below
 143  *
 144  * Syntax:
 145  *   <pluginspec> ::= <map> | <map>","<pluginspec>
 146  *   <map>        ::= <language name>":"<shared object path>
 147  *                  | <language name>":"<shared object path>":"<function name>
 148  */
 149 static void
 150 load_plugin_parser(const char *pluginspec)
 151 {
 152     char *p, *q;
 153     const char *lt_dl_name, *parser_name;
 154     struct plugin_entry *pent;
 155 
 156     pluginspec_saved = check_strdup(pluginspec);
 157     if (lt_dlinit() != 0)
 158         die("cannot initialize libltdl.");
 159     p = pluginspec_saved;
 160     while (*p != '\0') {
 161         pent = check_malloc(sizeof(*pent));
 162         pent->entry.lang_name = p;
 163         p = strchr(p, ':');
 164         if (p == NULL)
 165             die_with_code(2, "syntax error in pluginspec '%s'.", pluginspec);
 166         *p++ = '\0';
 167         if (*p == '\0')
 168             die_with_code(2, "syntax error in pluginspec '%s'.", pluginspec);
 169         lt_dl_name = p;
 170         p = strchr(p, ',');
 171         if (p != NULL)
 172             *p++ = '\0';
 173         q = strchr(lt_dl_name, ':');
 174 #ifdef _WIN32
 175         /* Assume a single-character name is a drive letter. */
 176         if (q == lt_dl_name + 1)
 177             q = strchr(q + 1, ':');
 178 #else
 179         /* skip quoted ':' */
 180         while (q && *q == ':' && *(q - 1) == '\\') {
 181             q = strchr(q + 1, ':');
 182         }
 183 #endif
 184         if (q == NULL) {
 185             parser_name = "parser";
 186         } else {
 187             *q++ = '\0';
 188             if (*q == '\0')
 189                 die_with_code(2, "syntax error in pluginspec '%s'.", pluginspec);
 190             parser_name = q;
 191         }
 192 #ifndef _WIN32
 193         strremovechar((char *)lt_dl_name, '\\');
 194 #endif
 195 #if defined(_WIN32) && !defined(__CYGWIN__)
 196         /* Bypass libtool and load the DLL directly, relative to us. */
 197         pent->handle = (lt_dlhandle)LoadLibrary(lt_dl_name);
 198         if (pent->handle == NULL) {
 199             q = strrchr(lt_dl_name, '/');
 200             if (q == NULL)
 201                 q = strrchr(lt_dl_name, '\\');
 202             if (q != NULL)
 203                 lt_dl_name = q + 1;
 204             q = strrchr(lt_dl_name, '.');
 205             if (q != NULL && strcmp(q, ".la") == 0)
 206                 *q = '\0';
 207             pent->handle = (lt_dlhandle)LoadLibrary(lt_dl_name);
 208             if (pent->handle == NULL) {
 209                 char dll[MAX_PATH*2];
 210                 GetModuleFileName(NULL, dll, MAX_PATH);
 211                 q = strrchr(dll, '\\');
 212                 sprintf(q+1, "..\\lib\\gtags\\%s", lt_dl_name);
 213                 pent->handle = (lt_dlhandle)LoadLibrary(dll);
 214             }
 215         }
 216         if (pent->handle == NULL)
 217             die_with_code(2, "cannot open shared object '%s'.", lt_dl_name);
 218         pent->entry.lt_dl_name = lt_dl_name;
 219         pent->entry.parser = (PVOID)GetProcAddress((HINSTANCE)pent->handle, parser_name);
 220 #else
 221         pent->handle = lt_dlopen(lt_dl_name);
 222         if (pent->handle == NULL) {
 223             /*
 224              * Retry after removing the extension, because some packages
 225              * don't have '.la' files.
 226              */
 227             q = strrchr(lt_dl_name, '.');
 228             if (q)
 229                 *q = '\0';
 230             pent->handle = lt_dlopenext(lt_dl_name);
 231             if (pent->handle == NULL)
 232                 die_with_code(2, "cannot open shared object '%s'.", lt_dl_name);
 233         }
 234         pent->entry.lt_dl_name = lt_dl_name;
 235         pent->entry.parser = lt_dlsym(pent->handle, parser_name);
 236 #endif
 237         if (pent->entry.parser == NULL)
 238             die_with_code(2, "cannot find symbol '%s' in '%s'.", parser_name, lt_dl_name);
 239         pent->entry.parser_name = parser_name;
 240         STAILQ_INSERT_TAIL(&plugin_list, pent, next);
 241         if (p == NULL)
 242             break;
 243     }
 244 }

 246 /**
 247  * unload_plugin_parser: Unload plug-in parsers.
 248  */
 249 static void
 250 unload_plugin_parser(void)
 251 {
 252     struct plugin_entry *pent;
 253 
 254     if (pluginspec_saved == NULL)
 255         return;
 256     while (!STAILQ_EMPTY(&plugin_list)) {
 257         pent = STAILQ_FIRST(&plugin_list);
 258 #if defined(_WIN32) && !defined(__CYGWIN__)
 259         FreeLibrary((HMODULE)pent->handle);
 260 #else
 261         lt_dlclose(pent->handle);
 262 #endif
 263         STAILQ_REMOVE_HEAD(&plugin_list, next);
 264         free(pent);
 265     }
 266     lt_dlexit();
 267     free(pluginspec_saved);
 268 }

 273 static const struct lang_entry lang_switch[] = {
 274     /* lang_name    parser_proc parser_name lt_dl_name, */
 275     /*              (for debug) (for debug) */
 276     {"c",       C,      "C",        "built-in"},    /* DEFAULT */
 277     {"yacc",    yacc,       "yacc",     "built-in"},
 278     {"cpp",     Cpp,        "Cpp",      "built-in"},
 279     {"java",    java,       "java",     "built-in"},
 280     {"php",     php,        "php",      "built-in"},
 281     {"asm",     assembly,   "assembly", "built-in"}
 282 };
 283 #define DEFAULT_ENTRY &lang_switch[0]
 284 /**
 285  * get language entry.
 286  *
 287  *      @param[in]      lang    language name (NULL means 'not specified'.)
 288  *      @return              language entry
 289  */
 290 static const struct lang_entry *
 291 get_lang_entry(const char *lang)
 292 {
 293     int i, size = sizeof(lang_switch) / sizeof(struct lang_entry);
 294     struct plugin_entry *pent;
 295 
 296     if (lang == NULL)
 297         die("get_lang_entry: something is wrong.");
 298     /*
 299      * Priority 1: locates in the plugin parser list.
 300      */
 301     STAILQ_FOREACH(pent, &plugin_list, next) {
 302         if (strcmp(lang, pent->entry.lang_name) == 0)
 303             return &pent->entry;
 304     }
 305     /*
 306      * Priority 2: locates in the built-in parser list.
 307      */
 308     for (i = 0; i < size; i++)
 309         if (!strcmp(lang, lang_switch[i].lang_name))
 310             return &lang_switch[i];
 311     /*
 312      * if specified language not found, it assumes default language, that is C.
 313      */
 314     warning("'%s' parser not found. C parser is used instead.", lang);
 315     return DEFAULT_ENTRY;
 316 }

 320  * [gtags.conf]

 322  * |...

 328  *  parser_init(langmap, plugin_parser);
 329  *
 330  * 2. Execute parsers
 331  *
 332  *  parse_file(...);
 333  *
 334  * 3. Unload parsers.
 335  *
 336  *  parser_exit();
 337  */
 338 /*
 339  * parser_init: load langmap and shared libraries.
 340  *
 341  *  @param[in]  langmap     the value of langmap=<langmap>
 342  *  @param[in]  pluginspec  the value of gtags_parser=<pluginspec>
 343  */
 344 void
 345 parser_init(const char *langmap, const char *pluginspec)
 346 {
 347     /* setup language mapping. */
 348     if (langmap == NULL)
 349         langmap = DEFAULTLANGMAP;
 350     langmap = trim_langmap(langmap);
 351     setup_langmap(langmap);
 352     langmap_saved = check_strdup(langmap);
 353 
 354     /* load shared objects. */
 355     if (pluginspec != NULL)
 356         load_plugin_parser(pluginspec);
 357 
 358     /*
 359      * This is a hack for FreeBSD.
 360      * In the near future, it will be removed.
 361      */
 362     if (test("r", NOTFUNCTION))
 363         load_notfunction(NOTFUNCTION);
 364 #ifdef __DJGPP__
 365     else if (test("r", DOS_NOTFUNCTION))
 366         load_notfunction(DOS_NOTFUNCTION);
 367 #endif
 368 }

 380 /**
 381  * getconf: return the value of the config variable by name.
 382  */
 383 char *
 384 getconf(const char *name)
 385 {
 386     int num = 0;
 387     STATIC_STRBUF(sb);
 388 
 389     strbuf_clear(sb);
 390     if (getconfs(name, sb))
 391         ;
 392     else if (getconfn(name, &num))
 393         strbuf_putn(sb, num);
 394     else if (getconfb(name))
 395         strbuf_putn(sb, 1);
 396     else
 397         return NULL;
 398     return check_strdup(strbuf_value(sb));
 399 }
 400 /**
 401  * parse_file: select and execute a parser.
 402  *
 403  *  @param[in]  path    path name
 404  *  @param[in]  flags   PARSER_WARNING: print warning messages
 405  *  @param[in]  put callback routine,
 406  *          each parser use this routine for output
 407  *  @param[in]  arg argument for callback routine
 408  */
 409 void
 410 parse_file(const char *path, int flags, PARSER_CALLBACK put, void *arg)
 411 {
 412     const struct lang_entry *ent = get_parser(path);
 413     if (ent) {
 414         if (flags & PARSER_EXPLAIN)
 415             fputs(get_explain(path, ent), stderr);
 416         execute_parser(ent, path, flags, put, arg);
 417     }
 418 }
 419 /**
 420  * get_parser: get a parser entry from a path.
 421  */
 422 const struct lang_entry *
 423 get_parser(const char *path)
 424 {
 425     const char *lang = decide_lang_path(path);
 426     if (lang == NULL)
 427         return NULL;
 428     /*
 429      * Select parser.
 430      * If lang == NULL then default parser is selected.
 431      */
 432     return get_lang_entry(lang);
 433 }
 434 /**
 435  * execute_parser: execute parser.

 437 void
 438 execute_parser(const struct lang_entry *ent, const char *path, int flags, PARSER_CALLBACK put, void *arg)
 439 {
 440     struct parser_param param;
 441     /*
 442      * call language specific parser.
 443      */
 444     param.size = sizeof(param);
 445     param.flags = flags;
 446     param.file = path;
 447     param.put = put;
 448     param.arg = arg;
 449     param.isnotfunction = isnotfunction;
 450     param.langmap = langmap_saved;
 451     param.getconf = getconf;
 452     param.die = die;
 453     param.warning = warning;
 454     param.message = message;
 455     ent->parser(&param);
 456 }
 457 /**
 458  * get_explain: get explain messages.

 460 const char *
 461 get_explain(const char *path, const struct lang_entry *ent)
 462 {
 463     STATIC_STRBUF(sb);
 464     strbuf_clear(sb);
 465 
 466     strbuf_sprintf(sb, " - File '%s' is handled as follows:\n", trimpath(path));
 467     strbuf_sprintf(sb, "\tsuffix:   |%s|\n", get_last_match());
 468     strbuf_sprintf(sb, "\tlanguage: |%s|\n", ent->lang_name);
 469     strbuf_sprintf(sb, "\tparser:   |%s|\n", ent->parser_name);
 470     strbuf_sprintf(sb, "\tlibrary:  |%s|\n", ent->lt_dl_name ? ent->lt_dl_name : "builtin library");
 471     return strbuf_value(sb);
 472 }
 473 void
 474 dbg_print(int level, const char *s)
 475 {
 476     fprintf(stderr, "[%04d]", lineno);
 477     for (; level > 0; level--)
 478         fprintf(stderr, "    ");
 479     fprintf(stderr, "%s\n", s);
