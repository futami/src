  67 /**
  68  * mpool_open --
  69  *  Initialize a memory pool.
  70  *
  71  *  @param key
  72  *  @param fd
  73  *  @param pagesize
  74  *  @param maxcache
  75  */
  76 MPOOL *
  77 mpool_open(key, fd, pagesize, maxcache)
  78     void *key;
  79     int fd;
  80     pgno_t pagesize, maxcache;
  81 {
  82     struct stat sb;
  83     MPOOL *mp;
  84     int entry;
  85 
  86     /*
  87      * Get information about the file.
  88      *
  89      * XXX
  90      * We don't currently handle pipes, although we should.
  91      */
  92     if (fstat(fd, &sb))
  93         return (NULL);
  94     if (!S_ISREG(sb.st_mode)) {
  95         errno = ESPIPE;
  96         return (NULL);
  97     }
  98 
  99     /* Allocate and initialize the MPOOL cookie. */
 100     if ((mp = (MPOOL *)calloc(1, sizeof(MPOOL))) == NULL)
 101         return (NULL);
 102     CIRCLEQ_INIT(&mp->lqh);
 103     for (entry = 0; entry < HASHSIZE; ++entry)
 104         CIRCLEQ_INIT(&mp->hqh[entry]);
 105     mp->maxcache = maxcache;
 106     mp->npages = sb.st_size / pagesize;
 107     mp->pagesize = pagesize;
 108     mp->fd = fd;
 109     return (mp);
 110 }

 133 /**
 134  * mpool_new --
 135  *  Get a new page of memory.
 136  *
 137  *  @param mp
 138  *  @param pgnoaddr
 139  */
 140 void *
 141 mpool_new(mp, pgnoaddr)
 142     MPOOL *mp;
 143     pgno_t *pgnoaddr;
 144 {
 145     struct _hqh *head;
 146     BKT *bp;
 147 
 148     if (mp->npages == MAX_PAGE_NUMBER) {
 149         (void)fprintf(stderr, "mpool_new: page allocation overflow.\n");
 150         abort();
 151     }
 152 #ifdef STATISTICS
 153     ++mp->pagenew;
 154 #endif
 155     /*
 156      * Get a BKT from the cache.  Assign a new page number, attach
 157      * it to the head of the hash chain, the tail of the lru chain,
 158      * and return.
 159      */
 160     if ((bp = mpool_bkt(mp)) == NULL)
 161         return (NULL);
 162     *pgnoaddr = bp->pgno = mp->npages++;
 163     bp->flags = MPOOL_PINNED;
 164 
 165     head = &mp->hqh[HASHKEY(bp->pgno)];
 166     CIRCLEQ_INSERT_HEAD(head, bp, hq);
 167     CIRCLEQ_INSERT_TAIL(&mp->lqh, bp, q);
 168     return (bp->page);
 169 }

 171 /**
 172  * mpool_get
 173  *  Get a page.
 174  *
 175  *  @param mp
 176  *  @param pgno
 177  *  @param flags
 178  */
 179 void *
 180 mpool_get(mp, pgno, flags)
 181     MPOOL *mp;
 182     pgno_t pgno;
 183     u_int flags;                /* XXX not used? */
 184 {
 185     struct _hqh *head;
 186     BKT *bp;
 187     off_t off;
 188     int nr;
 189 
 190     /* Check for attempt to retrieve a non-existent page. */
 191     if (pgno >= mp->npages) {
 192         errno = EINVAL;
 193         return (NULL);
 194     }
 195 
 196 #ifdef STATISTICS
 197     ++mp->pageget;
 198 #endif
 199 
 200     /* Check for a page that is cached. */
 201     if ((bp = mpool_look(mp, pgno)) != NULL) {
 202 #ifdef DEBUG
 203         if (bp->flags & MPOOL_PINNED) {
 204             (void)fprintf(stderr,
 205                 "mpool_get: page %d already pinned\n", bp->pgno);
 206             abort();
 207         }
 208 #endif
 209         /*
 210          * Move the page to the head of the hash chain and the tail
 211          * of the lru chain.
 212          */
 213         head = &mp->hqh[HASHKEY(bp->pgno)];
 214         CIRCLEQ_REMOVE(head, bp, hq);
 215         CIRCLEQ_INSERT_HEAD(head, bp, hq);
 216         CIRCLEQ_REMOVE(&mp->lqh, bp, q);
 217         CIRCLEQ_INSERT_TAIL(&mp->lqh, bp, q);
 218 
 219         /* Return a pinned page. */
 220         bp->flags |= MPOOL_PINNED;
 221         return (bp->page);
 222     }
 223 
 224     /* Get a page from the cache. */
 225     if ((bp = mpool_bkt(mp)) == NULL)
 226         return (NULL);
 227 
 228     /* Read in the contents. */
 229 #ifdef STATISTICS
 230     ++mp->pageread;
 231 #endif
 232 
 233     /*
 234      * If both of `off_t' and `long' are 32 bits, the right operand
 235      * of the multiplication is converted to `unsigned long',
 236      * and the multiplication is done with unsigned 32 bits.
 237      * It is equivalent to the case without cast.
 238      *
 239      * If `off_t' is 64 bits and `long' is 32 bits, the left operand
 240      * of the multiplication is converted to `off_t',
 241      * and the multiplication is done with signed 64 bits.
 242      * Adding cast avoids integer overflow.
 243      *
 244      * If both of `off_t' and `long' are 64 bits, the right operand
 245      * of the multiplication is converted to `unsigned long',
 246      * and the multiplication is done with unsigned 64 bits.
 247      * It is equivalent to the case without cast.
 248      */
 249     off = mp->pagesize * (off_t)pgno;
 250 
 251 #ifdef HAVE_PREAD
 252     if ((nr = pread(mp->fd, bp->page, mp->pagesize, off)) != mp->pagesize) {
 253         if (nr >= 0)
 254             errno = EFTYPE;
 255         return (NULL);
 256     }
 257 #else
 258     if (lseek(mp->fd, off, SEEK_SET) != off)
 259         return (NULL);
 260     if ((nr = read(mp->fd, bp->page, mp->pagesize)) != mp->pagesize) {
 261         if (nr >= 0)
 262             errno = EFTYPE;
 263         return (NULL);
 264     }
 265 #endif
 266 
 267     /* Set the page number, pin the page. */
 268     bp->pgno = pgno;
 269     bp->flags = MPOOL_PINNED;
 270 
 271     /*
 272      * Add the page to the head of the hash chain and the tail
 273      * of the lru chain.
 274      */
 275     head = &mp->hqh[HASHKEY(bp->pgno)];
 276     CIRCLEQ_INSERT_HEAD(head, bp, hq);
 277     CIRCLEQ_INSERT_TAIL(&mp->lqh, bp, q);
 278 
 279     /* Run through the user's filter. */
 280     if (mp->pgin != NULL)
 281         (mp->pgin)(mp->pgcookie, bp->pgno, bp->page);
 282 
 283     return (bp->page);
 284 }

 286 /**
 287  * mpool_put
 288  *  Return a page.
 289  *
 290  *  @param mp
 291  *  @param page
 292  *  @param flags
 293  */
 294 int
 295 mpool_put(mp, page, flags)
 296     MPOOL *mp;
 297     void *page;
 298     u_int flags;
 299 {
 300     BKT *bp;
 301 
 302 #ifdef STATISTICS
 303     ++mp->pageput;
 304 #endif
 305     bp = (BKT *)((char *)page - sizeof(BKT));
 306 #ifdef DEBUG
 307     if (!(bp->flags & MPOOL_PINNED)) {
 308         (void)fprintf(stderr,
 309             "mpool_put: page %d not pinned\n", bp->pgno);
 310         abort();
 311     }
 312 #endif
 313     bp->flags &= ~MPOOL_PINNED;
 314     bp->flags |= flags & MPOOL_DIRTY;
 315     return (RET_SUCCESS);
 316 }

 318 /**
 319  * mpool_close
 320  *  Close the buffer pool.
 321  *
 322  *  @param mp
 323  */
 324 int
 325 mpool_close(mp)
 326     MPOOL *mp;
 327 {
 328     BKT *bp;
 329 
 330     /* Free up any space allocated to the lru pages. */
 331     while ((bp = mp->lqh.cqh_first) != (void *)&mp->lqh) {
 332         CIRCLEQ_REMOVE(&mp->lqh, mp->lqh.cqh_first, q);
 333         free(bp);
 334     }
 335 
 336     /* Free the MPOOL cookie. */
 337     free(mp);
 338     return (RET_SUCCESS);
 339 }

 341 /**
 342  * mpool_sync
 343  *  Sync the pool to disk.
 344  *
 345  *  @param mp
 346  */
 347 int
 348 mpool_sync(mp)
 349     MPOOL *mp;
 350 {
 351     BKT *bp;
 352 
 353     /* Walk the lru chain, flushing any dirty pages to disk. */
 354     for (bp = mp->lqh.cqh_first;
 355         bp != (void *)&mp->lqh; bp = bp->q.cqe_next)
 356         if (bp->flags & MPOOL_DIRTY &&
 357             mpool_write(mp, bp) == RET_ERROR)
 358             return (RET_ERROR);
 359 
 360     /* Sync the file descriptor. */
 361     return (fsync(mp->fd) ? RET_ERROR : RET_SUCCESS);
 362 }

 364 /**
 365  * mpool_bkt
 366  *  Get a page from the cache (or create one).
 367  *
 368  *  @param mp
 369  */
 370 static BKT *
 371 mpool_bkt(mp)
 372     MPOOL *mp;
 373 {
 374     struct _hqh *head;
 375     BKT *bp;
 376 
 377     /* If under the max cached, always create a new page. */
 378     if (mp->curcache < mp->maxcache)
 379         goto new;
 380 
 381     /*
 382      * If the cache is max'd out, walk the lru list for a buffer we
 383      * can flush.  If we find one, write it (if necessary) and take it
 384      * off any lists.  If we don't find anything we grow the cache anyway.
 385      * The cache never shrinks.
 386      */
 387     for (bp = mp->lqh.cqh_first;
 388         bp != (void *)&mp->lqh; bp = bp->q.cqe_next)
 389         if (!(bp->flags & MPOOL_PINNED)) {
 390             /* Flush if dirty. */
 391             if (bp->flags & MPOOL_DIRTY &&
 392                 mpool_write(mp, bp) == RET_ERROR)
 393                 return (NULL);
 394 #ifdef STATISTICS
 395             ++mp->pageflush;
 396 #endif
 397             /* Remove from the hash and lru queues. */
 398             head = &mp->hqh[HASHKEY(bp->pgno)];
 399             CIRCLEQ_REMOVE(head, bp, hq);
 400             CIRCLEQ_REMOVE(&mp->lqh, bp, q);
 401 #ifdef DEBUG
 402             { void *spage;
 403                 spage = bp->page;
 404                 memset(bp, 0xff, sizeof(BKT) + mp->pagesize);
 405                 bp->page = spage;
 406             }
 407 #endif
 408             return (bp);
 409         }
 410 
 411 new:    if ((bp = (BKT *)malloc(sizeof(BKT) + mp->pagesize)) == NULL)
 412         return (NULL);
 413 #ifdef STATISTICS
 414     ++mp->pagealloc;
 415 #endif
 416 #if defined(DEBUG) || defined(PURIFY)
 417     memset(bp, 0xff, sizeof(BKT) + mp->pagesize);
 418 #endif
 419     bp->page = (char *)bp + sizeof(BKT);
 420     ++mp->curcache;
 421     return (bp);
 422 }

 424 /**
 425  * mpool_write
 426  *  Write a page to disk.
 427  *
 428  *  @param mp
 429  *  @param bp
 430  */
 431 static int
 432 mpool_write(mp, bp)
 433     MPOOL *mp;
 434     BKT *bp;
 435 {
 436     off_t off;
 437 
 438 #ifdef STATISTICS
 439     ++mp->pagewrite;
 440 #endif
 441 
 442     /* Run through the user's filter. */
 443     if (mp->pgout)
 444         (mp->pgout)(mp->pgcookie, bp->pgno, bp->page);
 445 
 446     /* See the comment in mpool_get for cast addition. */
 447     off = mp->pagesize * (off_t)bp->pgno;
 448 
 449 #ifdef HAVE_PWRITE
 450     if (pwrite(mp->fd, bp->page, mp->pagesize, off) != mp->pagesize)
 451         return (RET_ERROR);
 452 #else
 453     if (lseek(mp->fd, off, SEEK_SET) != off)
 454         return (RET_ERROR);
 455     if (write(mp->fd, bp->page, mp->pagesize) != mp->pagesize)
 456         return (RET_ERROR);
 457 #endif
 458 
 459     bp->flags &= ~MPOOL_DIRTY;
 460     return (RET_SUCCESS);
 461 }

 463 /**
 464  * mpool_look
 465  *  Lookup a page in the cache.
 466  *
 467  *  @param mp
 468  *  @param pgno
 469  */
 470 static BKT *
 471 mpool_look(mp, pgno)
 472     MPOOL *mp;
 473     pgno_t pgno;
 474 {
 475     struct _hqh *head;
 476     BKT *bp;
 477 
 478     head = &mp->hqh[HASHKEY(pgno)];
 479     for (bp = head->cqh_first; bp != (void *)head; bp = bp->hq.cqe_next)
 480         if (bp->pgno == pgno) {
 481 #ifdef STATISTICS
 482             ++mp->cachehit;
 483 #endif
 484             return (bp);
 485         }
 486 #ifdef STATISTICS
 487     ++mp->cachemiss;
 488 #endif
 489     return (NULL);
 490 }

 493 /**
 494  * mpool_stat
 495  *  Print out cache statistics.
 496  *
 497  *  @param mp
 498  */
 499 void
 500 mpool_stat(mp)
 501     MPOOL *mp;
 502 {
 503     BKT *bp;
 504     int cnt;
 505     char *sep;
 506 
 507     (void)fprintf(stderr, "%lu pages in the file\n", (long unsigned int)mp->npages);
 508     (void)fprintf(stderr,
 509         "page size %lu, cacheing %lu pages of %lu page max cache\n",
 510         mp->pagesize, (long unsigned int)mp->curcache, (long unsigned int)mp->maxcache);
 511     (void)fprintf(stderr, "%lu page puts, %lu page gets, %lu page new\n",
 512         mp->pageput, mp->pageget, mp->pagenew);
 513     (void)fprintf(stderr, "%lu page allocs, %lu page flushes\n",
 514         mp->pagealloc, mp->pageflush);
 515     if (mp->cachehit + mp->cachemiss)
 516         (void)fprintf(stderr,
 517             "%.0f%% cache hit rate (%lu hits, %lu misses)\n", 
 518             ((double)mp->cachehit / (mp->cachehit + mp->cachemiss))
 519             * 100, mp->cachehit, mp->cachemiss);
 520     (void)fprintf(stderr, "%lu page reads, %lu page writes\n",
 521         mp->pageread, mp->pagewrite);
 522 
 523     sep = "";
 524     cnt = 0;
 525     for (bp = mp->lqh.cqh_first;
 526         bp != (void *)&mp->lqh; bp = bp->q.cqe_next) {
 527         (void)fprintf(stderr, "%s%d", sep, bp->pgno);
 528         if (bp->flags & MPOOL_DIRTY)
 529             (void)fprintf(stderr, "d");
 530         if (bp->flags & MPOOL_PINNED)
 531             (void)fprintf(stderr, "P");
 532         if (++cnt == 10) {
 533             sep = "\n";
 534             cnt = 0;
 535         } else
 536             sep = ", ";
 537             
 538     }
 539     (void)fprintf(stderr, "\n");
 540 }
