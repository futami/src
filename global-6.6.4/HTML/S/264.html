  59 NAME
  60      abs2rel - make a relative path name from an absolute path
  61 
  62 SYNOPSIS
  63      char *
  64      abs2rel(const char *path, const char *base, char *result, size_t size)
  65 
  66 DESCRIPTION
  67      The abs2rel() function makes a relative path name from an absolute path
  68      name path based on a directory base and copies the resulting path name
  69      into the memory referenced by result.  The result argument must refer to
  70      a buffer capable of storing at least size characters.
  71 
  72      The resulting path name may include symbolic links.  The abs2rel() func-
  73      tion doesn't check whether or not any path exists.
  74 
  75 RETURN VALUES
  76      The abs2rel() function returns relative path name on success.  If an er-
  77      ror occurs, it returns NULL.
  78 
  79 ERRORS
  80      The abs2rel() function may fail and set the external variable errno to
  81      indicate the error.
  82 
  83      [EINVAL]           The base directory isn't an absolute path name or the
  84                         size argument is zero.
  85 
  86      [ERANGE]           The size argument is greater than zero but smaller
  87                         than the length of the pathname plus 1.
  88 
  89 EXAMPLE
  90          char result[MAXPATHLEN];
  91          char *path = abs2rel("/usr/src/sys", "/usr/local/lib", result, MAX-
  92      PATHLEN);
  93 
  94      yields:
  95 
  96          path == "../../src/sys"
  97 
  98      Similarly,
  99 
 100          path1 = abs2rel("/usr/src/sys", "/usr", result, MAXPATHLEN);
 101          path2 = abs2rel("/usr/src/sys", "/usr/src/sys", result, MAXPATHLEN);
 102 
 103      yields:
 104 
 105          path1 == "src/sys"
 106          path2 == "."
 107 
 108 
 109 BUGS
 110      If the base directory includes symbolic links, the abs2rel() function
 111      produces the wrong path.  For example, if '/sys' is a symbolic link to
 112      '/usr/src/sys',
 113 
 114          char *path = abs2rel("/usr/local/lib", "/sys", result, MAXPATHLEN);
 115 
 116      yields:
 117 
 118          path == "../usr/local/lib"         -- It's wrong!!
 119 
 120      You should convert the base directory into a real path in advance.
 121 
 122          path = abs2rel("/sys/kern", realpath("/sys", resolvedname), result,
 123      MAXPATHLEN);
 124 
 125      yields:
 126 
 127          path == "../../../sys/kern"        -- It's correct but ...
 128 
 129      That is correct, but a little redundant. If you wish get the simple an-
 130      swer 'kern', do the following.
 131 
 132          path = abs2rel(realpath("/sys/kern", r1), realpath("/sys", r2),
 133                                              result, MAXPATHLEN);
 134 
 135      The realpath() function assures correct result, but don't forget that
 136      realpath() requires that all but the last component of the path exist.

 189 /**
 190  * normalize: normalize path name
 191  *
 192  *  @param[in]  path    path name
 193  *  @param[in]  root    root of project (must be end with a '/')
 194  *  @param[in]  cwd current directory
 195  *  @param[out] result  normalized path name
 196  *  @param[in]  size    size of the result
 197  *  @return     ==NULL: error,
 198  *          !=NULL: result
 199  *
 200  *  [Note] Calls die() if the result path name is too long (MAXPATHLEN).
 201  */
 202 char *
 203 normalize(const char *path, const char *root, const char *cwd, char *result, int size)
 204 {
 205     char *p, abs[MAXPATHLEN];
 206 
 207     if (normalize_pathname(path, result, size) == NULL)
 208         goto toolong;
 209     if (isabspath(path)) {
 210         if (strlen(result) > MAXPATHLEN)
 211             goto toolong;
 212         strcpy(abs, result);
 213     } else {
 214         if (rel2abs(result, cwd, abs, sizeof(abs)) == NULL)
 215             goto toolong;
 216     }
 217     /*
 218      * Remove the root part of path and insert './'.
 219      *      rootdir  /a/b/
 220      *      path     /a/b/c/d.c -> c/d.c -> ./c/d.c
 221      */
 222     p = locatestring(abs, root, LOCATEFLAG);
 223     if (p == NULL) {
 224         p = locatestring(root, abs, LOCATEFLAG);
 225         /*
 226          * abs == /usr/src should be considered to be equal to root == /usr/src/.
 227          */
 228         if (p && !strcmp(p, "/"))
 229             result[0] = '\0';
 230         else
 231             return NULL;
 232     }
 233     strlimcpy(result, "./", size);
 234     strlimcpy(result + 2, p, size - 2);
 235     return result;
 236 toolong:
 237     die("path name is too long.");
 238 }
 239 /**
 240  * normalize_pathname: normalize relative path name.
 241  *
 242  *  @param[in]  path    relative path name
 243  *  @param[out] result  result buffer
 244  *  @param[in]  size    size of result buffer
 245  *  @return     != NULL: normalized path name,
 246  *          == NULL: error (ERANGE)
 247  *
 248  * Examples:
 249  * path         result
 250  * ---------------------------
 251  * /a           /a
 252  * ./a/./b/c        a/b/c
 253  * a////b///c       a/b/c
 254  * ../a/b/c     ../a/b/c
 255  * a/d/../b/c       a/b/c
 256  * a/../b/../c/../d d
 257  * a/../../d        ../d
 258  * /a/../../d       /d
 259  */
 260 char *
 261 normalize_pathname(const char *path, char *result, int size)
 262 {
 263     const char *savep, *p = path;
 264     char *final, *q = result;
 265     char *endp = result + size - 1;
 266 #if defined(_WIN32) || defined(__DJGPP__)
 267     char *spath = check_strdup(path), *pp = spath;
 268     while (*pp) {
 269         if (*pp == '\\')
 270             *pp = '/';
 271         ++pp;
 272     }
 273     p = spath;
 274 #endif
 275 
 276     /* accept the first '/' */
 277     if (isabspath(p)) {
 278         *q++ = *p++;
 279 #if defined(_WIN32) || defined(__DJGPP__)
 280         if (*p == ':') {
 281             *q++ = *p++;
 282             *q++ = *p++;
 283         }
 284 #endif
 285         final = q;
 286     }
 287     do {
 288         savep = p;
 289         while (!strncmp(p, "./", 2))    /* skip "./" at the head of the path */
 290             p += 2;
 291         while (!strncmp(p, "../", 3)) { /* accept the first "../" */
 292             if (q + 3 > endp)
 293                 goto erange;
 294             strcpy(q, "../");
 295             p += 3;
 296             q += 3;
 297         }
 298     } while (savep != p);
 299 
 300     final = q;
 301     while (*p) {
 302         if (*p == '/') {
 303             p++;
 304             do {
 305                 savep = p;
 306                 /* skip consecutive '/' */
 307                 while (*p == '/')   
 308                     p++;
 309                 /* skip consecutive './' */
 310                 while (!strncmp(p, "./", 2))
 311                     p += 2;
 312                 /* resolve '../'(parent directory) */
 313                 while (!strncmp(p, "../", 3)) {
 314                     p += 3;
 315                     if (q > final) {
 316                         while (q > final && *--q != '/')
 317                             ;
 318                     } else if (!(*result == '/' && result + 1 == q)) {
 319                         if (q + 3 > endp)
 320                             goto erange;
 321                         strcpy(q, "../");
 322                         q += 3;
 323                         final = q;
 324                     }
 325                 }
 326             } while (savep != p);
 327             if (q > endp)
 328                 goto erange;
 329             if (q > final) {
 330                 *q++ = '/';
 331             }
 332         } else {
 333             if (q > endp)
 334                 goto erange;
 335             *q++ = *p++;
 336         }
 337     }
 338     *q = '\0';
 339 #if defined(_WIN32) || defined(__DJGPP__)
 340     free(spath);
 341 #endif
 342     return result;
 343 erange:
 344     errno = ERANGE;
 345 #if defined(_WIN32) || defined(__DJGPP__)
 346     free(spath);
 347 #endif
 348     return NULL;
 349 }
 350 /**
 351  * abs2rel: convert an absolute path name into relative.
 352  *
 353  *  @param[in]  path    absolute path
 354  *  @param[in]  base    base directory (must be absolute path)
 355  *  @param[out] result  result buffer
 356  *  @param[in]  size    size of result buffer
 357  *  @return     != NULL: relative path,
 358  *          == NULL: error (ERANGE or EINVAL)
 359  */
 360 char *
 361 abs2rel(const char *path, const char *base, char *result, int size)
 362 {
 363     const char *pp, *bp, *branch, *branch_b;
 364     /*
 365      * endp points the last position which is safe in the result buffer.
 366      */
 367     const char *endp = result + size - 1;
 368     char *rp;
 369 #ifdef COLOR_PATH
 370     int escape_count = 0;
 371     const char *first_escape = NULL;
 372 #endif
 373 
 374     if (!isabspath(path)) {
 375         if (strlen(path) >= size)
 376             goto erange;
 377         strcpy(result, path);
 378         goto finish;
 379     } else if (!isabspath(base) || !size) {
 380         errno = EINVAL;
 381         return (NULL);
 382     } else if (size == 1)
 383         goto erange;
 384     /*
 385      * seek to branched point.
 386      */
 387     branch = path;
 388     branch_b = base;
 389     for (pp = path, bp = base; *pp && *bp; pp++, bp++) {
 390 #ifdef COLOR_PATH
 391         /* skip escape sequence */
 392         if (*pp == ESC) {
 393             escape_count++;
 394             if (first_escape == NULL)
 395                 first_escape = pp;
 396             while (*pp && *pp != EOE)
 397                 pp++;
 398             if (*pp == EOE)
 399                 pp++;
 400             else
 401                 die("invalid escape sequence in the path. '%s'", path);
 402         }
 403 #endif
 404         if (PATHCHAR(*pp) != PATHCHAR(*bp))
 405             break;
 406         if (*pp == '/') {
 407             branch = pp;
 408             branch_b = bp;
 409         }
 410     }
 411     if ((*pp == 0 || (*pp == '/' && *(pp + 1) == 0)) &&
 412         (*bp == 0 || (*bp == '/' && *(bp + 1) == 0))) {
 413         rp = result;
 414         *rp++ = '.';
 415         if (*pp == '/' || *(pp - 1) == '/')
 416             *rp++ = '/';
 417         if (rp > endp)
 418             goto erange;
 419         *rp = 0;
 420         goto finish;
 421     }
 422 #ifdef COLOR_PATH
 423     /* skip escape sequence */
 424     if (*pp == ESC) {
 425         escape_count++;
 426         if (first_escape == NULL)
 427             first_escape = pp;
 428         while (*pp && *pp != EOE)
 429             pp++;
 430         if (*pp == EOE)
 431             pp++;
 432         else
 433             die("invalid escape sequence in the path. '%s'", path);
 434     }
 435 #endif
 436     if ((*pp == 0 && *bp == '/') || (*pp == '/' && *bp == 0)) {
 437         branch = pp;
 438         branch_b = bp;
 439     }
 440     /*
 441      * up to root.
 442      */
 443     rp = result;
 444     for (bp = branch_b; *bp; bp++)
 445         if (*bp == '/' && *(bp + 1) != 0) {
 446             if (rp + 3 > endp)
 447                 goto erange;
 448             *rp++ = '.';
 449             *rp++ = '.';
 450             *rp++ = '/';
 451         }
 452     if (rp > endp)
 453         goto erange;
 454     *rp = 0;
 455     /*
 456      * down to leaf.
 457      */
 458     if (*branch) {
 459 #ifdef COLOR_PATH
 460         /*
 461          * Insert the first escape sequence here.
 462          * It is a case where a colored part is divided.
 463          */
 464         if (escape_count == 1) {
 465             const char *p = first_escape;
 466             while (*p && *p != EOE) {
 467                 if (rp > endp)
 468                     goto erange;
 469                 *rp++ = *p++;
 470             }
 471             *rp++ = *p;
 472         }
 473 #endif
 474         if (rp + strlen(branch + 1) > endp)
 475             goto erange;
 476         strcpy(rp, branch + 1);
 477     } else
 478         *--rp = 0;
 479 finish:
 480     return result;
 481 erange:
 482     errno = ERANGE;
 483     return (NULL);
 484 }
 485 /**
 486  * rel2abs: convert an relative path name into absolute.
 487  *
 488  *  @param[in]  path    relative path
 489  *  @param[in]  base    base directory (must be absolute path)
 490  *  @param[out] result  result buffer
 491  *  @param[in]  size    size of result buffer
 492  *  @return     != NULL: absolute path,
 493  *          == NULL: error (ERANGE or EINVAL)
 494  */
 495 char *
 496 rel2abs(const char *path, const char *base, char *result, int size)
 497 {
 498     const char *pp, *bp;
 499     /*
 500      * endp points the last position which is safe in the result buffer.
 501      */
 502     const char *endp = result + size - 1;
 503     char *rp;
 504     int length;
 505 
 506     if (isabspath(path)) {
 507         if (strlen(path) >= size)
 508             goto erange;
 509         strcpy(result, path);
 510         goto finish;
 511     } else if (!isabspath(base) || !size) {
 512         errno = EINVAL;
 513         return (NULL);
 514     } else if (size == 1)
 515         goto erange;
 516 
 517     length = strlen(base);
 518 
 519     if (!strcmp(path, ".") || !strcmp(path, "./")) {
 520         if (length >= size)
 521             goto erange;
 522         strcpy(result, base);
 523         /*
 524          * rp points the last char.
 525          */
 526         rp = result + length - 1;
 527         /*
 528          * remove the last '/'.
 529          */
 530         if (*rp == '/') {
 531             if (length > 1)
 532                 *rp = 0;
 533         } else
 534             rp++;
 535         /* rp point NULL char */
 536         if (*++path == '/') {
 537             /*
 538              * Append '/' to the tail of path name.
 539              */
 540             *rp++ = '/';
 541             if (rp > endp)
 542                 goto erange;
 543             *rp = 0;
 544         }
 545         goto finish;
 546     }
 547     bp = base + length;
 548     if (*(bp - 1) == '/')
 549         --bp;
 550     /*
 551      * up to root.
 552      */
 553     for (pp = path; *pp && *pp == '.'; ) {
 554         if (!strncmp(pp, "../", 3)) {
 555             pp += 3;
 556             while (bp > base && *--bp != '/')
 557                 ;
 558         } else if (!strncmp(pp, "./", 2)) {
 559             pp += 2;
 560         } else if (!strncmp(pp, "..\0", 3)) {
 561             pp += 2;
 562             while (bp > base && *--bp != '/')
 563                 ;
 564         } else
 565             break;
 566     }
 567     /*
 568      * down to leaf.
 569      */
 570     length = bp - base;
 571     if (length >= size)
 572         goto erange;
 573     strncpy(result, base, length);
 574     rp = result + length;
 575     if (*pp || *(pp - 1) == '/' || length == 0)
 576         *rp++ = '/';
 577     if (rp + strlen(pp) > endp)
 578         goto erange;
 579     strcpy(rp, pp);
 580 finish:
 581     return result;
 582 erange:
 583     errno = ERANGE;
 584     return (NULL);
 585 }
