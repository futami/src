  26 #ifndef PLACEFILE_GERBER_WRITER_H
  27 #define PLACEFILE_GERBER_WRITER_H
  28 
  29 #include <layers_id_colors_and_visibility.h>
  30 
  31 class BOARD;
  32 class MODULE;
  33 class D_PAD;
  34 
  35 /**

  36  * PLACEFILE_GERBER_WRITER is a class mainly used to create Gerber drill files

  37  */
  38 class PLACEFILE_GERBER_WRITER
  39 {
  40 public:
  41     PLACEFILE_GERBER_WRITER( BOARD* aPcb );
  42 
  43     virtual ~PLACEFILE_GERBER_WRITER()
  44     {
  45     }
  46 
  47 
  48     /**

  49      * Function SetOptions

  50      * Initialize internal parameters to match drill options

  51      * note: PTH and NPTH are always separate files in Gerber format

  52      * @param aOffset = drill coordinates offset

  53      */
  54     void SetOptions( wxPoint aOffset )
  55     {
  56         m_offset = aOffset;
  57     }
  58 
  59     /**

  60      * Creates an pnp gerber file

  61      * @param aFullFilename = the full filename

  62      * @param aLayer = layer (F_Cu or B_Cu) to generate

  63      * @param aIncludeBrdEdges = true to include board outlines

  64      * @return component count, or -1 if the file cannot be created

  65      */
  66     int  CreatePlaceFile( wxString& aFullFilename, PCB_LAYER_ID aLayer,
  67                           bool aIncludeBrdEdges );
  68 
  69     /**

  70      * @return a filename which identify the drill file function.

  71      * @param aFullBaseFilename = a full filename. it will be modified

  72      * to add "-pnp" and set the extension

  73      * @param aLayer = layer (F_Cu or B_Cu) to generate

  74      */
  75     const wxString GetPlaceFileName( const wxString& aFullBaseFilename,
  76                                      PCB_LAYER_ID aLayer ) const;
  77 
  78 private:
  79     BOARD*          m_pcb;
  80     /// The board layer currently used (typically F_Cu or B_Cu)

  81     PCB_LAYER_ID    m_layer;
  82     double          m_conversionUnits;  // scaling factor to convert the board unites to

  83                                         // Excellon/Gerber units (i.e inches or mm)

  84     wxPoint         m_offset;           // Drill offset coordinates

  85     bool            m_forceSmdItems;
  86     // True to plot a flashed marker shape at pad 1 position

  87     bool            m_plotPad1Marker;
  88     // True to plot a marker shape at other pads position

  89     // This is a flashed 0 sized round pad

  90     bool            m_plotOtherPadsMarker;
  91 
  92 
  93     /** convert a kicad footprint orientation to gerber rotation

  94      *  both are in degrees

  95      */
  96     double      mapRotationAngle( double aAngle );
  97 
  98     /** Find the pad(s) 1 (or pad "A1") of a footprint

  99      * Usefull to plot a marker at this (these) position(s)

 100      * @param aPadList is the list to fill

 101      * @param aFootprint is the footprint to test

 102      */
 103     void findPads1( std::vector<D_PAD*>& aPadList, MODULE* aFootprint ) const;
 104 };

 106 #endif  //  #ifndef PLACEFILE_GERBER_WRITER_H
