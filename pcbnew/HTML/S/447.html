  48 /**

  49  * FOOTPRINT_WIZARD

  50  * This is the parent class from where any footprint wizard class must

  51  * derive */
  52 class FOOTPRINT_WIZARD
  53 {
  54 public:
  55     FOOTPRINT_WIZARD() {}
  56     virtual ~FOOTPRINT_WIZARD();
  57 
  58     /**

  59      * Function GetName

  60      * @return the name of the wizard

  61      */
  62     virtual wxString        GetName() = 0;
  63 
  64     /**

  65      * Function GetImage

  66      * @return an svg image of the wizard to be rendered

  67      */
  68     virtual wxString        GetImage() = 0;
  69 
  70     /**

  71      * Function GetDescription

  72      * @return a description of the footprint wizard

  73      */
  74     virtual wxString        GetDescription() = 0;
  75 
  76     /**

  77      * Function GetNumParameterPages

  78      * @return the number of parameter pages that this wizard will show to the user

  79      */
  80     virtual int             GetNumParameterPages() = 0;
  81 
  82     /**

  83      * Function GetParameterPageName

  84      * @param aPage is the page we want the name of

  85      * @return a string with the page name

  86      */
  87     virtual wxString        GetParameterPageName( int aPage ) = 0;
  88 
  89     /**

  90      * Function GetParameterNames

  91      * @param aPage is the page we want the parameter names of

  92      * @return an array string with the parameter names on a certain page

  93      */
  94     virtual wxArrayString   GetParameterNames( int aPage ) = 0;
  95 
  96     /**

  97      * Function GetParameterTypes

  98      * @param aPage is the page we want the parameter types of

  99      * @return an array string with the parameter types on a certain page

 100      *          "IU" for internal units, "UNITS" for units (0,1,2,3...,N)

 101      */
 102     virtual wxArrayString   GetParameterTypes( int aPage ) = 0;
 103 
 104 
 105     /**

 106      * Function GetParameterValues

 107      * @param aPage is the page we want the parameter values of

 108      * @return an array of parameter values

 109      */
 110     virtual wxArrayString   GetParameterValues( int aPage ) = 0;
 111 
 112     /**

 113      * Function GetParameterErrors

 114      * @param aPage is the page we want to know the errors of

 115      * @return an array of errors (if any) for the parameters, empty strings for OK parameters

 116      */
 117     virtual wxArrayString   GetParameterErrors( int aPage ) = 0;
 118 
 119     /**

 120      * Function GetParameterHints

 121      * @param aPage is the page we want to know the hints of

 122      * @return an array of hints (if any) for the parameters, empty string for no hints

 123      */
 124     virtual wxArrayString   GetParameterHints( int aPage ) = 0;
 125 
 126     /**

 127      * Function GetParamaterDesignators

 128      * @param aPage is the page we want to know the designators of

 129      * @return an array of designators (blank strings for no designators

 130      */
 131     virtual wxArrayString    GetParameterDesignators( int aPage ) = 0;
 132 
 133     /**

 134      * Function SetParameterValues

 135      * @param aPage is the page we want to set the parameters in

 136      * @param aValues are the values we want to set into the parameters

 137      * @return an array of parameter values

 138      */
 139     virtual wxString        SetParameterValues( int aPage, wxArrayString& aValues ) = 0;
 140 
 141     /**

 142      * Function ResetParameters

 143      * Reset all wizard parameters to default values

 144      */
 145     virtual void ResetParameters() = 0;
 146 
 147     /**

 148      * Function GetModule

 149      * This method builds the module itself and returns it to the caller function

 150      * @return  PCB module built from the parameters given to the class

 151      * @param aMessage a wxString to store messages (if any) generated by the

 152      * footprint generator

 153      */
 154     virtual MODULE*         GetFootprint( wxString* aMessage ) = 0;
 155 
 156     /**

 157      * Function GetObject

 158      * This method gets the pointer to the object from where this wizard constructs

 159      * @return  it's a void pointer, as it could be a PyObject or any other

 160      */
 161     virtual void*           GetObject() = 0;
 162 
 163     /**

 164      * Function register_wizard

 165      * It's the standard method of a "FOOTPRINT_WIZARD" to register itself into

 166      * the FOOTPRINT_WIZARD_LIST singleton manager

 167      */
 168     void                    register_wizard();
 169 };
 170 
 171 
 172 class FOOTPRINT_WIZARD_LIST
 173 {
 174 private:
 175     /**

 176      * FOOTPRINT_WIZARD system wide static list

 177      */
 178     static std::vector<FOOTPRINT_WIZARD*> m_FootprintWizards;
 179 
 180 public:
 181 
 182     /**

 183      * Function register_wizard

 184      * A footprint wizard calls this static method when it wants to register itself

 185      * into the system wizards

 186      * Note: if it is already registered, this function do nothing

 187      * if n existing wizard with the same name exists, this existing wizard will be

 188      * unregistered.

 189      * @param aWizard is the footprint wizard to be registered

 190      */
 191     static void                 register_wizard( FOOTPRINT_WIZARD* aWizard );
 192 
 193     /**

 194      * Function deregister_object

 195      * Anyone calls this method to deregister an object which builds a wizard,

 196      * it will lookup on the vector calling GetObject until find, then removed

 197      * and deleted

 198      *

 199      * @param aObject is the footprint wizard object to be deregistered

 200      */
 201     static bool                 deregister_object( void* aObject );
 202 
 203     /**

 204      * Function GetWizard

 205      * @param aName is the footprint wizard name

 206      * @return a wizard object by it's name or NULL if it isn't available.

 207      */
 208     static FOOTPRINT_WIZARD*    GetWizard( const wxString& aName );
 209 
 210     /**

 211      * Function GetWizard

 212      * @return a wizard object by it's number or NULL if it isn't available.

 213      * @param  aIndex is the wizard index in list

 214      */
 215     static FOOTPRINT_WIZARD*    GetWizard( int aIndex );
 216 
 217     /**

 218      * Function GetWizardsCount

 219      * @return the number of wizards available into the system

 220      */
 221     static int                  GetWizardsCount();
 222 };
