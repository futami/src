  48 /**

  49  * LINE_PLACER

  50  *

  51  * Single track placement algorithm. Interactively routes a track.

  52  * Applies shove and walkaround algorithms when needed.

  53  */
  54 
  55 class DIFF_PAIR_PLACER : public PLACEMENT_ALGO
  56 {
  57 public:
  58     DIFF_PAIR_PLACER( ROUTER* aRouter );
  59     ~DIFF_PAIR_PLACER();
  60 
  61     /**

  62      * Function Start()

  63      *

  64      * Starts routing a single track at point aP, taking item aStartItem as anchor

  65      * (unless NULL).

  66      */
  67     bool Start( const VECTOR2I& aP, ITEM* aStartItem ) override;
  68 
  69     /**

  70      * Function Move()

  71      *

  72      * Moves the end of the currently routed trace to the point aP, taking

  73      * aEndItem as anchor (if not NULL).

  74      * (unless NULL).

  75      */
  76     bool Move( const VECTOR2I& aP, ITEM* aEndItem ) override;
  77 
  78     /**

  79      * Function FixRoute()

  80      *

  81      * Commits the currently routed track to the parent node, taking

  82      * aP as the final end point and aEndItem as the final anchor (if provided).

  83      * @return true, if route has been commited. May return false if the routing

  84      * result is violating design rules - in such case, the track is only committed

  85      * if Settings.CanViolateDRC() is on.

  86      */
  87     bool FixRoute( const VECTOR2I& aP, ITEM* aEndItem, bool aForceFinish ) override;
  88 
  89     /// @copydoc PLACEMENT_ALGO::CommitPlacement()

  90     bool CommitPlacement() override;
  91 
  92     /// @copydoc PLACEMENT_ALGO::AbortPlacement()

  93     bool AbortPlacement() override;
  94 
  95     /// @copydoc PLACEMENT_ALGO::HasPlacedAnything()

  96     bool HasPlacedAnything() const override;
  97 
  98     /**

  99      * Function ToggleVia()

 100      *

 101      * Enables/disables a via at the end of currently routed trace.

 102      */
 103     bool ToggleVia( bool aEnabled ) override;
 104 
 105     /**

 106      * Function SetLayer()

 107      *

 108      * Sets the current routing layer.

 109      */
 110     bool SetLayer( int aLayer ) override;
 111 
 112     /**

 113      * Function Traces()

 114      *

 115      * Returns the complete routed line, as a single-member ITEM_SET.

 116      */
 117     const ITEM_SET Traces() override;
 118 
 119     /**

 120      * Function CurrentEnd()

 121      *

 122      * Returns the current end of the line being placed. It may not be equal

 123      * to the cursor position due to collisions.

 124      */
 125     const VECTOR2I& CurrentEnd() const override
 126     {
 127         return m_currentEnd;
 128     }
 129 
 130     /**

 131      * Function CurrentNets()

 132      *

 133      * Returns the net code of currently routed track.

 134      */
 135     const std::vector<int> CurrentNets() const override;
 136 
 137     /**

 138      * Function CurrentLayer()

 139      *

 140      * Returns the layer of currently routed track.

 141      */
 142     int CurrentLayer() const override
 143     {
 144         return m_currentLayer;
 145     }
 146 
 147     /**

 148      * Function CurrentNode()

 149      *

 150      * Returns the most recent world state.

 151      */
 152     NODE* CurrentNode( bool aLoopsRemoved = false ) const override;
 153 
 154     /**

 155      * Function FlipPosture()

 156      *

 157      * Toggles the current posture (straight/diagonal) of the trace head.

 158      */
 159     void FlipPosture() override;
 160 
 161     /**

 162      * Function UpdateSizes()

 163      *

 164      * Performs on-the-fly update of the width, via diameter & drill size from

 165      * a settings class. Used to dynamically change these parameters as

 166      * the track is routed.

 167      */
 168     void UpdateSizes( const SIZES_SETTINGS& aSizes ) override;
 169 
 170     bool IsPlacingVia() const override { return m_placingVia; }
 171 
 172     void SetOrthoMode( bool aOrthoMode ) override;
 173 
 174     void GetModifiedNets( std::vector<int>& aNets ) const override;
 175 
 176 private:
 177     int viaGap() const;
 178     int gap() const;
 179 
 180     /**

 181      * Function route()

 182      *

 183      * Re-routes the current track to point aP. Returns true, when routing has

 184      * completed successfully (i.e. the trace end has reached point aP), and false

 185      * if the trace was stuck somewhere on the way. May call routeStep()

 186      * repetitively due to mouse smoothing.

 187      * @param aP ending point of current route.

 188      * @return true, if the routing is complete.

 189      */
 190     bool route( const VECTOR2I& aP );
 191 
 192     /**

 193      * Function updateLeadingRatLine()

 194      *

 195      * Draws the "leading" ratsnest line, which connects the end of currently

 196      * routed track and the nearest yet unrouted item. If the routing for

 197      * current net is complete, draws nothing.

 198      */
 199     void updateLeadingRatLine();
 200 
 201     /**

 202      * Function setWorld()

 204      * Sets the board to route.

 205      */
 206     void setWorld( NODE* aWorld );
 207 
 208     /**

 209      * Function startPlacement()

 210      *

 211      * Initializes placement of a new line with given parameters.

 212      */
 213     void initPlacement( );
 214 
 215     /**

 216      * Function setInitialDirection()

 218      * Sets preferred direction of the very first track segment to be laid.

 220      */
