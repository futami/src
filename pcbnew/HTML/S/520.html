  46 /**

  47  * CMP_READER

  48  * reads a component footprint link file (*.cmp) format.

  49  */
  50 class CMP_READER
  51 {
  52     LINE_READER* m_lineReader;            ///< The line reader to read.

  53 
  54 public:
  55     /**

  56      * CMP_READER constructor.

  57      * @param aLineReader is a LINE_READER (in fact a FILE_LINE_READER)

  58      * which is owned by me ( and deleted by me) to read

  59      * the component footprint link file.

  60      */
  61     CMP_READER( LINE_READER* aLineReader )
  62     {
  63         m_lineReader = aLineReader;
  64     }
  65 
  66     ~CMP_READER()
  67     {
  68         if( m_lineReader )
  69         {
  70             delete m_lineReader;
  71             m_lineReader = NULL;
  72         }
  73     }
  74 
  75     /**

  76      * Function Load

  77      * read the *.cmp file format contains the component footprint assignments created by CvPcb

  78      * into \a aNetlist.

  79      *

  80      * @param aNetlist is the #NETLIST to read into.

  81      *

  82      * @todo At some point in the future, use the footprint field in the new s-expression

  83      *       netlist file to assign a footprint to a component instead of using a secondary

  84      *       (*.cmp) file.

  85      *

  86      * Sample file footprint assignment entry:

  87      *

  88      * Cmp-Mod V01 Genere by CvPcb 29/10/2003-13: 11:6 *

  89      *  BeginCmp

  90      *  TimeStamp = /32307DE2/AA450F67;

  91      *  Reference = C1;

  92      *  ValeurCmp = 47uF;

  93      *  IdModule  = CP6;

  94      *  EndCmp

  95      *

  96      * @throw IO_ERROR if a the #LINE_READER IO error occurs.

  97      * @throw PARSE_ERROR if an error occurs while parsing the file.

  98      * @return true if OK, false if a component reference found in the

  99      * .cmp file is not found in netlist, which means the .cmp file

 100      * is not updated. This is an usual case, in CvPcb, but can be used to

 101      * print a warning in Pcbnew.

 102      */
 103     bool Load( NETLIST* aNetlist );
 104 };

 107 /**

 108  * NETLIST_READER

 109  * is a pure virtual class to derive a specific type of netlist reader from.

 110  */
 111 class NETLIST_READER
 112 {
 113 public:
 114 
 115     enum NETLIST_FILE_T
 116     {
 117         UNKNOWN = -1,
 118         ORCAD,
 119         LEGACY,
 120         KICAD,
 121 
 122         // Add new types here.  Don't forget to create the appropriate class derived from

 123         // NETCLASS_READER and add the entry to the NETLIST_READER::GetNetlistReader()

 124         // function.

 125     };
 126 
 127 
 128     /**

 129      * Constructor

 130      * @param aLineReader ownership is taken of this #LINE_READER.

 131      * @param aNetlist the #NETLIST object to read into.

 132      * @param aFootprintLinkReader ownership is taken of this #CMP_READER.

 133      */
 134     NETLIST_READER( LINE_READER*  aLineReader,
 135                     NETLIST*      aNetlist,
 136                     CMP_READER*   aFootprintLinkReader = NULL )
 137     {
 138         wxASSERT( aLineReader != NULL );
 139 
 140         m_lineReader           = aLineReader;
 141         m_footprintReader      = aFootprintLinkReader;
 142         m_netlist              = aNetlist;
 143         m_loadFootprintFilters = true;
 144         m_loadNets             = true;
 145     }
 146 
 147     virtual ~NETLIST_READER();
 148 
 149     /**

 150      * Function GuessNetlistFileType

 151      * looks at \a aFileHeaderLine to see if it matches any of the netlist file types it

 152      * knows about.

 153      *

 154      * @param aLineReader is the #LINE_READER object containing lines from the netlist to test.

 155      * @return the #NETLIST_FILE_T of \a aLineReader.

 156      */
 157     static NETLIST_FILE_T GuessNetlistFileType( LINE_READER* aLineReader );
 158 
 159     /**

 160      * Function GetNetlistReader

 161      * attempts to determine the net list file type of \a aNetlistFileName and return the

 162      * appropriate NETLIST_READER type.

 163      *

 164      * @param aNetlist is the netlist to load \a aNetlistFileName into.

 165      * @param aNetlistFileName is the full path and file name of the net list to read.

 166      * @param aCompFootprintFileName is the full path and file name of the component footprint

 167      *                               associations to read.  Set to wxEmptyString if loading the

 168      *                               footprint association file is not required.

 169      * @return the appropriate NETLIST_READER if \a aNetlistFileName is a valid netlist or

 170      *         NULL if \a aNetlistFileName is not a valid netlist files.

 171      */
 172     static NETLIST_READER* GetNetlistReader( NETLIST*        aNetlist,
 173                                              const wxString& aNetlistFileName,
 174                                              const wxString& aCompFootprintFileName = wxEmptyString );
 175 
 176     /**

 177      * Function LoadNetlist

 178      * loads the contents of the netlist file into \a aNetlist.

 179      *

 180      * @throw IO_ERROR if a file IO error occurs.

 181      * @throw PARSE_ERROR if an error occurs while parsing the file.

 182      */
 183     virtual void LoadNetlist() = 0;
 184 
 185     /**

 186      * Function GetLineReader()

 187      * @return the #LINE_READER associated with the #NETLIST_READER.

 188      */
 189     LINE_READER* GetLineReader();
 190 
 191 protected:
 192     NETLIST*     m_netlist;               ///< The net list to read the file(s) into.

 193     bool         m_loadFootprintFilters;  ///< Load the component footprint filters section if true.

 194     bool         m_loadNets;              ///< Load the nets section of the netlist file if true.

 195     LINE_READER* m_lineReader;            ///< The line reader of the netlist.

 196 
 197     /// The reader used to load the footprint links.  If NULL, footprint links are not read.

 198     CMP_READER*  m_footprintReader;
 199 };

 202 /**

 203  * LEGACY_NETLIST_READER

 204  * reads the KiCad legacy and the old Orcad netlist formats.

 205  *

 206  * The KiCad legacy netlist format was derived directly from an old Orcad netlist format.  The

 207  * primary difference is the header was changed so this reader can read both formats.

 208  */
 209 class LEGACY_NETLIST_READER : public NETLIST_READER
 210 {
 211     /**

 212      * Function loadComponent

 213      * read the \a aLine containing the description of a component from a legacy format

 214      * netlist and add it to the netlist.

 215      *

 216      * Analyze the first line of a component description in netlist:

 217      * ( /40C08647 $noname R20 4.7K {Lib=R}

 218      *

 219      * @param  aText contains the first line of description

 220      * @return the new component created by parsing \a aLine

 221      * @throw PARSE_ERROR when \a aLine is not a valid component description.

 222      */
 223     COMPONENT* loadComponent( char* aText );
 224 
 225     /**

 226      * Function loadFootprintFilters

 227      * loads the footprint filter section of netlist file.

 228      *

 229      * Sample legacy footprint filter section:

 230      *  { Allowed footprints by component:

 231      *  $component R11

 232      *  R?

 233      *  SM0603

 234      *  SM0805

 235      *  R?-*

 236      *  SM1206

 237      *  $endlist

 238      *  $endfootprintlist

 239      *  }

 240      *

 241      * @throw IO_ERROR if a file IO error occurs.

 242      * @throw PARSE_ERROR if an error occurs while parsing the file.

 243      */
 244     void loadFootprintFilters();
 245 
 246     /**

 247      * Function loadNet

 248      * read a component net description from \a aText.

 249      *

 250      * @param aText is current line read from the netlist.

 251      * @param aComponent is the component to add the net to.

 252      * @throw PARSE_ERROR if a error occurs reading \a aText.

 253      */
 254     void loadNet( char* aText, COMPONENT* aComponent );
 255 
 256 public:
 257 
 258     LEGACY_NETLIST_READER( LINE_READER*  aLineReader,
 259                            NETLIST*      aNetlist,
 260                            CMP_READER*   aFootprintLinkReader = NULL ) :
 261         NETLIST_READER( aLineReader, aNetlist, aFootprintLinkReader )
 262     {
 263     }
 264 
 265     /**

 266      * Function LoadNetlist

 267      * read the netlist file in the legacy format into \a aNetlist.

 268      *

 269      * The legacy netlist format is:

 270      * \# EESchema Netlist Version 1.0 generee le  18/5/2005-12:30:22

 271      *  (

 272      *  ( 40C08647 $noname R20 4,7K {Lib=R}

 273      *  (    1 VCC )

 274      *  (    2 MODB_1 )

 275      *  )

 276      *  ( 40C0863F $noname R18 4,7_k {Lib=R}

 277      *  (    1 VCC )

 278      *  (    2 MODA_1 )

 279      *  )

 280      *  }

 281      * \#End

 282      *

 283      * @throw IO_ERROR if a file IO error occurs.

 284      * @throw PARSE_ERROR if an error occurs while parsing the file.

 285      */
 286     virtual void LoadNetlist() override;
 287 };

 290 /**

 291  * KICAD_NETLIST_PARSER

 292  * is the parser for reading the KiCad s-expression netlist format.

 293  */
 294 class KICAD_NETLIST_PARSER : public NETLIST_LEXER
 295 {
 296 private:
 297     NL_T::T      token;
 298     LINE_READER* m_lineReader;  ///< The line reader used to parse the netlist.  Not owned.

 299     NETLIST*     m_netlist;     ///< The netlist to parse into.  Not owned.

 300 
 301     /**

 302      * Function skipCurrent

 303      * Skip the current token level, i.e

 304      * search for the RIGHT parenthesis which closes the current description

 305      */
 306     void skipCurrent();
 307 
 308     /**

 309      * Function parseComponent

 310      * parse a component description:

 311      * (comp (ref P1)

 312      * (value DB25FEMELLE)

 313      * (footprint DB25FC)

 314      * (libsource (lib conn) (part DB25))

 315      * (sheetpath (names /) (tstamps /))

 316      * (tstamp 3256759C))

 317      */
 318     void parseComponent();
 319 
 320     /**

 321      * Function parseNet

 322      * Parses a section like

 323      * (net (code 20) (name /PC-A0)

 324      *  (node (ref BUS1) (pin 62))

 325      *  (node (ref U3) (pin 3))

 326      *  (node (ref U9) (pin M6)))

 327      *

 328      * and set the corresponding pads netnames

 329      */
 330     void parseNet();
 331 
 332     /**

 333      * Function parseLibPartList

 334      * reads the section "libparts" in the netlist:

 335      * (libparts

 336      *   (libpart (lib device) (part C)

 337      *     (description "Condensateur non polarise")

 338      *     (footprints

 339      *       (fp SM*)

 340      *       (fp C?)

 341      *       (fp C1-1))

 342      *     (fields

 343      *       (field (name Reference) C)

 344      *       (field (name Value) C))

 345      *     (pins

 346      *       (pin (num 1) (name ~) (type passive))

 347      *       (pin (num 2) (name ~) (type passive))))

 348      *

 349      *  And add the strings giving the footprint filter (subsection footprints)

 350      *  of the corresponding module info

 351      *  <p>This section is used by CvPcb, and is not useful in Pcbnew,

 352      *  therefore it it not always read </p>

 353      */
 354     void parseLibPartList();
 355 
 356 
 357 public:
 358     KICAD_NETLIST_PARSER( LINE_READER* aReader, NETLIST* aNetlist );
 359 
 360     void SetLineReader( LINE_READER* aLineReader );
 361 
 362     void SetNetlist( NETLIST* aNetlist ) { m_netlist = aNetlist; }
 363 
 364     /**

 365      * Function Parse

 366      * parse the full netlist

 367      */
 368     void Parse();
 369 
 370     // Useful for debug only:

 371     const char* getTokenName( NL_T::T aTok )
 372     {
 373         return NETLIST_LEXER::TokenName( aTok );
 374     }
 375 };
