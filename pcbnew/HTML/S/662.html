  43 /**

  44  * LINE

  45  *

  46  * Represents a track on a PCB, connecting two non-trivial joints (that is,

  47  * vias, pads, junctions between multiple traces or two traces different widths

  48  * and combinations of these). PNS_LINEs are NOT stored in the model (NODE).

  49  * Instead, they are assembled on-the-fly, based on a via/pad/segment that

  50  * belongs to/starts/ends them.

  51  *

  52  * PNS_LINEs can be either loose (consisting of segments that do not belong to

  53  * any NODE) or owned (with segments taken from a NODE) - these are

  54  * returned by NODE::AssembleLine and friends.

  55  *

  56  * A LINE may have a VIA attached at its end (i.e. the last point) - this is used by via

  57  * dragging/force propagation stuff.

  58  */
  59 
  60 #define PNS_HULL_MARGIN 10

  62 class LINE : public LINK_HOLDER
  63 {
  64 public:
  65 
  66     /**

  67      * Constructor

  68      * Makes an empty line.

  69      */
  70     LINE() : LINK_HOLDER( LINE_T )
  71     {
  72         m_hasVia = false;
  73         m_width = 1;        // Dummy value

  74         m_snapThreshhold = 0;
  75     }
  76 
  77     LINE( const LINE& aOther );
  78 
  79     /**

  80      * Constructor

  81      * Copies properties (net, layers, etc.) from a base line and replaces the shape

  82      * by another

  83      **/
  84     LINE( const LINE& aBase, const SHAPE_LINE_CHAIN& aLine )
  85             : LINK_HOLDER( aBase ),
  86               m_line( aLine ),
  87               m_width( aBase.m_width ),
  88               m_snapThreshhold( aBase.m_snapThreshhold )
  89     {
  90         m_net = aBase.m_net;
  91         m_layers = aBase.m_layers;
  92         m_hasVia = false;
  93     }
  94 
  95     /**

  96      * Constructor

  97      * Constructs a LINE for a lone VIA (ie a stitching via).

  98      * @param aVia

  99      */
 100     LINE( const VIA& aVia ) :
 101         LINK_HOLDER( LINE_T )
 102     {
 103         m_hasVia = true;
 104         m_via = aVia;
 105         m_width = aVia.Diameter();
 106         m_net = aVia.Net();
 107         m_layers = aVia.Layers();
 108         m_rank = aVia.Rank();
 109         m_snapThreshhold = 0;
 110     }
 111 
 112     ~LINE();
