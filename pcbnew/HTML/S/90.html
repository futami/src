  98 void CONNECTIVITY_DATA::updateRatsnest()
  99 {
 100     #ifdef PROFILE
 101     PROF_COUNTER rnUpdate( "update-ratsnest" );
 102     #endif
 103     std::vector<RN_NET*> dirty_nets;
 104 
 105     // Start with net 1 as net 0 is reserved for not-connected

 106     // Nets without nodes are also ignored

 107     std::copy_if( m_nets.begin() + 1, m_nets.end(), std::back_inserter( dirty_nets ),
 108             [] ( RN_NET* aNet ) { return aNet->IsDirty() && aNet->GetNodeCount() > 0; } );
 109 
 110     // We don't want to spin up a new thread for fewer than 8 nets (overhead costs)

 111     size_t parallelThreadCount = std::min<size_t>( std::thread::hardware_concurrency(),
 112             ( dirty_nets.size() + 7 ) / 8 );
 113 
 114     std::atomic<size_t> nextNet( 0 );
 115     std::vector<std::future<size_t>> returns( parallelThreadCount );
 116 
 117     auto update_lambda = [&nextNet, &dirty_nets]() -> size_t
 118     {
 119         for( size_t i = nextNet++; i < dirty_nets.size(); i = nextNet++ )
 120             dirty_nets[i]->Update();
 121 
 122         return 1;
 123     };
 124 
 125     if( parallelThreadCount == 1 )
 126         update_lambda();
 127     else
 128     {
 129         for( size_t ii = 0; ii < parallelThreadCount; ++ii )
 130             returns[ii] = std::async( std::launch::async, update_lambda );
 131 
 132         // Finalize the ratsnest threads

 133         for( size_t ii = 0; ii < parallelThreadCount; ++ii )
 134             returns[ii].wait();
 135     }
 136 
 137     #ifdef PROFILE
 138     rnUpdate.Show();
 139     #endif /* PROFILE */
 140 }

 151 void CONNECTIVITY_DATA::RecalculateRatsnest( BOARD_COMMIT* aCommit  )
 152 {
 153     m_connAlgo->PropagateNets( aCommit );
 154 
 155     int lastNet = m_connAlgo->NetCount();
 156 
 157     if( lastNet >= (int) m_nets.size() )
 158     {
 159         unsigned int prevSize = m_nets.size();
 160         m_nets.resize( lastNet + 1 );
 161 
 162         for( unsigned int i = prevSize; i < m_nets.size(); i++ )
 163             m_nets[i] = new RN_NET;
 164     }
 165 
 166     auto clusters = m_connAlgo->GetClusters();
 167 
 168     int dirtyNets = 0;
 169 
 170     for( int net = 0; net < lastNet; net++ )
 171     {
 172         if( m_connAlgo->IsNetDirty( net ) )
 173         {
 174             m_nets[net]->Clear();
 175             dirtyNets++;
 176         }
 177     }
 178 
 179     for( const auto& c : clusters )
 180     {
 181         int net = c->OriginNet();
 182 
 183         // Don't add intentionally-kept zone islands to the ratsnest

 184         if( c->IsOrphaned() && c->Size() == 1 )
 185         {
 186             if( dynamic_cast<CN_ZONE*>( *c->begin() ) )
 187                 continue;
 188         }
 189 
 190         if( m_connAlgo->IsNetDirty( net ) )
 191         {
 192             addRatsnestCluster( c );
 193         }
 194     }
 195 
 196     m_connAlgo->ClearDirtyFlags();
 197 
 198     if( !m_skipRatsnest )
 199         updateRatsnest();
 200 }

 203 void CONNECTIVITY_DATA::BlockRatsnestItems( const std::vector<BOARD_ITEM*>& aItems )
 204 {
 205     std::vector<BOARD_CONNECTED_ITEM*> citems;
 206 
 207     for( auto item : aItems )
 208     {
 209         if( item->Type() == PCB_MODULE_T )
 210         {
 211             for( auto pad : static_cast<MODULE*>(item)->Pads() )
 212                 citems.push_back( pad );
 213         }
 214         else
 215         {
 216             citems.push_back( static_cast<BOARD_CONNECTED_ITEM*>(item) );
 217         }
 218     }
 219 
 220     for( const auto& item : citems )
 221     {
 222         if ( m_connAlgo->ItemExists( item ) )
 223         {
 224             auto& entry = m_connAlgo->ItemEntry( item );
 225 
 226             for( const auto& cnItem : entry.GetItems() )
 227             {
 228                 for( auto anchor : cnItem->Anchors() )
 229                     anchor->SetNoLine( true );
 230             }
 231         }
 232     }
 233 }

 257 void CONNECTIVITY_DATA::ComputeDynamicRatsnest( const std::vector<BOARD_ITEM*>& aItems )
 258 {
 259     m_dynamicRatsnest.clear();
 260 
 261     if( std::none_of( aItems.begin(), aItems.end(), []( const BOARD_ITEM* aItem )
 262             { return( aItem->Type() == PCB_TRACE_T || aItem->Type() == PCB_PAD_T ||
 263                       aItem->Type() == PCB_ARC_T || aItem->Type() == PCB_ZONE_AREA_T ||
 264                       aItem->Type() == PCB_MODULE_T || aItem->Type() == PCB_VIA_T ); } ) )
 265     {
 266         return ;
 267     }
 268 
 269     CONNECTIVITY_DATA connData( aItems, true );
 270     BlockRatsnestItems( aItems );
 271 
 272     for( unsigned int nc = 1; nc < connData.m_nets.size(); nc++ )
 273     {
 274         auto dynNet = connData.m_nets[nc];
 275 
 276         if( dynNet->GetNodeCount() != 0 )
 277         {
 278             auto ourNet = m_nets[nc];
 279             CN_ANCHOR_PTR nodeA, nodeB;
 280 
 281             if( ourNet->NearestBicoloredPair( *dynNet, nodeA, nodeB ) )
 282             {
 283                 RN_DYNAMIC_LINE l;
 284                 l.a = nodeA->Pos();
 285                 l.b = nodeB->Pos();
 286                 l.netCode = nc;
 287 
 288                 m_dynamicRatsnest.push_back( l );
 289             }
 290         }
 291     }
 292 
 293     const auto& edges = GetRatsnestForItems( aItems );
 294 
 295     for( const auto& edge : edges )
 296     {
 297         const auto& nodeA   = edge.GetSourceNode();
 298         const auto& nodeB   = edge.GetTargetNode();
 299         RN_DYNAMIC_LINE l;
 300 
 301         // Use the parents' positions

 302         l.a = nodeA->Parent()->GetPosition();
 303         l.b = nodeB->Parent()->GetPosition();
 304         l.netCode = 0;
 305         m_dynamicRatsnest.push_back( l );
 306     }
 307 }

 330 {

 332 
 333     for( auto net : m_nets )
 334     {
 335         if( !net )
 336             continue;
 337

 359 const std::vector<BOARD_CONNECTED_ITEM*> CONNECTIVITY_DATA::GetConnectedItems(
 360         const BOARD_CONNECTED_ITEM* aItem,
 361         const KICAD_T aTypes[],
 362         bool aIgnoreNetcodes ) const
 363 {
 364     std::vector<BOARD_CONNECTED_ITEM*> rv;
 365     const auto clusters = m_connAlgo->SearchClusters(
 366             aIgnoreNetcodes ?
 367                     CN_CONNECTIVITY_ALGO::CSM_PROPAGATE :
 368                     CN_CONNECTIVITY_ALGO::CSM_CONNECTIVITY_CHECK, aTypes,
 369             aIgnoreNetcodes ? -1 : aItem->GetNetCode() );
 370 
 371     for( auto cl : clusters )
 372     {
 373         if( cl->Contains( aItem ) )
 374         {
 375             for( const auto item : *cl )
 376             {
 377                 if( item->Valid() )
 378                     rv.push_back( item->Parent() );
 379             }
 380         }
 381     }
 382 
 383     return rv;
 384 }

 387 const std::vector<BOARD_CONNECTED_ITEM*> CONNECTIVITY_DATA::GetNetItems( int aNetCode,
 388         const KICAD_T aTypes[] ) const
 389 {
 390     std::vector<BOARD_CONNECTED_ITEM*> items;
 391     items.reserve( 32 );
 392 
 393     std::bitset<MAX_STRUCT_TYPE_ID> type_bits;
 394 
 395     for( unsigned int i = 0; aTypes[i] != EOT; ++i )
 396     {
 397         wxASSERT( aTypes[i] < MAX_STRUCT_TYPE_ID );
 398         type_bits.set( aTypes[i] );
 399     }
 400 
 401     m_connAlgo->ForEachItem( [&]( CN_ITEM& aItem ) {
 402         if( aItem.Valid() && ( aItem.Net() == aNetCode ) && type_bits[aItem.Parent()->Type()] )
 403             items.push_back( aItem.Parent() );
 404     } );
 405 
 406     std::sort( items.begin(), items.end() );
 407     items.erase( std::unique( items.begin(), items.end() ), items.end() );
 408     return items;
 409 }

 412 bool CONNECTIVITY_DATA::CheckConnectivity( std::vector<CN_DISJOINT_NET_ENTRY>& aReport )
 413 {
 414     RecalculateRatsnest();
 415 
 416     for( auto net : m_nets )
 417     {
 418         if( net )
 419         {
 420             for( const auto& edge : net->GetEdges() )
 421             {
 422                 CN_DISJOINT_NET_ENTRY ent;
 423                 ent.net = edge.GetSourceNode()->Parent()->GetNetCode();
 424                 ent.a   = edge.GetSourceNode()->Parent();
 425                 ent.b   = edge.GetTargetNode()->Parent();
 426                 ent.anchorA = edge.GetSourceNode()->Pos();
 427                 ent.anchorB = edge.GetTargetNode()->Pos();
 428                 aReport.push_back( ent );
 429             }
 430         }
 431     }
 432 
 433     return aReport.empty();
 434 }

 543 bool CONNECTIVITY_DATA::TestTrackEndpointDangling( TRACK* aTrack, wxPoint* aPos )
 544 {
 545     auto items = GetConnectivityAlgo()->ItemEntry( aTrack ).GetItems();
 546 
 547     // Not in the connectivity system.  This is a bug!

 548     if( items.empty() )
 549     {
 550         wxFAIL_MSG( "track not in connectivity system" );
 551         return false;
 552     }
 553 
 554     CN_ITEM* citem = items.front();
 555 
 556     if( !citem->Valid() )
 557         return false;
 558 
 559     for( const std::shared_ptr<CN_ANCHOR>& anchor : citem->Anchors() )
 560     {
 561         if( anchor->IsDangling() )
 562         {
 563             *aPos = (wxPoint) anchor->Pos();
 564             return true;
 565         }
 566     }
 567 
 568     return false;
 569 }

 634 const std::vector<CN_EDGE> CONNECTIVITY_DATA::GetRatsnestForItems( std::vector<BOARD_ITEM*> aItems )
 635 {
 636     std::set<int> nets;
 637     std::vector<CN_EDGE> edges;
 638     std::set<BOARD_CONNECTED_ITEM*> item_set;
 639 
 640     for( auto item : aItems )
 641     {
 642         auto conn_item = dyn_cast<BOARD_CONNECTED_ITEM*>( item );
 643 
 644         if( !conn_item )
 645             continue;
 646 
 647         if( item->Type() == PCB_MODULE_T )
 648         {
 649             auto component = static_cast<MODULE*>( item );
 650 
 651             for( auto pad : component->Pads() )
 652             {
 653                 nets.insert( pad->GetNetCode() );
 654                 item_set.insert( pad );
 655             }
 656         }
 657         else
 658         {
 659             item_set.insert( conn_item );
 660             nets.insert( conn_item->GetNetCode() );
 661         }
 662     }
 663 
 664     for( const auto& netcode : nets )
 665     {
 666         const auto& net = GetRatsnestForNet( netcode );
 667 
 668         for( const auto& edge : net->GetEdges() )
 669         {
 670             auto srcNode = edge.GetSourceNode();
 671             auto dstNode = edge.GetTargetNode();
 672 
 673             auto srcParent = srcNode->Parent();
 674             auto dstParent = dstNode->Parent();
 675 
 676             bool srcFound = ( item_set.find(srcParent) != item_set.end() );
 677             bool dstFound = ( item_set.find(dstParent) != item_set.end() );
 678 
 679             if ( srcFound && dstFound )
 680                 edges.push_back( edge );
 681         }
 682     }
 683 
 684     return edges;
 685 }

 688 const std::vector<CN_EDGE> CONNECTIVITY_DATA::GetRatsnestForComponent( MODULE* aComponent, bool aSkipInternalConnections )
 689 {
 690     std::set<int> nets;
 691     std::set<const D_PAD*> pads;
 692     std::vector<CN_EDGE> edges;
 693 
 694     for( auto pad : aComponent->Pads() )
 695     {
 696         nets.insert( pad->GetNetCode() );
 697         pads.insert( pad );
 698     }
 699 
 700     for( const auto& netcode : nets )
 701     {
 702         const auto& net = GetRatsnestForNet( netcode );
 703 
 704         for( const auto& edge : net->GetEdges() )
 705         {
 706             auto srcNode = edge.GetSourceNode();
 707             auto dstNode = edge.GetTargetNode();
 708 
 709             auto srcParent = static_cast<const D_PAD*>( srcNode->Parent() );
 710             auto dstParent = static_cast<const D_PAD*>( dstNode->Parent() );
 711 
 712             bool srcFound = ( pads.find(srcParent) != pads.end() );
 713             bool dstFound = ( pads.find(dstParent) != pads.end() );
 714 
 715             if ( srcFound && dstFound && !aSkipInternalConnections )
 716             {
 717                 edges.push_back( edge );
 718             }
 719             else if ( srcFound || dstFound )
 720             {
 721                 edges.push_back( edge );
 722             }
 723         }
 724     }
 725 
 726     return edges;
