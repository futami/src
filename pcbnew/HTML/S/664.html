 103 /**

 104  * LINE_PLACER

 105  *

 106  * Single track placement algorithm. Interactively routes a track.

 107  * Applies shove and walkaround algorithms when needed.

 108  */
 109 
 110 class LINE_PLACER : public PLACEMENT_ALGO
 111 {
 112 public:
 113     LINE_PLACER( ROUTER* aRouter );
 114     ~LINE_PLACER();
 115 
 116     /**

 117      * Function Start()

 118      *

 119      * Starts routing a single track at point aP, taking item aStartItem as anchor

 120      * (unless NULL).

 121      */
 122     bool Start( const VECTOR2I& aP, ITEM* aStartItem ) override;
 123 
 124     /**

 125      * Function Move()

 126      *

 127      * Moves the end of the currently routed trace to the point aP, taking

 128      * aEndItem as anchor (if not NULL).

 129      * (unless NULL).

 130      */
 131     bool Move( const VECTOR2I& aP, ITEM* aEndItem ) override;
 132 
 133     /**

 134      * Function FixRoute()

 135      *

 136      * Commits the currently routed track to the parent node, taking

 137      * aP as the final end point and aEndItem as the final anchor (if provided).

 138      * @return true, if route has been commited. May return false if the routing

 139      * result is violating design rules - in such case, the track is only committed

 140      * if Settings.CanViolateDRC() is on.

 141      */
 142     bool FixRoute( const VECTOR2I& aP, ITEM* aEndItem, bool aForceFinish ) override;
 143 
 144     bool UnfixRoute() override;
 145 
 146     bool CommitPlacement() override;
 147 
 148     bool AbortPlacement() override;
 149 
 150     bool HasPlacedAnything() const override;
 151 
 152     /**

 153      * Function ToggleVia()

 154      *

 155      * Enables/disables a via at the end of currently routed trace.

 156      */
 157     bool ToggleVia( bool aEnabled ) override;
 158 
 159     /**

 160      * Function SetLayer()

 161      *

 162      * Sets the current routing layer.

 163      */
 164     bool SetLayer( int aLayer ) override;
 165 
 166     /**

 167      * Function Head()

 168      *

 169      * Returns the "head" of the line being placed, that is the volatile part

 170      * that has not "settled" yet.

 171      */
 172     const LINE& Head() const { return m_head; }
 173 
 174     /**

 175      * Function Tail()

 176      *

 177      * Returns the "tail" of the line being placed, the part which has already wrapped around

 178      * and shoved some obstacles.

 179      */
 180     const LINE& Tail() const { return m_tail; }
 181 
 182     /**

 183      * Function Trace()

 184      *

 185      * Returns the complete routed line.

 186      */
 187     const LINE Trace() const;
 188 
 189     /**

 190      * Function Traces()

 191      *

 192      * Returns the complete routed line, as a single-member ITEM_SET.

 193      */
 194     const ITEM_SET Traces() override;
 195 
 196     /**

 197      * Function CurrentEnd()

 198      *

 199      * Returns the current end of the line being placed. It may not be equal

 200      * to the cursor position due to collisions.

 201      */
 202     const VECTOR2I& CurrentEnd() const override
 203     {
 204         return m_currentEnd;
 205     }
 206 
 207     /**

 208      * Function CurrentNet()

 209      *

 210      * Returns the net code of currently routed track.

 211      */
 212     const std::vector<int> CurrentNets() const override
 213     {
 214         return std::vector<int>( 1, m_currentNet );
 215     }
 216 
 217     /**

 218      * Function CurrentLayer()

 219      *

 220      * Returns the layer of currently routed track.

 221      */
 222     int CurrentLayer() const override
 223     {
 224         return m_currentLayer;
 225     }
 226 
 227     /**

 228      * Function CurrentNode()

 229      *

 230      * Returns the most recent world state.

 231      */
 232     NODE* CurrentNode( bool aLoopsRemoved = false ) const override;
 233 
 234     /**

 235      * Function FlipPosture()

 236      *

 237      * Toggles the current posture (straight/diagonal) of the trace head.

 238      */
 239     void FlipPosture() override;
 240 
 241     /**

 242      * Function UpdateSizes()

 243      *

 244      * Performs on-the-fly update of the width, via diameter & drill size from

 245      * a settings class. Used to dynamically change these parameters as

 246      * the track is routed.

 247      */
 248     void UpdateSizes( const SIZES_SETTINGS& aSizes ) override;
 249 
 250     void SetOrthoMode( bool aOrthoMode ) override;
 251 
 252     bool IsPlacingVia() const override { return m_placingVia; }
 253 
 254     void GetModifiedNets( std::vector<int>& aNets ) const override;
 255 
 256     /**

 257      * Function SplitAdjacentSegments()

 258      *

 259      * Checks if point aP lies on segment aSeg. If so, splits the segment in two,

 260      * forming a joint at aP and stores updated topology in node aNode.

 261      */
 262     bool SplitAdjacentSegments( NODE* aNode, ITEM* aSeg, const VECTOR2I& aP );
 263 
 264 
 265 private:
 266     /**

 267      * Function route()

 268      *

 269      * Re-routes the current track to point aP. Returns true, when routing has

 270      * completed successfully (i.e. the trace end has reached point aP), and false

 271      * if the trace was stuck somewhere on the way. May call routeStep()

 272      * repetitively due to mouse smoothing.

 273      * @param aP ending point of current route.

 274      * @return true, if the routing is complete.

 275      */
 276     bool route( const VECTOR2I& aP );
 277 
 278     /**

 279      * Function updateLeadingRatLine()

 280      *

 281      * Draws the "leading" ratsnest line, which connects the end of currently

 282      * routed track and the nearest yet unrouted item. If the routing for

 283      * current net is complete, draws nothing.

 284      */
 285     void updateLeadingRatLine();
 286 
 287     /**

 288      * Function setWorld()

 289      *

 290      * Sets the board to route.

 291      */
 292     void setWorld( NODE* aWorld );
 293 
 294     /**

 295      * Function startPlacement()

 296      *

 297      * Initializes placement of a new line with given parameters.

 298      */
 299     void initPlacement();
 300 
 301     /**

 302      * Function setInitialDirection()

 303      *

 304      * Sets preferred direction of the very first track segment to be laid.

 305      * Used by posture switching mechanism.

 306      */
 307     void setInitialDirection( const DIRECTION_45& aDirection );
 308 
 309     /**

 310      * Function removeLoops()

 311      *

 312      * Searches aNode for traces concurrent to aLatest and removes them. Updated

 313      * topology is stored in aNode.

 314      */
 315     void removeLoops( NODE* aNode, LINE& aLatest );
 316 
 317     /**

 318      * Function simplifyNewLine()

 319      *

 320      * Assembles a line starting from segment aLatest, removes collinear segments

 321      * and redundant vertexes. If a simplification bhas been found, replaces the

 322      * old line with the simplified one in aNode.

 323      */
 324     void simplifyNewLine( NODE* aNode, SEGMENT* aLatest );
 325 
 326     /**

 327      * Function checkObtusity()

 328      *

 329      * Helper function, checking if segments a and b form an obtuse angle

 330      * (in 45-degree regime).

 331      * @return true, if angle (aA, aB) is obtuse

 332      */
 333     bool checkObtusity( const SEG& aA, const SEG& aB ) const;
 334 
 335     /**

 336      * Function handleSelfIntersections()

 337      *

 338      * Checks if the head of the track intersects its tail. If so, cuts the

 339      * tail up to the intersecting segment and fixes the head direction to match

 340      * the last segment before the cut.

 341      * @return true if the line has been changed.

 342      */
 343     bool handleSelfIntersections();
 344 
 345     /**

 346      * Function handlePullback()

 347      *

 348      * Deals with pull-back: reduces the tail if head trace is moved backwards

 349      * wrs to the current tail direction.

 350      * @return true if the line has been changed.

 351      */
 352     bool handlePullback();
 353 
 354     /**

 355      * Function mergeHead()

 356      *

 357      * Moves "estabished" segments from the head to the tail if certain

 358      * conditions are met.

 359      * @return true, if the line has been changed.

 360      */
 361     bool mergeHead();
 362 
 363     /**

 364      * Function reduceTail()

 365      *

 366      * Attempts to reduce the numer of segments in the tail by trying to replace a

 367      * certain number of latest tail segments with a direct trace leading to aEnd

 368      * that does not collide with anything.

 369      * @param aEnd: current routing destination point.

 370      * @return true if the line has been changed.

 371      */
 372     bool reduceTail( const VECTOR2I& aEnd );
 373 
 374     /**

 375      * Function optimizeTailHeadTransition()

 376      *

 377      * Tries to reduce the corner count of the most recent part of tail/head by

 378      * merging obtuse/collinear segments.

 379      * @return true, if the line has been changed.

 380      */
 381     bool optimizeTailHeadTransition();
 382 
 383     /**

 384      * Function routeHead()

 385      *

 386      * Computes the head trace between the current start point (m_p_start) and

 387      * point aP, starting with direction defined in m_direction. The trace walks

 388      * around all colliding solid or non-movable items. Movable segments are

 389      * ignored, as they'll be handled later by the shove algorithm.

 390      */
 391     bool routeHead( const VECTOR2I& aP, LINE& aNewHead);
 392 
 393     /**

 394      * Function routeStep()

 395      *

 396      * Performs a single routing alorithm step, for the end point aP.

 397      * @param aP ending point of current route

 398      * @return true, if the line has been changed.

 399      */
 400     void routeStep( const VECTOR2I& aP );
 401 
 402     const LINE reduceToNearestObstacle( const LINE& aOriginalLine );
 403 
 404     bool rhStopAtNearestObstacle( const VECTOR2I& aP, LINE& aNewHead );
 405 
 406 
 407     ///> route step, walkaround mode

 408     bool rhWalkOnly( const VECTOR2I& aP, LINE& aNewHead);
 409 
 410     ///> route step, shove mode

 411     bool rhShoveOnly( const VECTOR2I& aP, LINE& aNewHead);
 412 
 413     ///> route step, mark obstacles mode

 414     bool rhMarkObstacles( const VECTOR2I& aP, LINE& aNewHead );
 415 
 416     const VIA makeVia( const VECTOR2I& aP );
 417 
 418     bool buildInitialLine( const VECTOR2I& aP, LINE& aHead );
 419 
 420     ///> current routing direction

 421     DIRECTION_45 m_direction;
 422 
 423     ///> routing direction for new traces

 424     DIRECTION_45 m_initial_direction;
 425 
 426     ///> routing "head": volatile part of the track from the previously

 427     ///  analyzed point to the current routing destination

 428     LINE m_head;
 429 
 430     ///> routing "tail": part of the track that has been already fixed due to collisions with obstacles

 431     LINE m_tail;
 432 
 433     ///> pointer to world to search colliding items

 434     NODE* m_world;
 435 
 436     ///> current routing start point (end of tail, beginning of head)

 437     VECTOR2I m_p_start;
 438 
 439     ///> The shove engine

 440     std::unique_ptr< SHOVE > m_shove;
 441 
 442     ///> Current world state

 443     NODE* m_currentNode;
 444 
 445     ///> Postprocessed world state (including marked collisions & removed loops)

 446     NODE* m_lastNode;
 447 
 448     SIZES_SETTINGS m_sizes;
 449 
 450     ///> Are we placing a via?

 451     bool m_placingVia;
 452 
 453     int m_currentNet;
 454     int m_currentLayer;
 455 
 456     VECTOR2I m_currentEnd, m_currentStart;
 457     LINE m_currentTrace;
 458 
 459     PNS_MODE m_currentMode;
 460     ITEM* m_startItem;
 461 
 462     bool m_idle;
 463     bool m_chainedPlacement;
 464     bool m_orthoMode;
 465     bool m_placementCorrect;
 466 
 467     FIXED_TAIL m_fixedTail;
 468     POSTURE_SOLVER m_postureSolver;
 469 };
