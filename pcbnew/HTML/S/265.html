  29 DIALOG_PNS_SETTINGS::DIALOG_PNS_SETTINGS( wxWindow* aParent, PNS::ROUTING_SETTINGS& aSettings ) :
  30     DIALOG_PNS_SETTINGS_BASE( aParent ), m_settings( aSettings )
  31 {
  32     // Add tool tip to the mode radio box, one by option

  33     // (cannot be made with wxFormBuilder for each item )

  34     m_mode->SetItemToolTip( 0, _( "DRC violation: highlight obstacles" ) );
  35     m_mode->SetItemToolTip( 1, _( "DRC violation: shove tracks and vias" ) );
  36     m_mode->SetItemToolTip( 2, _( "DRC violation: walk around obstacles" ) );
  37 
  38     // Load widgets' values from settings

  39     m_mode->SetSelection( m_settings.Mode() );
  40     m_shoveVias->SetValue( m_settings.ShoveVias() );
  41     m_backPressure->SetValue( m_settings.JumpOverObstacles() );
  42     m_removeLoops->SetValue( m_settings.RemoveLoops() );
  43     m_suggestEnding->SetValue( m_settings.SuggestFinish() );
  44     m_smartPads->SetValue( m_settings.SmartPads() );
  45     m_effort->SetValue( m_settings.OptimizerEffort() );
  46     m_smoothDragged->SetValue( m_settings.SmoothDraggedSegments() );
  47     m_violateDrc->SetValue( m_settings.CanViolateDRC() );
  48     m_freeAngleMode->SetValue( m_settings.GetFreeAngleMode() );
  49     m_dragToolMode->SetSelection ( m_settings.InlineDragEnabled() ? 1 : 0 );
  50     m_optimizeDraggedTrack->SetValue( m_settings.GetOptimizeDraggedTrack() );
  51     // Enable/disable some options

  52     wxCommandEvent event;
  53     onModeChange( event );
  54 
  55     // Don't show options that are not implemented

  56     m_suggestEnding->Hide();
  57     m_shoveVias->Hide();
  58 
  59     SetDefaultItem( m_stdButtonsOK );
  60     GetSizer()->Fit( this );
  61     GetSizer()->SetSizeHints( this );
  62 }
  63

  65 void DIALOG_PNS_SETTINGS::OnOkClick( wxCommandEvent& aEvent )
  66 {
  67     // Save widgets' values to settings

  68     m_settings.SetMode( (PNS::PNS_MODE) m_mode->GetSelection() );
  69     m_settings.SetShoveVias( m_shoveVias->GetValue() );
  70     m_settings.SetJumpOverObstacles( m_backPressure->GetValue() );
  71     m_settings.SetRemoveLoops( m_removeLoops->GetValue() );
  72     m_settings.SetSuggestFinish ( m_suggestEnding->GetValue() );
  73     m_settings.SetSmartPads( m_smartPads->GetValue() );
  74     m_settings.SetOptimizerEffort( (PNS::PNS_OPTIMIZATION_EFFORT) m_effort->GetValue() );
  75     m_settings.SetSmoothDraggedSegments( m_smoothDragged->GetValue() );
  76     m_settings.SetCanViolateDRC( m_violateDrc->GetValue() );
  77     m_settings.SetFreeAngleMode( m_freeAngleMode->GetValue() );
  78     m_settings.SetInlineDragEnabled( m_dragToolMode->GetSelection () ? true : false );
  79     m_settings.SetOptimizeDraggedTrack( m_optimizeDraggedTrack->GetValue() );
  80 
  81     aEvent.Skip();      // ends returning wxID_OK (default behavior)

  82 }
  83 
  84 
  85 void DIALOG_PNS_SETTINGS::onModeChange( wxCommandEvent& aEvent )
  86 {
  87     if( m_mode->GetSelection() == PNS::RM_MarkObstacles )
  88     {
  89         m_freeAngleMode->Enable();
  90         m_violateDrc->Enable();
  91     }
  92     else
  93     {
  94         m_freeAngleMode->Enable( false );
  95         m_violateDrc->Enable( false );
  96     }
  97 }
