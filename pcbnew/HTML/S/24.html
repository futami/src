  45 /**

  46  * AR_MATRIX

  47  * handle the matrix routing that describes the actual board

  48  */
  49 class AR_MATRIX
  50 {
  51 public:
  52     typedef unsigned char MATRIX_CELL;
  53     typedef int           DIST_CELL;
  54     typedef char          DIR_CELL;
  55 
  56     MATRIX_CELL* m_BoardSide[AR_MAX_ROUTING_LAYERS_COUNT]; // the image map of 2 board sides

  57     DIST_CELL*   m_DistSide[AR_MAX_ROUTING_LAYERS_COUNT];  // the image map of 2 board sides:

  58                                                            // distance to cells

  59     DIR_CELL* m_DirSide[AR_MAX_ROUTING_LAYERS_COUNT];      // the image map of 2 board sides:

  60                                                            // pointers back to source

  61     int      m_RoutingLayersCount; // Number of layers for autorouting (0 or 1)

  62     int      m_GridRouting;        // Size of grid for autoplace/autoroute

  63     EDA_RECT m_BrdBox;             // Actual board bounding box

  64     int      m_Nrows, m_Ncols;     // Matrix size

  65     int      m_MemSize;            // Memory requirement, just for statistics

  66     int      m_RouteCount;         // Number of routes

  67 
  68     PCB_LAYER_ID m_routeLayerTop;
  69     PCB_LAYER_ID m_routeLayerBottom;
  70 
  71 private:
  72     // a pointer to the current selected cell operation

  73     void ( AR_MATRIX::*m_opWriteCell )( int aRow, int aCol, int aSide, MATRIX_CELL aCell );
  74 
  75 public:
  76     enum CELL_OP
  77     {
  78         WRITE_CELL = 0,
  79         WRITE_OR_CELL = 1,
  80         WRITE_XOR_CELL = 2,
  81         WRITE_AND_CELL = 3,
  82         WRITE_ADD_CELL = 4
  83     };
  84 
  85     AR_MATRIX();
  86     ~AR_MATRIX();
  87 
  88     void WriteCell( int aRow, int aCol, int aSide, MATRIX_CELL aCell )
  89     {
  90         ( *this.*m_opWriteCell )( aRow, aCol, aSide, aCell );
  91     }
  92 
  93     /**

  94      * function GetBrdCoordOrigin

  95      * @return the board coordinate corresponding to the

  96      * routing matrix origin ( board coordinate offset )

  97      */
  98     wxPoint GetBrdCoordOrigin()
  99     {
 100         return m_BrdBox.GetOrigin();
 101     }
 102 
 103     /**

 104      * Function ComputeMatrixSize

 105      * calculates the number of rows and columns of dimensions of \a aPcb for routing and

 106      * automatic calculation of area.

 107      * @param aPcb = the physical board

 108      * @param aUseBoardEdgesOnly = true to use board edges only,

 109      *                           = false to use the full board bounding box (default)

 110      */
 111     bool ComputeMatrixSize( const EDA_RECT& aBoundingBox );
 112 
 113     /**

 114      * Function InitBoard

 115      * initializes the data structures.

 116      *

 117      * @return the amount of memory used or -1 if default.

 118      */
 119     int InitRoutingMatrix();
 120 
 121     void UnInitRoutingMatrix();
 122 
 123     // Initialize WriteCell to make the aLogicOp

 124     void SetCellOperation( CELL_OP aLogicOp );
 125 
 126     // functions to read/write one cell ( point on grid routing matrix:

 127     MATRIX_CELL GetCell( int aRow, int aCol, int aSide );
 128     void        SetCell( int aRow, int aCol, int aSide, MATRIX_CELL aCell );
 129     void        OrCell( int aRow, int aCol, int aSide, MATRIX_CELL aCell );
 130     void        XorCell( int aRow, int aCol, int aSide, MATRIX_CELL aCell );
 131     void        AndCell( int aRow, int aCol, int aSide, MATRIX_CELL aCell );
 132     void        AddCell( int aRow, int aCol, int aSide, MATRIX_CELL aCell );
 133     DIST_CELL   GetDist( int aRow, int aCol, int aSide );
 134     void        SetDist( int aRow, int aCol, int aSide, DIST_CELL );
 135 
 136     void TraceSegmentPcb( DRAWSEGMENT* pt_segm, int color, int marge, AR_MATRIX::CELL_OP op_logic );
 137     void CreateKeepOutRectangle(
 138             int ux0, int uy0, int ux1, int uy1, int marge, int aKeepOut, LSET aLayerMask );
 139     void PlacePad( D_PAD* aPad, int color, int marge, AR_MATRIX::CELL_OP op_logic );
 140     void TraceFilledRectangle( int ux0, int uy0, int ux1, int uy1, double angle, LSET aLayerMask,
 141             int color, AR_MATRIX::CELL_OP op_logic );
 142     void TraceFilledRectangle( int ux0, int uy0, int ux1, int uy1, LSET aLayerMask, int color,
 143             AR_MATRIX::CELL_OP op_logic );
 144 
 145 private:
 146     void drawSegmentQcq( int ux0, int uy0, int ux1, int uy1, int lg, LAYER_NUM layer, int color,
 147             CELL_OP op_logic );

 149     void traceCircle( int ux0, int uy0, int ux1, int uy1, int lg, LAYER_NUM layer, int color,
 150             AR_MATRIX::CELL_OP op_logic );
 151     void traceFilledCircle(
 152             int cx, int cy, int radius, LSET aLayerMask, int color, AR_MATRIX::CELL_OP op_logic );

 154             int color, AR_MATRIX::CELL_OP op_logic );
 155 };
