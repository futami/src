  74 /**

  75  * Struct OBSTACLE

  76  *

  77  * Holds an object colliding with another object, along with

  78  * some useful data about the collision.

  79  **/
  80 struct OBSTACLE
  81 {
  82     ///> Item we search collisions with

  83     const ITEM* m_head;
  84 
  85     ///> Item found to be colliding with m_head

  86     ITEM* m_item;
  87 
  88     ///> Hull of the colliding m_item

  89     SHAPE_LINE_CHAIN m_hull;
  90 
  91     ///> First and last intersection point between the head item and the hull

  92     ///> of the colliding m_item

  93     VECTOR2I m_ipFirst, m_ipLast;
  94 
  95     ///> ... and the distance thereof

  96     int m_distFirst, m_distLast;
  97 };

 133 /**

 134  * NODE

 135  *

 136  * Keeps the router "world" - i.e. all the tracks, vias, solids in a

 137  * hierarchical and indexed way.

 138  * Features:

 139  * - spatial-indexed container for PCB item shapes

 140  * - collision search & clearance checking

 141  * - assembly of lines connecting joints, finding loops and unique paths

 142  * - lightweight cloning/branching (for recursive optimization and shove

 143  * springback)

 144  **/
 145 class NODE

 167         m_maxClearance = aClearance;
 168     }
