 160     HYPERLYNX_PAD_STACK* addPadStack( HYPERLYNX_PAD_STACK stack )
 161     {
 162         for( auto p : m_padStacks )
 163         {
 164             if( *p == stack )
 165                 return p;
 166         }
 167 
 168         stack.SetId( m_padStacks.size() );
 169         m_padStacks.push_back( new HYPERLYNX_PAD_STACK( stack ) );
 170 
 171         return m_padStacks.back();
 172     }

 174     const std::string formatPadShape( HYPERLYNX_PAD_STACK& aStack )
 175     {
 176         int  shapeId = 0;
 177         char buf[1024];
 178 
 179         switch( aStack.m_shape )
 180         {
 181         case PAD_SHAPE_CIRCLE:
 182         case PAD_SHAPE_OVAL: shapeId = 0; break;
 183         case PAD_SHAPE_ROUNDRECT: shapeId = 2; break;
 184         case PAD_SHAPE_RECT: shapeId = 1; break;
 185         default:
 186             shapeId = 0;
 187 
 188             if( m_reporter )
 189             {
 190                 m_reporter->Report(
 191                         _( "File contains pad shapes that are not supported by the Hyperlynx exporter\n"
 192                            "(Supported shapes are oval, rectangle, circle.)\n"
 193                            "They have been exported as oval pads." ),
 194                         RPT_SEVERITY_WARNING );
 195             }
 196             break;
 197         }
 198 
 199         snprintf( buf, sizeof( buf ), "%d, %.9f, %.9f, %.1f, M", shapeId,
 200                 iu2hyp( aStack.m_sx ),
 201                 iu2hyp( aStack.m_sy ),
 202                 aStack.m_angle );
 203 
 204         return buf;
 205     }

 229 HYPERLYNX_PAD_STACK::HYPERLYNX_PAD_STACK( BOARD* aBoard, const D_PAD* aPad )
 230 {
 231     m_board = aBoard;
 232     m_sx    = aPad->GetSize().x;
 233     m_sy    = aPad->GetSize().y;
 234     m_angle = 180.0 - ( aPad->GetOrientation() / 10.0 );
 235 
 236     if( m_angle < 0.0 )
 237     {
 238         m_angle += 360.0;
 239     }
 240 
 241     m_layers = aPad->GetLayerSet();
 242     m_drill  = aPad->GetDrillSize().x;
 243     m_shape  = aPad->GetShape();
 244     m_type   = PAD_ATTRIB_STANDARD;
 245     m_id     = 0;
 246 }

 271 void HYPERLYNX_EXPORTER::writeSinglePadStack( HYPERLYNX_PAD_STACK& aStack )
 272 {
 273     LSET layerMask = LSET::AllCuMask() & m_board->GetEnabledLayers();
 274     LSET outLayers = aStack.m_layers & layerMask;
 275 
 276     if( outLayers.none() )
 277         return;
 278 
 279     m_out->Print( 0, "{PADSTACK=%d, %.9f\n", aStack.m_id, iu2hyp( aStack.m_drill ) );
 280 
 281     if( outLayers == layerMask )
 282     {
 283         m_out->Print( 1, "(\"%s\", %s)\n", "MDEF", formatPadShape( aStack ).c_str() );
 284     }
 285     else
 286     {
 287         for( auto l : outLayers.Seq() )
 288         {
 289             m_out->Print( 1, "(\"%s\", %s)\n", (const char*) m_board->GetLayerName( l ).c_str(),
 290                     formatPadShape( aStack ).c_str() );
 291         }
 292     }
 293 
 294     m_out->Print( 0, "}\n\n" );
 295 }

 298 bool HYPERLYNX_EXPORTER::writeBoardInfo()
 299 {
 300     SHAPE_POLY_SET outlines;
 301     wxString       errText;
 302     wxPoint        errLoc;
 303 
 304     m_out->Print( 0, "{BOARD \"%s\"\n", (const char*) m_board->GetFileName().c_str() );
 305 
 306     if( !m_board->GetBoardPolygonOutlines( outlines, &errText, &errLoc ) )
 307     {
 308         return false;
 309     }
 310 
 311     for( int o = 0; o < outlines.OutlineCount(); o++ )
 312     {
 313         const auto& outl = outlines.COutline( o );
 314 
 315         for( int i = 0; i < outl.SegmentCount(); i++ )
 316         {
 317             const auto& s = outl.CSegment( i );
 318             m_out->Print( 1, "(PERIMETER_SEGMENT X1=%.9f Y1=%.9f X2=%.9f Y2=%.9f)\n",
 319                     iu2hyp( s.A.x ), iu2hyp( s.A.y ), iu2hyp( s.B.x ), iu2hyp( s.B.y ) );
 320         }
 321     }
 322 
 323     m_out->Print( 0, "}\n\n" );
 324 
 325     return true;
 326 }

 329 bool HYPERLYNX_EXPORTER::writeStackupInfo()
 330 {
 331     /* Format:

 332      * {STACKUP

 333      * (SIGNAL T=thickness [P=plating_thickness] [C=constant] L=layer_name [M=material_name])             [comment]

 334      * (DIELECTRIC T=thickness [C=constant] [L=layer_name] [M=material_name]) [comment]

 335      * }

 336      * name lenght is <= 20 chars

 337      */
 338 
 339     auto layers = m_board->GetDesignSettings().GetEnabledLayers().CuStack();
 340 
 341     // Get the board physical stackup structure

 342     BOARD_STACKUP& stackup = m_board->GetDesignSettings().GetStackupDescriptor();
 343 
 344     m_out->Print( 0, "{STACKUP\n" );
 345 
 346     wxString layer_name;    // The last copper layer name used in stackup

 347 
 348     for( BOARD_STACKUP_ITEM* item: stackup.GetList() )
 349     {
 350         if( item->GetType() == BS_ITEM_TYPE_COPPER )
 351         {
 352             layer_name = m_board->GetLayerName( item->GetBrdLayerId() );
 353             int plating_thickness = 0;
 354             double resistivity = 1.724e-8;  // Good for copper

 355             m_out->Print( 1, "(SIGNAL T=%g P=%g C=%g L=\"%.20s\" M=COPPER)\n",
 356                           iu2hyp( item->GetThickness( 0 ) ),
 357                           iu2hyp( plating_thickness ),
 358                           resistivity,
 359                           TO_UTF8( layer_name ) );
 360         }
 361         else if( item->GetType() == BS_ITEM_TYPE_DIELECTRIC )
 362         {
 363             if( item->GetSublayersCount() < 2 )
 364             {
 365                 m_out->Print( 1, "(DIELECTRIC T=%g C=%g L=\"DE_%.17s\" M=\"%.20s\")\n",
 366                               iu2hyp( item->GetThickness( 0 ) ),
 367                               item->GetEpsilonR( 0 ),
 368                               TO_UTF8( layer_name ),
 369                               TO_UTF8( item->GetMaterial( 0 ) ) );
 370             }
 371             else for( int idx = 0; idx < item->GetSublayersCount(); idx++ )
 372             {
 373                 m_out->Print( 1, "(DIELECTRIC T=%g C=%g L=\"DE%d_%.16s\" M=\"%.20s\")\n",
 374                               iu2hyp( item->GetThickness( idx ) ),
 375                               item->GetEpsilonR( idx ),
 376                               idx, TO_UTF8( layer_name ),
 377                               TO_UTF8( item->GetMaterial( idx ) ) );
 378             }
 379         }
 380     }
 381 
 382     m_out->Print( 0, "}\n\n" );
 383 
 384     return true;
 385 }

 409 bool HYPERLYNX_EXPORTER::writePadStacks()
 410 {
 411     for( auto mod : m_board->Modules() )
 412     {
 413         for( auto pad : mod->Pads() )
 414         {
 415             auto ps = addPadStack( HYPERLYNX_PAD_STACK( m_board, pad ) );
 416             m_padMap[pad] = ps;
 417         }
 418     }
 419 
 420     for( auto trk : m_board->Tracks() )
 421     {
 422         if( VIA* via = dyn_cast<VIA*>( trk ) )
 423         {
 424             auto ps = addPadStack( HYPERLYNX_PAD_STACK( m_board, via ) );
 425             m_padMap[via] = ps;
 426         }
 427     }
 428 
 429     for( auto pstack : m_padStacks )
 430         writeSinglePadStack( *pstack );
 431 
 432     return true;
 433 }

 436 bool HYPERLYNX_EXPORTER::writeNetObjects( const std::vector<BOARD_ITEM*>& aObjects )
 437 {
 438     for( auto item : aObjects )
 439     {
 440         if( D_PAD* pad = dyn_cast<D_PAD*>( item ) )
 441         {
 442             auto pstackIter = m_padMap.find( pad );
 443 
 444             if( pstackIter != m_padMap.end() )
 445             {
 446                 wxString ref = pad->GetParent()->GetReference();
 447 
 448                 if( ref.IsEmpty() )
 449                     ref = "EMPTY";
 450 
 451                 auto padName = pad->GetName();
 452 
 453                 if( padName.IsEmpty() )
 454                     padName = "1";
 455 
 456 
 457                 m_out->Print( 1, "(PIN X=%.10f Y=%.10f R=\"%s.%s\" P=%d)\n",
 458                         iu2hyp( pad->GetPosition().x ), iu2hyp( pad->GetPosition().y ),
 459                         (const char*) ref.c_str(), (const char*) padName.c_str(),
 460                         pstackIter->second->GetId() );
 461             }
 462         }
 463         else if( VIA* via = dyn_cast<VIA*>( item ) )
 464         {
 465             auto pstackIter = m_padMap.find( via );
 466 
 467             if( pstackIter != m_padMap.end() )
 468             {
 469                 m_out->Print( 1, "(VIA X=%.10f Y=%.10f P=%d)\n", iu2hyp( via->GetPosition().x ),
 470                         iu2hyp( via->GetPosition().y ), pstackIter->second->GetId() );
 471             }
 472         }
 473         else if( TRACK* track = dyn_cast<TRACK*>( item ) )
 474         {
 475             const auto layerName = m_board->GetLayerName( track->GetLayer() );
 476 
 477             m_out->Print( 1, "(SEG X1=%.10f Y1=%.10f X2=%.10f Y2=%.10f W=%.10f L=\"%s\")\n",
 478                     iu2hyp( track->GetStart().x ), iu2hyp( track->GetStart().y ),
 479                     iu2hyp( track->GetEnd().x ), iu2hyp( track->GetEnd().y ),
 480                     iu2hyp( track->GetWidth() ), (const char*) layerName.c_str() );
 481         }
 482         else if( ZONE_CONTAINER* zone = dyn_cast<ZONE_CONTAINER*>( item ) )
 483         {
 484             for( PCB_LAYER_ID layer : zone->GetLayerSet().Seq() )
 485             {
 486                 const auto     layerName   = m_board->GetLayerName( layer );
 487                 SHAPE_POLY_SET filledShape = zone->GetFilledPolysList( layer );
 488 
 489                 filledShape.Simplify( SHAPE_POLY_SET::PM_FAST );
 490 
 491                 for( int i = 0; i < filledShape.OutlineCount(); i++ )
 492                 {
 493                     const auto& outl = filledShape.COutline( i );
 494 
 495                     auto p0 = outl.CPoint( 0 );
 496                     m_out->Print( 1, "{POLYGON T=POUR L=\"%s\" ID=%d X=%.10f Y=%.10f\n",
 497                             (const char*) layerName.c_str(), m_polyId, iu2hyp( p0.x ),
 498                             iu2hyp( p0.y ) );
 499 
 500                     for( int v = 0; v < outl.PointCount(); v++ )
 501                     {
 502                         m_out->Print( 2, "(LINE X=%.10f Y=%.10f)\n", iu2hyp( outl.CPoint( v ).x ),
 503                                 iu2hyp( outl.CPoint( v ).y ) );
 504                     }
 505 
 506                     m_out->Print( 2, "(LINE X=%.10f Y=%.10f)\n", iu2hyp( p0.x ), iu2hyp( p0.y ) );
 507                     m_out->Print( 1, "}\n" );
 508 
 509                     for( int h = 0; h < filledShape.HoleCount( i ); h++ )
 510                     {
 511                         const auto& holeShape = filledShape.CHole( i, h );
 512                         auto        ph0       = holeShape.CPoint( 0 );
 513 
 514                         m_out->Print( 1, "{POLYVOID ID=%d X=%.10f Y=%.10f\n", m_polyId,
 515                                 iu2hyp( ph0.x ), iu2hyp( ph0.y ) );
 516 
 517                         for( int v = 0; v < holeShape.PointCount(); v++ )
 518                         {
 519                             m_out->Print( 2, "(LINE X=%.10f Y=%.10f)\n",
 520                                     iu2hyp( holeShape.CPoint( v ).x ),
 521                                     iu2hyp( holeShape.CPoint( v ).y ) );
 522                         }
 523 
 524                         m_out->Print(
 525                                 2, "(LINE X=%.10f Y=%.10f)\n", iu2hyp( ph0.x ), iu2hyp( ph0.y ) );
 526                         m_out->Print( 1, "}\n" );
 527                     }
 528 
 529                     m_polyId++;
 530                 }
 531             }
 532         }
 533     }
 534 
 535     return true;
 536 }

 539 const std::vector<BOARD_ITEM*> HYPERLYNX_EXPORTER::collectNetObjects( int netcode )
 540 {
 541     std::vector<BOARD_ITEM*> rv;
 542 
 543     auto check = [&]( BOARD_CONNECTED_ITEM* item ) -> bool {
 544         if( ( item->GetLayerSet() & LSET::AllCuMask() ).none() )
 545             return false;
 546         if( item->GetNetCode() == netcode || ( netcode < 0 && item->GetNetCode() <= 0 ) )
 547             return true;
 548         return false;
 549     };
 550 
 551     for( auto mod : m_board->Modules() )
 552     {
 553         for( auto pad : mod->Pads() )
 554         {
 555             if( check( pad ) )
 556                 rv.push_back( pad );
 557         }
 558     }
 559 
 560     for( auto item : m_board->Tracks() )
 561         if( check( item ) )
 562             rv.push_back( item );
 563 
 564     for( int i = 0; i < m_board->GetAreaCount(); i++ )
 565     {
 566         auto zone = m_board->GetArea( i );
 567         if( check( zone ) )
 568             rv.push_back( zone );
 569     }
 570     return rv;
 571 }

 574 bool HYPERLYNX_EXPORTER::writeNets()
 575 {
 576     m_polyId = 1;
 577 
 578     for( const auto netInfo : m_board->GetNetInfo() )
 579     {
 580         int netcode = netInfo->GetNet();
 581         bool isNullNet = netInfo->GetNet() <= 0 || netInfo->GetNetname().IsEmpty();
 582 
 583         if( isNullNet )
 584             continue;
 585 
 586         auto netObjects = collectNetObjects( netcode );
 587 
 588         if( netObjects.size() )
 589         {
 590             m_out->Print( 0, "{NET=\"%s\"\n", (const char*) netInfo->GetNetname().c_str() );
 591             writeNetObjects( netObjects );
 592             m_out->Print( 0, "}\n\n" );
 593         }
 594     }
 595 
 596     auto nullNetObjects = collectNetObjects( -1 );
 597 
 598     int idx = 0;
 599 
 600     for( auto item : nullNetObjects )
 601     {
 602         m_out->Print( 0, "{NET=\"EmptyNet%d\"\n", idx );
 603         writeNetObjects( { item } );
 604         m_out->Print( 0, "}\n\n" );
 605         idx++;
 606     }
 607 
 608     return true;
 609 }
