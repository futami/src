  78 namespace KIGFX {
  79 /**

  80  * Special flavor of PCB_PAINTER that contains

  81  * modifications to handle printing options.

  82  */
  83 class PCB_PRINT_PAINTER : public PCB_PAINTER
  84 {
  85 public:
  86     PCB_PRINT_PAINTER( GAL* aGal );
  87 
  88     /**

  89      * Set drill marks visibility and options.

  90      * @param aRealSize when enabled, drill marks represent actual holes. Otherwise aSize

  91      * parameter is used.

  92      * @param aSize is drill mark size (internal units), valid only when aRealSize == false.

  93      */
  94     void SetDrillMarks( bool aRealSize, unsigned int aSize = 0 )
  95     {
  96         m_drillMarkReal = aRealSize;
  97         m_drillMarkSize = aSize;
  98     }
  99 
 100 protected:
 101     int getDrillShape( const D_PAD* aPad ) const override;
 102 
 103     VECTOR2D getDrillSize( const D_PAD* aPad ) const override;
 104 
 105     int getDrillSize( const VIA* aVia ) const override;
 106 
 107     ///> Flag deciding whether use the actual hole size or user-specified size for drill marks

 108     bool m_drillMarkReal;
 109 
 110     ///> User-specified size for drill marks (expressed in internal units)

 111     int m_drillMarkSize;
 112 };
