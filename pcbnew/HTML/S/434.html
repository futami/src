  61 void FOOTPRINT_EDIT_FRAME::LoadModuleFromLibrary( LIB_ID aFPID)
  62 {
  63     bool is_last_fp_from_brd = IsCurrentFPFromBoard();
  64 
  65     MODULE* module = LoadFootprint( aFPID );
  66 
  67     if( !module )
  68         return;
  69 
  70     if( !Clear_Pcb( true ) )
  71         return;
  72 
  73     GetCanvas()->GetViewControls()->SetCrossHairCursorPosition( VECTOR2D( 0, 0 ), false );
  74     AddModuleToBoard( module );
  75 
  76     auto fp = GetBoard()->GetFirstModule();
  77 
  78     if( fp )
  79     {
  80         fp->ClearFlags();
  81 
  82         // if either m_Reference or m_Value are gone, reinstall them -

  83         // otherwise you cannot see what you are doing on board

  84         TEXTE_MODULE* ref = &fp->Reference();
  85         TEXTE_MODULE* val = &fp->Value();
  86 
  87         if( val && ref )
  88         {
  89             ref->SetType( TEXTE_MODULE::TEXT_is_REFERENCE );    // just in case ...

  90 
  91             if( ref->GetLength() == 0 )
  92                 ref->SetText( wxT( "Ref**" ) );
  93 
  94             val->SetType( TEXTE_MODULE::TEXT_is_VALUE );        // just in case ...

  95 
  96             if( val->GetLength() == 0 )
  97                 val->SetText( wxT( "Val**" ) );
  98         }
  99     }
 100 
 101     Zoom_Automatique( false );
 102 
 103     Update3DView( true );
 104 
 105     GetScreen()->ClrModify();
 106 
 107     updateView();
 108     GetCanvas()->Refresh();
 109 
 110     // Update the save items if needed.

 111     if( is_last_fp_from_brd )
 112     {
 113         ReCreateMenuBar();
 114         ReCreateHToolbar();
 115     }
 116 
 117     m_treePane->GetLibTree()->ExpandLibId( aFPID );
 118     m_treePane->GetLibTree()->CenterLibId( aFPID );
 119     m_treePane->GetLibTree()->RefreshLibTree();        // update highlighting

 120 }

 123 void FOOTPRINT_EDIT_FRAME::Process_Special_Functions( wxCommandEvent& event )
 124 {
 125     int        id = event.GetId();
 126     wxPoint    pos;
 127 
 128     wxGetMousePosition( &pos.x, &pos.y );
 129     pos.y += 20;
 130 
 131     switch( id )
 132     {
 133     case ID_MODEDIT_NEW_MODULE:
 134         {
 135             LIB_ID selected = m_treePane->GetLibTree()->GetSelectedLibId();
 136             MODULE* module = CreateNewModule( wxEmptyString );
 137 
 138             if( !module )
 139                 break;
 140 
 141             if( !Clear_Pcb( true ) )
 142                 break;
 143 
 144             GetCanvas()->GetViewControls()->SetCrossHairCursorPosition( VECTOR2D( 0, 0 ), false );
 145             AddModuleToBoard( module );
 146 
 147             // Initialize data relative to nets and netclasses (for a new

 148             // module the defaults are used)

 149             // This is mandatory to handle and draw pads

 150             GetBoard()->BuildListOfNets();
 151             module->SetPosition( wxPoint( 0, 0 ) );
 152 
 153             if( GetBoard()->GetFirstModule() )
 154                 GetBoard()->GetFirstModule()->ClearFlags();
 155 
 156             Zoom_Automatique( false );
 157             GetScreen()->SetModify();
 158 
 159             // If selected from the library tree then go ahead and save it there

 160             if( !selected.GetLibNickname().empty() )
 161             {
 162                 LIB_ID fpid = module->GetFPID();
 163                 fpid.SetLibNickname( selected.GetLibNickname() );
 164                 module->SetFPID( fpid );
 165                 SaveFootprint( module );
 166                 GetScreen()->ClrModify();
 167             }
 168 
 169             updateView();
 170             GetCanvas()->Refresh();
 171             Update3DView( true );
 172 
 173             SyncLibraryTree( false );
 174         }
 175         break;
 176 
 177     case ID_MODEDIT_NEW_MODULE_FROM_WIZARD:
 178         {
 179             LIB_ID selected = m_treePane->GetLibTree()->GetSelectedLibId();
 180 
 181             if( IsContentModified() )
 182             {
 183                 if( !HandleUnsavedChanges( this, _( "The current footprint has been modified.  "
 184                                                     "Save changes?" ),
 185                                            [&]() -> bool {
 186                                                return SaveFootprint( GetBoard()->GetFirstModule() );
 187                                            } ) )
 188                 {
 189                     break;
 190                 }
 191             }
 192 
 193             auto wizard = (FOOTPRINT_WIZARD_FRAME*) Kiway().Player( FRAME_FOOTPRINT_WIZARD, true,
 194                                                                     this );
 195 
 196             if( wizard->ShowModal( NULL, this ) )
 197             {
 198                 // Creates the new footprint from python script wizard

 199                 MODULE* module = wizard->GetBuiltFootprint();
 200 
 201                 if( module == NULL )        // i.e. if create module command aborted

 202                     break;
 203 
 204                 Clear_Pcb( false );
 205 
 206                 GetCanvas()->GetViewControls()->SetCrossHairCursorPosition( VECTOR2D( 0, 0 ), false );
 207                 //  Add the new object to board

 208                 AddModuleToBoard( module );
 209 
 210                 // Initialize data relative to nets and netclasses (for a new

 211                 // module the defaults are used)

 212                 // This is mandatory to handle and draw pads

 213                 GetBoard()->BuildListOfNets();
 214                 module->SetPosition( wxPoint( 0, 0 ) );
 215                 module->ClearFlags();
 216 
 217                 Zoom_Automatique( false );
 218                 GetScreen()->SetModify();
 219 
 220                 // If selected from the library tree then go ahead and save it there

 221                 if( !selected.GetLibNickname().empty() )
 222                 {
 223                     LIB_ID fpid = module->GetFPID();
 224                     fpid.SetLibNickname( selected.GetLibNickname() );
 225                     module->SetFPID( fpid );
 226                     SaveFootprint( module );
 227                     GetScreen()->ClrModify();
 228                 }
 229 
 230                 updateView();
 231                 GetCanvas()->Refresh();
 232                 Update3DView( true );
 233 
 234                 SyncLibraryTree( false );
 235             }
 236 
 237             wizard->Destroy();
 238         }
 239         break;
 240 
 241     case ID_MODEDIT_SAVE:
 242         if( !GetBoard()->GetFirstModule() )     // no loaded footprint

 243             break;
 244 
 245         if( GetTargetFPID() == GetLoadedFPID() )
 246         {
 247             if( SaveFootprint( GetBoard()->GetFirstModule() ) )
 248             {
 249                 m_toolManager->GetView()->Update( GetBoard()->GetFirstModule() );
 250 
 251                 GetCanvas()->ForceRefresh();
 252                 GetScreen()->ClrModify();
 253             }
 254         }
 255 
 256         m_treePane->GetLibTree()->RefreshLibTree();
 257         break;
 258 
 259     case ID_MODEDIT_SAVE_AS:
 260         if( GetTargetFPID().GetLibItemName().empty() )
 261         {
 262             // Save Library As

 263             const wxString& src_libNickname = GetTargetFPID().GetLibNickname();
 264             wxString src_libFullName = Prj().PcbFootprintLibs()->GetFullURI( src_libNickname );
 265 
 266             if( SaveLibraryAs( src_libFullName ) )
 267                 SyncLibraryTree( true );
 268         }
 269         else if( GetTargetFPID() == GetLoadedFPID() )
 270         {
 271             // Save Board Footprint As

 272             MODULE* footprint = GetBoard()->GetFirstModule();
 273 
 274             if( footprint && SaveFootprintAs( footprint ) )
 275             {
 276                 m_footprintNameWhenLoaded = footprint->GetFPID().GetLibItemName();
 277                 m_toolManager->GetView()->Update( footprint );
 278                 GetScreen()->ClrModify();
 279 
 280                 GetCanvas()->ForceRefresh();
 281                 SyncLibraryTree( true );
 282             }
 283         }
 284         else
 285         {
 286             // Save Selected Footprint As

 287             MODULE* footprint = LoadFootprint( GetTargetFPID() );
 288 
 289             if( footprint && SaveFootprintAs( footprint ) )
 290                 SyncLibraryTree( true );
 291         }
 292 
 293         m_treePane->GetLibTree()->RefreshLibTree();
 294         break;
 295 
 296     case ID_ADD_FOOTPRINT_TO_BOARD:
 297         SaveFootprintToBoard( true );
 298         break;
 299 
 300     case ID_TOOLBARH_PCB_SELECT_LAYER:
 301         SetActiveLayer( ToLAYER_ID( m_selLayerBox->GetLayerSelection() ) );
 302 
 303         if( GetDisplayOptions().m_ContrastModeDisplay !=
 304             HIGH_CONTRAST_MODE::NORMAL )
 305         {
 306             GetCanvas()->Refresh();
 307         }
 308         break;
 309 
 310     case ID_MODEDIT_CHECK:
 311         // Currently: not implemented

 312         break;
 313 
 314     default:
 315         wxLogDebug( wxT( "FOOTPRINT_EDIT_FRAME::Process_Special_Functions error" ) );
 316         break;
 317     }
 318 }

 337 void FOOTPRINT_EDIT_FRAME::editFootprintProperties( MODULE* aModule )
 338 {
 339     LIB_ID oldFPID = aModule->GetFPID();
 340 
 341     DIALOG_FOOTPRINT_FP_EDITOR dialog( this, aModule );
 342     dialog.ShowModal();
 343 
 344     // Update library tree

 345     BASIC_FOOTPRINT_INFO footprintInfo( aModule );
 346     wxDataViewItem       treeItem = m_adapter->FindItem( oldFPID );
 347 
 348     if( treeItem.IsOk() )   // Can be not found in tree if the current footprint is imported from file

 349                             // therefore not yet in tree.

 350     {
 351         static_cast<LIB_TREE_NODE_LIB_ID*>( treeItem.GetID() )->Update( &footprintInfo );
 352         m_treePane->GetLibTree()->RefreshLibTree();
 353     }
 354 
 355     updateTitle();      // in case of a name change...

 356 
 357     UpdateMsgPanel();
 358 }

 361 void FOOTPRINT_EDIT_FRAME::OnEditItemRequest( BOARD_ITEM* aItem )
 362 {
 363     switch( aItem->Type() )
 364     {
 365     case PCB_PAD_T:
 366         InstallPadOptionsFrame( static_cast<D_PAD*>( aItem ) );
 367         break;
 368 
 369     case PCB_MODULE_T:
 370         editFootprintProperties( (MODULE*) aItem );
 371         GetCanvas()->Refresh();
 372         break;
 373 
 374     case PCB_MODULE_TEXT_T:
 375         InstallTextOptionsFrame( aItem );
 376         break;
 377 
 378     case PCB_MODULE_EDGE_T :
 379         InstallGraphicItemPropertiesDialog( aItem );
 380         break;
 381 
 382     case PCB_MODULE_ZONE_AREA_T:
 383     {
 384         ZONE_CONTAINER* zone = static_cast<ZONE_CONTAINER*>( aItem );
 385         bool success = false;
 386         if( zone )
 387         {
 388             ZONE_SETTINGS zoneSettings;
 389             zoneSettings << *zone;
 390             if( zone->GetIsKeepout() )
 391             {
 392                 success = InvokeKeepoutAreaEditor( this, &zoneSettings );
 393             }
 394             else if( zone->IsOnCopperLayer() )
 395             {
 396                 success = InvokeCopperZonesEditor( this, &zoneSettings );
 397             }
 398             else
 399             {
 400                 success = InvokeNonCopperZonesEditor( this, &zoneSettings );
 401             }
 402 
 403             if( success )
 404             {
 405                 BOARD_COMMIT commit( this );
 406                 commit.Modify( zone );
 407                 commit.Push( _( "Edit Zone" ) );
 408                 zoneSettings.ExportSetting( *zone );
 409             }
 410         }
 411     }
 412     break;
 413 
 414     default:
 415         wxASSERT( 0 );
 416         break;
 417     }
 418 }

 440 bool FOOTPRINT_EDIT_FRAME::OpenProjectFiles( const std::vector<wxString>& aFileSet, int aCtl )
 441 {
 442     if( ! Clear_Pcb( true ) )
 443         return false;                  // //this command is aborted

 444 
 445     GetCanvas()->GetViewControls()->SetCrossHairCursorPosition( VECTOR2D( 0, 0 ), false );
 446     Import_Module( aFileSet[0] );
 447 
 448     if( GetBoard()->GetFirstModule() )
 449         GetBoard()->GetFirstModule()->ClearFlags();
 450 
 451     GetScreen()->ClrModify();
 452     Zoom_Automatique( false );
 453     GetCanvas()->Refresh();
 454 
 455     return true;

 459 void FOOTPRINT_EDIT_FRAME::KiwayMailIn( KIWAY_EXPRESS& mail )
 460 {
 461     const std::string& payload = mail.GetPayload();
 462 
 463     switch( mail.Command() )
 464     {
 465     case MAIL_FP_EDIT:
 466         if( !payload.empty() )
 467         {
 468             wxFileName fpFileName( payload );
 469             wxString libNickname;
 470             wxString msg;
 471 
 472             FP_LIB_TABLE*        libTable = Prj().PcbFootprintLibs();
 473             const LIB_TABLE_ROW* libTableRow = libTable->FindRowByURI( fpFileName.GetPath() );
 474 
 475             if( !libTableRow )
 476             {
 477                 msg.Printf( _( "The current configuration does not include the footprint library\n"
 478                                "\"%s\".\nUse Manage Footprint Libraries to edit the configuration." ),
 479                             fpFileName.GetPath() );
 480                 DisplayErrorMessage( this, _( "Library not found in footprint library table." ), msg );
 481                 break;
 482             }
 483 
 484             libNickname = libTableRow->GetNickName();
 485 
 486             if( !libTable->HasLibrary( libNickname, true ) )
 487             {
 488                 msg.Printf( _( "The library with the nickname \"%s\" is not enabled\n"
 489                                "in the current configuration.  Use Manage Footprint Libraries to\n"
 490                                "edit the configuration." ), libNickname );
 491                 DisplayErrorMessage( this, _( "Footprint library not enabled." ), msg );
 492                 break;
 493             }
 494 
 495             LIB_ID  fpId( libNickname, fpFileName.GetName() );
 496 
 497             if( m_treePane )
 498             {
 499                 m_treePane->GetLibTree()->SelectLibId( fpId );
 500                 wxCommandEvent event( COMPONENT_SELECTED );
 501                 wxPostEvent( m_treePane, event );
 502             }
 503         }
 504 
 505         break;
 506 
 507     default:
 508         ;
 509     }
