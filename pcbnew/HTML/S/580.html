  34 /**

  35  * Common, abstract interface for edit frames.

  36  */
  37 class PCB_BASE_EDIT_FRAME : public PCB_BASE_FRAME
  38 {
  39 public:
  40     PCB_BASE_EDIT_FRAME( KIWAY* aKiway, wxWindow* aParent, FRAME_T aFrameType,
  41                 const wxString& aTitle, const wxPoint& aPos, const wxSize& aSize,
  42                 long aStyle, const wxString& aFrameName );
  43 
  44     virtual ~PCB_BASE_EDIT_FRAME();
  45 
  46     /**

  47      * Function CreateNewLibrary

  48      * If a library name is given, creates a new footprint library in the project folder

  49      * with the given name. If no library name is given it prompts user for a library path,

  50      * then creates a new footprint library at that location.

  51      * If library exists, user is warned about that, and is given a chance

  52      * to abort the new creation, and in that case existing library is first deleted.

  53      *

  54      * @return wxString - the newly created library path if library was successfully

  55      *   created, else wxEmptyString because user aborted or error.

  56      */
  57     wxString CreateNewLibrary(const wxString& aLibName = wxEmptyString);
  58 
  59     /**

  60      * Function AddLibrary

  61      * Add an existing library to either the global or project library table.

  62      * @param aFileName the library to add; a file open dialog will be displayed if empty.

  63      * @return true if successfully added

  64      */
  65     bool AddLibrary(const wxString& aLibName = wxEmptyString);
  66 
  67     /**

  68      * Function OnEditItemRequest

  69      * Install the corresponding dialog editor for the given item

  70      * @param aDC = the current device context

  71      * @param aItem = a pointer to the BOARD_ITEM to edit

  72      */
  73     virtual void OnEditItemRequest( BOARD_ITEM* aItem ) = 0;
  74 
  75     // Undo buffer handling

  76 
  77     /**

  78      * Function SaveCopyInUndoList

  79      * Creates a new entry in undo list of commands.

  80      * add a picker to handle aItemToCopy

  81      * @param aItemToCopy = the board item modified by the command to undo

  82      * @param aTypeCommand = command type (see enum UNDO_REDO_T)

  83      * @param aTransformPoint = the reference point of the transformation, for

  84      *                          commands like move

  85      */
  86     void SaveCopyInUndoList( BOARD_ITEM* aItemToCopy, UNDO_REDO_T aTypeCommand,
  87                             const wxPoint& aTransformPoint = wxPoint( 0, 0 ) ) override;
  88 
  89     /**

  90      * Function SaveCopyInUndoList

  91      * Creates a new entry in undo list of commands.

  92      * add a list of pickers to handle a list of items

  93      * @param aItemsList = the list of items modified by the command to undo

  94      * @param aTypeCommand = command type (see enum UNDO_REDO_T)

  95      * @param aTransformPoint = the reference point of the transformation,

  96      *                          for commands like move

  97      */
  98     void SaveCopyInUndoList( const PICKED_ITEMS_LIST& aItemsList, UNDO_REDO_T aTypeCommand,
  99                             const wxPoint& aTransformPoint = wxPoint( 0, 0 ) ) override;
 100 
 101     /**

 102      * Function RestoreCopyFromRedoList

 103      *  Redo the last edit:

 104      *  - Save the current board in Undo list

 105      *  - Get an old version of the board from Redo list

 106      *  @return none

 107      */
 108     void RestoreCopyFromRedoList( wxCommandEvent& aEvent );
 109 
 110     /**

 111      * Function RestoreCopyFromUndoList

 112      *  Undo the last edit:

 113      *  - Save the current board in Redo list

 114      *  - Get an old version of the board from Undo list

 115      *  @return none

 116      */
 117     void RestoreCopyFromUndoList( wxCommandEvent& aEvent );
 118 
 119     /**

 120      * Performs an undo of the last edit WITHOUT logging a corresponding redo.  Used to cancel

 121      * an in-progress operation.

 122      */
 123     void RollbackFromUndo();
 124 
 125     /**

 126      * Function PutDataInPreviousState

 127      * Used in undo or redo command.

 128      * Put data pointed by List in the previous state, i.e. the state memorized by List

 129      * @param aList = a PICKED_ITEMS_LIST pointer to the list of items to undo/redo

 130      * @param aRedoCommand = a bool: true for redo, false for undo

 131      * @param aRebuildRatsnet = a bool: true to rebuild ratsnest (normal use), false to just

 132      * retrieve last state (used in abort commands that do not need to rebuild ratsnest)

 133      */
 134     void PutDataInPreviousState( PICKED_ITEMS_LIST* aList,
 135                                  bool               aRedoCommand,
 136                                  bool               aRebuildRatsnet = true );
 137 
 138     /**

 139      * Function UndoRedoBlocked

 140      * Checks if the undo and redo operations are currently blocked.

 141      */
 142     bool UndoRedoBlocked() const
 143     {
 144         return m_undoRedoBlocked;
 145     }
 146 
 147     /**

 148      * Function UndoRedoBlock

 149      * Enables/disable undo and redo operations.

 150      */
 151     void UndoRedoBlock( bool aBlock = true )
 152     {
 153         m_undoRedoBlocked = aBlock;
 154     }

 157      * Function SetGridVisibility()

 161      * @param aVisible = true if the grid must be shown

 162      */
 163     void SetGridVisibility( bool aVisible ) override;

 172      * Function SetRotationAngle()

 173      * Sets the angle used for rotate operations.

 174      */
 175     void SetRotationAngle( int aRotationAngle );

 177     void InstallTextOptionsFrame( BOARD_ITEM* aText );

 194     /**

 195      * Function ClearUndoORRedoList

 196      * free the undo or redo list from List element

 197      *  Wrappers are deleted.

 198      *  datas pointed by wrappers are deleted if not in use in schematic

 199      *  i.e. when they are copy of a schematic item or they are no more in use

 200      *  (DELETED)

 201      * @param whichList = the UNDO_REDO_CONTAINER to clear

 202      * @param aItemCount = the count of items to remove. < 0 for all items

 203      * items are removed from the beginning of the list.

 204      * So this function can be called to remove old commands

 205      */
 206     void ClearUndoORRedoList( UNDO_REDO_LIST whichList, int aItemCount = -1 ) override;
