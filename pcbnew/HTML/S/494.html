  76     /**

  77      * PLUGIN_REGISTRY

  78      * Holds a list of available plugins, created using a singleton REGISTER_PLUGIN object.

  79      * This way, plugins can be added link-time.

  80      */
  81     class PLUGIN_REGISTRY
  82     {
  83         public:
  84             struct ENTRY
  85             {
  86                 PCB_FILE_T m_type;
  87                 std::function<PLUGIN*(void)> m_createFunc;
  88                 wxString m_name;
  89             };
  90 
  91             static PLUGIN_REGISTRY *Instance()
  92             {
  93                 static PLUGIN_REGISTRY *self = nullptr;
  94 
  95                 if( !self )
  96                 {
  97                     self = new PLUGIN_REGISTRY;
  98                 }
  99                 return self;
 100             }
 101 
 102             void Register( PCB_FILE_T aType, const wxString& aName, std::function<PLUGIN*(void)> aCreateFunc )
 103             {
 104                 ENTRY ent;
 105                 ent.m_type = aType;
 106                 ent.m_createFunc = aCreateFunc;
 107                 ent.m_name = aName;
 108                 m_plugins.push_back( ent );
 109             }
 110 
 111             PLUGIN* Create( PCB_FILE_T aFileType ) const
 112             {
 113                 for( auto& ent : m_plugins )
 114                 {
 115                     if ( ent.m_type == aFileType )
 116                     {
 117                         return ent.m_createFunc();
 118                     }
 119                 }
 120                 return nullptr;
 121             }
 122 
 123             const std::vector<ENTRY>& AllPlugins() const
 124             {
 125                 return m_plugins;
 126             }
 127 
 128         private:
 129             std::vector<ENTRY> m_plugins;
 130     };

 251 /**

 252  * PLUGIN

 253  * is a base class that BOARD loading and saving plugins should derive from.

 254  * Implementations can provide either Load() or Save() functions, or both.

 255  * PLUGINs throw exceptions, so it is best that you wrap your calls to these

 256  * functions in a try catch block.  Plugins throw exceptions because it is illegal

 257  * for them to have any user interface calls in them whatsoever, i.e. no windowing

 258  * or screen printing at all.

 259  *

 260  * <pre>

 261  *   try

 262  *   {

 263  *        IO_MGR::Load(...);

 264  *   or

 265  *        IO_MGR::Save(...);

 266  *   }

 267  *   catch( const IO_ERROR& ioe )

 268  *   {

 269  *        // grab text from ioe, show in error window.

 270  *   }

 271  * </pre>

 272  */

 292      * Function Load

 294      * knows about, into either a new BOARD or an existing one. This may be used to load an

 295      * entire new BOARD, or to augment an existing one if @a aAppendToMe is not NULL.

 296      *

 297      * @param aFileName is the name of the file to use as input and may be foreign in

 298      *  nature or native in nature.

 299      *

 300      * @param aAppendToMe is an existing BOARD to append to, but if NULL then

 301      *   this means "do not append, rather load anew".

 302      *

 574     /**

 575      * RELEASER

 576      * releases a PLUGIN in the context of a potential thrown exception, through

 577      * its destructor.

 578      */
 579     class RELEASER
 580     {
 581         PLUGIN* plugin;
 582 
 583         // private assignment operator so it's illegal

 584         RELEASER& operator=( RELEASER& aOther ) { return *this; }
 585 
 586         // private copy constructor so it's illegal

 587         RELEASER( const RELEASER& aOther ) {}
 588 
 589     public:
 590         RELEASER( PLUGIN* aPlugin = NULL ) :
 591             plugin( aPlugin )
 592         {
 593         }
 594 
 595         ~RELEASER()
 596         {
 597             if( plugin )
 598                 release();
 599         }
 600 
 601         void release()
 602         {
 603             IO_MGR::PluginRelease( plugin );
 604             plugin = NULL;
 605         }
 606 
 607         void set( PLUGIN* aPlugin )
 608         {
 609             if( plugin )
 610                 release();
 611             plugin = aPlugin;
 612         }
 613 
 614         operator PLUGIN* () const
 615         {
 616             return plugin;
 617         }
 618 
 619         PLUGIN* operator -> () const
 620         {
 621             return plugin;
 622         }
 623     };
