  30 class BOARD;
  31 class BOARD_DESIGN_SETTINGS;
  32 
  33 namespace PNS {
  34 
  35 class ITEM;
  36 
  37 class SIZES_SETTINGS {
  38 
  39 public:
  40     SIZES_SETTINGS()
  41             : m_trackWidth( 155000 ),
  42               m_diffPairWidth( 125000 ),
  43               m_diffPairGap( 180000 ),
  44               m_diffPairViaGap( 180000 ),
  45               m_viaDiameter( 600000 ),
  46               m_viaDrill( 250000 ),
  47               m_diffPairViaGapSameAsTraceGap( true ),
  48               m_viaType( VIATYPE::THROUGH ){};
  49 
  50     ~SIZES_SETTINGS() {};
  51 
  52     void Init( BOARD* aBoard, ITEM* aStartItem = NULL, int aNet = -1 );
  53     void ImportCurrent( BOARD_DESIGN_SETTINGS& aSettings );
  54 
  55     void ClearLayerPairs();
  56     void AddLayerPair( int aL1, int aL2 );
  57 
  58     int TrackWidth() const { return m_trackWidth; }
  59     void SetTrackWidth( int aWidth ) { m_trackWidth = aWidth; }
  60 
  61     int DiffPairWidth() const { return m_diffPairWidth; }
  62     int DiffPairGap() const { return m_diffPairGap; }
  63 
  64     int DiffPairViaGap() const {
  65         if( m_diffPairViaGapSameAsTraceGap )
  66             return m_diffPairGap;
  67         else
  68             return m_diffPairViaGap;
  69     }
  70 
  71     bool DiffPairViaGapSameAsTraceGap() const { return m_diffPairViaGapSameAsTraceGap; }
  72 
  73     void SetDiffPairWidth( int aWidth ) { m_diffPairWidth = aWidth; }
  74     void SetDiffPairGap( int aGap ) { m_diffPairGap = aGap; }
  75     void SetDiffPairViaGapSameAsTraceGap ( bool aEnable ) { m_diffPairViaGapSameAsTraceGap = aEnable; }
  76     void SetDiffPairViaGap( int aGap ) { m_diffPairViaGap = aGap; }
  77 
  78     int ViaDiameter() const { return m_viaDiameter; }
  79     void SetViaDiameter( int aDiameter ) { m_viaDiameter = aDiameter; }
  80 
  81     int ViaDrill() const { return m_viaDrill; }
  82     void SetViaDrill( int aDrill ) { m_viaDrill = aDrill; }
  83 
  84     OPT<int> PairedLayer( int aLayerId )
  85     {
  86         if( m_layerPairs.find(aLayerId) == m_layerPairs.end() )
  87             return OPT<int>();
  88 
  89         return m_layerPairs[aLayerId];
  90     }
  91 
  92     int GetLayerTop() const;
  93     int GetLayerBottom() const;
  94 
  95     void SetViaType( VIATYPE aViaType )
  96     {
  97         m_viaType = aViaType;
  98     }
  99 
 100     VIATYPE ViaType() const
 101     {
 102         return m_viaType;
 103     }
 104 
 105 private:
 106 
 107     int inheritTrackWidth( ITEM* aItem );
 108 
 109     int m_trackWidth;
 110     int m_diffPairWidth;
 111     int m_diffPairGap;
 112     int m_diffPairViaGap;
 113     int m_viaDiameter;
 114     int m_viaDrill;
 115 
 116     bool m_diffPairViaGapSameAsTraceGap;
 117 
 118     VIATYPE m_viaType;
 119 
 120     std::map<int, int> m_layerPairs;
 121 };
