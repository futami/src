  42 /**

  43  * DRAGGER

  44  *

  45  * Via, segment and corner dragging algorithm.

  46  */
  47 class DRAGGER : public DRAG_ALGO
  48 {
  49 public:
  50      DRAGGER( ROUTER* aRouter );
  51     ~DRAGGER();
  52 
  53     /**

  54      * Function Start()

  55      *

  56      * Starts routing a single track at point aP, taking item aStartItem as anchor

  57      * (unless NULL). Returns true if a dragging operation has started.

  58      */
  59     virtual bool Start( const VECTOR2I& aP, ITEM_SET& aPrimitives ) override;
  60 
  61     /**

  62      * Function Drag()

  63      *

  64      * Drags the current segment/corner/via to the point aP.

  65      * @return true, if dragging finished with success.

  66      */
  67     bool Drag( const VECTOR2I& aP ) override;
  68 
  69     /**

  70      * Function FixRoute()

  71      *

  72      * Checks if the result of current dragging operation is correct

  73      * and eventually commits it to the world.

  74      * @return true, if dragging finished with success.

  75      */
  76     bool FixRoute() override;
  77 
  78     /**

  79      * Function CurrentNode()

  80      *

  81      * Returns the most recent world state, including all

  82      * items changed due to dragging operation.

  83      */
  84     NODE* CurrentNode() const override;
  85 
  86     /**

  87      * Function Traces()

  88      *

  89      * Returns the set of dragged items.

  90      */
  91     const ITEM_SET Traces() override;
  92 
  93     void SetMode( int aDragMode ) override;
  94 
  95 private:
  96 
  97     const ITEM_SET findViaFanoutByHandle ( NODE *aNode, const VIA_HANDLE& handle );
  98     
  99     bool dragMarkObstacles( const VECTOR2I& aP );
 100     bool dragShove(const VECTOR2I& aP );
 101     bool dragWalkaround(const VECTOR2I& aP );
 102     bool startDragSegment( const VECTOR2D& aP, SEGMENT* aSeg );
 103     bool startDragArc( const VECTOR2D& aP, ARC* aArc );
 104     bool startDragVia( VIA* aVia );
 105     void dragViaMarkObstacles( const VIA_HANDLE& aHandle, NODE* aNode, const VECTOR2I& aP );
 106     void dragViaWalkaround( const VIA_HANDLE& aHandle, NODE* aNode, const VECTOR2I& aP );
 107     void optimizeAndUpdateDraggedLine( LINE& dragged, const VECTOR2I& aP );
 108 
 109     VIA_HANDLE m_initialVia;
 110     VIA_HANDLE m_draggedVia;
 111 
 112     NODE*    m_lastNode;
 113     int      m_mode;
 114     LINE     m_draggedLine;
 115     //VIA*     m_draggedVia;

 116     //LINE     m_lastValidDraggedLine;

 117     std::unique_ptr<SHOVE> m_shove;
 118     int      m_draggedSegmentIndex;
 119     bool     m_dragStatus;

 124     ITEM_SET m_draggedItems;

 126     ///< If true, moves the connection lines without maintaining 45° corners

 127     bool     m_freeAngleMode;
