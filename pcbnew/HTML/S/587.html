  33 static void onLayer( LIBEVAL::CONTEXT* aCtx, void *self )
  34 {
  35     LIBEVAL::VALUE* arg = aCtx->Pop();
  36     LIBEVAL::VALUE* result = aCtx->AllocValue();
  37 
  38     result->Set( 0.0 );
  39     aCtx->Push( result );
  40 
  41     if( !arg )
  42     {
  43         aCtx->ReportError( _( "Missing argument to 'onLayer()'" ) );
  44         return;
  45     }
  46 
  47     wxString     layerName = arg->AsString();
  48     PCB_LAYER_ID layer = ENUM_MAP<PCB_LAYER_ID>::Instance().ToEnum( layerName );
  49 
  50     if( layer == UNDEFINED_LAYER )
  51     {
  52         aCtx->ReportError( wxString::Format( _( "Unrecognized layer '%s' " ), layerName ) );
  53         return;
  54     }
  55 
  56     PCB_EXPR_VAR_REF* vref = static_cast<PCB_EXPR_VAR_REF*>( self );
  57     BOARD_ITEM*       item = vref ? vref->GetObject( aCtx ) : nullptr;
  58 
  59     if( item && item->IsOnLayer( layer ) )
  60         result->Set( 1.0 );
  61 }

  64 static void isPlated( LIBEVAL::CONTEXT* aCtx, void *self )
  65 {
  66     LIBEVAL::VALUE* result = aCtx->AllocValue();
  67 
  68     result->Set( 0.0 );
  69     aCtx->Push( result );
  70 
  71     PCB_EXPR_VAR_REF* vref = static_cast<PCB_EXPR_VAR_REF*>( self );
  72     BOARD_ITEM*       item = vref ? vref->GetObject( aCtx ) : nullptr;
  73     D_PAD*            pad = dynamic_cast<D_PAD*>( item );
  74 
  75     if( pad && pad->GetAttribute() == PAD_ATTRIB_STANDARD )
  76         result->Set( 1.0 );
  77 }

 102 LIBEVAL::VALUE PCB_EXPR_VAR_REF::GetValue( LIBEVAL::CONTEXT* aCtx )
 103 {
 104     BOARD_ITEM* item  = const_cast<BOARD_ITEM*>( GetObject( aCtx ) );
 105     auto        it = m_matchingTypes.find( TYPE_HASH( *item ) );
 106 
 107     if( it == m_matchingTypes.end() )
 108     {
 109         // Don't force user to type "A.Type == 'via' && A.Via_Type == 'buried'" when the

 110         // simplier "A.Via_Type == 'buried'" is perfectly clear.  Instead, return an undefined

 111         // value when the property doesn't appear on a particular object.

 112 
 113         return LIBEVAL::VALUE( "UNDEFINED" );
 114     }
 115     else
 116     {
 117         if( m_type == LIBEVAL::VT_NUMERIC )
 118             return LIBEVAL::VALUE( (double) item->Get<int>( it->second ) );
 119         else
 120         {
 121             wxString str;
 122 
 123             if( !m_isEnum )
 124             {
 125                 //printf("item %p Get string '%s'\n", item, (const char*) it->second->Name().c_str() );

 126                 str = item->Get<wxString>( it->second );
 127             }
 128             else
 129             {
 130                 const wxAny& any = item->Get( it->second );
 131                 any.GetAs<wxString>( &str );
 132                 //printf("item %p get enum: '%s'\n", item , (const char*) str.c_str() );

 133             }
 134 
 135             return LIBEVAL::VALUE( str );
 136         }
 137     }
 138 }

 152 LIBEVAL::VAR_REF* PCB_EXPR_UCODE::createVarRef( LIBEVAL::COMPILER *aCompiler,
 153                                                 const std::string& aVar,
 154                                                 const std::string& aField )
 155 {
 156     PROPERTY_MANAGER& propMgr = PROPERTY_MANAGER::Instance();
 157     PCB_EXPR_VAR_REF* vref = nullptr;
 158 
 159     if( aVar == "A" )
 160     {
 161         vref = new PCB_EXPR_VAR_REF( 0 );
 162     }
 163     else if( aVar == "B" )
 164     {
 165         vref = new PCB_EXPR_VAR_REF( 1 );
 166     }
 167     else
 168     {
 169         aCompiler->ReportError( "var" );
 170         return vref;
 171     }
 172 
 173     if( aField.empty() ) // return reference to base object

 174         return vref;
 175 
 176     std::string field( aField );
 177     std::replace( field.begin(), field.end(), '_', ' ');
 178 
 179     for( const PROPERTY_MANAGER::CLASS_INFO& cls : propMgr.GetAllClasses() )
 180     {
 181         if( propMgr.IsOfType( cls.type, TYPE_HASH( BOARD_ITEM ) ) )
 182         {
 183             PROPERTY_BASE* prop = propMgr.GetProperty( cls.type, field );
 184 
 185             if( prop )
 186             {
 187                 //printf("Field '%s' class %s ptr %p haschoices %d typeid %s\n", field.c_str(), (const char *) cls.name.c_str(), prop, !!prop->HasChoices(), typeid(*prop).name() );

 188                 vref->AddAllowedClass( cls.type, prop );
 189 
 190                 if( prop->TypeHash() == TYPE_HASH( int ) )
 191                 {
 192                     vref->SetType( LIBEVAL::VT_NUMERIC );
 193                 }
 194                 else if( prop->TypeHash() == TYPE_HASH( wxString ) )
 195                 {
 196                     vref->SetType( LIBEVAL::VT_STRING );
 197                 }
 198                 else if ( prop->HasChoices() )
 199                 {   // it's an enum, we treat it as string

 200                     vref->SetType( LIBEVAL::VT_STRING );
 201                     vref->SetIsEnum ( true );
 202                 }
 203                 else
 204                 {
 205                     (void) 0; // should we do anything here?

 206                     //msg.Printf("Unrecognized type for property '%s'", field.c_str() );

 207                     //aCompiler->ReportError( (const char*) msg.c_str() );

 208                 }
 209             }
 210         }
 211     }
 212 
 213     if( vref->GetType() == LIBEVAL::VT_UNDEFINED )
 214         aCompiler->ReportError( "field" );
 215 
 216     return vref;
 217 }

 220 class PCB_UNIT_RESOLVER : public LIBEVAL::UNIT_RESOLVER
 221 {
 222 public:
 223     virtual ~PCB_UNIT_RESOLVER()
 224     {
 225     }
 226 
 227     virtual const std::vector<std::string>& GetSupportedUnits() const override
 228     {
 229         static const std::vector<std::string> pcbUnits = { "mil", "mm", "in" };
 230 
 231         return pcbUnits;
 232     }
 233 
 234     virtual double Convert( const std::string& aString, int unitId ) const override
 235     {
 236         double v = atof( aString.c_str() );
 237 
 238         switch( unitId )
 239         {
 240         case 0:  return Mils2iu( v );
 241         case 1:  return Millimeter2iu( v );
 242         case 2:  return Mils2iu( v * 1000.0 );
 243         default: return v;
 244         }
 245     };
 246 };

 265 bool PCB_EXPR_EVALUATOR::Evaluate( const wxString& aExpr )
 266 {
 267     PCB_EXPR_UCODE   ucode;
 268     LIBEVAL::CONTEXT preflightContext;
 269 
 270     if( !m_compiler.Compile( (const char*) aExpr.c_str(), &ucode, &preflightContext ) )
 271     {
 272         m_errorStatus = m_compiler.GetErrorStatus();
 273         return false;
 274     }
 275 
 276 // fixme: handle error conditions

 277     LIBEVAL::CONTEXT ctx;
 278     LIBEVAL::VALUE*  result = ucode.Run( &ctx );
 279 
 280     if( result->GetType() == LIBEVAL::VT_NUMERIC )
 281         m_result = KiROUND( result->AsDouble() );
 282 
 283     return true;
 284 }
