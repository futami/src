  58 void filterNetClass( const BOARD& aBoard, NETCLASS& aNetClass )
  59 {
  60     auto connectivity = aBoard.GetConnectivity();
  61 
  62     for( NETCLASS::iterator it = aNetClass.begin(); it != aNetClass.end(); )
  63     {
  64         NETINFO_ITEM* netinfo = aBoard.FindNet( *it );
  65 
  66         if( netinfo && connectivity->GetNodeCount( netinfo->GetNet() ) <= 0 ) // hopefully there are no nets with negative

  67             aNetClass.Remove( it++ );                  // node count, but you never know..

  68         else
  69             ++it;
  70     }
  71 }

 182 void FP_CACHE::Save( MODULE* aModule )
 183 {
 184     m_cache_timestamp = 0;
 185 
 186     if( !m_lib_path.DirExists() && !m_lib_path.Mkdir() )
 187     {
 188         THROW_IO_ERROR( wxString::Format( _( "Cannot create footprint library path \"%s\"" ),
 189                                           m_lib_raw_path ) );
 190     }
 191 
 192     if( !m_lib_path.IsDirWritable() )
 193     {
 194         THROW_IO_ERROR( wxString::Format( _( "Footprint library path \"%s\" is read only" ),
 195                                           m_lib_raw_path ) );
 196     }
 197 
 198     for( MODULE_ITER it = m_modules.begin();  it != m_modules.end();  ++it )
 199     {
 200         if( aModule && aModule != it->second->GetModule() )
 201             continue;
 202 
 203         WX_FILENAME fn = it->second->GetFileName();
 204 
 205         wxString tempFileName =
 206 #ifdef USE_TMP_FILE
 207         wxFileName::CreateTempFileName( fn.GetPath() );
 208 #else
 209         fn.GetFullPath();
 210 #endif
 211         // Allow file output stream to go out of scope to close the file stream before

 212         // renaming the file.

 213         {
 214             wxLogTrace( traceKicadPcbPlugin, wxT( "Creating temporary library file %s" ),
 215                         GetChars( tempFileName ) );
 216 
 217             FILE_OUTPUTFORMATTER formatter( tempFileName );
 218 
 219             m_owner->SetOutputFormatter( &formatter );
 220             m_owner->Format( (BOARD_ITEM*) it->second->GetModule() );
 221         }
 222 
 223 #ifdef USE_TMP_FILE
 224         wxRemove( fn.GetFullPath() );     // it is not an error if this does not exist

 225 
 226         // Even on linux you can see an _intermittent_ error when calling wxRename(),

 227         // and it is fully inexplicable.  See if this dodges the error.

 228         wxMilliSleep( 250L );
 229 
 230         if( !wxRenameFile( tempFileName, fn.GetFullPath() ) )
 231         {
 232             wxString msg = wxString::Format(
 233                     _( "Cannot rename temporary file \"%s\" to footprint library file \"%s\"" ),
 234                     GetChars( tempFileName ),
 235                     GetChars( fn.GetFullPath() )
 236                     );
 237             THROW_IO_ERROR( msg );
 238         }
 239 #endif
 240         m_cache_timestamp += fn.GetTimestamp();
 241     }
 242 
 243     m_cache_timestamp += m_lib_path.GetModificationTime().GetValue().GetValue();
 244 
 245     // If we've saved the full cache, we clear the dirty flag.

 246     if( !aModule )
 247         m_cache_dirty = false;
 248 }

 251 void FP_CACHE::Load()
 252 {
 253     m_cache_dirty = false;
 254     m_cache_timestamp = 0;
 255 
 256     wxDir dir( m_lib_raw_path );
 257 
 258     if( !dir.IsOpened() )
 259     {
 260         wxString msg = wxString::Format( _( "Footprint library path '%s' does not exist "
 261                                             "(or is not a directory)." ),
 262                                          m_lib_raw_path );
 263         THROW_IO_ERROR( msg );
 264     }
 265 
 266     wxString fullName;
 267     wxString fileSpec = wxT( "*." ) + KiCadFootprintFileExtension;
 268 
 269     // wxFileName construction is egregiously slow.  Construct it once and just swap out

 270     // the filename thereafter.

 271     WX_FILENAME fn( m_lib_raw_path, wxT( "dummyName" ) );
 272 
 273     if( dir.GetFirst( &fullName, fileSpec ) )
 274     {
 275         wxString cacheError;
 276 
 277         do
 278         {
 279             fn.SetFullName( fullName );
 280 
 281             // Queue I/O errors so only files that fail to parse don't get loaded.

 282             try
 283             {
 284                 FILE_LINE_READER    reader( fn.GetFullPath() );
 285 
 286                 m_owner->m_parser->SetLineReader( &reader );
 287 
 288                 MODULE*     footprint = (MODULE*) m_owner->m_parser->Parse();
 289                 wxString    fpName = fn.GetName();
 290 
 291                 footprint->SetFPID( LIB_ID( wxEmptyString, fpName ) );
 292                 m_modules.insert( fpName, new FP_CACHE_ITEM( footprint, fn ) );
 293 
 294                 m_cache_timestamp += fn.GetTimestamp();
 295             }
 296             catch( const IO_ERROR& ioe )
 297             {
 298                 if( !cacheError.IsEmpty() )
 299                     cacheError += "\n\n";
 300 
 301                 cacheError += ioe.What();
 302             }
 303         } while( dir.GetNext( &fullName ) );
 304 
 305         if( !cacheError.IsEmpty() )
 306             THROW_IO_ERROR( cacheError );
 307     }
 308 }

 311 void FP_CACHE::Remove( const wxString& aFootprintName )
 312 {
 313     MODULE_CITER it = m_modules.find( aFootprintName );
 314 
 315     if( it == m_modules.end() )
 316     {
 317         wxString msg = wxString::Format( _( "library \"%s\" has no footprint \"%s\" to delete" ),
 318                                          m_lib_raw_path,
 319                                          aFootprintName );
 320         THROW_IO_ERROR( msg );
 321     }
 322 
 323     // Remove the module from the cache and delete the module file from the library.

 324     wxString fullPath = it->second->GetFileName().GetFullPath();
 325     m_modules.erase( aFootprintName );
 326     wxRemoveFile( fullPath );
 327 }

 542         B_Adhes,        // 32

 543         F_Adhes,

 547         F_SilkS,
 548         B_Mask,
 549         F_Mask,

 555         Margin,
 556         B_CrtYd,
 557         F_CrtYd,
 558         B_Fab,
 559         F_Fab
 560     };
 561 
 562     for( LSEQ seq = aBoard->GetEnabledLayers().Seq( non_cu, arrayDim( non_cu ) );  seq;  ++seq )

 602 void PCB_IO::format( BOARD* aBoard, int aNestLevel ) const
 603 {
 604 
 605     std::set<BOARD_ITEM*, BOARD_ITEM::ptr_cmp> sorted_modules( aBoard->Modules().begin(),
 606             aBoard->Modules().end() );
 607     std::set<BOARD_ITEM*, BOARD_ITEM::ptr_cmp> sorted_drawings( aBoard->Drawings().begin(),
 608             aBoard->Drawings().end() );
 609     std::set<TRACK*, TRACK::cmp_tracks> sorted_tracks( aBoard->Tracks().begin(),
 610             aBoard->Tracks().end() );
 611     std::set<BOARD_ITEM*, BOARD_ITEM::ptr_cmp> sorted_zones( aBoard->Zones().begin(),
 612             aBoard->Zones().end() );
 613 
 614     formatHeader( aBoard, aNestLevel );
 615 
 616     // Save the modules.

 617     for( auto module : sorted_modules )
 618     {
 619         Format( module, aNestLevel );
 620         m_out->Print( 0, "\n" );
 621     }
 622 
 623     // Save the graphical items on the board (not owned by a module)

 624     for( auto item : sorted_drawings )
 625         Format( item, aNestLevel );
 626 
 627     if( sorted_drawings.size() )
 628         m_out->Print( 0, "\n" );
 629 
 630     // Do not save MARKER_PCBs, they can be regenerated easily.

 631 
 632     // Save the tracks and vias.

 633     for( auto track : sorted_tracks )
 634         Format( track, aNestLevel );

1075 void PCB_IO::formatLayers( LSET aLayerMask, int aNestLevel ) const
1076 {
1077     std::string  output;
1078 
1079     if( aNestLevel == 0 )
1080         output += ' ';
1081 
1082     output += "(layers";
1083 
1084     static const LSET cu_all( LSET::AllCuMask() );
1085     static const LSET fr_bk( 2, B_Cu,       F_Cu );
1086     static const LSET adhes( 2, B_Adhes,    F_Adhes );
1087     static const LSET paste( 2, B_Paste,    F_Paste );
1088     static const LSET silks( 2, B_SilkS,    F_SilkS );
1089     static const LSET mask(  2, B_Mask,     F_Mask );
1090     static const LSET crt_yd(2, B_CrtYd,    F_CrtYd );
1091     static const LSET fab(   2, B_Fab,      F_Fab );
1092 
1093     LSET cu_mask = cu_all;
1094 
1095     // output copper layers first, then non copper

1096 
1097     if( ( aLayerMask & cu_mask ) == cu_mask )
1098     {
1099         output += " *.Cu";
1100         aLayerMask &= ~cu_all;          // clear bits, so they are not output again below

1101     }
1102     else if( ( aLayerMask & cu_mask ) == fr_bk )
1103     {
1104         output += " F&B.Cu";
1105         aLayerMask &= ~fr_bk;
1106     }
1107 
1108     if( ( aLayerMask & adhes ) == adhes )
1109     {
1110         output += " *.Adhes";
1111         aLayerMask &= ~adhes;
1112     }
1113 
1114     if( ( aLayerMask & paste ) == paste )
1115     {
1116         output += " *.Paste";
1117         aLayerMask &= ~paste;
1118     }
1119 
1120     if( ( aLayerMask & silks ) == silks )
1121     {
1122         output += " *.SilkS";
1123         aLayerMask &= ~silks;
1124     }
1125 
1126     if( ( aLayerMask & mask ) == mask )
1127     {
1128         output += " *.Mask";
1129         aLayerMask &= ~mask;
1130     }
1131 
1132     if( ( aLayerMask & crt_yd ) == crt_yd )
1133     {
1134         output += " *.CrtYd";
1135         aLayerMask &= ~crt_yd;
1136     }
1137 
1138     if( ( aLayerMask & fab ) == fab )
1139     {
1140         output += " *.Fab";
1141         aLayerMask &= ~fab;
1142     }
1143 
1144     // output any individual layers not handled in wildcard combos above

1145 
1146     wxString layerName;
1147 
1148     for( LAYER_NUM layer = 0; layer < PCB_LAYER_ID_COUNT; ++layer )
1149     {
1150         if( aLayerMask[layer] )
1151         {
1152             if( m_board && !( m_ctl & CTL_STD_LAYER_NAMES ) )
1153                 layerName = m_board->GetLayerName( PCB_LAYER_ID( layer ) );
1154 
1155             else    // I am being called from FootprintSave()

1156                 layerName = BOARD::GetStandardLayerName( PCB_LAYER_ID( layer ) );
1157 
1158             output += ' ';
1159             output += m_out->Quotew( layerName );
1160         }
1161     }
1162 
1163     m_out->Print( aNestLevel, "%s)", output.c_str() );
1164 }

1485     case TEXTE_MODULE::TEXT_is_VALUE:     type = "value";       break;
1486     case TEXTE_MODULE::TEXT_is_DIVERS:    type = "user";

1494     // Due to Pcbnew history, fp_text angle is saved as an absolute on screen angle,

1495     // but internally the angle is held relative to its parent footprint.  parent

1496     // may be NULL when saving a footprint outside a BOARD.

1497     double   orient = aText->GetTextAngle();
1498     MODULE*  parent = (MODULE*) aText->GetParent();
1499

1501     {
1502         // GetTextAngle() is always in -360..+360 range because of

1507 #if 0
1508         // This one could be considered reasonable if you like positive angles

1509         // in your board text.

1510         orient = NormalizeAnglePos( orient + parent->GetOrientation() );
1511 #else
1512         // Choose compatibility for now, even though this is only a 720 degree clamp

1513         // with two possible values for every angle.

1514         orient = NormalizeAngle360Min( orient + parent->GetOrientation() );
1515 #endif
1516     }
1517 
1518     if( orient != 0.0 )
1519         m_out->Print( 0, " %s", FormatAngle( orient ).c_str() );
1520 
1521     if( !aText->IsKeepUpright() )
1522         m_out->Print( 0, " unlocked" );

1526

1528         m_out->Print( 0, " hide" );
1529 
1530     m_out->Print( 0, "\n" );
1531 
1532     aText->EDA_TEXT::Format( m_out, aNestLevel, m_ctl | CTL_OMIT_HIDE );
1533 
1534     m_out->Print( aNestLevel + 1, "(tstamp %s)\n", TO_UTF8( aText->m_Uuid.AsString() ) );
1535 
1536     m_out->Print( aNestLevel, ")\n" );
