 145 void BOARD::SetProject( PROJECT* aProject )
 146 {
 147     m_project = aProject;
 148 
 149     if( aProject )
 150     {
 151         PROJECT_FILE& project = aProject->GetProjectFile();
 152 
 153         // Link the design settings object to the project file

 154         project.m_BoardSettings = &GetDesignSettings();
 155 
 156         // Set parent, which also will load the values from JSON stored in the project

 157         project.m_BoardSettings->SetParent( &project );
 158 
 159         // The DesignSettings' netclasses pointer will be pointing to its internal netclasses

 160         // list at this point. If we loaded anything into it from a legacy board file then we

 161         // want to transfer it over to the project netclasses list.

 162         if( m_LegacyNetclassesLoaded )
 163             project.NetSettings().m_NetClasses = GetDesignSettings().GetNetClasses();
 164 
 165         // Now update the DesignSettings' netclass pointer ot point into the project.

 166         GetDesignSettings().SetNetClasses( &project.NetSettings().m_NetClasses );
 167     }
 168 }

 269 const PCB_LAYER_ID BOARD::GetLayerID( const wxString& aLayerName ) const
 270 {
 271 
 272     // Look for the BOARD specific copper layer names

 273     for( LAYER_NUM layer = 0; layer < PCB_LAYER_ID_COUNT; ++layer )
 274     {
 275         if ( IsCopperLayer( layer ) && ( m_Layer[ layer ].m_name == aLayerName ) )
 276         {
 277             return ToLAYER_ID( layer );
 278         }
 279     }
 280 
 281     // Otherwise fall back to the system standard layer names

 282     for( LAYER_NUM layer = 0; layer < PCB_LAYER_ID_COUNT; ++layer )
 283     {
 284         if( GetStandardLayerName( ToLAYER_ID( layer ) ) == aLayerName )
 285         {
 286             return ToLAYER_ID( layer );
 287         }
 288     }
 289 
 290     return UNDEFINED_LAYER;
 291 }

 309 bool BOARD::SetLayerName( PCB_LAYER_ID aLayer, const wxString& aLayerName )
 310 {
 311     if( !IsCopperLayer( aLayer ) )
 312         return false;
 313 
 314     if( aLayerName == wxEmptyString )
 315         return false;
 316 
 317     // no quote chars in the name allowed

 318     if( aLayerName.Find( wxChar( '"' ) ) != wxNOT_FOUND )
 319         return false;
 320 
 321     wxString nameTemp = aLayerName;
 322 
 323     // replace any spaces with underscores before we do any comparing

 324     nameTemp.Replace( wxT( " " ), wxT( "_" ) );
 325 
 326     if( IsLayerEnabled( aLayer ) )
 327     {
 328         m_Layer[aLayer].m_name = nameTemp;
 329         return true;
 330     }
 331 
 332     return false;
 333 }

 489 {

 541 void BOARD::Add( BOARD_ITEM* aBoardItem, ADD_MODE aMode )
 542 {
 543     if( aBoardItem == NULL )
 544     {
 545         wxFAIL_MSG( wxT( "BOARD::Add() param error: aBoardItem NULL" ) );
 546         return;
 547     }
 548 
 549     switch( aBoardItem->Type() )
 550     {
 551     case PCB_NETINFO_T:
 552         m_NetInfo.AppendNet( (NETINFO_ITEM*) aBoardItem );
 553         break;
 554 
 555     // this one uses a vector

 556     case PCB_MARKER_T:
 557         m_markers.push_back( (MARKER_PCB*) aBoardItem );
 558         break;
 559 
 560     // this one uses a vector

 561     case PCB_ZONE_AREA_T:
 562         m_ZoneDescriptorList.push_back( (ZONE_CONTAINER*) aBoardItem );
 563         break;
 564 
 565     case PCB_TRACE_T:
 566     case PCB_VIA_T:
 567     case PCB_ARC_T:
 568 
 569         // N.B. This inserts a small memory leak as we lose the

 570         if( !IsCopperLayer( aBoardItem->GetLayer() ) )
 571         {
 572             wxFAIL_MSG( wxT( "BOARD::Add() Cannot place Track on non-copper layer" ) );
 573             return;
 574         }
 575 
 576         if( aMode == ADD_MODE::APPEND )
 577             m_tracks.push_back( static_cast<TRACK*>( aBoardItem ) );
 578         else
 579             m_tracks.push_front( static_cast<TRACK*>( aBoardItem ) );
 580 
 581         break;
 582 
 583     case PCB_MODULE_T:
 584         if( aMode == ADD_MODE::APPEND )
 585             m_modules.push_back( (MODULE*) aBoardItem );
 586         else
 587             m_modules.push_front( (MODULE*) aBoardItem );
 588 
 589         break;
 590 
 591     case PCB_DIMENSION_T:
 592     case PCB_LINE_T:
 593     case PCB_TEXT_T:
 594     case PCB_TARGET_T:
 595         if( aMode == ADD_MODE::APPEND )
 596             m_drawings.push_back( aBoardItem );
 597         else
 598             m_drawings.push_front( aBoardItem );
 599 
 600         break;
 601 
 602     // other types may use linked list

 603     default:
 604         {
 605             wxString msg;
 606             msg.Printf( wxT( "BOARD::Add() needs work: BOARD_ITEM type (%d) not handled" ),
 607                         aBoardItem->Type() );
 608             wxFAIL_MSG( msg );
 609             return;
 610         }
 611         break;
 612     }
 613 
 614     aBoardItem->SetParent( this );
 615     aBoardItem->ClearEditFlags();
 616     m_connectivity->Add( aBoardItem );
 617 
 618     InvokeListeners( &BOARD_LISTENER::OnBoardItemAdded, *this, aBoardItem );
 619 }

 622 void BOARD::Remove( BOARD_ITEM* aBoardItem )
 623 {
 624     // find these calls and fix them!  Don't send me no stinking' NULL.

 625     wxASSERT( aBoardItem );
 626 
 627     switch( aBoardItem->Type() )
 628     {
 629     case PCB_NETINFO_T:
 630     {
 631         NETINFO_ITEM* item = (NETINFO_ITEM*) aBoardItem;
 632         m_NetInfo.RemoveNet( item );
 633         break;
 634     }
 635 
 636     case PCB_MARKER_T:
 637 
 638         // find the item in the vector, then remove it

 639         for( unsigned i = 0; i<m_markers.size(); ++i )
 640         {
 641             if( m_markers[i] == (MARKER_PCB*) aBoardItem )
 642             {
 643                 m_markers.erase( m_markers.begin() + i );
 644                 break;
 645             }
 646         }
 647 
 648         break;
 649 
 650     case PCB_ZONE_AREA_T:    // this one uses a vector

 651         // find the item in the vector, then delete then erase it.

 652         for( unsigned i = 0; i<m_ZoneDescriptorList.size(); ++i )
 653         {
 654             if( m_ZoneDescriptorList[i] == (ZONE_CONTAINER*) aBoardItem )
 655             {
 656                 m_ZoneDescriptorList.erase( m_ZoneDescriptorList.begin() + i );
 657                 break;
 658             }
 659         }
 660         break;
 661 
 662     case PCB_MODULE_T:
 663         m_modules.erase( std::remove_if( m_modules.begin(), m_modules.end(),
 664                                          [aBoardItem]( BOARD_ITEM* aItem )
 665                                          {
 666                                              return aItem == aBoardItem;
 667                                          } ) );
 668         break;
 669 
 670     case PCB_TRACE_T:
 671     case PCB_ARC_T:
 672     case PCB_VIA_T:
 673         m_tracks.erase( std::remove_if( m_tracks.begin(), m_tracks.end(),
 674                                         [aBoardItem]( BOARD_ITEM* aItem )
 675                                         {
 676                                             return aItem == aBoardItem;
 677                                         } ) );
 678         break;
 679 
 680     case PCB_DIMENSION_T:
 681     case PCB_LINE_T:
 682     case PCB_TEXT_T:
 683     case PCB_TARGET_T:
 684         m_drawings.erase( std::remove_if( m_drawings.begin(), m_drawings.end(),

 921 SEARCH_RESULT BOARD::Visit( INSPECTOR inspector, void* testData, const KICAD_T scanTypes[] )
 922 {
 923     KICAD_T        stype;
 924     SEARCH_RESULT  result = SEARCH_RESULT::CONTINUE;
 925     const KICAD_T* p    = scanTypes;
 926     bool           done = false;
 927 
 928 #if 0 && defined(DEBUG)
 929     std::cout << GetClass().mb_str() << ' ';
 930 #endif
 931 
 932     while( !done )
 933     {
 934         stype = *p;
 935 
 936         switch( stype )
 937         {
 938         case PCB_T:
 939             result = inspector( this, testData );  // inspect me

 940             // skip over any types handled in the above call.

 941             ++p;
 942             break;
 943 
 944         /*  Instances of the requested KICAD_T live in a list, either one

 945          *   that I manage, or that my modules manage.  If it's a type managed

 946          *   by class MODULE, then simply pass it on to each module's

 947          *   MODULE::Visit() function by way of the

 948          *   IterateForward( m_Modules, ... ) call.

 949          */
 950 
 951         case PCB_MODULE_T:
 952         case PCB_PAD_T:
 953         case PCB_MODULE_TEXT_T:
 954         case PCB_MODULE_EDGE_T:
 955         case PCB_MODULE_ZONE_AREA_T:
 956 
 957             // this calls MODULE::Visit() on each module.

 958             result = IterateForward<MODULE*>( m_modules, inspector, testData, p );
 959 
 960             // skip over any types handled in the above call.

 961             for( ; ; )
 962             {
 963                 switch( stype = *++p )
 964                 {
 965                 case PCB_MODULE_T:
 966                 case PCB_PAD_T:
 967                 case PCB_MODULE_TEXT_T:
 968                 case PCB_MODULE_EDGE_T:
 969                 case PCB_MODULE_ZONE_AREA_T:
 970                     continue;
 971 
 972                 default:
 973                     ;
 974                 }
 975 
 976                 break;
 977             }
 978 
 979             break;
 980 
 981         case PCB_LINE_T:
 982         case PCB_TEXT_T:
 983         case PCB_DIMENSION_T:
 984         case PCB_TARGET_T:
 985             result = IterateForward<BOARD_ITEM*>( m_drawings, inspector, testData, p );
 986 
 987             // skip over any types handled in the above call.

 988             for( ; ; )
 989             {
 990                 switch( stype = *++p )
 991                 {
 992                 case PCB_LINE_T:
 993                 case PCB_TEXT_T:
 994                 case PCB_DIMENSION_T:
 995                 case PCB_TARGET_T:
 996                     continue;
 997 
 998                 default:
 999                     ;
1000                 }
1001 
1002                 break;
1003             }
1004 
1005             ;
1006             break;
1007 
1008 #if 0   // both these are on same list, so we must scan it twice in order

1009         // to get VIA priority, using new #else code below.

1010         // But we are not using separate lists for TRACKs and VIA, because

1011         // items are ordered (sorted) in the linked

1012         // list by netcode AND by physical distance:

1013         // when created, if a track or via is connected to an existing track or

1014         // via, it is put in linked list after this existing track or via

1015         // So usually, connected tracks or vias are grouped in this list

1016         // So the algorithm (used in ratsnest computations) which computes the

1017         // track connectivity is faster (more than 100 time regarding to

1018         // a non ordered list) because when it searches for a connection, first

1019         // it tests the near (near in term of linked list) 50 items

1020         // from the current item (track or via) in test.

1021         // Usually, because of this sort, a connected item (if exists) is

1022         // found.

1023         // If not found (and only in this case) an exhaustive (and time

1024         // consuming) search is made, but this case is statistically rare.

1025         case PCB_VIA_T:
1026         case PCB_TRACE_T:
1027         case PCB_ARC_T:
1028             result = IterateForward( m_Track, inspector, testData, p );
1029 
1030             // skip over any types handled in the above call.

1031             for( ; ; )
1032             {
1033                 switch( stype = *++p )
1034                 {
1035                 case PCB_VIA_T:
1036                 case PCB_TRACE_T:
1037                 case PCB_ARC_T:
1038                     continue;
1039 
1040                 default:
1041                     ;
1042                 }
1043 
1044                 break;
1045             }
1046 
1047             break;
1048 
1049 #else
1050         case PCB_VIA_T:
1051             result = IterateForward<TRACK*>( m_tracks, inspector, testData, p );
1052             ++p;
1053             break;
1054 
1055         case PCB_TRACE_T:
1056         case PCB_ARC_T:
1057             result = IterateForward<TRACK*>( m_tracks, inspector, testData, p );
1058             ++p;
1059             break;
1060 #endif
1061 
1062         case PCB_MARKER_T:
1063 
1064             // MARKER_PCBS are in the m_markers std::vector

1065             for( unsigned i = 0; i<m_markers.size(); ++i )
1066             {
1067                 result = m_markers[i]->Visit( inspector, testData, p );
1068 
1069                 if( result == SEARCH_RESULT::QUIT )
1070                     break;
1071             }
1072 
1073             ++p;
1074             break;
1075 
1076         case PCB_ZONE_AREA_T:
1077

1082

1084                     break;

1086

1088             break;
1089

1091             done = true;
1092             break;
1093         }
1094 
1095         if( result == SEARCH_RESULT::QUIT )
1096             break;

1384 D_PAD* BOARD::GetPad( std::vector<D_PAD*>& aPadList, const wxPoint& aPosition, LSET aLayerSet )
1385 {
1386     // Search aPadList for aPosition

1387     // aPadList is sorted by X then Y values, and a fast binary search is used

1388     int idxmax = aPadList.size()-1;
1389 
1390     int delta = aPadList.size();
1391 
1392     int idx = 0;        // Starting index is the beginning of list

1393 
1394     while( delta )
1395     {
1396         // Calculate half size of remaining interval to test.

1397         // Ensure the computed value is not truncated (too small)

1398         if( (delta & 1) && ( delta > 1 ) )
1399             delta++;
1400 
1401         delta /= 2;
1402 
1403         D_PAD* pad = aPadList[idx];
1404 
1405         if( pad->GetPosition() == aPosition )       // candidate found

1406         {
1407             // The pad must match the layer mask:

1408             if( ( aLayerSet & pad->GetLayerSet() ).any() )
1409                 return pad;
1410 
1411             // More than one pad can be at aPosition

1412             // search for a pad at aPosition that matched this mask

1413 
1414             // search next

1415             for( int ii = idx+1; ii <= idxmax; ii++ )
1416             {
1417                 pad = aPadList[ii];
1418 
1419                 if( pad->GetPosition() != aPosition )
1420                     break;
1421 
1422                 if( ( aLayerSet & pad->GetLayerSet() ).any() )
1423                     return pad;
1424             }
1425             // search previous

1426             for(  int ii = idx-1 ;ii >=0; ii-- )
1427             {
1428                 pad = aPadList[ii];
1429 
1430                 if( pad->GetPosition() != aPosition )
1431                     break;
1432 
1433                 if( ( aLayerSet & pad->GetLayerSet() ).any() )
1434                     return pad;
1435             }
1436 
1437             // Not found:

1438             return 0;
1439         }
1440 
1441         if( pad->GetPosition().x == aPosition.x )       // Must search considering Y coordinate

1442         {
1443             if( pad->GetPosition().y < aPosition.y )    // Must search after this item

1444             {
1445                 idx += delta;
1446 
1447                 if( idx > idxmax )
1448                     idx = idxmax;
1449             }
1450             else // Must search before this item

1451             {
1452                 idx -= delta;
1453 
1454                 if( idx < 0 )
1455                     idx = 0;
1456             }
1457         }
1458         else if( pad->GetPosition().x < aPosition.x ) // Must search after this item

1459         {
1460             idx += delta;
1461 
1462             if( idx > idxmax )
1463                 idx = idxmax;
1464         }
1465         else // Must search before this item

1466         {
1467             idx -= delta;
1468 
1469             if( idx < 0 )
1470                 idx = 0;
1471         }
1472     }
1473 
1474     return NULL;
1475 }

1518 std::tuple<int, double, double> BOARD::GetTrackLength( const TRACK& aTrack ) const
1519 {
1520     int    count = 0;
1521     double length = 0.0;
1522     double package_length = 0.0;
1523 
1524     constexpr KICAD_T types[] = { PCB_TRACE_T, PCB_ARC_T, PCB_VIA_T, PCB_PAD_T, EOT };
1525     auto              connectivity = GetBoard()->GetConnectivity();
1526 
1527     for( auto item : connectivity->GetConnectedItems(
1528                  static_cast<const BOARD_CONNECTED_ITEM*>( &aTrack ), types ) )
1529     {
1530         count++;
1531 
1532         if( auto track = dyn_cast<TRACK*>( item ) )
1533         {
1534             bool inPad = false;
1535 
1536             for( auto pad_it : connectivity->GetConnectedPads( item ) )
1537             {
1538                 auto pad = static_cast<D_PAD*>( pad_it );
1539 
1540                 if( pad->HitTest( track->GetStart(), track->GetWidth() / 2 )
1541                         && pad->HitTest( track->GetEnd(), track->GetWidth() / 2 ) )
1542                 {
1543                     inPad = true;
1544                     break;
1545                 }
1546             }
1547 
1548             if( !inPad )
1549                 length += track->GetLength();
1550         }
1551         else if( auto pad = dyn_cast<D_PAD*>( item ) )
1552             package_length += pad->GetPadToDieLength();
1553     }
1554 
1555     return std::make_tuple( count, length, package_length );

1559 MODULE* BOARD::GetFootprint( const wxPoint& aPosition, PCB_LAYER_ID aActiveLayer,
1560                              bool aVisibleOnly, bool aIgnoreLocked )
1561 {
1562     MODULE* module      = NULL;
1563     MODULE* alt_module  = NULL;
1564     int     min_dim     = 0x7FFFFFFF;
1565     int     alt_min_dim = 0x7FFFFFFF;
1566     bool    current_layer_back = IsBackLayer( aActiveLayer );
1567 
1568     for( auto pt_module : m_modules )
1569     {
1570         // is the ref point within the module's bounds?

1571         if( !pt_module->HitTest( aPosition ) )
1572             continue;
1573 
1574         // if caller wants to ignore locked modules, and this one is locked, skip it.

1575         if( aIgnoreLocked && pt_module->IsLocked() )
1576             continue;
1577 
1578         PCB_LAYER_ID layer = pt_module->GetLayer();
1579 
1580         // Filter non visible modules if requested

1581         if( !aVisibleOnly || IsModuleLayerVisible( layer ) )
1582         {
1583             EDA_RECT bb = pt_module->GetFootprintRect();
1584 
1585             int offx = bb.GetX() + bb.GetWidth() / 2;
1586             int offy = bb.GetY() + bb.GetHeight() / 2;
1587 
1588             // off x & offy point to the middle of the box.

1589             int dist = ( aPosition.x - offx ) * ( aPosition.x - offx ) +
1590                        ( aPosition.y - offy ) * ( aPosition.y - offy );
1591 
1592             if( current_layer_back == IsBackLayer( layer ) )
1593             {
1594                 if( dist <= min_dim )
1595                 {
1596                     // better footprint shown on the active side

1597                     module  = pt_module;
1598                     min_dim = dist;
1599                 }
1600             }
1601             else if( aVisibleOnly && IsModuleLayerVisible( layer ) )
1602             {
1603                 if( dist <= alt_min_dim )
1604                 {
1605                     // better footprint shown on the other side

1606                     alt_module  = pt_module;
1607                     alt_min_dim = dist;
1608                 }
1609             }
1610         }
1611     }
1612 
1613     if( module )
1614     {
1615         return module;
1616     }
1617 
1618     if( alt_module)
1619     {
1620         return alt_module;
1621     }
1622 
1623     return NULL;
1624 }

1708 bool BOARD::NormalizeAreaPolygon( PICKED_ITEMS_LIST * aNewZonesList, ZONE_CONTAINER* aCurrArea )
1709 {
1710     // mark all areas as unmodified except this one, if modified

1711     for( ZONE_CONTAINER* zone : m_ZoneDescriptorList )
1712         zone->SetLocalFlags( 0 );
1713 
1714     aCurrArea->SetLocalFlags( 1 );
1715 
1716     if( aCurrArea->Outline()->IsSelfIntersecting() )
1717     {
1718         aCurrArea->UnHatch();
1719 
1720         // Normalize copied area and store resulting number of polygons

1721         int n_poly = aCurrArea->Outline()->NormalizeAreaOutlines();
1722 
1723         // If clipping has created some polygons, we must add these new copper areas.

1724         if( n_poly > 1 )
1725         {
1726             ZONE_CONTAINER* NewArea;
1727 
1728             // Move the newly created polygons to new areas, removing them from the current area

1729             for( int ip = 1; ip < n_poly; ip++ )
1730             {
1731                 // Create new copper area and copy poly into it

1732                 SHAPE_POLY_SET* new_p = new SHAPE_POLY_SET( aCurrArea->Outline()->UnitSet( ip ) );
1733                 NewArea = AddArea( aNewZonesList, aCurrArea->GetNetCode(), aCurrArea->GetLayer(),
1734                                    wxPoint(0, 0), aCurrArea->GetHatchStyle() );
1735 
1736                 // remove the poly that was automatically created for the new area

1737                 // and replace it with a poly from NormalizeAreaOutlines

1738                 delete NewArea->Outline();
1739                 NewArea->SetOutline( new_p );
1740                 NewArea->Hatch();
1741                 NewArea->SetLocalFlags( 1 );
1742             }
1743 
1744             SHAPE_POLY_SET* new_p = new SHAPE_POLY_SET( aCurrArea->Outline()->UnitSet( 0 ) );
1745             delete aCurrArea->Outline();
1746             aCurrArea->SetOutline( new_p );
1747         }
1748     }
1749 
1750     aCurrArea->Hatch();
1751 
1752     return true;
1753 }
