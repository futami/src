  61 void PANEL_SETUP_RULES::onScintillaCharAdded( wxStyledTextEvent &aEvent )
  62 {
  63     constexpr int flags = wxSTC_FIND_REGEXP| wxSTC_FIND_POSIX;
  64 
  65     m_textEditor->SearchAnchor();
  66 
  67     int i = std::max( 0, m_textEditor->SearchPrev( flags, "\( *rule " ) );
  68     int currentPos = m_textEditor->GetCurrentPos();
  69 
  70     enum
  71     {
  72         NONE,
  73         STRING,
  74         SEXPR_OPEN,
  75         SEXPR_TOKEN,
  76         STRUCT_REF
  77     };
  78 
  79     std::stack<wxString> sexprs;
  80     wxString             partial;
  81     int                  context = NONE;
  82     int                  expr_context = NONE;
  83 
  84     for( ; i < currentPos; ++i )
  85     {
  86         wxChar c = m_textEditor->GetCharAt( i );
  87 
  88         if( c == '\\' )
  89         {
  90             i++;  // skip escaped char

  91         }
  92         else if( context == STRING )
  93         {
  94             if( c == '"' )
  95             {
  96                 context = NONE;
  97             }
  98             else
  99             {
 100                 if( expr_context == STRING )
 101                 {
 102                     if( c == '\'' )
 103                         expr_context = NONE;
 104                     else
 105                         partial += c;
 106                 }
 107                 else if( c == '\'' )
 108                 {
 109                     partial = wxEmptyString;
 110                     expr_context = STRING;
 111                 }
 112                 else if( c == '.' )
 113                 {
 114                     partial = wxEmptyString;
 115                     expr_context = STRUCT_REF;
 116                 }
 117                 else
 118                 {
 119                     partial += c;
 120                 }
 121             }
 122         }
 123         else if( c == '"' )
 124         {
 125             partial = wxEmptyString;
 126             context = STRING;
 127         }
 128         else if( c == '(' )
 129         {
 130             if( context == SEXPR_OPEN && !partial.IsEmpty() )
 131             {
 132                 m_textEditor->AutoCompCancel();
 133                 sexprs.push( partial );
 134             }
 135 
 136             partial = wxEmptyString;
 137             context = SEXPR_OPEN;
 138         }
 139         else if( c == ')' )
 140         {
 141             if( !sexprs.empty() )
 142                 sexprs.pop();
 143 
 144             context = NONE;
 145         }
 146         else if( c == ' ' )
 147         {
 148             if( context == SEXPR_OPEN && !partial.IsEmpty() )
 149             {
 150                 m_textEditor->AutoCompCancel();
 151                 sexprs.push( partial );
 152 
 153                 if( sexprs.size()
 154                     && ( sexprs.top() == "constraint" || sexprs.top() == "disallow" ) )
 155                 {
 156                     partial = wxEmptyString;
 157                     context = SEXPR_TOKEN;
 158                     continue;
 159                 }
 160             }
 161 
 162             context = NONE;
 163         }
 164         else
 165         {
 166             partial += c;
 167         }
 168     }
 169 
 170     wxString tokens;
 171 
 172     if( context == SEXPR_OPEN )
 173     {
 174         if( sexprs.empty() )
 175             tokens = "rule version";
 176         else if( sexprs.top() == "rule" )
 177             tokens = "condition constraint disallow";
 178         else if( sexprs.top() == "constraint" )
 179             tokens = "max min opt";
 180     }
 181     else if( context == SEXPR_TOKEN )
 182     {
 183         if( sexprs.empty() )
 184             /* badly formed grammar */;
 185         else if( sexprs.top() == "constraint" )
 186             tokens = "annulus_width clearance hole track_width";
 187         else if( sexprs.top() == "disallow" )
 188             tokens = "buried_via graphic hole micro_via pad text track via zone";
 189     }
 190     else if( context == STRING && expr_context == STRUCT_REF )
 191     {
 192         if( !sexprs.empty() && sexprs.top() == "condition" )
 193         {
 194             PROPERTY_MANAGER&  propMgr = PROPERTY_MANAGER::Instance();
 195             std::set<wxString> propNames;
 196 
 197             for( const PROPERTY_MANAGER::CLASS_INFO& cls : propMgr.GetAllClasses() )
 198             {
 199                 const PROPERTY_LIST& props = propMgr.GetProperties( cls.type );
 200 
 201                 for( PROPERTY_BASE* prop : props )
 202                 {
 203                     wxString ref( prop->Name() );
 204                     ref.Replace( " ", "_" );
 205                     propNames.insert( ref );
 206                 }
 207             }
 208 
 209             for( const wxString& propName : propNames )
 210                 tokens += " " + propName;
 211 
 212             PCB_EXPR_BUILTIN_FUNCTIONS& functions = PCB_EXPR_BUILTIN_FUNCTIONS::Instance();
 213 
 214             for( const wxString& funcSig : functions.GetSignatures() )
 215                 tokens += " " + funcSig;
 216         }
 217     }
 218 
 219     if( !tokens.IsEmpty() )
 220         m_scintillaTricks->DoAutocomplete( partial, wxSplit( tokens, ' ' ) );
 221 }

 224 bool PANEL_SETUP_RULES::TransferDataToWindow()
 225 {
 226     wxString   rulesFilepath = m_frame->Prj().AbsolutePath( "drc-rules" );
 227     wxFileName rulesFile( rulesFilepath );
 228 
 229     if( rulesFile.FileExists() )
 230         m_textEditor->LoadFile( rulesFile.GetFullPath() );
 231 
 232     m_originalText = m_textEditor->GetText();
 233 
 234     return true;
 235 }

 238 bool PANEL_SETUP_RULES::TransferDataFromWindow()
 239 {
 240     if( m_originalText == m_textEditor->GetText() )
 241         return true;
 242 
 243     try
 244     {
 245         std::vector<DRC_RULE*>     dummyRules;
 246 
 247         DRC_RULES_PARSER parser( m_frame->GetBoard(), m_textEditor->GetText(), _( "DRC rules" ) );
 248 
 249         parser.Parse( dummyRules );
 250     }
 251     catch( PARSE_ERROR& pe )
 252     {
 253         m_Parent->SetError( pe.What(), this, m_textEditor, pe.lineNumber, pe.byteIndex );
 254         return false;
 255     }
 256 
 257     if( m_textEditor->SaveFile( m_frame->Prj().AbsolutePath( "drc-rules" ) ) )
 258     {
 259         m_frame->GetToolManager()->GetTool<DRC>()->LoadRules();
 260         return true;
 261     }
 262 
 263     return false;
 264 }
