  29 int CN_ITEM::AnchorCount() const
  30 {
  31     if( !m_valid )
  32         return 0;
  33 
  34     switch( m_parent->Type() )
  35     {
  36     case PCB_PAD_T:
  37         return 5;  // center, north, south, east and west

  38     case PCB_TRACE_T:
  39     case PCB_ARC_T:
  40         return 2;  // start and end

  41     default:
  42         return 1;
  43     }
  44 }

  47 const VECTOR2I CN_ITEM::GetAnchor( int n ) const
  48 {
  49     VECTOR2I pt0;
  50 
  51     if( !m_valid )
  52         return pt0;
  53 
  54     switch( m_parent->Type() )
  55     {
  56     case PCB_PAD_T:
  57     {
  58         D_PAD* pad = (D_PAD*) m_parent;
  59 
  60         if( n == 0 )
  61             return VECTOR2I( pad->GetPosition() );
  62 
  63         // ShapePos() is the geometric center (not anchor) for the pad

  64         pt0 = pad->ShapePos();
  65         VECTOR2I pt1 = pt0;
  66 
  67         switch( pad->GetShape() )
  68         {
  69         case PAD_SHAPE_TRAPEZOID:
  70             // Because the trap delta is applied as +1/2 at one end and -1/2 at the other,

  71             // the midpoint is actually unchanged.  Therefore all the cardinal points are

  72             // the same as for a rectangle.

  73             KI_FALLTHROUGH;
  74 
  75         case PAD_SHAPE_RECT:
  76         case PAD_SHAPE_CIRCLE:
  77         case PAD_SHAPE_OVAL:
  78         case PAD_SHAPE_ROUNDRECT:
  79         case PAD_SHAPE_CHAMFERED_RECT:
  80             switch( n )
  81             {
  82             case 1: pt1.y -= pad->GetSize().y / 2; break;    // North

  83             case 2: pt1.y += pad->GetSize().y / 2; break;    // South

  84             case 3: pt1.x -= pad->GetSize().x / 2; break;    // East

  85             case 4: pt1.x += pad->GetSize().x / 2; break;    // West

  86             default:                               break;    // Wicked witch

  87             }
  88 
  89             if( pad->GetOrientation() )
  90                 RotatePoint( pt1, pad->ShapePos(), pad->GetOrientation() );
  91 
  92             // Thermal spokes on circular pads form an 'X' instead of a '+'

  93             if( pad->GetShape() == PAD_SHAPE_CIRCLE )
  94                 RotatePoint( pt1, pad->ShapePos(), 450 );
  95 
  96             return pt1;
  97 
  98         case PAD_SHAPE_CUSTOM:
  99         {
 100             switch( n )
 101             {
 102             case 1: pt1.y = INT_MIN / 2; break;    // North

 103             case 2: pt1.y = INT_MAX / 2; break;    // South

 104             case 3: pt1.x = INT_MIN / 2; break;    // East

 105             case 4: pt1.x = INT_MAX / 2; break;    // West

 106             default:                     break;    // Wicked witch

 107             }
 108 
 109             if( pad->GetOrientation() )
 110                 RotatePoint( pt1, pad->ShapePos(), pad->GetOrientation() );
 111 
 112             const std::shared_ptr<SHAPE_POLY_SET>& padPolySet = pad->GetEffectivePolygon();
 113             const SHAPE_LINE_CHAIN&                padOutline = padPolySet->COutline( 0 );
 114             SHAPE_LINE_CHAIN::INTERSECTIONS        intersections;
 115 
 116             padOutline.Intersect( SEG( pt0, pt1 ), intersections );
 117 
 118             if( intersections.empty() )
 119             {
 120                 // There should always be at least some copper outside the hole and/or

 121                 // shapePos center

 122                 assert( false );
 123                 return pt0;
 124             }
 125 
 126             return intersections[ intersections.size() - 1 ].p;
 127         }
 128         }
 129 
 130         break;
 131     }
 132     case PCB_TRACE_T:
 133     case PCB_ARC_T:
 134         if( n == 0 )
 135             return static_cast<const TRACK*>( m_parent )->GetStart();
 136         else
 137             return static_cast<const TRACK*>( m_parent )->GetEnd();
 138 
 139     case PCB_VIA_T:
 140         return static_cast<const VIA*>( m_parent )->GetStart();
 141 
 142     default:
 143         assert( false );
 144         break;
 145     }
 146 
 147     return pt0;
 148 }

 163 int CN_ZONE::AnchorCount() const
 164 {
 165     if( !Valid() )
 166         return 0;
 167 
 168     const auto  zone    = static_cast<const ZONE_CONTAINER*>( Parent() );
 169     const auto& outline = zone->GetFilledPolysList( m_layer ).COutline( m_subpolyIndex );
 170 
 171     return outline.PointCount() ? 1 : 0;

 199 CN_ITEM* CN_LIST::Add( D_PAD* pad )
 200  {
 201     if( !pad->IsOnCopperLayer() )
 202          return nullptr;
 203 
 204      auto item = new CN_ITEM( pad, false, 1 );
 205      item->AddAnchor( pad->ShapePos() );
 206      item->SetLayers( LAYER_RANGE( F_Cu, B_Cu ) );
 207 
 208      switch( pad->GetAttribute() )
 209      {
 210      case PAD_ATTRIB_SMD:
 211      case PAD_ATTRIB_HOLE_NOT_PLATED:
 212      case PAD_ATTRIB_CONN:
 213      {
 214          LSET lmsk = pad->GetLayerSet();
 215 
 216          for( int i = 0; i <= MAX_CU_LAYERS; i++ )
 217          {
 218              if( lmsk[i] )
 219              {
 220                  item->SetLayer( i );
 221                  break;
 222              }
 223          }
 224          break;
 225      }
 226      default:
 227          break;
 228      }
 229 
 230      addItemtoTree( item );
 231      m_items.push_back( item );
 232      SetDirty();
 233      return item;
 234 }

 272  const std::vector<CN_ITEM*> CN_LIST::Add( ZONE_CONTAINER* zone, PCB_LAYER_ID aLayer )
 273  {
 274      const auto& polys = zone->GetFilledPolysList( aLayer );
 275 
 276      std::vector<CN_ITEM*> rv;
 277 
 278      for( int j = 0; j < polys.OutlineCount(); j++ )
 279      {
 280          CN_ZONE* zitem = new CN_ZONE( zone, aLayer, false, j );
 281          const auto& outline = zone->GetFilledPolysList( aLayer ).COutline( j );
 282 
 283          for( int k = 0; k < outline.PointCount(); k++ )
 284              zitem->AddAnchor( outline.CPoint( k ) );
 285 
 286          m_items.push_back( zitem );
 287          zitem->SetLayer( aLayer );
 288          addItemtoTree( zitem );
 289          rv.push_back( zitem );
 290          SetDirty();
 291      }
 292 
 293      return rv;
 294  }

 297 void CN_LIST::RemoveInvalidItems( std::vector<CN_ITEM*>& aGarbage )
 298 {
 299     if( !m_hasInvalid )
 300         return;
 301 
 302     auto lastItem = std::remove_if(m_items.begin(), m_items.end(), [&aGarbage] ( CN_ITEM* item )
 303     {
 304         if( !item->Valid() )
 305         {
 306             aGarbage.push_back ( item );
 307             return true;
 308         }
 309 
 310         return false;
 311     } );
 312 
 313     m_items.resize( lastItem - m_items.begin() );
 314 
 315     for( auto item : m_items )
 316         item->RemoveInvalidRefs();
 317 
 318     for( auto item : aGarbage )
 319         m_index.Remove( item );
 320 
 321     m_hasInvalid = false;
 322 }

 341 bool CN_ANCHOR::IsDangling() const
 342 {
 343     if( !m_cluster )
 344         return true;
 345 
 346     // the minimal number of items connected to item_ref

 347     // at this anchor point to decide the anchor is *not* dangling

 348     size_t minimal_count = 1;
 349     size_t connected_count = m_item->ConnectedItems().size();
 350 
 351     // a via can be removed if connected to only one other item.

 352     if( Parent()->Type() == PCB_VIA_T )
 353         return connected_count < 2;
 354 
 355     if( m_item->AnchorCount() == 1 )
 356         return connected_count < minimal_count;
 357 
 358     // Items with multiple anchors have usually items connected to each anchor.

 359     // We want only the item count of this anchor point

 360     connected_count = 0;
 361     for( auto item : m_item->ConnectedItems() )
 362     {
 363         if( item->Parent()->Type() == PCB_ZONE_AREA_T )
 364         {
 365             ZONE_CONTAINER* zone = static_cast<ZONE_CONTAINER*>( item->Parent() );
 366 
 367             if( zone->HitTestFilledArea( static_cast<PCB_LAYER_ID>( item->Layer() ),
 368                                          static_cast<wxPoint>( Pos() ) ) )
 369                 connected_count++;
 370         }
 371         else if( item->Parent()->HitTest( (wxPoint) Pos() ) )
 372             connected_count++;
 373     }
 374 
 375     return connected_count < minimal_count;
 376 }

 379 int CN_ANCHOR::ConnectedItemsCount() const
 380 {
 381     if( !m_cluster )
 382         return 0;
 383 
 384     int connected_count = 0;
 385 
 386     for( auto item : m_item->ConnectedItems() )
 387     {
 388         if( item->Parent()->Type() == PCB_ZONE_AREA_T )
 389         {
 390             ZONE_CONTAINER* zone = static_cast<ZONE_CONTAINER*>( item->Parent() );
 391 
 392             if( zone->HitTestFilledArea( static_cast<PCB_LAYER_ID>( item->Layer() ),
 393                                          wxPoint( Pos().x, Pos().y ) ) )
 394                 connected_count++;
 395         }
 396         else if( item->Parent()->HitTest( wxPoint( Pos().x, Pos().y ) ) )
 397             connected_count++;
 398     }
 399 
 400     return connected_count;
 401 }

 453 void CN_CLUSTER::Add( CN_ITEM* item )
 454 {
 455     m_items.push_back( item );
 456 
 457     if( item->Net() <= 0 )
 458         return;
 459 
 460     if( m_originNet <= 0 )
 461     {
 462         m_originNet = item->Net();
 463     }
 464 
 465     if( item->Parent()->Type() == PCB_PAD_T )
 466     {
 467         if( !m_originPad )
 468         {
 469             m_originPad = item;
 470             m_originNet = item->Net();
 471         }
 472 
 473         if( m_originPad && item->Net() != m_originNet )
 474         {
 475             m_conflicting = true;
 476         }
 477     }
