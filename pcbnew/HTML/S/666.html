  22 #ifndef PCBNEW_ROUTER_PNS_LINK_HOLDER_H_
  23 #define PCBNEW_ROUTER_PNS_LINK_HOLDER_H_
  24 
  25 #include "pns_item.h"
  26 #include "pns_linked_item.h"
  27 
  28 namespace PNS
  29 {
  30 class LINK_HOLDER : public ITEM
  31 {
  32 public:
  33     typedef std::vector<LINKED_ITEM*> LINKS;
  34 
  35     LINK_HOLDER( PnsKind aKind ) : ITEM( aKind )
  36     {}
  37 
  38     ///> Adds a reference to an item registered in a NODE that is a part of this line.

  39     void Link( LINKED_ITEM* aLink )
  40     {
  41         m_links.push_back( aLink );
  42     }
  43 
  44     ///> Returns the list of links from the owning node that constitute this

  45     ///> line (or NULL if the line is not linked)

  46     LINKS& Links()
  47     {
  48         return m_links;
  49     }
  50 
  51     bool IsLinked() const
  52     {
  53         return m_links.size() != 0;
  54     }
  55 
  56     ///> Checks if the segment aLink is a part of the line.

  57     bool ContainsLink( const LINKED_ITEM* aItem ) const
  58     {
  59         return std::find( m_links.begin(), m_links.end(),
  60                 aItem ) != m_links.end();
  61     }
  62 
  63     LINKED_ITEM* GetLink( int aIndex ) const
  64     {
  65         return m_links[aIndex];
  66     }
  67 
  68     ///> Erases the linking information. Used to detach the line from the owning node.

  69     virtual void ClearLinks()
  70     {
  71         m_links.clear();
  72     }
  73 
  74     ///> Returns the number of segments that were assembled together to form this line.

  75     int LinkCount() const
  76     {
  77         return m_links.size();
  78     }
  79 
  80     void ShowLinks() const
  81     {
  82         #if 0 // fixme: move outside header

  83         if( !IsLinked() )
  84         {
  85             wxLogTrace( "PNS", "item %p: no links", this );
  86             return;
  87         }
  88 
  89         wxLogTrace( "PNS", "item %p: %d links", this, (int) m_links.size() );
  90 
  91         for( int i = 0; i < (int) m_links.size(); i++ )
  92             wxLogTrace( "PNS", "item %d: %p\n", i, m_links[i] );
  93         #endif
  94     }
  95 
  96 protected:
  97     ///> Copies m_links from the line aParent.

  98     void copyLinks( const LINK_HOLDER* aParent )
  99     {
 100         m_links = aParent->m_links;
 101     }
 102 
 103     ///> List of segments in the owning NODE (ITEM::m_owner) that constitute this line, or NULL

 104     ///> if the line is not a part of any node.

 105     LINKS m_links;
 106 };
 107 
 108 } // namespace PNS

 109 #endif /* PCBNEW_ROUTER_PNS_LINK_HOLDER_H_ */
