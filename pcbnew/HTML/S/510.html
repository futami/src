  31 // Microwave shapes that are created as board modules when the user requests them.

  32 enum class MICROWAVE_FOOTPRINT_SHAPE
  33 {
  34     GAP,
  35     STUB,
  36     STUB_ARC,
  37     FUNCTION_SHAPE,
  38 };
  39 
  40 /**

  41  * Parameters for construction of a microwave inductor

  42  */
  43 struct MICROWAVE_INDUCTOR_PATTERN
  44 {
  45 public:
  46     wxPoint m_Start;
  47     wxPoint m_End;
  48     int     m_length;       // full length trace.

  49     int     m_Width;        // Trace width.

  50 };
  51 
  52 
  53 /**

  54  * MICROWAVE_TOOL

  55  *

  56  * Tool responsible for adding microwave features to PCBs

  57  */
  58 class MICROWAVE_TOOL : public PCB_TOOL_BASE
  59 {
  60 public:
  61     MICROWAVE_TOOL();
  62     ~MICROWAVE_TOOL();
  63 
  64     ///> React to model/view changes

  65     void Reset( RESET_REASON aReason ) override;
  66 
  67     ///> Bind handlers to corresponding TOOL_ACTIONs

  68     void setTransitions() override;
  69 
  70 private:
  71     ///> Main interactive tool

  72     int addMicrowaveFootprint( const TOOL_EVENT& aEvent );
  73 
  74     ///> Create an inductor between the two points

  75     void createInductorBetween( const VECTOR2I& aStart, const VECTOR2I& aEnd );
  76 
  77     ///> Draw a microwave inductor interactively

  78     int drawMicrowaveInductor( const TOOL_EVENT& aEvent );
  79 
  80     /**

  81      * Creates a module "GAP" or "STUB" used in micro wave designs.

  82      *  This module has 2 pads:

  83      *  PAD_ATTRIB_SMD, rectangular, H size = V size = current track width.

  84      *  the "gap" is isolation created between this 2 pads

  85      *

  86      * @param aComponentShape is the component to create

  87      * @return the new module

  88      */
  89     MODULE* createFootprint( MICROWAVE_FOOTPRINT_SHAPE aFootprintShape );
  90 
  91     MODULE* createPolygonShape();
  92 
  93     /**

  94      * Creates an S-shaped coil footprint for microwave applications.

  95      */
  96     MODULE* createMicrowaveInductor( MICROWAVE_INDUCTOR_PATTERN& aPattern, wxString& aErrorMessage );
  97 
  98     /**

  99      * Create a basic footprint for micro wave applications.

 100      *

 101      * The default pad settings are:

 102      *  PAD_ATTRIB_SMD, rectangular, H size = V size = current track width.

 103      *

 104      * @param aValue is the text value

 105      * @param aTextSize is the size of ref and value texts ( <= 0 to use board default values )

 106      * @param aPadCount is number of pads

 107      * @return the new module

 108      */
 109     MODULE* createBaseFootprint( const wxString& aValue, int aTextSize, int aPadCount );
 110 };
