  24 #ifndef PCB_VIEWER_TOOLS_H_
  25 #define PCB_VIEWER_TOOLS_H_
  26 
  27 #include <class_board.h>
  28 #include <tool/tool_interactive.h>
  29 
  30 class PCB_BASE_FRAME;
  31 
  32 /**

  33  * PCB_VIEWER_TOOLS

  34  *

  35  * Tools useful for viewing footprints.

  36  * This tool is designed to be lighter-weight so that it doesn't

  37  * bring in as many pcbnew dependencies (since it is used in cvpcb).

  38  */
  39 class PCB_VIEWER_TOOLS : public TOOL_INTERACTIVE
  40 {
  41 public:
  42     PCB_VIEWER_TOOLS() :
  43         TOOL_INTERACTIVE( "pcbnew.PCBViewerTools" ),
  44         m_footprintFrame( false )
  45     {}
  46 
  47     ~PCB_VIEWER_TOOLS() override {}
  48 
  49     bool Init() override;
  50 
  51     /// @copydoc TOOL_BASE::Reset()

  52     void Reset( RESET_REASON aReason ) override;
  53 
  54     ///> Launches a tool to measure between points

  55     int MeasureTool( const TOOL_EVENT& aEvent );
  56 
  57     // Display modes

  58     int ShowPadNumbers( const TOOL_EVENT& aEvent );
  59     int PadDisplayMode( const TOOL_EVENT& aEvent );
  60     int GraphicOutlines( const TOOL_EVENT& aEvent );
  61     int TextOutlines( const TOOL_EVENT& aEvent );
  62     int ZoomAutomatically( const TOOL_EVENT& aEvent );
  63 
  64     /// Show the 3D viewer

  65     int Show3DViewer( const TOOL_EVENT& aEvent );
  66 
  67     ///> Sets up handlers for various events.

  68     void setTransitions() override;
  69 
  70     /**

  71      * Toggles edit module mode. When enabled, one may select parts of modules individually

  72      * (graphics, pads, etc.), so they can be modified.

  73      * @param aEnabled decides if the mode should be enabled.

  74      */
  75     void SetFootprintFrame( bool aIsFrame )
  76     {
  77         m_footprintFrame = aIsFrame;
  78     }
  79 
  80     bool IsFootprintFrame() const
  81     {
  82         return m_footprintFrame;
  83     }
  84 
  85 protected:
  86     bool m_footprintFrame;  ///< Is this tool associated with a footprint frame

  87 
  88     PCB_BASE_FRAME* frame() const
  89     {
  90         return getEditFrame<PCB_BASE_FRAME>();
  91     }
  92 
  93     KIGFX::PCB_VIEW* view() const
  94     {
  95         return static_cast<KIGFX::PCB_VIEW*>( getView() );
  96     }
  97 
  98     const PCB_DISPLAY_OPTIONS& displayOptions() const
  99     {
 100         return frame()->GetDisplayOptions();
 101     }
 102 
 103     PCB_DRAW_PANEL_GAL* canvas() const
 104     {
 105         return static_cast<PCB_DRAW_PANEL_GAL*>( frame()->GetCanvas() );
 106     }
 107 
 108     BOARD* board() const
 109     {
 110         return getModel<BOARD>();
 111     }
 112 
 113     MODULE* module() const
 114     {
 115         return board()->GetFirstModule();
 116     }
 117 };
