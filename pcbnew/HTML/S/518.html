  59 bool PCB_EDIT_FRAME::ReadNetlistFromFile( const wxString &aFilename,
  60                                           NETLIST& aNetlist,
  61                                           REPORTER& aReporter )
  62 {
  63     wxString msg;
  64 
  65     try
  66     {
  67         std::unique_ptr<NETLIST_READER> netlistReader( NETLIST_READER::GetNetlistReader(
  68                 &aNetlist, aFilename, wxEmptyString ) );
  69 
  70         if( !netlistReader.get() )
  71         {
  72             msg.Printf( _( "Cannot open netlist file \"%s\"." ), GetChars( aFilename ) );
  73             wxMessageBox( msg, _( "Netlist Load Error." ), wxOK | wxICON_ERROR, this );
  74             return false;
  75         }
  76 
  77         SetLastPath( LAST_PATH_NETLIST, aFilename );
  78         netlistReader->LoadNetlist();
  79         LoadFootprints( aNetlist, aReporter );
  80     }
  81     catch( const IO_ERROR& ioe )
  82     {
  83         msg.Printf( _( "Error loading netlist.\n%s" ), ioe.What().GetData() );
  84         wxMessageBox( msg, _( "Netlist Load Error" ), wxOK | wxICON_ERROR );
  85         return false;
  86     }
  87 
  88     SetLastPath( LAST_PATH_NETLIST, aFilename );
  89 
  90     return true;
  91 }

  94 void PCB_EDIT_FRAME::OnNetlistChanged( BOARD_NETLIST_UPDATER& aUpdater, bool* aRunDragCommand )
  95 {
  96     std::string dummyPayload;
  97     Kiway().ExpressMail( FRAME_SCH, MAIL_SCH_CLEAN_NETCLASSES, dummyPayload, this );
  98 
  99     BOARD* board = GetBoard();
 100 
 101     SetMsgPanel( board );
 102 
 103     // Update rendered tracks and vias net labels

 104     // TODO is there a way to extract information about which nets were modified?

 105     for( auto track : board->Tracks() )
 106         GetCanvas()->GetView()->Update( track );
 107 
 108     std::vector<MODULE*> newFootprints = aUpdater.GetAddedComponents();
 109 
 110     // Spread new footprints.

 111     wxPoint areaPosition = (wxPoint) GetCanvas()->GetViewControls()->GetCursorPosition();
 112     EDA_RECT bbox = board->GetBoundingBox();
 113 
 114     GetToolManager()->RunAction( PCB_ACTIONS::selectionClear, true );
 115 
 116     SpreadFootprints( &newFootprints, areaPosition );
 117 
 118     // Start drag command for new modules

 119     if( !newFootprints.empty() )
 120     {
 121         for( MODULE* footprint : newFootprints )
 122             GetToolManager()->RunAction( PCB_ACTIONS::selectItem, true, footprint );
 123 
 124         *aRunDragCommand = true;
 125 
 126         // Now fix a reference point to move the footprints.

 127         // We use the first footprint in list as reference point

 128         // The graphic cursor will be on this fp when moving the footprints.

 129         SELECTION_TOOL*   selTool = GetToolManager()->GetTool<SELECTION_TOOL>();
 130         PCBNEW_SELECTION& selection = selTool->GetSelection();
 131         selection.SetReferencePoint( newFootprints[0]->GetPosition() );
 132     }
 133 
 134     GetCanvas()->Refresh();
 135 }

 140 void PCB_EDIT_FRAME::LoadFootprints( NETLIST& aNetlist, REPORTER& aReporter )
 141 {
 142     wxString   msg;
 143     LIB_ID     lastFPID;
 144     COMPONENT* component;
 145     MODULE*    module = nullptr;
 146     MODULE*    fpOnBoard;
 147 
 148     if( aNetlist.IsEmpty() || Prj().PcbFootprintLibs()->IsEmpty() )
 149         return;
 150 
 151     aNetlist.SortByFPID();
 152 
 153     for( unsigned ii = 0; ii < aNetlist.GetCount(); ii++ )
 154     {
 155         component = aNetlist.GetComponent( ii );
 156 
 157 #if ALLOW_PARTIAL_FPID
 158         // The FPID is ok as long as there is a footprint portion coming from eeschema.

 159         if( !component->GetFPID().GetLibItemName().size() )
 160 #else
 161         if( component->GetFPID().empty() )
 162 #endif
 163         {
 164             msg.Printf( _( "No footprint defined for symbol \"%s\".\n" ),
 165                         component->GetReference() );
 166             aReporter.Report( msg, RPT_SEVERITY_ERROR );
 167 
 168             continue;
 169         }
 170 
 171         // Check if component footprint is already on BOARD and only load the footprint from

 172         // the library if it's needed.  Nickname can be blank.

 173         if( aNetlist.IsFindByTimeStamp() )
 174             fpOnBoard = m_Pcb->FindModuleByPath( aNetlist.GetComponent( ii )->GetPath() );
 175         else
 176             fpOnBoard = m_Pcb->FindModuleByReference( aNetlist.GetComponent( ii )->GetReference() );
 177 
 178         bool footprintMisMatch = fpOnBoard && fpOnBoard->GetFPID() != component->GetFPID();
 179 
 180         if( footprintMisMatch && !aNetlist.GetReplaceFootprints() )
 181         {
 182             msg.Printf( _( "Footprint of %s changed: board footprint \"%s\", netlist footprint \"%s\"." ),
 183                         component->GetReference(),
 184                         GetChars( fpOnBoard->GetFPID().Format() ),
 185                         GetChars( component->GetFPID().Format() ) );
 186             aReporter.Report( msg, RPT_SEVERITY_WARNING );
 187 
 188             continue;
 189         }
 190 
 191         if( !aNetlist.GetReplaceFootprints() )
 192             footprintMisMatch = false;
 193 
 194         if( fpOnBoard && !footprintMisMatch )   // nothing else to do here

 195             continue;
 196 
 197         if( component->GetFPID() != lastFPID )
 198         {
 199             module = NULL;
 200 
 201 #if ALLOW_PARTIAL_FPID
 202             // The LIB_ID is ok as long as there is a footprint portion coming

 203             // the library if it's needed.  Nickname can be blank.

 204             if( !component->GetFPID().GetLibItemName().size() )
 205 #else
 206             if( !component->GetFPID().IsValid() )
 207 #endif
 208             {
 209                 msg.Printf( _( "%s footprint ID \"%s\" is not valid." ),
 210                             component->GetReference(),
 211                             GetChars( component->GetFPID().Format() ) );
 212                 aReporter.Report( msg, RPT_SEVERITY_ERROR );
 213 
 214                 continue;
 215             }
 216 
 217             // loadFootprint() can find a footprint with an empty nickname in fpid.

 218             module = PCB_BASE_FRAME::loadFootprint( component->GetFPID() );
 219 
 220             if( module )
 221             {
 222                 lastFPID = component->GetFPID();
 223             }
 224             else
 225             {
 226                 msg.Printf( _( "%s footprint \"%s\" not found in any libraries in the footprint library table.\n" ),
 227                             component->GetReference(),
 228                             GetChars( component->GetFPID().GetLibItemName() ) );
 229                 aReporter.Report( msg, RPT_SEVERITY_ERROR );
 230 
 231                 continue;
 232             }
 233         }
 234         else
 235         {
 236             // Footprint already loaded from a library, duplicate it (faster)

 237             if( module == NULL )
 238                 continue;            // Module does not exist in any library.

 239 
 240             module = new MODULE( *module );
 241         }
 242 
 243         if( module )
 244             component->SetModule( module );
 245     }
