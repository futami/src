  24 #ifndef TOOLS_ZONE_CREATE_HELPER__H_
  25 #define TOOLS_ZONE_CREATE_HELPER__H_
  26 
  27 #include <preview_items/polygon_geom_manager.h>
  28 #include <preview_items/polygon_item.h>
  29 
  30 #include <tools/drawing_tool.h>
  31 
  32 namespace KIGFX
  33 {
  34 class VIEW;
  35 }
  36 
  37 /**

  38  * This class is an adjuct helper to the DRAWING_TOOL interactive

  39  * tool, which handles incoming geometry changes from a

  40  * POLYGON_GEOM_MANAGER and translates that into a ZONE_CONTAINER

  41  * based on given parameters

  42  */
  43 class ZONE_CREATE_HELPER : public POLYGON_GEOM_MANAGER::CLIENT
  44 {
  45 public:
  46 
  47     /**

  48      * Parameters used to fully describe a zone creation process

  49      */
  50     struct PARAMS
  51     {
  52         ///> Should create a keepout zone?

  53         bool m_keepout;
  54 
  55         ///> Layer to begin drawing

  56         PCB_LAYER_ID m_layer;
  57 
  58         ///> The zone mode to operate in

  59         ZONE_MODE m_mode;
  60 
  61         ///> Zone settings source (for similar and cutout zones)

  62         ZONE_CONTAINER* m_sourceZone;
  63 
  64         ///> Zone leader mode

  65         POLYGON_GEOM_MANAGER::LEADER_MODE m_leaderMode;
  66     };
  67 
  68     /**

  69      * @param aTool the DRAWING_TOOL to provide the zone tool to

  70      * @param aParams the parameters to use to guide the zone creation

  71      */
  72     ZONE_CREATE_HELPER( DRAWING_TOOL& aTool, PARAMS& aParams );
  73 
  74     virtual ~ZONE_CREATE_HELPER();
  75 
  76     /*

  77      * Interface for receiving POLYGON_GEOM_MANAGER update

  78      */
  79 
  80     void OnGeometryChange( const POLYGON_GEOM_MANAGER& aMgr ) override;
  81 
  82     bool OnFirstPoint( POLYGON_GEOM_MANAGER& aMgr ) override;
  83 
  84     void OnComplete( const POLYGON_GEOM_MANAGER& aMgr ) override;
  85 
  86     /**

  87      * Function createNewZone()

  88      *

  89      * Prompt the user for new zone settings, and create a new zone with

  90      * those settings

  91      *

  92      * @param aKeepout should the zone be a keepout

  93      * @return the new zone, can be null if the user aborted

  94      */
  95     std::unique_ptr<ZONE_CONTAINER> createNewZone( bool aKeepout );
  96 
  97     /**

  98      * Function createZoneFromExisting

  99      *

 100      * Create a new zone with the settings from an existing zone

 101      *

 102      * @param aSrcZone the zone to copy settings from

 103      * @return the new zone

 104      */
 105     std::unique_ptr<ZONE_CONTAINER> createZoneFromExisting( const ZONE_CONTAINER& aSrcZone );
 106 
 107     /**

 108      * Function performZoneCutout()

 109      *

 110      * Cut one zone out of another one (i.e. subtraction) and

 111      * update the zone.

 112      *

 113      * @param aZone the zone to removed area from

 114      * @param aCutout the area to remove

 115      */
 116     void performZoneCutout( ZONE_CONTAINER& aZone, ZONE_CONTAINER& aCutout );
 117 
 118     /**

 119      * Commit the current zone-in-progress to the BOARD. This might

 120      * be adding a new zone, or modifying an existing zone with a

 121      * cutout, depending on parameters.

 122      *

 123      * @param aZone - the drawn zone outline to commit

 124      */
 125     void commitZone( std::unique_ptr<ZONE_CONTAINER> aZone );
 126 
 127 private:
 128 
 129     DRAWING_TOOL& m_tool;
 130 
 131     ///> Parameters of the zone to be drawn

 132     PARAMS& m_params;
 133 
 134     ///> The preview item to display

 135     KIGFX::PREVIEW::POLYGON_ITEM m_previewItem;
 136 
 137     ///> view that show the preview item

 138     KIGFX::VIEW& m_parentView;
 139 
 140     ///> The zone-in-progress

 141     std::unique_ptr<ZONE_CONTAINER> m_zone;
 142 };
 143 
 144 #endif /* __DRAWING_TOOL_H */
