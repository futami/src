  33 class IMPORTED_SHAPE

  41 class IMPORTED_LINE : public IMPORTED_SHAPE
  42 {
  43 public:
  44     IMPORTED_LINE( const VECTOR2D& aStart, const VECTOR2D& aEnd, double aWidth )
  45         : m_start( aStart ), m_end( aEnd ), m_width( aWidth )
  46     {
  47     }
  48 
  49     void ImportTo( GRAPHICS_IMPORTER& aImporter ) const override
  50     {
  51         aImporter.AddLine( m_start, m_end, m_width );
  52     }
  53 
  54 private:
  55     const VECTOR2D m_start;
  56     const VECTOR2D m_end;
  57     double m_width;
  58 };
