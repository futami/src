 103 void PCB_POLYGON::FormPolygon( XNODE*   aNode, VERTICES_ARRAY* aPolygon,
 104                                const wxString& aDefaultMeasurementUnit,
 105                                const wxString& aActualConversion )
 106 {
 107     XNODE*      lNode;
 108     double      x, y;
 109 
 110     lNode = FindNode( aNode, wxT( "pt" ) );
 111 
 112     while( lNode )
 113     {
 114         if( lNode->GetName() == wxT( "pt" ) )
 115         {
 116             SetDoublePrecisionPosition(
 117                 lNode->GetNodeContent(), aDefaultMeasurementUnit, &x, &y, aActualConversion );
 118             aPolygon->Add( new wxRealPoint( x, y ) );
 119         }
 120 
 121         lNode = lNode->GetNext();
 122     }
 123 }

 126 bool PCB_POLYGON::Parse( XNODE*          aNode,
 127                          const wxString& aDefaultMeasurementUnit,
 128                          const wxString& aActualConversion )
 129 {
 130     XNODE*      lNode;
 131     wxString    propValue;
 132 
 133     lNode = FindNode( aNode, wxT( "netNameRef" ) );
 134 
 135     if( lNode )
 136     {
 137         lNode->GetAttribute( wxT( "Name" ), &propValue );
 138         propValue.Trim( false );
 139         propValue.Trim( true );
 140         m_net = propValue;
 141         m_netCode = GetNetCode( m_net );
 142     }
 143 
 144     // retrieve polygon outline

 145     FormPolygon( aNode, &m_outline, aDefaultMeasurementUnit, aActualConversion );
 146 
 147     m_positionX = m_outline[0]->x;
 148     m_positionY = m_outline[0]->y;
 149 
 150     // fill the polygon with the same contour as its outline is

 151     m_islands.Add( new VERTICES_ARRAY );
 152     FormPolygon( aNode, m_islands[0], aDefaultMeasurementUnit, aActualConversion );
 153 
 154     return true;
 155 }

 158 void PCB_POLYGON::AddToModule( MODULE* aModule )
 159 {
 160     if( IsNonCopperLayer( m_KiCadLayer ) )
 161     {
 162         EDGE_MODULE* dwg = new EDGE_MODULE( aModule, S_POLYGON );
 163         aModule->Add( dwg );
 164 
 165         dwg->SetWidth( 0 );
 166         dwg->SetLayer( m_KiCadLayer );
 167 
 168         auto outline = new std::vector<wxPoint>;
 169         for( auto point : m_outline )
 170             outline->push_back( wxPoint( point->x, point->y ) );
 171 
 172         dwg->SetPolyPoints( *outline );
 173         dwg->SetStart0( *outline->begin() );
 174         dwg->SetEnd0( outline->back() );
 175         dwg->SetDrawCoord();
 176 
 177         delete( outline );
 178     }
 179 }

 182 void PCB_POLYGON::AddToBoard()
 183 {
 184     int i = 0;
 185 
 186     if( m_outline.GetCount() > 0 )
 187     {
 188         ZONE_CONTAINER* zone = new ZONE_CONTAINER( m_board );
 189         m_board->Add( zone, ADD_MODE::APPEND );
 190 
 191         zone->SetLayer( m_KiCadLayer );
 192         zone->SetNetCode( m_netCode );
 193 
 194         // add outline

 195         for( i = 0; i < (int) m_outline.GetCount(); i++ )
 196         {
 197             zone->AppendCorner( wxPoint( KiROUND( m_outline[i]->x ),
 198                                          KiROUND( m_outline[i]->y ) ), -1 );
 199         }
 200 
 201         zone->SetZoneClearance( m_width );
 202 
 203         zone->SetPriority( m_priority );
 204 
 205         zone->SetHatch( ZONE_HATCH_STYLE::DIAGONAL_EDGE, zone->GetDefaultHatchPitch(), true );
 206 
 207         if ( m_objType == wxT( 'K' ) )
 208         {
 209             zone->SetIsKeepout( true );
 210             zone->SetDoNotAllowTracks( true );
 211             zone->SetDoNotAllowVias( true );
 212             zone->SetDoNotAllowPads( true );
 213             zone->SetDoNotAllowCopperPour( true );
 214             zone->SetDoNotAllowFootprints( false );
 215         }
 216         else if( m_objType == wxT( 'C' ) )
 217         {
 218             // convert cutouts to keepouts because standalone cutouts are not supported in KiCad

 219             zone->SetIsKeepout( true );
 220             zone->SetDoNotAllowCopperPour( true );
 221             zone->SetDoNotAllowTracks( false );
 222             zone->SetDoNotAllowVias( false );
 223             zone->SetDoNotAllowPads( false );
 224             zone->SetDoNotAllowFootprints( false );
 225         }
 226 
 227         //if( m_filled )

 228         //    zone->BuildFilledPolysListData( m_board );

 229     }
 230 }

 241 void PCB_POLYGON::SetPosOffset( int aX_offs, int aY_offs )
 242 {
 243     int i, island;
 244 
 245     PCB_COMPONENT::SetPosOffset( aX_offs, aY_offs );
 246 
 247     for( i = 0; i < (int) m_outline.GetCount(); i++ )
 248     {
 249         m_outline[i]->x += aX_offs;
 250         m_outline[i]->y += aY_offs;
 251     }
 252 
 253     for( island = 0; island < (int) m_islands.GetCount(); island++ )
 254     {
 255         for( i = 0; i < (int) m_islands[island]->GetCount(); i++ )
 256         {
 257             (*m_islands[island])[i]->x  += aX_offs;
 258             (*m_islands[island])[i]->y  += aY_offs;
 259         }
 260     }
 261 
 262     for( island = 0; island < (int) m_cutouts.GetCount(); island++ )
 263     {
 264         for( i = 0; i < (int) m_cutouts[island]->GetCount(); i++ )
 265         {
 266             (*m_cutouts[island])[i]->x  += aX_offs;
 267             (*m_cutouts[island])[i]->y  += aY_offs;
 268         }
 269     }
 270 }
