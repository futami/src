  34 ///< Shapes of available meanders

  35 enum MEANDER_TYPE {
  36         MT_SINGLE,          // _|^|_, single-sided

  37         MT_START,           // _|^|

  38         MT_FINISH,          // |^|_

  39         MT_TURN,            // |^| or |_|

  40         MT_CHECK_START,     // try fitting a start type, but don't produce a line

  41         MT_CHECK_FINISH,    // try fitting a finish type, but don't produce a line

  42         MT_CORNER,          // line corner

  43         MT_EMPTY            // no meander (straight line)

  44 };
  45 
  46 ///> meander corner shape

  47 enum MEANDER_STYLE {
  48     MEANDER_STYLE_ROUND = 1,          // rounded (90 degree arc)

  49     MEANDER_STYLE_CHAMFER             // chamfered (45 degree segment)

  50 };

 102 /**

 103  * MEANDER_SETTINGS

 104  *

 105  * Holds the geometry of a single meander.

 106  */
 107 class MEANDER_SHAPE
 108 {
 109 public:
 110     /**

 111      * Constructor

 112      *

 113      * @param aPlacer the meander placer instance

 114      * @param aWidth width of the meandered line

 115      * @param aIsDual when true, the shape contains two meandered

 116      *                lines at a given offset (diff pairs)

 117      */
 118     MEANDER_SHAPE( MEANDER_PLACER_BASE* aPlacer, int aWidth, bool aIsDual = false ) :
 119         m_placer( aPlacer ),
 120         m_dual( aIsDual ),
 121         m_width( aWidth ),
 122         m_baselineOffset( 0 )
 123     {
 124         // Do not leave unitialized members, and keep static analyser quiet:

 125         m_type = MT_SINGLE;
 126         m_amplitude = 0;
 127         m_side = false;
 128         m_baseIndex = 0;
 129         m_currentTarget = NULL;
 130         m_meanCornerRadius = 0;
 131     }
 132 
 133     /**

 134      * Function SetType()

 135      *

 136      * Sets the type of the meander.

 137      */
 138     void SetType( MEANDER_TYPE aType )
 139     {
 140         m_type = aType;
 141     }
 142 
 143     /**

 144      * Function Type()

 145      *

 146      * @return the type of the meander.

 147      */
 148     MEANDER_TYPE Type() const
 149     {
 150         return m_type;
 151     }
 152 
 153     /**

 154      * Function SetBaseIndex()

 155      *

 156      * Sets an auxillary index of the segment being meandered in its original LINE.

 157      */
 158     void SetBaseIndex( int aIndex )
 159     {
 160         m_baseIndex = aIndex;
 161     }
 162 
 163     /**

 164      * Function BaseIndex()

 165      *

 166      * @return auxillary index of the segment being meandered in its original LINE.

 167      */
 168     int BaseIndex() const
 169     {
 170         return m_baseIndex;
 171     }
 172 
 173     /**

 174      * Function Amplitude()

 175      *

 176      * @return the amplitude of the meander shape.

 177      */
 178     int Amplitude() const
 179     {
 180         return m_amplitude;
 181     }
 182 
 183     /**

 184      * Function MakeCorner()

 185      *

 186      * Creates a dummy meander shape representing a line corner. Used to define

 187      * the starts/ends of meandered segments.

 188      * @param aP1 corner point of the 1st line

 189      * @param aP2 corner point of the 2nd line (if m_dual == true)

 190      */
 191     void MakeCorner( VECTOR2I aP1, VECTOR2I aP2 = VECTOR2I( 0, 0 ) );
 192 
 193     /**

 194      * Function Resize()

 195      *

 196      * Changes the amplitude of the meander shape to aAmpl and recalculates

 197      * the resulting line chain.

 198      * @param aAmpl new amplitude.

 199      */
 200     void Resize( int aAmpl );
 201 
 202     /**

 203      * Function Recalculate()

 204      *

 205      * Recalculates the line chain representing the meanders's shape.

 206      */
 207     void Recalculate();
 208 
 209     /**

 210      * Function IsDual()

 211      *

 212      * @return true if the shape represents 2 parallel lines (diff pair).

 213      */
 214     bool IsDual() const
 215     {
 216         return m_dual;
 217     }
 218 
 219     /**

 220      * Function Side()

 221      *

 222      * @return true if the meander is to the right of its base segment.

 223      */
 224     bool Side() const
 225     {
 226         return m_side;
 227     }
 228 
 229     /**

 230      * Function End()

 231      *

 232      * @return end vertex of the base segment of the meander shape.

 233      */
 234     VECTOR2I End() const
 235     {
 236         return m_clippedBaseSeg.B;
 237     }
 238 
 239     /**

 240      * Function CLine()

 241      *

 242      * @return the line chain representing the shape of the meander.

 243      */
 244     const SHAPE_LINE_CHAIN& CLine( int aShape ) const
 245     {
 246         return m_shapes[aShape];
 247     }
 248 
 249     /**

 250      * Function MakeEmpty()

 251      *

 252      * Replaces the meander with straight bypass line(s), effectively

 253      * clearing it.

 254      */
 255     void MakeEmpty();
 256 
 257     /**

 258      * Function Fit()

 259      *

 260      * Attempts to fit a meander of a given type onto a segment, avoiding

 261      * collisions with other board features.

 262      * @param aType type of meander shape

 263      * @param aSeg base segment for meandering

 264      * @param aP start point of the meander

 265      * @param aSide side of aSeg to put the meander on (true = right)

 266      * @return true on success.

 267      */
 268     bool Fit( MEANDER_TYPE aType, const SEG& aSeg, const VECTOR2I& aP, bool aSide );
 269 
 270     /**

 271      * Function BaseSegment()

 272      *

 273      * Returns the base segment the meadner was fitted to.

 274      * @return the base segment.

 275      */
 276     const SEG& BaseSegment() const
 277     {
 278         return m_clippedBaseSeg;
 279     }
 280 
 281     /**

 282      * Function BaselineLength()

 283      *

 284      * @return length of the base segment for the meander (i.e.

 285      *         the minimum tuned length.

 286      */
 287     int BaselineLength() const;
 288 
 289     /**

 290      * Function MaxTunableLength()

 291      *

 292      * @return the length of the fitted line chain.

 293      */
 294     int MaxTunableLength() const;
 295 
 296     /**

 297      * Function Settings()

 298      *

 299      * @return the current meandering settings.

 300      */
 301     const MEANDER_SETTINGS& Settings() const;
 302 
 303     /**

 304      * Function Width()

 305      *

 306      * @return width of the meandered line.

 307      */
 308     int Width() const
 309     {
 310         return m_width;
 311     }
 312 
 313     /**

 314      * Function SetBaselineOffset()

 315      *

 316      * Sets the parallel offset between the base segment and the meandered

 317      * line. Used for dual menaders (diff pair) only.

 318      * @param aOffset the offset

 319      */
 320     void SetBaselineOffset( int aOffset )
 321     {
 322         m_baselineOffset = aOffset;
 323     }
 324 
 325 private:
 326     friend class MEANDERED_LINE;
 327 
 328     ///> starts turtle drawing

 329     void start( SHAPE_LINE_CHAIN* aTarget, const VECTOR2D& aWhere, const VECTOR2D& aDir );
 330     ///> moves turtle forward by aLength

 331     void forward( int aLength );
 332     ///> turns the turtle by aAngle

 333     void turn( int aAngle );
 334     ///> tells the turtle to draw a mitered corner of given radius and turn direction

 335     void miter( int aRadius, bool aSide );
 336     ///> tells the turtle to draw an U-like shape

 337     void uShape( int aSides, int aCorner, int aTop );
 338 
 339     ///> generates a 90-degree circular arc

 340     SHAPE_LINE_CHAIN makeMiterShape( VECTOR2D aP, VECTOR2D aDir, bool aSide );
 341 
 342     ///> reflects a point onto other side of a given segment

 343     VECTOR2I reflect( VECTOR2I aP, const SEG& aLine );
 344 
 345     ///> produces a meander shape of given type

 346     SHAPE_LINE_CHAIN genMeanderShape( VECTOR2D aP, VECTOR2D aDir, bool aSide, MEANDER_TYPE aType, int aAmpl, int aBaselineOffset = 0 );
 347 
 348     ///> recalculates the clipped baseline after the parameters of

 349     ///> the meander have been changed.

 350     void updateBaseSegment();
 351 
 352     ///> returns sanitized corner radius value

 353     int cornerRadius() const;
 354 
 355     ///> returns sanitized spacing value

 356     int spacing() const;
 357 
 358     ///> the type

 359     MEANDER_TYPE m_type;
 360     ///> the placer that placed this meander

 361     MEANDER_PLACER_BASE* m_placer;
 362     ///> dual or single line

 363     bool m_dual;
 364     ///> width of the line

 365     int m_width;
 366     ///> amplitude of the meander

 367     int m_amplitude;
 368     ///> offset wrs the base segment (dual only)

 369     int m_baselineOffset;
 370     ///> average radius of meander corners (for correction of DP meanders)

 371     int m_meanCornerRadius;
 372     ///> first point of the meandered line

 373     VECTOR2I m_p0;
 374     ///> base segment (unclipped)

 375     SEG m_baseSeg;
 376     ///> base segment (clipped)

 377     SEG m_clippedBaseSeg;
 378     ///> side (true = right)

 379     bool m_side;
 380     ///> the actual shapes (0 used for single, both for dual)

 381     SHAPE_LINE_CHAIN m_shapes[2];
 382     ///> index of the meandered segment in the base line

 383     int m_baseIndex;
 384     ///> current turtle direction

 385     VECTOR2D m_currentDir;
 386     ///> current turtle position

 387     VECTOR2D m_currentPos;
 388     ///> the line the turtle is drawing on

 389     SHAPE_LINE_CHAIN* m_currentTarget;
 390 };

 393 /**

 394  * MEANDERED_LINE

 395  *

 396  * Represents a set of meanders fitted over a single or two lines.

 397  */
 398 class MEANDERED_LINE
 399 {
 400 public:
 401     MEANDERED_LINE()
 402     {
 403         // Do not leave unitialized members, and keep static analyser quiet:

 404         m_placer = NULL;
 405         m_dual = false;
 406         m_width = 0;
 407         m_baselineOffset = 0;
 408     }
 409 
 410     /**

 411      * Constructor

 412      *

 413      * @param aPlacer the meander placer instance

 414      * @param aIsDual when true, the meanders are generated for two coupled lines

 415      */
 416     MEANDERED_LINE( MEANDER_PLACER_BASE* aPlacer, bool aIsDual = false ) :
 417         m_placer( aPlacer ),
 418         m_dual( aIsDual )
 419     {
 420         // Do not leave unitialized members, and keep static analyser quiet:

 421         m_width = 0;
 422         m_baselineOffset = 0;
 423     }
 424 
 425     ~MEANDERED_LINE()
 426     {
 427         Clear();
 428     }
 429 
 430     /**

 431      * Function AddCorner()

 432      *

 433      * Creates a dummy meander shape representing a line corner. Used to define

 434      * the starts/ends of meandered segments.

 435      * @param aA corner point of the 1st line

 436      * @param aB corner point of the 2nd line (if m_dual == true)

 437      */
 438     void AddCorner( const VECTOR2I& aA, const VECTOR2I& aB = VECTOR2I( 0, 0 ) );
 439 
 440     /**

 441      * Function AddMeander()

 442      *

 443      * Adds a new meander shape the the meandered line.

 444      * @param aShape the meander shape to add

 445      */
 446     void AddMeander( MEANDER_SHAPE* aShape );
 447 
 448     /**

 449      * Function Clear()

 450      *

 451      * Clears the line geometry, removing all corners and meanders.

 452      */
 453     void Clear();
 454 
 455     /**

 456      * Function SetWidth()

 457      *

 458      * Sets the line width.

 459      */
 460     void SetWidth( int aWidth )
 461     {
 462         m_width = aWidth;
 463     }
 464 
 465     /**

 466      * Function MeanderSegment()

 467      *

 468      * Fits maximum amplitude meanders on a given segment and adds to the

 469      * current line.

 470      * @param aSeg the base segment to meander

 471      * @param aBaseIndex index of the base segment in the original line

 472      */
 473     void MeanderSegment( const SEG& aSeg, int aBaseIndex = 0 );
 474 
 475     /// @copydoc MEANDER_SHAPE::SetBaselineOffset()

 476     void SetBaselineOffset( int aOffset )
 477     {
 478         m_baselineOffset = aOffset;
 479     }
 480 
 481     /**

 482      * Function Meanders()

 483      *

 484      * @return set of meander shapes for this line

 485      */
 486     std::vector<MEANDER_SHAPE*>& Meanders()
 487     {
 488         return m_meanders;
 489     }
 490 
 491     /**

 492      * Function CheckSelfIntersections()

 493      *

 494      * Checks if the given shape is intersecting with any other meander in

 495      * the current line.

 496      * @param aShape the shape to check

 497      * @param aClearance clearance value

 498      * @return true, if the meander shape is not colliding

 499      */
 500     bool CheckSelfIntersections( MEANDER_SHAPE* aShape, int aClearance );
 501 
 502     /**

 503      * Function Settings()

 504      *

 505      * @return the current meandering settings.

 506      */
 507     const MEANDER_SETTINGS& Settings() const;
 508 
 509 private:
 510     VECTOR2I m_last;
 511 
 512     MEANDER_PLACER_BASE* m_placer;
 513     std::vector<MEANDER_SHAPE*> m_meanders;
 514 
 515     bool m_dual;
 516     int m_width;
 517     int m_baselineOffset;
 518 };
