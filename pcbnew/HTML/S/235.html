  66 DIALOG_NETLIST::DIALOG_NETLIST( PCB_EDIT_FRAME* aParent, wxString& aNetlistFullFilename )

  72 {

  77 
  78     m_cbUpdateFootprints->SetValue( cfg->m_NetlistDialog.update_footprints );
  79     m_cbDeleteShortingTracks->SetValue( cfg->m_NetlistDialog.delete_shorting_tracks );

  81     m_cbDeleteSinglePadNets->SetValue( cfg->m_NetlistDialog.delete_single_pad_nets );
  82 
  83     m_cbWarnNoNetPad->SetValue( m_warnForNoNetPads );

 125 void DIALOG_NETLIST::OnOpenNetlistClick( wxCommandEvent& event )
 126 {
 127     wxString dirPath = wxFileName( Prj().GetProjectFullName() ).GetPath();
 128 
 129     wxString filename = m_parent->GetLastPath( LAST_PATH_NETLIST );
 130 
 131     if( !filename.IsEmpty() )
 132     {
 133         wxFileName fn = filename;
 134         dirPath = fn.GetPath();
 135         filename = fn.GetFullName();
 136     }
 137 
 138     wxFileDialog FilesDialog( this, _( "Select Netlist" ), dirPath, filename,
 139                               NetlistFileWildcard(), wxFD_DEFAULT_STYLE | wxFD_FILE_MUST_EXIST );
 140 
 141     if( FilesDialog.ShowModal() != wxID_OK )
 142         return;
 143 
 144     m_NetlistFilenameCtrl->SetValue( FilesDialog.GetPath() );
 145     onFilenameChanged();
 146 }

 148 void DIALOG_NETLIST::OnUpdatePCB( wxCommandEvent& event )
 149 {
 150     BOARD*     pcb = m_parent->GetBoard();
 151     wxFileName fn = m_NetlistFilenameCtrl->GetValue();
 152 
 153     if( !fn.IsOk() )
 154     {
 155         wxMessageBox( _("Please, choose a valid netlist file.") );
 156         return;
 157     }
 158 
 159     if( !fn.FileExists() )
 160     {
 161         wxMessageBox( _("The netlist file does not exist.") );
 162         return;
 163     }
 164 
 165     // Give the user a chance to bail out when making changes from a netlist.

 166     if( pcb->IsEmpty() || IsOK( this, _( "The changes made cannot be undone.  "
 167                                          "Are you sure you want to update the PCB?" ) ) )
 168     {
 169         m_MessageWindow->SetLabel( _( "Changes Applied To PCB" ) );
 170         loadNetlist( false );
 171 
 172         m_sdbSizer1Cancel->SetDefault();
 173     }
 174 }

 177 void DIALOG_NETLIST::OnTestFootprintsClick( wxCommandEvent& event )
 178 {
 179     if( m_parent->GetBoard()->GetFirstModule() == nullptr )
 180     {
 181         DisplayInfoMessage( this, _( "No footprints." ) );
 182         return;
 183     }
 184 
 185     wxString        netlistFilename = m_NetlistFilenameCtrl->GetValue();
 186     NETLIST         netlist;
 187     wxBusyCursor    dummy;         // Shows an hourglass while calculating.

 188 
 189     if( !m_parent->ReadNetlistFromFile( netlistFilename, netlist, NULL_REPORTER::GetInstance() ) )
 190         return;
 191 
 192     HTML_MESSAGE_BOX dlg( this, _( "Check footprints" ) );
 193     std::vector<DRC_ITEM*> drcItems;
 194 
 195     TestFootprints( netlist, m_parent->GetBoard(), drcItems );
 196 
 197     for( DRC_ITEM* item : drcItems )
 198         dlg.AddHTML_Text( item->ShowHtml( m_parent ) );
 199 
 200     dlg.ShowModal();
 201 }

 211 void DIALOG_NETLIST::onFilenameChanged()
 212 {
 213     if( m_initialized )
 214     {
 215         wxFileName fn = m_NetlistFilenameCtrl->GetValue();
 216 
 217         if( fn.IsOk() )
 218         {
 219             if( fn.FileExists() )
 220             {
 221                 m_netlistPath = m_NetlistFilenameCtrl->GetValue();
 222                 loadNetlist( true );
 223             }
 224             else
 225             {
 226                 m_MessageWindow->Clear();
 227                 REPORTER& reporter = m_MessageWindow->Reporter();
 228                 reporter.Report( _( "The netlist file does not exist." ), RPT_SEVERITY_ERROR );
 229             }
 230         }
 231     }
 232 }

 263 void DIALOG_NETLIST::loadNetlist( bool aDryRun )
 264 {
 265     wxString netlistFileName = m_NetlistFilenameCtrl->GetValue();
 266     wxFileName fn = netlistFileName;
 267 
 268     if( !fn.IsOk() || !fn.FileExists() )
 269         return;
 270 
 271     m_MessageWindow->Clear();
 272     REPORTER& reporter = m_MessageWindow->Reporter();
 273 
 274     wxBusyCursor busy;
 275 
 276     wxString msg;
 277     msg.Printf( _( "Reading netlist file \"%s\".\n" ), GetChars( netlistFileName ) );
 278     reporter.ReportHead( msg, RPT_SEVERITY_INFO );
 279 
 280     if( m_matchByTimestamp->GetSelection() == 1 )
 281         msg = _( "Using reference designators to match components and footprints.\n" );
 282     else
 283         msg = _( "Using tstamps (unique IDs) to match components and footprints.\n" );
 284 
 285     reporter.ReportHead( msg, RPT_SEVERITY_INFO );
 286     m_MessageWindow->SetLazyUpdate( true ); // Use lazy update to speed the creation of the report

 287                                             // (the window is not updated for each message)

 288     m_matchByUUID = m_matchByTimestamp->GetSelection() == 0;
 289 
 290     NETLIST netlist;
 291 
 292     netlist.SetDeleteExtraFootprints( m_cbDeleteExtraFootprints->GetValue() );
 293     netlist.SetFindByTimeStamp( m_matchByUUID );
 294     netlist.SetReplaceFootprints( m_cbUpdateFootprints->GetValue() );
 295 
 296     if( !m_parent->ReadNetlistFromFile( netlistFileName, netlist, reporter ) )
 297         return;
 298 
 299     BOARD_NETLIST_UPDATER updater( m_parent, m_parent->GetBoard() );
 300     updater.SetReporter ( &reporter );
 301     updater.SetIsDryRun( aDryRun );
 302     updater.SetLookupByTimestamp( m_matchByUUID );
 303     updater.SetDeleteUnusedComponents ( m_cbDeleteExtraFootprints->GetValue() );
 304     updater.SetReplaceFootprints( m_cbUpdateFootprints->GetValue() );
 305     updater.SetDeleteSinglePadNets( m_cbDeleteSinglePadNets->GetValue() );
 306     m_warnForNoNetPads = m_cbWarnNoNetPad->GetValue();
 307     updater.SetWarnPadNoNetInNetlist( m_warnForNoNetPads );
 308     updater.UpdateNetlist( netlist );
 309 
 310     // The creation of the report was made without window update: the full page must be displayed

 311     m_MessageWindow->Flush( true );
 312 
 313     if( aDryRun )
 314         return;
 315 
 316     m_parent->OnNetlistChanged( updater, &m_runDragCommand );
 317 }
