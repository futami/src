  25 #ifndef PCB_SELECTION_CONDITIONS_H_
  26 #define PCB_SELECTION_CONDITIONS_H_
  27 
  28 #include <tool/selection.h>
  29 #include <tool/selection_conditions.h>
  30 #include <class_board_item.h>
  31 
  32 
  33 class PCB_SELECTION_CONDITIONS : public SELECTION_CONDITIONS
  34 {
  35 public:
  36     /**

  37      * Function OnlyConnectedItems

  38      * Tests if selection contains exclusively connected items (pads, tracks, vias, zones).

  39      * @param aSelection is the selection to be tested.

  40      * @return True if there are only connected items connected.

  41      */
  42     static bool OnlyConnectedItems( const SELECTION& aSelection );
  43 
  44     /**

  45      * Function SameNet

  46      * Creates a functor that tests if selection contains items belonging to the same net or are

  47      * unconnected if aAllowUnconnected == true.

  48      * @param aAllowUnconnected determines if unconnected items (with no net code assigned) should

  49      * be treated as connected to the same net.

  50      * @return Functor testing if selected items are belonging to the same net.

  51      */
  52     static SELECTION_CONDITION SameNet( bool aAllowUnconnected = false );
  53 
  54     /**

  55      * Function SameLayer

  56      * Creates a functor that tests if selection contains items that belong exclusively to the same

  57      * layer. In case of items belonging to multiple layers, it is enough to have a single common

  58      * layer with other items.

  59      * @return Functor testing if selected items share at least one common layer.

  60      */
  61     static SELECTION_CONDITION SameLayer();
  62 
  63 
  64 private:
  65     ///> Helper function used by SameNet()

  66     static bool sameNetFunc( const SELECTION& aSelection, bool aAllowUnconnected );
  67 
  68     ///> Helper function used by SameLayer()

  69     static bool sameLayerFunc( const SELECTION& aSelection );
  70 };
  71 
  72 #endif /* PCB_SELECTION_CONDITIONS_H_ */
