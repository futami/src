  40 class PCB_TARGET : public BOARD_ITEM
  41 {
  42     int     m_Shape;            // bit 0 : 0 = draw +, 1 = draw X

  43     int     m_Size;
  44     int     m_Width;
  45     wxPoint m_Pos;
  46 
  47 public:
  48     PCB_TARGET( BOARD_ITEM* aParent );
  49 
  50     // Do not create a copy constructor.  The one generated by the compiler is adequate.

  51 
  52     PCB_TARGET( BOARD_ITEM* aParent, int aShape, PCB_LAYER_ID aLayer,
  53                 const wxPoint& aPos, int aSize, int aWidth );
  54 
  55     // Do not create a copy constructor & operator=.

  56     // The ones generated by the compiler are adequate.

  57 
  58     ~PCB_TARGET();
  59 
  60     static inline bool ClassOf( const EDA_ITEM* aItem )
  61     {
  62         return aItem && PCB_TARGET_T == aItem->Type();
  63     }
  64 
  65     void SetPosition( const wxPoint& aPos ) override { m_Pos = aPos; }
  66     wxPoint GetPosition() const override { return m_Pos; }
  67 
  68     void SetShape( int aShape )     { m_Shape = aShape; }
  69     int GetShape() const            { return m_Shape; }
  70 
  71     void SetSize( int aSize )       { m_Size = aSize; }
  72     int GetSize() const             { return m_Size; }
  73 
  74     void SetWidth( int aWidth )     { m_Width = aWidth; }
  75     int GetWidth() const            { return m_Width; }
  76 
  77     void Move( const wxPoint& aMoveVector ) override
  78     {
  79         m_Pos += aMoveVector;
  80     }
  81 
  82     void Rotate( const wxPoint& aRotCentre, double aAngle ) override;
  83 
  84     void Flip( const wxPoint& aCentre, bool aFlipLeftRight ) override;
  85 
  86     wxString GetClass() const override
  87     {
  88         return wxT( "PCB_TARGET" );
  89     }
  90 
  91     bool HitTest( const wxPoint& aPosition, int aAccuracy = 0 ) const override;
  92     bool HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy = 0 ) const override;
  93 
  94     // Virtual function

  95     const EDA_RECT GetBoundingBox() const override;
  96 
  97     wxString GetSelectMenuText( EDA_UNITS aUnits ) const override;
  98 
  99     BITMAP_DEF GetMenuImage() const override;
