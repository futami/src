  26 #ifndef DIALOG_FIND_BASE_H
  27 #define DIALOG_FIND_BASE_H
  28 
  29 #include <boost/function.hpp>
  30 #include <sys/types.h>
  31 #include <wx/event.h>
  32 #include <deque>
  33 
  34 #include <class_board_item.h>

  36 #include <dialog_find_base.h>
  37 
  38 using namespace std;
  39 
  40 class DIALOG_FIND : public DIALOG_FIND_BASE
  41 {
  42 public:
  43     DIALOG_FIND( PCB_BASE_FRAME* aParent );
  44 
  45     /**

  46      * Returns the currently found item or nullptr in the case of no items found

  47      * @return

  48      */
  49     inline BOARD_ITEM* GetItem() const
  50     {
  51         if( m_it != m_hitList.end() )
  52             return *m_it;
  53         else
  54             return nullptr;
  55     }
  56 
  57     /**

  58      * Function to be called on each found event.  Must be able to handle nullptr in the

  59      * case where no item is found

  60      * @param aCallback

  61      */
  62     void SetCallback( boost::function<void( BOARD_ITEM* )> aCallback )
  63     {
  64         m_highlightCallback = aCallback;
  65     }
  66 
  67 
  68 private:
  69     PCB_BASE_FRAME*                     m_frame;
  70     std::deque<BOARD_ITEM*>             m_hitList;
  71     std::deque<BOARD_ITEM*>::iterator   m_it;
  72     bool                                m_upToDate;
  73 
  74     boost::function<void( BOARD_ITEM* )> m_highlightCallback;
  75 
  76     void onTextEnter( wxCommandEvent& event ) override;
  77     void onFindNextClick( wxCommandEvent& event ) override;
  78     void onFindPreviousClick( wxCommandEvent& event ) override;
  79     void onSearchAgainClick( wxCommandEvent& event ) override;
  80     void onClose( wxCommandEvent& event ) override;
  81     void search( bool direction );
  82 };
  83 
  84 #endif /* DIALOG_FIND_BASE_H */
