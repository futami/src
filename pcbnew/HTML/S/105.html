  48 /**

  49  * Dialog class to select a footprint from the libraries. This is the master

  50  * View class in a Model-View-Adapter (mediated MVC) architecture. The other

  51  * pieces are in:

  52  *

  53  * - Adapter: CMP_TREE_MODEL_ADAPTER in common/cmp_tree_model_adapter.h

  54  * - Model: CMP_TREE_NODE and descendants in common/cmp_tree_model.h

  55  *

  56  * Because everything is tied together in the adapter class, see that file

  57  * for thorough documentation. A simple example usage follows:

  58  *

  59  *     // Create the adapter class

  60  *     auto adapter( FP_TREE_MODEL_ADAPTER::Create( Prj().PcbFootprintLibs() ) );

  61  *

  62  *     // Perform any configuration of adapter properties here

  63  *     adapter->SetPreselectNode( "LIB_NICKNAME", "FP_NAME", 2 );

  64  *

  65  *     // Initialize model from #FP_LIB_TABLE

  66  *     libNicknames = libs->GetLogicalLibs();

  67  *

  68  *     for( auto nickname : libNicknames )

  69  *     {

  70  *         adapter->AddLibrary( nickname );

  71  *     }

  72  *

  73  *     // Create and display dialog

  74  *     DIALOG_CHOOSE_FOOTPRINT dlg( this, title, adapter, 1 );

  75  *     bool selected = ( dlg.ShowModal() != wxID_CANCEL );

  76  *

  77  *     // Receive part

  78  *     if( selected )

  79  *     {

  80  *         int unit;

  81  *         #LIB_ID id = dlg.GetSelectedAlias( &unit );

  82  *         do_something( id, unit );

  83  *     }

  84  *

  85  */
  86 class DIALOG_CHOOSE_FOOTPRINT : public DIALOG_SHIM
  87 {
  88 public:
  89     /**

  90      * Create dialog to choose component.

  91      *

  92      * @param aParent   a PCB_BASE_FRAME parent window.

  93      * @param aAdapter  FP_TREE_MODEL_ADAPTER::PTR. See CMP_TREE_MODEL_ADAPTER

  94      *                  for documentation.

  95      */
  96     DIALOG_CHOOSE_FOOTPRINT( PCB_BASE_FRAME* aParent, const wxString& aTitle,
  97                              FP_TREE_MODEL_ADAPTER::PTR& aAdapter );
  98 
  99     ~DIALOG_CHOOSE_FOOTPRINT();
 100 
 101     /**

 102      * To be called after this dialog returns from ShowModal().

 103      *

 104      * @return the #LIB_ID of the symbol that has been selected.

 105      */
 106     LIB_ID GetSelectedLibId() const;

 108     /** Function IsExternalBrowserSelected

 109      *

 110      * @return true, iff the user pressed the thumbnail view of the component to

 111      *               launch the component browser.

 112      */
 113     bool IsExternalBrowserSelected() const

 115         return m_external_browser_requested;
 116     }
