 150 void PANEL_SETUP_BOARD_STACKUP::onAddDielectricLayer( wxCommandEvent& event )
 151 {
 152     // Build Dielectric layers list:

 153     wxArrayString d_list;
 154     std::vector<int> rows;  // indexes of row values for each selectable item

 155     int row = -1;
 156 
 157     for( BOARD_STACKUP_ROW_UI_ITEM& item : m_rowUiItemsList )
 158     {
 159         row++;
 160 
 161         if( !item.m_isEnabled )
 162             continue;
 163 
 164         BOARD_STACKUP_ITEM* brd_stackup_item = item.m_Item;
 165 
 166         if( brd_stackup_item->GetType() == BS_ITEM_TYPE_DIELECTRIC )
 167         {
 168             if( brd_stackup_item->GetSublayersCount() > 1 )
 169             {
 170                 d_list.Add( wxString::Format( _( "Layer \"%s\" (sublayer %d/%d)" ),
 171                                 brd_stackup_item->FormatDielectricLayerName(),
 172                                 item.m_SubItem+1, brd_stackup_item->GetSublayersCount() ) );
 173             }
 174             else
 175                 d_list.Add( brd_stackup_item->FormatDielectricLayerName() );
 176 
 177             rows.push_back( row );
 178         }
 179     }
 180 
 181     // Show list

 182     int index = wxGetSingleChoiceIndex( wxEmptyString, _("Dielectric Layers List"),
 183                                         d_list);
 184 
 185     if( index < 0 )
 186         return;
 187 
 188     row = rows[index];
 189 
 190     BOARD_STACKUP_ITEM* brd_stackup_item = m_rowUiItemsList[row].m_Item;
 191     int new_sublayer = m_rowUiItemsList[row].m_SubItem;
 192 
 193     // Insert a new item after the selected item

 194     brd_stackup_item->AddDielectricPrms( new_sublayer+1 );
 195 
 196     rebuildLayerStackPanel();
 197 }

 200 void PANEL_SETUP_BOARD_STACKUP::onRemoveDielectricLayer( wxCommandEvent& event )
 201 {
 202     // Build deletable Dielectric layers list.

 203     // A layer can be deleted if there are 2 (or more) dielectric sub-layers

 204     // between 2 copper layers

 205     wxArrayString d_list;
 206     std::vector<int> rows;  // indexes of row values for each selectable item

 207 
 208     int ui_row = 0;     // The row index in m_rowUiItemsList of items in choice list

 209 
 210     // Build the list of dielectric layers:

 211     for( auto item : m_stackup.GetList() )
 212     {
 213         if( !item->IsEnabled() || item->GetType() != BS_ITEM_TYPE_DIELECTRIC ||
 214             item->GetSublayersCount() <= 1 )
 215         {
 216             ui_row++;
 217             continue;
 218         }
 219 
 220         for( int ii = 0; ii < item->GetSublayersCount(); ii++ )
 221         {
 222             d_list.Add( wxString::Format( "Layer \"%s\" sublayer %d/%d",
 223                             item->FormatDielectricLayerName(), ii+1,
 224                             item->GetSublayersCount() ) );
 225 
 226             rows.push_back( ui_row++ );
 227         }
 228     }
 229 
 230     // Show choice list

 231     int index = wxGetSingleChoiceIndex( wxEmptyString, _("Dielectric Layers List"),
 232                                         d_list );
 233 
 234     if( index < 0 )
 235         return;
 236 
 237     ui_row = rows[index];
 238 
 239     BOARD_STACKUP_ITEM* brd_stackup_item = m_rowUiItemsList[ui_row].m_Item;
 240     int sublayer = m_rowUiItemsList[ui_row].m_SubItem;
 241 
 242     // Remove the selected sub item for the selected dielectric layer

 243     brd_stackup_item->RemoveDielectricPrms( sublayer );
 244 
 245     rebuildLayerStackPanel();
 246 }

 331 void PANEL_SETUP_BOARD_STACKUP::synchronizeWithBoard( bool aFullSync )
 332 {
 333     BOARD_STACKUP& brd_stackup = m_brdSettings->GetStackupDescriptor();
 334 
 335     if( aFullSync )
 336     {
 337         int thickness = m_brdSettings->GetBoardThickness();
 338         m_thicknessCtrl->SetValue( StringFromValue( m_units, thickness, true, true ) );
 339 
 340         m_rbDielectricConstraint->SetSelection( brd_stackup.m_HasDielectricConstrains ? 1 : 0 );
 341         m_choiceEdgeConn->SetSelection( brd_stackup.m_EdgeConnectorConstraints );
 342         m_cbCastellatedPads->SetValue( brd_stackup.m_CastellatedPads );
 343         m_cbEgdesPlated->SetValue( brd_stackup.m_EdgePlating );
 344 
 345         // find the choice depending on the initial finish setting

 346         wxArrayString initial_finish_list = GetCopperFinishStandardList( false );
 347         unsigned idx;
 348 
 349         for( idx = 0; idx < initial_finish_list.GetCount(); idx++ )
 350         {
 351             if( initial_finish_list[idx] ==  brd_stackup.m_FinishType )
 352                 break;
 353         }
 354 
 355         // Now init the choice (use last choice: "User defined" if not found )

 356         if( idx >= initial_finish_list.GetCount() )
 357             idx = initial_finish_list.GetCount()-1;
 358 
 359         m_choiceFinish->SetSelection( idx );
 360     }
 361 
 362     int row = 0;
 363 
 364     for( BOARD_STACKUP_ROW_UI_ITEM& ui_row_item : m_rowUiItemsList )
 365     {
 366         BOARD_STACKUP_ITEM* item = ui_row_item.m_Item;
 367         int sub_item = ui_row_item.m_SubItem;
 368 
 369         if( item->GetType() == BS_ITEM_TYPE_DIELECTRIC )
 370         {
 371             wxChoice* choice = dynamic_cast<wxChoice*>( ui_row_item.m_LayerTypeCtrl );
 372 
 373             if( choice )
 374                 choice->SetSelection( item->GetTypeName() == KEY_CORE ? 0 : 1 );
 375         }
 376 
 377         if( item->IsMaterialEditable() )
 378         {
 379             wxTextCtrl* matName = dynamic_cast<wxTextCtrl*>( ui_row_item.m_MaterialCtrl );
 380 
 381             if( matName )
 382             {
 383                 if( IsPrmSpecified( item->GetMaterial( sub_item ) ) )
 384                     matName->SetValue( item->GetMaterial( sub_item ) );
 385                 else
 386                     matName->SetValue( wxGetTranslation( NotSpecifiedPrm() ) );
 387             }
 388         }
 389 
 390         if( item->IsThicknessEditable() )
 391         {
 392             wxTextCtrl* textCtrl = dynamic_cast<wxTextCtrl*>( ui_row_item.m_ThicknessCtrl );
 393 
 394             if( textCtrl )
 395                 textCtrl->SetValue( StringFromValue( m_units,
 396                                         item->GetThickness( sub_item ), true, true ) );
 397 
 398             if( item->GetType() == BS_ITEM_TYPE_DIELECTRIC )
 399             {
 400                 wxCheckBox* cb_box = dynamic_cast<wxCheckBox*> ( ui_row_item.m_ThicknessLockCtrl );
 401 
 402                 if( cb_box )
 403                     cb_box->SetValue( item->IsThicknessLocked( sub_item ) );
 404             }
 405         }
 406 
 407         if( item->IsColorEditable() )
 408         {
 409             auto bm_combo = dynamic_cast<wxBitmapComboBox*>( ui_row_item.m_ColorCtrl );
 410             int  color_idx = 0;
 411 
 412             if( item->GetColor().StartsWith( "#" ) )  // User defined color

 413             {
 414                 wxColour color( item->GetColor() );
 415                 m_UserColors[row] = color;
 416                 color_idx = GetColorUserDefinedListIdx();
 417 
 418                 if( bm_combo )      // Update user color shown in the wxBitmapComboBox

 419                 {
 420                     bm_combo->SetString( color_idx, color.GetAsString( wxC2S_HTML_SYNTAX ) );
 421                     wxBitmap layerbmp( m_colorSwatchesSize.x, m_colorSwatchesSize.y );
 422                     LAYER_SELECTOR::DrawColorSwatch( layerbmp, COLOR4D(), COLOR4D( color ) );
 423                     bm_combo->SetItemBitmap( color_idx, layerbmp );
 424                 }
 425             }
 426             else
 427             {
 428                 const FAB_LAYER_COLOR* color_list = GetColorStandardList();
 429 
 430                 for( int ii = 0; ii < GetColorStandardListCount(); ii++ )
 431                 {
 432                     if( color_list[ii].m_ColorName == item->GetColor() )
 433                     {
 434                         color_idx = ii;
 435                         break;
 436                     }
 437                 }
 438             }
 439 
 440             if( bm_combo )
 441                 bm_combo->SetSelection( color_idx );
 442         }
 443 
 444         if( item->HasEpsilonRValue() )
 445         {
 446             wxString txt;
 447             txt.Printf( "%.2f", item->GetEpsilonR( sub_item ) );
 448             wxTextCtrl* textCtrl = dynamic_cast<wxTextCtrl*>( ui_row_item.m_EpsilonCtrl );
 449 
 450             if( textCtrl )
 451                 textCtrl->SetValue( txt );
 452         }
 453 
 454         if( item->HasLossTangentValue() )
 455         {
 456             wxString txt;
 457             txt.Printf( "%g", item->GetLossTangent( sub_item ) );
 458             wxTextCtrl* textCtrl = dynamic_cast<wxTextCtrl*>( ui_row_item.m_LossTgCtrl );
 459 
 460             if( textCtrl )
 461                 textCtrl->SetValue( txt );
 462         }
 463     }
 464 
 465     // Now enable/disable stackup items, according to the m_enabledLayers config

 466     showOnlyActiveLayers();
 467 
 468     updateIconColor();
 469 }

 472 void PANEL_SETUP_BOARD_STACKUP::showOnlyActiveLayers()
 473 {
 474 
 475     // Now enable/disable stackup items, according to the m_enabledLayers config

 476     // Calculate copper layer count from m_enabledLayers, and *do not use* brd_stackup

 477     // for that, because it is not necessary up to date

 478     // (for instance after modifying the layer count from the panel layers in dialog)

 479     LSET copperMask = m_enabledLayers & ( LSET::ExternalCuMask() | LSET::InternalCuMask() );
 480     int copperLayersCount = copperMask.count();
 481 
 482     for( BOARD_STACKUP_ROW_UI_ITEM& ui_row_item: m_rowUiItemsList )
 483     {
 484         bool show_item;
 485         BOARD_STACKUP_ITEM* item = ui_row_item.m_Item;
 486 
 487         if( item->GetType() == BS_ITEM_TYPE_DIELECTRIC )
 488             // the m_DielectricLayerId is not a copper layer id, it is a dielectric idx from 1

 489             show_item = item->GetDielectricLayerId() < copperLayersCount;
 490         else
 491             show_item = m_enabledLayers[item->GetBrdLayerId()];
 492 
 493         item->SetEnabled( show_item );
 494 
 495         ui_row_item.m_isEnabled = show_item;
 496 
 497         // Show or not items of this row:

 498         ui_row_item.m_Icon->Show( show_item );
 499         ui_row_item.m_LayerName->Show( show_item );
 500         ui_row_item.m_LayerTypeCtrl->Show( show_item );
 501         ui_row_item.m_MaterialCtrl->Show( show_item );
 502 
 503         if( ui_row_item.m_MaterialButt )
 504             ui_row_item.m_MaterialButt->Show( show_item );
 505 
 506         ui_row_item.m_ThicknessCtrl->Show( show_item );
 507         ui_row_item.m_ThicknessLockCtrl->Show( show_item );
 508         ui_row_item.m_ColorCtrl->Show( show_item );
 509         ui_row_item.m_EpsilonCtrl->Show( show_item );
 510         ui_row_item.m_LossTgCtrl->Show( show_item );
 511     }
 512 }

1158             if( !checkBox->GetValue() )  // Only not locked dielectric thickness can be modified

1159                 continue;
1160             else
1161             {
1162                 fixed_thickness_cnt++;
1163 
1164                 if( item->GetThickness( sublayer_idx ) < 0 )
1165                     thickness_error = true;
1166             }
1167         }
1168 
1169         thickness += item->GetThickness( sublayer_idx );
1170     }
1171 
1172     if( thickness_error )
1173     {
1174         wxMessageBox( _( "A locked dielectric thickness is < 0\n"
1175                          "Unlock it or change its thickness") );
1176         return;
1177     }
1178 
1179     // the number of adjustable dielectric layers must obvioulsly be > 0

1180     // So verify the user has at least one dielectric layer free

1181     int adjustableDielectricCount = dielectricCount - fixed_thickness_cnt;
1182 
1183     if( adjustableDielectricCount <= 0 )
1184     {
1185         wxMessageBox( _( "Cannot calculate dielectric thickness\n"
1186                          "At least one dielectric layer must be not locked") );
1187         return;
1188     }
1189 
1190     int dielectric_thickness = GetPcbThickness() - thickness;
1191 
1192     if( dielectric_thickness <= 0 ) // fixed thickness is too big: cannot calculate free thickness

1193     {
1194         wxMessageBox( _( "Cannot calculate dielectric thickness\n"
1195                          "Fixed thickness too big or board thickness too small") );
1196         return;
1197     }
1198 
1199     dielectric_thickness /= adjustableDielectricCount;
1200 
1201     // Update items thickness, and the values displayed on screen

1202     for( BOARD_STACKUP_ROW_UI_ITEM& ui_item : m_rowUiItemsList )
1203     {
1204         BOARD_STACKUP_ITEM* item = ui_item.m_Item;
1205         int sublayer_idx = ui_item.m_SubItem;
1206 
1207         if( item->GetType() == BS_ITEM_TYPE_DIELECTRIC && ui_item.m_isEnabled )
1208         {
1209             wxCheckBox* checkBox = static_cast<wxCheckBox*>( ui_item.m_ThicknessLockCtrl );
1210 
1211             if( !checkBox->GetValue() )  // Not locked thickness: can be modified

1212             {
1213                 item->SetThickness( dielectric_thickness, sublayer_idx );
1214                 wxTextCtrl* textCtrl = static_cast<wxTextCtrl*>( ui_item.m_ThicknessCtrl );
1215                 textCtrl->SetValue( StringFromValue( m_units, item->GetThickness( sublayer_idx ),
1216                                                      true, true ) );
1217             }
1218         }
1219     }
1220 }
