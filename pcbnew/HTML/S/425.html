  30 #ifndef _GENDRILL_GERBER_WRITER_
  31 #define _GENDRILL_GERBER_WRITER_
  32 
  33 #include <gendrill_file_writer_base.h>
  34 
  35 class BOARD;
  36 
  37 /**

  38  * GERBER_WRITER is a class mainly used to create Gerber drill files

  39  */
  40 class GERBER_WRITER: public GENDRILL_WRITER_BASE
  41 {
  42 public:
  43     GERBER_WRITER( BOARD* aPcb );
  44 
  45     virtual ~GERBER_WRITER()
  46     {
  47     }
  48 
  49     /**

  50      * Function SetFormat

  51      * Initialize internal parameters to match the given format

  52      * @param aRightDigits = number of digits for mantissa part of coordinates (5 or 6)

  53      */
  54     void SetFormat( int aRightDigits = 6 );
  55 
  56     /**

  57      * Function SetOptions

  58      * Initialize internal parameters to match drill options

  59      * note: PTH and NPTH are always separate files in Gerber format

  60      * @param aOffset = drill coordinates offset

  61      */
  62     void SetOptions( wxPoint aOffset )
  63     {
  64         m_offset = aOffset;
  65         m_merge_PTH_NPTH = false;
  66     }
  67 
  68     /**

  69      * Function CreateDrillandMapFilesSet

  70      * Creates the full set of Excellon drill file for the board

  71      * filenames are computed from the board name, and layers id

  72      * @param aPlotDirectory = the output folder

  73      * @param aGenDrill = true to generate the EXCELLON drill file

  74      * @param aGenMap = true to generate a drill map file

  75      * @param aReporter = a REPORTER to return activity or any message (can be NULL)

  76      */
  77     void CreateDrillandMapFilesSet( const wxString& aPlotDirectory,
  78                                     bool aGenDrill, bool aGenMap,
  79                                     REPORTER * aReporter = NULL );
  80 
  81 private:
  82     /**

  83      * Function createDrillFile

  84      * Creates an Excellon drill file

  85      * @param aFullFilename = the full filename

  86      * @param aIsNpth = true for a NPTH file, false for a PTH file

  87      * @param aLayerPair = first board layer and the last board layer for this drill file

  88      * for blind buried vias, they are not always top and bottom layers

  89      * @return hole count, or -1 if the file cannot be created

  90      */
  91     int  createDrillFile( wxString& aFullFilename, bool aIsNpth, DRILL_LAYER_PAIR aLayerPair );
  92 
  93     /**

  94      * @return a filename which identify the drill file function.

  95      * it is the board name with the layer pair names added, and for separate

  96      * (PTH and NPTH) files, "-NPH" or "-NPTH" added

  97      * @param aPair = the layer pair

  98      * @param aNPTH = true to generate the filename of NPTH holes

  99      * @param aMerge_PTH_NPTH = true to generate the filename of a file which containd both

 100      * NPH and NPTH holes

 101      */
 102     virtual const wxString getDrillFileName( DRILL_LAYER_PAIR aPair, bool aNPTH,
 103                                              bool aMerge_PTH_NPTH ) const override;
 104 };

 106 #endif  //  #ifndef _GENDRILL_GERBER_WRITER_
