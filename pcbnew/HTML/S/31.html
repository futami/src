  59 COMMIT& BOARD_COMMIT::Stage( EDA_ITEM* aItem, CHANGE_TYPE aChangeType )
  60 {
  61     // if aItem belongs a footprint, the full footprint will be saved

  62     // because undo/redo does not handle "sub items" modifications

  63     if( aItem && aItem->Type() != PCB_MODULE_T && aChangeType == CHT_MODIFY )
  64     {
  65         EDA_ITEM* item = aItem->GetParent();
  66 
  67         if( item && item->Type() == PCB_MODULE_T )  // means aItem belongs a footprint

  68             aItem = item;
  69     }
  70 
  71     return COMMIT::Stage( aItem, aChangeType );
  72 }

  84 void BOARD_COMMIT::Push( const wxString& aMessage, bool aCreateUndoEntry, bool aSetDirtyBit )
  85 {
  86     // Objects potentially interested in changes:

  87     PICKED_ITEMS_LIST   undoList;
  88     KIGFX::VIEW*        view = m_toolMgr->GetView();
  89     BOARD*              board = (BOARD*) m_toolMgr->GetModel();
  90     PCB_BASE_FRAME*     frame = (PCB_BASE_FRAME*) m_toolMgr->GetToolHolder();
  91     auto                connectivity = board->GetConnectivity();
  92     std::set<EDA_ITEM*> savedModules;
  93     SELECTION_TOOL*     selTool = m_toolMgr->GetTool<SELECTION_TOOL>();
  94     bool                itemsDeselected = false;
  95 
  96     if( Empty() )
  97         return;
  98 
  99     for( COMMIT_LINE& ent : m_changes )
 100     {
 101         int changeType = ent.m_type & CHT_TYPE;
 102         int changeFlags = ent.m_type & CHT_FLAGS;
 103         BOARD_ITEM* boardItem = static_cast<BOARD_ITEM*>( ent.m_item );
 104 
 105         // Module items need to be saved in the undo buffer before modification

 106         if( m_editModules )
 107         {
 108             // Be sure that we are storing a module

 109             if( ent.m_item->Type() != PCB_MODULE_T )
 110                 ent.m_item = ent.m_item->GetParent();
 111 
 112             // We have not saved the module yet, so let's create an entry

 113             if( savedModules.count( ent.m_item ) == 0 )
 114             {
 115                 if( !ent.m_copy )
 116                 {
 117                     wxASSERT( changeType != CHT_MODIFY );     // too late to make a copy..

 118                     ent.m_copy = ent.m_item->Clone();
 119                 }
 120 
 121                 wxASSERT( ent.m_item->Type() == PCB_MODULE_T );
 122                 wxASSERT( ent.m_copy->Type() == PCB_MODULE_T );
 123 
 124                 if( aCreateUndoEntry )
 125                 {
 126                     ITEM_PICKER itemWrapper( nullptr, ent.m_item, UR_CHANGED );
 127                     itemWrapper.SetLink( ent.m_copy );
 128                     undoList.PushItem( itemWrapper );
 129                     frame->SaveCopyInUndoList( undoList, UR_CHANGED );
 130                 }
 131 
 132                 savedModules.insert( ent.m_item );
 133                 static_cast<MODULE*>( ent.m_item )->SetLastEditTime();
 134             }
 135         }
 136 
 137         switch( changeType )
 138         {
 139             case CHT_ADD:
 140             {
 141                 if( m_editModules )
 142                 {
 143                     // modules inside modules are not supported yet

 144                     wxASSERT( boardItem->Type() != PCB_MODULE_T );
 145 
 146                     boardItem->SetParent( board->Modules().front() );
 147 
 148                     if( !( changeFlags & CHT_DONE ) )
 149                         board->Modules().front()->Add( boardItem );
 150                 }
 151                 else if( boardItem->Type() == PCB_MODULE_TEXT_T ||
 152                          boardItem->Type() == PCB_MODULE_EDGE_T ||
 153                          boardItem->Type() == PCB_MODULE_ZONE_AREA_T )
 154                 {
 155                     wxASSERT( boardItem->GetParent() &&
 156                               boardItem->GetParent()->Type() == PCB_MODULE_T );
 157                 }
 158                 else
 159                 {
 160                     if( aCreateUndoEntry )
 161                         undoList.PushItem( ITEM_PICKER( nullptr, boardItem, UR_NEW ) );
 162 
 163                     if( !( changeFlags & CHT_DONE ) )
 164                         board->Add( boardItem );        // handles connectivity

 165                 }
 166 
 167                 if( boardItem->Type() != PCB_NETINFO_T )
 168                     view->Add( boardItem );
 169 
 170                 break;
 171             }
 172 
 173             case CHT_REMOVE:
 174             {
 175                 if( !m_editModules && aCreateUndoEntry )
 176                     undoList.PushItem( ITEM_PICKER( nullptr, boardItem, UR_DELETED ) );
 177 
 178                 if( boardItem->IsSelected() )
 179                 {
 180                     selTool->RemoveItemFromSel( boardItem, true /* quiet mode */ );
 181                     itemsDeselected = true;
 182                 }
 183 
 184                 switch( boardItem->Type() )
 185                 {
 186                 // Module items

 187                 case PCB_PAD_T:
 188                 case PCB_MODULE_EDGE_T:
 189                 case PCB_MODULE_TEXT_T:
 190                 case PCB_MODULE_ZONE_AREA_T:
 191                     // This level can only handle module items when editing modules

 192                     wxASSERT( m_editModules );
 193 
 194                     if( boardItem->Type() == PCB_MODULE_TEXT_T )
 195                     {
 196                         TEXTE_MODULE* text = static_cast<TEXTE_MODULE*>( boardItem );
 197 
 198                         // don't allow deletion of Reference or Value

 199                         if( text->GetType() != TEXTE_MODULE::TEXT_is_DIVERS )
 200                             break;
 201                     }
 202 
 203                     view->Remove( boardItem );
 204 
 205                     if( !( changeFlags & CHT_DONE ) )
 206                     {
 207                         MODULE* module = static_cast<MODULE*>( boardItem->GetParent() );
 208                         wxASSERT( module && module->Type() == PCB_MODULE_T );
 209                         module->Delete( boardItem );
 210                     }
 211 
 212                     break;
 213 
 214                 // Board items

 215                 case PCB_LINE_T:                // a segment not on copper layers

 216                 case PCB_TEXT_T:                // a text on a layer

 217                 case PCB_TRACE_T:               // a track segment (segment on a copper layer)

 218                 case PCB_ARC_T:                 // an arced track segment (segment on a copper layer)

 219                 case PCB_VIA_T:                 // a via (like track segment on a copper layer)

 220                 case PCB_DIMENSION_T:           // a dimension (graphic item)

 221                 case PCB_TARGET_T:              // a target (graphic item)

 222                 case PCB_MARKER_T:              // a marker used to show something

 223                 case PCB_ZONE_AREA_T:
 224                     view->Remove( boardItem );
 225 
 226                     if( !( changeFlags & CHT_DONE ) )
 227                         board->Remove( boardItem );
 228 
 229                     break;
 230 
 231                 case PCB_MODULE_T:
 232                 {
 233                     // There are no modules inside a module yet

 234                     wxASSERT( !m_editModules );
 235 
 236                     MODULE* module = static_cast<MODULE*>( boardItem );
 237                     view->Remove( module );
 238                     module->ClearFlags();
 239 
 240                     if( !( changeFlags & CHT_DONE ) )
 241                         board->Remove( module );        // handles connectivity

 242                 }
 243                 break;
 244 
 245                 // Metadata items

 246                 case PCB_NETINFO_T:
 247                     if( !( changeFlags & CHT_DONE ) )
 248                         board->Remove( boardItem );
 249                     break;
 250 
 251                 default:                        // other types do not need to (or should not) be handled

 252                     wxASSERT( false );
 253                     break;
 254                 }
 255 
 256                 break;
 257             }
 258 
 259             case CHT_MODIFY:
 260             {
 261                 if( !m_editModules && aCreateUndoEntry )
 262                 {
 263                     ITEM_PICKER itemWrapper( nullptr, boardItem, UR_CHANGED );
 264                     wxASSERT( ent.m_copy );
 265                     itemWrapper.SetLink( ent.m_copy );
 266                     undoList.PushItem( itemWrapper );
 267                 }
 268 
 269                 if( ent.m_copy )
 270                     connectivity->MarkItemNetAsDirty( static_cast<BOARD_ITEM*>( ent.m_copy ) );
 271 
 272                 connectivity->Update( boardItem );
 273                 view->Update( boardItem );
 274                 board->OnItemChanged( boardItem );
 275 
 276                 // if no undo entry is needed, the copy would create a memory leak

 277                 if( !aCreateUndoEntry )
 278                     delete ent.m_copy;
 279 
 280                 break;
 281             }
 282 
 283             default:
 284                 wxASSERT( false );
 285                 break;
 286         }
 287     }
 288 
 289     if ( !m_editModules )
 290     {
 291         size_t num_changes = m_changes.size();
 292 
 293         connectivity->RecalculateRatsnest( this );
 294         connectivity->ClearDynamicRatsnest();
 295         frame->GetCanvas()->RedrawRatsnest();
 296 
 297         if( m_changes.size() > num_changes )
 298         {
 299             for( size_t i = num_changes; i < m_changes.size(); ++i )
 300             {
 301                 COMMIT_LINE& ent = m_changes[i];
 302 
 303                 // This should only be modifications from the connectivity algo

 304                 wxASSERT( ( ent.m_type & CHT_TYPE ) == CHT_MODIFY );
 305 
 306                 auto boardItem = static_cast<BOARD_ITEM*>( ent.m_item );
 307 
 308                 if( aCreateUndoEntry )
 309                 {
 310                     ITEM_PICKER itemWrapper( nullptr, boardItem, UR_CHANGED );
 311                     wxASSERT( ent.m_copy );
 312                     itemWrapper.SetLink( ent.m_copy );
 313                     undoList.PushItem( itemWrapper );
 314                 }
 315                 else
 316                 {
 317                     delete ent.m_copy;
 318                 }
 319 
 320                 view->Update( boardItem );
 321             }
 322         }
 323     }
 324 
 325     if( !m_editModules && aCreateUndoEntry )
 326         frame->SaveCopyInUndoList( undoList, UR_UNSPECIFIED );
 327 
 328     m_toolMgr->PostEvent( { TC_MESSAGE, TA_MODEL_CHANGE, AS_GLOBAL } );
 329 
 330     if( itemsDeselected )
 331         m_toolMgr->PostEvent( EVENTS::UnselectedEvent );
 332 
 333     if( aSetDirtyBit )
 334         frame->OnModify();
 335 
 336     frame->UpdateMsgPanel();
 337 
 338     clear();
 339 }

 342 EDA_ITEM* BOARD_COMMIT::parentObject( EDA_ITEM* aItem ) const
 343 {
 344     switch( aItem->Type() )
 345     {
 346         case PCB_PAD_T:
 347         case PCB_MODULE_EDGE_T:
 348         case PCB_MODULE_TEXT_T:
 349         case PCB_MODULE_ZONE_AREA_T:
 350             return aItem->GetParent();
 351 
 352         case PCB_ZONE_AREA_T:
 353             wxASSERT( !dynamic_cast<MODULE*>( aItem->GetParent() ) );
 354             return aItem;
 355 
 356         default:
 357             break;
 358     }
 359 
 360     return aItem;
 361 }

 364 void BOARD_COMMIT::Revert()
 365 {
 366     PICKED_ITEMS_LIST undoList;
 367     KIGFX::VIEW* view = m_toolMgr->GetView();
 368     BOARD* board = (BOARD*) m_toolMgr->GetModel();
 369     auto connectivity = board->GetConnectivity();
 370 
 371     for( auto it = m_changes.rbegin(); it != m_changes.rend(); ++it )
 372     {
 373         COMMIT_LINE& ent = *it;
 374         BOARD_ITEM* item = static_cast<BOARD_ITEM*>( ent.m_item );
 375         BOARD_ITEM* copy = static_cast<BOARD_ITEM*>( ent.m_copy );
 376         int changeType = ent.m_type & CHT_TYPE;
 377         int changeFlags = ent.m_type & CHT_FLAGS;
 378 
 379         switch( changeType )
 380         {
 381         case CHT_ADD:
 382             if( !( changeFlags & CHT_DONE ) )
 383                 break;
 384 
 385             view->Remove( item );
 386             connectivity->Remove( item );
 387             board->Remove( item );
 388             break;
 389 
 390         case CHT_REMOVE:
 391             if( !( changeFlags & CHT_DONE ) )
 392                 break;
 393 
 394             view->Add( item );
 395             connectivity->Add( item );
 396             board->Add( item );
 397             break;
 398 
 399         case CHT_MODIFY:
 400         {
 401             view->Remove( item );
 402             connectivity->Remove( item );
 403 
 404             item->SwapData( copy );
 405 
 406             view->Add( item );
 407             connectivity->Add( item );
 408             board->OnItemChanged( item );
 409             delete copy;
 410             break;
 411         }
 412 
 413         default:
 414             wxASSERT( false );
 415             break;
 416         }
 417     }
 418 
 419     if ( !m_editModules )
 420         connectivity->RecalculateRatsnest();
 421 
 422     SELECTION_TOOL* selTool = m_toolMgr->GetTool<SELECTION_TOOL>();
 423     selTool->RebuildSelection();
 424 
 425     clear();
 426 }
