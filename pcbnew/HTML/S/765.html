  61 void EditToolSelectionFilter( GENERAL_COLLECTOR& aCollector, int aFlags )
  62 {
  63     // Iterate from the back so we don't have to worry about removals.

  64     for( int i = aCollector.GetCount() - 1; i >= 0; --i )
  65     {
  66         BOARD_ITEM* item = aCollector[ i ];
  67 
  68         if( ( aFlags & EXCLUDE_LOCKED ) && item->IsLocked() )
  69         {
  70             aCollector.Remove( item );
  71         }
  72         else if( item->Type() == PCB_MODULE_ZONE_AREA_T )
  73         {
  74             MODULE* mod = static_cast<MODULE*>( item->GetParent() );
  75 
  76             // case 1: handle locking

  77             if( ( aFlags & EXCLUDE_LOCKED ) && mod && mod->IsLocked() )
  78             {
  79                 aCollector.Remove( item );
  80             }
  81 
  82             // case 2: selection contains both the module and its pads - remove the pads

  83             if( !( aFlags & INCLUDE_PADS_AND_MODULES ) && mod && aCollector.HasItem( mod ) )
  84                 aCollector.Remove( item );
  85         }
  86         else if( item->Type() == PCB_PAD_T )
  87         {
  88             MODULE* mod = static_cast<MODULE*>( item->GetParent() );
  89 
  90             // case 1: handle locking

  91             if( ( aFlags & EXCLUDE_LOCKED ) && mod && mod->IsLocked() )
  92             {
  93                 aCollector.Remove( item );
  94             }
  95             else if( ( aFlags & EXCLUDE_LOCKED_PADS ) && mod && mod->PadsLocked() )
  96             {
  97                 // Pad locking is considerably "softer" than item locking

  98                 aCollector.Remove( item );
  99 
 100                 if( !mod->IsLocked() && !aCollector.HasItem( mod ) )
 101                     aCollector.Append( mod );
 102             }
 103 
 104             // case 2: selection contains both the module and its pads - remove the pads

 105             if( !( aFlags & INCLUDE_PADS_AND_MODULES ) && mod && aCollector.HasItem( mod ) )
 106                 aCollector.Remove( item );
 107         }
 108         else if( ( aFlags & EXCLUDE_TRANSIENTS ) && item->Type() == PCB_MARKER_T )
 109         {
 110             aCollector.Remove( item );
 111         }
 112     }
 113 }

 156 bool EDIT_TOOL::Init()
 157 {
 158     // Find the selection tool, so they can cooperate

 159     m_selectionTool = static_cast<SELECTION_TOOL*>( m_toolMgr->FindTool( "pcbnew.InteractiveSelection" ) );
 160     wxASSERT_MSG( m_selectionTool, "pcbnew.InteractiveSelection tool is not available" );
 161 
 162     auto editingModuleCondition =
 163             [ this ] ( const SELECTION& aSelection )
 164             {
 165                 return m_editModules;
 166             };
 167 
 168     auto singleModuleCondition = SELECTION_CONDITIONS::OnlyType( PCB_MODULE_T )
 169                                     && SELECTION_CONDITIONS::Count( 1 );
 170 
 171     auto noActiveToolCondition =
 172             [ this ] ( const SELECTION& aSelection )
 173             {
 174                 return frame()->ToolStackIsEmpty();
 175             };
 176 
 177     auto notMovingCondition =
 178             [ this ] ( const SELECTION& aSelection )
 179             {
 180                 return !frame()->IsCurrentTool( PCB_ACTIONS::move )
 181                        && !frame()->IsCurrentTool( PCB_ACTIONS::moveWithReference );
 182             };
 183 
 184     // Add context menu entries that are displayed when selection tool is active

 185     CONDITIONAL_MENU& menu = m_selectionTool->GetToolMenu().GetMenu();
 186 
 187     menu.AddItem( PCB_ACTIONS::move, SELECTION_CONDITIONS::NotEmpty && notMovingCondition );
 188     menu.AddItem( PCB_ACTIONS::inlineBreakTrack, SELECTION_CONDITIONS::Count( 1 )
 189                       && SELECTION_CONDITIONS::OnlyTypes( GENERAL_COLLECTOR::Tracks ) );
 190     menu.AddItem( PCB_ACTIONS::drag45Degree, SELECTION_CONDITIONS::OnlyTypes( GENERAL_COLLECTOR::Tracks ) );
 191     menu.AddItem( PCB_ACTIONS::dragFreeAngle, SELECTION_CONDITIONS::OnlyTypes( GENERAL_COLLECTOR::Tracks ) );
 192     menu.AddItem( PCB_ACTIONS::rotateCcw, SELECTION_CONDITIONS::NotEmpty );
 193     menu.AddItem( PCB_ACTIONS::rotateCw, SELECTION_CONDITIONS::NotEmpty );
 194     menu.AddItem( PCB_ACTIONS::flip, SELECTION_CONDITIONS::NotEmpty );
 195     menu.AddItem( PCB_ACTIONS::mirror, editingModuleCondition && SELECTION_CONDITIONS::NotEmpty );
 196 
 197     menu.AddItem( ACTIONS::doDelete, SELECTION_CONDITIONS::NotEmpty );
 198     menu.AddItem( PCB_ACTIONS::properties, SELECTION_CONDITIONS::Count( 1 )
 199                       || SELECTION_CONDITIONS::OnlyTypes( GENERAL_COLLECTOR::Tracks ) );
 200 
 201     menu.AddItem( ACTIONS::duplicate, SELECTION_CONDITIONS::NotEmpty );
 202 
 203     // Add the submenu for create array and special move

 204     auto specialToolsSubMenu = std::make_shared<SPECIAL_TOOLS_CONTEXT_MENU>();
 205     specialToolsSubMenu->SetTool( this );
 206     menu.AddSeparator();
 207     m_selectionTool->GetToolMenu().AddSubMenu( specialToolsSubMenu );
 208     menu.AddMenu( specialToolsSubMenu.get(), SELECTION_CONDITIONS::NotEmpty );
 209 
 210     menu.AddSeparator();
 211     menu.AddItem( ACTIONS::cut, SELECTION_CONDITIONS::NotEmpty );
 212     menu.AddItem( ACTIONS::copy, SELECTION_CONDITIONS::NotEmpty );
 213     // Selection tool handles the context menu for some other tools, such as the Picker.

 214     // Don't add things like Paste when another tool is active.

 215     menu.AddItem( ACTIONS::paste, noActiveToolCondition );
 216 
 217     // Footprint actions

 218     menu.AddSeparator();
 219     menu.AddItem( PCB_ACTIONS::editFpInFpEditor, singleModuleCondition );
 220     menu.AddItem( PCB_ACTIONS::updateFootprint, singleModuleCondition );
 221     menu.AddItem( PCB_ACTIONS::changeFootprint, singleModuleCondition );
 222 
 223     return true;
 224 }

 227 int EDIT_TOOL::GetAndPlace( const TOOL_EVENT& aEvent )
 228 {
 229     SELECTION_TOOL* selectionTool = m_toolMgr->GetTool<SELECTION_TOOL>();
 230     MODULE*         module = getEditFrame<PCB_BASE_FRAME>()->GetFootprintFromBoardByReference();
 231 
 232     if( module )
 233     {
 234         m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
 235         m_toolMgr->RunAction( PCB_ACTIONS::selectItem, true, (void*) module );
 236 
 237         selectionTool->GetSelection().SetReferencePoint( module->GetPosition() );
 238         m_toolMgr->RunAction( PCB_ACTIONS::move, false );
 239     }
 240 
 241     return 0;

 245 bool EDIT_TOOL::invokeInlineRouter( int aDragMode )
 246 {
 247     ROUTER_TOOL* theRouter = m_toolMgr->GetTool<ROUTER_TOOL>();
 248 
 249     if( !theRouter )
 250         return false;
 251 
 252   // make sure we don't accidentally invoke inline routing mode while the router is already active!

 253     if( theRouter->IsToolActive() )
 254         return false;
 255 
 256     if( theRouter->CanInlineDrag() )
 257     {
 258         m_toolMgr->RunAction( PCB_ACTIONS::routerInlineDrag, true, aDragMode );
 259         return true;
 260     }
 261 
 262     return false;
 263 }

 299 int EDIT_TOOL::doMoveSelection( const TOOL_EVENT& aEvent, bool aPickReference )
 300 {
 301     PCB_BASE_EDIT_FRAME*  editFrame = getEditFrame<PCB_BASE_EDIT_FRAME>();
 302     KIGFX::VIEW_CONTROLS* controls  = getViewControls();
 303     VECTOR2I originalCursorPos = controls->GetCursorPosition();
 304 
 305     // Be sure that there is at least one item that we can modify. If nothing was selected before,

 306     // try looking for the stuff under mouse cursor (i.e. Kicad old-style hover selection)

 307     PCBNEW_SELECTION& selection = m_selectionTool->RequestSelection(
 308             []( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector )
 309             {
 310                 EditToolSelectionFilter( aCollector, EXCLUDE_TRANSIENTS );
 311             } );
 312 
 313     if( m_dragging || selection.Empty() )
 314         return 0;
 315 
 316     LSET item_layers = selection.GetSelectionLayers();
 317     bool unselect    = selection.IsHover(); // N.B. This must be saved before the re-selection below

 318 
 319     // Now filter out locked pads.  We cannot do this in the first RequestSelection() as we need

 320     // the item_layers when a pad is the selection front (ie: will become curr_tiem).

 321     selection = m_selectionTool->RequestSelection(
 322             []( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector )
 323             {
 324                 EditToolSelectionFilter( aCollector, EXCLUDE_LOCKED_PADS );
 325             } );
 326 
 327     if( selection.Empty() )
 328         return 0;
 329 
 330     std::string tool = aEvent.GetCommandStr().get();
 331     editFrame->PushTool( tool );
 332     Activate();
 333     controls->ShowCursor( true );
 334     controls->SetAutoPan( true );
 335 
 336     std::vector<BOARD_ITEM*> sel_items;
 337 
 338     for( EDA_ITEM* item : selection )
 339     {
 340         BOARD_ITEM* boardItem = dynamic_cast<BOARD_ITEM*>( item );
 341         MODULE*     module    = dynamic_cast<MODULE*>( item );
 342 
 343         if( boardItem )
 344             sel_items.push_back( boardItem );
 345 
 346         if( module )
 347         {
 348             for( D_PAD* pad : module->Pads() )
 349                 sel_items.push_back( pad );
 350         }
 351     }
 352 
 353     bool        restore_state = false;
 354     VECTOR2I    totalMovement;
 355     GRID_HELPER grid( m_toolMgr, editFrame->GetMagneticItemsSettings() );
 356     TOOL_EVENT* evt = const_cast<TOOL_EVENT*>( &aEvent );
 357     VECTOR2I    prevPos;
 358 
 359     // Prime the pump

 360     m_toolMgr->RunAction( ACTIONS::refreshPreview );
 361 
 362     // Main loop: keep receiving events

 363     do
 364     {
 365         editFrame->GetCanvas()->SetCurrentCursor( wxCURSOR_ARROW );
 366         grid.SetSnap( !evt->Modifier( MD_SHIFT ) );
 367         grid.SetUseGrid( !evt->Modifier( MD_ALT ) );
 368         controls->SetSnapping( !evt->Modifier( MD_ALT ) );
 369 
 370         if( evt->IsAction( &PCB_ACTIONS::move ) || evt->IsMotion()
 371                 || evt->IsAction( &PCB_ACTIONS::drag ) || evt->IsDrag( BUT_LEFT )
 372                 || evt->IsAction( &ACTIONS::refreshPreview )
 373                 || evt->IsAction( &PCB_ACTIONS::moveWithReference ) )
 374         {
 375             if( m_dragging && evt->Category() == TC_MOUSE )
 376             {
 377                 VECTOR2I mousePos( controls->GetMousePosition() );
 378 
 379                 m_cursor = grid.BestSnapAnchor( mousePos, item_layers, sel_items );
 380 
 381                 if( controls->GetSettings().m_lastKeyboardCursorPositionValid )
 382                 {
 383                     long action = controls->GetSettings().m_lastKeyboardCursorCommand;
 384 
 385                     // The arrow keys are by definition SINGLE AXIS.  Do not allow the other

 386                     // axis to be snapped to the grid.

 387                     if( action == ACTIONS::CURSOR_LEFT || action == ACTIONS::CURSOR_RIGHT )
 388                         m_cursor.y = prevPos.y;
 389                     else if( action == ACTIONS::CURSOR_UP || action == ACTIONS::CURSOR_DOWN )
 390                         m_cursor.x = prevPos.x;
 391                 }
 392 
 393                 controls->ForceCursorPosition( true, m_cursor );
 394                 selection.SetReferencePoint( m_cursor );
 395 
 396                 VECTOR2I movement( m_cursor - prevPos );
 397                 prevPos = m_cursor;
 398                 totalMovement += movement;
 399 
 400                 // Drag items to the current cursor position

 401                 for( EDA_ITEM* item : sel_items )
 402                 {
 403                     // Don't double move footprint pads, fields, etc.

 404                     if( !item->GetParent() || !item->GetParent()->IsSelected() )
 405                         static_cast<BOARD_ITEM*>( item )->Move( movement );
 406                 }
 407 
 408                 frame()->UpdateMsgPanel();
 409             }
 410             else if( !m_dragging ) // Prepare to start dragging

 411             {
 412                 if( !( evt->IsAction( &PCB_ACTIONS::move )
 413                        || evt->IsAction( &PCB_ACTIONS::moveWithReference ) )
 414                     && isInteractiveDragEnabled() )
 415                 {
 416                     if( invokeInlineRouter( PNS::DM_ANY ) )
 417                         break;
 418                 }
 419 
 420                 // deal with locked items (override lock or abort the operation)

 421                 SELECTION_LOCK_FLAGS lockFlags = m_selectionTool->CheckLock();
 422 
 423                 if( lockFlags == SELECTION_LOCKED )
 424                     break;
 425 
 426                 m_dragging = true;
 427 
 428                 // When editing modules, all items have the same parent

 429                 if( EditingModules() )
 430                 {
 431                     m_commit->Modify( selection.Front() );
 432                 }
 433                 else
 434                 {
 435                     // Save items, so changes can be undone

 436                     for( EDA_ITEM* item : selection )
 437                     {
 438                         // Don't double move footprint pads, fields, etc.

 439                         if( item->GetParent() && item->GetParent()->IsSelected() )
 440                             continue;
 441 
 442                         m_commit->Modify( item );
 443                     }
 444                 }
 445 
 446                 editFrame->UndoRedoBlock( true );
 447                 m_cursor = controls->GetCursorPosition();
 448 
 449                 if( selection.HasReferencePoint() )
 450                 {
 451                     // start moving with the reference point attached to the cursor

 452                     grid.SetAuxAxes( false );
 453 
 454                     auto delta = m_cursor - selection.GetReferencePoint();
 455 
 456                     // Drag items to the current cursor position

 457                     for( EDA_ITEM* item : selection )
 458                     {
 459                         // Don't double move footprint pads, fields, etc.

 460                         if( item->GetParent() && item->GetParent()->IsSelected() )
 461                             continue;
 462 
 463                         static_cast<BOARD_ITEM*>( item )->Move( delta );
 464                     }
 465 
 466                     selection.SetReferencePoint( m_cursor );
 467                 }
 468                 else
 469                 {
 470                     std::vector<BOARD_ITEM*> items;
 471 
 472                     for( EDA_ITEM* item : selection )
 473                         items.push_back( static_cast<BOARD_ITEM*>( item ) );
 474 
 475                     m_cursor = grid.BestDragOrigin( originalCursorPos, items );
 476 
 477                     // Set the current cursor position to the first dragged item origin, so the

 478                     // movement vector could be computed later

 479                     if( aPickReference )
 480                     {
 481                         VECTOR2I ref;
 482 
 483                         if( pickReferencePoint( _( "Select reference point for move..." ),
 484                                                 "", "", ref ) )
 485                         {
 486                             selection.SetReferencePoint( ref );
 487                             controls->ForceCursorPosition( true, ref );
 488                             m_cursor = ref;
 489                         }
 490                         else
 491                         {
 492                             // Cancel before move started

 493                             break;
 494                         }
 495                     }
 496                     else
 497                     {
 498                         selection.SetReferencePoint( m_cursor );
 499                         grid.SetAuxAxes( true, m_cursor );
 500                     }
 501                 }
 502 
 503                 controls->SetCursorPosition( m_cursor, false );
 504 
 505                 prevPos = m_cursor;
 506                 controls->SetAutoPan( true );
 507             }
 508 
 509             m_toolMgr->PostEvent( EVENTS::SelectedItemsModified );
 510             m_toolMgr->RunAction( PCB_ACTIONS::updateLocalRatsnest, false );
 511         }
 512 
 513         else if( evt->IsCancelInteractive() || evt->IsActivate() )
 514         {
 515             restore_state = true; // Canceling the tool means that items have to be restored

 516             break;                // Finish

 517         }
 518 
 519         else if( evt->IsAction( &ACTIONS::undo ) )
 520         {
 521             restore_state = true; // Perform undo locally

 522             break;                // Finish

 523         }
 524 
 525         // Dispatch TOOL_ACTIONs

 526         else if( evt->Category() == TC_COMMAND )
 527         {
 528             if( evt->IsAction( &ACTIONS::doDelete ) )
 529             {
 530                 break; // finish -- there is no further processing for removed items

 531             }
 532             else if( evt->IsAction( &ACTIONS::duplicate ) )
 533             {
 534                 break; // finish -- Duplicate tool will start a new Move with the dup'ed items

 535             }
 536             else if( evt->IsAction( &PCB_ACTIONS::moveExact ) )
 537             {
 538                 // Reset positions so the Move Exactly is from the start.

 539                 for( EDA_ITEM* item : selection )
 540                 {
 541                     BOARD_ITEM* i = static_cast<BOARD_ITEM*>( item );
 542                     i->Move( -totalMovement );
 543                 }
 544 
 545                 break; // finish -- we moved exactly, so we are finished

 546             }
 547         }
 548 
 549         else if( evt->IsMouseUp( BUT_LEFT ) || evt->IsClick( BUT_LEFT ) )
 550         {
 551             break; // finish

 552         }
 553 
 554         else
 555         {
 556             evt->SetPassEvent();
 557         }
 558 
 559     } while( ( evt = Wait() ) ); // Assignment (instead of equality test) is intentional

 560 
 561     m_lockedSelected = false;
 562     controls->ForceCursorPosition( false );
 563     controls->ShowCursor( false );
 564     controls->SetSnapping( false );
 565     controls->SetAutoPan( false );
 566 
 567     m_dragging = false;
 568     editFrame->UndoRedoBlock( false );
 569 
 570     // Discard reference point when selection is "dropped" onto the board (ie: not dragging anymore)

 571     selection.ClearReferencePoint();
 572 
 573     // If canceled, we need to remove the dynamic ratsnest from the screen

 574     if( restore_state )
 575     {
 576         m_toolMgr->RunAction( PCB_ACTIONS::hideDynamicRatsnest, true );
 577         m_commit->Revert();
 578     }
 579     else
 580     {
 581         m_commit->Push( _( "Drag" ) );
 582     }
 583 
 584     if( unselect )
 585         m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
 586 
 587     editFrame->PopTool( tool );
 588 
 589     return 0;
 590 }

 593 int EDIT_TOOL::ChangeTrackWidth( const TOOL_EVENT& aEvent )
 594 {
 595     const auto& selection = m_selectionTool->RequestSelection(
 596             []( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector ) {
 597                 EditToolSelectionFilter( aCollector, EXCLUDE_TRANSIENTS );
 598             } );
 599 
 600     for( EDA_ITEM* item : selection )
 601     {
 602         if( auto via = dyn_cast<VIA*>( item ) )
 603         {
 604             m_commit->Modify( item );
 605 
 606             int new_width;
 607             int new_drill;
 608 
 609             if( via->GetViaType() == VIATYPE::MICROVIA )
 610             {
 611                 auto net = via->GetNet();
 612 
 613                 new_width = net->GetMicroViaSize();
 614                 new_drill = net->GetMicroViaDrillSize();
 615             }
 616             else
 617             {
 618                 new_width = board()->GetDesignSettings().GetCurrentViaSize();
 619                 new_drill = board()->GetDesignSettings().GetCurrentViaDrill();
 620             }
 621 
 622             via->SetDrill( new_drill );
 623             via->SetWidth( new_width );
 624         }
 625         else if ( auto track = dyn_cast<TRACK*>( item ) )
 626         {
 627             m_commit->Modify( item );
 628 
 629             int new_width = board()->GetDesignSettings().GetCurrentTrackWidth();
 630             track->SetWidth( new_width );
 631         }
 632     }
 633 
 634     m_commit->Push( _("Edit track width/via size") );
 635 
 636     if( selection.IsHover() )
 637     {
 638         m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
 639 
 640         // Notify other tools of the changes -- This updates the visual ratsnest

 641         m_toolMgr->ProcessEvent( EVENTS::SelectedItemsModified );
 642     }
 643 
 644     return 0;
 645 }

 648 int EDIT_TOOL::Properties( const TOOL_EVENT& aEvent )
 649 {
 650     PCB_BASE_EDIT_FRAME*    editFrame = getEditFrame<PCB_BASE_EDIT_FRAME>();
 651     const PCBNEW_SELECTION& selection = m_selectionTool->RequestSelection(
 652             []( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector )
 653             {
 654                 EditToolSelectionFilter( aCollector, EXCLUDE_TRANSIENTS );
 655             } );
 656

 662     }
 663     else if( selection.Size() == 1 )

 669         editFrame->OnEditItemRequest( item );

 673     }

 681     }
 682 
 683     if( selection.IsHover() )
 684     {
 685         m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
 686 
 687         // Notify other tools of the changes -- This updates the visual ratsnest

 688         m_toolMgr->ProcessEvent( EVENTS::SelectedItemsModified );
 689     }
 690 
 691     return 0;

 740 /*!

 741  * Mirror a point about the vertical axis passing through another point

 742  */
 743 static wxPoint mirrorPointX( const wxPoint& aPoint, const wxPoint& aMirrorPoint )
 744 {
 745     wxPoint mirrored = aPoint;
 746 
 747     mirrored.x -= aMirrorPoint.x;
 748     mirrored.x = -mirrored.x;
 749     mirrored.x += aMirrorPoint.x;
 750 
 751     return mirrored;
 752 }

 755 /**

 756  * Mirror a pad in the vertical axis passing through a point (mirror left to right)

 757  */
 758 static void mirrorPadX( D_PAD& aPad, const wxPoint& aMirrorPoint )
 759 {
 760     if( aPad.GetShape() == PAD_SHAPE_CUSTOM )
 761         aPad.FlipPrimitives( true );  // mirror primitives left to right

 762 
 763     wxPoint tmpPt = mirrorPointX( aPad.GetPosition(), aMirrorPoint );
 764     aPad.SetPosition( tmpPt );
 765 
 766     aPad.SetX0( aPad.GetPosition().x );
 767 
 768     tmpPt = aPad.GetOffset();
 769     tmpPt.x = -tmpPt.x;
 770     aPad.SetOffset( tmpPt );
 771 
 772     auto tmpz = aPad.GetDelta();
 773     tmpz.x = -tmpz.x;
 774     aPad.SetDelta( tmpz );
 775 
 776     aPad.SetOrientation( -aPad.GetOrientation() );
 777 }

 780 int EDIT_TOOL::Mirror( const TOOL_EVENT& aEvent )
 781 {
 782     auto& selection = m_selectionTool->RequestSelection(
 783             []( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector )
 784             {
 785                 EditToolSelectionFilter( aCollector, EXCLUDE_LOCKED_PADS | EXCLUDE_TRANSIENTS );
 786             },
 787             nullptr, !m_dragging );
 788 
 789     if( selection.Empty() )
 790         return 0;
 791 
 792     updateModificationPoint( selection );
 793     auto refPoint = selection.GetReferencePoint();
 794     wxPoint mirrorPoint( refPoint.x, refPoint.y );
 795 
 796     // When editing modules, all items have the same parent

 797     if( EditingModules() )
 798         m_commit->Modify( selection.Front() );
 799 
 800     for( EDA_ITEM* item : selection )
 801     {
 802         // only modify items we can mirror

 803         switch( item->Type() )
 804         {
 805         case PCB_MODULE_EDGE_T:
 806         case PCB_MODULE_TEXT_T:
 807         case PCB_MODULE_ZONE_AREA_T:
 808         case PCB_PAD_T:
 809             // Only create undo entry for items on the board

 810             if( !item->IsNew() && !EditingModules() )
 811                 m_commit->Modify( item );
 812 
 813             break;
 814         default:
 815             continue;
 816         }
 817 
 818         // modify each object as necessary

 819         switch( item->Type() )
 820         {
 821         case PCB_MODULE_EDGE_T:
 822         {
 823             auto& edge = static_cast<EDGE_MODULE&>( *item );
 824             edge.Mirror( mirrorPoint, false );
 825             break;
 826         }
 827 
 828         case PCB_MODULE_ZONE_AREA_T:
 829         {
 830             auto& zone = static_cast<MODULE_ZONE_CONTAINER&>( *item );
 831             zone.Mirror( mirrorPoint, false );
 832             break;
 833         }
 834 
 835         case PCB_MODULE_TEXT_T:
 836         {
 837             auto& modText = static_cast<TEXTE_MODULE&>( *item );
 838             modText.Mirror( mirrorPoint, false );
 839             break;
 840         }
 841 
 842         case PCB_PAD_T:
 843         {
 844             auto& pad = static_cast<D_PAD&>( *item );
 845             mirrorPadX( pad, mirrorPoint );
 846             break;
 847         }
 848 
 849         default:
 850             // it's likely the commit object is wrong if you get here

 851             assert( false );
 852             break;
 853         }
 854     }
 855 
 856     if( !m_dragging )
 857         m_commit->Push( _( "Mirror" ) );
 858 
 859     if( selection.IsHover() && !m_dragging )
 860         m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
 861 
 862     m_toolMgr->ProcessEvent( EVENTS::SelectedItemsModified );
 863 
 864     if( m_dragging )
 865         m_toolMgr->RunAction( PCB_ACTIONS::updateLocalRatsnest, false );
 866 
 867     return 0;
 868 }

 922 int EDIT_TOOL::Remove( const TOOL_EVENT& aEvent )
 923 {
 924     ROUTER_TOOL* routerTool = m_toolMgr->GetTool<ROUTER_TOOL>();
 925 
 926     // Do not delete items while actively routing.

 927     if( routerTool && routerTool->Router() && routerTool->Router()->RoutingInProgress() )
 928         return 1;
 929 
 930     std::vector<BOARD_ITEM*> lockedItems;
 931     Activate();
 932 
 933     // get a copy instead of reference (as we're going to clear the selection before removing items)

 934     PCBNEW_SELECTION selectionCopy;
 935     bool isCut = aEvent.Parameter<PCB_ACTIONS::REMOVE_FLAGS>() == PCB_ACTIONS::REMOVE_FLAGS::CUT;
 936     bool isAlt = aEvent.Parameter<PCB_ACTIONS::REMOVE_FLAGS>() == PCB_ACTIONS::REMOVE_FLAGS::ALT;
 937 
 938     // If we are in a "Cut" operation, then the copied selection exists already

 939     if( isCut )
 940     {
 941         selectionCopy = m_selectionTool->GetSelection();
 942     }
 943     else
 944     {
 945         selectionCopy = m_selectionTool->RequestSelection(
 946             []( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector )
 947             {
 948                 EditToolSelectionFilter( aCollector, EXCLUDE_LOCKED_PADS | EXCLUDE_TRANSIENTS );
 949             } );
 950     }
 951 
 952     bool isHover = selectionCopy.IsHover();
 953 
 954     // in "alternative" mode, deletion is not just a simple list of selected items,

 955     // it removes whole tracks, not just segments

 956     if( isAlt && isHover
 957             && ( selectionCopy.HasType( PCB_TRACE_T ) || selectionCopy.HasType( PCB_VIA_T ) ) )
 958     {
 959         m_toolMgr->RunAction( PCB_ACTIONS::selectConnection, true );
 960     }
 961 
 962     if( selectionCopy.Empty() )
 963         return 0;
 964 
 965     // N.B. Setting the CUT flag prevents lock filtering as we only want to delete the items that

 966     // were copied to the clipboard, no more, no fewer.  Filtering for locked item, if any will be done

 967     // in the copyToClipboard() routine

 968     if( !m_lockedSelected && !isCut )
 969     {
 970         // Second RequestSelection removes locked items but keeps a copy of their pointers

 971         selectionCopy = m_selectionTool->RequestSelection(
 972                 []( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector )
 973                 {
 974                     EditToolSelectionFilter( aCollector, EXCLUDE_LOCKED );
 975                 },
 976                 &lockedItems );
 977     }
 978 
 979 
 980     // As we are about to remove items, they have to be removed from the selection first

 981     m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
 982 
 983     for( EDA_ITEM* item : selectionCopy )
 984     {
 985         if( m_editModules )
 986         {
 987             m_commit->Remove( item );
 988             continue;
 989         }
 990 
 991         switch( item->Type() )
 992         {
 993         case PCB_MODULE_TEXT_T:
 994             {
 995                 auto text = static_cast<TEXTE_MODULE*>( item );
 996                 auto parent = static_cast<MODULE*>( item->GetParent() );
 997 
 998                 if( text->GetType() == TEXTE_MODULE::TEXT_is_DIVERS )
 999                 {
1000                     m_commit->Modify( text );
1001                     getView()->Remove( text );
1002                     parent->Remove( text );
1003                 }
1004             }
1005             break;
1006 
1007         case PCB_PAD_T:
1008             {
1009                 auto pad = static_cast<D_PAD*>( item );
1010                 auto parent = static_cast<MODULE*>( item->GetParent() );
1011 
1012                 m_commit->Modify( parent );
1013                 getView()->Remove( pad );
1014                 parent->Remove( pad );
1015             }
1016             break;
1017 
1018         case PCB_MODULE_ZONE_AREA_T:
1019             {
1020                 auto zone = static_cast<MODULE_ZONE_CONTAINER*>( item );
1021                 auto parent = static_cast<MODULE*>( item->GetParent() );
1022 
1023                 m_commit->Modify( parent );
1024                 getView()->Remove( zone );
1025                 parent->Remove( zone );
1026             }
1027             break;
1028 
1029         case PCB_ZONE_AREA_T:
1030             // We process the zones special so that cutouts can be deleted when the delete tool

1031             // is called from inside a cutout when the zone is selected.

1032             {
1033                 // Only interact with cutouts when deleting and a single item is selected

1034                 if( !isCut && selectionCopy.GetSize() == 1 )
1035                 {
1036                     VECTOR2I curPos = getViewControls()->GetCursorPosition();
1037                     auto     zone   = static_cast<ZONE_CONTAINER*>( item );
1038 
1039                     int outlineIdx, holeIdx;
1040 
1041                     if( zone->HitTestCutout( curPos, &outlineIdx, &holeIdx ) )
1042                     {
1043                         // Remove the cutout

1044                         m_commit->Modify( zone );
1045                         zone->RemoveCutout( outlineIdx, holeIdx );
1046 
1047                         std::vector<ZONE_CONTAINER*> toFill;
1048                         toFill.emplace_back( zone );
1049 
1050                         // Fill the modified zone

1051                         ZONE_FILLER filler( board() );
1052                         filler.InstallNewProgressReporter( frame(), _( "Fill Zone" ), 4 );
1053                         filler.Fill( toFill );
1054 
1055                         // Update the display

1056                         zone->Hatch();
1057                         canvas()->Refresh();
1058 
1059                         // Restore the selection on the original zone

1060                         m_toolMgr->RunAction( PCB_ACTIONS::selectItem, true, zone );
1061 
1062                         break;
1063                     }
1064                 }
1065 
1066                 // Remove the entire zone otherwise

1067                 m_commit->Remove( item );
1068             }
1069             break;
1070 
1071         default:
1072             m_commit->Remove( item );
1073             break;
1074         }
1075     }
1076 
1077     if( isCut )
1078         m_commit->Push( _( "Cut" ) );
1079     else
1080         m_commit->Push( _( "Delete" ) );
1081 
1082     if( !m_lockedSelected && !lockedItems.empty() )
1083     {
1084         ///> Popup nag for deleting locked items

1085         STATUS_TEXT_POPUP statusPopup( frame() );
1086 
1087         m_lockedSelected = true;
1088         m_toolMgr->RunAction( PCB_ACTIONS::selectItems, true, &lockedItems );
1089         statusPopup.SetText( _( "Delete again to remove locked items" ) );
1090         statusPopup.PopupFor( 2000 );
1091         statusPopup.Move( wxGetMousePosition() + wxPoint( 20, 20 ) );
1092 
1093         Activate();
1094 
1095         while( m_lockedSelected && statusPopup.IsShown() )
1096         {
1097             statusPopup.Move( wxGetMousePosition() + wxPoint( 20, 20 ) );
1098             Wait();
1099         }
1100 
1101         // Ensure statusPopup is hidden after use and before deleting it:

1102         statusPopup.Hide();
1103     }
1104 
1105     m_lockedSelected = false;
1106 
1107     return 0;
1108 }

1190 int EDIT_TOOL::Duplicate( const TOOL_EVENT& aEvent )
1191 {
1192     bool increment = aEvent.IsAction( &PCB_ACTIONS::duplicateIncrement );
1193 
1194     // Be sure that there is at least one item that we can modify

1195     const auto& selection = m_selectionTool->RequestSelection(
1196             []( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector )
1197             {
1198                 EditToolSelectionFilter( aCollector, EXCLUDE_LOCKED_PADS | EXCLUDE_TRANSIENTS );
1199             } );
1200 
1201     if( selection.Empty() )
1202         return 0;
1203 
1204     // we have a selection to work on now, so start the tool process

1205     PCB_BASE_EDIT_FRAME* editFrame = getEditFrame<PCB_BASE_EDIT_FRAME>();
1206 
1207     // If the selection was given a hover, we do not keep the selection after completion

1208     bool is_hover = selection.IsHover();
1209 
1210     std::vector<BOARD_ITEM*> new_items;
1211     new_items.reserve( selection.Size() );
1212 
1213     BOARD_ITEM* orig_item = nullptr;
1214     BOARD_ITEM* dupe_item = nullptr;
1215 
1216     // Each selected item is duplicated and pushed to new_items list

1217     // Old selection is cleared, and new items are then selected.

1218     for( EDA_ITEM* item : selection )
1219     {
1220         orig_item = static_cast<BOARD_ITEM*>( item );
1221 
1222         if( m_editModules )
1223         {
1224             MODULE* editModule = editFrame->GetBoard()->GetFirstModule();
1225             dupe_item = editModule->DuplicateItem( orig_item );
1226 
1227             if( increment && item->Type() == PCB_PAD_T
1228                     && PAD_NAMING::PadCanHaveName( *static_cast<D_PAD*>( dupe_item ) ) )
1229             {
1230                 PAD_TOOL* padTool = m_toolMgr->GetTool<PAD_TOOL>();
1231                 wxString padName = padTool->GetLastPadName();
1232                 padName = editModule->GetNextPadName( padName );
1233                 padTool->SetLastPadName( padName );
1234                 static_cast<D_PAD*>( dupe_item )->SetName( padName );
1235             }
1236         }
1237         else if( orig_item->GetParent() && orig_item->GetParent()->Type() == PCB_MODULE_T )
1238         {
1239             MODULE* parent = static_cast<MODULE*>( orig_item->GetParent() );
1240 
1241             m_commit->Modify( parent );
1242             dupe_item = parent->DuplicateItem( orig_item, true /* add to parent */ );
1243         }
1244         else
1245         {
1246             switch( orig_item->Type() )
1247             {
1248             case PCB_MODULE_T:
1249             case PCB_TEXT_T:
1250             case PCB_LINE_T:
1251             case PCB_TRACE_T:
1252             case PCB_VIA_T:
1253             case PCB_ZONE_AREA_T:
1254             case PCB_TARGET_T:
1255             case PCB_DIMENSION_T:
1256                 dupe_item = orig_item->Duplicate();
1257                 break;
1258 
1259             default:
1260                 // Silently drop other items (such as footprint texts) from duplication

1261                 break;
1262             }
1263         }
1264 
1265         if( dupe_item )
1266         {
1267             // Clear the selection flag here, otherwise the SELECTION_TOOL

1268             // will not properly select it later on

1269             dupe_item->ClearSelected();
1270 
1271             new_items.push_back( dupe_item );
1272             m_commit->Add( dupe_item );
1273         }
1274     }
1275 
1276     // Clear the old selection first

1277     m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
1278 
1279     // Select the new items

1280     m_toolMgr->RunAction( PCB_ACTIONS::selectItems, true, &new_items );
1281 
1282     // record the new items as added

1283     if( !selection.Empty() )
1284     {
1285         editFrame->DisplayToolMsg( wxString::Format( _( "Duplicated %d item(s)" ),
1286                                                      (int) new_items.size() ) );
1287 
1288         // If items were duplicated, pick them up

1289         // this works well for "dropping" copies around and pushes the commit

1290         TOOL_EVENT evt = PCB_ACTIONS::move.MakeEvent();
1291         Move( evt );
1292 
1293         // After moving the new items, we need to refresh the group and view flags

1294         m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
1295 
1296         if( !is_hover )
1297             m_toolMgr->RunAction( PCB_ACTIONS::selectItems, true, &new_items );
1298     }
1299 
1300     return 0;
1301 }

1348 bool EDIT_TOOL::updateModificationPoint( PCBNEW_SELECTION& aSelection )
1349 {
1350     if( m_dragging && aSelection.HasReferencePoint() )
1351         return false;
1352 
1353     // When there is only one item selected, the reference point is its position...

1354     if( aSelection.Size() == 1 )
1355     {
1356         auto item =  static_cast<BOARD_ITEM*>( aSelection.Front() );
1357         auto pos = item->GetPosition();
1358         aSelection.SetReferencePoint( VECTOR2I( pos.x, pos.y ) );
1359     }
1360     // ...otherwise modify items with regard to the grid-snapped cursor position

1361     else
1362     {
1363         m_cursor = getViewControls()->GetCursorPosition( true );
1364         aSelection.SetReferencePoint( m_cursor );
1365     }
1366 
1367     return true;
1368 }

1371 bool EDIT_TOOL::pickReferencePoint( const wxString& aTooltip, const wxString& aSuccessMessage,
1372                                     const wxString& aCanceledMessage, VECTOR2I& aReferencePoint )
1373 {
1374     std::string         tool = "pcbnew.InteractiveEdit.selectReferencePoint";
1375     STATUS_TEXT_POPUP   statusPopup( frame() );
1376     PCBNEW_PICKER_TOOL* picker = m_toolMgr->GetTool<PCBNEW_PICKER_TOOL>();
1377     OPT<VECTOR2I>       pickedPoint;
1378     bool                done = false;
1379 
1380     statusPopup.SetText( aTooltip );
1381 
1382     picker->SetClickHandler(
1383             [&]( const VECTOR2D& aPoint ) -> bool
1384             {
1385                 pickedPoint = aPoint;
1386 
1387                 if( !aSuccessMessage.empty() )
1388                 {
1389                     statusPopup.SetText( aSuccessMessage );
1390                     statusPopup.Expire( 800 );
1391                 }
1392                 else
1393                 {
1394                     statusPopup.Hide();
1395                 }
1396 
1397                 return false; // we don't need any more points

1398             } );
1399 
1400     picker->SetMotionHandler(
1401             [&]( const VECTOR2D& aPos )
1402             {
1403                 statusPopup.Move( wxGetMousePosition() + wxPoint( 20, -50 ) );
1404             } );
1405 
1406     picker->SetCancelHandler(
1407             [&]()
1408             {
1409                 if( !aCanceledMessage.empty() )
1410                 {
1411                     statusPopup.SetText( aCanceledMessage );
1412                     statusPopup.Expire( 800 );
1413                 }
1414                 else
1415                 {
1416                     statusPopup.Hide();
1417                 }
1418             } );
1419 
1420     picker->SetFinalizeHandler(
1421             [&]( const int& aFinalState )
1422             {
1423                 done = true;
1424             } );
1425 
1426     statusPopup.Move( wxGetMousePosition() + wxPoint( 20, -50 ) );
1427     statusPopup.Popup();
1428 
1429     m_toolMgr->RunAction( ACTIONS::pickerTool, true, &tool );
1430 
1431     while( !done )
1432         Wait();
1433 
1434     // Ensure statusPopup is hidden after use and before deleting it:

1435     statusPopup.Hide();
1436 
1437     if( pickedPoint.is_initialized() )
1438         aReferencePoint = pickedPoint.get();
1439 
1440     return pickedPoint.is_initialized();
1441 }

1444 int EDIT_TOOL::copyToClipboard( const TOOL_EVENT& aEvent )
1445 {
1446     CLIPBOARD_IO io;
1447 
1448     Activate();
1449 
1450     PCBNEW_SELECTION& selection = m_selectionTool->RequestSelection(
1451             []( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector ) {
1452                 EditToolSelectionFilter( aCollector, EXCLUDE_LOCKED_PADS | EXCLUDE_TRANSIENTS );
1453             } );
1454 
1455     if( selection.Empty() )
1456         return 1;
1457 
1458     VECTOR2I refPoint;
1459     bool     rv = pickReferencePoint( _( "Select reference point for the copy..." ),
1460             _( "Selection copied." ), _( "Copy cancelled." ), refPoint );
1461     frame()->SetMsgPanel( board() );
1462 
1463     if( !rv )
1464         return 1;
1465 
1466     selection.SetReferencePoint( refPoint );
1467 
1468     io.SetBoard( board() );
1469     io.SaveSelection( selection );
1470 
1471     return 0;
1472 }

1475 int EDIT_TOOL::cutToClipboard( const TOOL_EVENT& aEvent )
1476 {
1477     if( !copyToClipboard( aEvent ) )
1478     {
1479         // N.B. Setting the CUT flag prevents lock filtering as we only want to delete the items that

1480         // were copied to the clipboard, no more, no fewer.  Filtering for locked item, if any will be done

1481         // in the copyToClipboard() routine

1482         TOOL_EVENT evt( aEvent.Category(), aEvent.Action(), TOOL_ACTION_SCOPE::AS_GLOBAL );
1483         evt.SetParameter( PCB_ACTIONS::REMOVE_FLAGS::CUT );
1484         Remove( evt );
1485     }
1486 
1487     return 0;
1488 }
