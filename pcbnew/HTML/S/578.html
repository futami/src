  63 // A helper class to plot board items

  64 class BRDITEMS_PLOTTER : public PCB_PLOT_PARAMS
  65 {
  66     PLOTTER*    m_plotter;
  67     BOARD*      m_board;
  68     LSET        m_layerMask;
  69 
  70 public:
  71     BRDITEMS_PLOTTER( PLOTTER* aPlotter, BOARD* aBoard, const PCB_PLOT_PARAMS& aPlotOpts )
  72             : PCB_PLOT_PARAMS( aPlotOpts )
  73     {
  74         m_plotter = aPlotter;
  75         m_board = aBoard;
  76     }
  77 
  78     /**

  79      * @return a 'width adjustment' for the postscript engine

  80      * (useful for controlling toner bleeding during direct transfer)

  81      * added to track width and via/pads size

  82      */
  83     int getFineWidthAdj()
  84     {
  85         if( GetFormat() == PLOT_FORMAT::POST )
  86             return GetWidthAdjust();
  87         else
  88             return 0;
  89     }
  90 
  91     // Basic functions to plot a board item

  92     void SetLayerSet( LSET aLayerMask )     { m_layerMask = aLayerMask; }
  93     void PlotFootprintGraphicItems( MODULE* aModule );
  94     void PlotFootprintGraphicItem( EDGE_MODULE* aEdge );
  95     void PlotFootprintTextItem( TEXTE_MODULE* aTextMod, COLOR4D aColor );
  96 
  97     /*

  98      * Reference, Value, and other fields are plotted only if the corresponding option is enabled.

  99      * Invisible text fields are plotted only if PlotInvisibleText option is set.

 100      */
 101     void PlotFootprintTextItems( MODULE* aModule );
 102 
 103     void PlotDimension( DIMENSION* Dimension );
 104     void PlotPcbTarget( PCB_TARGET* PtMire );
 105     void PlotFilledAreas( ZONE_CONTAINER* aZone, SHAPE_POLY_SET& aPolysList );
 106     void PlotTextePcb( TEXTE_PCB* pt_texte );
 107     void PlotDrawSegment( DRAWSEGMENT* PtSegm );
 108 
 109     /**

 110      * Plot a pad.

 111      * unlike other items, a pad had not a specific color,

 112      * and be drawn as a non filled item although the plot mode is filled

 113      * color and plot mode are needed by this function

 114      */
 115     void PlotPad( D_PAD* aPad, COLOR4D aColor, EDA_DRAW_MODE_T aPlotMode );
 116 
 117     /**

 118      * plot items like text and graphics,

 119      *  but not tracks and modules

 120      */
 121     void PlotBoardGraphicItems();
 122 
 123     /** Function PlotDrillMarks

 124      * Draw a drill mark for pads and vias.

 125      * Must be called after all drawings, because it

 126      * redraw the drill mark on a pad or via, as a negative (i.e. white) shape in

 127      * FILLED plot mode (for PS and PDF outputs)

 128      */
 129     void PlotDrillMarks();
 130 
 131     /**

 132      * Function getColor

 133      * @return the layer color

 134      * @param aLayer = the layer id

 135      * White color is special: cannot be seen on a white paper

 136      * and in B&W mode, is plotted as white but other colors are plotted in BLACK

 137      * so the returned color is LIGHTGRAY when the layer color is WHITE

 138      */
 139     COLOR4D getColor( LAYER_NUM aLayer );
 140 
 141 private:
 142     /** Helper function to plot a single drill mark. It compensate and clamp

 143      * the drill mark size depending on the current plot options

 144      */
 145     void plotOneDrillMark( PAD_DRILL_SHAPE_T aDrillShape,
 146                            const wxPoint& aDrillPos, wxSize aDrillSize,
 147                            const wxSize& aPadSize,
 148                            double aOrientation, int aSmallDrill );
 149 
 150 };

 169                         const PCB_PLOT_PARAMS& aPlotOpt );

 171 /**

 172  * Function PlotStandardLayer

 173  * plot copper or technical layers.

 174  * not used for silk screen layers, because these layers have specific

 175  * requirements, mainly for pads

 176  * @param aBoard = the board to plot

 177  * @param aPlotter = the plotter to use

 178  * @param aLayerMask = the mask to define the layers to plot

 179  * @param aPlotOpt = the plot options (files, sketch). Has meaning for some formats only

 180  *

 181  * aPlotOpt has 3 important options to control this plot,

 182  * which are set, depending on the layer type to plot

 183  *      SetEnablePlotVia( bool aEnable )

 184  *          aEnable = true to plot vias, false to skip vias (has meaning

 185  *                      only for solder mask layers).

 186  *      SetSkipPlotNPTH_Pads( bool aSkip )

 187  *          aSkip = true to skip NPTH Pads, when the pad size and the pad hole

 188  *                  have the same size. Used in GERBER format only.

 189  *      SetDrillMarksType( DrillMarksType aVal ) controle the actual hole:

 190  *              no hole, small hole, actual hole

 191  */
 192 void PlotStandardLayer( BOARD* aBoard, PLOTTER* aPlotter, LSET aLayerMask,
 193                         const PCB_PLOT_PARAMS& aPlotOpt );
 194
