 110 PCB_DRAW_PANEL_GAL::PCB_DRAW_PANEL_GAL( wxWindow* aParentWindow, wxWindowID aWindowId,
 111                                         const wxPoint& aPosition, const wxSize& aSize,
 112                                         KIGFX::GAL_DISPLAY_OPTIONS& aOptions, GAL_TYPE aGalType ) :
 113         EDA_DRAW_PANEL_GAL( aParentWindow, aWindowId, aPosition, aSize, aOptions, aGalType )
 114 {
 115     m_view = new KIGFX::PCB_VIEW( true );
 116     m_view->SetGAL( m_gal );
 117 
 118     m_painter = std::make_unique<KIGFX::PCB_PAINTER>( m_gal );
 119     m_view->SetPainter( m_painter.get() );
 120 
 121     setDefaultLayerOrder();
 122     setDefaultLayerDeps();
 123 
 124     // View controls is the first in the event handler chain, so the Tool Framework operates

 125     // on updated viewport data.

 126     m_viewControls = new KIGFX::WX_VIEW_CONTROLS( m_view, this );
 127 
 128     // Load display options (such as filled/outline display of items).

 129     // Can be made only if the parent window is an EDA_DRAW_FRAME (or a derived class)

 130     // which is not always the case (namely when it is used from a wxDialog like the pad editor)

 131     PCB_BASE_FRAME* frame = dynamic_cast<PCB_BASE_FRAME*>( GetParentEDAFrame() );
 132 
 133     if( frame )
 134         static_cast<KIGFX::PCB_VIEW*>( m_view )->UpdateDisplayOptions(
 135                 frame->GetDisplayOptions() );
 136 }

 144 void PCB_DRAW_PANEL_GAL::DisplayBoard( BOARD* aBoard )
 145 {
 146 
 147     m_view->Clear();
 148 
 149     auto zones = aBoard->Zones();
 150     std::atomic<size_t> next( 0 );
 151     std::atomic<size_t> count_done( 0 );
 152     size_t parallelThreadCount = std::max<size_t>( std::thread::hardware_concurrency(), 2 );
 153 
 154     for( size_t ii = 0; ii < parallelThreadCount; ++ii )
 155     {
 156         std::thread t = std::thread( [ &count_done, &next, &zones ]( )
 157         {
 158             for( size_t i = next.fetch_add( 1 ); i < zones.size(); i = next.fetch_add( 1 ) )
 159                 zones[i]->CacheTriangulation();
 160 
 161             count_done++;
 162         } );
 163 
 164         t.detach();
 165     }
 166 
 167     if( m_worksheet )
 168         m_worksheet->SetFileName( TO_UTF8( aBoard->GetFileName() ) );
 169 
 170     // Load drawings

 171     for( auto drawing : const_cast<BOARD*>(aBoard)->Drawings() )
 172         m_view->Add( drawing );
 173 
 174     // Load tracks

 175     for( auto track : aBoard->Tracks() )
 176         m_view->Add( track );
 177 
 178     // Load modules and its additional elements

 179     for( auto module : aBoard->Modules() )
 180         m_view->Add( module );
 181 
 182     // DRC markers

 183     for( auto marker : aBoard->Markers() )
 184         m_view->Add( marker );
 185 
 186     // Finalize the triangulation threads

 187     while( count_done < parallelThreadCount )
 188         std::this_thread::sleep_for( std::chrono::milliseconds( 10 ) );
 189 
 190     // Load zones

 191     for( auto zone : aBoard->Zones() )
 192         m_view->Add( zone );
 193 
 194     // Ratsnest

 195     m_ratsnest = std::make_unique<KIGFX::RATSNEST_VIEWITEM>( aBoard->GetConnectivity() );
 196     m_view->Add( m_ratsnest.get() );
 197 }

 207 void PCB_DRAW_PANEL_GAL::UpdateColors()
 208 {
 209     COLOR_SETTINGS* cs = nullptr;
 210 
 211     auto frame = dynamic_cast<PCB_BASE_FRAME*>( GetParentEDAFrame() );
 212 
 213     if( frame )
 214     {
 215         cs = frame->GetColorSettings();
 216     }
 217     else
 218     {
 219         PCBNEW_SETTINGS* app = Pgm().GetSettingsManager().GetAppSettings<PCBNEW_SETTINGS>();
 220 
 221         if( app )
 222             cs = Pgm().GetSettingsManager().GetColorSettings( app->m_ColorTheme );
 223         else
 224             cs = Pgm().GetSettingsManager().GetColorSettings();
 225     }
 226 
 227     wxCHECK_RET( cs, "null COLOR_SETTINGS" );
 228 
 229     Pgm().GetSettingsManager().SaveColorSettings( cs, "board" );
 230 
 231     auto rs = static_cast<KIGFX::PCB_RENDER_SETTINGS*>( m_view->GetPainter()->GetSettings() );
 232     rs->LoadColors( cs );
 233 
 234     m_gal->SetGridColor( cs->GetColor( LAYER_GRID ) );
 235     m_gal->SetCursorColor( cs->GetColor( LAYER_CURSOR ) );
 236 }

 239 void PCB_DRAW_PANEL_GAL::SetHighContrastLayer( PCB_LAYER_ID aLayer )
 240 {
 241     // Set display settings for high contrast mode

 242     KIGFX::RENDER_SETTINGS* rSettings = m_view->GetPainter()->GetSettings();
 243 
 244     SetTopLayer( aLayer );
 245 
 246     rSettings->ClearActiveLayers();
 247     rSettings->SetActiveLayer( aLayer );
 248 
 249     if( IsCopperLayer( aLayer ) )
 250     {
 251         // Bring some other layers to the front in case of copper layers and make them colored

 252         // fixme do not like the idea of storing the list of layers here,

 253         // should be done in some other way I guess..

 254         LAYER_NUM layers[] = {
 255                 GetNetnameLayer( aLayer ),
 256                 LAYER_VIA_THROUGH, LAYER_VIAS_HOLES, LAYER_VIAS_NETNAMES,
 257                 LAYER_PADS_TH, LAYER_PADS_PLATEDHOLES, LAYER_PADS_NETNAMES,
 258                 LAYER_NON_PLATEDHOLES, LAYER_SELECT_OVERLAY, LAYER_GP_OVERLAY,
 259                 LAYER_RATSNEST, LAYER_CURSOR
 260         };
 261 
 262         for( unsigned int i : layers )
 263             rSettings->SetActiveLayer( i );
 264 
 265         // Pads should be shown too

 266         if( aLayer == B_Cu )
 267         {
 268             rSettings->SetActiveLayer( LAYER_PAD_BK );
 269             rSettings->SetActiveLayer( LAYER_MOD_BK );
 270             rSettings->SetActiveLayer( LAYER_PAD_BK_NETNAMES );
 271         }
 272         else if( aLayer == F_Cu )
 273         {
 274             rSettings->SetActiveLayer( LAYER_PAD_FR );
 275             rSettings->SetActiveLayer( LAYER_MOD_FR );
 276             rSettings->SetActiveLayer( LAYER_PAD_FR_NETNAMES );
 277         }
 278     }
 279 
 280     m_view->UpdateAllLayersColor();
 281 }

 284 void PCB_DRAW_PANEL_GAL::SetTopLayer( PCB_LAYER_ID aLayer )
 285 {
 286     m_view->ClearTopLayers();
 287     setDefaultLayerOrder();
 288     m_view->SetTopLayer( aLayer );
 289 
 290     // Layers that should always have on-top attribute enabled

 291     const std::vector<LAYER_NUM> layers = {
 292             LAYER_VIA_THROUGH, LAYER_VIAS_HOLES, LAYER_VIAS_NETNAMES, LAYER_PADS_TH,
 293             LAYER_PADS_PLATEDHOLES, LAYER_PADS_NETNAMES, LAYER_NON_PLATEDHOLES,
 294             LAYER_SELECT_OVERLAY, LAYER_GP_OVERLAY, LAYER_RATSNEST, LAYER_DRC_ERROR,
 295             LAYER_DRC_WARNING, LAYER_DRC_EXCLUSION
 296     };
 297 
 298     for( auto layer : layers )
 299         m_view->SetTopLayer( layer );
 300 
 301     // Extra layers that are brought to the top if a F.* or B.* is selected

 302     const std::vector<LAYER_NUM> frontLayers = {
 303         F_Cu, F_Adhes, F_Paste, F_SilkS, F_Mask, F_Fab, F_CrtYd, LAYER_PAD_FR,
 304         LAYER_PAD_FR_NETNAMES, NETNAMES_LAYER_INDEX( F_Cu )
 305     };
 306 
 307     const std::vector<LAYER_NUM> backLayers = {
 308         B_Cu, B_Adhes, B_Paste, B_SilkS, B_Mask, B_Fab, B_CrtYd, LAYER_PAD_BK,
 309         LAYER_PAD_BK_NETNAMES, NETNAMES_LAYER_INDEX( B_Cu )
 310     };
 311 
 312     const std::vector<LAYER_NUM>* extraLayers = NULL;
 313 
 314     // Bring a few more extra layers to the top depending on the selected board side

 315     if( IsFrontLayer( aLayer ) )
 316         extraLayers = &frontLayers;
 317     else if( IsBackLayer( aLayer ) )
 318         extraLayers = &backLayers;
 319 
 320     if( extraLayers )
 321     {
 322         for( auto layer : *extraLayers )
 323             m_view->SetTopLayer( layer );
 324 
 325         // Move the active layer to the top

 326         if( !IsCopperLayer( aLayer ) )
 327             m_view->SetLayerOrder( aLayer, m_view->GetLayerOrder( GAL_LAYER_ORDER[0] ) );
 328     }
 329     else if( IsCopperLayer( aLayer ) )
 330     {
 331         // Display labels for copper layers on the top

 332         m_view->SetTopLayer( GetNetnameLayer( aLayer ) );
 333     }
 334 
 335     m_view->EnableTopLayer( true );
 336     m_view->UpdateAllLayersOrder();
 337 }

 340 void PCB_DRAW_PANEL_GAL::SyncLayersVisibility( const BOARD* aBoard )
 341 {
 342     // Load layer & elements visibility settings

 343     for( LAYER_NUM i = 0; i < PCB_LAYER_ID_COUNT; ++i )
 344         m_view->SetLayerVisible( i, aBoard->IsLayerVisible( PCB_LAYER_ID( i ) ) );
 345 
 346     for( GAL_LAYER_ID i = GAL_LAYER_ID_START; i < GAL_LAYER_ID_END; ++i )
 347         m_view->SetLayerVisible( i, aBoard->IsElementVisible( i ) );
 348 
 349     // Always enable netname layers, as their visibility is controlled by layer dependencies

 350     for( LAYER_NUM i = NETNAMES_LAYER_ID_START; i < NETNAMES_LAYER_ID_END; ++i )
 351         m_view->SetLayerVisible( i, true );
 352 
 353     // Enable some layers that are GAL specific

 354     m_view->SetLayerVisible( LAYER_PADS_PLATEDHOLES, true );
 355     m_view->SetLayerVisible( LAYER_VIAS_HOLES, true );
 356     m_view->SetLayerVisible( LAYER_GP_OVERLAY, true );
 357     m_view->SetLayerVisible( LAYER_SELECT_OVERLAY, true );
 358     m_view->SetLayerVisible( LAYER_RATSNEST, true );
 359 }

 362 void PCB_DRAW_PANEL_GAL::GetMsgPanelInfo( EDA_DRAW_FRAME* aFrame,
 363                                           std::vector<MSG_PANEL_ITEM>& aList )
 364 {
 365     BOARD* board = static_cast<PCB_BASE_FRAME*>( GetParentEDAFrame() )->GetBoard();
 366     wxString txt;
 367     int viasCount = 0;
 368     int trackSegmentsCount = 0;
 369 
 370     for( auto item : board->Tracks() )
 371     {
 372         if( item->Type() == PCB_VIA_T )
 373             viasCount++;
 374         else
 375             trackSegmentsCount++;
 376     }
 377 
 378     txt.Printf( wxT( "%d" ), board->GetPadCount() );
 379     aList.emplace_back( _( "Pads" ), txt, DARKGREEN );
 380 
 381     txt.Printf( wxT( "%d" ), viasCount );
 382     aList.emplace_back( _( "Vias" ), txt, DARKGREEN );
 383 
 384     txt.Printf( wxT( "%d" ), trackSegmentsCount );
 385     aList.emplace_back( _( "Track Segments" ), txt, DARKGREEN );
 386 
 387     txt.Printf( wxT( "%d" ), board->GetNodesCount() );
 388     aList.emplace_back( _( "Nodes" ), txt, DARKCYAN );
 389 
 390     txt.Printf( wxT( "%d" ), board->GetNetCount() - 1 /* don't include "No Net" in count */ );
 391     aList.emplace_back( _( "Nets" ), txt, RED );
 392 
 393     txt.Printf( wxT( "%d" ), board->GetConnectivity()->GetUnconnectedCount() );
 394     aList.emplace_back( _( "Unrouted" ), txt, BLUE );
 395 }

 398 void PCB_DRAW_PANEL_GAL::OnShow()
 399 {
 400     PCB_BASE_FRAME* frame = dynamic_cast<PCB_BASE_FRAME*>( GetParentEDAFrame() );
 401 
 402     try
 403     {
 404         // Check if the current rendering backend can be properly initialized

 405         m_view->UpdateItems();
 406     }
 407     catch( const std::runtime_error& e )
 408     {
 409         // Fallback to software renderer

 410         DisplayError( GetParent(), e.what() );
 411         SwitchBackend( GAL_TYPE_CAIRO );
 412 
 413         if( frame )
 414             frame->ActivateGalCanvas();
 415     }
 416 
 417     if( frame )
 418     {
 419         SetTopLayer( frame->GetActiveLayer() );
 420         KIGFX::PAINTER* painter = m_view->GetPainter();
 421         auto settings = static_cast<KIGFX::PCB_RENDER_SETTINGS*>( painter->GetSettings() );
 422         settings->LoadDisplayOptions( frame->GetDisplayOptions(), frame->ShowPageLimits() );
 423     }
 424 }

 464 void PCB_DRAW_PANEL_GAL::setDefaultLayerDeps()
 465 {
 466     // caching makes no sense for Cairo and other software renderers

 467     auto target = m_backend == GAL_TYPE_OPENGL ? KIGFX::TARGET_CACHED : KIGFX::TARGET_NONCACHED;
 468 
 469     for( int i = 0; i < KIGFX::VIEW::VIEW_MAX_LAYERS; i++ )
 470         m_view->SetLayerTarget( i, target );
 471 
 472     for( LAYER_NUM i = 0; (unsigned) i < sizeof( GAL_LAYER_ORDER ) / sizeof( LAYER_NUM ); ++i )
 473     {
 474         LAYER_NUM layer = GAL_LAYER_ORDER[i];
 475         wxASSERT( layer < KIGFX::VIEW::VIEW_MAX_LAYERS );
 476 
 477         // Set layer display dependencies & targets

 478         if( IsCopperLayer( layer ) )
 479             m_view->SetRequired( GetNetnameLayer( layer ), layer );
 480         else if( IsNetnameLayer( layer ) )
 481             m_view->SetLayerDisplayOnly( layer );
 482     }
 483 
 484     m_view->SetLayerTarget( LAYER_ANCHOR, KIGFX::TARGET_NONCACHED );
 485     m_view->SetLayerDisplayOnly( LAYER_ANCHOR );
 486 
 487     // Some more required layers settings

 488     m_view->SetRequired( LAYER_VIAS_HOLES, LAYER_VIA_THROUGH );
 489     m_view->SetRequired( LAYER_VIAS_NETNAMES, LAYER_VIA_THROUGH );
 490     m_view->SetRequired( LAYER_PADS_PLATEDHOLES, LAYER_PADS_TH );
 491     m_view->SetRequired( LAYER_NON_PLATEDHOLES, LAYER_PADS_TH );
 492     m_view->SetRequired( LAYER_PADS_NETNAMES, LAYER_PADS_TH );
 493 
 494     // Front modules

 495     m_view->SetRequired( LAYER_PAD_FR, F_Cu );
 496     m_view->SetRequired( LAYER_MOD_TEXT_FR, LAYER_MOD_FR );
 497     m_view->SetRequired( LAYER_PAD_FR_NETNAMES, LAYER_PAD_FR );
 498 
 499     // Back modules

 500     m_view->SetRequired( LAYER_PAD_BK, B_Cu );
 501     m_view->SetRequired( LAYER_MOD_TEXT_BK, LAYER_MOD_BK );
 502     m_view->SetRequired( LAYER_PAD_BK_NETNAMES, LAYER_PAD_BK );
 503 
 504     m_view->SetLayerTarget( LAYER_SELECT_OVERLAY , KIGFX::TARGET_OVERLAY );
 505     m_view->SetLayerDisplayOnly( LAYER_SELECT_OVERLAY ) ;
 506     m_view->SetLayerTarget( LAYER_GP_OVERLAY , KIGFX::TARGET_OVERLAY );
 507     m_view->SetLayerDisplayOnly( LAYER_GP_OVERLAY ) ;
 508     m_view->SetLayerTarget( LAYER_RATSNEST, KIGFX::TARGET_OVERLAY );
 509     m_view->SetLayerDisplayOnly( LAYER_RATSNEST );
 510 
 511     m_view->SetLayerTarget( LAYER_WORKSHEET, KIGFX::TARGET_NONCACHED );
 512     m_view->SetLayerDisplayOnly( LAYER_WORKSHEET ) ;
 513     m_view->SetLayerDisplayOnly( LAYER_GRID );
 514 }
