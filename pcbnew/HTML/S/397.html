  37 /**

  38  * DRC_TEST_PROVIDER

  39  * is a base class that represents a DRC "provider" which runs some DRC functions over a

  40  * #BOARD and spits out #PCB_MARKERs as needed.

  41  */
  42 class DRC_TEST_PROVIDER
  43 {
  44 public:
  45     /**

  46      * A callable that can handle a single generated PCB_MARKER

  47      */
  48     using MARKER_HANDLER = std::function<void( MARKER_PCB* )>;
  49 
  50     /**

  51      * Runs this provider against the given PCB with configured options (if any).

  52      *

  53      * Note: Board is non-const, as some DRC functions modify the board (e.g. zone fill

  54      * or polygon coalescing)

  55      */
  56     virtual bool RunDRC( EDA_UNITS aUnits, BOARD& aBoard ) = 0;
  57 
  58     virtual ~DRC_TEST_PROVIDER() {}
  59 
  60 protected:
  61     DRC_TEST_PROVIDER( MARKER_HANDLER aMarkerHandler ) :
  62             m_marker_handler( std::move( aMarkerHandler ) )
  63     {
  64     }
  65 
  66     /**

  67      * Pass a given marker to the marker handler

  68      */
  69     void HandleMarker( MARKER_PCB* aMarker ) const
  70     {
  71         m_marker_handler( aMarker );
  72     }
  73 
  74 private:
  75     /// The handler for any generated markers

  76     MARKER_HANDLER m_marker_handler;
  77 };

  80 /**

  81  * BOARD_DRC_ITEMS_PROVIDER

  82  * is an implementation of the RC_ITEMS_PROVIDER interface which uses a BOARD instance

  83  * to fulfill the interface.

  84  */
  85 class BOARD_DRC_ITEMS_PROVIDER : public RC_ITEMS_PROVIDER
  86 {
  87 private:
  88     BOARD*                   m_board;
  89 
  90     int                      m_severities;
  91     std::vector<MARKER_PCB*> m_filteredMarkers;
  92 
  93 public:
  94     BOARD_DRC_ITEMS_PROVIDER( BOARD* aBoard ) :
  95             m_board( aBoard ),
  96             m_severities( 0 )
  97     {
  98     }
  99 
 100     void SetSeverities( int aSeverities ) override
 101     {
 102         m_severities = aSeverities;
 103 
 104         BOARD_DESIGN_SETTINGS& bds = m_board->GetDesignSettings();
 105 
 106         m_filteredMarkers.clear();
 107 
 108         for( MARKER_PCB* marker : m_board->Markers() )
 109         {
 110             int markerSeverity;
 111 
 112             if( marker->IsExcluded() )
 113                 markerSeverity = RPT_SEVERITY_EXCLUSION;
 114             else
 115                 markerSeverity = bds.GetSeverity( marker->GetRCItem()->GetErrorCode() );
 116 
 117             if( markerSeverity & m_severities )
 118                 m_filteredMarkers.push_back( marker );
 119         }
 120     }
 121 
 122     int GetCount( int aSeverity = -1 ) override
 123     {
 124         if( aSeverity < 0 )
 125             return m_filteredMarkers.size();
 126 
 127         BOARD_DESIGN_SETTINGS& bds = m_board->GetDesignSettings();
 128 
 129         int count = 0;
 130 
 131         for( MARKER_PCB* marker : m_board->Markers() )
 132         {
 133             int markerSeverity;
 134 
 135             if( marker->IsExcluded() )
 136                 markerSeverity = RPT_SEVERITY_EXCLUSION;
 137             else
 138                 markerSeverity = bds.GetSeverity( marker->GetRCItem()->GetErrorCode() );
 139 
 140             if( markerSeverity == aSeverity )
 141                 count++;
 142         }
 143 
 144         return count;
 145     }
 146 
 147     DRC_ITEM* GetItem( int aIndex ) override
 148     {
 149         MARKER_PCB* marker = m_filteredMarkers[ aIndex ];
 150 
 151         return marker ? static_cast<DRC_ITEM*>( marker->GetRCItem() ) : nullptr;
 152     }
 153 
 154     void DeleteItem( int aIndex, bool aDeep ) override
 155     {
 156         MARKER_PCB* marker = m_filteredMarkers[ aIndex ];
 157         m_filteredMarkers.erase( m_filteredMarkers.begin() + aIndex );
 158 
 159         if( aDeep )
 160             m_board->Delete( marker );
 161     }
 162 
 163     void DeleteAllItems() override
 164     {
 165         m_board->DeleteMARKERs();
 166         m_filteredMarkers.clear();
 167     }
 168 };

 171 /**

 172  * VECTOR_DRC_ITEMS_PROVIDER

 173  * is an implementation of the interface named DRC_ITEMS_PROVIDER which uses a vector

 174  * of pointers to DRC_ITEMs to fulfill the interface.  No ownership is taken of the

 175  * vector.

 176  */
 177 class VECTOR_DRC_ITEMS_PROVIDER : public RC_ITEMS_PROVIDER
 178 {
 179     PCB_BASE_FRAME*         m_frame;
 180     std::vector<DRC_ITEM*>* m_sourceVector;     // owns its DRC_ITEMs

 181 
 182     int                     m_severities;
 183     std::vector<DRC_ITEM*>  m_filteredVector;   // does not own its DRC_ITEMs

 184 
 185 public:
 186 
 187     VECTOR_DRC_ITEMS_PROVIDER( PCB_BASE_FRAME* aFrame, std::vector<DRC_ITEM*>* aList ) :
 188             m_frame( aFrame ),
 189             m_sourceVector( aList ),
 190             m_severities( 0 )
 191     {
 192     }
 193 
 194     void SetSeverities( int aSeverities ) override
 195     {
 196         m_severities = aSeverities;
 197 
 198         BOARD_DESIGN_SETTINGS& bds = m_frame->GetBoard()->GetDesignSettings();
 199 
 200         m_filteredVector.clear();
 201 
 202         if( m_sourceVector )
 203         {
 204             for( DRC_ITEM* item : *m_sourceVector )
 205             {
 206                 if( bds.GetSeverity( item->GetErrorCode() ) & aSeverities )
 207                     m_filteredVector.push_back( item );
 208             }
 209         }
 210     }
 211 
 212     int  GetCount( int aSeverity = -1 ) override
 213     {
 214         if( aSeverity < 0 )
 215             return m_filteredVector.size();
 216 
 217         int count = 0;
 218         BOARD_DESIGN_SETTINGS& bds = m_frame->GetBoard()->GetDesignSettings();
 219 
 220         if( m_sourceVector )
 221         {
 222             for( DRC_ITEM* item : *m_sourceVector )
 223             {
 224                 if( bds.GetSeverity( item->GetErrorCode() ) == aSeverity )
 225                     count++;
 226             }
 227         }
 228 
 229         return count;
 230     }
 231 
 232     DRC_ITEM* GetItem( int aIndex ) override
 233     {
 234         return (m_filteredVector)[aIndex];
 235     }
 236 
 237     void DeleteItem( int aIndex, bool aDeep ) override
 238     {
 239         DRC_ITEM* item = m_filteredVector[aIndex];
 240         m_filteredVector.erase( m_filteredVector.begin() + aIndex );
 241 
 242         if( aDeep )
 243         {
 244             for( size_t i = 0; i < m_sourceVector->size(); ++i )
 245             {
 246                 if( m_sourceVector->at( i ) == item )
 247                 {
 248                     delete item;
 249                     m_sourceVector->erase( m_sourceVector->begin() + i );
 250                     break;
 251                 }
 252             }
 253         }
 254     }
 255 
 256     void DeleteAllItems() override
 257     {
 258         if( m_sourceVector )
 259         {
 260             for( DRC_ITEM* item : *m_sourceVector )
 261                 delete item;
 262 
 263             m_sourceVector->clear();
 264         }
 265 
 266         m_filteredVector.clear();   // no ownership of DRC_ITEM pointers

 267     }
 268 };
