  54 D_PAD::D_PAD( MODULE* parent ) :
  55     BOARD_CONNECTED_ITEM( parent, PCB_PAD_T )
  56 {
  57     m_Size.x = m_Size.y   = Mils2iu( 60 );  // Default pad size 60 mils.

  58     m_Drill.x = m_Drill.y = Mils2iu( 30 );  // Default drill size 30 mils.

  59     m_Orient              = 0;              // Pad rotation in 1/10 degrees.

  60     m_LengthPadToDie      = 0;
  61 
  62     if( m_Parent  &&  m_Parent->Type() == PCB_MODULE_T )
  63     {
  64         m_Pos = GetParent()->GetPosition();
  65     }
  66 
  67     SetShape( PAD_SHAPE_CIRCLE );                   // Default pad shape is PAD_CIRCLE.

  68     SetAnchorPadShape( PAD_SHAPE_CIRCLE );          // Default shape for custom shaped pads

  69                                                     // is PAD_CIRCLE.

  70     SetDrillShape( PAD_DRILL_SHAPE_CIRCLE );        // Default pad drill shape is a circle.

  71     m_Attribute           = PAD_ATTRIB_STANDARD;    // Default pad type is NORMAL (thru hole)

  72     SetProperty( PAD_PROP_NONE );                   // no special fabrication property

  73     m_LocalClearance      = 0;
  74     m_LocalSolderMaskMargin  = 0;
  75     m_LocalSolderPasteMargin = 0;
  76     m_LocalSolderPasteMarginRatio = 0.0;
  77     // Parameters for round rect only:

  78     m_roundedCornerScale = 0.25;                    // from  IPC-7351C standard

  79     // Parameters for chamfered rect only:

  80     m_chamferScale = 0.2;                           // Size of chamfer: ratio of smallest of X,Y size

  81     m_chamferPositions  = RECT_NO_CHAMFER;          // No chamfered corner

  82 
  83     m_ZoneConnection    = ZONE_CONNECTION::INHERITED; // Use parent setting by default

  84     m_ThermalWidth      = 0;                        // Use parent setting by default

  85     m_ThermalGap        = 0;                        // Use parent setting by default

  86 
  87     m_customShapeClearanceArea = CUST_PAD_SHAPE_IN_ZONE_OUTLINE;
  88 
  89     // Set layers mask to default for a standard thru hole pad.

  90     m_layerMask           = StandardMask();
  91 
  92     SetSubRatsnest( 0 );                       // used in ratsnest calculations

  93 
  94     m_shapesDirty = true;
  95     m_effectiveBoundingRadius = 0;
  96 }

 238 void D_PAD::BuildEffectiveShapes() const
 239 {
 240     m_effectiveShapes.clear();
 241     m_effectiveHoleShape = nullptr;
 242 
 243     auto add = [this]( SHAPE* aShape )
 244                {
 245                    m_effectiveShapes.emplace_back( aShape );
 246                };
 247 
 248     wxPoint shapePos = ShapePos();  // Fetch only once; rotation involves trig

 249     PAD_SHAPE_T effectiveShape = GetShape();
 250 
 251     if( GetShape() == PAD_SHAPE_CUSTOM )
 252         effectiveShape = GetAnchorPadShape();
 253 
 254     switch( effectiveShape )
 255     {
 256     case PAD_SHAPE_CIRCLE:
 257         add( new SHAPE_CIRCLE( shapePos, m_Size.x / 2 ) );
 258         break;
 259 
 260     case PAD_SHAPE_OVAL:
 261     {
 262         wxSize  half_size = m_Size / 2;
 263         int     half_width = std::min( half_size.x, half_size.y );
 264         wxPoint half_len( half_size.x - half_width, half_size.y - half_width );
 265 
 266         RotatePoint( &half_len, m_Orient );
 267 
 268         add( new SHAPE_SEGMENT( shapePos - half_len, shapePos + half_len, half_width * 2 ) );
 269     }
 270         break;
 271 
 272     case PAD_SHAPE_RECT:
 273         if( m_Orient == 0 || m_Orient == 1800 )
 274         {
 275             add( new SHAPE_RECT( shapePos - m_Size / 2, m_Size.x, m_Size.y ) );
 276             break;
 277         }
 278         else if( m_Orient == 900 || m_Orient == -900 )
 279         {
 280             wxSize rot_size( m_Size.y, m_Size.x );
 281             add( new SHAPE_RECT( shapePos - rot_size / 2, rot_size.x, rot_size.y ) );
 282             break;
 283         }
 284 
 285         // Not at a cartesian angle; fall through to general case

 286         KI_FALLTHROUGH;
 287 
 288     case PAD_SHAPE_TRAPEZOID:
 289     case PAD_SHAPE_ROUNDRECT:
 290     {
 291         int     r = GetRoundRectCornerRadius();
 292         wxPoint half_size( m_Size.x / 2, m_Size.y / 2 );
 293         wxSize  trap_delta( 0, 0 );
 294 
 295         if( effectiveShape == PAD_SHAPE_ROUNDRECT )
 296             half_size -= wxPoint( r, r );
 297         else if( effectiveShape == PAD_SHAPE_TRAPEZOID )
 298             trap_delta = m_DeltaSize / 2;
 299 
 300         SHAPE_LINE_CHAIN corners;
 301 
 302         corners.Append( -half_size.x + trap_delta.y, -half_size.y - trap_delta.x );
 303         corners.Append(  half_size.x - trap_delta.y, -half_size.y + trap_delta.x );
 304         corners.Append(  half_size.x + trap_delta.y,  half_size.y - trap_delta.x );
 305         corners.Append( -half_size.x - trap_delta.y,  half_size.y + trap_delta.x );
 306 
 307         corners.Rotate( -DECIDEG2RAD( m_Orient ) );
 308         corners.Move( shapePos );
 309 
 310         add( new SHAPE_SIMPLE( corners ) );
 311 
 312         if( effectiveShape == PAD_SHAPE_ROUNDRECT )
 313         {
 314             add( new SHAPE_SEGMENT( corners.CPoint( 0 ), corners.CPoint( 1 ), r * 2 ) );
 315             add( new SHAPE_SEGMENT( corners.CPoint( 1 ), corners.CPoint( 2 ), r * 2 ) );
 316             add( new SHAPE_SEGMENT( corners.CPoint( 2 ), corners.CPoint( 3 ), r * 2 ) );
 317             add( new SHAPE_SEGMENT( corners.CPoint( 3 ), corners.CPoint( 0 ), r * 2 ) );
 318         }
 319     }
 320         break;
 321 
 322     case PAD_SHAPE_CHAMFERED_RECT:
 323     {
 324         SHAPE_POLY_SET outline;
 325         auto board = GetBoard();
 326         int maxError = ARC_HIGH_DEF;
 327 
 328         if( board )
 329             maxError = board->GetDesignSettings().m_MaxError;
 330 
 331         TransformRoundChamferedRectToPolygon( outline, shapePos, GetSize(), m_Orient,
 332                                               GetRoundRectCornerRadius(), GetChamferRectRatio(),
 333                                               GetChamferPositions(), maxError );
 334 
 335         add( new SHAPE_SIMPLE( outline.COutline( 0 ) ) );
 336     }
 337         break;
 338 
 339     default:
 340         wxFAIL_MSG( "D_PAD::buildEffectiveShapes: Unsupported pad shape: "
 341                     + PAD_SHAPE_T_asString( effectiveShape ) );
 342         break;
 343     }
 344 
 345     if( GetShape() == PAD_SHAPE_CUSTOM )
 346     {
 347         for( const std::shared_ptr<DRAWSEGMENT>& primitive : m_editPrimitives )
 348         {
 349             for( SHAPE* shape : primitive->MakeEffectiveShapes() )
 350             {
 351                 shape->Rotate( -DECIDEG2RAD( m_Orient ) );
 352                 shape->Move( shapePos );
 353                 add( shape );
 354             }
 355         }
 356     }
 357 
 358     // Polygon

 359     //

 360     m_effectivePolygon = std::make_shared<SHAPE_POLY_SET>();
 361     TransformShapeWithClearanceToPolygon( *m_effectivePolygon, 0 );
 362 
 363     // Bounding box and radius

 364     //

 365     m_effectiveBoundingRadius = 0;
 366 
 367     for( int cnt = 0; cnt < m_effectivePolygon->OutlineCount(); ++cnt )
 368     {
 369         const SHAPE_LINE_CHAIN& poly = m_effectivePolygon->COutline( cnt );
 370 
 371         for( int ii = 0; ii < poly.PointCount(); ++ii )
 372         {
 373             int dist = KiROUND( ( poly.CPoint( ii ) - m_Pos ).EuclideanNorm() );
 374             m_effectiveBoundingRadius = std::max( m_effectiveBoundingRadius, dist );
 375         }
 376     }
 377 
 378     m_effectiveBoundingRadius += 1;
 379 
 380     // reset the bbox to uninitialized state to prepare for merging

 381     m_effectiveBoundingBox = EDA_RECT();
 382 
 383     for( const std::shared_ptr<SHAPE>& shape : m_effectiveShapes )
 384     {
 385         BOX2I r = shape->BBox();
 386         m_effectiveBoundingBox.Merge( EDA_RECT( (wxPoint) r.GetOrigin(),
 387                                       wxSize( r.GetWidth(), r.GetHeight() ) ) );
 388     }
 389     // Hole shape

 390     //

 391     wxSize  half_size = m_Drill / 2;
 392     int     half_width = std::min( half_size.x, half_size.y );
 393     wxPoint half_len( half_size.x - half_width, half_size.y - half_width );
 394 
 395     RotatePoint( &half_len, m_Orient );
 396 
 397     m_effectiveHoleShape = std::make_shared<SHAPE_SEGMENT>( m_Pos - half_len, m_Pos + half_len,
 398                                                             half_width * 2 );
 399 
 400     // All done

 401     //

 402     m_shapesDirty = false;
 403 }

 415 void D_PAD::SetDrawCoord()
 416 {
 417     MODULE* module = (MODULE*) m_Parent;
 418 
 419     m_Pos = m_Pos0;
 420 
 421     if( module == NULL )
 422         return;
 423 
 424     double angle = module->GetOrientation();
 425 
 426     RotatePoint( &m_Pos.x, &m_Pos.y, angle );
 427     m_Pos += module->GetPosition();
 428 }

 432 {
 433     MODULE* module = (MODULE*) m_Parent;
 434 
 435     if( module == NULL )
 436     {
 437         m_Pos0 = m_Pos;
 438         return;
 439     }
 440 
 441     m_Pos0 = m_Pos - module->GetPosition();
 442     RotatePoint( &m_Pos0.x, &m_Pos0.y, -module->GetOrientation() );
 443 }

 474 void D_PAD::Flip( const wxPoint& aCentre, bool aFlipLeftRight )
 475 {
 476     if( aFlipLeftRight )
 477     {
 478         MIRROR( m_Pos.x, aCentre.x );
 479         MIRROR( m_Pos0.x, 0 );
 480         MIRROR( m_Offset.x, 0 );
 481         MIRROR( m_DeltaSize.x, 0 );
 482     }
 483     else
 484     {
 485         MIRROR( m_Pos.y, aCentre.y );
 486         MIRROR( m_Pos0.y, 0 );
 487         MIRROR( m_Offset.y, 0 );
 488         MIRROR( m_DeltaSize.y, 0 );
 489     }
 490 
 491     SetOrientation( -GetOrientation() );
 492 
 493     // flip pads layers

 494     // PADS items are currently on all copper layers, or

 495     // currently, only on Front or Back layers.

 496     // So the copper layers count is not taken in account

 497     SetLayerSet( FlipLayerMask( m_layerMask ) );
 498 
 499     // Flip the basic shapes, in custom pads

 500     FlipPrimitives( aFlipLeftRight );
 501 
 502     m_shapesDirty = true;
 503 }

 518 {

 556 
 557 int D_PAD::GetSolderMaskMargin() const
 558 {
 559     // The pad inherits the margin only to calculate a default shape,

 560     // therefore only if it is also a copper layer

 561     // Pads defined only on mask layers (and perhaps on other tech layers) use the shape

 562     // defined by the pad settings only

 563     bool isOnCopperLayer = ( m_layerMask & LSET::AllCuMask() ).any();
 564 
 565     if( !isOnCopperLayer )
 566         return 0;
 567 
 568     int     margin = m_LocalSolderMaskMargin;
 569 
 570     MODULE* module = GetParent();
 571 
 572     if( module )
 573     {
 574         if( margin == 0 )
 575         {
 576             if( module->GetLocalSolderMaskMargin() )
 577                 margin = module->GetLocalSolderMaskMargin();
 578         }
 579 
 580         if( margin == 0 )
 581         {
 582             BOARD* brd = GetBoard();
 583 
 584             if( brd )
 585                 margin = brd->GetDesignSettings().m_SolderMaskMargin;
 586         }
 587     }
 588 
 589     // ensure mask have a size always >= 0

 590     if( margin < 0 )
 591     {
 592         int minsize = -std::min( m_Size.x, m_Size.y ) / 2;
 593 
 594         if( margin < minsize )
 595             margin = minsize;
 596     }
 597 
 598     return margin;
 599 }

 602 wxSize D_PAD::GetSolderPasteMargin() const
 603 {
 604     // The pad inherits the margin only to calculate a default shape,

 605     // therefore only if it is also a copper layer.

 606     // Pads defined only on mask layers (and perhaps on other tech layers) use the shape

 607     // defined by the pad settings only

 608     bool isOnCopperLayer = ( m_layerMask & LSET::AllCuMask() ).any();
 609 
 610     if( !isOnCopperLayer )
 611         return wxSize( 0, 0 );
 612 
 613     int     margin = m_LocalSolderPasteMargin;
 614     double  mratio = m_LocalSolderPasteMarginRatio;
 615 
 616     MODULE* module = GetParent();
 617 
 618     if( module )
 619     {
 620         if( margin == 0 )
 621             margin = module->GetLocalSolderPasteMargin();
 622 
 623         auto brd = GetBoard();
 624 
 625         if( margin == 0 && brd )
 626         {
 627                 margin = brd->GetDesignSettings().m_SolderPasteMargin;
 628         }
 629 
 630         if( mratio == 0.0 )
 631             mratio = module->GetLocalSolderPasteMarginRatio();
 632 
 633         if( mratio == 0.0 && brd )
 634         {
 635             mratio = brd->GetDesignSettings().m_SolderPasteMarginRatio;
 636         }
 637     }
 638 
 639     wxSize pad_margin;
 640     pad_margin.x = margin + KiROUND( m_Size.x * mratio );
 641     pad_margin.y = margin + KiROUND( m_Size.y * mratio );
 642 
 643     // ensure mask have a size always >= 0

 644     if( pad_margin.x < -m_Size.x / 2 )
 645         pad_margin.x = -m_Size.x / 2;
 646 
 647     if( pad_margin.y < -m_Size.y / 2 )
 648         pad_margin.y = -m_Size.y / 2;
 649 
 650     return pad_margin;
 651 }

 790 bool D_PAD::HitTest( const wxPoint& aPosition, int aAccuracy ) const
 791 {
 792     VECTOR2I delta = aPosition - GetPosition();
 793     int      boundingRadius = GetBoundingRadius() + aAccuracy;
 794

 801 
 802 bool D_PAD::HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy ) const
 803 {
 804     auto getArea = []( const SHAPE_POLY_SET& aPoly ) -> double
 805                    {
 806                        return aPoly.OutlineCount() ? aPoly.COutline( 0 ).Area() : 0;
 807                    };
 808 
 809     EDA_RECT arect = aRect;
 810     arect.Normalize();
 811     arect.Inflate( aAccuracy );
 812 
 813     EDA_RECT bbox = GetBoundingBox();
 814 
 815     if( !arect.Intersects( bbox ) )
 816         return false;
 817 
 818     // This covers total containment for all test cases

 819     if( arect.Contains( bbox ) )
 820         return true;
 821 
 822     SHAPE_POLY_SET selRect;
 823     selRect.NewOutline();
 824     selRect.Append( arect.GetOrigin() );
 825     selRect.Append( VECTOR2I( arect.GetRight(), arect.GetTop() ) );
 826     selRect.Append( VECTOR2I( arect.GetRight(), arect.GetBottom() ) );
 827     selRect.Append( VECTOR2I( arect.GetLeft(), arect.GetBottom() ) );
 828 
 829     selRect.BooleanIntersection( *GetEffectivePolygon(), SHAPE_POLY_SET::PM_FAST );
 830 
 831     double padArea = getArea( *GetEffectivePolygon() );
 832     double intersection = getArea( selRect );
 833 
 834     if( intersection > ( padArea * 0.99 ) )
 835         return true;
 836     else
 837         return !aContained && intersection > 0;
 838 }

 840 
 841 bool D_PAD::Collide( const D_PAD* aPad, int aMinClearance, int* aActual )
 842 {
 843     int center2center = KiROUND( EuclideanNorm( aPad->ShapePos() - ShapePos() ) );
 844 
 845     // Quick test: Clearance is OK if the bounding circles are further away than aMinClearance

 846     if( center2center - GetBoundingRadius() - aPad->GetBoundingRadius() >= aMinClearance )
 847         return false;
 848 
 849     int actual = INT_MAX;
 850 
 851     for( const std::shared_ptr<SHAPE>& aShape : GetEffectiveShapes() )
 852     {
 853         for( const std::shared_ptr<SHAPE>& bShape : aPad->GetEffectiveShapes() )
 854         {
 855             int this_dist;
 856 
 857             if( aShape->Collide( bShape.get(), aMinClearance, &this_dist ) )
 858                 actual = std::min( actual, this_dist );
 859         }
 860     }
 861 
 862     if( actual < INT_MAX )
 863     {
 864         // returns the actual clearance (clearance < aMinClearance) for diags:

 865         if( aActual )
 866             *aActual = std::max( 0, actual );
 867 
 868         return true;
 869     }
 870 
 871     return false;
 872 }

 874 
 875 bool D_PAD::Collide( const SHAPE_SEGMENT* aSeg, int aMinClearance, int* aActual )
 876 {
 877     int actual = INT_MAX;
 878 
 879     for( const std::shared_ptr<SHAPE>& shape : GetEffectiveShapes() )
 880     {
 881         int this_dist;
 882 
 883         if( shape->Collide( aSeg, aMinClearance, &this_dist ) )
 884             actual = std::min( actual, this_dist );
 885     }
 886 
 887     if( actual < INT_MAX )
 888     {
 889         if( aActual )
 890             *aActual = std::max( 0, actual );
 891 
 892         return true;
 893     }
 894 
 895     return false;
 896 }

 928         return diff;
 929 
 930     if( ( diff = padref->m_DeltaSize.y - padcmp->m_DeltaSize.y ) != 0 )

1033 void D_PAD::ViewGetLayers( int aLayers[], int& aCount ) const
1034 {
1035     aCount = 0;
1036 
1037     // These 2 types of pads contain a hole

1038     if( m_Attribute == PAD_ATTRIB_STANDARD )
1039         aLayers[aCount++] = LAYER_PADS_PLATEDHOLES;
1040 
1041     if( m_Attribute == PAD_ATTRIB_HOLE_NOT_PLATED )
1042         aLayers[aCount++] = LAYER_NON_PLATEDHOLES;
1043 
1044     if( IsOnLayer( F_Cu ) && IsOnLayer( B_Cu ) )
1045     {
1046         // Multi layer pad

1047         aLayers[aCount++] = LAYER_PADS_TH;
1048         aLayers[aCount++] = LAYER_PADS_NETNAMES;
1049     }
1050     else if( IsOnLayer( F_Cu ) )
1051     {
1052         aLayers[aCount++] = LAYER_PAD_FR;
1053 
1054         // Is this a PTH pad that has only front copper?  If so, we need to also display the

1055         // net name on the PTH netname layer so that it isn't blocked by the drill hole.

1056         if( m_Attribute == PAD_ATTRIB_STANDARD )
1057             aLayers[aCount++] = LAYER_PADS_NETNAMES;
1058         else
1059             aLayers[aCount++] = LAYER_PAD_FR_NETNAMES;
1060     }
1061     else if( IsOnLayer( B_Cu ) )
1062     {
1063         aLayers[aCount++] = LAYER_PAD_BK;
1064 
1065         // Is this a PTH pad that has only back copper?  If so, we need to also display the

1066         // net name on the PTH netname layer so that it isn't blocked by the drill hole.

1067         if( m_Attribute == PAD_ATTRIB_STANDARD )
1068             aLayers[aCount++] = LAYER_PADS_NETNAMES;
1069         else
1070             aLayers[aCount++] = LAYER_PAD_BK_NETNAMES;
1071     }
1072     else
1073     {
1074         // Internal layers only.  (Not yet supported in GUI, but is being used by Python

1075         // footprint generators and will be needed anyway once pad stacks are supported.)

1076         for ( int internal = In1_Cu; internal < In30_Cu; ++internal )
1077         {
1078             if( IsOnLayer( (PCB_LAYER_ID) internal ) )
1079                 aLayers[aCount++] = internal;
1080         }
1081     }
1082 
1083     // Check non-copper layers. This list should include all the layers that the

1084     // footprint editor allows a pad to be placed on.

1085     static const PCB_LAYER_ID layers_mech[] = { F_Mask, B_Mask, F_Paste, B_Paste,
1086         F_Adhes, B_Adhes, F_SilkS, B_SilkS, Dwgs_User, Eco1_User, Eco2_User };
1087 
1088     for( PCB_LAYER_ID each_layer : layers_mech )
1089     {
1090         if( IsOnLayer( each_layer ) )
1091             aLayers[aCount++] = each_layer;
1092     }
1093 
1094 #ifdef __WXDEBUG__
1095     if( aCount == 0 )    // Should not occur

1096     {
1097         wxString msg;
1098         msg.Printf( wxT( "footprint %s, pad %s: could not find valid layer for pad" ),
1099                 GetParent() ? GetParent()->GetReference() : "<null>",
1100                 GetName().IsEmpty() ? "(unnamed)" : GetName() );
1101         wxLogWarning( msg );
1102     }
1103 #endif
1104 }

1107 unsigned int D_PAD::ViewGetLOD( int aLayer, KIGFX::VIEW* aView ) const
1108 {
1109     if( aView->GetPrintMode() > 0 )  // In printing mode the pad is always drawable

1110         return 0;
1111 
1112     const int HIDE = std::numeric_limits<unsigned int>::max();
1113     BOARD* board = GetBoard();
1114 
1115     // Handle Render tab switches

1116     if( ( GetAttribute() == PAD_ATTRIB_STANDARD || GetAttribute() == PAD_ATTRIB_HOLE_NOT_PLATED )
1117          && !aView->IsLayerVisible( LAYER_PADS_TH ) )
1118         return HIDE;
1119 
1120     if( !IsFlipped() && !aView->IsLayerVisible( LAYER_MOD_FR ) )
1121         return HIDE;
1122 
1123     if( IsFlipped() && !aView->IsLayerVisible( LAYER_MOD_BK ) )
1124         return HIDE;
1125 
1126     if( IsFrontLayer( ( PCB_LAYER_ID )aLayer ) && !aView->IsLayerVisible( LAYER_PAD_FR ) )
1127         return HIDE;
1128 
1129     if( IsBackLayer( ( PCB_LAYER_ID )aLayer ) && !aView->IsLayerVisible( LAYER_PAD_BK ) )
1130         return HIDE;
1131 
1132     // Only draw the pad if at least one of the layers it crosses is being displayed

1133     if( board && !( board->GetVisibleLayers() & GetLayerSet() ).any() )
1134         return HIDE;
1135 
1136     // Netnames will be shown only if zoom is appropriate

1137     if( IsNetnameLayer( aLayer ) )
1138     {
1139         int divisor = std::min( GetBoundingBox().GetWidth(), GetBoundingBox().GetHeight() );
1140 
1141         // Pad sizes can be zero briefly when someone is typing a number like "0.5"

1142         // in the pad properties dialog

1143         if( divisor == 0 )
1144             return HIDE;
1145 
1146         return ( Millimeter2iu( 5 ) / divisor );
1147     }
1148 
1149     // Other layers are shown without any conditions

1150     return 0;
1151 }
