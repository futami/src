  60 /**

  61  * LAYER_WIDGET

  62  * is abstract and is used to manage a list of layers, with the notion of

  63  * a "current" layer, and layer specific visibility control.  You must derive from

  64  * it to use it so you can implement the abstract functions which recieve the

  65  * events.  Each layer is given its own color, and that color can be changed

  66  * within the UI provided here.  This widget knows nothing of the client code, meaning

  67  * it has no knowledge of a BOARD or anything.  To use it you must derive from

  68  * this class and implement the abstract functions:

  69  * <p> void OnLayerColorChange( int aLayer, int aColor );

  70  * <p> bool OnLayerSelect( int aLayer );

  71  * <p> void OnLayerVisible( int aLayer, bool isVisible );

  72  * <p> void OnRenderColorChange( int id, int aColor );

  73  * <p> void OnRenderEnable( int id, bool isEnabled );

  74  *

  75  * Please note that even if designed toward layers, it is used to

  76  * contain other stuff, too (the second page in pcbnew contains render

  77  * items, for example)

  78  */
  79 class LAYER_WIDGET : public wxPanel

  81 public:
  82     /**

  83      * Struct ROW

  84      * provides all the data needed to add a row to a LAYER_WIDGET.  This is

  85      * part of the public API for a LAYER_WIDGET.

  86      */
  87     struct ROW
  88     {
  89         wxString    rowName;    ///< the prompt or layername

  90         int         id;         ///< either a layer or "visible element" id

  91         COLOR4D     color;      ///< COLOR4D::UNSPECIFIED if none.

  92         bool        state;      ///< initial wxCheckBox state

  93         wxString    tooltip;    ///< if not empty, use this tooltip on row

  94         bool        changeable; ///< if true, the state can be changed

  95         bool        spacer;     ///< if true, this row is a spacer

  96         COLOR4D     defaultColor; ///< The default color for the row

  97 
  98         ROW( const wxString& aRowName, int aId, COLOR4D aColor = COLOR4D::UNSPECIFIED,
  99              const wxString& aTooltip = wxEmptyString, bool aState = true,
 100              bool aChangeable = true, COLOR4D aDefaultColor = COLOR4D::UNSPECIFIED )
 101         {
 102             rowName = aRowName;
 103             id      = aId;
 104             color   = aColor;
 105             state   = aState;
 106             tooltip = aTooltip;
 107             changeable = aChangeable;
 108             spacer = false;
 109             defaultColor = aDefaultColor;
 110         }
 111 
 112         ROW()
 113         {
 114             id = 0;
 115             color = COLOR4D::UNSPECIFIED;
 116             state = true;
 117             changeable = true;
 118             spacer = true;
 119             defaultColor = COLOR4D::UNSPECIFIED;
 120         }
 121     };

 432     //-----<abstract functions>-------------------------------------------

 433 
 434     /**

 435      * Function OnLayerColorChange

 436      * is called to notify client code about a layer color change.  Derived

 437      *es will handle this accordingly.

 438      * @param aLayer is the board layer to change

 439      * @param aColor is the new color

 440      */
 441     virtual void OnLayerColorChange( int aLayer, COLOR4D aColor ) = 0;
 442 
 443     /**

 444      * Function OnLayerSelect

 445      * is called to notify client code whenever the user selects a different

 446      * layer.  Derived classes will handle this accordingly, and can deny

 447      * the change by returning false.

 448      * @param aLayer is the board layer to select

 449      */
 450     virtual bool OnLayerSelect( int aLayer ) = 0;
 451 
 452     /**

 453      * Function OnLayerVisible

 454      * is called to notify client code about a layer visibility change.

 455      *

 456      * @param aLayer is the board layer to select

 457      * @param isVisible is the new visible state

 458      * @param isFinal is true when this is the last of potentially several

 459      *  such calls, and can be used to decide when to update the screen only

 460      *  one time instead of several times in the midst of a multiple layer change.

 461      */
 462     virtual void OnLayerVisible( LAYER_NUM aLayer, bool isVisible, bool isFinal = true ) = 0;
 463 
 464     /**

 465      * Function OnLayerRightClick

 466      * is called to notify client code about a layer being right-clicked.

 470      */

 472 
 473     /**

 479      * @param aColor is the new color

 482
