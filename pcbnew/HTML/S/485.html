  26 #ifndef GRAPHICS_IMPORTER_PCBNEW_H
  27 #define GRAPHICS_IMPORTER_PCBNEW_H
  28 
  29 #include "graphics_importer.h"
  30 
  31 #include <layers_id_colors_and_visibility.h>
  32 
  33 class BOARD_ITEM;
  34 class BOARD;
  35 class MODULE;
  36 class DRAWSEGMENT;
  37 class EDA_TEXT;
  38 
  39 class GRAPHICS_IMPORTER_PCBNEW : public GRAPHICS_IMPORTER
  40 {
  41 public:
  42     GRAPHICS_IMPORTER_PCBNEW();
  43 
  44     /**

  45      * @brief Sets the target layer for the imported shapes.

  46      * @param aLayer is the layer to be used by the imported shapes.

  47      */
  48     void SetLayer( PCB_LAYER_ID aLayer )
  49     {
  50         m_layer = aLayer;
  51     }
  52 
  53     /**

  54      * @brief Returns the target layer for the imported shapes.

  55      */
  56     PCB_LAYER_ID GetLayer() const
  57     {
  58         return m_layer;
  59     }
  60 
  61     void AddLine( const VECTOR2D& aOrigin, const VECTOR2D& aEnd, double aWidth ) override;
  62 
  63     void AddCircle( const VECTOR2D& aOrigin, double aRadius, double aWidth ) override;
  64 
  65     void AddArc( const VECTOR2D& aCenter, const VECTOR2D& aStart, double aAngle,
  66                  double aWidth ) override;
  67 
  68     void AddPolygon( const std::vector< VECTOR2D >& aVertices, double aWidth ) override;
  69 
  70     void AddText( const VECTOR2D& aOrigin, const wxString& aText,
  71             double aHeight, double aWidth, double aThickness, double aOrientation,
  72             EDA_TEXT_HJUSTIFY_T aHJustify, EDA_TEXT_VJUSTIFY_T aVJustify ) override;
  73 
  74     void AddSpline( const VECTOR2D& aStart, const VECTOR2D& aBezierControl1,
  75                     const VECTOR2D& aBezierControl2, const VECTOR2D& aEnd , double aWidth ) override;
  76 
  77     /** convert a imported coordinate to a board coordinate, according to

  78      * the internal units, the user scale and offset

  79      * @param aCoordinate is the imported coordinate in mm

  80      */
  81     wxPoint MapCoordinate( const VECTOR2D& aCoordinate );
  82 
  83     /** @return a line thickness in a board Iu value, according to

  84      * the internal units.

  85      * if aLineWidth < 0, the default ine thickness value is returned

  86      * @param aLineWidth is the line thickness in mm to convert

  87      */
  88     int MapLineWidth( double aLineWidth );
  89 
  90 protected:
  91     ///> Create an object representing a graphical shape.

  92     virtual std::unique_ptr<DRAWSEGMENT> createDrawing() = 0;
  93 
  94     ///> Create an object representing a text. Both pointers point to different parts of the

  95     ///> same object, the EDA_TEXT pointer is simply for convenience.

  96     virtual std::pair<std::unique_ptr<BOARD_ITEM>, EDA_TEXT*> createText() = 0;
  97 
  98     ///> Target layer for the imported shapes.

  99     PCB_LAYER_ID m_layer;
 100 };
