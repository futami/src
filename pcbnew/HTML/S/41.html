  36 // A enum to manage the different layers inside the stackup layers.

  37 // Note the stackup layers include both dielectric and some layers handled by the board editor

  38 // Therefore a stackup layer item is not exactely like a board layer

  39 enum BOARD_STACKUP_ITEM_TYPE
  40 {
  41     BS_ITEM_TYPE_UNDEFINED,     // For not yet initialized BOARD_STACKUP_ITEM item

  42     BS_ITEM_TYPE_COPPER,        // A initialized BOARD_STACKUP_ITEM item for copper layers

  43     BS_ITEM_TYPE_DIELECTRIC,    // A initialized BOARD_STACKUP_ITEM item for the

  44                                 // dielectric between copper layers

  45     BS_ITEM_TYPE_SOLDERPASTE,   // A initialized BOARD_STACKUP_ITEM item for solder paste layers

  46     BS_ITEM_TYPE_SOLDERMASK,    // A initialized BOARD_STACKUP_ITEM item for solder mask layers

  47                                 // note: this is a specialized dielectric material

  48     BS_ITEM_TYPE_SILKSCREEN,    // A initialized BOARD_STACKUP_ITEM item for silkscreen layers

  49 };

  51 // A enum to manage edge connector fab info

  52 enum BS_EDGE_CONNECTOR_CONSTRAINTS
  53 {
  54     BS_EDGE_CONNECTOR_NONE,     // No edge connector in board

  55     BS_EDGE_CONNECTOR_IN_USE,   // some edge connector in board

  56     BS_EDGE_CONNECTOR_BEVELLED  // Some connector in board, and the connector must be bevelled

  60 /**

  61  * A helper class to manage a dielectric layer set of parameters

  62  */
  63 class DIELECTRIC_PRMS
  64 {
  65     friend class BOARD_STACKUP_ITEM;
  66 
  67 private:
  68     wxString m_Material;    /// type of material (for dielectric and solder mask)

  69     int m_Thickness;        /// the physical layer thickness in internal units

  70     bool m_ThicknessLocked; /// true for dielectric layers with a fixed thickness

  71                             /// (for impendace controled purposes), unused for other layers

  72     double m_EpsilonR;      /// For dielectric (and solder mask) the dielectric constant

  73     double m_LossTangent;   /// For dielectric (and solder mask) the dielectric loss

  74 
  75 public:
  76     DIELECTRIC_PRMS() :
  77         m_Thickness(0), m_ThicknessLocked( false ),
  78         m_EpsilonR( 1.0 ), m_LossTangent( 0.0 )
  79     {}
  80 };

  87 class BOARD_STACKUP_ITEM
  88 {
  89 public:

  91     BOARD_STACKUP_ITEM( BOARD_STACKUP_ITEM& aOther );
  92 
  93 
  94 private:
  95     BOARD_STACKUP_ITEM_TYPE m_Type;
  96     wxString m_LayerName;   /// name of layer as shown in layer manager. Usefull to create reports

  97     wxString m_TypeName;    /// type name of layer (copper, silk screen, core, prepreg ...)

  98     wxString m_Color;       /// mainly for silkscreen and solder mask

  99     PCB_LAYER_ID m_LayerId; /// the layer id (F.Cu to B.Cu, F.Silk, B.silk, F.Mask, B.Mask)

 100                             /// and UNDEFINED_LAYER (-1) for dielectic layers that are not

 101                             /// really layers for the board editor

 102     int m_DielectricLayerId;/// the "layer" id for dielectric layers,

 103                             /// from 1 (top) to 31 (bottom)

 104                             /// (only 31 dielectric layers for 32 copper layers)

 105     /// List of dielectric parameters

 106     /// usually only one item, but in complex (microwave) boards, one can have

 107     /// more than one dielectic layer between 2 copper layers, and therfore

 108     /// more than one item in list

 109     std::vector<DIELECTRIC_PRMS> m_DielectricPrmsList;
 110

 201 /**

 202  * this class manage the layers needed to make a physical board

 203  * they are solder mask, silk screen, copper and dielectric

 204  * Some other layers, used in fabrication, are not managed here because they

 205  * are not used to make a physical board itself

 206  * Note also there are a few other parameters realed to the physical stackup,

 207  * like finish type, impedance control and a few others

 208  */
 209 class BOARD_STACKUP
 210 {
 211     // The list of items describing the stackup for fabrication.

 212     // this is not just copper layers, but also mask dielectric layers

 213     std::vector<BOARD_STACKUP_ITEM*> m_list;
 214 
 215 public:
 216     /** The name of external copper finish

 217      */
 218     wxString m_FinishType;
 219 
 220     /** True if some layers have impedance controlled tracks or have specific

 221      * constrains for micro-wave applications

 222      * If the board has dielectric constrains, the .gbrjob will contain

 223      * info about dielectric constrains: loss tangent and Epsilon rel.

 224      * If not, these values will be not specified in job file.

 225      */
 226     bool m_HasDielectricConstrains;
 227 
 228     /** True if some layers (copper and/or dielectric) have specific thickness

 229      */
 230     bool m_HasThicknessConstrains;
 231 
 232     /** If the board has edge connector cards, some constrains can be specifed

 233      * in job file:

 234      *  BS_EDGE_CONNECTOR_NONE = no edge connector

 235      *  BS_EDGE_CONNECTOR_IN_USE = board has edge connectors

 236      *  BS_EDGE_CONNECTOR_BEVELLED = edge connectors are bevelled

 237      */
 238     BS_EDGE_CONNECTOR_CONSTRAINTS m_EdgeConnectorConstraints;
 239 
 240     bool m_CastellatedPads;         ///< True if castellated pads exist

 241     bool m_EdgePlating;             ///< True if the edge board is plated

 242 
 243 public:
 244     BOARD_STACKUP();
 245     BOARD_STACKUP( BOARD_STACKUP& aOther );
 246     BOARD_STACKUP& operator=( const BOARD_STACKUP& aOther );
 247 
 248     ~BOARD_STACKUP() { RemoveAll(); }
 249 
 250     std::vector<BOARD_STACKUP_ITEM*>& GetList() { return m_list; }
 251 
 252     /// @return a reference to the layer aIndex, or nullptr if not exists

 253     BOARD_STACKUP_ITEM* GetStackupLayer( int aIndex );
 254 
 255     /** @return  the board layers full mask allowed in the stackup list

 256      * i.e. the SilkS, Mask, Paste and all copper layers

 257      */
 258     static LSET StackupAllowedBrdLayers()
 259     {
 260         return LSET( 6, F_SilkS, F_Mask, F_Paste, B_SilkS, B_Mask, B_Paste )
 261                | LSET::ExternalCuMask() | LSET::InternalCuMask();
 262     }
 263 
 264 
 265     /// Delete all items in list and clear the list

 266     void RemoveAll();
 267 
 268     /// @return the number of layers in the stackup

 269     int GetCount() { return (int) m_list.size(); }
 270 
 271     /// @return the board thickness ( in UI) from the thickness of BOARD_STACKUP_ITEM list

 272     int BuildBoardTicknessFromStackup() const;
 273 
 274     /// Add a new item in stackup layer

 275     void Add( BOARD_STACKUP_ITEM* aItem ) { m_list.push_back( aItem ); }
 276 
 277     /**

 278      * Synchronize the BOARD_STACKUP_ITEM* list with the board.

 279      * Not enabled layers are removed

 280      * Missing layers are added

 281      * @param aSettings, is the current board setting.

 282      * @return true if changes are made

 283      */
 284     bool SynchronizeWithBoard( BOARD_DESIGN_SETTINGS* aSettings );
 285 
 286     /**

 287      * Creates a default stackup, according to the current BOARD_DESIGN_SETTINGS settings.

 288      * @param aSettings is the current board setting.

 289      * if nullptr, build a full stackup (with 32 copper layers)

 290      * @param aActiveCopperLayersCount is used only if aSettings == nullptr is the number

 291      * of copper layers to use to calculate a default dielectric thickness.

 292      * ((<= 0 to use all copper layers)

 293      */
 294     void BuildDefaultStackupList( BOARD_DESIGN_SETTINGS* aSettings, int aActiveCopperLayersCount = 0 );
 295 
 296     /**

 297      * Writes the stackup info on board file

 298      * @param aFormatter is the OUTPUTFORMATTER used to create the file

 299      * @param aBoard is the board

 300      * @param aNestLevel is the index to nest level to indent the lines in file

 301      */
 302     void FormatBoardStackup( OUTPUTFORMATTER* aFormatter,
 303                              BOARD* aBoard, int aNestLevel ) const;
 304 };
