  44 /**

  45  * DRAWING_TOOL

  46  *

  47  * Tool responsible for drawing graphical elements like lines, arcs, circles, etc.

  48  */
  49 
  50 class DRAWING_TOOL : public PCB_TOOL_BASE
  51 {
  52 public:
  53     DRAWING_TOOL();
  54     ~DRAWING_TOOL();
  55 
  56     /// @copydoc TOOL_INTERACTIVE::Init()

  57     bool Init() override;
  58 
  59     /// @copydoc TOOL_INTERACTIVE::Reset()

  60     void Reset( RESET_REASON aReason ) override;
  61 
  62     ///> The possible drawing modes of DRAWING_TOOL

  63     enum class MODE
  64     {
  65         NONE,
  66         LINE,
  67         RECTANGLE,
  68         CIRCLE,
  69         ARC,
  70         TEXT,
  71         ANCHOR,
  72         DXF,
  73         DIMENSION,
  74         KEEPOUT,
  75         ZONE,
  76         GRAPHIC_POLYGON,
  77         VIA
  78     };
  79 
  80     /**

  81      * Function GetDrawingMode

  82      *

  83      * Returns the current drawing mode of the DRAWING_TOOL, or

  84      * MODE::NONE if not currently in any drawing mode

  85      */
  86     MODE GetDrawingMode() const;
  87 
  88     /**

  89      * Function DrawLine()

  90      * Starts interactively drawing a line. After invoking the function it expects the user

  91      * to click at least two times to determine the origin and the end for a line. If there are

  92      * more clicks, the line is drawn as a continous polyline.

  93      */
  94     int DrawLine( const TOOL_EVENT& aEvent );
  95 
  96     /**

  97      * Function DrawRectangle()

  98      * Starts interactively drawing a rectangle. After invoking the function it expects the user

  99      * to first click on a point that is going to be used as the top-left of the rectangle. The

 100      * second click determines the bottom-right.

 101      */
 102     int DrawRectangle( const TOOL_EVENT& aEvent );
 103 
 104     /**

 105      * Function DrawCircle()

 106      * Starts interactively drawing a circle. After invoking the function it expects the user

 107      * to first click on a point that is going to be used as the center of the circle. The second

 108      * click determines the circle radius.

 109      */
 110     int DrawCircle( const TOOL_EVENT& aEvent );
 111 
 112     /**

 113      * Function DrawArc()

 114      * Starts interactively drawing an arc. After invoking the function it expects the user

 115      * to first click on a point that is going to be used as the center of the arc. The second

 116      * click determines the origin and radius, the third one - the angle.

 117      */
 118     int DrawArc( const TOOL_EVENT& aEvent );
 119 
 120     /**

 121      * Function PlaceText()

 122      * Displays a dialog that allows one to input text and its settings and then

 123      * lets the user decide where to place the text in editor.

 124      */
 125     int PlaceText( const TOOL_EVENT& aEvent );
 126 
 127     /**

 128      * Function DrawDimension()

 129      * Starts interactively drawing a dimension. After invoking the function it expects the user

 130      * to first click on a point that is going to be used as the origin of the dimension.

 131      * The second click determines the end and the third click modifies its height.

 132      */
 133     int DrawDimension( const TOOL_EVENT& aEvent );
 134 
 135     /**

 136      * Function DrawZone()

 137      * Starts interactively drawing a zone. After invoking the function a zone settings dialog

 138      * is displayed. After confirmation it allows the user to set points that are going to be used

 139      * as a boundary polygon of the zone. Double click or clicking on the origin of the boundary

 140      * polyline finishes the drawing.

 141      *

 142      * The event parameter indicates which type of zone to draw:

 143      *  ADD      add a new zone/keepout with fresh settings

 144      *  CUTOUT   add a cutout to an existing zone

 145      *  SIMILAR  add a new zone with the same settings as an existing one

 146      */
 147     int DrawZone(  const TOOL_EVENT& aEvent );
 148 
 149     int DrawVia( const TOOL_EVENT& aEvent );
 150 
 151     /**

 152      * Function PlaceImportedGraphics()

 153      * Places a drawing imported from a DXF or SVG file in module editor.

 154      */
 155     int PlaceImportedGraphics( const TOOL_EVENT& aEvent );
 156 
 157     /**

 158      * Function SetAnchor()

 159      * Places the footprint anchor (only in module editor).

 160      */
 161     int SetAnchor( const TOOL_EVENT& aEvent );
 162 
 163     ///> Sets up handlers for various events.

 164     void setTransitions() override;
 165 
 166 private:
 167 
 168     /**

 169      * Starts drawing a selected shape (i.e. DRAWSEGMENT).

 170      * @param aShape is the type of created shape (@see STROKE_T).

 171      * @param aGraphic is an object that is going to be used by the tool for drawing. Must be

 172      *                 already created. The tool deletes the object if it is not added to a BOARD.

 173      * @param aStartingPoint is a starting point for this new DRAWSEGMENT. If it exists the new

 174      *                       item has its start point set to aStartingPoint, and its settings

 175      *                       (width, layer) set to the current default values.

 176      * @return False if the tool was cancelled before the origin was set or origin and end are

 177      *         the same point.

 178      */
 179     bool drawSegment( const std::string& aTool, int aShape, DRAWSEGMENT** aGraphic,
 180                       OPT<VECTOR2D> aStartingPoint );
 181 
 182     /**

 183      * Starts drawing an arc.

 184      * @param aGraphic is an object that is going to be used by the tool for drawing. Must be

 185      *                 already created. The tool deletes the object if it is not added to a BOARD.

 186      * @return False if the tool was cancelled before the origin was set or origin and end are

 187      *         the same point.

 188      */
 189     bool drawArc( const std::string& aTool, DRAWSEGMENT** aGraphic, bool aImmediateMode );
 190 
 191     /**

 192      * Draws a polygon, that is added as a zone or a keepout area.

 193      *

 194      * @param aKeepout dictates if the drawn polygon is a zone or a

 195      * keepout area.

 196      * @param aMode dictates the mode of the zone tool:

 197      *  ADD      add a new zone/keepout with fresh settings

 198      *  CUTOUT   add a cutout to an existing zone

 199      *  SIMILAR  add a new zone with the same settings as an existing one

 200      */
 201 
 202     /**

 203      * Function getSourceZoneForAction()

 204      *

 205      * Gets a source zone item for an action that takes an existing zone

 206      * into account (for example a cutout of an existing zone). The source

 207      * zone is taken from the current selection

 208      *

 209      * @param aMode mode of the zone tool

 210      * @param aZone updated pointer to a suitable source zone,

 211      * or nullptr if none found, or the action doesn't need a source

 212      * @return true if a suitable zone was found, or the action doesn't

 213      * need a zone. False if the action needs a zone but none was found.

 214      */
 215     bool getSourceZoneForAction( ZONE_MODE aMode, ZONE_CONTAINER** aZone );
 216 
 217     /**

 218      * Function constrainDimension()

 219      * Forces the dimension lime to be drawn on multiple of 45 degrees

 220      * @param aDimension is the dimension element currently being drawn

 221      */
 222     void constrainDimension( DIMENSION* aDim );
 223 
 224     ///> Returns the appropriate width for a segment depending on the settings.

 225     int getSegmentWidth( PCB_LAYER_ID aLayer ) const;
 226 
 227     KIGFX::VIEW* m_view;
 228     KIGFX::VIEW_CONTROLS* m_controls;
 229     BOARD* m_board;
 230     PCB_BASE_EDIT_FRAME* m_frame;
 231     MODE m_mode;
 232 
 233     unsigned int m_lineWidth;               // Current line width for multi-segment drawing

 234     static const unsigned int WIDTH_STEP;   // Amount of width change with one -/+ key press

 235 
 236     // give internal access to drawing helper classes

 237     friend class ZONE_CREATE_HELPER;
 238 };

 240 #endif /* __DRAWING_TOOL_H */
