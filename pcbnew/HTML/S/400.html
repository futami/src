  69 void DRC_RULES_PARSER::Parse( std::vector<DRC_RULE*>& aRules )
  70 {
  71     bool haveVersion = false;
  72 
  73     for( T token = NextTok();  token != T_EOF;  token = NextTok() )
  74     {
  75         if( token != T_LEFT )
  76             Expecting( T_LEFT );
  77 
  78         token = NextTok();
  79 
  80         if( !haveVersion && token != T_version )
  81             Expecting( "version" );
  82 
  83         switch( token )
  84         {
  85         case T_version:
  86             NeedNUMBER( "version" );
  87             m_requiredVersion = (int)strtol( CurText(), NULL, 10 );
  88             m_tooRecent = ( m_requiredVersion > DRC_RULE_FILE_VERSION );
  89             haveVersion = true;
  90             NeedRIGHT();
  91             break;
  92 
  93         case T_rule:
  94             aRules.push_back( parseDRC_RULE() );
  95             break;
  96 
  97         default:
  98             Expecting( "selector or rule" );
  99         }
 100     }
 101 }

 104 DRC_RULE* DRC_RULES_PARSER::parseDRC_RULE()
 105 {
 106     DRC_RULE* rule = new DRC_RULE();
 107     T         token = NextTok();
 108 
 109     if( !IsSymbol( token ) )
 110         Expecting( "rule name" );
 111 
 112     rule->m_Name = FromUTF8();
 113 
 114     for( token = NextTok();  token != T_RIGHT;  token = NextTok() )
 115     {
 116         if( token != T_LEFT )
 117             Expecting( T_LEFT );
 118 
 119         token = NextTok();
 120 
 121         switch( token )
 122         {
 123         case T_disallow:
 124             switch( NextTok() )
 125             {
 126             case T_track:      rule->m_DisallowFlags |= DISALLOW_TRACKS;     break;
 127             case T_via:        rule->m_DisallowFlags |= DISALLOW_VIAS;       break;
 128             case T_micro_via:  rule->m_DisallowFlags |= DISALLOW_MICRO_VIAS; break;
 129             case T_buried_via: rule->m_DisallowFlags |= DISALLOW_BB_VIAS;    break;
 130             case T_pad:        rule->m_DisallowFlags |= DISALLOW_PADS;       break;
 131             case T_zone:       rule->m_DisallowFlags |= DISALLOW_ZONES;      break;
 132             case T_text:       rule->m_DisallowFlags |= DISALLOW_TEXTS;      break;
 133             case T_graphic:    rule->m_DisallowFlags |= DISALLOW_GRAPHICS;   break;
 134             case T_hole:       rule->m_DisallowFlags |= DISALLOW_HOLES;      break;
 135             case T_footprint:  rule->m_DisallowFlags |= DISALLOW_FOOTPRINTS; break;
 136             default: Expecting( "track, via, micro_via, blind_via, pad, zone, text, "
 137                                 "graphic, or hole" );
 138             }
 139 
 140             rule->m_ConstraintFlags = DISALLOW_CONSTRAINT;
 141             NeedRIGHT();
 142             break;
 143 
 144         case T_constraint:
 145             parseConstraint( rule );
 146             break;
 147 
 148         case T_condition:
 149             NeedSYMBOL();
 150             rule->m_Condition.m_Expression = FromUTF8();
 151 
 152             if( !rule->m_Condition.Compile() )
 153             {
 154                 LIBEVAL::ERROR_STATUS error = rule->m_Condition.GetCompilationError();
 155                 THROW_PARSE_ERROR( error.message, CurSource(), CurLine(), CurLineNumber(),
 156                                    CurOffset() + error.srcPos );
 157             }
 158 
 159             NeedRIGHT();
 160             break;
 161 
 162         default:
 163             Expecting( "disallow, constraint or condition" );
 164         }
 165     }
 166 
 167     return rule;
 168 }

 171 void DRC_RULES_PARSER::parseConstraint( DRC_RULE* aRule )
 172 {
 173     T   token;
 174     int constraintType;
 175     int value;
 176 
 177     switch( NextTok() )
 178     {
 179     case T_clearance:     constraintType = CLEARANCE_CONSTRAINT; break;
 180     case T_track_width:   constraintType = TRACK_CONSTRAINT;     break;
 181     case T_annulus_width: constraintType = ANNULUS_CONSTRAINT;   break;
 182     case T_hole:          constraintType = HOLE_CONSTRAINT;      break;
 183     default: Expecting( "clearance, track_width, annulus_width, or hole" ); return;
 184     }
 185 
 186     aRule->m_ConstraintFlags |= constraintType;
 187 
 188     for( token = NextTok();  token != T_RIGHT;  token = NextTok() )
 189     {
 190         if( token != T_LEFT )
 191             Expecting( T_LEFT );
 192 
 193         token = NextTok();
 194 
 195         switch( token )
 196         {
 197         case T_min:
 198             NextTok();
 199             parseValueWithUnits( FromUTF8(), value );
 200 
 201             switch( constraintType )
 202             {
 203             case CLEARANCE_CONSTRAINT: aRule->m_Clearance.Min = value;       break;
 204             case TRACK_CONSTRAINT:     aRule->m_TrackConstraint.Min = value; break;
 205             case ANNULUS_CONSTRAINT:   aRule->m_MinAnnulusWidth = value;     break;
 206             case HOLE_CONSTRAINT:      aRule->m_MinHole = value;             break;
 207             }
 208 
 209             NeedRIGHT();
 210             break;
 211 
 212         case T_max:
 213             NextTok();
 214             parseValueWithUnits( FromUTF8(), value );
 215 
 216             switch( constraintType )
 217             {
 218             case CLEARANCE_CONSTRAINT: aRule->m_Clearance.Max = value;       break;
 219             case TRACK_CONSTRAINT:     aRule->m_TrackConstraint.Max = value; break;
 220             default: Expecting( "min" );
 221             }
 222 
 223             NeedRIGHT();
 224             break;
 225 
 226         case T_opt:
 227             NextTok();
 228             parseValueWithUnits( FromUTF8(), value );
 229 
 230             switch( constraintType )
 231             {
 232             case CLEARANCE_CONSTRAINT: aRule->m_Clearance.Opt = value;       break;
 233             case TRACK_CONSTRAINT:     aRule->m_TrackConstraint.Opt = value; break;
 234             default: Expecting( "min" );
 235             }
 236 
 237             NeedRIGHT();
 238             break;
 239 
 240         default:
 241             Expecting( "allow or constraint" );
 242         }
 243     }
 244 }

 247 void DRC_RULES_PARSER::parseValueWithUnits( const wxString& aExpr, int& aResult )
 248 {
 249     PCB_EXPR_EVALUATOR evaluator;
 250 
 251     bool ok = evaluator.Evaluate( aExpr );
 252 
 253     if( !ok )
 254     {
 255         LIBEVAL::ERROR_STATUS error = evaluator.GetErrorStatus();
 256         THROW_PARSE_ERROR( error.message, CurSource(), CurLine(), CurLineNumber(),
 257                            CurOffset() + error.srcPos );
 258     }
 259 
 260     aResult = evaluator.Result();
 261 };
