  31 #ifndef PCB_LAYER_WIDGET_H
  32 #define PCB_LAYER_WIDGET_H
  33 
  34 #include <layer_widget.h>
  35 
  36 /**

  37  * PCB_LAYER_WIDGET

  38  * is here to implement the abstract functions of LAYER_WIDGET so they

  39  * may be tied into the PCB_EDIT_FRAME's data and so we can add a popup

  40  * menu which is specific to PCBNEW's needs.

  41  */
  42 class PCB_LAYER_WIDGET : public LAYER_WIDGET
  43 {
  44 public:
  45 
  46     /**

  47      * Constructor

  48      * @param aParent is the parent window

  49      * @param aFocusOwner is the window that should be sent the focus after

  50      * @param aFpEditorMode false for the board editor (default), true for fp editor

  51      *  when true, some options or layers which cannot be used in editor mode are not

  52      * displayed

  53      */
  54     PCB_LAYER_WIDGET( PCB_BASE_FRAME* aParent, wxWindow* aFocusOwner, bool aFpEditorMode = false );
  55 
  56     void ReFill();
  57 
  58     /**

  59      * Function ReFillRender

  60      * rebuilds Render for instance after the config is read

  61      */
  62     void ReFillRender();
  63 
  64     /**

  65      * Function SyncLayerVisibilities

  66      * updates each "Layer" checkbox in this layer widget according

  67      * to each layer's current visibility determined by IsLayerVisible(), and is

  68      * helpful immediately after loading a BOARD which may have state information in it.

  69      */
  70     void SyncLayerVisibilities();
  71 
  72     /**

  73      * Function SyncLayerAlphaIndicators

  74      * updates each "Layer"s alpha indicator to show if the board is currently being

  75      * rendered with more transparency or less.

  76      */
  77     void SyncLayerAlphaIndicators();
  78 
  79     /**

  80      * Updates the color for each layer and item from the active color theme

  81      */
  82     void SyncLayerColors();
  83 
  84     /**

  85      * Function SetLayersManagerTabsText

  86      * Update the layer manager tabs labels

  87      * Useful when changing Language or to set labels to a non default value

  88      */
  89     void SetLayersManagerTabsText();
  90 
  91     //-----<implement LAYER_WIDGET abstract callback functions>-----------

  92     void OnLayerColorChange( int aLayer, COLOR4D aColor ) override;
  93     bool OnLayerSelect( int aLayer ) override;
  94     void OnLayerVisible( int aLayer, bool isVisible, bool isFinal ) override;
  95     void OnLayerRightClick( wxMenu& aMenu ) override;
  96     void OnRenderColorChange( int aId, COLOR4D aColor ) override;
  97     void OnRenderEnable( int aId, bool isEnabled ) override;
  98     //-----</implement LAYER_WIDGET abstract callback functions>----------

  99 
 100     /**

 101      * Function OnLayerSelected

 102      * ensure the active layer is visible, and other layers not visible

 103      * when m_alwaysShowActiveLayer is true

 104      * Otherwise do nothing.

 105      * @return true m_alwaysShowActiveLayer is true and the canvas is refreshed,

 106      * and false if do nothing

 107      */
 108     bool OnLayerSelected();     // postprocess after an active layer selection

 109                                 // ensure active layer visible if

 110                                 // m_alwaysShowActiveCopperLayer is true;

 111 
 112     /**

 113      * Function addRightClickMenuItems

 114      * add menu items to a menu that should be shown when right-clicking

 115      * the PCB layer widget.

 116      */
 117     void AddRightClickMenuItems( wxMenu& menu );
 118 
 119 
 120 protected:
 121 
 122     static const LAYER_WIDGET::ROW  s_render_rows[];
 123     bool m_alwaysShowActiveCopperLayer;         // If true: Only shows the current active layer

 124                                                 // even if it is changed

 125     bool m_fp_editor_mode;
 126 
 127     PCB_BASE_FRAME* myframe;
 128 
 129     // popup menu ids.

 130     enum POPUP_ID
 131     {
 132         ID_SHOW_ALL_COPPER_LAYERS = LAYER_WIDGET::ID_LAST_VALUE,
 133         ID_HIDE_ALL_COPPER_LAYERS,
 134         ID_HIDE_ALL_COPPER_LAYERS_BUT_ACTIVE,
 135         ID_ALWAYS_HIDE_ALL_COPPER_LAYERS_BUT_ACTIVE,
 136         ID_SHOW_NO_LAYERS,
 137         ID_SHOW_ALL_LAYERS,
 138         ID_SHOW_ONLY_FRONT,
 139         ID_SHOW_ONLY_INNER,
 140         ID_SHOW_ONLY_BACK,
 141         ID_SHOW_ONLY_FRONT_ASSEMBLY,
 142         ID_SHOW_ONLY_BACK_ASSEMBLY,
 143         ID_HIDE_ALL_NON_COPPER,
 144         ID_SHOW_ALL_NON_COPPER,
 145         ID_LAST_VALUE
 146     };
 147 
 148     virtual COLOR4D getBackgroundLayerColor() override;
 149 
 150     /**

 151      * Function isAllowedInFpMode

 152      * @return true if item aId has meaning in footprint editor mode.

 153      * and therefore is shown in render panel

 154      */
 155     bool isAllowedInFpMode( int aId );
 156 
 157     /**

 158      * Function isLayerAllowedInFpMode

 159      *

 160      * User layers, which are not paired, are not shown in layers manager.  However a not

 161      * listed layer can be reachable in the graphic item properties dialog.

 162      *

 163      * @param aLayer is the layer id to test

 164      * @return true if PCB_LAYER_ID aLayer has meaning in footprint editor mode.

 165      * and therefore is shown in render panel

 166      */
 167     bool isLayerAllowedInFpMode( PCB_LAYER_ID aLayer );
 168 
 169     /**

 170      * Function OnRightDownLayers

 171      * puts up a popup menu for the layer panel.

 172      */
 173     void onRightDownLayers( wxMouseEvent& event );
 174 
 175     void onPopupSelection( wxCommandEvent& event );
 176 
 177     /// this is for the popup menu, the right click handler has to be installed

 178     /// on every child control within the layer panel.

 179     void installRightLayerClickHandler();
 180 };
 181 
 182 #endif  // PCB_LAYER_WIDGET_H
