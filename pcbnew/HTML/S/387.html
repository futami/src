  39 /// DRC error codes:

  40 enum PCB_DRC_CODE {
  41     DRCE_FIRST = 1,
  42     DRCE_UNCONNECTED_ITEMS = DRCE_FIRST,    ///< items are unconnected

  43     DRCE_ALLOWED_ITEMS,                     ///< a disallowed item has been used

  44     DRCE_CLEARANCE,                         ///< items are too close together

  45     DRCE_TRACKS_CROSSING,                   ///< tracks are crossing

  46     DRCE_COPPER_EDGE_CLEARANCE,             ///< a copper item is too close to the board edge

  47     DRCE_ZONES_INTERSECT,                   ///< copper area outlines intersect

  48     DRCE_ZONE_HAS_EMPTY_NET,                ///< copper area has a net but no pads in nets, which is suspicious

  49     DRCE_DANGLING_VIA,                      ///< via which isn't connected to anything

  50     DRCE_DANGLING_TRACK,                    ///< track with at least one end not connected to anything

  51     DRCE_DRILLED_HOLES_TOO_CLOSE,           ///< overlapping drilled holes break drill bits

  52     DRCE_TRACK_WIDTH,                       ///< Track width is too small or too large

  53     DRCE_TOO_SMALL_VIA,                     ///< Too small via size

  54     DRCE_VIA_ANNULUS,                       ///< Via size and drill leave annulus too small or too large

  55     DRCE_TOO_SMALL_DRILL,                   ///< Too small via or pad drill

  56     DRCE_VIA_HOLE_BIGGER,                   ///< via's hole is bigger than its diameter

  57     DRCE_PADSTACK,                          ///< something is wrong with a pad or via stackup

  58     DRCE_TOO_SMALL_MICROVIA,                ///< Too small micro via size

  59     DRCE_TOO_SMALL_MICROVIA_DRILL,          ///< Too small micro via drill

  60     DRCE_KEEPOUT,                           ///< A disallowed object is inside a keepout

  61     DRCE_OVERLAPPING_FOOTPRINTS,            ///< footprint courtyards overlap

  62     DRCE_MISSING_COURTYARD,                 ///< footprint has no courtyard defined

  63     DRCE_MALFORMED_COURTYARD,               ///< footprint has a courtyard but malformed

  64                                             ///< (not convertible to a closed polygon with holes)

  65     DRCE_PTH_IN_COURTYARD,
  66     DRCE_NPTH_IN_COURTYARD,
  67     DRCE_DISABLED_LAYER_ITEM,               ///< item on a disabled layer

  68     DRCE_INVALID_OUTLINE,                   ///< invalid board outline

  69     DRCE_MISSING_FOOTPRINT,                 ///< footprint not found for netlist item

  70     DRCE_DUPLICATE_FOOTPRINT,               ///< more than one footprints found for netlist item

  71     DRCE_EXTRA_FOOTPRINT,                   ///< netlist item not found for footprint

  72 
  73     DRCE_UNRESOLVED_VARIABLE,
  74 
  75     DRCE_LAST = DRCE_UNRESOLVED_VARIABLE
  76 };

  97 /**

  98  * Design Rule Checker object that performs all the DRC tests.  The output of

  99  * the checking goes to the BOARD file in the form of two MARKER lists.  Those

 100  * two lists are displayable in the drc dialog box.  And they can optionally

 101  * be sent to a text file on disk.

 102  * This class is given access to the windows and the BOARD

 103  * that it needs via its constructor or public access functions.

 104  */
 105 class DRC : public PCB_TOOL_BASE
 106 {
 107     friend class DIALOG_DRC;
 108 
 109 public:
 110     DRC();
 111     ~DRC();
 112 
 113     /// @copydoc TOOL_INTERACTIVE::Reset()

 114     void Reset( RESET_REASON aReason ) override;
 115 
 116 private:
 117     bool     m_doUnconnectedTest;       // enable unconnected tests

 118     bool     m_testTracksAgainstZones;  // enable zone to items clearance tests

 119     bool     m_doKeepoutTest;           // enable keepout areas to items clearance tests

 120     bool     m_refillZones;             // refill zones if requested (by user).

 121     bool     m_reportAllTrackErrors;    // Report all tracks errors (or only 4 first errors)

 122     bool     m_testFootprints;          // Test footprints against schematic

 123 
 124     PCB_EDIT_FRAME*            m_editFrame;        // The pcb frame editor which owns the board

 125     BOARD*                     m_pcb;
 126     SHAPE_POLY_SET             m_board_outlines;   // The board outline including cutouts

 127     bool                       m_board_outline_valid;
 128     DIALOG_DRC*                m_drcDialog;
 129 
 130     std::vector<DRC_ITEM*>     m_unconnected;      // list of unconnected pads

 131     std::vector<DRC_ITEM*>     m_footprints;       // list of footprint warnings

 132     bool                       m_drcRun;           // indicates DRC has been run at least once

 133     bool                       m_footprintsTested; // indicates footprints were tested in last run

 134 
 135     std::vector<DRC_RULE*>     m_rules;
 136 
 137     // Temp variables for performance during a single DRC run

 138     //

 139     // wxString's c'tor is surprisingly expensive, and in the world of DRC everything matters

 140     //

 141     wxString                   m_msg;
 142     wxString                   m_clearanceSource;
 143     int                        m_largestClearance;
 144 
 145 private:
 146     ///> Sets up handlers for various events.

 147     void setTransitions() override;
 148 
 149     /**

 150      * Update needed pointers from the one pointer which is known not to change.

 151      */
 152     void updatePointers();
 153 
 154     EDA_UNITS userUnits() const { return m_editFrame->GetUserUnits(); }
 155 
 156     /**

 157      * Adds a DRC marker to the PCB through the COMMIT mechanism.

 158      */
 159     void addMarkerToPcb( BOARD_COMMIT& aCommit, MARKER_PCB* aMarker );
 160 
 161     //-----<categorical group tests>-----------------------------------------

 162 
 163     /**

 164      * Perform the DRC on all tracks.

 165      *

 166      * This test can take a while, a progress bar can be displayed

 167      * @param aActiveWindow = the active window ued as parent for the progress bar

 168      * @param aShowProgressBar = true to show a progress bar

 169      * (Note: it is shown only if there are many tracks)

 170      */
 171     void testTracks( BOARD_COMMIT& aCommit, wxWindow * aActiveWindow, bool aShowProgressBar );
 172 
 173     void testPadClearances( BOARD_COMMIT& aCommit );
 174 
 175     void testUnconnected();
 176 
 177     void testZones( BOARD_COMMIT& aCommit );
 178 
 179     void testCopperDrawItem( BOARD_COMMIT& aCommit, BOARD_ITEM* aDrawing );
 180 
 181     void testCopperTextAndGraphics( BOARD_COMMIT& aCommit );
 182 
 183     // Tests for items placed on disabled layers (causing false connections).

 184     void testDisabledLayers( BOARD_COMMIT& aCommit );
 185 
 186     /**

 187      * Test that the board outline is contiguous and composed of valid elements

 188      */
 189     void testOutline( BOARD_COMMIT& aCommit );
 190 
 191     //-----<single "item" tests>-----------------------------------------

 192 
 193     /**

 194      * Test the clearance between aRefPad and other pads.

 195      *

 196      * The pad list must be sorted by x coordinate.

 197      *

 198      * @param aRefPad is the pad to test

 199      * @param aStart is the first pad of the list to test against aRefPad

 200      * @param aEnd is the end of the list and is not included

 201      * @param x_limit is used to stop the test

 202      * (i.e. when the current pad pos X in list exceeds this limit, because the list

 203      * is sorted by X coordinate)

 204      */
 205     bool doPadToPadsDrc( BOARD_COMMIT& aCommit, D_PAD* aRefPad, D_PAD** aStart, D_PAD** aEnd,
 206                          int x_limit );
 207 
 208     /**

 209      * Test the current segment.

 210      *

 211      * @param aRefSeg The segment to test

 212      * @param aStartIt the iterator to the first track to test

 213      * @param aEndIt the marker for the iterator end

 214      * @param aTestZones true if should do copper zones test. This can be very time consumming

 215      * @return bool - true if no problems, else false and m_currentMarker is

 216      *          filled in with the problem information.

 217      */
 218     void doTrackDrc( BOARD_COMMIT& aCommit, TRACK* aRefSeg, TRACKS::iterator aStartIt,
 219                      TRACKS::iterator aEndIt, bool aTestZones );
 220 
 221 public:
 222     /**

 223      * Load the DRC rules.  Must be called after the netclasses have been read.

 224      */
 225     bool LoadRules();
 226 
 227     /**

 228      * Fetches a reasonable point for marking a violoation between two non-point objects.

 229      */
 230     static wxPoint GetLocation( TRACK* aTrack, ZONE_CONTAINER* aConflictZone );
 231     static wxPoint GetLocation( TRACK* aTrack, const SEG& aConflictSeg );
 232 
 233     /**

 234      * Open a dialog and prompts the user, then if a test run button is

 235      * clicked, runs the test(s) and creates the MARKERS.  The dialog is only

 236      * created if it is not already in existence.

 237      *

 238      * @param aParent is the parent window for wxWidgets. Usually the PCB editor frame

 239      * but can be another dialog

 240      * if aParent == NULL (default), the parent will be the PCB editor frame

 241      * and the dialog will be not modal (just float on parent

 242      * if aParent is specified, the dialog will be modal.

 243      * The modal mode is mandatory if the dialog is created from another dialog, not

 244      * from the PCB editor frame

 245      */
 246     void ShowDRCDialog( wxWindow* aParent );
 247 
 248     int ShowDRCDialog( const TOOL_EVENT& aEvent );
 249 
 250     /**

 251      * Check to see if the DRC dialog is currently shown

 252      *

 253      * @return true if the dialog is shown

 254      */
 255     bool IsDRCDialogShown();
 256 
 257     /**

 258      * Deletes this ui dialog box and zeros out its pointer to remember

 259      * the state of the dialog's existence.

 260      *

 261      * @param aReason Indication of which button was clicked to cause the destruction.

 262      * if aReason == wxID_OK, design parameters values which can be entered from the dialog

 263      * will bbe saved in design parameters list

 264      */
 265     void DestroyDRCDialog( int aReason );
 266 
 267     /**

 268      * Run all the tests specified with a previous call to

 269      * SetSettings()

 270      * @param aMessages = a wxTextControl where to display some activity messages. Can be NULL

 271      */
 272     void RunTests( wxTextCtrl* aMessages = NULL );
 273 };
