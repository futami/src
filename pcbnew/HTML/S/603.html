  30 #ifndef PLOTCONTROLLER_H_
  31 #define PLOTCONTROLLER_H_
  32 
  33 #include <pcb_plot_params.h>
  34 #include <layers_id_colors_and_visibility.h>
  35 
  36 class PLOTTER;
  37 class BOARD;
  38 
  39 
  40 /**

  41  * Batch plotter state object. Keeps the plot options and handles multiple

  42  * plot requests

  43  * Especially useful in Python scripts

  44  */
  45 class PLOT_CONTROLLER
  46 {
  47 public:
  48     /** Batch plotter constructor, nothing interesting here */
  49     PLOT_CONTROLLER( BOARD *aBoard );
  50 
  51     /** Batch plotter destructor, ensures that the last plot is closed

  52      */
  53     ~PLOT_CONTROLLER();
  54 
  55     /**

  56      * Accessor to the plot parameters and options

  57      */
  58     PCB_PLOT_PARAMS& GetPlotOptions() { return m_plotOptions; }
  59 
  60     void SetLayer( LAYER_NUM aLayer ) { m_plotLayer = aLayer; }
  61     LAYER_NUM GetLayer() { return m_plotLayer; }
  62 
  63 
  64     /**

  65      * @return true if a plotter is initialized and can be used

  66      */
  67     bool IsPlotOpen() const { return m_plotter != NULL; }
  68 
  69     /** Close the current plot, nothing happens if it isn't open

  70      */
  71     void ClosePlot();
  72 
  73     /** Open a new plotfile; works as a factory for plotter objects

  74      * @param aSuffix is a string added to the base filename (derived from

  75      * the board filename) to identify the plot file

  76      * @param aFormat is the plot file format identifier

  77      * @param aSheetDesc

  78      */
  79     bool OpenPlotfile( const wxString& aSuffix, PLOT_FORMAT aFormat, const wxString& aSheetDesc );
  80 
  81     /** Plot a single layer on the current plotfile

  82      * m_plotLayer is the layer to plot

  83      */
  84     bool PlotLayer();
  85 
  86     /**

  87      * @return the current plot full filename, set by OpenPlotfile

  88      */
  89     const wxString GetPlotFileName() { return m_plotFile.GetFullPath(); }
  90 
  91     /**

  92      * @return the current plot full filename, set by OpenPlotfile

  93      */
  94     const wxString GetPlotDirName() { return m_plotFile.GetPathWithSep(); }
  95 
  96     /**

  97      * Plotters can plot in Black and White mode or Color mode

  98      * SetColorMode activate/de-actiavte the Color mode.

  99      * @param aColorMode = true to activate the plot color mode

 100      */
 101     void SetColorMode( bool  );
 102 
 103     /**

 104      * @return  true if the current plot color mode is Color,

 105      *   false if the current plot color mode is Black and White

 106      */
 107     bool GetColorMode();
 108 
 109 private:
 110     /// the layer to plot

 111     LAYER_NUM m_plotLayer;
 112 
 113     /// Option bank

 114     PCB_PLOT_PARAMS m_plotOptions;
 115 
 116     /// This is the plotter object; it starts NULL and become instantiated

 117     /// when a plotfile is requested

 118     PLOTTER* m_plotter;
 119 
 120     /// The board we're plotting

 121     BOARD* m_board;
 122 
 123     /// The current plot filename, set by OpenPlotfile

 124     wxFileName m_plotFile;
 125 };

 127 #endif
