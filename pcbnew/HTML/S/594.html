  59 /**

  60  * PCB_RENDER_SETTINGS

  61  * Stores PCB specific render settings.

  62  */
  63 class PCB_RENDER_SETTINGS : public RENDER_SETTINGS
  64 {
  65 public:
  66     friend class PCB_PAINTER;
  67 
  68     ///> Flags to control clearance lines visibility

  69     enum CLEARANCE_MODE
  70     {
  71         CL_NONE             = 0x00,
  72 
  73         // Object type

  74         CL_PADS             = 0x01,
  75         CL_VIAS             = 0x02,
  76         CL_TRACKS           = 0x04,
  77 
  78         // Existence

  79         CL_NEW              = 0x08,
  80         CL_EDITED           = 0x10,
  81         CL_EXISTING         = 0x20
  82     };
  83 
  84     ///> Determines how zones should be displayed

  85     enum DISPLAY_ZONE_MODE
  86     {
  87         DZ_HIDE_FILLED = 0,
  88         DZ_SHOW_FILLED,
  89         DZ_SHOW_OUTLINED
  90     };
  91 
  92     ///> Determines how net color overrides should be applied

  93     enum class NET_COLOR_MODE
  94     {
  95         OFF,        ///< Net (and netclass) colors are not shown

  96         RATSNEST,   ///< Net/netclass colors are shown on ratsnest lines only

  97         ALL         ///< Net/netclass colors are shown on all net copper

  98     };
  99 
 100     PCB_RENDER_SETTINGS();
 101 
 102     /**

 103      * Function LoadDisplayOptions

 104      * Loads settings related to display options (high-contrast mode, full or outline modes

 105      * for vias/pads/tracks and so on).

 106      * @param aOptions are settings that you want to use for displaying items.

 107      */
 108     void LoadDisplayOptions( const PCB_DISPLAY_OPTIONS& aOptions, bool aShowPageLimits );
 109 
 110     /**

 111      * Loads net-specific render settings

 112      * @param aSettings is the NET_SETTINGS for the current proejct

 113      * @param aList is the list of nets on the board

 114      * @param aHiddenNets is a list of nets to hide from the ratsnest

 115      */
 116     void LoadNetSettings( const NET_SETTINGS& aSettings, const NETINFO_LIST& aList,
 117                           const std::set<int>& aHiddenNets );
 118 
 119     virtual void LoadColors( const COLOR_SETTINGS* aSettings ) override;
 120 
 121     /// @copydoc RENDER_SETTINGS::GetColor()

 122     virtual const COLOR4D& GetColor( const VIEW_ITEM* aItem, int aLayer ) const override;
 123 
 124     /**

 125      * Function SetSketchMode

 126      * Turns on/off sketch mode for given item layer.

 127      * @param aItemLayer is the item layer that is changed.

 128      * @param aEnabled decides if it is drawn in sketch mode (true for sketched mode,

 129      * false for filled mode).

 130      */
 131     inline void SetSketchMode( int aItemLayer, bool aEnabled )
 132     {
 133         m_sketchMode[aItemLayer] = aEnabled;
 134     }
 135 
 136     /**

 137      * Function GetSketchMode

 138      * Returns sketch mode setting for a given item layer.

 139      * @param aItemLayer is the item layer that is changed.

 140      */
 141     inline bool GetSketchMode( int aItemLayer ) const
 142     {
 143         return m_sketchMode[aItemLayer];
 144     }
 145 
 146     /**

 147      * Turns on/off sketch mode for graphic items (DRAWSEGMENTs, texts).

 148      * @param aEnabled decides if it is drawn in sketch mode (true for sketched mode,

 149      * false for filled mode).

 150      */
 151     inline void SetSketchModeGraphicItems( bool aEnabled )
 152     {
 153         m_sketchGraphics = aEnabled;
 154     }
 155 
 156     /**

 157      * Turns on/off drawing outline and hatched lines for zones.

 158      */
 159     void EnableZoneOutlines( bool aEnabled )
 160     {
 161         m_zoneOutlines = aEnabled;
 162     }
 163 
 164     inline bool IsBackgroundDark() const override
 165     {
 166         auto luma = m_layerColors[ LAYER_PCB_BACKGROUND ].GetBrightness();
 167 
 168         return luma < 0.5;
 169     }
 170 
 171     const COLOR4D& GetBackgroundColor() override { return m_layerColors[ LAYER_PCB_BACKGROUND ]; }
 172 
 173     void SetBackgroundColor( const COLOR4D& aColor ) override
 174     {
 175         m_layerColors[ LAYER_PCB_BACKGROUND ] = aColor;
 176     }
 177 
 178     const COLOR4D& GetGridColor() override { return m_layerColors[ LAYER_GRID ]; }
 179 
 180     const COLOR4D& GetCursorColor() override { return m_layerColors[ LAYER_CURSOR ]; }
 181 
 182     /** Switch the contrast mode setting (HIGH_CONTRAST_MODE:NORMAL, DIMMED or HIDDEN )

 183      * to control how the non active layers are shown

 184      */
 185     void SetContrastModeDisplay( HIGH_CONTRAST_MODE aMode ) { m_contrastModeDisplay = aMode; }
 186 
 187     /** return the contrast mode setting (HIGH_CONTRAST_MODE:NORMAL, DIMMED or HIDDEN )

 188      */
 189     HIGH_CONTRAST_MODE GetContrastModeDisplay() { return m_contrastModeDisplay; }
 190 
 191     inline bool GetCurvedRatsnestLinesEnabled() const { return m_curvedRatsnestlines; }
 192 
 193     inline bool GetGlobalRatsnestLinesEnabled() const { return m_globalRatsnestlines; }
 194 
 195     bool GetDrawIndividualViaLayers() const { return m_drawIndividualViaLayers; }
 196     void SetDrawIndividualViaLayers( bool aFlag ) { m_drawIndividualViaLayers = aFlag; }
 197 
 198     NET_COLOR_MODE GetNetColorMode() const { return m_netColorMode; }
 199     void SetNetColorMode( NET_COLOR_MODE aMode ) { m_netColorMode = aMode; }
 200 
 201     std::map<wxString, KIGFX::COLOR4D>& GetNetclassColorMap() { return m_netclassColors; }
 202 
 203     std::map<int, KIGFX::COLOR4D>& GetNetColorMap() { return m_netColors; }
 204 
 205     std::set<int>& GetHiddenNets() { return m_hiddenNets; }
 206     const std::set<int>& GetHiddenNets() const { return m_hiddenNets; }
 207 
 208 protected:
 209     ///> Flag determining if items on a given layer should be drawn as an outline or a filled item

 210     bool    m_sketchMode[GAL_LAYER_ID_END];
 211 
 212     ///> Flag determining if graphic items should be outlined or stroked

 213     bool    m_sketchGraphics;
 214 
 215     ///> Flag determining if text items should be outlined or stroked

 216     bool    m_sketchText;
 217 
 218     ///> Flag determining if pad numbers should be visible

 219     bool    m_padNumbers;
 220 
 221     ///> Flag determining if net names should be visible for pads

 222     bool    m_netNamesOnPads;
 223 
 224     ///> Flag determining if net names should be visible for tracks

 225     bool    m_netNamesOnTracks;
 226 
 227     ///> Flag determining if net names should be visible for vias

 228     bool    m_netNamesOnVias;
 229 
 230     ///> Flag determining if zones should have outlines drawn

 231     bool    m_zoneOutlines;
 232 
 233     ///> Flag determining if ratsnest lines should be drawn curved

 234     bool    m_curvedRatsnestlines = true;
 235 
 236     ///> Flag determining if ratsnest lines are shown by default

 237     bool    m_globalRatsnestlines = true;
 238 
 239     bool    m_drawIndividualViaLayers = false;
 240 
 241     ///> Maximum font size for netnames (and other dynamically shown strings)

 242     static const double MAX_FONT_SIZE;
 243 
 244     ///> Option for different display modes for zones

 245     DISPLAY_ZONE_MODE m_displayZone;
 246 
 247     ///> Clearance visibility settings

 248     int m_clearance;
 249 
 250     ///> Color used for highlighting selection candidates

 251     COLOR4D m_selectionCandidateColor;
 252 
 253     ///> How to display nets and netclasses with color overrides

 254     NET_COLOR_MODE m_netColorMode;
 255 
 256     ///> Overrides for specific netclass colors

 257     std::map<wxString, KIGFX::COLOR4D> m_netclassColors;
 258 
 259     ///> Overrides for specific net colors, stored as netcodes for the ratsnest to access easily

 260     std::map<int, KIGFX::COLOR4D> m_netColors;
 261 
 262     ///> Set of net codes that should not have their ratsnest displayed

 263     std::set<int> m_hiddenNets;
 264 
 265     ///> How to display inactive layers (HIGH_CONTRAST_MODE:NORMAL, DIMMED or HIDDEN )

 266     HIGH_CONTRAST_MODE m_contrastModeDisplay;
 267 };

 270 /**

 271  * PCB_PAINTER

 272  * Contains methods for drawing PCB-specific items.

 273  */
 274 class PCB_PAINTER : public PAINTER
 275 {
 276 public:
 277     PCB_PAINTER( GAL* aGal );
 278 
 279     /// @copydoc PAINTER::ApplySettings()

 280     virtual void ApplySettings( const RENDER_SETTINGS* aSettings ) override
 281     {
 282         m_pcbSettings = *static_cast<const PCB_RENDER_SETTINGS*>( aSettings );
 283     }
 284 
 285     /// @copydoc PAINTER::GetAdapter()

 286     virtual PCB_RENDER_SETTINGS* GetSettings() override
 287     {
 288         return &m_pcbSettings;
 289     }
 290 
 291     /// @copydoc PAINTER::Draw()

 292     virtual bool Draw( const VIEW_ITEM* aItem, int aLayer ) override;
 293 
 294 protected:
 295     PCB_RENDER_SETTINGS m_pcbSettings;
 296 
 297     // Drawing functions for various types of PCB-specific items

 298     void draw( const TRACK* aTrack, int aLayer );
 299     void draw( const ARC* aArc, int aLayer );
 300     void draw( const VIA* aVia, int aLayer );
 301     void draw( const D_PAD* aPad, int aLayer );
 302     void draw( const DRAWSEGMENT* aSegment, int aLayer );
 303     void draw( const TEXTE_PCB* aText, int aLayer );
 304     void draw( const TEXTE_MODULE* aText, int aLayer );
 305     void draw( const MODULE* aModule, int aLayer );
 306     void draw( const ZONE_CONTAINER* aZone, int aLayer );
 307     void draw( const DIMENSION* aDimension, int aLayer );
 308     void draw( const PCB_TARGET* aTarget );
 309     void draw( const MARKER_PCB* aMarker );
 310 
 311     /**

 312      * Function getLineThickness()

 313      * Get the thickness to draw for a line (e.g. 0 thickness lines

 314      * get a minimum value).

 315      * @param aActualThickness line own thickness

 316      * @return the thickness to draw

 317      */
 318     int getLineThickness( int aActualThickness ) const;
 319 
 320     /**

 321      * Return drill shape of a pad.

 322      */
 323     virtual int getDrillShape( const D_PAD* aPad ) const;
 324 
 325     /**

 326      * Return drill size for a pad (internal units).

 327      */
 328     virtual VECTOR2D getDrillSize( const D_PAD* aPad ) const;
 329 
 330     /**

 331      * Return drill diameter for a via (internal units).

 332      */
 333     virtual int getDrillSize( const VIA* aVia ) const;
 334 };
