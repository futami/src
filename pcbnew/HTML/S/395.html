  41 bool DRC_KEEPOUT_TESTER::RunDRC( EDA_UNITS aUnits, BOARD& aBoard )
  42 {
  43     bool success = true;
  44 
  45     m_units = aUnits;
  46     m_board = &aBoard;
  47 
  48     // Get a list of all zones to inspect, from both board and footprints

  49     std::list<ZONE_CONTAINER*> areasToInspect = m_board->GetZoneList( true );
  50 
  51     // Test keepout areas for vias, tracks and pads inside keepout areas

  52     for( ZONE_CONTAINER* area : areasToInspect )
  53     {
  54         m_keepoutFlags = area->GetKeepouts( &m_sources );
  55 
  56         if( m_keepoutFlags > 0 )
  57         {
  58             m_zone = area;
  59             m_zoneBBox = area->GetBoundingBox();
  60 
  61             success &= checkTracksAndVias();
  62             success &= checkFootprints();
  63             success &= checkDrawings();
  64         }
  65     }
  66 
  67     return success;
  68 }

  71 bool DRC_KEEPOUT_TESTER::checkTracksAndVias()
  72 {
  73     constexpr int VIA_MASK = DISALLOW_VIAS | DISALLOW_MICRO_VIAS | DISALLOW_BB_VIAS;
  74     constexpr int CHECK_VIAS_MASK = VIA_MASK | DISALLOW_HOLES;
  75     constexpr int CHECK_TRACKS_AND_VIAS_MASK = CHECK_VIAS_MASK | DISALLOW_TRACKS;
  76 
  77     if(( m_keepoutFlags & CHECK_TRACKS_AND_VIAS_MASK ) == 0 )
  78         return true;
  79 
  80     bool success = true;
  81 
  82     for( TRACK* segm : m_board->Tracks() )
  83     {
  84         if( !m_zoneBBox.Intersects( segm->GetBoundingBox() ) )
  85             continue;
  86 
  87         if( segm->Type() == PCB_TRACE_T && ( m_keepoutFlags & DISALLOW_TRACKS ) != 0 )
  88         {
  89             // Ignore if the keepout zone is not on the same layer

  90             if( !m_zone->IsOnLayer( segm->GetLayer() ) )
  91                 continue;
  92 
  93             SEG trackSeg( segm->GetStart(), segm->GetEnd() );
  94 
  95             if( m_zone->Outline()->Collide( trackSeg, segm->GetWidth() / 2 ) )
  96             {
  97                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_KEEPOUT );
  98 
  99                 m_msg.Printf( drcItem->GetErrorText() + _( " (%s)" ),
 100                               m_sources.at( DISALLOW_TRACKS ) );
 101 
 102                 drcItem->SetErrorMessage( m_msg );
 103                 drcItem->SetItems( segm, m_zone );
 104 
 105                 HandleMarker( new MARKER_PCB( drcItem, DRC::GetLocation( segm, m_zone ) ) );
 106                 success = false;
 107             }
 108         }
 109         else if( segm->Type() == PCB_VIA_T && ( m_keepoutFlags & CHECK_VIAS_MASK ) != 0 )
 110         {
 111             VIA* via  = static_cast<VIA*>( segm );
 112             SEG  seg( via->GetPosition(), via->GetPosition() );
 113             int  test = 0;
 114             int  clearance = via->GetWidth() / 2;
 115 
 116             if( ( m_keepoutFlags & DISALLOW_VIAS ) > 0 )
 117             {
 118                 test = DISALLOW_VIAS;
 119             }
 120             else if( via->GetViaType() == VIATYPE::MICROVIA
 121                         && ( m_keepoutFlags & DISALLOW_MICRO_VIAS ) > 0 )
 122             {
 123                 test = DISALLOW_MICRO_VIAS;
 124             }
 125             else if( via->GetViaType() == VIATYPE::BLIND_BURIED
 126                         && ( m_keepoutFlags & DISALLOW_BB_VIAS ) > 0 )
 127             {
 128                 test = DISALLOW_BB_VIAS;
 129             }
 130             else if( ( m_keepoutFlags & DISALLOW_HOLES ) > 0 )
 131             {
 132                 test = DISALLOW_HOLES;
 133                 clearance = via->GetDrillValue() / 2;
 134             }
 135             else
 136                 continue;
 137 
 138             if( m_zone->Outline()->Collide( seg, clearance ) )
 139             {
 140                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_KEEPOUT );
 141                 m_msg.Printf( drcItem->GetErrorText() + _( " (%s)" ), m_sources.at( test ) );
 142                 drcItem->SetErrorMessage( m_msg );
 143                 drcItem->SetItems( segm, m_zone );
 144 
 145                 HandleMarker( new MARKER_PCB( drcItem, DRC::GetLocation( segm, m_zone ) ) );
 146                 success = false;
 147             }
 148         }
 149     }
 150 
 151     return success;
 152 }

 155 bool DRC_KEEPOUT_TESTER::checkFootprints()
 156 {
 157     constexpr int CHECK_PADS_MASK = DISALLOW_PADS | DISALLOW_HOLES;
 158     constexpr int CHECK_FOOTPRINTS_MASK = CHECK_PADS_MASK | DISALLOW_FOOTPRINTS;
 159 
 160     if(( m_keepoutFlags & CHECK_FOOTPRINTS_MASK ) == 0 )
 161         return true;
 162 
 163     bool success = true;
 164 
 165     for( MODULE* fp : m_board->Modules() )
 166     {
 167         if( !m_zoneBBox.Intersects( fp->GetBoundingBox() ) )
 168             continue;
 169 
 170         if( ( m_keepoutFlags & DISALLOW_FOOTPRINTS ) > 0
 171                 && ( fp->IsFlipped() ? m_zone->CommonLayerExists( LSET::BackMask() )
 172                                      : m_zone->CommonLayerExists( LSET::FrontMask() ) ) )
 173         {
 174             SHAPE_POLY_SET poly;
 175 
 176             if( fp->BuildPolyCourtyard() )
 177                 poly = fp->IsFlipped() ? fp->GetPolyCourtyardBack() : fp->GetPolyCourtyardFront();
 178 
 179             if( poly.OutlineCount() == 0 )
 180                 poly = fp->GetBoundingPoly();
 181 
 182             // Build the common area between footprint and the keepout area:

 183             poly.BooleanIntersection( *m_zone->Outline(), SHAPE_POLY_SET::PM_FAST );
 184 
 185             // If it's not empty then we have a violation

 186             if( poly.OutlineCount() )
 187             {
 188                 const VECTOR2I& pt = poly.CVertex( 0, 0, -1 );
 189                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_KEEPOUT );
 190 
 191                 m_msg.Printf( drcItem->GetErrorText() + _( " (%s)" ),
 192                               m_sources.at( DISALLOW_FOOTPRINTS ) );
 193 
 194                 drcItem->SetErrorMessage( m_msg );
 195                 drcItem->SetItems( fp, m_zone );
 196 
 197                 HandleMarker( new MARKER_PCB( drcItem, (wxPoint) pt ) );
 198                 success = false;
 199             }
 200         }
 201 
 202         if( ( m_keepoutFlags & CHECK_PADS_MASK ) > 0 )
 203         {
 204             success &= checkPads( fp );
 205         }
 206     }
 207 
 208     return success;
 209 }

 212 bool DRC_KEEPOUT_TESTER::checkPads( MODULE* aModule )
 213 {
 214     bool success = true;
 215 
 216     for( D_PAD* pad : aModule->Pads() )
 217     {
 218         if( !m_zone->CommonLayerExists( pad->GetLayerSet() ) )
 219             continue;
 220 
 221         // Fast test to detect a pad inside the keepout area bounding box.

 222         EDA_RECT padBBox( pad->GetPosition(), wxSize() );
 223         padBBox.Inflate( pad->GetBoundingRadius() );
 224 
 225         if( !m_zoneBBox.Intersects( padBBox ) )
 226             continue;
 227 
 228         if( ( m_keepoutFlags & DISALLOW_PADS ) > 0 )
 229         {
 230             SHAPE_POLY_SET outline = *pad->GetEffectivePolygon();
 231 
 232             // Build the common area between pad and the keepout area:

 233             outline.BooleanIntersection( *m_zone->Outline(), SHAPE_POLY_SET::PM_FAST );
 234 
 235             // If it's not empty then we have a violation

 236             if( outline.OutlineCount() )
 237             {
 238                 const VECTOR2I& pt = outline.CVertex( 0, 0, -1 );
 239                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_KEEPOUT );
 240 
 241                 m_msg.Printf( drcItem->GetErrorText() + _( " (%s)" ),
 242                               m_sources.at( DISALLOW_PADS ) );
 243 
 244                 drcItem->SetErrorMessage( m_msg );
 245                 drcItem->SetItems( pad, m_zone );
 246 
 247                 HandleMarker( new MARKER_PCB( drcItem, (wxPoint) pt ) );
 248                 success = false;
 249             }
 250         }
 251         else if( ( m_keepoutFlags & DISALLOW_HOLES ) > 0 )
 252         {
 253             const SHAPE_SEGMENT* slot = pad->GetEffectiveHoleShape();
 254 
 255             if( m_zone->Outline()->Collide( slot->GetSeg(), slot->GetWidth() / 2 ) )
 256             {
 257                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_KEEPOUT );
 258 
 259                 m_msg.Printf( drcItem->GetErrorText() + _( " (%s)" ),
 260                               m_sources.at( DISALLOW_HOLES ) );
 261 
 262                 drcItem->SetErrorMessage( m_msg );
 263                 drcItem->SetItems( pad, m_zone );
 264 
 265                 HandleMarker( new MARKER_PCB( drcItem, pad->GetPosition() ) );
 266                 success = false;
 267             }
 268         }
 269     }
 270 
 271     return success;
 272 }

 275 bool DRC_KEEPOUT_TESTER::checkDrawings()
 276 {
 277     constexpr int CHECK_DRAWINGS_MASK = DISALLOW_TEXTS | DISALLOW_GRAPHICS;
 278     constexpr KICAD_T graphicTypes[] = { PCB_LINE_T, PCB_DIMENSION_T, PCB_TARGET_T, EOT };
 279 
 280     if(( m_keepoutFlags & CHECK_DRAWINGS_MASK ) == 0 )
 281         return true;
 282 
 283     bool success = true;
 284 
 285     for( BOARD_ITEM* drawing : m_board->Drawings() )
 286     {
 287         if( !m_zoneBBox.Intersects( drawing->GetBoundingBox() ) )
 288             continue;
 289 
 290         int  sourceId = 0;
 291 
 292         if( drawing->IsType( graphicTypes ) && ( m_keepoutFlags & DISALLOW_GRAPHICS ) > 0 )
 293             sourceId = DISALLOW_GRAPHICS;
 294         else if( drawing->Type() == PCB_TEXT_T && ( m_keepoutFlags & DISALLOW_TEXTS ) > 0 )
 295             sourceId = DISALLOW_TEXTS;
 296         else
 297             continue;
 298 
 299         SHAPE_POLY_SET poly;
 300         drawing->TransformShapeWithClearanceToPolygon( poly, 0 );
 301 
 302         // Build the common area between footprint and the keepout area:

 303         poly.BooleanIntersection( *m_zone->Outline(), SHAPE_POLY_SET::PM_FAST );
 304 
 305         // If it's not empty then we have a violation

 306         if( poly.OutlineCount() )
 307         {
 308             const VECTOR2I& pt = poly.CVertex( 0, 0, -1 );
 309             DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_KEEPOUT );
 310             m_msg.Printf( drcItem->GetErrorText() + _( " (%s)" ), m_sources.at( sourceId ) );
 311             drcItem->SetErrorMessage( m_msg );
 312             drcItem->SetItems( drawing, m_zone );
 313 
 314             HandleMarker( new MARKER_PCB( drcItem, (wxPoint) pt ) );
 315             success = false;
 316         }
 317     }
 318 
 319     return success;
 320 }
