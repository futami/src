  64 void RATSNEST_VIEWITEM::ViewDraw( int aLayer, KIGFX::VIEW* aView ) const
  65 {
  66     std::unique_lock<std::mutex> lock( m_data->GetLock(), std::try_to_lock );
  67 
  68     if( !lock )
  69         return;
  70 
  71     constexpr int CROSS_SIZE = 200000;
  72 
  73     auto gal = aView->GetGAL();
  74   gal->SetIsStroke( true );
  75     gal->SetIsFill( false );
  76     gal->SetLineWidth( 1.0 );
  77     auto rs = static_cast<PCB_RENDER_SETTINGS*>( aView->GetPainter()->GetSettings() );
  78 
  79     COLOR4D defaultColor = rs->GetColor( nullptr, LAYER_RATSNEST );
  80     COLOR4D color        = defaultColor;
  81 
  82     bool colorByNet = rs->GetNetColorMode() != PCB_RENDER_SETTINGS::NET_COLOR_MODE::OFF;
  83 
  84     std::set<int>        highlightedNets = rs->GetHighlightNetCodes();
  85     const std::set<int>& hiddenNets      = rs->GetHiddenNets();
  86 
  87     std::map<int, KIGFX::COLOR4D>& netColors = rs->GetNetColorMap();
  88 
  89     const bool curved_ratsnest = rs->GetCurvedRatsnestLinesEnabled();
  90 
  91     // Draw the "dynamic" ratsnest (i.e. for objects that may be currently being moved)

  92     for( const RN_DYNAMIC_LINE& l : m_data->GetDynamicRatsnest() )
  93     {
  94         if( hiddenNets.count( l.netCode ) )
  95             continue;
  96 
  97         if( colorByNet && netColors.count( l.netCode ) )
  98             color = netColors.at( l.netCode );
  99         else
 100             color = defaultColor;
 101 
 102         gal->SetStrokeColor( color.Brightened( 0.5 ) );
 103 
 104         if ( l.a == l.b )
 105         {
 106             gal->DrawLine( VECTOR2I( l.a.x - CROSS_SIZE, l.a.y - CROSS_SIZE ),
 107                            VECTOR2I( l.b.x + CROSS_SIZE, l.b.y + CROSS_SIZE ) );
 108             gal->DrawLine( VECTOR2I( l.a.x - CROSS_SIZE, l.a.y + CROSS_SIZE ),
 109                            VECTOR2I( l.b.x + CROSS_SIZE, l.b.y - CROSS_SIZE ) );
 110         }
 111         else
 112         {
 113             if( curved_ratsnest )
 114             {
 115                 auto dx = l.b.x - l.a.x;
 116                 auto dy = l.b.y - l.a.y;
 117                 const auto center = VECTOR2I( l.a.x + 0.5 * dx - 0.1 * dy,
 118                         l.a.y + 0.5 * dy + 0.1 * dx );
 119                 gal->DrawCurve( l.a, center, center, l.b );
 120             }
 121             else
 122             {
 123                 gal->DrawLine( l.a, l.b );
 124             }
 125         }
 126     }
 127 
 128     for( int i = 1 /* skip "No Net" at [0] */; i < m_data->GetNetCount(); ++i )
 129     {
 130         if( hiddenNets.count( i ) )
 131             continue;
 132 
 133         RN_NET* net = m_data->GetRatsnestForNet( i );
 134 
 135         if( !net )
 136             continue;
 137 
 138         if( colorByNet && netColors.count( i ) )
 139             color = netColors.at( i );
 140         else
 141             color = defaultColor;
 142 
 143         // Draw the "static" ratsnest

 144         if( highlightedNets.count( i ) )
 145             gal->SetStrokeColor( color.Brightened(0.8) );
 146         else
 147             gal->SetStrokeColor( color );  // using the default ratsnest color for not highlighted

 148 
 149         for( const auto& edge : net->GetUnconnected() )
 150         {
 151             //if ( !edge.IsVisible() )

 152             //    continue;

 153 
 154             const auto& sourceNode = edge.GetSourceNode();
 155             const auto& targetNode = edge.GetTargetNode();
 156             const VECTOR2I source( sourceNode->Pos() );
 157             const VECTOR2I target( targetNode->Pos() );
 158 
 159             if( !sourceNode->Valid() || !targetNode->Valid() )
 160                 continue;
 161 
 162             bool enable =  !sourceNode->GetNoLine() && !targetNode->GetNoLine();
 163             bool show;
 164 
 165             // If the global ratsnest is currently enabled, the local ratsnest

 166             // should be easy to turn off, so either element can disable it

 167             // If the global ratsnest is disabled, the local ratsnest should be easy to turn on

 168             // so either element can enable it.

 169             if( rs->GetGlobalRatsnestLinesEnabled() )
 170                 show = sourceNode->Parent()->GetLocalRatsnestVisible() &&
 171                        targetNode->Parent()->GetLocalRatsnestVisible();
 172             else
 173                 show = sourceNode->Parent()->GetLocalRatsnestVisible() ||
 174                        targetNode->Parent()->GetLocalRatsnestVisible();
 175 
 176             if ( enable && show )
 177             {
 178                 if ( source == target )
 179                 {
 180                     gal->DrawLine( VECTOR2I( source.x - CROSS_SIZE, source.y - CROSS_SIZE ),
 181                                    VECTOR2I( source.x + CROSS_SIZE, source.y + CROSS_SIZE ) );
 182                     gal->DrawLine( VECTOR2I( source.x - CROSS_SIZE, source.y + CROSS_SIZE ),
 183                                    VECTOR2I( source.x + CROSS_SIZE, source.y - CROSS_SIZE ) );
 184                 }
 185                 else
 186                 {
 187                     if( curved_ratsnest )
 188                     {
 189                         auto dx = target.x - source.x;
 190                         auto dy = target.y - source.y;
 191                         const auto center = VECTOR2I( source.x + 0.5 * dx - 0.1 * dy,
 192                                 source.y + 0.5 * dy + 0.1 * dx );
 193                         gal->DrawCurve( source, center, center, target );
 194                     }
 195                     else
 196                     {
 197                         gal->DrawLine( source, target );
 198                     }
 199                 }
 200             }
 201         }
 202     }
 203 }
