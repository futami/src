  54 bool PCB_EDIT_FRAME::ImportSpecctraSession( const wxString& fullFileName )
  55 {
  56     // To avoid issues with undo/redo lists (dangling pointers)

  57     // clear the lists

  58     // todo: use undo/redo feature

  59     ClearUndoRedoList();
  60 
  61     SPECCTRA_DB     db;
  62     LOCALE_IO       toggle;
  63 
  64     try
  65     {
  66         db.LoadSESSION( fullFileName );
  67         db.FromSESSION( GetBoard() );
  68     }
  69     catch( const IO_ERROR& ioe )
  70     {
  71         wxString msg = _( "Board may be corrupted, do not save it.\n Fix problem and try again" );
  72 
  73         wxString extra = ioe.What();
  74 
  75         DisplayErrorMessage( this, msg, extra);
  76         return false;
  77     }
  78 
  79     OnModify();
  80 
  81     GetBoard()->GetConnectivity()->Clear();
  82     GetBoard()->GetConnectivity()->Build( GetBoard() );
  83 
  84     if( GetCanvas() )    // Update view:

  85     {
  86         // Update footprint positions

  87         GetCanvas()->GetView()->RecacheAllItems();
  88 
  89         // add imported tracks (previous tracks are removed, therfore all are new)

  90         for( auto track : GetBoard()->Tracks() )
  91             GetCanvas()->GetView()->Add( track );
  92    }
  93 
  94     SetStatusText( wxString( _( "Session file imported and merged OK." ) ) );
  95 
  96     Refresh();
  97 
  98     return true;
  99 }

 105 /**

 106  * Function scale

 107  * converts a session file distance to KiCad units of deci-mils.

 108  * @param distance The session file length to convert.

 109  * @param aResolution The session UNIT_RES which holds the engineering unit

 110  *  specifier

 111  * @return int - The KiCad length in internal unit

 112  */
 113 static int scale( double distance, UNIT_RES* aResolution )
 114 {
 115     double  resValue = aResolution->GetValue();
 116     double  factor;
 117 
 118     switch( aResolution->GetEngUnits() )
 119     {
 120     default:
 121     case T_inch:
 122         factor = 25.4e6;        // nanometers per inch

 123         break;
 124     case T_mil:
 125         factor = 25.4e3;        // nanometers per mil

 126         break;
 127     case T_cm:
 128         factor = 1e7;           // nanometers per cm

 129         break;
 130     case T_mm:
 131         factor = 1e6;           // nanometers per mm

 132         break;
 133     case T_um:
 134         factor = 1e3;           // nanometers per um

 135         break;
 136     }
 137 
 138     int ret = KiROUND( factor * distance / resValue );
 139 
 140     return ret;
 141 }

 144 /**

 145  * Function mapPt

 146  * translates a point from the Specctra Session format coordinate system

 147  * to the KiCad coordinate system.

 148  * @param aPoint The session point to translate

 149  * @param aResolution - The amount to scale the point.

 150  * @return wxPoint - The KiCad coordinate system point.

 151  */
 152 static wxPoint mapPt( const POINT& aPoint, UNIT_RES* aResolution )
 153 {
 154     wxPoint ret(  scale( aPoint.x, aResolution ),
 155                  -scale( aPoint.y, aResolution ) );    // negate y

 156 
 157     return ret;
 158 }

 161 TRACK* SPECCTRA_DB::makeTRACK( PATH* aPath, int aPointIndex, int aNetcode )
 162 {
 163     int layerNdx = findLayerName( aPath->layer_id );
 164 
 165     if( layerNdx == -1 )
 166     {
 167         wxString layerName = FROM_UTF8( aPath->layer_id.c_str() );
 168         THROW_IO_ERROR( wxString::Format( _("Session file uses invalid layer id \"%s\""),
 169                                           GetChars( layerName ) ) );
 170     }
 171 
 172     TRACK* track = new TRACK( sessionBoard );
 173 
 174     track->SetStart( mapPt( aPath->points[aPointIndex+0], routeResolution ) );
 175     track->SetEnd( mapPt( aPath->points[aPointIndex+1], routeResolution ) );
 176     track->SetLayer( pcbLayer2kicad[layerNdx] );
 177     track->SetWidth( scale( aPath->aperture_width, routeResolution ) );
 178     track->SetNetCode( aNetcode );
 179 
 180     return track;
 181 }

 184 ::VIA* SPECCTRA_DB::makeVIA( PADSTACK* aPadstack, const POINT& aPoint,
 185             int aNetCode, int aViaDrillDefault )
 186 {
 187     ::VIA*  via = 0;
 188     SHAPE*  shape;
 189 
 190     int     shapeCount = aPadstack->Length();
 191     int     drill_diam_iu = -1;
 192     int     copperLayerCount = sessionBoard->GetCopperLayerCount();
 193 
 194 
 195     // The drill diameter is encoded in the padstack name if Pcbnew did the DSN export.

 196     // It is after the colon and before the last '_'

 197     int     drillStartNdx = aPadstack->padstack_id.find( ':' );
 198 
 199     if( drillStartNdx != -1 )
 200     {
 201         ++drillStartNdx;    // skip over the ':'

 202 
 203         int drillEndNdx = aPadstack->padstack_id.rfind( '_' );
 204         if( drillEndNdx != -1 )
 205         {
 206             std::string diam_txt( aPadstack->padstack_id,
 207                             drillStartNdx, drillEndNdx-drillStartNdx );
 208 
 209             double drill_um = strtod( diam_txt.c_str(), 0 );
 210 
 211             drill_diam_iu = int( drill_um * (IU_PER_MM / 1000.0) );
 212 
 213             if( drill_diam_iu == aViaDrillDefault )
 214                 drill_diam_iu = UNDEFINED_DRILL_DIAMETER;
 215         }
 216     }
 217 
 218     if( shapeCount == 0 )
 219     {
 220         THROW_IO_ERROR( _( "Session via padstack has no shapes" ) );
 221     }
 222     else if( shapeCount == 1 )
 223     {
 224         shape = (SHAPE*) (*aPadstack)[0];
 225         DSN_T type = shape->shape->Type();
 226         if( type != T_circle )
 227             THROW_IO_ERROR( wxString::Format( _( "Unsupported via shape: %s"),
 228                                               GetChars( GetTokenString( type ) ) ) );
 229 
 230         CIRCLE* circle = (CIRCLE*) shape->shape;
 231         int viaDiam = scale( circle->diameter, routeResolution );
 232 
 233         via = new ::VIA( sessionBoard );
 234         via->SetPosition( mapPt( aPoint, routeResolution ) );
 235         via->SetDrill( drill_diam_iu );
 236         via->SetViaType( VIATYPE::THROUGH );
 237         via->SetWidth( viaDiam );
 238         via->SetLayerPair( F_Cu, B_Cu );
 239     }
 240     else if( shapeCount == copperLayerCount )
 241     {
 242         shape = (SHAPE*) (*aPadstack)[0];
 243         DSN_T type = shape->shape->Type();
 244         if( type != T_circle )
 245             THROW_IO_ERROR( wxString::Format( _( "Unsupported via shape: %s"),
 246                                               GetChars( GetTokenString( type ) ) ) );
 247 
 248         CIRCLE* circle = (CIRCLE*) shape->shape;
 249         int viaDiam = scale( circle->diameter, routeResolution );
 250 
 251         via = new ::VIA( sessionBoard );
 252         via->SetPosition( mapPt( aPoint, routeResolution ) );
 253         via->SetDrill( drill_diam_iu );
 254         via->SetViaType( VIATYPE::THROUGH );
 255         via->SetWidth( viaDiam );
 256         via->SetLayerPair( F_Cu, B_Cu );
 257     }
 258     else    // VIA_MICROVIA or VIA_BLIND_BURIED

 259     {
 260         int topLayerNdx = -1;           // session layer detectors

 261         int botLayerNdx = INT_MAX;
 262 
 263         int viaDiam = -1;
 264 
 265         for( int i=0; i<shapeCount;  ++i )
 266         {
 267             shape = (SHAPE*) (*aPadstack)[i];
 268             DSN_T type = shape->shape->Type();
 269             if( type != T_circle )
 270                 THROW_IO_ERROR( wxString::Format( _( "Unsupported via shape: %s"),
 271                                                   GetChars( GetTokenString( type ) ) ) );
 272 
 273             CIRCLE* circle = (CIRCLE*) shape->shape;
 274 
 275             int layerNdx = findLayerName( circle->layer_id );
 276             if( layerNdx == -1 )
 277             {
 278                 wxString layerName = FROM_UTF8( circle->layer_id.c_str() );
 279                 THROW_IO_ERROR( wxString::Format( _("Session file uses invalid layer id \"%s\""),
 280                                                   GetChars( layerName ) ) );
 281             }
 282 
 283             if( layerNdx > topLayerNdx )
 284                 topLayerNdx = layerNdx;
 285 
 286             if( layerNdx < botLayerNdx )
 287                 botLayerNdx = layerNdx;
 288 
 289             if( viaDiam == -1 )
 290                 viaDiam = scale( circle->diameter, routeResolution );
 291         }
 292 
 293         via = new ::VIA( sessionBoard );
 294         via->SetPosition( mapPt( aPoint, routeResolution ) );
 295         via->SetDrill( drill_diam_iu );
 296 
 297         if( (topLayerNdx==0 && botLayerNdx==1)
 298          || (topLayerNdx==copperLayerCount-2 && botLayerNdx==copperLayerCount-1))
 299             via->SetViaType( VIATYPE::MICROVIA );
 300         else
 301             via->SetViaType( VIATYPE::BLIND_BURIED );
 302 
 303         via->SetWidth( viaDiam );
 304 
 305         PCB_LAYER_ID topLayer = pcbLayer2kicad[topLayerNdx];
 306         PCB_LAYER_ID botLayer = pcbLayer2kicad[botLayerNdx];
 307 
 308         via->SetLayerPair( topLayer, botLayer );
 309     }
 310 
 311     wxASSERT( via );
 312 
 313     via->SetNetCode( aNetCode );
 314     return via;
 315 }

 321 void SPECCTRA_DB::FromSESSION( BOARD* aBoard )
 322 {
 323     sessionBoard = aBoard;      // not owned here

 324 
 325     if( !session )
 326         THROW_IO_ERROR( _("Session file is missing the \"session\" section") );
 327 
 328     if( !session->route )
 329         THROW_IO_ERROR( _("Session file is missing the \"routes\" section") );
 330 
 331     if( !session->route->library )
 332         THROW_IO_ERROR( _("Session file is missing the \"library_out\" section") );
 333 
 334     // delete all the old tracks and vias

 335     aBoard->Tracks().clear();
 336 
 337     aBoard->DeleteMARKERs();
 338 
 339     buildLayerMaps( aBoard );
 340 
 341     if( session->placement )
 342     {
 343         // Walk the PLACEMENT object's COMPONENTs list, and for each PLACE within

 344         // each COMPONENT, reposition and re-orient each component and put on

 345         // correct side of the board.

 346         COMPONENTS& components = session->placement->components;
 347         for( COMPONENTS::iterator comp=components.begin();  comp!=components.end();  ++comp )
 348         {
 349             PLACES& places = comp->places;
 350             for( unsigned i=0; i<places.size();  ++i )
 351             {
 352                 PLACE* place = &places[i];  // '&' even though places[] holds a pointer!

 353 
 354                 wxString reference = FROM_UTF8( place->component_id.c_str() );
 355                 MODULE* module = aBoard->FindModuleByReference( reference );
 356 
 357                 if( !module )
 358                 {
 359                     THROW_IO_ERROR( wxString::Format( _( "Reference '%s' not found." ),
 360                                                       reference ) );
 361                 }
 362 
 363                 if( !place->hasVertex )
 364                     continue;
 365 
 366                 UNIT_RES* resolution = place->GetUnits();
 367                 wxASSERT( resolution );
 368 
 369                 wxPoint newPos = mapPt( place->vertex, resolution );
 370                 module->SetPosition( newPos );
 371 
 372                 if( place->side == T_front )
 373                 {
 374                     // convert from degrees to tenths of degrees used in KiCad.

 375                     int orientation = KiROUND( place->rotation * 10.0 );
 376 
 377                     if( module->GetLayer() != F_Cu )
 378                     {
 379                         // module is on copper layer (back)

 380                         module->Flip( module->GetPosition(), false );
 381                     }
 382 
 383                     module->SetOrientation( orientation );
 384                 }
 385                 else if( place->side == T_back )
 386                 {
 387                     int orientation = KiROUND( (place->rotation + 180.0) * 10.0 );
 388 
 389                     if( module->GetLayer() != B_Cu )
 390                     {
 391                         // module is on component layer (front)

 392                         module->Flip( module->GetPosition(), false );
 393                     }
 394 
 395                     module->SetOrientation( orientation );
 396                 }
 397                 else
 398                 {
 399                     // as I write this, the PARSER *is* catching this, so we should never see below:

 400                     wxFAIL_MSG( wxT("DSN::PARSER did not catch an illegal side := 'back|front'") );
 401                 }
 402             }
 403         }
 404     }
 405 
 406     routeResolution = session->route->GetUnits();
 407 
 408     // Walk the NET_OUTs and create tracks and vias anew.

 409     NET_OUTS& net_outs = session->route->net_outs;
 410     for( NET_OUTS::iterator net = net_outs.begin(); net!=net_outs.end(); ++net )
 411     {
 412         int netoutCode = 0;
 413 
 414         // page 143 of spec says wire's net_id is optional

 415         if( net->net_id.size() )
 416         {
 417             wxString netName = FROM_UTF8( net->net_id.c_str() );
 418             NETINFO_ITEM* netinfo = aBoard->FindNet( netName );
 419 
 420             if( netinfo )
 421                 netoutCode = netinfo->GetNet();
 422             else  // else netCode remains 0

 423             {
 424                 // int breakhere = 1;

 425             }
 426         }
 427 
 428         WIRES& wires = net->wires;
 429         for( unsigned i = 0; i<wires.size(); ++i )
 430         {
 431             WIRE*   wire  = &wires[i];
 432             DSN_T   shape = wire->shape->Type();
 433 
 434             if( shape != T_path )
 435             {
 436                 /*  shape == T_polygon is expected from freerouter if you have

 437                     a zone on a non "power" type layer, i.e. a T_signal layer

 438                     and the design does a round trip back in as session here.

 439                     We kept our own zones in the BOARD, so ignore this so called

 440                     'wire'.

 441 

 442                 wxString netId = FROM_UTF8( wire->net_id.c_str() );

 443                 THROW_IO_ERROR( wxString::Format( _("Unsupported wire shape: \"%s\" for net: \"%s\""),

 444                                                     DLEX::GetTokenString(shape).GetData(),

 445                                                     netId.GetData()

 446                     ) );

 447                 */
 448             }
 449             else
 450             {
 451                 PATH*   path = (PATH*) wire->shape;
 452                 for( unsigned pt=0;  pt<path->points.size()-1;  ++pt )
 453                 {
 454                     TRACK* track = makeTRACK( path, pt, netoutCode );
 455                     aBoard->Add( track );
 456                 }
 457             }
 458         }
 459 
 460         WIRE_VIAS& wire_vias = net->wire_vias;
 461         LIBRARY& library = *session->route->library;
 462         for( unsigned i=0;  i<wire_vias.size();  ++i )
 463         {
 464             int         netCode = 0;
 465 
 466             // page 144 of spec says wire_via's net_id is optional

 467             if( net->net_id.size() )
 468             {
 469                 wxString netName = FROM_UTF8( net->net_id.c_str() );
 470                 NETINFO_ITEM* netvia = aBoard->FindNet( netName );
 471 
 472                 if( netvia )
 473                     netCode = netvia->GetNet();
 474 
 475                 // else netCode remains 0

 476             }
 477 
 478             WIRE_VIA* wire_via = &wire_vias[i];
 479 
 480             // example: (via Via_15:8_mil 149000 -71000 )

 481 
 482             PADSTACK* padstack = library.FindPADSTACK( wire_via->GetPadstackId() );
 483             if( !padstack )
 484             {
 485                 // Dick  Feb 29, 2008:

 486                 // Freerouter has a bug where it will not round trip all vias.

 487                 // Vias which have a (use_via) element will be round tripped.

 488                 // Vias which do not, don't come back in in the session library,

 489                 // even though they may be actually used in the pre-routed,

 490                 // protected wire_vias. So until that is fixed, create the

 491                 // padstack from its name as a work around.

 492 
 493 
 494                 // Could use a STRING_FORMATTER here and convert the entire

 495                 // wire_via to text and put that text into the exception.

 496                 wxString psid( FROM_UTF8( wire_via->GetPadstackId().c_str() ) );
 497 
 498                 THROW_IO_ERROR( wxString::Format( _("A wire_via references a missing padstack \"%s\""),
 499                                                   GetChars( psid ) ) );
 500             }
 501 
 502             NETCLASSPTR netclass = aBoard->GetDesignSettings().GetNetClasses().GetDefault();
 503 
 504             int via_drill_default = netclass->GetViaDrill();
 505 
 506             for( unsigned v=0;  v<wire_via->vertexes.size();  ++v )
 507             {
 508                 ::VIA* via = makeVIA( padstack, wire_via->vertexes[v], netCode, via_drill_default );
 509                 aBoard->Add( via );
 510             }
 511         }
 512     }
 513 }
