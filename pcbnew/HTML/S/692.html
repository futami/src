  22 #ifndef __PNS_SOLID_H
  23 #define __PNS_SOLID_H
  24 
  25 #include <math/vector2d.h>
  26 
  27 #include <geometry/seg.h>
  28 #include <geometry/shape.h>
  29 #include <geometry/shape_line_chain.h>
  30 
  31 #include "pns_item.h"
  32 
  33 namespace PNS {
  34 
  35 class SOLID : public ITEM
  36 {
  37 public:
  38     SOLID() : ITEM( SOLID_T ), m_shape( NULL )
  39     {
  40         m_movable = false;
  41         m_padToDie = 0;
  42     }
  43 
  44     ~SOLID()
  45     {
  46         delete m_shape;
  47     }
  48 
  49     SOLID( const SOLID& aSolid ) :
  50         ITEM( aSolid )
  51     {
  52         m_shape = aSolid.m_shape->Clone();
  53         m_pos = aSolid.m_pos;
  54         m_padToDie = aSolid.m_padToDie;
  55     }
  56 
  57     static inline bool ClassOf( const ITEM* aItem )
  58     {
  59         return aItem && SOLID_T == aItem->Kind();
  60     }
  61 
  62     ITEM* Clone() const override;
  63 
  64     const SHAPE* Shape() const override { return m_shape; }
  65 
  66     const SHAPE_LINE_CHAIN Hull( int aClearance = 0, int aWalkaroundThickness = 0 ) const override;
  67 
  68     void SetShape( SHAPE* shape )
  69     {
  70         if( m_shape )
  71             delete m_shape;
  72 
  73         m_shape = shape;
  74     }
  75 
  76     const VECTOR2I& Pos() const
  77     {
  78         return m_pos;
  79     }
  80 
  81     void SetPos( const VECTOR2I& aCenter );
  82     
  83     int GetPadToDie() const
  84     {
  85         return m_padToDie;
  86     }
  87 
  88     void SetPadToDie( int aLen )
  89     {
  90         m_padToDie = aLen;
  91     }
  92 
  93     virtual VECTOR2I Anchor( int aN ) const override
  94     {
  95         return m_pos;
  96     }
  97 
  98     virtual int AnchorCount() const override
  99     {
 100         return 1;
 101     }
 102 
 103     VECTOR2I Offset() const
 104     {
 105         return m_offset;
 106     }
 107 
 108     void SetOffset( const VECTOR2I& aOffset )
 109     {
 110         m_offset = aOffset;
 111     }
 112 
 113 private:
 114     VECTOR2I    m_pos;
 115     SHAPE*      m_shape;
 116     VECTOR2I    m_offset;
 117     int         m_padToDie;
 118 };
 119 
 120 }
