  39 void DRC::doTrackDrc( BOARD_COMMIT& aCommit, TRACK* aRefSeg, TRACKS::iterator aStartIt,
  40                       TRACKS::iterator aEndIt, bool aTestZones )
  41 {
  42     BOARD_DESIGN_SETTINGS&     bds = m_pcb->GetDesignSettings();
  43 
  44     SHAPE_SEGMENT refSeg( aRefSeg->GetStart(), aRefSeg->GetEnd(), aRefSeg->GetWidth() );
  45     PCB_LAYER_ID  refLayer = aRefSeg->GetLayer();
  46     LSET          refLayerSet = aRefSeg->GetLayerSet();
  47 
  48     EDA_RECT      refSegBB = aRefSeg->GetBoundingBox();
  49     int           refSegWidth = aRefSeg->GetWidth();
  50 
  51 
  52     /******************************************/
  53     /* Phase 0 : via DRC tests :              */
  54     /******************************************/
  55 
  56     if( aRefSeg->Type() == PCB_VIA_T )
  57     {
  58         VIA *refvia = static_cast<VIA*>( aRefSeg );
  59         int viaAnnulus = ( refvia->GetWidth() - refvia->GetDrill() ) / 2;
  60         int minAnnulus = refvia->GetMinAnnulus( &m_clearanceSource );
  61 
  62         // test if the via size is smaller than minimum

  63         if( refvia->GetViaType() == VIATYPE::MICROVIA )
  64         {
  65             if( viaAnnulus < minAnnulus )
  66             {
  67                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_VIA_ANNULUS );
  68 
  69                 m_msg.Printf( _( "Via annulus too small (%s %s; actual %s)" ),
  70                               m_clearanceSource,
  71                               MessageTextFromValue( userUnits(), minAnnulus, true ),
  72                               MessageTextFromValue( userUnits(), viaAnnulus, true ) );
  73 
  74                 drcItem->SetErrorMessage( m_msg );
  75                 drcItem->SetItems( refvia );
  76 
  77                 MARKER_PCB* marker = new MARKER_PCB( drcItem, refvia->GetPosition() );
  78                 addMarkerToPcb( aCommit, marker );
  79             }
  80 
  81             if( refvia->GetWidth() < bds.m_MicroViasMinSize )
  82             {
  83                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_TOO_SMALL_MICROVIA );
  84 
  85                 m_msg.Printf( drcItem->GetErrorText() + _( " (board minimum %s; actual %s)" ),
  86                               MessageTextFromValue( userUnits(), bds.m_MicroViasMinSize, true ),
  87                               MessageTextFromValue( userUnits(), refvia->GetWidth(), true ) );
  88 
  89                 drcItem->SetErrorMessage( m_msg );
  90                 drcItem->SetItems( refvia );
  91 
  92                 MARKER_PCB* marker = new MARKER_PCB( drcItem, refvia->GetPosition() );
  93                 addMarkerToPcb( aCommit, marker );
  94             }
  95         }
  96         else
  97         {
  98             if( bds.m_ViasMinAnnulus > minAnnulus )
  99             {
 100                 minAnnulus = bds.m_ViasMinAnnulus;
 101                 m_clearanceSource = _( "board minimum" );
 102             }
 103 
 104             if( viaAnnulus < minAnnulus )
 105             {
 106                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_VIA_ANNULUS );
 107 
 108                 m_msg.Printf( _( "Via annulus too small (%s %s; actual %s)" ),
 109                               m_clearanceSource,
 110                               MessageTextFromValue( userUnits(), minAnnulus, true ),
 111                               MessageTextFromValue( userUnits(), viaAnnulus, true ) );
 112 
 113                 drcItem->SetErrorMessage( m_msg );
 114                 drcItem->SetItems( refvia );
 115 
 116                 MARKER_PCB* marker = new MARKER_PCB( drcItem, refvia->GetPosition() );
 117                 addMarkerToPcb( aCommit, marker );
 118             }
 119 
 120             if( refvia->GetWidth() < bds.m_ViasMinSize )
 121             {
 122                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_TOO_SMALL_VIA );
 123 
 124                 m_msg.Printf( drcItem->GetErrorText() + _( " (board minimum %s; actual %s)" ),
 125                               MessageTextFromValue( userUnits(), bds.m_ViasMinSize, true ),
 126                               MessageTextFromValue( userUnits(), refvia->GetWidth(), true ) );
 127 
 128                 drcItem->SetErrorMessage( m_msg );
 129                 drcItem->SetItems( refvia );
 130 
 131                 MARKER_PCB* marker = new MARKER_PCB( drcItem, refvia->GetPosition() );
 132                 addMarkerToPcb( aCommit, marker );
 133             }
 134         }
 135 
 136         // test if via's hole is bigger than its diameter

 137         // This test is necessary since the via hole size and width can be modified

 138         // and a default via hole can be bigger than some vias sizes

 139         if( refvia->GetDrillValue() > refvia->GetWidth() )
 140         {
 141             DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_VIA_HOLE_BIGGER );
 142 
 143             m_msg.Printf( drcItem->GetErrorText() + _( " (diameter %s; drill %s)" ),
 144                           MessageTextFromValue( userUnits(), refvia->GetWidth(), true ),
 145                           MessageTextFromValue( userUnits(), refvia->GetDrillValue(), true ) );
 146 
 147             drcItem->SetErrorMessage( m_msg );
 148             drcItem->SetItems( refvia );
 149 
 150             MARKER_PCB* marker = new MARKER_PCB( drcItem, refvia->GetPosition() );
 151             addMarkerToPcb( aCommit, marker );
 152         }
 153 
 154         // test if the type of via is allowed due to design rules

 155         if( refvia->GetViaType() == VIATYPE::MICROVIA && !bds.m_MicroViasAllowed )
 156         {
 157             DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_ALLOWED_ITEMS );
 158 
 159             m_msg.Printf( _( "Microvia not allowed (board design rule constraints)" ) );
 160             drcItem->SetErrorMessage( m_msg );
 161             drcItem->SetItems( refvia );
 162 
 163             MARKER_PCB* marker = new MARKER_PCB( drcItem, refvia->GetPosition() );
 164             addMarkerToPcb( aCommit, marker );
 165         }
 166 
 167         // test if the type of via is allowed due to design rules

 168         if( refvia->GetViaType() == VIATYPE::BLIND_BURIED && !bds.m_BlindBuriedViaAllowed )
 169         {
 170             DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_ALLOWED_ITEMS );
 171 
 172             m_msg.Printf( _( "Blind/buried via not allowed (board design rule constraints)" ) );
 173             drcItem->SetErrorMessage( m_msg );
 174             drcItem->SetItems( refvia );
 175 
 176             MARKER_PCB* marker = new MARKER_PCB( drcItem, refvia->GetPosition() );
 177             addMarkerToPcb( aCommit, marker );
 178         }
 179 
 180         // For microvias: test if they are blind vias and only between 2 layers

 181         // because they are used for very small drill size and are drill by laser

 182         // and **only one layer** can be drilled

 183         if( refvia->GetViaType() == VIATYPE::MICROVIA )
 184         {
 185             PCB_LAYER_ID layer1, layer2;
 186             bool         err = true;
 187 
 188             refvia->LayerPair( &layer1, &layer2 );
 189 
 190             if( layer1 > layer2 )
 191                 std::swap( layer1, layer2 );
 192 
 193             if( layer2 == B_Cu && layer1 == bds.GetCopperLayerCount() - 2 )
 194                 err = false;
 195             else if( layer1 == F_Cu  &&  layer2 == In1_Cu  )
 196                 err = false;
 197 
 198             if( err )
 199             {
 200                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_PADSTACK );
 201 
 202                 m_msg.Printf( _( "Microvia through too many layers (%s and %s not adjacent)" ),
 203                               m_pcb->GetLayerName( layer1 ),
 204                               m_pcb->GetLayerName( layer2 ) );
 205 
 206                 drcItem->SetErrorMessage( m_msg );
 207                 drcItem->SetItems( refvia );
 208 
 209                 MARKER_PCB* marker = new MARKER_PCB( drcItem, refvia->GetPosition() );
 210                 addMarkerToPcb( aCommit, marker );
 211             }
 212         }
 213 
 214     }
 215     else    // This is a track segment

 216     {
 217         int minWidth, maxWidth;
 218         aRefSeg->GetWidthConstraints( &minWidth, &maxWidth, &m_clearanceSource );
 219 
 220         int errorCode = 0;
 221         int constraintWidth;
 222 
 223         if( refSegWidth < minWidth )
 224         {
 225             errorCode = DRCE_TRACK_WIDTH;
 226             constraintWidth = minWidth;
 227         }
 228         else if( refSegWidth > maxWidth )
 229         {
 230             errorCode = DRCE_TRACK_WIDTH;
 231             constraintWidth = maxWidth;
 232         }
 233 
 234         if( errorCode )
 235         {
 236             wxPoint refsegMiddle = ( aRefSeg->GetStart() + aRefSeg->GetEnd() ) / 2;
 237 
 238             DRC_ITEM* drcItem = DRC_ITEM::Create( errorCode );
 239 
 240             m_msg.Printf( drcItem->GetErrorText() + _( " (%s %s; actual %s)" ),
 241                           m_clearanceSource,
 242                           MessageTextFromValue( userUnits(), constraintWidth, true ),
 243                           MessageTextFromValue( userUnits(), refSegWidth, true ) );
 244 
 245             drcItem->SetErrorMessage( m_msg );
 246             drcItem->SetItems( aRefSeg );
 247 
 248             MARKER_PCB* marker = new MARKER_PCB( drcItem, refsegMiddle );
 249             addMarkerToPcb( aCommit, marker );
 250         }
 251     }
 252 
 253 
 254     /******************************************/
 255     /* Phase 1 : test DRC track to pads :     */
 256     /******************************************/
 257 
 258     // Compute the min distance to pads

 259     for( MODULE* mod : m_pcb->Modules() )
 260     {
 261         // Don't preflight at the module level.  Getting a module's bounding box goes

 262         // through all its pads anyway (so it's no faster), and also all its drawings

 263         // (so it's in fact slower).

 264 
 265         for( D_PAD* pad : mod->Pads() )
 266         {
 267             // Preflight based on bounding boxes.

 268             EDA_RECT inflatedBB = refSegBB;
 269             inflatedBB.Inflate( pad->GetBoundingRadius() + m_largestClearance );
 270 
 271             if( !inflatedBB.Contains( pad->GetPosition() ) )
 272                 continue;
 273 
 274             if( !( pad->GetLayerSet() & refLayerSet ).any() )
 275                 continue;
 276 
 277             // No need to check pads with the same net as the refSeg.

 278             if( pad->GetNetCode() && aRefSeg->GetNetCode() == pad->GetNetCode() )
 279                 continue;
 280 
 281             if( pad->GetDrillSize().x > 0 )
 282             {
 283                 const SHAPE_SEGMENT* slot = pad->GetEffectiveHoleShape();
 284                 DRC_RULE*            rule = GetRule( aRefSeg, pad, CLEARANCE_CONSTRAINT );
 285 
 286                 int minClearance;
 287                 int actual;
 288 
 289                 if( rule )
 290                 {
 291                     m_clearanceSource = wxString::Format( _( "'%s' rule" ), rule->m_Name );
 292                     minClearance = rule->m_Clearance.Min;
 293                 }
 294                 else
 295                 {
 296                     minClearance = aRefSeg->GetClearance( nullptr, &m_clearanceSource );
 297                 }
 298 
 299                 if( slot->Collide( &refSeg, minClearance + bds.GetDRCEpsilon(), &actual ) )
 300                 {
 301                     DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
 302 
 303                     m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
 304                                   m_clearanceSource,
 305                                   MessageTextFromValue( userUnits(), minClearance, true ),
 306                                   MessageTextFromValue( userUnits(), actual, true ) );
 307 
 308                     drcItem->SetErrorMessage( m_msg );
 309                     drcItem->SetItems( aRefSeg, pad );
 310 
 311                     MARKER_PCB* marker = new MARKER_PCB( drcItem, pad->GetPosition() );
 312                     addMarkerToPcb( aCommit, marker );
 313 
 314                     if( !m_reportAllTrackErrors )
 315                         return;
 316                 }
 317             }
 318 
 319             int minClearance = aRefSeg->GetClearance( pad, &m_clearanceSource );
 320             int actual;
 321 
 322             if( pad->Collide( &refSeg, minClearance - bds.GetDRCEpsilon(), &actual ) )
 323             {
 324                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
 325 
 326                 m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
 327                               m_clearanceSource,
 328                               MessageTextFromValue( userUnits(), minClearance, true ),
 329                               MessageTextFromValue( userUnits(), actual, true ) );
 330 
 331                 drcItem->SetErrorMessage( m_msg );
 332                 drcItem->SetItems( aRefSeg, pad );
 333 
 334                 MARKER_PCB* marker = new MARKER_PCB( drcItem, pad->GetPosition() );
 335                 addMarkerToPcb( aCommit, marker );
 336 
 337                 if( !m_reportAllTrackErrors )
 338                     return;
 339             }
 340         }
 341     }
 342 
 343     /***********************************************/
 344     /* Phase 2: test DRC with other track segments */
 345     /***********************************************/
 346 
 347     // Test the reference segment with other track segments

 348     for( auto it = aStartIt; it != aEndIt; it++ )
 349     {
 350         TRACK* track = *it;
 351 
 352         // No problem if segments have the same net code:

 353         if( aRefSeg->GetNetCode() == track->GetNetCode() )
 354             continue;
 355 
 356         // No problem if tracks are on different layers:

 357         // Note that while the general case of GetLayerSet intersection always works,

 358         // the others are much faster.

 359         bool sameLayers;
 360 
 361         if( aRefSeg->Type() == PCB_VIA_T )
 362         {
 363             if( track->Type() == PCB_VIA_T )
 364                 sameLayers = ( refLayerSet & track->GetLayerSet() ).any();
 365             else
 366                 sameLayers = refLayerSet.test( track->GetLayer() );
 367         }
 368         else
 369         {
 370             if( track->Type() == PCB_VIA_T )
 371                 sameLayers = track->GetLayerSet().test( refLayer );
 372             else
 373                 sameLayers = track->GetLayer() == refLayer;
 374         }
 375 
 376         if( !sameLayers )
 377             continue;
 378 
 379         // Preflight based on worst-case inflated bounding boxes:

 380         EDA_RECT trackBB = track->GetBoundingBox();
 381         trackBB.Inflate( m_largestClearance );
 382 
 383         if( !trackBB.Intersects( refSegBB ) )
 384             continue;
 385 
 386         int           minClearance = aRefSeg->GetClearance( track, &m_clearanceSource );
 387         int           actual;
 388         SHAPE_SEGMENT trackSeg( track->GetStart(), track->GetEnd(), track->GetWidth() );
 389 
 390         // Check two tracks crossing first as it reports a DRCE without distances

 391         if( OPT_VECTOR2I intersection = refSeg.GetSeg().Intersect( trackSeg.GetSeg() ) )
 392         {
 393             DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_TRACKS_CROSSING );
 394             drcItem->SetErrorMessage( m_msg );
 395             drcItem->SetItems( aRefSeg, track );
 396 
 397             MARKER_PCB* marker = new MARKER_PCB( drcItem, (wxPoint) intersection.get() );
 398             addMarkerToPcb( aCommit, marker );
 399 
 400             if( !m_reportAllTrackErrors )
 401                 return;
 402         }
 403         else if( refSeg.Collide( &trackSeg, minClearance, &actual ) )
 404         {
 405             wxPoint   pos = GetLocation( aRefSeg, trackSeg.GetSeg() );
 406             DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
 407 
 408             m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
 409                           m_clearanceSource,
 410                           MessageTextFromValue( userUnits(), minClearance, true ),
 411                           MessageTextFromValue( userUnits(), actual, true ) );
 412 
 413             drcItem->SetErrorMessage( m_msg );
 414             drcItem->SetItems( aRefSeg, track );
 415 
 416             MARKER_PCB* marker = new MARKER_PCB( drcItem, pos );
 417             addMarkerToPcb( aCommit, marker );
 418 
 419             if( !m_reportAllTrackErrors )
 420                 return;
 421         }
 422     }
 423 
 424     /***************************************/
 425     /* Phase 3: test DRC with copper zones */
 426     /***************************************/
 427     // Can be *very* time consumming.

 428     if( aTestZones )
 429     {
 430         SEG testSeg( aRefSeg->GetStart(), aRefSeg->GetEnd() );
 431 
 432         for( ZONE_CONTAINER* zone : m_pcb->Zones() )
 433         {
 434             if( !( refLayerSet & zone->GetLayerSet() ).any() || zone->GetIsKeepout() )
 435                 continue;
 436 
 437             for( PCB_LAYER_ID layer : zone->GetLayerSet().Seq() )
 438             {
 439                 if( zone->GetFilledPolysList( layer ).IsEmpty() )
 440                     continue;
 441 
 442                 if( zone->GetNetCode() && zone->GetNetCode() == aRefSeg->GetNetCode() )
 443                     continue;
 444 
 445                 // to avoid false positive, due to rounding issues and approxiamtions

 446                 // in distance and clearance calculations, use a small threshold for distance

 447                 // (1 micron)

 448                 #define THRESHOLD_DIST Millimeter2iu( 0.001 )
 449 
 450                 int minClearance = aRefSeg->GetClearance( zone, &m_clearanceSource );
 451                 int widths       = refSegWidth / 2;
 452                 int allowedDist  = minClearance + widths + THRESHOLD_DIST;
 453                 int actual;
 454 
 455                 if( zone->GetFilledPolysList( layer ).Collide( testSeg, allowedDist, &actual ) )
 456                 {
 457                     actual = std::max( 0, actual - widths );
 458                     DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
 459 
 460                     m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
 461                                   m_clearanceSource,
 462                                   MessageTextFromValue( userUnits(), minClearance, true ),
 463                                   MessageTextFromValue( userUnits(), actual, true ) );
 464 
 465                     drcItem->SetErrorMessage( m_msg );
 466                     drcItem->SetItems( aRefSeg, zone );
 467 
 468                     MARKER_PCB* marker = new MARKER_PCB( drcItem, GetLocation( aRefSeg, zone ) );
 469                     addMarkerToPcb( aCommit, marker );
 470                 }
 471             }
 472         }
 473     }
 474 
 475     /***********************************************/
 476     /* Phase 4: test DRC with to board edge        */
 477     /***********************************************/
 478     if( m_board_outline_valid )
 479     {
 480         int minClearance = bds.m_CopperEdgeClearance;
 481         m_clearanceSource = _( "board edge" );
 482 
 483         static DRAWSEGMENT dummyEdge;
 484         dummyEdge.SetLayer( Edge_Cuts );
 485 
 486         if( aRefSeg->GetRuleClearance( &dummyEdge, &minClearance, &m_clearanceSource ) )
 487         {
 488             /* minClearance and m_clearanceSource set in GetRuleClearance() */;
 489         }
 490 
 491         SEG testSeg( aRefSeg->GetStart(), aRefSeg->GetEnd() );
 492         int halfWidth = refSegWidth / 2;
 493         int center2centerAllowed = minClearance + halfWidth;
 494 
 495         for( auto it = m_board_outlines.IterateSegmentsWithHoles(); it; it++ )
 496         {
 497             SEG::ecoord center2center_squared = testSeg.SquaredDistance(  *it );
 498 
 499             if( center2center_squared < SEG::Square( center2centerAllowed ) )
 500             {
 501                 VECTOR2I pt = testSeg.NearestPoint( *it );
 502 
 503                 KICAD_T        types[] = { PCB_LINE_T, EOT };
 504                 DRAWSEGMENT*   edge = nullptr;
 505                 INSPECTOR_FUNC inspector =
 506                         [&] ( EDA_ITEM* item, void* testData )
 507                         {
 508                             DRAWSEGMENT* test_edge = dynamic_cast<DRAWSEGMENT*>( item );
 509 
 510                             if( !test_edge || test_edge->GetLayer() != Edge_Cuts )
 511                                 return SEARCH_RESULT::CONTINUE;
 512 
 513                             if( test_edge->HitTest( (wxPoint) pt, minClearance + halfWidth ) )
 514                             {
 515                                 edge = test_edge;
 516                                 return SEARCH_RESULT::QUIT;
 517                             }
 518 
 519                             return SEARCH_RESULT::CONTINUE;
 520                         };
 521 
 522                 // Best-efforts search for edge segment

 523                 BOARD::IterateForward<BOARD_ITEM*>( m_pcb->Drawings(), inspector, nullptr, types );
 524 
 525                 int       actual  = std::max( 0.0, sqrt( center2center_squared ) - halfWidth );
 526                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_COPPER_EDGE_CLEARANCE );
 527 
 528                 m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
 529                               m_clearanceSource,
 530                               MessageTextFromValue( userUnits(), minClearance, true ),
 531                               MessageTextFromValue( userUnits(), actual, true ) );
 532 
 533                 drcItem->SetErrorMessage( m_msg );
 534                 drcItem->SetItems( aRefSeg, edge );
 535 
 536                 MARKER_PCB* marker = new MARKER_PCB( drcItem, (wxPoint) pt );
 537                 addMarkerToPcb( aCommit, marker );
 538             }
 539         }
 540     }
