  26 #ifndef GRAPHICS_IMPORT_PLUGIN_H
  27 #define GRAPHICS_IMPORT_PLUGIN_H
  28 
  29 #include <wildcards_and_files_ext.h>
  30 #include <wx/arrstr.h>
  31 
  32 class GRAPHICS_IMPORTER;
  33 
  34 /**

  35  * Interface for vector graphics import plugins.

  36  */
  37 class GRAPHICS_IMPORT_PLUGIN
  38 {
  39 public:
  40     virtual ~GRAPHICS_IMPORT_PLUGIN()
  41     {
  42     }
  43 
  44     /**

  45      * Set the receiver of the imported shapes.

  46      */
  47     virtual void SetImporter( GRAPHICS_IMPORTER* aImporter )
  48     {
  49         m_importer = aImporter;
  50     }
  51 
  52     /**

  53      * Return the plugin name.

  54      *

  55      * This string will be used as the description in the file dialog.

  56      */
  57     virtual const wxString GetName() const = 0;
  58 
  59     /**

  60      * Return a vector of the file extensions handled by this plugin.

  61      */
  62     virtual const std::vector<std::string> GetFileExtensions() const = 0;
  63 
  64     /**

  65      * Return a list of wildcards that contains the file extensions

  66      * handled by this plugin, separated with a semi-colon.

  67      */
  68     wxString GetWildcards() const
  69     {
  70         wxString ret;
  71         bool first = true;
  72 
  73         for( const auto& extension : GetFileExtensions() )
  74         {
  75             if( first )
  76                 first = false;
  77             else
  78                 ret += ";";
  79 
  80             ret += "*." + formatWildcardExt( extension );
  81         }
  82 
  83         return ret;
  84     }
  85 
  86     /**

  87      * Load file for import.

  88      *

  89      * It is necessary to have the GRAPHICS_IMPORTER object set before.

  90      */
  91     virtual bool Load( const wxString& aFileName ) = 0;
  92 
  93     /**

  94      * Return image height from original imported file.

  95      *

  96      * @return Original Image height in mm.

  97      */
  98     virtual double GetImageHeight() const = 0;
  99 
 100     /**

 101      * Return image width from original imported file.

 102      *

 103      * @return Original Image width in mm.

 104      */
 105     virtual double GetImageWidth() const = 0;
 106 
 107     /**

 108      * Actually imports the file.

 109      *

 110      * It is necessary to have loaded the file beforehand.

 111      */
 112     virtual bool Import() = 0;
 113 
 114     virtual void SetLineWidthMM( double aLineWidth ) {}
 115 
 116     /**

 117      * Collect warning and error messages after loading/importing.

 118      *

 119      * @return the list of messages in one string. Each message ends by '\n'

 120      */
 121     const virtual std::string& GetMessages() const = 0;
 122 
 123 protected:
 124     ///> Importer used to create objects representing the imported shapes.

 125     GRAPHICS_IMPORTER* m_importer;
 126 };
 127 
 128 
 129 #endif /* GRAPHICS_IMPORT_PLUGIN_H */
