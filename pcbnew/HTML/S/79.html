  25 /**

  26  * @file class_track.h

  27  * @brief A single base class (TRACK) represents both tracks and vias, with subclasses

  28  * for curved tracks (ARC) and vias (VIA).  All told there are three KICAD_Ts:

  29  * PCB_TRACK_T, PCB_ARC_T, and PCB_VIA_T.

  30  *

  31  * For vias there is a further VIATYPE which indicates THROUGH, BLIND_BURIED, or

  32  * MICROVIA, which are supported by the synthetic KICAD_Ts PCB_LOCATE_STDVIA_T,

  33  * PCB_LOCATE_BBVIA_T, and PCB_LOCATE_UVIA_T.

  34  */

  66 // Via types

  67 // Note that this enum must be synchronized to GAL_LAYER_ID

  68 enum class VIATYPE : int
  69 {
  70     THROUGH      = 3, /* Always a through hole via */
  71     BLIND_BURIED = 2, /* this via can be on internal layers */
  72     MICROVIA     = 1, /* this via which connect from an external layer

  73                                 * to the near neighbor internal layer */
  74     NOT_DEFINED  = 0  /* not yet used */
  75 };

 348 class VIA : public TRACK

 350 public:
 351     VIA( BOARD_ITEM* aParent );

 353     static inline bool ClassOf( const EDA_ITEM *aItem )
 354     {
 355         return aItem && PCB_VIA_T == aItem->Type();
 356     }
 357 
 358     // Do not create a copy constructor.  The one generated by the compiler is adequate.

 359 
 360     bool IsType( const KICAD_T aScanTypes[] ) const override
 361     {
 362         if( BOARD_CONNECTED_ITEM::IsType( aScanTypes ) )
 363             return true;
 364 
 365         for( const KICAD_T* p = aScanTypes; *p != EOT; ++p )
 366         {
 367             if( *p == PCB_LOCATE_STDVIA_T && m_ViaType == VIATYPE::THROUGH )
 368                 return true;
 369             else if( *p == PCB_LOCATE_UVIA_T && m_ViaType == VIATYPE::MICROVIA )
 370                 return true;
 371             else if( *p == PCB_LOCATE_BBVIA_T && m_ViaType == VIATYPE::BLIND_BURIED )
 372                 return true;
 373         }
 374 
 375         return false;
 376     }
 377
