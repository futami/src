  52 bool PCB_COPPER_POUR::Parse( XNODE*          aNode,
  53                              const wxString& aDefaultMeasurementUnit,
  54                              const wxString& aActualConversion )
  55 {
  56     XNODE*          lNode;
  57     wxString        pourType, str, propValue;
  58     int             pourSpacing, thermalWidth;
  59 
  60     lNode = FindNode( aNode, wxT( "netNameRef" ) );
  61 
  62     if( lNode )
  63     {
  64         lNode->GetAttribute( wxT( "Name" ), &propValue );
  65         propValue.Trim( false );
  66         propValue.Trim( true );
  67         m_net = propValue;
  68         m_netCode = GetNetCode( m_net );
  69     }
  70 
  71     if( FindNode( aNode, wxT( "width" ) ) )
  72         SetWidth( FindNode( aNode, wxT( "width" ) )->GetNodeContent(),
  73                   aDefaultMeasurementUnit, &m_width, aActualConversion );
  74 
  75     if( FindNode( aNode, wxT( "pourSpacing" ) ) )
  76         SetWidth( FindNode( aNode, wxT( "pourSpacing" ) )->GetNodeContent(),
  77                   aDefaultMeasurementUnit, &pourSpacing, aActualConversion );
  78 
  79     if( FindNode( aNode, wxT( "thermalWidth" ) ) )
  80         SetWidth( FindNode( aNode, wxT( "thermalWidth" ) )->GetNodeContent(),
  81                   aDefaultMeasurementUnit, &thermalWidth, aActualConversion );
  82 
  83     if( FindNode( aNode, wxT( "island" ) ) )
  84         m_filled = true;
  85 
  86     lNode = FindNode( aNode, wxT( "pcbPoly" ) );
  87 
  88     if( lNode )
  89     {
  90         // retrieve copper pour outline

  91         FormPolygon( lNode, &m_outline, aDefaultMeasurementUnit, aActualConversion );
  92 
  93         m_positionX = m_outline[0]->x;
  94         m_positionY = m_outline[0]->y;
  95     }
  96     else
  97     {
  98         return false;
  99     }
 100 
 101     return true;
 102 }
