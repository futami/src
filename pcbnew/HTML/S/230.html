  86 void DIALOG_MOVE_EXACT::buildRotationAnchorMenu()
  87 {
  88     wxArrayString menuItems;
  89 
  90     for( auto anchorID : m_menuIDs )
  91     {
  92         switch( anchorID )
  93         {
  94         case ROTATE_AROUND_ITEM_ANCHOR:
  95             menuItems.push_back( _( "Rotate around item anchor" ) );
  96             break;
  97         case ROTATE_AROUND_SEL_CENTER:
  98             menuItems.push_back( _( "Rotate around selection center" ) );
  99             break;
 100         case ROTATE_AROUND_USER_ORIGIN:
 101             menuItems.push_back( _( "Rotate around local coordinates origin" ) );
 102             break;
 103         case ROTATE_AROUND_AUX_ORIGIN:
 104             menuItems.push_back( _( "Rotate around drill/place origin" ) );
 105             break;
 106         }
 107     }
 108 
 109     m_anchorOptions->Set( menuItems );
 110 }

 113 void DIALOG_MOVE_EXACT::ToPolarDeg( double x, double y, double& r, double& q )
 114 {
 115     // convert to polar coordinates

 116     r = hypot( x, y );
 117 
 118     q = ( r != 0) ? RAD2DEG( atan2( y, x ) ) : 0;
 119 }

 122 bool DIALOG_MOVE_EXACT::GetTranslationInIU( wxRealPoint& val, bool polar )
 123 {
 124     if( polar )
 125     {
 126         const double r = m_moveX.GetDoubleValue();
 127         const double q = m_moveY.GetDoubleValue();
 128 
 129         val.x = r * cos( DEG2RAD( q / 10.0 ) );
 130         val.y = r * sin( DEG2RAD( q / 10.0 ) );
 131     }
 132     else
 133     {
 134         // direct read

 135         val.x = m_moveX.GetDoubleValue();
 136         val.y = m_moveY.GetDoubleValue();
 137     }
 138 
 139     // no validation to do here, but in future, you could return false here

 140     return true;
 141 }

 144 void DIALOG_MOVE_EXACT::OnPolarChanged( wxCommandEvent& event )
 145 {
 146     bool newPolar = m_polarCoords->IsChecked();
 147     double moveX = m_moveX.GetDoubleValue();
 148     double moveY = m_moveY.GetDoubleValue();
 149     updateDialogControls( newPolar );
 150 
 151     if( newPolar )
 152     {
 153         if( moveX != m_stateX || moveY != m_stateY )
 154         {
 155             m_stateX = moveX;
 156             m_stateY = moveY;
 157             ToPolarDeg( m_stateX, m_stateY, m_stateRadius, m_stateTheta );
 158             m_stateTheta *= 10.0;
 159 
 160             m_moveX.SetDoubleValue( m_stateRadius );
 161             m_stateRadius = m_moveX.GetDoubleValue();
 162             m_moveY.SetDoubleValue( m_stateTheta );
 163             m_stateTheta = m_moveY.GetDoubleValue();
 164         }
 165         else
 166         {
 167             m_moveX.SetDoubleValue( m_stateRadius );
 168             m_moveY.SetDoubleValue( m_stateTheta );
 169         }
 170     }
 171     else
 172     {
 173         if( moveX != m_stateRadius || moveY != m_stateTheta )
 174         {
 175             m_stateRadius = moveX;
 176             m_stateTheta = moveY;
 177             m_stateX = m_stateRadius * cos( DEG2RAD( m_stateTheta / 10.0 ) );
 178             m_stateY = m_stateRadius * sin( DEG2RAD( m_stateTheta / 10.0 ) );
 179 
 180             m_moveX.SetDoubleValue( m_stateX );
 181             m_stateX = m_moveX.GetDoubleValue();
 182             m_moveY.SetDoubleValue( m_stateY );
 183             m_stateY = m_moveY.GetDoubleValue();
 184         }
 185         else
 186         {
 187             m_moveX.SetDoubleValue( m_stateX );
 188             m_moveY.SetDoubleValue( m_stateY );
 189         }
 190     }
 191 }

 235 bool DIALOG_MOVE_EXACT::TransferDataFromWindow()
 236 {
 237     // for the output, we only deliver a Cartesian vector

 238     wxRealPoint translation;
 239     bool ok = GetTranslationInIU( translation, m_polarCoords->IsChecked() );
 240     m_translation.x = KiROUND(translation.x);
 241     m_translation.y = KiROUND(translation.y);
 242     m_rotation = m_rotate.GetValue();
 243     m_rotationAnchor = m_menuIDs[ m_anchorOptions->GetSelection() ];
 244 
 245     if( ok )
 246     {
 247         // save the settings

 248         m_options.polarCoords = m_polarCoords->GetValue();
 249         m_options.entry1 = m_moveX.GetOriginalText();
 250         m_options.entry2 = m_moveY.GetOriginalText();
 251         m_options.entryRotation = m_rotate.GetOriginalText();
 252         m_options.entryAnchorSelection = (size_t) std::max( m_anchorOptions->GetSelection(), 0 );
 253         return true;
 254     }
 255 
 256     return false;
 257 }

 271 void DIALOG_MOVE_EXACT::OnTextChanged( wxCommandEvent& event )
 272 {
 273 
 274     double delta_x = m_moveX.GetDoubleValue();
 275     double delta_y = m_moveY.GetDoubleValue();
 276     double max_border = std::numeric_limits<int>::max() * 0.7071;
 277 
 278     if( m_bbox.GetLeft() + delta_x < -max_border ||
 279             m_bbox.GetRight() + delta_x > max_border ||
 280             m_bbox.GetTop() + delta_y < -max_border ||
 281             m_bbox.GetBottom() + delta_y > max_border )
 282     {
 283         const wxString invalid_length = _( "Invalid movement values.  Movement would place selection "
 284                                            "outside of the maximum board area." );
 285 
 286         m_xEntry->SetToolTip( invalid_length );
 287         m_xEntry->SetForegroundColour( *wxRED );
 288         m_yEntry->SetToolTip( invalid_length );
 289         m_yEntry->SetForegroundColour( *wxRED );
 290         m_stdButtons->GetAffirmativeButton()->Disable();
 291     }
 292     else
 293     {
 294         m_xEntry->SetToolTip( "" );
 295         m_xEntry->SetForegroundColour( wxSystemSettings::GetColour( wxSYS_COLOUR_WINDOWTEXT ) );
 296         m_yEntry->SetToolTip( "" );
 297         m_yEntry->SetForegroundColour( wxSystemSettings::GetColour( wxSYS_COLOUR_WINDOWTEXT ) );
 298         m_stdButtons->GetAffirmativeButton()->Enable();
 299         event.Skip();
 300     }
