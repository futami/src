  36 class BOARD;
  37 class PLOTTER;

  40 /**

  41  * EXCELLON_WRITER is a class mainly used to create Excellon drill files

  42  * However, this class is also used to create drill maps and drill report

  43  */
  44 class EXCELLON_WRITER: public GENDRILL_WRITER_BASE
  45 {
  46 private:
  47     FILE*                    m_file;                // The output file

  48     bool                     m_minimalHeader;       // True to use minimal header

  49     bool                     m_mirror;
  50     bool                     m_useRouteModeForOval; // True to use a route command for oval holes

  51                                                     // False to use a G85 canned mode for oval holes

  52 
  53 
  54 public:
  55     EXCELLON_WRITER( BOARD* aPcb );
  56 
  57     virtual ~EXCELLON_WRITER()
  58     {
  59     }
  60 
  61     /**

  62      * Return the plot offset (usually the position

  63      * of the auxiliary axis

  64      */
  65     wxPoint GetOffset() { return m_offset; }
  66 
  67     /**

  68      *

  69      */
  70     void SetRouteModeForOvalHoles( bool aUseRouteModeForOvalHoles )
  71     {
  72         m_useRouteModeForOval = aUseRouteModeForOvalHoles;
  73     }
  74 
  75     /**

  76      * Function SetFormat

  77      * Initialize internal parameters to match the given format

  78      * @param aMetric = true for metric coordinates, false for imperial units

  79      * @param aZerosFmt =  DECIMAL_FORMAT, SUPPRESS_LEADING, SUPPRESS_TRAILING, KEEP_ZEROS

  80      * @param aLeftDigits = number of digits for integer part of coordinates

  81      *          if <= 0 (default), a suitable value will be used, depending on units

  82      * @param aRightDigits = number of digits for mantissa part of coordinates

  83      *          if <= 0 (default), a suitable value will be used, depending on units

  84      */
  85     void SetFormat( bool aMetric, ZEROS_FMT aZerosFmt = DECIMAL_FORMAT,
  86                     int aLeftDigits = 0, int aRightDigits = 0 );
  87 
  88 
  89 
  90     /**

  91      * Function SetOptions

  92      * Initialize internal parameters to match drill options

  93      * @param aMirror = true to create mirrored coordinates (Y coordinates negated)

  94      * @param aMinimalHeader = true to use a minimal header (no comments, no info)

  95      * @param aOffset = drill coordinates offset

  96      * @param aMerge_PTH_NPTH = true to create only one file containing PTH and NPTH

  97      * false to create 2 separate files : one for PTH and one for NPTH

  98      */
  99     void SetOptions( bool aMirror, bool aMinimalHeader, wxPoint aOffset, bool aMerge_PTH_NPTH )
 100     {
 101         m_mirror = aMirror;
 102         m_offset = aOffset;
 103         m_minimalHeader = aMinimalHeader;
 104         m_merge_PTH_NPTH = aMerge_PTH_NPTH;
 105     }
 106 
 107     /**

 108      * Function CreateDrillandMapFilesSet

 109      * Creates the full set of Excellon drill file for the board

 110      * filenames are computed from the board name, and layers id

 111      * @param aPlotDirectory = the output folder

 112      * @param aGenDrill = true to generate the EXCELLON drill file

 113      * @param aGenMap = true to generate a drill map file

 114      * @param aReporter = a REPORTER to return activity or any message (can be NULL)

 115      */
 116     void CreateDrillandMapFilesSet( const wxString& aPlotDirectory,
 117                                     bool aGenDrill, bool aGenMap,
 118                                     REPORTER * aReporter = NULL );
 119 
 120 
 121 private:
 122     /**

 123      * Function CreateDrillFile

 124      * Creates an Excellon drill file

 125      * @param aFile = an opened file to write to will be closed by CreateDrillFile

 126      * @return hole count

 127      */
 128     int  createDrillFile( FILE * aFile, DRILL_LAYER_PAIR aLayerPair,
 129                           bool aGenerateNPTH_list );
 130 
 131 
 132     /* Print the DRILL file header. The full header is somethink like:

 133      * M48

 134      * ;DRILL file {PCBNEW (2007-11-29-b)} date 17/1/2008-21:02:35

 135      * ;FORMAT={ <precision> / absolute / <units> / <numbers format>}

 136      * ; #@! TF.FileFunction,Plated,1,4,PTH

 137      * ; #@! TF.CreationDate,2018-11-23T15:59:51+01:00

 138      * ; #@! TF.GenerationSoftware,Kicad,Pcbnew,2017.04

 139      * FMAT,2

 140      * INCH,TZ

 141      */
 142     void writeEXCELLONHeader( DRILL_LAYER_PAIR aLayerPair,
 143                               bool aGenerateNPTH_list);
 144 
 145     void writeEXCELLONEndOfFile();
 146 
 147     /* Created a line like:

 148      * X48000Y19500

 149      * According to the selected format

 150      */
 151     void writeCoordinates( char* aLine, double aCoordX, double aCoordY );
 152 };
