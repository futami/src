 197 DRC_ITEM* DRC_ITEM::Create( int aErrorCode )
 198 {
 199     switch( aErrorCode )
 200     {
 201     case DRCE_UNCONNECTED_ITEMS:        return new DRC_ITEM( unconnectedItems );
 202     case DRCE_ALLOWED_ITEMS:            return new DRC_ITEM( itemsNotAllowed );
 203     case DRCE_CLEARANCE:                return new DRC_ITEM( clearance );
 204     case DRCE_TRACKS_CROSSING:          return new DRC_ITEM( tracksCrossing );
 205     case DRCE_COPPER_EDGE_CLEARANCE:    return new DRC_ITEM( copperEdgeClearance );
 206     case DRCE_ZONES_INTERSECT:          return new DRC_ITEM( zonesIntersect );
 207     case DRCE_ZONE_HAS_EMPTY_NET:       return new DRC_ITEM( zoneHasEmptyNet );
 208     case DRCE_DANGLING_VIA:             return new DRC_ITEM( viaDangling );
 209     case DRCE_DANGLING_TRACK:           return new DRC_ITEM( trackDangling );
 210     case DRCE_DRILLED_HOLES_TOO_CLOSE:  return new DRC_ITEM( holeNearHole );
 211     case DRCE_TRACK_WIDTH:              return new DRC_ITEM( trackWidth );
 212     case DRCE_TOO_SMALL_VIA:            return new DRC_ITEM( viaTooSmall );
 213     case DRCE_VIA_ANNULUS:              return new DRC_ITEM( viaAnnulus );
 214     case DRCE_TOO_SMALL_DRILL:          return new DRC_ITEM( drillTooSmall );
 215     case DRCE_VIA_HOLE_BIGGER:          return new DRC_ITEM( viaHoleLargerThanPad );
 216     case DRCE_PADSTACK:                 return new DRC_ITEM( padstack );
 217     case DRCE_TOO_SMALL_MICROVIA:       return new DRC_ITEM( microviaTooSmall );
 218     case DRCE_TOO_SMALL_MICROVIA_DRILL: return new DRC_ITEM( microviaDrillTooSmall );
 219     case DRCE_KEEPOUT:                  return new DRC_ITEM( keepout );
 220     case DRCE_OVERLAPPING_FOOTPRINTS:   return new DRC_ITEM( courtyardsOverlap );
 221     case DRCE_MISSING_COURTYARD:        return new DRC_ITEM( missingCourtyard );
 222     case DRCE_MALFORMED_COURTYARD:      return new DRC_ITEM( malformedCourtyard );
 223     case DRCE_PTH_IN_COURTYARD:         return new DRC_ITEM( pthInsideCourtyard );
 224     case DRCE_NPTH_IN_COURTYARD:        return new DRC_ITEM( npthInsideCourtyard );
 225     case DRCE_DISABLED_LAYER_ITEM:      return new DRC_ITEM( itemOnDisabledLayer );
 226     case DRCE_INVALID_OUTLINE:          return new DRC_ITEM( invalidOutline );
 227     case DRCE_MISSING_FOOTPRINT:        return new DRC_ITEM( duplicateFootprints );
 228     case DRCE_DUPLICATE_FOOTPRINT:      return new DRC_ITEM( missingFootprint );
 229     case DRCE_EXTRA_FOOTPRINT:          return new DRC_ITEM( extraFootprint );
 230     case DRCE_UNRESOLVED_VARIABLE:      return new DRC_ITEM( unresolvedVariable );
 231 
 232     default:
 233         wxFAIL_MSG( "Unknown DRC error code" );
 234         return nullptr;
 235     }
 236 }

 239 DRC_ITEM* DRC_ITEM::Create( const wxString& aErrorKey )
 240 {
 241     for( const RC_ITEM& item : allItemTypes )
 242     {
 243         if( aErrorKey == item.GetSettingsKey() )
 244             return new DRC_ITEM( static_cast<const DRC_ITEM&>( item ) );
 245     }
 246 
 247     // This can happen if a project has old-format exclusions.  Just drop these items.

 248     return nullptr;
 249 }

 252 wxString escapeHtml( wxString aString )
 253 {
 254     aString.Replace( wxT("<"), wxT("&lt;") );
 255     aString.Replace( wxT(">"), wxT("&gt;") );
 256     return aString;
 257 }

 260 wxString DRC_ITEM::ShowHtml( PCB_BASE_FRAME* aFrame ) const
 261 {
 262     BOARD_ITEM* mainItem = nullptr;
 263     BOARD_ITEM* auxItem = nullptr;
 264     wxString    msg = m_errorMessage.IsEmpty() ? GetErrorText() : m_errorMessage;
 265     wxString    mainText;
 266     wxString    auxText;
 267 
 268     if( m_mainItemUuid != niluuid )
 269         mainItem = aFrame->GetBoard()->GetItem( m_mainItemUuid );
 270 
 271     if( m_auxItemUuid != niluuid )
 272         auxItem = aFrame->GetBoard()->GetItem( m_auxItemUuid );
 273 
 274     if( mainItem )
 275         mainText = mainItem->GetSelectMenuText( aFrame->GetUserUnits() );
 276 
 277     if( auxItem )
 278         auxText = auxItem->GetSelectMenuText( aFrame->GetUserUnits() );
 279 
 280     if( mainItem && auxItem )
 281     {
 282         // an html fragment for the entire message in the listbox.  feel free

 283         // to add color if you want:

 284         return wxString::Format( wxT( "<b>%s</b><br>&nbsp;&nbsp; %s<br>&nbsp;&nbsp; %s" ),
 285                                  escapeHtml( msg ),
 286                                  escapeHtml( mainText ),
 287                                  escapeHtml( auxText ) );
 288     }
 289     else if( mainItem )
 290     {
 291         return wxString::Format( wxT( "<b>%s</b><br>&nbsp;&nbsp; %s" ),
 292                                  escapeHtml( msg ),
 293                                  escapeHtml( mainText ) );
 294     }
 295     else
 296     {
 297         return wxString::Format( wxT( "<b>%s</b>" ),
 298                                  escapeHtml( msg ) );
 299     }
 300 }
