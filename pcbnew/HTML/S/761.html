  36 void FOOTPRINT_VIEWER_FRAME::ReCreateHToolbar()
  37 {
  38     // Note:

  39     // To rebuild the aui toolbar, the more easy way is to clear ( calling m_mainToolBar.Clear() )

  40     // all wxAuiToolBarItems.

  41     // However the wxAuiToolBarItems are not the owners of controls managed by

  42     // them ( m_zoomSelectBox and m_gridSelectBox ), and therefore do not delete them

  43     // So we do not recreate them after clearing the tools.

  44 
  45     if( m_mainToolBar )
  46         m_mainToolBar->ClearToolbar();
  47     else
  48         m_mainToolBar = new ACTION_TOOLBAR( this, wxID_ANY, wxDefaultPosition, wxDefaultSize,
  49                                             KICAD_AUI_TB_STYLE | wxAUI_TB_HORZ_LAYOUT  );
  50 
  51     // Set up toolbar

  52     m_mainToolBar->AddTool( ID_MODVIEW_PREVIOUS, wxEmptyString,
  53                             KiScaledBitmap( lib_previous_xpm, this ),
  54                             _( "Display previous footprint" ) );
  55     m_mainToolBar->AddTool( ID_MODVIEW_NEXT, wxEmptyString,
  56                             KiScaledBitmap( lib_next_xpm, this ),
  57                             _( "Display next footprint" ) );
  58 
  59     m_mainToolBar->AddScaledSeparator( this );
  60     m_mainToolBar->Add( ACTIONS::zoomRedraw );
  61     m_mainToolBar->Add( ACTIONS::zoomInCenter );
  62     m_mainToolBar->Add( ACTIONS::zoomOutCenter );
  63     m_mainToolBar->Add( ACTIONS::zoomFitScreen );
  64     m_mainToolBar->Add( ACTIONS::zoomTool,                       ACTION_TOOLBAR::TOGGLE );
  65     m_mainToolBar->Add( PCB_ACTIONS::zoomFootprintAutomatically, ACTION_TOOLBAR::TOGGLE );
  66 
  67     m_mainToolBar->AddScaledSeparator( this );
  68     m_mainToolBar->Add( ACTIONS::show3DViewer );
  69     m_mainToolBar->AddTool( ID_ADD_FOOTPRINT_TO_BOARD, wxEmptyString,
  70                             KiScaledBitmap( export_xpm, this ),
  71                             _( "Insert footprint in board" ) );
  72 
  73     m_mainToolBar->AddScaledSeparator( this );
  74 
  75     // Grid selection choice box.

  76     if( m_gridSelectBox == nullptr )
  77         m_gridSelectBox = new wxChoice( m_mainToolBar, ID_ON_GRID_SELECT,
  78                                     wxDefaultPosition, wxDefaultSize, 0, NULL );
  79 
  80     UpdateGridSelectBox();
  81     m_mainToolBar->AddControl( m_gridSelectBox );
  82 
  83     m_mainToolBar->AddScaledSeparator( this );
  84 
  85     // Zoom selection choice box.

  86     if( m_zoomSelectBox == nullptr )
  87         m_zoomSelectBox = new wxChoice( m_mainToolBar, ID_ON_ZOOM_SELECT,
  88                                     wxDefaultPosition, wxDefaultSize, 0, NULL );
  89 
  90     UpdateZoomSelectBox();
  91     m_mainToolBar->AddControl( m_zoomSelectBox );
  92 
  93     // after adding the buttons to the toolbar, must call Realize() to

  94     // reflect the changes

  95     m_mainToolBar->Realize();
  96 }

  99 void FOOTPRINT_VIEWER_FRAME::ReCreateOptToolbar()
 100 {
 101     if( m_optionsToolBar )
 102         m_optionsToolBar->ClearToolbar();
 103     else
 104         m_optionsToolBar = new ACTION_TOOLBAR( this, ID_OPT_TOOLBAR, wxDefaultPosition, wxDefaultSize,

 107     m_optionsToolBar->Add( ACTIONS::selectionTool,          ACTION_TOOLBAR::TOGGLE );

 111     m_optionsToolBar->Add( ACTIONS::toggleGrid,             ACTION_TOOLBAR::TOGGLE );
 112     m_optionsToolBar->Add( ACTIONS::togglePolarCoords,      ACTION_TOOLBAR::TOGGLE );
 113     m_optionsToolBar->Add( ACTIONS::imperialUnits,          ACTION_TOOLBAR::TOGGLE );
 114     m_optionsToolBar->Add( ACTIONS::metricUnits,            ACTION_TOOLBAR::TOGGLE );
 115     m_optionsToolBar->Add( ACTIONS::toggleCursorStyle,      ACTION_TOOLBAR::TOGGLE );
 116 
 117     m_optionsToolBar->AddScaledSeparator( this );
 118     m_optionsToolBar->Add( PCB_ACTIONS::showPadNumbers,     ACTION_TOOLBAR::TOGGLE );
 119     m_optionsToolBar->Add( PCB_ACTIONS::padDisplayMode,     ACTION_TOOLBAR::TOGGLE );
 120     m_optionsToolBar->Add( PCB_ACTIONS::textOutlines,       ACTION_TOOLBAR::TOGGLE );
 121     m_optionsToolBar->Add( PCB_ACTIONS::graphicsOutlines,   ACTION_TOOLBAR::TOGGLE );
 122 
 123     m_optionsToolBar->Realize();
 124 }

 133 void FOOTPRINT_VIEWER_FRAME::SyncToolbars()
 134 {
 135     m_mainToolBar->Toggle( ACTIONS::zoomTool, IsCurrentTool( ACTIONS::zoomTool ) );
 136     m_mainToolBar->Toggle( PCB_ACTIONS::zoomFootprintAutomatically, GetAutoZoom() );
 137     m_mainToolBar->Refresh();
 138 
 139     m_optionsToolBar->Toggle( ACTIONS::toggleGrid,    IsGridVisible() );
 140     m_optionsToolBar->Toggle( ACTIONS::selectionTool, IsCurrentTool( ACTIONS::selectionTool ) );
 141     m_optionsToolBar->Toggle( ACTIONS::measureTool,   IsCurrentTool( ACTIONS::measureTool ) );
 142     m_optionsToolBar->Toggle( ACTIONS::metricUnits,   GetUserUnits() != EDA_UNITS::INCHES );
 143     m_optionsToolBar->Toggle( ACTIONS::imperialUnits, GetUserUnits() == EDA_UNITS::INCHES );
 144 
 145     const PCB_DISPLAY_OPTIONS& opts = GetDisplayOptions();
 146 
 147     m_optionsToolBar->Toggle( PCB_ACTIONS::showPadNumbers,     opts.m_DisplayPadNum );
 148     m_optionsToolBar->Toggle( PCB_ACTIONS::padDisplayMode,     !opts.m_DisplayPadFill );
 149     m_optionsToolBar->Toggle( PCB_ACTIONS::textOutlines,       !opts.m_DisplayTextFill );
 150     m_optionsToolBar->Toggle( PCB_ACTIONS::graphicsOutlines,   !opts.m_DisplayGraphicsFill );
 151 
 152     m_optionsToolBar->Refresh();
 153 }

 156 void FOOTPRINT_VIEWER_FRAME::ReCreateMenuBar()
 157 {
 158     SELECTION_TOOL* selTool = m_toolManager->GetTool<SELECTION_TOOL>();
 159     // wxWidgets handles the Mac Application menu behind the scenes, but that means

 160     // we always have to start from scratch with a new wxMenuBar.

 161     wxMenuBar*  oldMenuBar = GetMenuBar();
 162     WX_MENUBAR* menuBar    = new WX_MENUBAR();
 163 
 164     //----- File menu -----------------------------------------------------------

 165     //

 166     CONDITIONAL_MENU* fileMenu = new CONDITIONAL_MENU( false, selTool );
 167 
 168     fileMenu->AddClose( _( "Footprint Viewer" ) );
 169 
 170     fileMenu->Resolve();
 171 
 172     //----- View menu -----------------------------------------------------------

 173     //

 174     CONDITIONAL_MENU* viewMenu = new CONDITIONAL_MENU( false, selTool );
 175 
 176     viewMenu->AddSeparator();
 177     viewMenu->AddItem( ACTIONS::zoomInCenter,     SELECTION_CONDITIONS::ShowAlways );
 178     viewMenu->AddItem( ACTIONS::zoomOutCenter,    SELECTION_CONDITIONS::ShowAlways );
 179     viewMenu->AddItem( ACTIONS::zoomFitScreen,    SELECTION_CONDITIONS::ShowAlways );
 180     viewMenu->AddItem( ACTIONS::zoomRedraw,       SELECTION_CONDITIONS::ShowAlways );
 181 
 182     viewMenu->AddSeparator();
 183     viewMenu->AddItem( ACTIONS::show3DViewer,     SELECTION_CONDITIONS::ShowAlways );
 184 
 185     viewMenu->Resolve();
 186 
 187     //----- Menubar -------------------------------------------------------------

 188     //

 189     menuBar->Append( fileMenu, _( "&File" ) );
 190     menuBar->Append( viewMenu, _( "&View" ) );
 191     AddStandardHelpMenu( menuBar );
 192 
 193     SetMenuBar( menuBar );
 194     delete oldMenuBar;
 195 }
