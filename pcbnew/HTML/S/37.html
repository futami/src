  27 #ifndef BOARD_ITEM_CONTAINER_H
  28 #define BOARD_ITEM_CONTAINER_H
  29 
  30 #include <class_board_item.h>
  31 #include <zone_settings.h>
  32 
  33 enum class ADD_MODE
  34 {
  35     INSERT,
  36     APPEND
  37 };
  38 
  39 /**

  40  * @brief Abstract interface for BOARD_ITEMs capable of storing other items inside.

  41  * @see MODULE

  42  * @see BOARD

  43  */
  44 class BOARD_ITEM_CONTAINER : public BOARD_ITEM
  45 {
  46 public:
  47     BOARD_ITEM_CONTAINER( BOARD_ITEM* aParent, KICAD_T aType )
  48         : BOARD_ITEM( aParent, aType )
  49     {
  50     }
  51 
  52     /**

  53      * @brief Adds an item to the container.

  54      * @param aMode decides whether the item is added in the beginning or at the end of the list.

  55      */
  56     virtual void Add( BOARD_ITEM* aItem, ADD_MODE aMode = ADD_MODE::INSERT ) = 0;
  57 
  58     /**

  59      * @brief Removes an item from the container.

  60      */
  61     virtual void Remove( BOARD_ITEM* aItem ) = 0;
  62 
  63     /**

  64      * @brief Removes an item from the container and deletes it.

  65      */
  66     virtual void Delete( BOARD_ITEM* aItem )
  67     {
  68         Remove( aItem );
  69         delete aItem;
  70     }
  71 
  72     /**

  73      * @brief Fetch the zone settings for this container

  74      */
  75     virtual const ZONE_SETTINGS& GetZoneSettings() const
  76     {
  77         return m_zoneSettings;
  78     }
  79 
  80     /**

  81      * @brief Set the zone settings for this container

  82      * @param aSettings new Zone settings for this container

  83      */
  84     virtual void SetZoneSettings( const ZONE_SETTINGS& aSettings )
  85     {
  86         m_zoneSettings = aSettings;
  87     }
  88 
  89 private:
  90     ZONE_SETTINGS m_zoneSettings;
  91 };
  92 
  93 #endif /* BOARD_ITEM_CONTAINER_H */
