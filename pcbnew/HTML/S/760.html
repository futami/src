  35 void FOOTPRINT_EDIT_FRAME::ReCreateHToolbar()
  36 {
  37     // Note:

  38     // To rebuild the aui toolbar, the more easy way is to clear ( calling m_mainToolBar.Clear() )

  39     // all wxAuiToolBarItems.

  40     // However the wxAuiToolBarItems are not the owners of controls managed by

  41     // them ( m_zoomSelectBox and m_gridSelectBox ), and therefore do not delete them

  42     // So we do not recreate them after clearing the tools.

  43 
  44     if( m_mainToolBar )
  45         m_mainToolBar->ClearToolbar();
  46     else
  47         m_mainToolBar = new ACTION_TOOLBAR( this, ID_H_TOOLBAR, wxDefaultPosition, wxDefaultSize,
  48                                             KICAD_AUI_TB_STYLE | wxAUI_TB_HORZ_LAYOUT );
  49 
  50     wxString msg;
  51 
  52     // Set up toolbar

  53     m_mainToolBar->Add( PCB_ACTIONS::newFootprint );
  54 #ifdef KICAD_SCRIPTING
  55     m_mainToolBar->Add( PCB_ACTIONS::createFootprint );
  56 #endif
  57 
  58     if( IsCurrentFPFromBoard() )
  59         m_mainToolBar->Add( PCB_ACTIONS::saveToBoard );
  60     else
  61         m_mainToolBar->Add( PCB_ACTIONS::saveToLibrary );
  62 
  63     m_mainToolBar->AddScaledSeparator( this );
  64     m_mainToolBar->Add( ACTIONS::print );
  65 
  66     m_mainToolBar->AddScaledSeparator( this );
  67     m_mainToolBar->Add( ACTIONS::undo );
  68     m_mainToolBar->Add( ACTIONS::redo );
  69 
  70     m_mainToolBar->AddScaledSeparator( this );
  71     m_mainToolBar->Add( ACTIONS::zoomRedraw );
  72     m_mainToolBar->Add( ACTIONS::zoomInCenter );
  73     m_mainToolBar->Add( ACTIONS::zoomOutCenter );
  74     m_mainToolBar->Add( ACTIONS::zoomFitScreen );
  75     m_mainToolBar->Add( ACTIONS::zoomTool, ACTION_TOOLBAR::TOGGLE );
  76 
  77     m_mainToolBar->AddScaledSeparator( this );
  78     m_mainToolBar->Add( PCB_ACTIONS::footprintProperties );
  79     m_mainToolBar->Add( PCB_ACTIONS::defaultPadProperties );
  80 
  81     m_mainToolBar->AddScaledSeparator( this );
  82     m_mainToolBar->AddTool( ID_MODEDIT_LOAD_MODULE_FROM_BOARD, wxEmptyString,
  83                             KiScaledBitmap( load_module_board_xpm, this ),
  84                             _( "Load footprint from current board" ) );
  85 
  86     m_mainToolBar->AddTool( ID_ADD_FOOTPRINT_TO_BOARD, wxEmptyString,
  87                             KiScaledBitmap( export_xpm, this ),
  88                             _( "Insert footprint into current board" ) );
  89 
  90 #if 0       // Currently there is no check footprint function defined, so do not show this tool

  91     m_mainToolBar->AddScaledSeparator( this );
  92     m_mainToolBar->AddTool( ID_MODEDIT_CHECK, wxEmptyString,
  93                             KiScaledBitmap( module_check_xpm, this ),
  94                             _( "Check footprint" ) );
  95 #endif
  96 
  97     m_mainToolBar->AddScaledSeparator( this );
  98 
  99     // Grid selection choice box.

 100     if( m_gridSelectBox == nullptr )
 101         m_gridSelectBox = new wxChoice( m_mainToolBar, ID_ON_GRID_SELECT,
 102                                     wxDefaultPosition, wxDefaultSize, 0, NULL );
 103 
 104     UpdateGridSelectBox();
 105     m_mainToolBar->AddControl( m_gridSelectBox );
 106 
 107     m_mainToolBar->AddScaledSeparator( this );
 108 
 109     // Zoom selection choice box.

 110     if( m_zoomSelectBox == nullptr )
 111         m_zoomSelectBox = new wxChoice( m_mainToolBar, ID_ON_ZOOM_SELECT,
 112                                     wxDefaultPosition, wxDefaultSize, 0, NULL );
 113 
 114     UpdateZoomSelectBox();
 115     m_mainToolBar->AddControl( m_zoomSelectBox );
 116 
 117     m_mainToolBar->AddScaledSeparator( this );
 118 
 119     // Layer selection choice box.

 120     if( m_selLayerBox == nullptr )
 121     {
 122         m_selLayerBox = new PCB_LAYER_BOX_SELECTOR( m_mainToolBar, ID_TOOLBARH_PCB_SELECT_LAYER );
 123         m_selLayerBox->SetBoardFrame( this );
 124 
 125         // Some layers cannot be seclect (they are shown in the layer manager

 126         // only to set the color and visibility, but not for selection)

 127         // Disable them in layer box

 128         m_selLayerBox->SetNotAllowedLayerSet( LSET::ForbiddenFootprintLayers() );
 129         m_selLayerBox->Resync();
 130     }
 131 
 132     ReCreateLayerBox( false );
 133     m_mainToolBar->AddControl( m_selLayerBox );
 134 
 135     // after adding the buttons to the toolbar, must call Realize() to reflect the changes

 136     m_mainToolBar->Realize();
 137 }

 197 void FOOTPRINT_EDIT_FRAME::ReCreateLayerBox( bool aForceResizeToolbar )
 198 {
 199     if( m_selLayerBox == NULL || m_mainToolBar == NULL )
 200         return;
 201 
 202     m_selLayerBox->SetToolTip( _( "+/- to switch" ) );
 203     m_selLayerBox->Resync();
 204 
 205     if( aForceResizeToolbar )
 206     {
 207         // the layer box can have its size changed

 208         // Update the aui manager, to take in account the new size

 209         m_auimgr.Update();
 210     }
 211 }

 220 void FOOTPRINT_EDIT_FRAME::SyncToolbars()
 221 {
 222 #define TOGGLE_TOOL( toolbar, tool ) toolbar->Toggle( tool, true, IsCurrentTool( tool ) )
 223 
 224     if( !m_mainToolBar || !m_optionsToolBar || !m_drawToolBar )
 225         return;
 226 
 227     auto& opts = GetDisplayOptions();
 228 
 229     if( IsCurrentFPFromBoard() )
 230         m_mainToolBar->Toggle( PCB_ACTIONS::saveToBoard,   IsContentModified() );
 231     else
 232         m_mainToolBar->Toggle( PCB_ACTIONS::saveToLibrary, IsContentModified() );
 233 
 234     m_mainToolBar->Toggle( ACTIONS::undo, GetUndoCommandCount() > 0 );
 235     m_mainToolBar->Toggle( ACTIONS::redo, GetRedoCommandCount() > 0 );
 236     TOGGLE_TOOL( m_mainToolBar, ACTIONS::zoomTool );
 237     m_mainToolBar->Toggle( PCB_ACTIONS::footprintProperties, GetBoard()->GetFirstModule() );
 238     m_mainToolBar->Refresh();
 239 
 240     bool hcm = opts.m_ContrastModeDisplay != HIGH_CONTRAST_MODE::NORMAL;
 241 
 242     m_optionsToolBar->Toggle( ACTIONS::toggleGrid,              IsGridVisible() );
 243     m_optionsToolBar->Toggle( ACTIONS::metricUnits,         GetUserUnits() != EDA_UNITS::INCHES );
 244     m_optionsToolBar->Toggle( ACTIONS::imperialUnits,       GetUserUnits() == EDA_UNITS::INCHES );
 245     m_optionsToolBar->Toggle( ACTIONS::togglePolarCoords,       GetShowPolarCoords() );
 246     m_optionsToolBar->Toggle( PCB_ACTIONS::padDisplayMode,      !opts.m_DisplayPadFill );
 247     m_optionsToolBar->Toggle( PCB_ACTIONS::textOutlines,        !opts.m_DisplayTextFill );
 248     m_optionsToolBar->Toggle( PCB_ACTIONS::graphicsOutlines,    !opts.m_DisplayGraphicsFill );
 249     m_optionsToolBar->Toggle( ACTIONS::highContrastMode,        hcm );
 250     m_optionsToolBar->Toggle( PCB_ACTIONS::toggleFootprintTree, IsSearchTreeShown() );
 251     m_optionsToolBar->Refresh();
 252 
 253     if( GetLoadedFPID().empty() )
 254     {
 255         // If there is no footprint loaded, disable the editing tools

 256         m_drawToolBar->Toggle( PCB_ACTIONS::placePad,        false, false );
 257         m_drawToolBar->Toggle( PCB_ACTIONS::drawLine,        false, false );
 258         m_drawToolBar->Toggle( PCB_ACTIONS::drawRectangle,   false, false );
 259         m_drawToolBar->Toggle( PCB_ACTIONS::drawCircle,      false, false );
 260         m_drawToolBar->Toggle( PCB_ACTIONS::drawArc,         false, false );
 261         m_drawToolBar->Toggle( PCB_ACTIONS::drawPolygon,     false, false );
 262         m_drawToolBar->Toggle( PCB_ACTIONS::drawZoneKeepout, false, false );
 263         m_drawToolBar->Toggle( PCB_ACTIONS::placeText,       false, false );
 264         m_drawToolBar->Toggle( ACTIONS::deleteTool,          false, false );
 265         m_drawToolBar->Toggle( PCB_ACTIONS::setAnchor,       false, false );
 266         m_drawToolBar->Toggle( PCB_ACTIONS::gridSetOrigin,   false, false );
 267         m_drawToolBar->Toggle( ACTIONS::measureTool,         false, false );
 268     }
 269     else
 270     {
 271         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::placePad );
 272         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::drawLine );
 273         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::drawRectangle );
 274         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::drawCircle );
 275         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::drawArc );
 276         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::drawPolygon );
 277         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::drawZoneKeepout );
 278         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::placeText );
 279         TOGGLE_TOOL( m_drawToolBar, ACTIONS::deleteTool );
 280         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::setAnchor );
 281         TOGGLE_TOOL( m_drawToolBar, PCB_ACTIONS::gridSetOrigin );
 282         TOGGLE_TOOL( m_drawToolBar, ACTIONS::measureTool );
 283     }
 284 
 285     TOGGLE_TOOL( m_drawToolBar, ACTIONS::selectionTool );
 286     m_drawToolBar->Refresh();
 287 }
