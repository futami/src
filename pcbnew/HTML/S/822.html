 102 bool ZONE_FILLER::Fill( const std::vector<ZONE_CONTAINER*>& aZones, bool aCheck )
 103 {
 104     std::vector<std::pair<ZONE_CONTAINER*, PCB_LAYER_ID>> toFill;
 105     std::vector<CN_ZONE_ISOLATED_ISLAND_LIST> islandsList;
 106 
 107     auto connectivity = m_board->GetConnectivity();
 108     bool filledPolyWithOutline = not m_board->GetDesignSettings().m_ZoneUseNoOutlineInFill;
 109 
 110     std::unique_lock<std::mutex> lock( connectivity->GetLock(), std::try_to_lock );
 111 
 112     if( !lock )
 113         return false;
 114 
 115     if( m_progressReporter )
 116     {
 117         m_progressReporter->Report( aCheck ? _( "Checking zone fills..." ) : _( "Building zone fills..." ) );
 118         m_progressReporter->SetMaxProgress( aZones.size() );
 119     }
 120 
 121     // The board outlines is used to clip solid areas inside the board (when outlines are valid)

 122     m_boardOutline.RemoveAllContours();
 123     m_brdOutlinesValid = m_board->GetBoardPolygonOutlines( m_boardOutline );
 124 
 125     // Update the bounding box shape caches in the pads to prevent multi-threaded rebuilds

 126     for( auto module : m_board->Modules() )
 127     {
 128         for( auto pad : module->Pads() )
 129         {
 130             if( pad->IsDirty() )
 131                 pad->BuildEffectiveShapes();
 132         }
 133     }
 134 
 135     for( auto zone : aZones )
 136     {
 137         // Keepout zones are not filled

 138         if( zone->GetIsKeepout() )
 139             continue;
 140 
 141         if( m_commit )
 142             m_commit->Modify( zone );
 143 
 144         // calculate the hash value for filled areas. it will be used later

 145         // to know if the current filled areas are up to date

 146         for( PCB_LAYER_ID layer : zone->GetLayerSet().Seq() )
 147         {
 148             zone->BuildHashValue( layer );
 149 
 150             // Add the zone to the list of zones to test or refill

 151             toFill.emplace_back( std::make_pair( zone, layer ) );
 152         }
 153 
 154         islandsList.emplace_back( CN_ZONE_ISOLATED_ISLAND_LIST( zone ) );
 155 
 156         // Remove existing fill first to prevent drawing invalid polygons

 157         // on some platforms

 158         zone->UnFill();
 159     }
 160 
 161     std::atomic<size_t> nextItem( 0 );
 162     size_t              parallelThreadCount =
 163             std::min<size_t>( std::thread::hardware_concurrency(), aZones.size() );
 164     std::vector<std::future<size_t>> returns( parallelThreadCount );
 165 
 166     auto fill_lambda = [&] ( PROGRESS_REPORTER* aReporter ) -> size_t
 167     {
 168         size_t num = 0;
 169 
 170         for( size_t i = nextItem++; i < toFill.size(); i = nextItem++ )
 171         {
 172             PCB_LAYER_ID    layer = toFill[i].second;
 173             ZONE_CONTAINER* zone  = toFill[i].first;
 174 
 175             zone->SetFilledPolysUseThickness( filledPolyWithOutline );
 176 
 177             SHAPE_POLY_SET rawPolys, finalPolys;
 178             fillSingleZone( zone, layer, rawPolys, finalPolys );
 179 
 180             std::unique_lock<std::mutex> zoneLock( zone->GetLock() );
 181 
 182             zone->SetRawPolysList( layer, rawPolys );
 183             zone->SetFilledPolysList( layer, finalPolys );
 184             zone->SetIsFilled( true );
 185 
 186             if( m_progressReporter )
 187                 m_progressReporter->AdvanceProgress();
 188 
 189             num++;
 190         }
 191 
 192         return num;
 193     };
 194 
 195     if( parallelThreadCount <= 1 )
 196         fill_lambda( m_progressReporter );
 197     else
 198     {
 199         for( size_t ii = 0; ii < parallelThreadCount; ++ii )
 200             returns[ii] = std::async( std::launch::async, fill_lambda, m_progressReporter );
 201 
 202         for( size_t ii = 0; ii < parallelThreadCount; ++ii )
 203         {
 204             // Here we balance returns with a 100ms timeout to allow UI updating

 205             std::future_status status;
 206             do
 207             {
 208                 if( m_progressReporter )
 209                     m_progressReporter->KeepRefreshing();
 210 
 211                 status = returns[ii].wait_for( std::chrono::milliseconds( 100 ) );
 212             } while( status != std::future_status::ready );
 213         }
 214     }
 215 
 216     // Now update the connectivity to check for copper islands

 217     if( m_progressReporter )
 218     {
 219         m_progressReporter->AdvancePhase();
 220         m_progressReporter->Report( _( "Removing insulated copper islands..." ) );
 221         m_progressReporter->KeepRefreshing();
 222     }
 223 
 224     connectivity->SetProgressReporter( m_progressReporter );
 225     connectivity->FindIsolatedCopperIslands( islandsList );
 226 
 227     // Now remove insulated copper islands and islands outside the board edge

 228     bool outOfDate = false;
 229 
 230     for( auto& zone : islandsList )
 231     {
 232         for( PCB_LAYER_ID layer : zone.m_zone->GetLayerSet().Seq() )
 233         {
 234             if( !zone.m_islands.count( layer ) )
 235                 continue;
 236 
 237             std::vector<int>& islands = zone.m_islands.at( layer );
 238 
 239             std::sort( islands.begin(), islands.end(), std::greater<int>() );
 240             SHAPE_POLY_SET poly = zone.m_zone->GetFilledPolysList( layer );
 241 
 242             long long int       minArea = zone.m_zone->GetMinIslandArea();
 243             ISLAND_REMOVAL_MODE mode    = zone.m_zone->GetIslandRemovalMode();
 244 
 245             // Remove solid areas outside the board cutouts and the insulated islands

 246             // only zones with net code > 0 can have insulated islands by definition

 247             if( zone.m_zone->GetNetCode() > 0 )
 248             {
 249                 // solid areas outside the board cutouts are also removed, because they are usually

 250                 // insulated islands

 251                 for( auto idx : islands )
 252                 {
 253                     if( mode == ISLAND_REMOVAL_MODE::ALWAYS
 254                             || ( mode == ISLAND_REMOVAL_MODE::AREA
 255                                     && poly.Outline( idx ).Area() < minArea )
 256                             || !m_boardOutline.Contains( poly.Polygon( idx ).front().CPoint( 0 ) ) )
 257                         poly.DeletePolygon( idx );
 258                     else
 259                         zone.m_zone->SetIsIsland( layer, idx );
 260                 }
 261             }
 262             // Zones with no net can have areas outside the board cutouts.

 263             // By definition, Zones with no net have no isolated island

 264             // (in fact all filled areas are isolated islands)

 265             // but they can have some areas outside the board cutouts.

 266             // A filled area outside the board cutouts has all points outside cutouts,

 267             // so we only need to check one point for each filled polygon.

 268             // Note also non copper zones are already clipped

 269             else if( m_brdOutlinesValid && zone.m_zone->IsOnCopperLayer() )
 270             {
 271                 for( int idx = 0; idx < poly.OutlineCount(); )
 272                 {
 273                     if( poly.Polygon( idx ).empty()
 274                             || !m_boardOutline.Contains( poly.Polygon( idx ).front().CPoint( 0 ) ) )
 275                     {
 276                         poly.DeletePolygon( idx );
 277                     }
 278                     else
 279                         idx++;
 280                 }
 281             }
 282 
 283             zone.m_zone->SetFilledPolysList( layer, poly );
 284             zone.m_zone->CalculateFilledArea();
 285 
 286             if( aCheck && zone.m_zone->GetHashValue( layer ) != poly.GetHash() )
 287                 outOfDate = true;
 288         }
 289     }
 290 
 291     if( aCheck && outOfDate )
 292     {
 293         PROGRESS_REPORTER_HIDER raii( m_progressReporter );
 294         KIDIALOG dlg( m_progressReporter->GetParent(),
 295                       _( "Zone fills are out-of-date. Refill?" ),
 296                       _( "Confirmation" ), wxOK | wxCANCEL | wxICON_WARNING );
 297         dlg.SetOKCancelLabels( _( "Refill" ), _( "Continue without Refill" ) );
 298         dlg.DoNotShowCheckbox( __FILE__, __LINE__ );
 299 
 300         if( dlg.ShowModal() == wxID_CANCEL )
 301         {
 302             if( m_commit )
 303                 m_commit->Revert();
 304 
 305             connectivity->SetProgressReporter( nullptr );
 306             return false;
 307         }
 308     }
 309 
 310     if( m_progressReporter )
 311     {
 312         m_progressReporter->AdvancePhase();
 313         m_progressReporter->Report( _( "Performing polygon fills..." ) );
 314         m_progressReporter->SetMaxProgress( toFill.size() );
 315     }
 316 
 317 
 318     nextItem = 0;
 319 
 320     auto tri_lambda = [&] ( PROGRESS_REPORTER* aReporter ) -> size_t
 321     {
 322         size_t num = 0;
 323 
 324         for( size_t i = nextItem++; i < islandsList.size(); i = nextItem++ )
 325         {
 326             islandsList[i].m_zone->CacheTriangulation();
 327             num++;
 328 
 329             if( m_progressReporter )
 330                 m_progressReporter->AdvanceProgress();
 331         }
 332 
 333         return num;
 334     };
 335 
 336     if( parallelThreadCount <= 1 )
 337         tri_lambda( m_progressReporter );
 338     else
 339     {
 340         for( size_t ii = 0; ii < parallelThreadCount; ++ii )
 341             returns[ii] = std::async( std::launch::async, tri_lambda, m_progressReporter );
 342 
 343         for( size_t ii = 0; ii < parallelThreadCount; ++ii )
 344         {
 345             // Here we balance returns with a 100ms timeout to allow UI updating

 346             std::future_status status;
 347             do
 348             {
 349                 if( m_progressReporter )
 350                     m_progressReporter->KeepRefreshing();
 351 
 352                 status = returns[ii].wait_for( std::chrono::milliseconds( 100 ) );
 353             } while( status != std::future_status::ready );
 354         }
 355     }
 356 
 357     if( m_progressReporter )
 358     {
 359         m_progressReporter->AdvancePhase();
 360         m_progressReporter->Report( _( "Committing changes..." ) );
 361         m_progressReporter->KeepRefreshing();
 362     }
 363 
 364     connectivity->SetProgressReporter( nullptr );
 365 
 366     if( m_commit )
 367     {
 368         m_commit->Push( _( "Fill Zone(s)" ), false );
 369     }
 370     else
 371     {
 372         for( auto& i : toFill )
 373             connectivity->Update( i.first );
 374 
 375         connectivity->RecalculateRatsnest();
 376     }
 377 
 378     return true;
 379 }

 382 /**

 383  * Return true if the given pad has a thermal connection with the given zone.

 384  */
 385 bool hasThermalConnection( D_PAD* pad, const ZONE_CONTAINER* aZone )
 386 {
 387     // Rejects non-standard pads with tht-only thermal reliefs

 388     if( aZone->GetPadConnection( pad ) == ZONE_CONNECTION::THT_THERMAL
 389             && pad->GetAttribute() != PAD_ATTRIB_STANDARD )
 390     {
 391         return false;
 392     }
 393 
 394     if( aZone->GetPadConnection( pad ) != ZONE_CONNECTION::THERMAL
 395             && aZone->GetPadConnection( pad ) != ZONE_CONNECTION::THT_THERMAL )
 396     {
 397         return false;
 398     }
 399 
 400     if( pad->GetNetCode() != aZone->GetNetCode() || pad->GetNetCode() <= 0 )
 401         return false;
 402 
 403     EDA_RECT item_boundingbox = pad->GetBoundingBox();
 404     int thermalGap = aZone->GetThermalReliefGap( pad );
 405     item_boundingbox.Inflate( thermalGap, thermalGap );
 406 
 407     return item_boundingbox.Intersects( aZone->GetBoundingBox() );
 408 }

 411 /**

 412  * Setup aDummyPad to have the same size and shape of aPad's hole.  This allows us to create

 413  * thermal reliefs and clearances for holes using the pad code.

 414  */
 415 static void setupDummyPadForHole( const D_PAD* aPad, D_PAD& aDummyPad )
 416 {
 417     aDummyPad.SetNetCode( aPad->GetNetCode() );
 418     aDummyPad.SetSize( aPad->GetDrillSize() );
 419     aDummyPad.SetOrientation( aPad->GetOrientation() );
 420     aDummyPad.SetShape( aPad->GetDrillShape() == PAD_DRILL_SHAPE_OBLONG ? PAD_SHAPE_OVAL
 421                                                                         : PAD_SHAPE_CIRCLE );
 422     aDummyPad.SetPosition( aPad->GetPosition() );
 423 }

 426 /**

 427  * Add a knockout for a pad.  The knockout is 'aGap' larger than the pad (which might be

 428  * either the thermal clearance or the electrical clearance).

 429  */
 430 void ZONE_FILLER::addKnockout( D_PAD* aPad, int aGap, SHAPE_POLY_SET& aHoles )
 431 {
 432     if( aPad->GetShape() == PAD_SHAPE_CUSTOM )
 433     {
 434         SHAPE_POLY_SET poly;
 435         aPad->TransformShapeWithClearanceToPolygon( poly, aGap, m_high_def );
 436 
 437         // the pad shape in zone can be its convex hull or the shape itself

 438         if( aPad->GetCustomShapeInZoneOpt() == CUST_PAD_SHAPE_IN_ZONE_CONVEXHULL )
 439         {
 440             std::vector<wxPoint> convex_hull;
 441             BuildConvexHull( convex_hull, poly );
 442 
 443             aHoles.NewOutline();
 444 
 445             for( const wxPoint& pt : convex_hull )
 446                 aHoles.Append( pt );
 447         }
 448         else
 449             aHoles.Append( poly );
 450     }
 451     else
 452     {
 453         // Optimizing polygon vertex count: the high definition is used for round

 454         // and oval pads (pads with large arcs) but low def for other shapes (with

 455         // small arcs)

 456         if( aPad->GetShape() == PAD_SHAPE_CIRCLE || aPad->GetShape() == PAD_SHAPE_OVAL ||
 457           ( aPad->GetShape() == PAD_SHAPE_ROUNDRECT && aPad->GetRoundRectRadiusRatio() > 0.4 ) )
 458             aPad->TransformShapeWithClearanceToPolygon( aHoles, aGap, m_high_def );
 459         else
 460             aPad->TransformShapeWithClearanceToPolygon( aHoles, aGap, m_low_def );
 461     }
 462 }

 465 /**

 466  * Add a knockout for a graphic item.  The knockout is 'aGap' larger than the item (which

 467  * might be either the electrical clearance or the board edge clearance).

 468  */
 469 void ZONE_FILLER::addKnockout( BOARD_ITEM* aItem, int aGap, bool aIgnoreLineWidth,
 470                                SHAPE_POLY_SET& aHoles )
 471 {
 472     switch( aItem->Type() )
 473     {
 474     case PCB_LINE_T:
 475     {
 476         DRAWSEGMENT* seg = (DRAWSEGMENT*) aItem;
 477         seg->TransformShapeWithClearanceToPolygon( aHoles, aGap, m_high_def, aIgnoreLineWidth );
 478         break;
 479     }
 480     case PCB_TEXT_T:
 481     {
 482         TEXTE_PCB* text = (TEXTE_PCB*) aItem;
 483         text->TransformBoundingBoxWithClearanceToPolygon( &aHoles, aGap );
 484         break;
 485     }
 486     case PCB_MODULE_EDGE_T:
 487     {
 488         EDGE_MODULE* edge = (EDGE_MODULE*) aItem;
 489         edge->TransformShapeWithClearanceToPolygon( aHoles, aGap, m_high_def, aIgnoreLineWidth );
 490         break;
 491     }
 492     case PCB_MODULE_TEXT_T:
 493     {
 494         TEXTE_MODULE* text = (TEXTE_MODULE*) aItem;
 495 
 496         if( text->IsVisible() )
 497             text->TransformBoundingBoxWithClearanceToPolygon( &aHoles, aGap );
 498 
 499         break;
 500     }
 501     default:
 502         break;
 503     }
 504 }

 507 /**

 508  * Removes thermal reliefs from the shape for any pads connected to the zone.  Does NOT add

 509  * in spokes, which must be done later.

 510  */
 511 void ZONE_FILLER::knockoutThermalReliefs( const ZONE_CONTAINER* aZone, PCB_LAYER_ID aLayer,
 512                                           SHAPE_POLY_SET& aFill )
 513 {
 514     SHAPE_POLY_SET holes;
 515 
 516     // Use a dummy pad to calculate relief when a pad has a hole but is not on the zone's

 517     // copper layer.  The dummy pad has the size and shape of the original pad's hole. We have

 518     // to give it a parent because some functions expect a non-null parent to find clearance

 519     // data, etc.

 520     MODULE  dummymodule( m_board );
 521     D_PAD   dummypad( &dummymodule );
 522 
 523     for( auto module : m_board->Modules() )
 524     {
 525         for( auto pad : module->Pads() )
 526         {
 527             if( !hasThermalConnection( pad, aZone ) )
 528                 continue;
 529 
 530             // If the pad isn't on the current layer but has a hole, knock out a thermal relief

 531             // for the hole.

 532             if( !pad->IsOnLayer( aLayer ) )
 533             {
 534                 if( pad->GetDrillSize().x == 0 && pad->GetDrillSize().y == 0 )
 535                     continue;
 536 
 537                 setupDummyPadForHole( pad, dummypad );
 538                 pad = &dummypad;
 539             }
 540 
 541             addKnockout( pad, aZone->GetThermalReliefGap( pad ), holes );
 542         }
 543     }
 544 
 545     holes.Simplify( SHAPE_POLY_SET::PM_FAST );
 546     aFill.BooleanSubtract( holes, SHAPE_POLY_SET::PM_FAST );
 547 }

 550 /**

 551  * Removes clearance from the shape for copper items which share the zone's layer but are

 552  * not connected to it.

 553  */
 554 void ZONE_FILLER::buildCopperItemClearances( const ZONE_CONTAINER* aZone, PCB_LAYER_ID aLayer,
 555                                              SHAPE_POLY_SET& aHoles )
 556 {
 557     static DRAWSEGMENT dummyEdge;
 558     dummyEdge.SetLayer( Edge_Cuts );
 559 
 560     // a small extra clearance to be sure actual track clearance is not smaller

 561     // than requested clearance due to many approximations in calculations,

 562     // like arc to segment approx, rounding issues...

 563     // 2 microns are a good value

 564     int extra_margin = Millimeter2iu( ADVANCED_CFG::GetCfg().m_extraClearance );
 565 
 566     BOARD_DESIGN_SETTINGS& bds = m_board->GetDesignSettings();
 567     int                    zone_clearance = aZone->GetLocalClearance();
 568     EDA_RECT               zone_boundingbox = aZone->GetBoundingBox();
 569 
 570     // items outside the zone bounding box are skipped, so it needs to be inflated by

 571     // the largest clearance value found in the netclasses and rules

 572     int biggest_clearance = std::max( zone_clearance, bds.GetBiggestClearanceValue() );
 573     zone_boundingbox.Inflate( biggest_clearance + extra_margin );
 574 
 575     // Use a dummy pad to calculate hole clearance when a pad has a hole but is not on the

 576     // zone's copper layer.  The dummy pad has the size and shape of the original pad's hole.

 577     // We have to give it a parent because some functions expect a non-null parent to find

 578     // clearance data, etc.

 579     MODULE  dummymodule( m_board );
 580     D_PAD   dummypad( &dummymodule );
 581 
 582     // Add non-connected pad clearances

 583     //

 584     for( auto module : m_board->Modules() )
 585     {
 586         for( auto pad : module->Pads() )
 587         {
 588             if( !pad->IsOnLayer( aLayer ) )
 589             {
 590                 if( pad->GetDrillSize().x == 0 && pad->GetDrillSize().y == 0 )
 591                     continue;
 592 
 593                 setupDummyPadForHole( pad, dummypad );
 594                 pad = &dummypad;
 595             }
 596 
 597             if( pad->GetNetCode() != aZone->GetNetCode() || pad->GetNetCode() <= 0
 598                     || aZone->GetPadConnection( pad ) == ZONE_CONNECTION::NONE )
 599             {
 600                 if( pad->GetBoundingBox().Intersects( zone_boundingbox ) )
 601                 {
 602                     int gap;
 603 
 604                     // for pads having the same netcode as the zone, the net clearance has no

 605                     // meaning so use the greater of the zone clearance and the thermal relief

 606                     if( pad->GetNetCode() > 0 && pad->GetNetCode() == aZone->GetNetCode() )
 607                         gap = std::max( zone_clearance, aZone->GetThermalReliefGap( pad ) );
 608                     else
 609                         gap = aZone->GetClearance( pad );
 610 
 611                     addKnockout( pad, gap, aHoles );
 612                 }
 613             }
 614         }
 615     }
 616 
 617     // Add non-connected track clearances

 618     //

 619     for( auto track : m_board->Tracks() )
 620     {
 621         if( !track->IsOnLayer( aLayer ) )
 622             continue;
 623 
 624         if( track->GetNetCode() == aZone->GetNetCode()  && ( aZone->GetNetCode() != 0) )
 625             continue;
 626 
 627         if( track->GetBoundingBox().Intersects( zone_boundingbox ) )
 628         {
 629             int gap = aZone->GetClearance( track ) + extra_margin;
 630 
 631             track->TransformShapeWithClearanceToPolygon( aHoles, gap, m_low_def );
 632         }
 633     }
 634 
 635     // Add graphic item clearances.  They are by definition unconnected, and have no clearance

 636     // definitions of their own.

 637     //

 638     auto doGraphicItem =
 639             [&]( BOARD_ITEM* aItem )
 640             {
 641                 // A item on the Edge_Cuts is always seen as on any layer:

 642                 if( !aItem->IsOnLayer( aLayer ) && !aItem->IsOnLayer( Edge_Cuts ) )
 643                     return;
 644 
 645                 if( aItem->GetBoundingBox().Intersects( zone_boundingbox ) )
 646                 {
 647                     bool ignoreLineWidth = aItem->IsOnLayer( Edge_Cuts );
 648                     int  gap = aZone->GetClearance( aItem );
 649 
 650                     addKnockout( aItem, gap, ignoreLineWidth, aHoles );
 651                 }
 652             };
 653 
 654     for( auto module : m_board->Modules() )
 655     {
 656         doGraphicItem( &module->Reference() );
 657         doGraphicItem( &module->Value() );
 658 
 659         for( auto item : module->GraphicalItems() )
 660             doGraphicItem( item );
 661     }
 662 
 663     for( auto item : m_board->Drawings() )
 664         doGraphicItem( item );
 665 
 666     // Add zones outlines having an higher priority and keepout

 667     //

 668     for( ZONE_CONTAINER* zone : m_board->GetZoneList( true ) )
 669     {
 670 
 671         // If the zones share no common layers

 672         if( !zone->GetLayerSet().test( aLayer ) )
 673             continue;
 674 
 675         if( !zone->GetIsKeepout() && zone->GetPriority() <= aZone->GetPriority() )
 676             continue;
 677 
 678         if( zone->GetIsKeepout() && !zone->GetDoNotAllowCopperPour() )
 679             continue;
 680 
 681         // A higher priority zone or keepout area is found: remove this area

 682         EDA_RECT item_boundingbox = zone->GetBoundingBox();
 683 
 684         if( item_boundingbox.Intersects( zone_boundingbox ) )
 685         {
 686             // Add the zone outline area.  Don't use any clearance for keepouts, or for zones

 687             // with the same net (they will be connected but will honor their own clearance,

 688             // thermal connections, etc.).

 689             int  gap = 0;
 690 
 691             if( !zone->GetIsKeepout() && aZone->GetNetCode() != zone->GetNetCode() )
 692                 gap = aZone->GetClearance( zone );
 693 
 694             zone->TransformOutlinesShapeWithClearanceToPolygon( aHoles, gap );
 695         }
 696     }
 697 
 698     aHoles.Simplify( SHAPE_POLY_SET::PM_FAST );
 699 }

 702 /**

 703  * 1 - Creates the main zone outline using a correction to shrink the resulting area by

 704  *     m_ZoneMinThickness / 2.  The result is areas with a margin of m_ZoneMinThickness / 2

 705  *     so that when drawing outline with segments having a thickness of m_ZoneMinThickness the

 706  *     outlines will match exactly the initial outlines

 707  * 2 - Knocks out thermal reliefs around thermally-connected pads

 708  * 3 - Builds a set of thermal spoke for the whole zone

 709  * 4 - Knocks out unconnected copper items, deleting any affected spokes

 710  * 5 - Removes unconnected copper islands, deleting any affected spokes

 711  * 6 - Adds in the remaining spokes

 712  */
 713 void ZONE_FILLER::computeRawFilledArea( const ZONE_CONTAINER* aZone, PCB_LAYER_ID aLayer,
 714                                         const SHAPE_POLY_SET& aSmoothedOutline,
 715                                         std::set<VECTOR2I>* aPreserveCorners,
 716                                         SHAPE_POLY_SET& aRawPolys,
 717                                         SHAPE_POLY_SET& aFinalPolys )
 718 {
 719     m_high_def = m_board->GetDesignSettings().m_MaxError;
 720     m_low_def = std::min( ARC_LOW_DEF, int( m_high_def*1.5 ) );   // Reasonable value

 721 
 722     // Features which are min_width should survive pruning; features that are *less* than

 723     // min_width should not.  Therefore we subtract epsilon from the min_width when

 724     // deflating/inflating.

 725     int half_min_width = aZone->GetMinThickness() / 2;
 726     int epsilon = Millimeter2iu( 0.001 );
 727     int numSegs = std::max( GetArcToSegmentCount( half_min_width, m_high_def, 360.0 ), 6 );
 728 
 729     // solid polygons are deflated and inflated during calculations.

 730     // Polygons deflate usually do not create issues.

 731     // Polygons inflate is a tricky transform, because it can create excessively long and narrow 'spikes'

 732     // especially for acute angles.

 733     // But in very case, the inflate transform caannot create bigger shapes than initial shapes.

 734     // so the corner strategy is very important.

 735     // The best is SHAPE_POLY_SET::ROUND_ALL_CORNERS.

 736     // unfortunately, it creates a lot of small segments.

 737     // SHAPE_POLY_SET::ALLOW_ACUTE_CORNERS is not acceptable

 738     // So for intermediate transforms, we use CHAMFER_ALL_CORNERS.

 739     // For final transform, we use ROUND_ALL_CORNERS

 740     SHAPE_POLY_SET::CORNER_STRATEGY intermediatecornerStrategy = SHAPE_POLY_SET::CHAMFER_ALL_CORNERS;
 741     SHAPE_POLY_SET::CORNER_STRATEGY finalcornerStrategy = SHAPE_POLY_SET::ROUND_ALL_CORNERS;
 742 
 743     std::deque<SHAPE_LINE_CHAIN> thermalSpokes;
 744     SHAPE_POLY_SET clearanceHoles;
 745 
 746     std::unique_ptr<SHAPE_FILE_IO> dumper( new SHAPE_FILE_IO(
 747                     s_DumpZonesWhenFilling ? "zones_dump.txt" : "", SHAPE_FILE_IO::IOM_APPEND ) );
 748 
 749     aRawPolys = aSmoothedOutline;
 750 
 751     if( s_DumpZonesWhenFilling )
 752         dumper->BeginGroup( "clipper-zone" );
 753 
 754     knockoutThermalReliefs( aZone, aLayer, aRawPolys );
 755 
 756     if( s_DumpZonesWhenFilling )
 757         dumper->Write( &aRawPolys, "solid-areas-minus-thermal-reliefs" );
 758 
 759     buildCopperItemClearances( aZone, aLayer, clearanceHoles );
 760 
 761     if( s_DumpZonesWhenFilling )
 762         dumper->Write( &aRawPolys, "clearance holes" );
 763 
 764     buildThermalSpokes( aZone, aLayer, thermalSpokes );
 765 
 766     // Create a temporary zone that we can hit-test spoke-ends against.  It's only temporary

 767     // because the "real" subtract-clearance-holes has to be done after the spokes are added.

 768     static const bool USE_BBOX_CACHES = true;
 769     SHAPE_POLY_SET testAreas = aRawPolys;
 770     testAreas.BooleanSubtract( clearanceHoles, SHAPE_POLY_SET::PM_FAST );
 771 
 772     // Prune features that don't meet minimum-width criteria

 773     if( half_min_width - epsilon > epsilon )
 774     {
 775         testAreas.Deflate( half_min_width - epsilon, numSegs, intermediatecornerStrategy );
 776         testAreas.Inflate( half_min_width - epsilon, numSegs, intermediatecornerStrategy );
 777     }
 778 
 779     // Spoke-end-testing is hugely expensive so we generate cached bounding-boxes to speed

 780     // things up a bit.

 781     testAreas.BuildBBoxCaches();
 782 
 783     for( const SHAPE_LINE_CHAIN& spoke : thermalSpokes )
 784     {
 785         const VECTOR2I& testPt = spoke.CPoint( 3 );
 786 
 787         // Hit-test against zone body

 788         if( testAreas.Contains( testPt, -1, 1, USE_BBOX_CACHES ) )
 789         {
 790             aRawPolys.AddOutline( spoke );
 791             continue;
 792         }
 793 
 794         // Hit-test against other spokes

 795         for( const SHAPE_LINE_CHAIN& other : thermalSpokes )
 796         {
 797             if( &other != &spoke && other.PointInside( testPt, 1, USE_BBOX_CACHES  ) )
 798             {
 799                 aRawPolys.AddOutline( spoke );
 800                 break;
 801             }
 802         }
 803     }
 804 
 805     // Ensure previous changes (adding thermal stubs) do not add

 806     // filled areas outside the zone boundary

 807     aRawPolys.BooleanIntersection( aSmoothedOutline, SHAPE_POLY_SET::PM_FAST );
 808     aRawPolys.Simplify( SHAPE_POLY_SET::PM_FAST );
 809 
 810     if( s_DumpZonesWhenFilling )
 811         dumper->Write( &aRawPolys, "solid-areas-with-thermal-spokes" );
 812 
 813     aRawPolys.BooleanSubtract( clearanceHoles, SHAPE_POLY_SET::PM_FAST );
 814     // Prune features that don't meet minimum-width criteria

 815     if( half_min_width - epsilon > epsilon )
 816         aRawPolys.Deflate( half_min_width - epsilon, numSegs, intermediatecornerStrategy );
 817 
 818     if( s_DumpZonesWhenFilling )
 819         dumper->Write( &aRawPolys, "solid-areas-before-hatching" );
 820 
 821     // Now remove the non filled areas due to the hatch pattern

 822     if( aZone->GetFillMode() == ZONE_FILL_MODE::HATCH_PATTERN )
 823         addHatchFillTypeOnZone( aZone, aLayer, aRawPolys );
 824 
 825     if( s_DumpZonesWhenFilling )
 826         dumper->Write( &aRawPolys, "solid-areas-after-hatching" );
 827 
 828     // Re-inflate after pruning of areas that don't meet minimum-width criteria

 829     if( aZone->GetFilledPolysUseThickness() )
 830     {
 831         // If we're stroking the zone with a min_width stroke then this will naturally

 832         // inflate the zone by half_min_width

 833     }
 834     else if( half_min_width - epsilon > epsilon )
 835     {
 836         aRawPolys.Simplify( SHAPE_POLY_SET::PM_FAST );
 837         aRawPolys.Inflate( half_min_width - epsilon, numSegs, finalcornerStrategy );
 838 
 839         // If we've deflated/inflated by something near our corner radius then we will have

 840         // ended up with too-sharp corners.  Apply outline smoothing again.

 841         if( aZone->GetMinThickness() > (int)aZone->GetCornerRadius() )
 842             aRawPolys.BooleanIntersection( aSmoothedOutline, SHAPE_POLY_SET::PM_FAST );
 843     }
 844 
 845     aRawPolys.Fracture( SHAPE_POLY_SET::PM_FAST );
 846 
 847     if( s_DumpZonesWhenFilling )
 848         dumper->Write( &aRawPolys, "areas_fractured" );
 849 
 850     aFinalPolys = aRawPolys;
 851 
 852     if( s_DumpZonesWhenFilling )
 853         dumper->EndGroup();
 854 }

 857 /*

 858  * Build the filled solid areas data from real outlines (stored in m_Poly)

 859  * The solid areas can be more than one on copper layers, and do not have holes

 860  * ( holes are linked by overlapping segments to the main outline)

 861  */
 862 bool ZONE_FILLER::fillSingleZone( ZONE_CONTAINER* aZone, PCB_LAYER_ID aLayer,
 863                                   SHAPE_POLY_SET& aRawPolys, SHAPE_POLY_SET& aFinalPolys )
 864 {
 865     SHAPE_POLY_SET smoothedPoly;
 866     std::set<VECTOR2I> colinearCorners;
 867     aZone->GetColinearCorners( m_board, colinearCorners );
 868 
 869     /*

 870      * convert outlines + holes to outlines without holes (adding extra segments if necessary)

 871      * m_Poly data is expected normalized, i.e. NormalizeAreaOutlines was used after building

 872      * this zone

 873      */
 874     if ( !aZone->BuildSmoothedPoly( smoothedPoly, &colinearCorners ) )
 875         return false;
 876 
 877     if( aZone->IsOnCopperLayer() )
 878     {
 879         computeRawFilledArea(
 880                 aZone, aLayer, smoothedPoly, &colinearCorners, aRawPolys, aFinalPolys );
 881     }
 882     else
 883     {
 884         // Features which are min_width should survive pruning; features that are *less* than

 885         // min_width should not.  Therefore we subtract epsilon from the min_width when

 886         // deflating/inflating.

 887         int half_min_width = aZone->GetMinThickness() / 2;
 888         int epsilon = Millimeter2iu( 0.001 );
 889         int numSegs = std::max( GetArcToSegmentCount( half_min_width, m_high_def, 360.0 ), 6 );
 890 
 891         if( m_brdOutlinesValid )
 892             smoothedPoly.BooleanIntersection( m_boardOutline, SHAPE_POLY_SET::PM_STRICTLY_SIMPLE );
 893 
 894         smoothedPoly.Deflate( half_min_width/* - epsilon*/, numSegs );
 895 
 896         // Remove the non filled areas due to the hatch pattern

 897         if( aZone->GetFillMode() == ZONE_FILL_MODE::HATCH_PATTERN )
 898             addHatchFillTypeOnZone( aZone, aLayer, smoothedPoly );
 899 
 900         // Re-inflate after pruning of areas that don't meet minimum-width criteria

 901         if( aZone->GetFilledPolysUseThickness() )
 902         {
 903             // If we're stroking the zone with a min_width stroke then this will naturally

 904             // inflate the zone by half_min_width

 905         }
 906         else if( half_min_width - epsilon > epsilon )
 907             smoothedPoly.Deflate( -( half_min_width - epsilon ), numSegs );
 908 
 909         aRawPolys = smoothedPoly;
 910         aFinalPolys = smoothedPoly;
 911 
 912         aFinalPolys.Fracture( SHAPE_POLY_SET::PM_STRICTLY_SIMPLE );
 913     }
 914 
 915     aZone->SetNeedRefill( false );
 916     return true;
 917 }

 920 /**

 921  * Function buildThermalSpokes

 922  */
 923 void ZONE_FILLER::buildThermalSpokes( const ZONE_CONTAINER* aZone, PCB_LAYER_ID aLayer,
 924                                       std::deque<SHAPE_LINE_CHAIN>& aSpokesList )
 925 {
 926     auto zoneBB = aZone->GetBoundingBox();
 927     int  zone_clearance = aZone->GetZoneClearance();
 928     int  biggest_clearance = m_board->GetDesignSettings().GetBiggestClearanceValue();
 929     biggest_clearance = std::max( biggest_clearance, zone_clearance );
 930     zoneBB.Inflate( biggest_clearance );
 931 
 932     // Is a point on the boundary of the polygon inside or outside?  This small epsilon lets

 933     // us avoid the question.

 934     int epsilon = KiROUND( IU_PER_MM * 0.04 );  // about 1.5 mil

 935 
 936     for( auto module : m_board->Modules() )
 937     {
 938         for( auto pad : module->Pads() )
 939         {
 940             if( !hasThermalConnection( pad, aZone ) )
 941                 continue;
 942 
 943             // We currently only connect to pads, not pad holes

 944             if( !pad->IsOnLayer( aLayer ) )
 945                 continue;
 946 
 947             int thermalReliefGap = aZone->GetThermalReliefGap( pad );
 948 
 949             // Calculate thermal bridge half width

 950             int spoke_w = aZone->GetThermalReliefCopperBridge( pad );
 951             // Avoid spoke_w bigger than the smaller pad size, because

 952             // it is not possible to create stubs bigger than the pad.

 953             // Possible refinement: have a separate size for vertical and horizontal stubs

 954             spoke_w = std::min( spoke_w, pad->GetSize().x );
 955             spoke_w = std::min( spoke_w, pad->GetSize().y );
 956 
 957             // Cannot create stubs having a width < zone min thickness

 958             if( spoke_w <= aZone->GetMinThickness() )
 959                 continue;
 960 
 961             int spoke_half_w = spoke_w / 2;
 962 
 963             // Quick test here to possibly save us some work

 964             BOX2I itemBB = pad->GetBoundingBox();
 965             itemBB.Inflate( thermalReliefGap + epsilon );
 966 
 967             if( !( itemBB.Intersects( zoneBB ) ) )
 968                 continue;
 969 
 970             // Thermal spokes consist of segments from the pad center to points just outside

 971             // the thermal relief.

 972             //

 973             // We use the bounding-box to lay out the spokes, but for this to work the

 974             // bounding box has to be built at the same rotation as the spokes.

 975             // We have to use a dummy pad to avoid dirtying the cached shapes

 976             wxPoint shapePos = pad->ShapePos();
 977             double  padAngle = pad->GetOrientation();
 978             D_PAD   dummy_pad( *pad );
 979             dummy_pad.SetOrientation( 0.0 );
 980             dummy_pad.SetPosition( { 0, 0 } );
 981 
 982             BOX2I reliefBB = dummy_pad.GetBoundingBox();
 983             reliefBB.Inflate( thermalReliefGap + epsilon );
 984 
 985             // For circle pads, the thermal spoke orientation is 45 deg

 986             if( pad->GetShape() == PAD_SHAPE_CIRCLE )
 987                 padAngle = s_RoundPadThermalSpokeAngle;
 988 
 989             for( int i = 0; i < 4; i++ )
 990             {
 991                 SHAPE_LINE_CHAIN spoke;
 992                 switch( i )
 993                 {
 994                 case 0:       // lower stub

 995                     spoke.Append( +spoke_half_w,       -spoke_half_w );
 996                     spoke.Append( -spoke_half_w,       -spoke_half_w );
 997                     spoke.Append( -spoke_half_w,       reliefBB.GetBottom() );
 998                     spoke.Append( 0,                   reliefBB.GetBottom() );  // test pt

 999                     spoke.Append( +spoke_half_w,       reliefBB.GetBottom() );
1000                     break;
1001 
1002                 case 1:       // upper stub

1003                     spoke.Append( +spoke_half_w,       spoke_half_w );
1004                     spoke.Append( -spoke_half_w,       spoke_half_w );
1005                     spoke.Append( -spoke_half_w,       reliefBB.GetTop() );
1006                     spoke.Append( 0,                   reliefBB.GetTop() );     // test pt

1007                     spoke.Append( +spoke_half_w,       reliefBB.GetTop() );
1008                     break;
1009 
1010                 case 2:       // right stub

1011                     spoke.Append( -spoke_half_w,       spoke_half_w );
1012                     spoke.Append( -spoke_half_w,       -spoke_half_w );
1013                     spoke.Append( reliefBB.GetRight(), -spoke_half_w );
1014                     spoke.Append( reliefBB.GetRight(), 0 );                     // test pt

1015                     spoke.Append( reliefBB.GetRight(), spoke_half_w );
1016                     break;
1017 
1018                 case 3:       // left stub

1019                     spoke.Append( spoke_half_w,        spoke_half_w );
1020                     spoke.Append( spoke_half_w,        -spoke_half_w );
1021                     spoke.Append( reliefBB.GetLeft(),  -spoke_half_w );
1022                     spoke.Append( reliefBB.GetLeft(),  0 );                     // test pt

1023                     spoke.Append( reliefBB.GetLeft(),  spoke_half_w );
1024                     break;
1025                 }
1026 
1027                 spoke.Rotate( -DECIDEG2RAD( padAngle ) );
1028                 spoke.Move( shapePos );
1029 
1030                 spoke.SetClosed( true );
1031                 spoke.GenerateBBoxCache();
1032                 aSpokesList.push_back( std::move( spoke ) );
1033             }
1034         }
1035     }
1036 }

1038 
1039 void ZONE_FILLER::addHatchFillTypeOnZone( const ZONE_CONTAINER* aZone, PCB_LAYER_ID aLayer,
1040                                           SHAPE_POLY_SET& aRawPolys )
1041 {
1042     // Build grid:

1043 
1044     // obviously line thickness must be > zone min thickness.

1045     // It can happens if a board file was edited by hand by a python script

1046     // Use 1 micron margin to be *sure* there is no issue in Gerber files

1047     // (Gbr file unit = 1 or 10 nm) due to some truncation in coordinates or calculations

1048     // This margin also avoid problems due to rounding coordinates in next calculations

1049     // that can create incorrect polygons

1050     int thickness = std::max( aZone->GetHatchFillTypeThickness(),
1051                               aZone->GetMinThickness()+Millimeter2iu( 0.001 ) );
1052 
1053     int linethickness = thickness - aZone->GetMinThickness();
1054     int gridsize = thickness + aZone->GetHatchFillTypeGap();
1055     double orientation = aZone->GetHatchFillTypeOrientation();
1056 
1057     SHAPE_POLY_SET filledPolys = aRawPolys;
1058     // Use a area that contains the rotated bbox by orientation,

1059     // and after rotate the result by -orientation.

1060     if( orientation != 0.0 )
1061     {
1062         filledPolys.Rotate( M_PI/180.0 * orientation, VECTOR2I( 0,0 ) );
1063     }
1064 
1065     BOX2I bbox = filledPolys.BBox( 0 );
1066 
1067     // Build hole shape

1068     // the hole size is aZone->GetHatchFillTypeGap(), but because the outline thickness

1069     // is aZone->GetMinThickness(), the hole shape size must be larger

1070     SHAPE_LINE_CHAIN hole_base;
1071     int hole_size = aZone->GetHatchFillTypeGap() + aZone->GetMinThickness();
1072     VECTOR2I corner( 0, 0 );;
1073     hole_base.Append( corner );
1074     corner.x += hole_size;
1075     hole_base.Append( corner );
1076     corner.y += hole_size;
1077     hole_base.Append( corner );
1078     corner.x = 0;
1079     hole_base.Append( corner );
1080     hole_base.SetClosed( true );
1081 
1082     // Calculate minimal area of a grid hole.

1083     // All holes smaller than a threshold will be removed

1084     double minimal_hole_area = hole_base.Area() / 2;
1085 
1086     // Now convert this hole to a smoothed shape:

1087     if( aZone->GetHatchFillTypeSmoothingLevel()  > 0 )
1088     {
1089         // the actual size of chamfer, or rounded corner radius is the half size

1090         // of the HatchFillTypeGap scaled by aZone->GetHatchFillTypeSmoothingValue()

1091         // aZone->GetHatchFillTypeSmoothingValue() = 1.0 is the max value for the chamfer or the

1092         // radius of corner (radius = half size of the hole)

1093         int smooth_value = KiROUND( aZone->GetHatchFillTypeGap()
1094                                     * aZone->GetHatchFillTypeSmoothingValue() / 2 );
1095 
1096         // Minimal optimization:

1097         // make smoothing only for reasonnable smooth values, to avoid a lot of useless segments

1098         // and if the smooth value is small, use chamfer even if fillet is requested

1099         #define SMOOTH_MIN_VAL_MM 0.02
1100         #define SMOOTH_SMALL_VAL_MM 0.04
1101         if( smooth_value > Millimeter2iu( SMOOTH_MIN_VAL_MM ) )
1102         {
1103             SHAPE_POLY_SET smooth_hole;
1104             smooth_hole.AddOutline( hole_base );
1105             int smooth_level = aZone->GetHatchFillTypeSmoothingLevel();
1106 
1107             if( smooth_value < Millimeter2iu( SMOOTH_SMALL_VAL_MM ) && smooth_level > 1 )
1108                 smooth_level = 1;
1109             // Use a larger smooth_value to compensate the outline tickness

1110             // (chamfer is not visible is smooth value < outline thickess)

1111             smooth_value += aZone->GetMinThickness()/2;
1112 
1113             // smooth_value cannot be bigger than the half size oh the hole:

1114             smooth_value = std::min( smooth_value, aZone->GetHatchFillTypeGap()/2 );
1115             // the error to approximate a circle by segments when smoothing corners by a arc

1116             int error_max = std::max( Millimeter2iu( 0.01), smooth_value/20 );
1117 
1118             switch( smooth_level )
1119             {
1120             case 1:
1121                 // Chamfer() uses the distance from a corner to create a end point

1122                 // for the chamfer.

1123                 hole_base = smooth_hole.Chamfer( smooth_value ).Outline( 0 );
1124                 break;
1125 
1126             default:
1127                 if( aZone->GetHatchFillTypeSmoothingLevel() > 2 )
1128                     error_max /= 2;    // Force better smoothing

1129                 hole_base = smooth_hole.Fillet( smooth_value, error_max ).Outline( 0 );
1130                 break;
1131 
1132             case 0:
1133                 break;
1134             };
1135         }
1136     }
1137 
1138     // Build holes

1139     SHAPE_POLY_SET holes;
1140 
1141     for( int xx = 0; ; xx++ )
1142     {
1143         int xpos = xx * gridsize;
1144 
1145         if( xpos > bbox.GetWidth() )
1146             break;
1147 
1148         for( int yy = 0; ; yy++ )
1149         {
1150             int ypos = yy * gridsize;
1151 
1152             if( ypos > bbox.GetHeight() )
1153                 break;
1154 
1155             // Generate hole

1156             SHAPE_LINE_CHAIN hole( hole_base );
1157             hole.Move( VECTOR2I( xpos, ypos ) );
1158             holes.AddOutline( hole );
1159         }
1160     }
1161 
1162     holes.Move( bbox.GetPosition() );
1163 
1164     // Clamp holes to the area of filled zones with a outline thickness

1165     // > aZone->GetMinThickness() to be sure the thermal pads can be built

1166     int outline_margin = std::max( (aZone->GetMinThickness()*10)/9, linethickness/2 );
1167     filledPolys.Deflate( outline_margin, 16 );
1168     holes.BooleanIntersection( filledPolys, SHAPE_POLY_SET::PM_FAST );
1169 
1170     if( orientation != 0.0 )
1171         holes.Rotate( -M_PI/180.0 * orientation, VECTOR2I( 0,0 ) );
1172 
1173     // Now filter truncated holes to avoid small holes in pattern

1174     // It happens for holes near the zone outline

1175     for( int ii = 0; ii < holes.OutlineCount(); )
1176     {
1177         double area = holes.Outline( ii ).Area();
1178 
1179         if( area < minimal_hole_area ) // The current hole is too small: remove it

1180             holes.DeletePolygon( ii );
1181         else
1182             ++ii;
1183     }
1184
