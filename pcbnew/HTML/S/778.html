  25 #ifndef PICKER_TOOL_H
  26 #define PICKER_TOOL_H
  27 
  28 #include <boost/optional/optional.hpp>
  29 #include <tools/pcb_tool_base.h>
  30 
  31 /**

  32  * @brief Generic tool for picking a point.

  33  */
  34 class PCBNEW_PICKER_TOOL : public PCB_TOOL_BASE
  35 {
  36 public:
  37     PCBNEW_PICKER_TOOL();
  38     ~PCBNEW_PICKER_TOOL() override { }
  39 
  40     ///> Event handler types.

  41     typedef std::function<bool(const VECTOR2D&)> CLICK_HANDLER;
  42     typedef std::function<void(const VECTOR2D&)> MOTION_HANDLER;
  43     typedef std::function<void(void)> CANCEL_HANDLER;
  44     typedef std::function<void(const int&)> FINALIZE_HANDLER;
  45 
  46     enum pickerEndState
  47     {
  48         WAIT_CANCEL,
  49         CLICK_CANCEL,
  50         END_ACTIVATE,
  51         EVT_CANCEL,
  52         EXCEPTION_CANCEL
  53     };
  54 
  55     ///> @copydoc TOOL_INTERACTIVE::Reset()

  56     void Reset( RESET_REASON aReason ) override {}
  57 
  58     ///> Main event loop.

  59     int Main( const TOOL_EVENT& aEvent );
  60 
  61     /**

  62      * Function SetLayerSet()

  63      * Sets the tool's snap layer set

  64      */
  65     inline void SetLayerSet( LSET aLayerSet ) { m_layerMask = aLayerSet; }
  66 
  67     inline void SetCursor( const wxCursor& aCursor ) { m_cursor = aCursor; }
  68 
  69     /**

  70      * Function SetClickHandler()

  71      * Sets a handler for mouse click event. Handler may decide to receive further click by

  72      * returning true.

  73      */
  74     inline void SetClickHandler( CLICK_HANDLER aHandler )
  75     {
  76         wxASSERT( !m_clickHandler );
  77         m_clickHandler = aHandler;
  78     }
  79 
  80     /**

  81      * Function SetMotionHandler()

  82      * Sets a handler for mouse motion.  Used for roll-over highlighting.

  83      */
  84     inline void SetMotionHandler( MOTION_HANDLER aHandler )
  85     {
  86         wxASSERT( !m_motionHandler );
  87         m_motionHandler = aHandler;
  88     }
  89 
  90     /**

  91      * Function SetCancelHandler()

  92      * Sets a handler for cancel events (ESC or context-menu Cancel).

  93      */
  94     inline void SetCancelHandler( CANCEL_HANDLER aHandler )
  95     {
  96         wxASSERT( !m_cancelHandler );
  97         m_cancelHandler = aHandler;
  98     }
  99 
 100     /**

 101      * Function SetFinalizeHandler()

 102      * Sets a handler for the finalize event. Takes the state of the exit from the Main loop

 103      */
 104     inline void SetFinalizeHandler( FINALIZE_HANDLER aHandler )
 105     {
 106         wxASSERT( !m_finalizeHandler );
 107         m_finalizeHandler = aHandler;
 108     }
 109 
 110 private:
 111     ///> @copydoc TOOL_INTERACTIVE::setTransitions();

 112     void setTransitions() override;
 113 
 114     ///> Reinitializes tool to its initial state.

 115     void reset();
 116 
 117     ///> Applies the requested VIEW_CONTROLS settings.

 118     void setControls();
 119 
 120 private:
 121     ///> The layer set to use for optional snapping

 122     LSET                  m_layerMask;
 123     wxCursor              m_cursor;
 124 
 125     OPT<CLICK_HANDLER>    m_clickHandler;
 126     OPT<MOTION_HANDLER>   m_motionHandler;
 127     OPT<CANCEL_HANDLER>   m_cancelHandler;
 128     OPT<FINALIZE_HANDLER> m_finalizeHandler;
 129 
 130     OPT<VECTOR2D>         m_picked;
 131 };
 132 
 133 #endif /* PICKER_TOOL_H */
