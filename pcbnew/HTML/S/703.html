  41 NODE::OPT_OBSTACLE WALKAROUND::nearestObstacle( const LINE& aPath )
  42 {
  43     NODE::OPT_OBSTACLE obs = m_world->NearestObstacle( &aPath, m_itemMask, m_restrictedSet.empty() ? NULL : &m_restrictedSet );
  44 
  45     if( m_restrictedSet.empty() )
  46         return obs;
  47 
  48     else if( obs && m_restrictedSet.find ( obs->m_item ) != m_restrictedSet.end() )
  49         return obs;
  50 
  51     return NODE::OPT_OBSTACLE();
  52 }

 157 bool clipToLoopStart( SHAPE_LINE_CHAIN& l )
 158 {
 159     auto ip = l.SelfIntersecting();
 160 
 161     if(!ip)
 162         return false;
 163     else {
 164         int pidx = l.Split( ip->p );
 165         auto lead = l.Slice(0, pidx);
 166         auto tail = l.Slice(pidx + 1, -1);
 167 
 168         int pidx2 = tail.Split( ip->p );
 169         
 170         auto dbg = ROUTER::GetInstance()->GetInterface()->GetDebugDecorator();
 171         dbg->AddPoint( ip->p, 5 );
 172         
 173         l = lead;
 174         l.Append( tail.Slice( 0, pidx2 ) );
 175         //l = l.Slice(0, pidx);

 176         return true;
 177     }

 184 const WALKAROUND::RESULT WALKAROUND::Route( const LINE& aInitialPath )
 185 {
 186     LINE path_cw( aInitialPath ), path_ccw( aInitialPath );
 187     WALKAROUND_STATUS s_cw = IN_PROGRESS, s_ccw = IN_PROGRESS;
 188     SHAPE_LINE_CHAIN best_path;
 189     RESULT result;
 190 
 191     // special case for via-in-the-middle-of-track placement

 192     if( aInitialPath.PointCount() <= 1 )
 193     {
 194         if( aInitialPath.EndsWithVia() && m_world->CheckColliding( &aInitialPath.Via(), m_itemMask ) )
 195             return RESULT( STUCK, STUCK );
 196 
 197         return RESULT( DONE, DONE, aInitialPath, aInitialPath );
 198     }
 199 
 200     start( aInitialPath );
 201 
 202     m_currentObstacle[0] = m_currentObstacle[1] = nearestObstacle( aInitialPath );
 203     m_recursiveBlockageCount = 0;
 204 
 205     result.lineCw = aInitialPath;
 206     result.lineCcw = aInitialPath;
 207 
 208     if( m_forceWinding )
 209     {
 210         s_cw = m_forceCw ? IN_PROGRESS : STUCK;
 211         s_ccw = m_forceCw ? STUCK : IN_PROGRESS;
 212         m_forceSingleDirection = true;
 213     } else {
 214         m_forceSingleDirection = false;
 215     }
 216 
 217     while( m_iteration < m_iterationLimit )
 218     {
 219         if( s_cw != STUCK )
 220             s_cw = singleStep( path_cw, true );
 221 
 222         if( s_ccw != STUCK )
 223             s_ccw = singleStep( path_ccw, false );
 224 
 225         //Dbg()->AddLine( path_cw.CLine(), 2, 10000 );

 226 
 227 
 228         //printf("iter %d s_cw %d s_ccw %d\n", m_iteration, s_cw, s_ccw );

 229         
 230         auto old = path_cw.CLine();
 231 
 232         if( clipToLoopStart( path_cw.Line() ))
 233         {
 234             //printf("ClipCW\n");

 235             //Dbg()->AddLine( old, 1, 40000 );

 236             s_cw = ALMOST_DONE;
 237         }
 238 
 239         if( clipToLoopStart( path_ccw.Line() ))
 240         {
 241             //printf("ClipCCW\n");

 242             s_ccw = ALMOST_DONE;
 243         }
 244 
 245         
 246         if( s_cw != IN_PROGRESS )
 247         {
 248             result.lineCw = path_cw;
 249             result.statusCw = s_cw;
 250         }
 251 
 252         if( s_ccw != IN_PROGRESS )
 253         {
 254             result.lineCcw = path_ccw;
 255             result.statusCcw = s_ccw;
 256         }
 257 
 258         if( s_cw != IN_PROGRESS && s_ccw != IN_PROGRESS )
 259             break;
 260 
 261         m_iteration++;
 262     }
 263 
 264     if( s_cw == IN_PROGRESS )
 265     {
 266         result.lineCw = path_cw;
 267         result.statusCw = ALMOST_DONE;
 268     }
 269 
 270     if( s_ccw == IN_PROGRESS )
 271     {
 272         result.lineCcw = path_ccw;
 273         result.statusCcw = ALMOST_DONE;
 274     }
 275 
 276     result.lineCw.Line().Simplify();
 277     result.lineCcw.Line().Simplify();
 278 
 279     if( result.lineCw.SegmentCount() < 1 || result.lineCw.CPoint( 0 ) != aInitialPath.CPoint( 0 ) )
 280     {
 281         result.statusCw = STUCK;
 282     }
 283 
 284     if( result.lineCw.PointCount() > 0 && result.lineCw.CPoint( -1 ) != aInitialPath.CPoint( -1 ) )
 285     {
 286         result.statusCw = ALMOST_DONE;
 287     }
 288 
 289     if( result.lineCcw.SegmentCount() < 1 || result.lineCcw.CPoint( 0 ) != aInitialPath.CPoint( 0 ) )
 290     {
 291         result.statusCcw = STUCK;
 292     }
 293 
 294     if( result.lineCcw.PointCount() > 0 && result.lineCcw.CPoint( -1 ) != aInitialPath.CPoint( -1 ) )
 295     {
 296         result.statusCcw = ALMOST_DONE;
 297     }
 298 
 299     return result;
 300 }

 304 WALKAROUND::WALKAROUND_STATUS WALKAROUND::Route( const LINE& aInitialPath,
 305         LINE& aWalkPath, bool aOptimize )
 306 {
 307     LINE path_cw( aInitialPath ), path_ccw( aInitialPath );
 308     WALKAROUND_STATUS s_cw = IN_PROGRESS, s_ccw = IN_PROGRESS;
 309     SHAPE_LINE_CHAIN best_path;
 310 
 311     // special case for via-in-the-middle-of-track placement

 312     if( aInitialPath.PointCount() <= 1 )
 313     {
 314         if( aInitialPath.EndsWithVia() && m_world->CheckColliding( &aInitialPath.Via(), m_itemMask ) )
 315             return STUCK;
 316 
 317         aWalkPath = aInitialPath;
 318         return DONE;
 319     }
 320 
 321     start( aInitialPath );
 322 
 323     m_currentObstacle[0] = m_currentObstacle[1] = nearestObstacle( aInitialPath );
 324     m_recursiveBlockageCount = 0;
 325 
 326     aWalkPath = aInitialPath;
 327 
 328     if( m_forceWinding )
 329     {
 330         s_cw = m_forceCw ? IN_PROGRESS : STUCK;
 331         s_ccw = m_forceCw ? STUCK : IN_PROGRESS;
 332         m_forceSingleDirection = true;
 333     } else {
 334         m_forceSingleDirection = false;
 335     }
 336 
 337     while( m_iteration < m_iterationLimit )
 338     {
 339         if( s_cw != STUCK )
 340             s_cw = singleStep( path_cw, true );
 341 
 342         if( s_ccw != STUCK )
 343             s_ccw = singleStep( path_ccw, false );
 344 
 345         if( ( s_cw == DONE && s_ccw == DONE ) || ( s_cw == STUCK && s_ccw == STUCK ) )
 346         {
 347             int len_cw  = path_cw.CLine().Length();
 348             int len_ccw = path_ccw.CLine().Length();
 349 
 350             if( m_forceLongerPath )
 351                 aWalkPath = ( len_cw > len_ccw ? path_cw : path_ccw );
 352             else
 353                 aWalkPath = ( len_cw < len_ccw ? path_cw : path_ccw );
 354 
 355             break;
 356         }
 357         else if( s_cw == DONE && !m_forceLongerPath )
 358         {
 359             aWalkPath = path_cw;
 360             break;
 361         }
 362         else if( s_ccw == DONE && !m_forceLongerPath )
 363         {
 364             aWalkPath = path_ccw;
 365             break;
 366         }
 367 
 368         m_iteration++;
 369     }
 370 
 371     if( m_iteration == m_iterationLimit )
 372     {
 373         int len_cw  = path_cw.CLine().Length();
 374         int len_ccw = path_ccw.CLine().Length();
 375 
 376         if( m_forceLongerPath )
 377             aWalkPath = ( len_cw > len_ccw ? path_cw : path_ccw );
 378         else
 379             aWalkPath = ( len_cw < len_ccw ? path_cw : path_ccw );
 380     }
 381 
 382     if( m_cursorApproachMode )
 383     {
 384         // int len_cw = path_cw.GetCLine().Length();

 385         // int len_ccw = path_ccw.GetCLine().Length();

 386         bool found = false;
 387 
 388         SHAPE_LINE_CHAIN l = aWalkPath.CLine();
 389 
 390         for( int i = 0; i < l.SegmentCount(); i++ )
 391         {
 392             const SEG s = l.Segment( i );
 393 
 394             VECTOR2I nearest = s.NearestPoint( m_cursorPos );
 395             VECTOR2I::extended_type dist_a = ( s.A - m_cursorPos ).SquaredEuclideanNorm();
 396             VECTOR2I::extended_type dist_b = ( s.B - m_cursorPos ).SquaredEuclideanNorm();
 397             VECTOR2I::extended_type dist_n = ( nearest - m_cursorPos ).SquaredEuclideanNorm();
 398 
 399             if( dist_n <= dist_a && dist_n < dist_b )
 400             {
 401                 l.Remove( i + 1, -1 );
 402                 l.Append( nearest );
 403                 l.Simplify();
 404                 found = true;
 405                 break;
 406             }
 407         }
 408 
 409         if( found )
 410         {
 411             aWalkPath = aInitialPath;
 412             aWalkPath.SetShape( l );
 413         }
 414     }
 415 
 416     aWalkPath.Line().Simplify();
 417 
 418     if( aWalkPath.SegmentCount() < 1 )
 419         return STUCK;
 420     if( aWalkPath.CPoint( -1 ) != aInitialPath.CPoint( -1 ) )
 421         return ALMOST_DONE;
 422     if( aWalkPath.CPoint( 0 ) != aInitialPath.CPoint( 0 ) )
 423         return STUCK;
 424 
 425     WALKAROUND_STATUS st = s_ccw == DONE || s_cw == DONE ? DONE : STUCK;
 426 
 427     if( st == DONE )
 428     {
 429         if( aOptimize )
 430             OPTIMIZER::Optimize( &aWalkPath, OPTIMIZER::MERGE_OBTUSE, m_world );
 431     }
 432 
 433     return st;
 434 }
