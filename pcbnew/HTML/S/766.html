  60 /**

  61  * EDIT_TOOL

  62  *

  63  * The interactive edit tool. Allows one to move, rotate, flip and change properties of items selected

  64  * using the pcbnew.InteractiveSelection tool.

  65  */
  66 
  67 class EDIT_TOOL : public PCB_TOOL_BASE
  68 {
  69 public:
  70     EDIT_TOOL();
  71 
  72     /// @copydoc TOOL_INTERACTIVE::Reset()

  73     void Reset( RESET_REASON aReason ) override;
  74 
  75     /// @copydoc TOOL_INTERACTIVE::Init()

  76     bool Init() override;
  77 
  78     ///> Find an item and start moving.

  79     int GetAndPlace( const TOOL_EVENT& aEvent );
  80 
  81     /**

  82      * Function Move()

  83      * Main loop in which events are handled.

  84      */
  85     int Move( const TOOL_EVENT& aEvent );
  86 
  87     /**

  88      * Function Drag()

  89      * Invoke the PNS router to drag tracks.

  90      */
  91     int Drag( const TOOL_EVENT& aEvent );
  92 
  93     /**

  94      * Function Properties()

  95      * Displays properties window for the selected object.

  96      */
  97     int Properties( const TOOL_EVENT& aEvent );
  98 
  99     /**

 100      * Function Rotate()

 101      * Rotates currently selected items.

 102      */
 103     int Rotate( const TOOL_EVENT& aEvent );
 104 
 105     /**

 106      * Function Flip()

 107      * Rotates currently selected items. The rotation point is the current cursor position.

 108      */
 109     int Flip( const TOOL_EVENT& aEvent );
 110 
 111     /**

 112      * Function Mirror

 113      * Mirrors the current selection. The mirror axis passes through the current point.

 114      */
 115     int Mirror( const TOOL_EVENT& aEvent );
 116 
 117     int ChangeTrackWidth( const TOOL_EVENT& aEvent );
 118 
 119     /**

 120      * Function Remove()

 121      * Deletes currently selected items. The rotation point is the current cursor position.

 122      */
 123     int Remove( const TOOL_EVENT& aEvent );
 124 
 125     /**

 126      * Function Duplicate()

 127      * Duplicates the current selection and starts a move action.

 128      */
 129     int Duplicate( const TOOL_EVENT& aEvent );
 130 
 131     /**

 132      * Function MoveExact()

 133      * Invokes a dialog box to allow moving of the item by an exact amount.

 134      */
 135     int MoveExact( const TOOL_EVENT& aEvent );
 136 
 137     /**

 138      * Moves an item but with a reference point selected first

 139      */
 140     int MoveWithReference( const TOOL_EVENT& aEvent );
 141 
 142     /**

 143      * Function CreateArray()

 144      * Creates an array of the selected items, invoking the array editor dialog to set the options.

 145      */
 146     int CreateArray( const TOOL_EVENT& aEvent );
 147 
 148     /**

 149      * Function FootprintFilter()

 150      * A selection filter which prunes the selection to contain only items of type PCB_MODULE_T

 151      */
 152     static void FootprintFilter( const VECTOR2I&, GENERAL_COLLECTOR& aCollector );
 153 
 154     /**

 155      * Function PadFilter()

 156      * A selection filter which prunes the selection to contain only items of type PCB_PAD_T

 157      */
 158     static void PadFilter( const VECTOR2I&, GENERAL_COLLECTOR& aCollector );
 159 
 160     ///> Sets up handlers for various events.

 161     void setTransitions() override;
 162 
 163     /**

 164      * Function copyToClipboard()

 165      * Sends the current selection to the clipboard by formatting it as a fake pcb

 166      * see AppendBoardFromClipboard for importing

 167      */
 168     int copyToClipboard( const TOOL_EVENT& aEvent );
 169 
 170     /**

 171      * Function cutToClipboard()

 172      * Cuts the current selection to the clipboard by formatting it as a fake pcb

 173      * see AppendBoardFromClipboard for importing

 174      */
 175     int cutToClipboard( const TOOL_EVENT& aEvent );
 176 
 177     BOARD_COMMIT* GetCurrentCommit() const { return m_commit.get(); }
 178 
 179 private:
 180     ///> Returns the right modification point (e.g. for rotation), depending on the number of

 181     ///> selected items.

 182     bool updateModificationPoint( PCBNEW_SELECTION& aSelection );
 183 
 184     bool invokeInlineRouter( int aDragMode );
 185     bool isInteractiveDragEnabled() const;
 186 
 187     int doMoveSelection( const TOOL_EVENT& aEvent, bool aPickReference = false );
 188 
 189     bool pickReferencePoint( const wxString& aTooltip, const wxString& aSuccessMessage,
 190                              const wxString& aCanceledMessage, VECTOR2I& aReferencePoint );
 191 
 192 private:
 193     SELECTION_TOOL* m_selectionTool;   // Selection tool used for obtaining selected items

 194     bool            m_dragging;        // Indicates objects are being dragged right now

 195     bool            m_lockedSelected;  // Determines if we prompt before removing locked objects

 196     VECTOR2I        m_cursor;          // Last cursor position (needed for getModificationPoint()

 197                                        // to avoid changes of edit reference point).

 198     std::unique_ptr<BOARD_COMMIT> m_commit;
 199 };
