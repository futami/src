 110 PNS_PCBNEW_RULE_RESOLVER::PNS_PCBNEW_RULE_RESOLVER( BOARD* aBoard, PNS::ROUTER_IFACE* aRouterIface ) :
 111     m_routerIface( aRouterIface ),
 112     m_board( aBoard )
 113 {
 114     m_netClearanceCache.resize( m_board->GetNetCount() );
 115 
 116     // Build clearance cache for net classes

 117     for( unsigned int i = 0; i < m_board->GetNetCount(); i++ )
 118     {
 119         NETINFO_ITEM* ni = m_board->FindNet( i );
 120 
 121         if( ni == NULL )
 122             continue;
 123 
 124         CLEARANCE_ENT ent;
 125         ent.coupledNet = DpCoupledNet( i );
 126 
 127         wxString netClassName = ni->GetClassName();
 128         NETCLASSPTR nc = m_board->GetDesignSettings().GetNetClasses().Find( netClassName );
 129 
 130         int clearance = nc->GetClearance();
 131         ent.clearance = clearance;
 132         ent.dpClearance = nc->GetDiffPairGap();
 133         m_netClearanceCache[i] = ent;
 134 
 135         wxLogTrace( "PNS", "Add net %u netclass %s clearance %d Diff Pair clearance %d",
 136                 i, netClassName.mb_str(), clearance, ent.dpClearance );
 137     }
 138 
 139     // Build clearance cache for pads

 140     for( auto mod : m_board->Modules() )
 141     {
 142         auto moduleClearance = mod->GetLocalClearance();
 143 
 144         for( auto pad : mod->Pads() )
 145         {
 146             int padClearance = pad->GetLocalClearance();
 147 
 148             if( padClearance > 0 )
 149                 m_localClearanceCache[ pad ] = padClearance;
 150 
 151             else if( moduleClearance > 0 )
 152                 m_localClearanceCache[ pad ] = moduleClearance;
 153         }
 154     }
 155 
 156     auto defaultRule = m_board->GetDesignSettings().GetNetClasses().Find ("Default");
 157 
 158     if( defaultRule )
 159     {
 160         m_defaultClearance = defaultRule->GetClearance();
 161     }
 162     else
 163     {
 164         m_defaultClearance = Millimeter2iu(0.254);
 165     }
 166 }

 174 int PNS_PCBNEW_RULE_RESOLVER::holeRadius( const PNS::ITEM* aItem ) const
 175 {
 176     if( aItem->Kind() == PNS::ITEM::SOLID_T )
 177     {
 178         const D_PAD* pad = dynamic_cast<const D_PAD*>( aItem->Parent() );
 179 
 180         if( pad && pad->GetDrillShape() == PAD_DRILL_SHAPE_CIRCLE )
 181             return pad->GetDrillSize().x / 2;
 182     }
 183     else if( aItem->Kind() == PNS::ITEM::VIA_T )
 184     {
 185         const ::VIA* via = dynamic_cast<const ::VIA*>( aItem->Parent() );
 186 
 187         if( via )
 188             return via->GetDrillValue() / 2;
 189     }
 190 
 191     return 0;
 192 }

 195 bool PNS_PCBNEW_RULE_RESOLVER::CollideHoles( const PNS::ITEM* aA, const PNS::ITEM* aB,
 196                                              bool aNeedMTV, VECTOR2I* aMTV ) const
 197 {
 198     VECTOR2I pos_a = aA->Shape()->Centre();
 199     VECTOR2I pos_b = aB->Shape()->Centre();
 200 
 201     // Holes with identical locations are allowable

 202     if( pos_a == pos_b )
 203         return false;
 204 
 205     int radius_a = holeRadius( aA );
 206     int radius_b = holeRadius( aB );
 207 
 208     // Do both objects have holes?

 209     if( radius_a > 0 && radius_b > 0 )
 210     {
 211         int holeToHoleMin = m_board->GetDesignSettings().m_HoleToHoleMin;
 212 
 213         ecoord min_dist = holeToHoleMin + radius_a + radius_b;
 214         ecoord min_dist_sq = min_dist * min_dist;
 215 
 216         const VECTOR2I delta = pos_b - pos_a;
 217 
 218         ecoord dist_sq = delta.SquaredEuclideanNorm();
 219 
 220         if( dist_sq == 0 || dist_sq < min_dist_sq )
 221         {
 222             if( aNeedMTV )
 223                 *aMTV = delta.Resize( min_dist - sqrt( dist_sq ) + 3 );  // fixme: apparent rounding error

 224 
 225             return true;
 226         }
 227     }
 228 
 229     return false;
 230 }

 233 int PNS_PCBNEW_RULE_RESOLVER::localPadClearance( const PNS::ITEM* aItem ) const
 234 {
 235     if( !aItem->Parent() || aItem->Parent()->Type() != PCB_PAD_T )
 236         return 0;
 237 
 238     const D_PAD* pad = static_cast<D_PAD*>( aItem->Parent() );
 239 
 240     auto i = m_localClearanceCache.find( pad );
 241 
 242     if( i == m_localClearanceCache.end() )
 243         return 0;
 244 
 245     return i->second;
 246 }

 249 int PNS_PCBNEW_RULE_RESOLVER::Clearance( const PNS::ITEM* aA, const PNS::ITEM* aB ) const
 250 {
 251     int net_a = aA->Net();
 252     int cl_a = Clearance( net_a );
 253     int net_b = aB->Net();
 254     int cl_b = Clearance( net_b );
 255 
 256     // Pad clearance is 0 if the ITEM* is not a pad

 257     int pad_a = localPadClearance( aA );
 258     int pad_b = localPadClearance( aB );
 259 
 260     if( pad_a > 0 )
 261         cl_a = pad_a;
 262 
 263     if( pad_b > 0 )
 264         cl_b = pad_b;
 265 
 266     return std::max( cl_a, cl_b );
 267 }

 281 int PNS_PCBNEW_RULE_RESOLVER::matchDpSuffix( const wxString& aNetName, wxString& aComplementNet,
 282                                              wxString& aBaseDpName )
 283 {
 284     int rv = 0;
 285 
 286     if( aNetName.EndsWith( "+" ) )
 287     {
 288         aComplementNet = "-";
 289         rv = 1;
 290     }
 291     else if( aNetName.EndsWith( "P" ) )
 292     {
 293         aComplementNet = "N";
 294         rv = 1;
 295     }
 296     else if( aNetName.EndsWith( "-" ) )
 297     {
 298         aComplementNet = "+";
 299         rv = -1;
 300     }
 301     else if( aNetName.EndsWith( "N" ) )
 302     {
 303         aComplementNet = "P";
 304         rv = -1;
 305     }
 306     // Match P followed by 2 digits

 307     else if( aNetName.Right( 2 ).IsNumber() && aNetName.Right( 3 ).Left( 1 ) == "P" )
 308     {
 309         aComplementNet = "N" + aNetName.Right( 2 );
 310         rv = 1;
 311     }
 312     // Match P followed by 1 digit

 313     else if( aNetName.Right( 1 ).IsNumber() && aNetName.Right( 2 ).Left( 1 ) == "P" )
 314     {
 315         aComplementNet = "N" + aNetName.Right( 1 );
 316         rv = 1;
 317     }
 318     // Match N followed by 2 digits

 319     else if( aNetName.Right( 2 ).IsNumber() && aNetName.Right( 3 ).Left( 1 ) == "N" )
 320     {
 321         aComplementNet = "P" + aNetName.Right( 2 );
 322         rv = -1;
 323     }
 324     // Match N followed by 1 digit

 325     else if( aNetName.Right( 1 ).IsNumber() && aNetName.Right( 2 ).Left( 1 ) == "N" )
 326     {
 327         aComplementNet = "P" + aNetName.Right( 1 );
 328         rv = -1;
 329     }
 330     if( rv != 0 )
 331     {
 332         aBaseDpName = aNetName.Left( aNetName.Length() - aComplementNet.Length() );
 333         aComplementNet = aBaseDpName + aComplementNet;
 334     }
 335 
 336     return rv;
 337 }

 340 int PNS_PCBNEW_RULE_RESOLVER::DpCoupledNet( int aNet )
 341 {
 342     wxString refName = m_board->FindNet( aNet )->GetNetname();
 343     wxString dummy, coupledNetName;
 344 
 345     if( matchDpSuffix( refName, coupledNetName, dummy ) )
 346     {
 347         NETINFO_ITEM* net = m_board->FindNet( coupledNetName );
 348 
 349         if( !net )
 350             return -1;
 351 
 352         return net->GetNet();
 353     }
 354 
 355     return -1;
 356 }

 374 bool PNS_PCBNEW_RULE_RESOLVER::DpNetPair( PNS::ITEM* aItem, int& aNetP, int& aNetN )
 375 {
 376     if( !aItem || !aItem->Parent() || !aItem->Parent()->GetNet() )
 377         return false;
 378 
 379     wxString netNameP = aItem->Parent()->GetNet()->GetNetname();
 380     wxString netNameN, netNameCoupled, netNameBase;
 381 
 382     int r = matchDpSuffix( netNameP, netNameCoupled, netNameBase );
 383 
 384     if( r == 0 )
 385         return false;
 386     else if( r == 1 )
 387     {
 388         netNameN = netNameCoupled;
 389     }
 390     else
 391     {
 392         netNameN = netNameP;
 393         netNameP = netNameCoupled;
 394     }
 395 
 396 //    wxLogTrace( "PNS","p %s n %s base %s\n", (const char *)netNameP.c_str(), (const char *)netNameN.c_str(), (const char *)netNameBase.c_str() );

 397 
 398     NETINFO_ITEM* netInfoP = m_board->FindNet( netNameP );
 399     NETINFO_ITEM* netInfoN = m_board->FindNet( netNameN );
 400 
 401     //wxLogTrace( "PNS","ip %p in %p\n", netInfoP, netInfoN);

 402 
 403     if( !netInfoP || !netInfoN )
 404         return false;
 405 
 406     aNetP = netInfoP->GetNet();
 407     aNetN = netInfoN->GetNet();
 408 
 409     return true;
 410 }

 569 std::unique_ptr<PNS::SOLID> PNS_KICAD_IFACE_BASE::syncPad( D_PAD* aPad )
 570 {
 571     LAYER_RANGE layers( 0, MAX_CU_LAYERS - 1 );
 572 
 573     // ignore non-copper pads

 574     if( ( aPad->GetLayerSet() & LSET::AllCuMask()).none() )
 575         return NULL;
 576 
 577     switch( aPad->GetAttribute() )
 578     {
 579     case PAD_ATTRIB_STANDARD:
 580         break;
 581 
 582     case PAD_ATTRIB_SMD:
 583     case PAD_ATTRIB_HOLE_NOT_PLATED:
 584     case PAD_ATTRIB_CONN:
 585         {
 586             LSET lmsk = aPad->GetLayerSet();
 587             bool is_copper = false;
 588 
 589             for( int i = 0; i < MAX_CU_LAYERS; i++ )
 590             {
 591                 if( lmsk[i] )
 592                 {
 593                     is_copper = true;
 594 
 595                     if( aPad->GetAttribute() != PAD_ATTRIB_HOLE_NOT_PLATED )
 596                         layers = LAYER_RANGE( i );
 597 
 598                     break;
 599                 }
 600             }
 601 
 602             if( !is_copper )
 603                 return NULL;
 604         }
 605         break;
 606 
 607     default:
 608         wxLogTrace( "PNS", "unsupported pad type 0x%x", aPad->GetAttribute() );
 609         return NULL;
 610     }
 611 
 612     std::unique_ptr< PNS::SOLID > solid( new PNS::SOLID );
 613 
 614     solid->SetLayers( layers );
 615     solid->SetNet( aPad->GetNetCode() );
 616     solid->SetParent( aPad );
 617     solid->SetPadToDie( aPad->GetPadToDieLength() );
 618 
 619     wxPoint wx_c = aPad->ShapePos();
 620     wxSize  wx_sz = aPad->GetSize();
 621     wxPoint offset = aPad->GetOffset();
 622 
 623     VECTOR2I c( wx_c.x, wx_c.y );
 624 
 625     RotatePoint( &offset, aPad->GetOrientation() );
 626 
 627     solid->SetPos( VECTOR2I( c.x - offset.x, c.y - offset.y ) );
 628     solid->SetOffset( VECTOR2I( offset.x, offset.y ) );
 629 
 630     if( aPad->GetEffectiveShapes().size() == 1 )
 631     {
 632         solid->SetShape( aPad->GetEffectiveShapes()[0]->Clone() );
 633     }
 634     else
 635     {
 636         // JEY TODO:

 637         // TOM TODO: move to SHAPE_COMPOUND...

 638 
 639         const std::shared_ptr<SHAPE_POLY_SET>& outline = aPad->GetEffectivePolygon();
 640         SHAPE_SIMPLE*                          shape = new SHAPE_SIMPLE();
 641 
 642         for( auto iter = outline->CIterate( 0 ); iter; iter++ )
 643             shape->Append( *iter );
 644 
 645         solid->SetShape( shape );
 646     }
 647 
 648     return solid;
 649 }

 652 std::unique_ptr<PNS::SEGMENT> PNS_KICAD_IFACE_BASE::syncTrack( TRACK* aTrack )
 653 {
 654     std::unique_ptr< PNS::SEGMENT > segment(
 655         new PNS::SEGMENT( SEG( aTrack->GetStart(), aTrack->GetEnd() ), aTrack->GetNetCode() )
 656     );
 657 
 658     segment->SetWidth( aTrack->GetWidth() );
 659     segment->SetLayers( LAYER_RANGE( aTrack->GetLayer() ) );

 708 bool PNS_KICAD_IFACE_BASE::syncZone( PNS::NODE* aWorld, ZONE_CONTAINER* aZone )
 709 {
 710     SHAPE_POLY_SET poly;
 711 
 712     // TODO handle no-via restriction

 713     if( !aZone->GetIsKeepout() || !aZone->GetDoNotAllowTracks() )
 714         return false;
 715 
 716     // Some intersecting zones, despite being on the same layer with the same net, cannot be

 717     // merged due to other parameters such as fillet radius.  The copper pour will end up

 718     // effectively merged though, so we want to keep the corners of such intersections sharp.

 719     std::set<VECTOR2I> colinearCorners;
 720     aZone->GetColinearCorners( m_board, colinearCorners );
 721 
 722     aZone->BuildSmoothedPoly( poly, &colinearCorners );
 723     poly.CacheTriangulation();
 724 
 725     if( !poly.IsTriangulationUpToDate() )
 726     {
 727         KIDIALOG dlg( nullptr, wxString::Format( _( "Malformed keep-out zone at (%d, %d)" ),
 728                 aZone->GetPosition().x, aZone->GetPosition().y ), KIDIALOG::KD_WARNING );
 729         dlg.ShowDetailedText(
 730                 wxString::Format( _( "%s\nThis zone cannot be handled by the track layout tool.\n"
 731                                      "Please verify it is not a self-intersecting polygon." ),
 732                         aZone->GetSelectMenuText( EDA_UNITS::MILLIMETRES ) ) );
 733         dlg.DoNotShowCheckbox( __FILE__, __LINE__ );
 734         dlg.ShowModal();
 735 
 736         return false;
 737     }
 738 
 739     LSET layers = aZone->GetLayerSet();
 740 
 741     for( int layer = F_Cu; layer <= B_Cu; layer++ )
 742     {
 743         if ( ! layers[layer] )
 744             continue;
 745 
 746         for( int outline = 0; outline < poly.OutlineCount(); outline++ )
 747         {
 748             auto tri = poly.TriangulatedPolygon( outline );
 749 
 750             for( size_t i = 0; i < tri->GetTriangleCount(); i++)
 751             {
 752                 VECTOR2I a, b, c;
 753                 tri->GetTriangle( i, a, b, c );
 754                 auto triShape = new SHAPE_SIMPLE;
 755 
 756                 triShape->Append( a );
 757                 triShape->Append( b );
 758                 triShape->Append( c );
 759 
 760                 std::unique_ptr< PNS::SOLID > solid( new PNS::SOLID );
 761 
 762                 solid->SetLayer( layer );
 763                 solid->SetNet( -1 );
 764                 solid->SetParent( aZone );
 765                 solid->SetShape( triShape );
 766                 solid->SetRoutable( false );
 767 
 768                 aWorld->Add( std::move( solid ) );
 769             }
 770         }
 771     }
 772 
 773     return true;
 774 }

 778 {
 779     if( !IsCopperLayer( aLayer ) )
 780         return false;
 781 
 782     int textWidth = aText->GetEffectiveTextPenWidth();
 783     std::vector<wxPoint> textShape;

 785     aText->TransformTextShapeToSegmentList( textShape );

 788         return false;

 791     {
 792         VECTOR2I start( textShape[jj] );
 793         VECTOR2I end( textShape[jj+1] );

 797         solid->SetNet( -1 );
 798         solid->SetParent( nullptr );

 801 
 802         aWorld->Add( std::move( solid ) );
 803     }
 804 
 805     return true;
 806 
 807     /* A coarser (but faster) method:

 808      *

 809     SHAPE_POLY_SET outline;

 810     SHAPE_SIMPLE* shape = new SHAPE_SIMPLE();

 811 

 812     aText->TransformBoundingBoxWithClearanceToPolygon( &outline, 0 );

 813 

 814     for( auto iter = outline.CIterate( 0 ); iter; iter++ )

 815         shape->Append( *iter );

 816 

 817     solid->SetShape( shape );

 818 

 819     solid->SetLayer( aLayer );

 820     solid->SetNet( -1 );

 821     solid->SetParent( nullptr );

 822     solid->SetRoutable( false );

 823     aWorld->Add( std::move( solid ) );

 824     return true;

 825      */
 826 }

 879 bool PNS_KICAD_IFACE::IsItemVisible( const PNS::ITEM* aItem )
 880 {
 881     // by default, all items are visible (new ones created by the router have parent == NULL as they have not been

 882     // committed yet to the BOARD)

 883     if( !m_view || !aItem->Parent() )
 884         return true;
 885 
 886     auto item = aItem->Parent();
 887     bool isOnVisibleLayer = true;
 888 
 889     if( m_view->GetPainter()->GetSettings()->GetHighContrast() )
 890     {
 891         int  layers[KIGFX::VIEW::VIEW_MAX_LAYERS];
 892         int  layers_count;
 893         auto activeLayers = m_view->GetPainter()->GetSettings()->GetActiveLayers();
 894 
 895         isOnVisibleLayer = false;
 896         item->ViewGetLayers( layers, layers_count );
 897 
 898         for( int i = 0; i < layers_count; ++i )
 899         {
 900             // Item is on at least one of the active layers

 901             if( activeLayers.count( layers[i] ) > 0 )
 902             {
 903                 isOnVisibleLayer = true;
 904                 break;
 905             }
 906         }
 907     }
 908 
 909     if( m_view->IsVisible( item ) && isOnVisibleLayer
 910             && item->ViewGetLOD( item->GetLayer(), m_view ) < m_view->GetScale() )
 911         return true;
 912 
 913     // Items hidden in the router are not hidden on the board

 914     if( m_hiddenItems.find( item ) != m_hiddenItems.end() )
 915         return true;
 916 
 917     return false;
 918 }

 921 void PNS_KICAD_IFACE_BASE::SyncWorld( PNS::NODE *aWorld )
 922 {
 923     int worstPadClearance = 0;
 924 
 925     m_world = aWorld;
 926 
 927     if( !m_board )
 928     {
 929         wxLogTrace( "PNS", "No board attached, aborting sync." );
 930         return;
 931     }
 932 
 933     for( auto gitem : m_board->Drawings() )
 934     {
 935         if ( gitem->Type() == PCB_LINE_T )
 936         {
 937             syncGraphicalItem( aWorld, static_cast<DRAWSEGMENT*>( gitem ) );
 938         }
 939         else if( gitem->Type() == PCB_TEXT_T )
 940         {
 941             syncTextItem( aWorld, static_cast<TEXTE_PCB*>( gitem ), gitem->GetLayer() );
 942         }
 943     }
 944 
 945     for( auto zone : m_board->Zones() )
 946     {
 947         syncZone( aWorld, zone );
 948     }
 949 
 950     for( auto module : m_board->Modules() )
 951     {
 952         for( auto pad : module->Pads() )
 953         {
 954             if( auto solid = syncPad( pad ) )
 955                 aWorld->Add( std::move( solid ) );
 956 
 957             worstPadClearance = std::max( worstPadClearance, pad->GetLocalClearance() );
 958         }
 959 
 960         syncTextItem( aWorld, &module->Reference(), module->Reference().GetLayer() );
 961         syncTextItem( aWorld, &module->Value(), module->Value().GetLayer() );
 962 
 963         for( MODULE_ZONE_CONTAINER* zone : module->Zones() )
 964             syncZone( aWorld, zone );
 965 
 966         if( module->IsNetTie() )
 967             continue;
 968 
 969         for( auto mgitem : module->GraphicalItems() )
 970         {
 971             if( mgitem->Type() == PCB_MODULE_EDGE_T )
 972             {
 973                 syncGraphicalItem( aWorld, static_cast<DRAWSEGMENT*>( mgitem ) );
 974             }
 975             else if( mgitem->Type() == PCB_MODULE_TEXT_T )
 976             {
 977                 syncTextItem( aWorld, static_cast<TEXTE_MODULE*>( mgitem ), mgitem->GetLayer() );
 978             }
 979         }
 980     }
 981 
 982     for( auto t : m_board->Tracks() )
 983     {
 984         KICAD_T type = t->Type();
 985 
 986         if( type == PCB_TRACE_T )
 987         {
 988             if( auto segment = syncTrack( t ) )
 989                 aWorld->Add( std::move( segment ) );
 990         }
 991         else if( type == PCB_ARC_T )
 992         {
 993             if( auto arc = syncArc( static_cast<ARC*>( t ) ) )
 994                 aWorld->Add( std::move( arc ) );
 995         }
 996         else if( type == PCB_VIA_T )
 997         {
 998             if( auto via = syncVia( static_cast<VIA*>( t ) ) )
 999                 aWorld->Add( std::move( via ) );
1000         }
1001     }
1002 
1003     int worstRuleClearance = m_board->GetDesignSettings().GetBiggestClearanceValue();
1004 
1005     delete m_ruleResolver;
1006     m_ruleResolver = new PNS_PCBNEW_RULE_RESOLVER( m_board, this );
1007 
1008     aWorld->SetRuleResolver( m_ruleResolver );
1009     aWorld->SetMaxClearance( 4 * std::max(worstPadClearance, worstRuleClearance ) );
1010 }

1125 void PNS_KICAD_IFACE::AddItem( PNS::ITEM* aItem )
1126 {
1127     BOARD_CONNECTED_ITEM* newBI = NULL;
1128 
1129     switch( aItem->Kind() )
1130     {
1131     case PNS::ITEM::ARC_T:
1132     {
1133         auto arc = static_cast<PNS::ARC*>( aItem );
1134         ARC* new_arc = new ARC( m_board, static_cast<const SHAPE_ARC*>( arc->Shape() ) );
1135         new_arc->SetWidth( arc->Width() );
1136         new_arc->SetLayer( ToLAYER_ID( arc->Layers().Start() ) );
1137         new_arc->SetNetCode( std::max<int>( 0, arc->Net() ) );
1138         newBI = new_arc;
1139         break;
1140     }
1141 
1142     case PNS::ITEM::SEGMENT_T:
1143     {
1144         PNS::SEGMENT* seg = static_cast<PNS::SEGMENT*>( aItem );
1145         TRACK* track = new TRACK( m_board );
1146         const SEG& s = seg->Seg();
1147         track->SetStart( wxPoint( s.A.x, s.A.y ) );
1148         track->SetEnd( wxPoint( s.B.x, s.B.y ) );
1149         track->SetWidth( seg->Width() );
1150         track->SetLayer( ToLAYER_ID( seg->Layers().Start() ) );
1151         track->SetNetCode( seg->Net() > 0 ? seg->Net() : 0 );
1152         newBI = track;
1153         break;
1154     }
1155 
1156     case PNS::ITEM::VIA_T:
1157     {
1158         VIA* via_board = new VIA( m_board );
1159         PNS::VIA* via = static_cast<PNS::VIA*>( aItem );
1160         via_board->SetPosition( wxPoint( via->Pos().x, via->Pos().y ) );
1161         via_board->SetWidth( via->Diameter() );
1162         via_board->SetDrill( via->Drill() );
1163         via_board->SetNetCode( via->Net() > 0 ? via->Net() : 0 );
1164         via_board->SetViaType( via->ViaType() ); // MUST be before SetLayerPair()

1165         via_board->SetLayerPair( ToLAYER_ID( via->Layers().Start() ),
1166                                  ToLAYER_ID( via->Layers().End() ) );
1167         newBI = via_board;
1168         break;
1169     }
1170 
1171     case PNS::ITEM::SOLID_T:
1172     {
1173         auto pad = static_cast<D_PAD*>( aItem->Parent() );
1174         auto pos = static_cast<PNS::SOLID*>( aItem )->Pos();
1175 
1176         m_moduleOffsets[ pad ].p_new = pos;
1177         return;
1178     }
1179 
1180     default:
1181         break;
1182     }
1183 
1184     if( newBI )
1185     {
1186         //newBI->SetLocalRatsnestVisible( m_dispOptions->m_ShowGlobalRatsnest );

1187         aItem->SetParent( newBI );
1188         newBI->ClearFlags();
1189 
1190         m_commit->Add( newBI );
1191     }
1192 }

1195 void PNS_KICAD_IFACE::Commit()
1196 {
1197     std::set<MODULE*> processedMods;
1198 
1199     EraseView();
1200 
1201     for( auto mo : m_moduleOffsets )
1202     {
1203         auto offset = mo.second.p_new - mo.second.p_old;
1204         auto mod = mo.first->GetParent();
1205 
1206         VECTOR2I p_orig = mod->GetPosition();
1207         VECTOR2I p_new = p_orig + offset;
1208 
1209         if( processedMods.find( mod ) != processedMods.end() )
1210             continue;
1211 
1212         processedMods.insert( mod );
1213         m_commit->Modify( mod );
1214         mod->SetPosition( wxPoint( p_new.x, p_new.y ));
1215     }
1216 
1217     m_moduleOffsets.clear();
1218 
1219     m_commit->Push( _( "Interactive Router" ) );
1220     m_commit = std::make_unique<BOARD_COMMIT>( m_tool );
1221 }
