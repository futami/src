  40 void PCB_BRIGHT_BOX::ViewDraw( int aLayer, KIGFX::VIEW* aView ) const
  41 {
  42     if( !m_item )
  43         return;
  44 
  45     if( m_item->Type() == PCB_TRACE_T )
  46     {
  47         const TRACK* track = static_cast<const TRACK*>( m_item );
  48 
  49         auto gal = aView->GetGAL();
  50 
  51         gal->SetIsStroke( true );
  52         gal->SetIsFill( false );
  53         gal->SetLineWidth( m_lineWidth );
  54         gal->SetStrokeColor( m_color );
  55 
  56         gal->DrawSegment( track->GetStart(), track->GetEnd(), track->GetWidth() );
  57     }
  58     else
  59     {
  60         BRIGHT_BOX::ViewDraw( aLayer, aView );
  61     }
  62 }
