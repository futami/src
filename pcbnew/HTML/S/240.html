  99 void DIALOG_NON_COPPER_ZONES_EDITOR::OnUpdateUI( wxUpdateUIEvent& )
 100 {
 101     if( m_cornerSmoothingType != m_cornerSmoothingChoice->GetSelection() )
 102     {
 103         m_cornerSmoothingType = m_cornerSmoothingChoice->GetSelection();
 104 
 105         if( m_cornerSmoothingChoice->GetSelection() == ZONE_SETTINGS::SMOOTHING_CHAMFER )
 106             m_cornerRadiusLabel->SetLabel( _( "Chamfer distance:" ) );
 107         else
 108             m_cornerRadiusLabel->SetLabel( _( "Fillet radius:" ) );
 109     }
 110 
 111     m_cornerRadiusCtrl->Enable(m_cornerSmoothingType > ZONE_SETTINGS::SMOOTHING_NONE );
 112 }

 115 bool DIALOG_NON_COPPER_ZONES_EDITOR::TransferDataToWindow()
 116 {
 117     m_cornerSmoothingChoice->SetSelection( m_settings.GetCornerSmoothingType() );
 118     m_cornerRadius.SetValue( m_settings.GetCornerRadius() );
 119 
 120     m_minWidth.SetValue( m_settings.m_ZoneMinThickness );
 121     m_ConstrainOpt->SetValue( m_settings.m_Zone_45_Only );
 122 
 123     switch( m_settings.m_Zone_HatchingStyle )
 124     {
 125     case ZONE_HATCH_STYLE::NO_HATCH:
 126         m_OutlineAppearanceCtrl->SetSelection( 0 );
 127         break;
 128     case ZONE_HATCH_STYLE::DIAGONAL_EDGE:
 129         m_OutlineAppearanceCtrl->SetSelection( 1 );
 130         break;
 131     case ZONE_HATCH_STYLE::DIAGONAL_FULL:
 132         m_OutlineAppearanceCtrl->SetSelection( 2 );
 133         break;
 134     }
 135 
 136     SetInitialFocus( m_OutlineAppearanceCtrl );
 137 
 138     switch( m_settings.m_FillMode )
 139     {
 140     case ZONE_FILL_MODE::HATCH_PATTERN:
 141         m_GridStyleCtrl->SetSelection( 1 ); break;
 142     default:
 143         m_GridStyleCtrl->SetSelection( 0 ); break;
 144     }
 145 
 146     m_gridStyleRotation.SetUnits( EDA_UNITS::DEGREES );
 147     m_gridStyleRotation.SetValue( m_settings.m_HatchFillTypeOrientation*10 ); // IU is decidegree

 148 
 149     // Gives a reasonable value to grid style parameters, if currently there are no defined

 150     // parameters for grid pattern thickness and gap (if the value is 0)

 151     // the grid pattern thickness default value is (arbitrary) m_ZoneMinThickness * 4

 152     // or 1mm

 153     // the grid pattern gap default value is (arbitrary) m_ZoneMinThickness * 6

 154     // or 1.5 mm

 155     int bestvalue = m_settings.m_HatchFillTypeThickness;
 156 
 157     if( bestvalue <= 0 )     // No defined value for m_HatchFillTypeThickness

 158         bestvalue = std::max( m_settings.m_ZoneMinThickness * 4, Millimeter2iu( 1.0 ) );
 159 
 160     m_gridStyleThickness.SetValue( std::max( bestvalue, m_settings.m_ZoneMinThickness ) );
 161 
 162     bestvalue = m_settings.m_HatchFillTypeGap;
 163 
 164     if( bestvalue <= 0 )     // No defined value for m_HatchFillTypeGap

 165         bestvalue = std::max( m_settings.m_ZoneMinThickness * 6, Millimeter2iu( 1.5 ) );
 166 
 167     m_gridStyleGap.SetValue( std::max( bestvalue, m_settings.m_ZoneMinThickness ) );
 168 
 169     m_spinCtrlSmoothLevel->SetValue( m_settings.m_HatchFillTypeSmoothingLevel );
 170     m_spinCtrlSmoothValue->SetValue( m_settings.m_HatchFillTypeSmoothingValue );
 171 
 172     // Enable/Disable some widgets

 173     wxCommandEvent event;
 174     OnStyleSelection( event );
 175 
 176     return true;
 177 }

 191 void DIALOG_NON_COPPER_ZONES_EDITOR::OnLayerSelection( wxDataViewEvent& event )
 192 {
 193     if( event.GetColumn() != 0 )
 194         return;
 195 
 196     int row  = m_layers->ItemToRow( event.GetItem() );
 197     bool val = m_layers->GetToggleValue( row, 0 );
 198 
 199     wxVariant layerID;
 200     m_layers->GetValue( layerID, row, 2 );
 201     m_settings.m_Layers.set( ToLAYER_ID( layerID.GetInteger() ), val );
 202 }

 205 bool DIALOG_NON_COPPER_ZONES_EDITOR::TransferDataFromWindow()
 206 {
 207     m_settings.SetCornerSmoothingType( m_cornerSmoothingChoice->GetSelection() );
 208 
 209     m_settings.SetCornerRadius( m_settings.GetCornerSmoothingType() == ZONE_SETTINGS::SMOOTHING_NONE
 210                                 ? 0 : m_cornerRadius.GetValue() );
 211 
 212     if( !m_gridStyleRotation.Validate( -1800, 1800 ) )
 213         return false;
 214 
 215     m_settings.m_ZoneMinThickness = m_minWidth.GetValue();
 216 
 217     switch( m_OutlineAppearanceCtrl->GetSelection() )
 218     {
 219     case 0:
 220         m_settings.m_Zone_HatchingStyle = ZONE_HATCH_STYLE::NO_HATCH;
 221         break;
 222     case 1:
 223         m_settings.m_Zone_HatchingStyle = ZONE_HATCH_STYLE::DIAGONAL_EDGE;
 224         break;
 225     case 2:
 226         m_settings.m_Zone_HatchingStyle = ZONE_HATCH_STYLE::DIAGONAL_FULL;
 227         break;
 228     }
 229 
 230     if( m_GridStyleCtrl->GetSelection() > 0 )
 231         m_settings.m_FillMode = ZONE_FILL_MODE::HATCH_PATTERN;
 232     else
 233         m_settings.m_FillMode = ZONE_FILL_MODE::POLYGONS;
 234 
 235 
 236     if( m_settings.m_FillMode == ZONE_FILL_MODE::HATCH_PATTERN )
 237     {
 238         int minThickness = m_minWidth.GetValue();
 239 
 240         if( !m_gridStyleThickness.Validate( minThickness, INT_MAX ) )
 241             return false;
 242 
 243         if( !m_gridStyleGap.Validate( minThickness, INT_MAX ) )
 244             return false;
 245     }
 246 
 247 
 248     m_settings.m_HatchFillTypeOrientation = m_gridStyleRotation.GetValue()/10.0; // value is returned in deci-degree

 249     m_settings.m_HatchFillTypeThickness = m_gridStyleThickness.GetValue();
 250     m_settings.m_HatchFillTypeGap = m_gridStyleGap.GetValue();
 251     m_settings.m_HatchFillTypeSmoothingLevel = m_spinCtrlSmoothLevel->GetValue();
 252     m_settings.m_HatchFillTypeSmoothingValue = m_spinCtrlSmoothValue->GetValue();
 253 
 254     auto cfg = m_parent->GetPcbNewSettings();
 255     cfg->m_Zones.hatching_style = static_cast<int>( m_settings.m_Zone_HatchingStyle );
 256 
 257     m_settings.m_Zone_45_Only = m_ConstrainOpt->GetValue();
 258 
 259     // Get the layer selection for this zone

 260     int layer = -1;
 261     for( int ii = 0; ii < m_layers->GetItemCount(); ++ii )
 262     {
 263         if( m_layers->GetToggleValue( (unsigned) ii, 0 ) )
 264         {
 265             layer = ii;
 266             break;
 267         }
 268     }
 269 
 270     if( layer < 0 )
 271     {
 272         DisplayError( this, _( "No layer selected." ) );
 273         return false;
 274     }
 275 
 276     *m_ptr = m_settings;
 277     return true;
 278 }
