 249 void DIALOG_PAD_PROPERTIES::prepareCanvas()
 250 {
 251     // Initialize the canvas to display the pad

 252 #ifdef __WXMAC__
 253     // Cairo renderer doesn't handle Retina displays

 254     EDA_DRAW_PANEL_GAL::GAL_TYPE backend = EDA_DRAW_PANEL_GAL::GAL_TYPE_OPENGL;
 255 #else
 256     EDA_DRAW_PANEL_GAL::GAL_TYPE backend = EDA_DRAW_PANEL_GAL::GAL_TYPE_CAIRO;
 257 #endif
 258     m_padPreviewGAL = new PCB_DRAW_PANEL_GAL( this, -1, wxDefaultPosition, wxDefaultSize,
 259                                               m_parent->GetGalDisplayOptions(), backend );
 260 
 261     m_padPreviewSizer->Add( m_padPreviewGAL, 12, wxEXPAND | wxALL, 5 );
 262 
 263     // Show the X and Y axis. It is usefull because pad shape can have an offset

 264     // or be a complex shape.

 265     KIGFX::COLOR4D axis_color = LIGHTBLUE;
 266 
 267     m_axisOrigin = new KIGFX::ORIGIN_VIEWITEM( axis_color, KIGFX::ORIGIN_VIEWITEM::CROSS,
 268                                                Millimeter2iu( 0.2 ),
 269                                                VECTOR2D( m_dummyPad->GetPosition() ) );
 270     m_axisOrigin->SetDrawAtZero( true );
 271 
 272     m_padPreviewGAL->UpdateColors();
 273     m_padPreviewGAL->SetStealsFocus( false );
 274 
 275     m_padPreviewGAL->GetViewControls()->ApplySettings(
 276             m_parent->GetCanvas()->GetViewControls()->GetSettings() );
 277 
 278     m_padPreviewGAL->Show();
 279 
 280     KIGFX::VIEW* view = m_padPreviewGAL->GetView();
 281 
 282     // fix the pad render mode (filled/not filled)

 283     auto settings = static_cast<KIGFX::PCB_RENDER_SETTINGS*>( view->GetPainter()->GetSettings() );
 284     bool sketchMode = m_cbShowPadOutline->IsChecked();
 285     settings->SetSketchMode( LAYER_PADS_TH, sketchMode );
 286     settings->SetSketchMode( LAYER_PAD_FR, sketchMode );
 287     settings->SetSketchMode( LAYER_PAD_BK, sketchMode );
 288     settings->SetSketchModeGraphicItems( sketchMode );
 289 
 290     // gives a non null grid size (0.001mm) because GAL layer does not like a 0 size grid:

 291     double gridsize = 0.001 * IU_PER_MM;
 292     view->GetGAL()->SetGridSize( VECTOR2D( gridsize, gridsize ) );
 293     // And do not show the grid:

 294     view->GetGAL()->SetGridVisibility( false );
 295     view->Add( m_dummyPad );
 296     view->Add( m_axisOrigin );
 297 
 298     m_padPreviewGAL->StartDrawing();
 299     Connect( wxEVT_SIZE, wxSizeEventHandler( DIALOG_PAD_PROPERTIES::OnResize ) );
 300 }

 322 void DIALOG_PAD_PROPERTIES::onCornerRadiusChange( wxCommandEvent& event )
 323 {
 324     if( m_dummyPad->GetShape() != PAD_SHAPE_ROUNDRECT &&
 325         m_dummyPad->GetShape() != PAD_SHAPE_CHAMFERED_RECT )
 326         return;
 327 
 328     double rrRadius = m_cornerRadius.GetValue();
 329 
 330     if( rrRadius < 0.0 )
 331     {
 332         rrRadius = 0.0;
 333         m_tcCornerRadius->ChangeValue( wxString::Format( "%.1f", rrRadius ) );
 334     }
 335 
 336     transferDataToPad( m_dummyPad );
 337     m_dummyPad->SetRoundRectCornerRadius( rrRadius );
 338 
 339     auto ratio = wxString::Format( "%.1f", m_dummyPad->GetRoundRectRadiusRatio() * 100 );
 340     m_tcCornerSizeRatio->ChangeValue( ratio );
 341     m_tcCornerSizeRatio1->ChangeValue( ratio );
 342     redraw();
 343 }

 346 void DIALOG_PAD_PROPERTIES::onCornerSizePercentChange( wxCommandEvent& event )
 347 {
 348     if( m_dummyPad->GetShape() != PAD_SHAPE_ROUNDRECT &&
 349         m_dummyPad->GetShape() != PAD_SHAPE_CHAMFERED_RECT )
 350     {
 351         return;
 352     }
 353 
 354     wxObject* ctrl = event.GetEventObject();
 355     wxString  value = event.GetString();
 356     bool      changed = false;
 357 
 358     if( ctrl == m_tcCornerSizeRatio || ctrl == m_tcCornerSizeRatio1 )
 359     {
 360         double ratioPercent;
 361 
 362         if( value.ToDouble( &ratioPercent ) )
 363         {
 364             // Clamp ratioPercent to acceptable value (0.0 to 50.0)

 365             if( ratioPercent < 0.0 )
 366             {
 367                 ratioPercent = 0.0;
 368                 value.Printf( "%.1f", ratioPercent );
 369                 m_tcCornerSizeRatio->ChangeValue( value );
 370                 m_tcCornerSizeRatio1->ChangeValue( value );
 371             }
 372 
 373             if( ratioPercent > 50.0 )
 374             {
 375                 ratioPercent = 0.5;
 376                 value.Printf( "%.1f", ratioPercent*100.0 );
 377                 m_tcCornerSizeRatio->ChangeValue( value );
 378                 m_tcCornerSizeRatio1->ChangeValue( value );
 379             }
 380 
 381             if( ctrl == m_tcCornerSizeRatio )
 382                 m_tcCornerSizeRatio1->ChangeValue( value );
 383             else
 384                 m_tcCornerSizeRatio->ChangeValue( value );
 385 
 386             changed = true;
 387         }
 388     }
 389     else if( ctrl == m_tcChamferRatio || ctrl == m_tcChamferRatio1 )
 390     {
 391         double ratioPercent;
 392 
 393         if( value.ToDouble( &ratioPercent ) )
 394         {
 395             // Clamp ratioPercent to acceptable value (0.0 to 50.0)

 396             if( ratioPercent < 0.0 )
 397             {
 398                 ratioPercent = 0.0;
 399                 value.Printf( "%.1f", ratioPercent );
 400                 m_tcChamferRatio->ChangeValue( value );
 401                 m_tcChamferRatio1->ChangeValue( value );
 402             }
 403 
 404             if( ratioPercent > 50.0 )
 405             {
 406                 ratioPercent = 0.5;
 407                 value.Printf( "%.1f", ratioPercent*100.0 );
 408                 m_tcChamferRatio->ChangeValue( value );
 409                 m_tcChamferRatio1->ChangeValue( value );
 410             }
 411 
 412             if( ctrl == m_tcChamferRatio )
 413                 m_tcChamferRatio1->ChangeValue( value );
 414             else
 415                 m_tcChamferRatio->ChangeValue( value );
 416 
 417             changed = true;
 418         }
 419     }
 420 
 421     if( changed )
 422     {
 423         transferDataToPad( m_dummyPad );
 424         m_cornerRadius.ChangeValue( m_dummyPad->GetRoundRectCornerRadius() );
 425         redraw();
 426     }
 427 }

1036 // test if all values are acceptable for the pad

1037 bool DIALOG_PAD_PROPERTIES::padValuesOK()
1038 {
1039     bool error = transferDataToPad( m_dummyPad );
1040     bool skip_tstoffset = false;    // the offset prm is not always tested

1041 
1042     wxArrayString error_msgs;
1043     wxString msg;
1044 
1045     // Test for incorrect values

1046     if( (m_dummyPad->GetSize().x <= 0) ||
1047        ((m_dummyPad->GetSize().y <= 0) && (m_dummyPad->GetShape() != PAD_SHAPE_CIRCLE)) )
1048     {
1049         error_msgs.Add( _( "Pad size must be greater than zero" ) );
1050     }
1051 
1052     if( (m_dummyPad->GetSize().x < m_dummyPad->GetDrillSize().x) ||
1053         (m_dummyPad->GetSize().y < m_dummyPad->GetDrillSize().y) )
1054     {
1055         error_msgs.Add(  _( "Incorrect value for pad drill: pad drill bigger than pad size" ) );
1056         skip_tstoffset = true;  // offset prm will be not tested because if the drill value

1057                                 // is incorrect the offset prm is always seen as incorrect, even if it is 0

1058     }
1059 
1060     if( m_dummyPad->GetLocalClearance() < 0 )
1061     {
1062         error_msgs.Add( _( "Pad local clearance must be zero or greater than zero" ) );
1063     }
1064 
1065     // Some pads need a negative solder mask clearance (mainly for BGA with small pads)

1066     // However the negative solder mask clearance must not create negative mask size

1067     // Therefore test for minimal acceptable negative value

1068     // Hovewer, a negative value can give strange result with custom shapes, so it is not

1069     // allowed for custom pad shape

1070     if( m_dummyPad->GetLocalSolderMaskMargin() < 0 )
1071     {
1072         if( m_dummyPad->GetShape() == PAD_SHAPE_CUSTOM )
1073             error_msgs.Add( _( "Pad local solder mask clearance must be zero or greater than zero" ) );
1074         else
1075         {
1076             int min_smClearance = -std::min( m_dummyPad->GetSize().x, m_dummyPad->GetSize().y )/2;
1077 
1078             if( m_dummyPad->GetLocalSolderMaskMargin() <= min_smClearance )
1079             {
1080                 error_msgs.Add( wxString::Format(
1081                                 _( "Pad local solder mask clearance must be greater than %s" ),
1082                                 StringFromValue( GetUserUnits(), min_smClearance, true, true ) ) );
1083             }
1084         }
1085     }
1086 
1087     // Some pads need a positive solder paste clearance (mainly for BGA with small pads)

1088     // Hovewer, a positive value can create issues if the resulting shape is too big.

1089     // (like a solder paste creating a solder paste area on a neighbour pad or on the solder mask)

1090     // So we could ask for user to confirm the choice
