  28 DIALOG_FILTER_SELECTION::DIALOG_FILTER_SELECTION( PCB_BASE_FRAME* aParent, OPTIONS& aOptions ) :
  29     DIALOG_FILTER_SELECTION_BASE( aParent ),
  30     m_options( aOptions )
  31 {
  32     m_Include_Modules->SetValue( m_options.includeModules );
  33     m_IncludeLockedModules->SetValue( m_options.includeLockedModules );
  34 
  35     if( m_Include_Modules->GetValue() )
  36         m_IncludeLockedModules->Enable();
  37     else
  38         m_IncludeLockedModules->Disable();
  39 
  40     m_Include_Tracks->SetValue( m_options.includeTracks );
  41     m_Include_Vias->SetValue( m_options.includeVias );
  42     m_Include_Zones->SetValue( m_options.includeZones );
  43     m_Include_Draw_Items->SetValue( m_options.includeItemsOnTechLayers );
  44     m_Include_Edges_Items->SetValue( m_options.includeBoardOutlineLayer );
  45     m_Include_PcbTextes->SetValue( m_options.includePcbTexts );
  46 
  47     m_sdbSizer1OK->SetDefault();
  48     SetFocus();
  49     GetSizer()->SetSizeHints( this );
  50     Centre();
  51 }

  54 void DIALOG_FILTER_SELECTION::checkBoxClicked( wxCommandEvent& aEvent )
  55 {
  56     if( m_Include_Modules->GetValue() )
  57         m_IncludeLockedModules->Enable();
  58     else
  59         m_IncludeLockedModules->Disable();
  60 }

  63 void DIALOG_FILTER_SELECTION::ExecuteCommand( wxCommandEvent& event )
  64 {
  65     m_options.includeModules           = m_Include_Modules->GetValue();
  66     m_options.includeLockedModules     = m_IncludeLockedModules->GetValue();
  67     m_options.includeTracks            = m_Include_Tracks->GetValue();
  68     m_options.includeVias              = m_Include_Vias->GetValue();
  69     m_options.includeZones             = m_Include_Zones->GetValue();
  70     m_options.includeItemsOnTechLayers = m_Include_Draw_Items->GetValue();
  71     m_options.includeBoardOutlineLayer = m_Include_Edges_Items->GetValue();
  72     m_options.includePcbTexts          = m_Include_PcbTextes->GetValue();
  73 
  74     EndModal( wxID_OK );
