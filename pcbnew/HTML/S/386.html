 110 void DRC::ShowDRCDialog( wxWindow* aParent )
 111 {
 112     bool show_dlg_modal = true;
 113 
 114     // the dialog needs a parent frame. if it is not specified, this is

 115     // the PCB editor frame specified in DRC class.

 116     if( !aParent )
 117     {
 118         // if any parent is specified, the dialog is modal.

 119         // if this is the default PCB editor frame, it is not modal

 120         show_dlg_modal = false;
 121         aParent = m_editFrame;
 122     }
 123 
 124     Activate();
 125     m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
 126 
 127     if( !m_drcDialog )
 128     {
 129         m_drcDialog = new DIALOG_DRC( this, m_editFrame, aParent );
 130         updatePointers();
 131 
 132         if( show_dlg_modal )
 133             m_drcDialog->ShowModal();
 134         else
 135             m_drcDialog->Show( true );
 136     }
 137     else    // The dialog is just not visible (because the user has double clicked on an error item)

 138     {
 139         updatePointers();
 140         m_drcDialog->Show( true );
 141     }
 142 }

 183 bool DRC::LoadRules()
 184 {
 185     wxString   rulesFilepath = m_editFrame->Prj().AbsolutePath( "drc-rules" );
 186     wxFileName rulesFile( rulesFilepath );
 187 
 188     if( rulesFile.FileExists() )
 189     {
 190         m_rules.clear();
 191 
 192         FILE* fp = wxFopen( rulesFilepath, wxT( "rt" ) );
 193 
 194         if( fp )
 195         {
 196             try
 197             {
 198                 DRC_RULES_PARSER parser( m_pcb, fp, rulesFilepath );
 199                 parser.Parse( m_rules );
 200             }
 201             catch( PARSE_ERROR& pe )
 202             {
 203                 // Don't leave possibly malformed stuff around for us to trip over

 204                 m_rules.clear();
 205 
 206                 wxSafeYield( m_editFrame );
 207                 m_editFrame->ShowBoardSetupDialog( _( "Rules" ), pe.What(), ID_RULES_EDITOR,
 208                                                    pe.lineNumber, pe.byteIndex );
 209 
 210                 return false;
 211             }
 212         }
 213     }
 214 
 215     std::reverse( std::begin( m_rules ), std::end( m_rules ) );
 216 
 217     BOARD_DESIGN_SETTINGS& bds = m_pcb->GetDesignSettings();
 218     bds.m_DRCRules = m_rules;
 219 
 220     return true;
 221 }

 224 void DRC::RunTests( wxTextCtrl* aMessages )
 225 {
 226     // Make absolutely sure these are up-to-date

 227     if( !LoadRules() )
 228         return;
 229 
 230     wxASSERT( m_pcb == m_editFrame->GetBoard() );
 231 
 232     BOARD_COMMIT           commit( m_editFrame );
 233     BOARD_DESIGN_SETTINGS& bds = m_pcb->GetDesignSettings();
 234 
 235     m_largestClearance = bds.GetBiggestClearanceValue();
 236 
 237     if( !bds.Ignore( DRCE_INVALID_OUTLINE )
 238         || !bds.Ignore( DRCE_COPPER_EDGE_CLEARANCE ) )
 239     {
 240         if( aMessages )
 241         {
 242             aMessages->AppendText( _( "Board Outline...\n" ) );
 243             wxSafeYield();
 244         }
 245 
 246         testOutline( commit );
 247     }
 248 
 249     // test pad to pad clearances, nothing to do with tracks, vias or zones.

 250     if( !bds.Ignore( DRCE_COPPER_EDGE_CLEARANCE ) )
 251     {
 252         if( aMessages )
 253         {
 254             aMessages->AppendText( _( "Pad clearances...\n" ) );
 255             wxSafeYield();
 256         }
 257 
 258         testPadClearances( commit );
 259     }
 260 
 261     // test drilled holes

 262     if( !bds.Ignore( DRCE_DRILLED_HOLES_TOO_CLOSE )
 263         || !bds.Ignore( DRCE_TOO_SMALL_DRILL )
 264         || !bds.Ignore( DRCE_TOO_SMALL_MICROVIA_DRILL ) )
 265     {
 266         if( aMessages )
 267         {
 268             aMessages->AppendText( _( "Drill sizes and clearances...\n" ) );
 269             wxSafeYield();
 270         }
 271 
 272         DRC_DRILLED_HOLE_TESTER tester( [&]( MARKER_PCB* aMarker )
 273                                         {
 274                                             addMarkerToPcb( commit, aMarker );
 275                                         } );
 276 
 277         tester.RunDRC( userUnits(), *m_pcb );
 278     }
 279 
 280     // caller (a wxTopLevelFrame) is the wxDialog or the Pcb Editor frame that call DRC:

 281     wxWindow* caller = aMessages ? aMessages->GetParent() : m_editFrame;
 282 
 283     if( m_refillZones )
 284     {
 285         if( aMessages )
 286             aMessages->AppendText( _( "Refilling all zones...\n" ) );
 287 
 288         m_toolMgr->GetTool<ZONE_FILLER_TOOL>()->FillAllZones( caller );
 289     }
 290     else
 291     {
 292         if( aMessages )
 293             aMessages->AppendText( _( "Checking zone fills...\n" ) );
 294 
 295         m_toolMgr->GetTool<ZONE_FILLER_TOOL>()->CheckAllZones( caller );
 296     }
 297 
 298     // test track and via clearances to other tracks, pads, and vias

 299     if( aMessages )
 300     {
 301         aMessages->AppendText( _( "Track clearances...\n" ) );
 302         wxSafeYield();
 303     }
 304 
 305     testTracks( commit, aMessages ? aMessages->GetParent() : m_editFrame, true );
 306 
 307     // test zone clearances to other zones

 308     if( aMessages )
 309     {
 310         aMessages->AppendText( _( "Zone to zone clearances...\n" ) );
 311         wxSafeYield();
 312     }
 313 
 314     testZones( commit );
 315 
 316     // find and gather unconnected pads.

 317     if( m_doUnconnectedTest
 318         && !bds.Ignore( DRCE_UNCONNECTED_ITEMS ) )
 319     {
 320         if( aMessages )
 321         {
 322             aMessages->AppendText( _( "Unconnected pads...\n" ) );
 323             aMessages->Refresh();
 324         }
 325 
 326         testUnconnected();
 327     }
 328 
 329     // find and gather vias, tracks, pads inside keepout areas.

 330     if( m_doKeepoutTest )
 331     {
 332         if( aMessages )
 333         {
 334             aMessages->AppendText( _( "Keepout areas ...\n" ) );
 335             aMessages->Refresh();
 336         }
 337 
 338         DRC_KEEPOUT_TESTER tester( [&]( MARKER_PCB* aMarker )
 339                                    {
 340                                        addMarkerToPcb( commit, aMarker );
 341                                    } );
 342 
 343         tester.RunDRC( userUnits(), *m_pcb );
 344     }
 345 
 346     // find and gather vias, tracks, pads inside text boxes.

 347     if( !bds.Ignore( DRCE_CLEARANCE ) )
 348     {
 349         if( aMessages )
 350         {
 351             aMessages->AppendText( _( "Text and graphic clearances...\n" ) );
 352             wxSafeYield();
 353         }
 354 
 355         testCopperTextAndGraphics( commit );
 356     }
 357 
 358     // test courtyards

 359     if( !bds.Ignore( DRCE_OVERLAPPING_FOOTPRINTS )
 360         || !bds.Ignore( DRCE_MISSING_COURTYARD )
 361         || !bds.Ignore( DRCE_MALFORMED_COURTYARD )
 362         || !bds.Ignore( DRCE_PTH_IN_COURTYARD )
 363         || !bds.Ignore( DRCE_NPTH_IN_COURTYARD ) )
 364     {
 365         if( aMessages )
 366         {
 367             aMessages->AppendText( _( "Courtyard areas...\n" ) );
 368             aMessages->Refresh();
 369         }
 370 
 371         DRC_COURTYARD_TESTER tester( [&]( MARKER_PCB* aMarker )
 372                                      {
 373                                          addMarkerToPcb( commit, aMarker );
 374                                      } );
 375 
 376         tester.RunDRC( userUnits(), *m_pcb );
 377     }
 378 
 379     for( DRC_ITEM* footprintItem : m_footprints )
 380         delete footprintItem;
 381 
 382     m_footprints.clear();
 383     m_footprintsTested = false;
 384 
 385     if( m_testFootprints && !Kiface().IsSingle() )
 386     {
 387         if( aMessages )
 388         {
 389             aMessages->AppendText( _( "Checking footprints against schematic...\n" ) );
 390             aMessages->Refresh();
 391         }
 392 
 393         NETLIST netlist;
 394         m_editFrame->FetchNetlistFromSchematic( netlist, PCB_EDIT_FRAME::ANNOTATION_DIALOG );
 395 
 396         if( m_drcDialog )
 397             m_drcDialog->Raise();
 398 
 399         TestFootprints( netlist, m_pcb, m_footprints );
 400         m_footprintsTested = true;
 401     }
 402 
 403     // Check if there are items on disabled layers

 404     if( !bds.Ignore( DRCE_DISABLED_LAYER_ITEM ) )
 405     {
 406         if( aMessages )
 407         {
 408             aMessages->AppendText( _( "Items on disabled layers...\n" ) );
 409             aMessages->Refresh();
 410         }
 411 
 412         testDisabledLayers( commit );
 413     }
 414 
 415     if( !bds.Ignore( DRCE_UNRESOLVED_VARIABLE ) )
 416     {
 417         if( aMessages )
 418         {
 419             aMessages->AppendText( _( "Unresolved text variables...\n" ) );
 420             aMessages->Refresh();
 421         }
 422 
 423         DRC_TEXTVAR_TESTER tester( [&]( MARKER_PCB* aMarker )
 424                                    {
 425                                        addMarkerToPcb( commit, aMarker );
 426                                    },
 427                                    m_editFrame->GetCanvas()->GetWorksheet() );
 428 
 429         tester.RunDRC( userUnits(), *m_pcb );
 430     }
 431 
 432     commit.Push( wxEmptyString, false, false );
 433     m_drcRun = true;
 434 
 435     // update the m_drcDialog listboxes

 436     updatePointers();
 437 
 438     if( aMessages )
 439     {
 440         // no newline on this one because it is last, don't want the window

 441         // to unnecessarily scroll.

 442         aMessages->AppendText( _( "Finished" ) );
 443     }
 444 }

 465 void DRC::testPadClearances( BOARD_COMMIT& aCommit )
 466 {
 467     BOARD_DESIGN_SETTINGS& bds = m_pcb->GetDesignSettings();
 468     std::vector<D_PAD*>    sortedPads;
 469 
 470     m_pcb->GetSortedPadListByXthenYCoord( sortedPads );
 471 
 472     if( sortedPads.empty() )
 473         return;
 474 
 475     // find the max size of the pads (used to stop the pad-to-pad tests)

 476     int max_size = 0;
 477 
 478     for( D_PAD* pad : sortedPads )
 479     {
 480         // GetBoundingRadius() is the radius of the minimum sized circle fully containing the pad

 481         int radius = pad->GetBoundingRadius();
 482 
 483         if( radius > max_size )
 484             max_size = radius;
 485     }
 486 
 487     // Better to be fast than accurate; this keeps us from having to look up / calculate the

 488     // actual clearances

 489     max_size += m_largestClearance;
 490 
 491     // Upper limit of pad list (limit not included)

 492     D_PAD** listEnd = &sortedPads[0] + sortedPads.size();
 493 
 494     // Test the pads

 495     for( auto& pad : sortedPads )
 496     {
 497         if( !bds.Ignore( DRCE_COPPER_EDGE_CLEARANCE ) && m_board_outline_valid )
 498         {
 499             int minClearance = bds.m_CopperEdgeClearance;
 500             m_clearanceSource = _( "board edge" );
 501 
 502             static DRAWSEGMENT dummyEdge;
 503             dummyEdge.SetLayer( Edge_Cuts );
 504 
 505             if( pad->GetRuleClearance( &dummyEdge, &minClearance, &m_clearanceSource ) )
 506             {
 507                 /* minClearance and m_clearanceSource set in GetRuleClearance() */;
 508             }
 509 
 510             for( auto it = m_board_outlines.IterateSegmentsWithHoles(); it; it++ )
 511             {
 512                 SHAPE_SEGMENT edge( *it );
 513                 int actual;
 514 
 515                 if( pad->Collide( &edge, minClearance, &actual ) )
 516                 {
 517                     DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_COPPER_EDGE_CLEARANCE );
 518 
 519                     m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
 520                                   m_clearanceSource,
 521                                   MessageTextFromValue( userUnits(), minClearance, true ),
 522                                   MessageTextFromValue( userUnits(), actual, true ) );
 523 
 524                     drcItem->SetErrorMessage( m_msg );
 525                     drcItem->SetItems( pad );
 526 
 527                     MARKER_PCB* marker = new MARKER_PCB( drcItem, pad->GetPosition() );
 528                     addMarkerToPcb( aCommit, marker );
 529 
 530                     break;
 531                 }
 532             }
 533         }
 534 
 535         if( !bds.Ignore( DRCE_CLEARANCE ) )
 536         {
 537             int x_limit = pad->GetPosition().x + pad->GetBoundingRadius() + max_size;
 538 
 539             doPadToPadsDrc( aCommit, pad, &pad, listEnd, x_limit );
 540         }
 541     }
 542 }

 545 void DRC::testTracks( BOARD_COMMIT& aCommit, wxWindow *aActiveWindow, bool aShowProgressBar )
 546 {
 547     wxProgressDialog* progressDialog = NULL;
 548     const int         delta = 500;  // This is the number of tests between 2 calls to the

 549                                     // progress bar

 550     int               count = m_pcb->Tracks().size();
 551     int               deltamax = count/delta;
 552 
 553     if( aShowProgressBar && deltamax > 3 )
 554     {
 555         // Do not use wxPD_APP_MODAL style here: it is not necessary and create issues

 556         // on OSX

 557         progressDialog = new wxProgressDialog( _( "Track clearances" ), wxEmptyString,
 558                                                deltamax, aActiveWindow,
 559                                                wxPD_AUTO_HIDE | wxPD_CAN_ABORT | wxPD_ELAPSED_TIME );
 560         progressDialog->Update( 0, wxEmptyString );
 561     }
 562 
 563     std::shared_ptr<CONNECTIVITY_DATA> connectivity = m_pcb->GetConnectivity();
 564     BOARD_DESIGN_SETTINGS&             settings = m_pcb->GetDesignSettings();
 565 
 566 
 567     if( !m_pcb->GetDesignSettings().Ignore( DRCE_DANGLING_TRACK )
 568             || !m_pcb->GetDesignSettings().Ignore( DRCE_DANGLING_VIA ) )
 569     {
 570         connectivity->Clear();
 571         connectivity->Build( m_pcb ); // just in case. This really needs to be reliable.

 572     }
 573 
 574     int ii = 0;
 575     count = 0;
 576 
 577     for( auto seg_it = m_pcb->Tracks().begin(); seg_it != m_pcb->Tracks().end(); seg_it++ )
 578     {
 579         if( ii++ > delta )
 580         {
 581             ii = 0;
 582             count++;
 583 
 584             if( progressDialog )
 585             {
 586                 if( !progressDialog->Update( count, wxEmptyString ) )
 587                     break;  // Aborted by user

 588 #ifdef __WXMAC__
 589                 // Work around a dialog z-order issue on OS X

 590                 if( count == deltamax )
 591                     aActiveWindow->Raise();
 592 #endif
 593             }
 594         }
 595 
 596         // Test new segment against tracks and pads, optionally against copper zones

 597         doTrackDrc( aCommit, *seg_it, seg_it + 1, m_pcb->Tracks().end(), m_testTracksAgainstZones );
 598 
 599         // Test for dangling items

 600         int code = (*seg_it)->Type() == PCB_VIA_T ? DRCE_DANGLING_VIA : DRCE_DANGLING_TRACK;
 601         wxPoint pos;
 602 
 603         if( !settings.Ignore( code ) && connectivity->TestTrackEndpointDangling( *seg_it, &pos ) )
 604         {
 605             DRC_ITEM* drcItem = DRC_ITEM::Create( code );
 606             drcItem->SetItems( *seg_it );
 607 
 608             MARKER_PCB* marker = new MARKER_PCB( drcItem, pos );
 609             addMarkerToPcb( aCommit, marker );
 610         }
 611     }
 612 
 613     if( progressDialog )
 614         progressDialog->Destroy();
 615 }

 618 void DRC::testUnconnected()
 619 {
 620     for( DRC_ITEM* unconnectedItem : m_unconnected )
 621         delete unconnectedItem;
 622 
 623     m_unconnected.clear();
 624 
 625     auto connectivity = m_pcb->GetConnectivity();
 626 
 627     connectivity->Clear();
 628     connectivity->Build( m_pcb ); // just in case. This really needs to be reliable.

 629     connectivity->RecalculateRatsnest();
 630 
 631     std::vector<CN_EDGE> edges;
 632     connectivity->GetUnconnectedEdges( edges );
 633 
 634     for( const CN_EDGE& edge : edges )
 635     {
 636         DRC_ITEM* item = DRC_ITEM::Create( DRCE_UNCONNECTED_ITEMS );
 637         item->SetItems( edge.GetSourceNode()->Parent(), edge.GetTargetNode()->Parent() );
 638         m_unconnected.push_back( item );
 639     }
 640 }

 643 void DRC::testZones( BOARD_COMMIT& aCommit )
 644 {
 645     BOARD_DESIGN_SETTINGS& bds = m_pcb->GetDesignSettings();
 646 
 647     // Test copper areas for valid netcodes

 648     // if a netcode is < 0 the netname was not found when reading a netlist

 649     // if a netcode is == 0 the netname is void, and the zone is not connected.

 650     // This is allowed, but i am not sure this is a good idea

 651     //

 652     // In recent Pcbnew versions, the netcode is always >= 0, but an internal net name

 653     // is stored, and initialized from the file or the zone properties editor.

 654     // if it differs from the net name from net code, there is a DRC issue

 655 
 656     std::vector<SHAPE_POLY_SET> smoothed_polys;
 657     smoothed_polys.resize( m_pcb->GetAreaCount() );
 658 
 659     for( int ii = 0; ii < m_pcb->GetAreaCount(); ii++ )
 660     {
 661         ZONE_CONTAINER* zone = m_pcb->GetArea( ii );
 662 
 663         if( !bds.Ignore( DRCE_ZONE_HAS_EMPTY_NET ) && zone->IsOnCopperLayer() )
 664         {
 665             int netcode = zone->GetNetCode();
 666             // a netcode < 0 or > 0 and no pad in net is a error or strange

 667             // perhaps a "dead" net, which happens when all pads in this net were removed

 668             // Remark: a netcode < 0 should not happen (this is more a bug somewhere)

 669             int pads_in_net = ( netcode > 0 ) ? m_pcb->GetConnectivity()->GetPadCount( netcode ) : 1;
 670 
 671             if( ( netcode < 0 ) || pads_in_net == 0 )
 672             {
 673                 DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_ZONE_HAS_EMPTY_NET );
 674                 drcItem->SetItems( zone );
 675 
 676                 MARKER_PCB* marker = new MARKER_PCB( drcItem, zone->GetPosition() );
 677                 addMarkerToPcb( aCommit, marker );
 678             }
 679         }
 680 
 681         ZONE_CONTAINER*    zoneRef = m_pcb->GetArea( ii );
 682         std::set<VECTOR2I> colinearCorners;
 683         zoneRef->GetColinearCorners( m_pcb, colinearCorners );
 684 
 685         zoneRef->BuildSmoothedPoly( smoothed_polys[ii], &colinearCorners );
 686     }
 687 
 688     // iterate through all areas

 689     for( int ia = 0; ia < m_pcb->GetAreaCount(); ia++ )
 690     {
 691         ZONE_CONTAINER* zoneRef = m_pcb->GetArea( ia );
 692 
 693         if( !zoneRef->IsOnCopperLayer() )
 694             continue;
 695 
 696         // If we are testing a single zone, then iterate through all other zones

 697         // Otherwise, we have already tested the zone combination

 698         for( int ia2 = ia + 1; ia2 < m_pcb->GetAreaCount(); ia2++ )
 699         {
 700             ZONE_CONTAINER* zoneToTest = m_pcb->GetArea( ia2 );
 701 
 702             if( zoneRef == zoneToTest )
 703                 continue;
 704 
 705             // test for same layer

 706             if( zoneRef->GetLayer() != zoneToTest->GetLayer() )
 707                 continue;
 708 
 709             // Test for same net

 710             if( zoneRef->GetNetCode() == zoneToTest->GetNetCode() && zoneRef->GetNetCode() >= 0 )
 711                 continue;
 712 
 713             // test for different priorities

 714             if( zoneRef->GetPriority() != zoneToTest->GetPriority() )
 715                 continue;
 716 
 717             // test for different types

 718             if( zoneRef->GetIsKeepout() != zoneToTest->GetIsKeepout() )
 719                 continue;
 720 
 721             // Examine a candidate zone: compare zoneToTest to zoneRef

 722 
 723             // Get clearance used in zone to zone test.  The policy used to

 724             // obtain that value is now part of the zone object itself by way of

 725             // ZONE_CONTAINER::GetClearance().

 726             int zone2zoneClearance = zoneRef->GetClearance( zoneToTest, &m_clearanceSource );
 727 
 728             // Keepout areas have no clearance, so set zone2zoneClearance to 1

 729             // ( zone2zoneClearance = 0  can create problems in test functions)

 730             if( zoneRef->GetIsKeepout() )
 731                 zone2zoneClearance = 1;
 732 
 733             // test for some corners of zoneRef inside zoneToTest

 734             for( auto iterator = smoothed_polys[ia].IterateWithHoles(); iterator; iterator++ )
 735             {
 736                 VECTOR2I currentVertex = *iterator;
 737                 wxPoint pt( currentVertex.x, currentVertex.y );
 738 
 739                 if( smoothed_polys[ia2].Contains( currentVertex ) )
 740                 {
 741                     DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_ZONES_INTERSECT );
 742                     drcItem->SetItems( zoneRef, zoneToTest );
 743 
 744                     MARKER_PCB* marker = new MARKER_PCB( drcItem, pt );
 745                     addMarkerToPcb( aCommit, marker );
 746                 }
 747             }
 748 
 749             // test for some corners of zoneToTest inside zoneRef

 750             for( auto iterator = smoothed_polys[ia2].IterateWithHoles(); iterator; iterator++ )
 751             {
 752                 VECTOR2I currentVertex = *iterator;
 753                 wxPoint pt( currentVertex.x, currentVertex.y );
 754 
 755                 if( smoothed_polys[ia].Contains( currentVertex ) )
 756                 {
 757                     DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_ZONES_INTERSECT );
 758                     drcItem->SetItems( zoneToTest, zoneRef );
 759 
 760                     MARKER_PCB* marker = new MARKER_PCB( drcItem, pt );
 761                     addMarkerToPcb( aCommit, marker );
 762                 }
 763             }
 764 
 765             // Iterate through all the segments of refSmoothedPoly

 766             std::map<wxPoint, int> conflictPoints;
 767 
 768             for( auto refIt = smoothed_polys[ia].IterateSegmentsWithHoles(); refIt; refIt++ )
 769             {
 770                 // Build ref segment

 771                 SEG refSegment = *refIt;
 772 
 773                 // Iterate through all the segments in smoothed_polys[ia2]

 774                 for( auto testIt = smoothed_polys[ia2].IterateSegmentsWithHoles(); testIt; testIt++ )
 775                 {
 776                     // Build test segment

 777                     SEG testSegment = *testIt;
 778                     wxPoint pt;
 779 
 780                     int ax1, ay1, ax2, ay2;
 781                     ax1 = refSegment.A.x;
 782                     ay1 = refSegment.A.y;
 783                     ax2 = refSegment.B.x;
 784                     ay2 = refSegment.B.y;
 785 
 786                     int bx1, by1, bx2, by2;
 787                     bx1 = testSegment.A.x;
 788                     by1 = testSegment.A.y;
 789                     bx2 = testSegment.B.x;
 790                     by2 = testSegment.B.y;
 791 
 792                     int d = GetClearanceBetweenSegments( bx1, by1, bx2, by2,
 793                                                          0,
 794                                                          ax1, ay1, ax2, ay2,
 795                                                          0,
 796                                                          zone2zoneClearance,
 797                                                          &pt.x, &pt.y );
 798 
 799                     if( d < zone2zoneClearance )
 800                     {
 801                         if( conflictPoints.count( pt ) )
 802                             conflictPoints[ pt ] = std::min( conflictPoints[ pt ], d );
 803                         else
 804                             conflictPoints[ pt ] = d;
 805                     }
 806                 }
 807             }
 808 
 809             for( const std::pair<const wxPoint, int>& conflict : conflictPoints )
 810             {
 811                 int       actual = conflict.second;
 812                 DRC_ITEM* drcItem;
 813 
 814                 if( actual <= 0 )
 815                 {
 816                     drcItem = DRC_ITEM::Create( DRCE_ZONES_INTERSECT );
 817                 }
 818                 else
 819                 {
 820                     drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
 821 
 822                     m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
 823                                   m_clearanceSource,
 824                                   MessageTextFromValue( userUnits(), zone2zoneClearance, true ),
 825                                   MessageTextFromValue( userUnits(), conflict.second, true ) );
 826 
 827                     drcItem->SetErrorMessage( m_msg );
 828                 }
 829 
 830                 drcItem->SetItems( zoneRef, zoneToTest );
 831 
 832                 MARKER_PCB* marker = new MARKER_PCB( drcItem, conflict.first );
 833                 addMarkerToPcb( aCommit, marker );
 834             }
 835         }
 836     }
 837 }

 878 void DRC::testCopperDrawItem( BOARD_COMMIT& aCommit, BOARD_ITEM* aItem )
 879 {
 880     EDA_RECT            bbox;
 881     std::vector<SHAPE*> itemShapes;
 882     DRAWSEGMENT*        drawItem = dynamic_cast<DRAWSEGMENT*>( aItem );
 883     EDA_TEXT*           textItem = dynamic_cast<EDA_TEXT*>( aItem );
 884 
 885     if( drawItem )
 886     {
 887         bbox = drawItem->GetBoundingBox();
 888         itemShapes = drawItem->MakeEffectiveShapes();
 889     }
 890     else if( textItem )
 891     {
 892         bbox = textItem->GetTextBox();
 893 
 894         int penWidth = textItem->GetEffectiveTextPenWidth();
 895         std::vector<wxPoint> pts;
 896         textItem->TransformTextShapeToSegmentList( pts );
 897 
 898         for( unsigned jj = 0; jj < pts.size(); jj += 2 )
 899             itemShapes.push_back( new SHAPE_SEGMENT( pts[jj], pts[jj+1], penWidth ) );
 900     }
 901     else
 902     {
 903         wxFAIL_MSG( "unknown item type in testCopperDrawItem()" );
 904         return;
 905     }
 906 
 907     SHAPE_RECT bboxShape( bbox.GetX(), bbox.GetY(), bbox.GetWidth(), bbox.GetHeight() );
 908 
 909     // Test tracks and vias

 910     for( auto track : m_pcb->Tracks() )
 911     {
 912         if( !track->IsOnLayer( aItem->GetLayer() ) )
 913             continue;
 914 
 915         int     minClearance = track->GetClearance( aItem, &m_clearanceSource );
 916         int     actual = INT_MAX;
 917         wxPoint pos;
 918 
 919         SHAPE_SEGMENT trackSeg( track->GetStart(), track->GetEnd(), track->GetWidth() );
 920 
 921         // Fast test to detect a track segment candidate inside the text bounding box

 922         if( !bboxShape.Collide( &trackSeg, 0 ) )
 923             continue;
 924 
 925         for( const SHAPE* shape : itemShapes )
 926         {
 927             int this_dist;
 928 
 929             if( shape->Collide( &trackSeg, minClearance, &this_dist ) )
 930             {
 931                 if( this_dist < actual )
 932                 {
 933                     actual = this_dist;
 934                     pos = (wxPoint) shape->Centre();
 935                 }
 936             }
 937         }
 938 
 939         if( actual < INT_MAX )
 940         {
 941             DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
 942 
 943             m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
 944                           m_clearanceSource,
 945                           MessageTextFromValue( userUnits(), minClearance, true ),
 946                           MessageTextFromValue( userUnits(), std::max( 0, actual ), true ) );
 947 
 948             drcItem->SetErrorMessage( m_msg );
 949             drcItem->SetItems( track, aItem );
 950 
 951             MARKER_PCB* marker = new MARKER_PCB( drcItem, pos );
 952             addMarkerToPcb( aCommit, marker );
 953         }
 954     }
 955 
 956     // Test pads

 957     for( auto pad : m_pcb->GetPads() )
 958     {
 959         if( !pad->IsOnLayer( aItem->GetLayer() ) )
 960             continue;
 961 
 962         // Graphic items are allowed to act as net-ties within their own footprint

 963         if( drawItem && pad->GetParent() == drawItem->GetParent() )
 964             continue;
 965 
 966         int minClearance = pad->GetClearance( aItem, &m_clearanceSource );
 967         int actual = INT_MAX;
 968 
 969         // Fast test to detect a pad candidate inside the text bounding box

 970         // Finer test (time consumming) is made only for pads near the text.

 971         int bb_radius = pad->GetBoundingRadius() + minClearance;
 972 
 973         if( !bboxShape.Collide( SEG( pad->GetPosition(), pad->GetPosition() ), bb_radius ) )
 974             continue;
 975 
 976         for( const std::shared_ptr<SHAPE>& aShape : pad->GetEffectiveShapes() )
 977         {
 978             for( const SHAPE* bShape : itemShapes )
 979             {
 980                 int this_dist;
 981 
 982                 if( aShape->Collide( bShape, minClearance, &this_dist ) )
 983                     actual = std::min( actual, this_dist );
 984             }
 985         }
 986 
 987         if( actual < INT_MAX )
 988         {
 989             DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
 990 
 991             m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
 992                           m_clearanceSource,
 993                           MessageTextFromValue( userUnits(), minClearance, true ),
 994                           MessageTextFromValue( userUnits(), std::max( 0, actual ), true ) );
 995 
 996             drcItem->SetErrorMessage( m_msg );
 997             drcItem->SetItems( pad, aItem );
 998 
 999             MARKER_PCB* marker = new MARKER_PCB( drcItem, pad->GetPosition() );
1000             addMarkerToPcb( aCommit, marker );
1001         }
1002     }
1003 
1004     for( SHAPE* shape : itemShapes )
1005         delete shape;
1006 }

1099 bool DRC::doPadToPadsDrc( BOARD_COMMIT& aCommit, D_PAD* aRefPad, D_PAD** aStart, D_PAD** aEnd,
1100                           int x_limit )
1101 {
1102     const static LSET all_cu = LSET::AllCuMask();
1103 
1104     LSET layerMask = aRefPad->GetLayerSet() & all_cu;
1105 
1106     for( D_PAD** pad_list = aStart;  pad_list<aEnd;  ++pad_list )
1107     {
1108         D_PAD* pad = *pad_list;
1109 
1110         if( pad == aRefPad )
1111             continue;
1112 
1113         // We can stop the test when pad->GetPosition().x > x_limit

1114         // because the list is sorted by X values

1115         if( pad->GetPosition().x > x_limit )
1116             break;
1117 
1118         // No problem if pads which are on copper layers are on different copper layers,

1119         // (pads can be only on a technical layer, to build complex pads)

1120         // but their hole (if any ) can create DRC error because they are on all

1121         // copper layers, so we test them

1122         if( ( pad->GetLayerSet() & layerMask ) == 0 &&
1123             ( pad->GetLayerSet() & all_cu ) != 0 &&
1124             ( aRefPad->GetLayerSet() & all_cu ) != 0 )
1125         {
1126             // if holes are in the same location and have the same size and shape,

1127             // this can be accepted

1128             if( pad->GetPosition() == aRefPad->GetPosition()
1129                 && pad->GetDrillSize() == aRefPad->GetDrillSize()
1130                 && pad->GetDrillShape() == aRefPad->GetDrillShape() )
1131             {
1132                 if( aRefPad->GetDrillShape() == PAD_DRILL_SHAPE_CIRCLE )
1133                     continue;
1134 
1135                 // for oval holes: must also have the same orientation

1136                 if( pad->GetOrientation() == aRefPad->GetOrientation() )
1137                     continue;
1138             }
1139 
1140             if( pad->GetDrillSize().x )
1141             {
1142                 int minClearance = aRefPad->GetClearance( nullptr, &m_clearanceSource );
1143                 int actual;
1144 
1145                 if( aRefPad->Collide( pad->GetEffectiveHoleShape(), minClearance, &actual ) )
1146                 {
1147                     DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
1148 
1149                     m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
1150                                   m_clearanceSource,
1151                                   MessageTextFromValue( userUnits(), minClearance, true ),
1152                                   MessageTextFromValue( userUnits(), actual, true ) );
1153 
1154                     drcItem->SetErrorMessage( m_msg );
1155                     drcItem->SetItems( pad, aRefPad );
1156 
1157                     MARKER_PCB* marker = new MARKER_PCB( drcItem, pad->GetPosition() );
1158                     addMarkerToPcb( aCommit, marker );
1159                     return false;
1160                 }
1161             }
1162 
1163             if( aRefPad->GetDrillSize().x )
1164             {
1165                 int minClearance = pad->GetClearance( nullptr, &m_clearanceSource );
1166                 int actual;
1167 
1168                 if( pad->Collide( aRefPad->GetEffectiveHoleShape(), minClearance, &actual ) )
1169                 {
1170                     DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
1171 
1172                     m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
1173                                   m_clearanceSource,
1174                                   MessageTextFromValue( userUnits(), minClearance, true ),
1175                                   MessageTextFromValue( userUnits(), actual, true ) );
1176 
1177                     drcItem->SetErrorMessage( m_msg );
1178                     drcItem->SetItems( aRefPad, pad );
1179 
1180                     MARKER_PCB* marker = new MARKER_PCB( drcItem, aRefPad->GetPosition() );
1181                     addMarkerToPcb( aCommit, marker );
1182                     return false;
1183                 }
1184             }
1185 
1186             continue;
1187         }
1188 
1189         // The pad must be in a net (i.e pt_pad->GetNet() != 0 ),

1190         // But no problem if pads have the same netcode (same net)

1191         if( pad->GetNetCode() && ( aRefPad->GetNetCode() == pad->GetNetCode() ) )
1192             continue;
1193 
1194         // if pads are from the same footprint

1195         if( pad->GetParent() == aRefPad->GetParent() )
1196         {
1197             // and have the same pad number ( equivalent pads  )

1198 
1199             // one can argue that this 2nd test is not necessary, that any

1200             // two pads from a single module are acceptable.  This 2nd test

1201             // should eventually be a configuration option.

1202             if( pad->PadNameEqual( aRefPad ) )
1203                 continue;
1204         }
1205 
1206         // if either pad has no drill and is only on technical layers, not a clearance violation

1207         if( ( ( pad->GetLayerSet() & layerMask ) == 0 && !pad->GetDrillSize().x ) ||
1208             ( ( aRefPad->GetLayerSet() & layerMask ) == 0 && !aRefPad->GetDrillSize().x ) )
1209         {
1210             continue;
1211         }
1212 
1213         int  minClearance = aRefPad->GetClearance( pad, &m_clearanceSource );
1214         int  clearanceAllowed = minClearance - m_pcb->GetDesignSettings().GetDRCEpsilon();
1215         int  actual;
1216 
1217         if( aRefPad->Collide( pad, clearanceAllowed, &actual ) )
1218         {
1219             DRC_ITEM* drcItem = DRC_ITEM::Create( DRCE_CLEARANCE );
1220 
1221             m_msg.Printf( drcItem->GetErrorText() + _( " (%s clearance %s; actual %s)" ),
1222                           m_clearanceSource,
1223                           MessageTextFromValue( userUnits(), minClearance, true ),
1224                           MessageTextFromValue( userUnits(), actual, true ) );
1225 
1226             drcItem->SetErrorMessage( m_msg );
1227             drcItem->SetItems( aRefPad, pad );
1228 
1229             MARKER_PCB* marker = new MARKER_PCB( drcItem, aRefPad->GetPosition() );
1230             addMarkerToPcb( aCommit, marker );
1231             return false;
1232         }
1233     }
1234 
1235     return true;
1236 }

1245 const int UI_EPSILON = Mils2iu( 5 );
1246 
1247 
1248 wxPoint DRC::GetLocation( TRACK* aTrack, ZONE_CONTAINER* aConflictZone )
1249 {
1250     SHAPE_POLY_SET* conflictOutline;
1251 
1252     PCB_LAYER_ID l = aTrack->GetLayer();
1253 
1254     if( aConflictZone->IsFilled() && aConflictZone->HasFilledPolysForLayer( l ) )
1255         conflictOutline = const_cast<SHAPE_POLY_SET*>( &aConflictZone->GetFilledPolysList( l ) );
1256     else
1257         conflictOutline = aConflictZone->Outline();
1258 
1259     wxPoint pt1 = aTrack->GetPosition();
1260     wxPoint pt2 = aTrack->GetEnd();
1261 
1262     // If the mid-point is in the zone, then that's a fine place for the marker

1263     if( conflictOutline->SquaredDistance( ( pt1 + pt2 ) / 2 ) == 0 )
1264         return ( pt1 + pt2 ) / 2;
1265 
1266     // Otherwise do a binary search for a "good enough" marker location

1267     else
1268     {
1269         while( GetLineLength( pt1, pt2 ) > UI_EPSILON )
1270         {
1271             if( conflictOutline->SquaredDistance( pt1 ) < conflictOutline->SquaredDistance( pt2 ) )
1272                 pt2 = ( pt1 + pt2 ) / 2;
