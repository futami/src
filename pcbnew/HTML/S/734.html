  62 PyObject* PYTHON_ACTION_PLUGIN::CallMethod( const char* aMethod, PyObject* aArglist )
  63 {
  64     PyLOCK lock;
  65 
  66     PyErr_Clear();
  67     // pFunc is a new reference to the desired method

  68     PyObject* pFunc = PyObject_GetAttrString( this->m_PyAction, aMethod );
  69 
  70     if( pFunc && PyCallable_Check( pFunc ) )
  71     {
  72         PyObject* result = PyObject_CallObject( pFunc, aArglist );
  73 
  74         if( PyErr_Occurred() )
  75         {
  76             wxMessageBox( PyErrStringWithTraceback(),
  77                     _( "Exception on python action plugin code" ),
  78                     wxICON_ERROR | wxOK );
  79         }
  80 
  81         if( result )
  82         {
  83             Py_XDECREF( pFunc );
  84             return result;
  85         }
  86     }
  87     else
  88     {
  89         wxString msg = wxString::Format( _( "Method \"%s\" not found, or not callable" ), aMethod );
  90         wxMessageBox( msg, _( "Unknown Method" ), wxICON_ERROR | wxOK );
  91     }
  92 
  93     if( pFunc )
  94     {
  95         Py_XDECREF( pFunc );
  96     }
  97 
  98     return NULL;
  99 }

 213 void PCB_EDIT_FRAME::RunActionPlugin( ACTION_PLUGIN* aActionPlugin )
 214 {
 215 
 216     PICKED_ITEMS_LIST itemsList;
 217     BOARD*  currentPcb  = GetBoard();
 218     bool    fromEmpty   = false;
 219 
 220     itemsList.m_Status = UR_CHANGED;
 221 
 222     // Append tracks:

 223     for( auto item : currentPcb->Tracks() )
 224     {
 225         ITEM_PICKER picker( nullptr, item, UR_CHANGED );
 226         itemsList.PushItem( picker );
 227     }
 228 
 229     // Append modules:

 230     for( auto item : currentPcb->Modules() )
 231     {
 232         ITEM_PICKER picker( nullptr, item, UR_CHANGED );
 233         itemsList.PushItem( picker );
 234     }
 235 
 236     // Append drawings

 237     for( auto item : currentPcb->Drawings() )
 238     {
 239         ITEM_PICKER picker( nullptr, item, UR_CHANGED );
 240         itemsList.PushItem( picker );
 241     }
 242 
 243     // Append zones outlines

 244     for( int ii = 0; ii < currentPcb->GetAreaCount(); ii++ )
 245     {
 246         ITEM_PICKER picker( nullptr, (EDA_ITEM*) currentPcb->GetArea( ii ), UR_CHANGED );
 247         itemsList.PushItem( picker );
 248     }
 249 
 250     if( itemsList.GetCount() > 0 )
 251         SaveCopyInUndoList( itemsList, UR_CHANGED, wxPoint( 0.0, 0.0 ) );
 252     else
 253         fromEmpty = true;
 254 
 255     itemsList.ClearItemsList();
 256 
 257     // Execute plugin itself...

 258     ACTION_PLUGINS::SetActionRunning( true );
 259     aActionPlugin->Run();
 260     ACTION_PLUGINS::SetActionRunning( false );
 261 
 262     // Get back the undo buffer to fix some modifications

 263     PICKED_ITEMS_LIST* oldBuffer = NULL;
 264 
 265     if( fromEmpty )
 266     {
 267         oldBuffer = new PICKED_ITEMS_LIST();
 268         oldBuffer->m_Status = UR_NEW;
 269     }
 270     else
 271     {
 272         oldBuffer = PopCommandFromUndoList();
 273         wxASSERT( oldBuffer );
 274     }
 275 
 276     // Try do discover what was modified

 277     PICKED_ITEMS_LIST deletedItemsList;
 278 
 279     // The list of existing items after running the action script

 280     std::set<BOARD_ITEM*> currItemList;
 281     // Append tracks:

 282     for( TRACK* item : currentPcb->Tracks() )
 283         currItemList.insert( item );
 284 
 285     // Append modules:

 286     for( MODULE* item : currentPcb->Modules() )
 287         currItemList.insert( item );
 288 
 289     // Append drawings

 290     for( BOARD_ITEM* item : currentPcb->Drawings() )
 291         currItemList.insert( item );
 292 
 293     // Append zones outlines

 294     for( int ii = 0; ii < currentPcb->GetAreaCount(); ii++ )
 295         currItemList.insert( currentPcb->GetArea( ii ) );
 296 
 297     // Found deleted modules

 298     for( unsigned int i = 0; i < oldBuffer->GetCount(); i++ )
 299     {
 300         BOARD_ITEM* item = (BOARD_ITEM*) oldBuffer->GetPickedItem( i );
 301         ITEM_PICKER picker( nullptr, item, UR_DELETED );
 302 
 303         wxASSERT( item );
 304 
 305         if( currItemList.find( item ) == currItemList.end() )
 306             deletedItemsList.PushItem( picker );
 307     }
 308 
 309     // Mark deleted elements in undolist

 310     for( unsigned int i = 0; i < deletedItemsList.GetCount(); i++ )
 311     {
 312         oldBuffer->PushItem( deletedItemsList.GetItemWrapper( i ) );
 313     }
 314 
 315     // Find new modules

 316     for( MODULE* item : currentPcb->Modules() )
 317     {
 318         if( !oldBuffer->ContainsItem( item ) )
 319         {
 320             ITEM_PICKER picker( nullptr, item, UR_NEW );
 321             oldBuffer->PushItem( picker );
 322         }
 323     }
 324 
 325     for( TRACK* item : currentPcb->Tracks() )
 326     {
 327         if( !oldBuffer->ContainsItem( item ) )
 328         {
 329             ITEM_PICKER picker( nullptr, item, UR_NEW );
 330             oldBuffer->PushItem( picker );
 331         }
 332     }
 333 
 334     for( BOARD_ITEM* item : currentPcb->Drawings() )
 335     {
 336         if( !oldBuffer->ContainsItem( item ) )
 337         {
 338             ITEM_PICKER picker( nullptr, item, UR_NEW );
 339             oldBuffer->PushItem( picker );
 340         }
 341     }
 342 
 343     for( int ii = 0; ii < currentPcb->GetAreaCount(); ii++ )
 344     {
 345         if( !oldBuffer->ContainsItem( (EDA_ITEM*) currentPcb->GetArea( ii ) ) )
 346         {
 347             ITEM_PICKER picker( nullptr, (EDA_ITEM*) currentPcb->GetArea( ii ), UR_NEW );
 348             oldBuffer->PushItem( picker );
 349         }
 350     }
 351 
 352     if( oldBuffer->GetCount() )
 353     {
 354         OnModify();
 355         PushCommandToUndoList( oldBuffer );
 356     }
 357     else
 358     {
 359         delete oldBuffer;
 360     }
 361 
 362     ActivateGalCanvas();
 363 }

 366 void PCB_EDIT_FRAME::buildActionPluginMenus( ACTION_MENU* actionMenu )
 367 {
 368     if( !actionMenu ) // Should not occur.

 369         return;
 370 
 371     for( int ii = 0; ii < ACTION_PLUGINS::GetActionsCount(); ii++ )
 372     {
 373         wxMenuItem* item;
 374         ACTION_PLUGIN* ap = ACTION_PLUGINS::GetAction( ii );
 375         const wxBitmap& bitmap = ap->iconBitmap.IsOk() ? ap->iconBitmap : KiBitmap( hammer_xpm );
 376 
 377         item = AddMenuItem( actionMenu, wxID_ANY,  ap->GetName(), ap->GetDescription(), bitmap );
 378 
 379         Connect( item->GetId(), wxEVT_COMMAND_MENU_SELECTED,
 380                 wxCommandEventHandler( PCB_EDIT_FRAME::OnActionPluginMenu ) );
 381 
 382         ACTION_PLUGINS::SetActionMenu( ii, item->GetId() );
 383     }
 384 }

 387 void PCB_EDIT_FRAME::AddActionPluginTools()
 388 {
 389     bool need_separator = true;
 390     const auto& orderedPlugins = GetOrderedActionPlugins();
 391 
 392     for( const auto& ap : orderedPlugins )
 393     {
 394         if( GetActionPluginButtonVisible( ap->GetPluginPath(), ap->GetShowToolbarButton() ) )
 395         {
 396 
 397             if( need_separator )
 398             {
 399                 m_mainToolBar->AddScaledSeparator( this );
 400                 need_separator = false;
 401             }
 402 
 403             // Add button

 404             wxBitmap bitmap;
 405 
 406             if ( ap->iconBitmap.IsOk() )
 407                 bitmap = KiScaledBitmap( ap->iconBitmap, this );
 408             else
 409                 bitmap = KiScaledBitmap( hammer_xpm, this );
 410 
 411             wxAuiToolBarItem* button = m_mainToolBar->AddTool(
 412                     wxID_ANY, wxEmptyString, bitmap, ap->GetName() );
 413 
 414             Connect( button->GetId(), wxEVT_COMMAND_MENU_SELECTED,
 415                     wxCommandEventHandler( PCB_EDIT_FRAME::OnActionPluginButton ) );
 416 
 417             // Link action plugin to button

 418             ACTION_PLUGINS::SetActionButton( ap, button->GetId() );
 419         }
 420     }
 421 }

 424 std::vector<ACTION_PLUGIN*> PCB_EDIT_FRAME::GetOrderedActionPlugins()
 425 {
 426     std::vector<ACTION_PLUGIN*> plugins;
 427     std::vector<ACTION_PLUGIN*> orderedPlugins;
 428 
 429     for( int i = 0; i < ACTION_PLUGINS::GetActionsCount(); i++ )
 430         plugins.push_back( ACTION_PLUGINS::GetAction( i ) );
 431 
 432     // First add plugins that have entries in settings

 433     for( const auto& pair : m_Settings->m_VisibleActionPlugins )
 434     {
 435         auto loc = std::find_if( plugins.begin(), plugins.end(),
 436                 [pair] ( ACTION_PLUGIN* plugin )
 437                 {
 438                     return plugin->GetPluginPath() == pair.first;
 439                 } );
 440 
 441         if( loc != plugins.end() )
 442         {
 443             orderedPlugins.push_back( *loc );
 444             plugins.erase( loc );
 445         }
 446     }
 447 
 448     // Now append new plugins that have not been configured yet

 449     for( auto remaining_plugin : plugins )
 450         orderedPlugins.push_back( remaining_plugin );
 451 
 452     return orderedPlugins;
 453 }

 456 bool PCB_EDIT_FRAME::GetActionPluginButtonVisible( const wxString& aPluginPath, bool aPluginDefault )
 457 {
 458     auto& settings = m_Settings->m_VisibleActionPlugins;
 459 
 460     for( const auto& entry : settings )
 461     {
 462         if( entry.first == aPluginPath )
 463             return entry.second;
 464     }
 465 
 466     // Plugin is not in settings, return default.

 467     return  aPluginDefault;
 468 }
