 231 bool DIALOG_FOOTPRINT_BOARD_EDITOR::TransferDataToWindow()
 232 {
 233     if( !wxDialog::TransferDataToWindow() )
 234         return false;
 235 
 236     if( !m_PanelGeneral->TransferDataToWindow() )
 237         return false;
 238 
 239     if( !m_Panel3D->TransferDataToWindow() )
 240         return false;
 241 
 242     // Module Texts

 243 
 244     m_texts->push_back( m_footprint->Reference() );
 245     m_texts->push_back( m_footprint->Value() );
 246 
 247     for( auto item : m_footprint->GraphicalItems() )
 248     {
 249         auto textModule = dyn_cast<TEXTE_MODULE*>( item );
 250 
 251         if( textModule )
 252             m_texts->push_back( *textModule );
 253     }
 254 
 255     // notify the grid

 256     wxGridTableMessage tmsg( m_texts, wxGRIDTABLE_NOTIFY_ROWS_APPENDED, m_texts->GetNumberRows() );
 257     m_itemsGrid->ProcessTableMessage( tmsg );
 258 
 259     // Module Properties

 260 
 261     m_posX.SetValue( m_footprint->GetPosition().x );
 262     m_posY.SetValue( m_footprint->GetPosition().y );
 263 
 264     m_BoardSideCtrl->SetSelection( (m_footprint->GetLayer() == B_Cu) ? 1 : 0 );
 265 
 266     m_OrientValue = m_footprint->GetOrientation() / 10.0;
 267 
 268     if( m_OrientValue == 0.0 )
 269         m_Orient0->SetValue( true );
 270     else if( m_OrientValue == 90.0 || m_OrientValue == -270.0 )
 271         m_Orient90->SetValue( true );
 272     else if( m_OrientValue == 270.0 || m_OrientValue == -90.0 )
 273         m_Orient270->SetValue( true );
 274     else if( m_OrientValue == 180.0 || m_OrientValue == -180.0 )
 275         m_Orient180->SetValue( true );
 276     else
 277         m_OrientOther->SetValue( true );
 278 
 279     updateOrientationControl();
 280 
 281     if( m_footprint->IsLocked() )
 282         m_AutoPlaceCtrl->SetSelection( 2 );
 283     else if( m_footprint->PadsLocked() )
 284         m_AutoPlaceCtrl->SetSelection( 1 );
 285     else
 286         m_AutoPlaceCtrl->SetSelection( 0 );
 287 
 288     m_AutoPlaceCtrl->SetItemToolTip( 0, _( "Component can be freely moved and auto placed. User "
 289                                            "can arbitrarily select and edit component's pads." ) );
 290     m_AutoPlaceCtrl->SetItemToolTip( 1, _( "Component can be freely moved and auto placed, but "
 291                                            "its pads cannot be selected or edited." ) );
 292     m_AutoPlaceCtrl->SetItemToolTip( 2, _( "Component is locked: it cannot be freely moved or "
 293                                            "auto placed." ) );
 294 
 295     m_CostRot90Ctrl->SetValue( m_footprint->GetPlacementCost90() );
 296     m_CostRot180Ctrl->SetValue( m_footprint->GetPlacementCost180() );
 297 
 298     m_AttributsCtrl->SetItemToolTip( 0, _( "Use this attribute for most non SMD footprints\n"
 299             "Footprints with this option are not put in the footprint position list file" ) );
 300     m_AttributsCtrl->SetItemToolTip( 1, _( "Use this attribute for SMD footprints.\n"
 301             "Only footprints with this option are put in the footprint position list file" ) );
 302     m_AttributsCtrl->SetItemToolTip( 2, _( "Use this attribute for \"virtual\" footprints drawn "
 303             "on board\nsuch as an edge connector (old ISA PC bus for instance)" ) );
 304 
 305     switch( m_footprint->GetAttributes() & 255 )
 306     {
 307     case MOD_CMS:     m_AttributsCtrl->SetSelection( 1 ); break;
 308     case MOD_VIRTUAL: m_AttributsCtrl->SetSelection( 2 ); break;
 309     case 0:
 310     default:          m_AttributsCtrl->SetSelection( 0 ); break;
 311     }
 312 
 313     // Local Clearances

 314 
 315     m_netClearance.SetValue( m_footprint->GetLocalClearance() );
 316     m_solderMask.SetValue( m_footprint->GetLocalSolderMaskMargin() );
 317     m_solderPaste.SetValue( m_footprint->GetLocalSolderPasteMargin() );
 318 
 319     // Prefer "-0" to "0" for normally negative values

 320     if( m_footprint->GetLocalSolderPasteMargin() == 0 )
 321         m_SolderPasteMarginCtrl->SetValue( wxT( "-" ) + m_SolderPasteMarginCtrl->GetValue() );
 322 
 323     // Add solder paste margin ratio in percent

 324     // for the usual default value 0.0, display -0.0 (or -0,0 in some countries)

 325     wxString msg;
 326     msg.Printf( wxT( "%f" ), m_footprint->GetLocalSolderPasteMarginRatio() * 100.0 );
 327 
 328     if( m_footprint->GetLocalSolderPasteMarginRatio() == 0.0 &&
 329         msg[0] == '0')  // Sometimes Printf adds a sign if the value is very small (0.0)

 330         m_SolderPasteMarginRatioCtrl->SetValue( wxT("-") + msg );
 331     else
 332         m_SolderPasteMarginRatioCtrl->SetValue( msg );
 333 
 334     switch( m_footprint->GetZoneConnection() )
 335     {
 336     default:
 337     case ZONE_CONNECTION::INHERITED:
 338         m_ZoneConnectionChoice->SetSelection( 0 );
 339         break;
 340     case ZONE_CONNECTION::FULL:
 341         m_ZoneConnectionChoice->SetSelection( 1 );
 342         break;
 343     case ZONE_CONNECTION::THERMAL:
 344         m_ZoneConnectionChoice->SetSelection( 2 );
 345         break;
 346     case ZONE_CONNECTION::NONE:
 347         m_ZoneConnectionChoice->SetSelection( 3 );
 348         break;
 349     }
 350 
 351     // 3D Settings

 352 
 353     wxString default_path;
 354     wxGetEnv( KISYS3DMOD, &default_path );
 355 #ifdef __WINDOWS__
 356     default_path.Replace( wxT( "/" ), wxT( "\\" ) );
 357 #endif
 358 
 359     m_shapes3D_list.clear();
 360     m_modelsGrid->DeleteRows( 0, m_modelsGrid->GetNumberRows() );
 361 
 362     wxString origPath, alias, shortPath;
 363     FILENAME_RESOLVER* res = Prj().Get3DCacheManager()->GetResolver();
 364 
 365     for( const MODULE_3D_SETTINGS& model : m_footprint->Models() )
 366     {
 367         m_shapes3D_list.push_back( model );
 368         origPath = model.m_Filename;
 369 
 370         if( res && res->SplitAlias( origPath, alias, shortPath ) )
 371             origPath = alias + wxT( ":" ) + shortPath;
 372 
 373         m_modelsGrid->AppendRows( 1 );
 374         int row = m_modelsGrid->GetNumberRows() - 1;
 375         m_modelsGrid->SetCellValue( row, 0, origPath );
 376         m_modelsGrid->SetCellValue( row, 1, model.m_Show ? wxT( "1" ) : wxT( "0" ) );
 377     }
 378 
 379     select3DModel( 0 );   // will clamp idx within bounds

 380     m_PreviewPane->UpdateDummyModule();
 381 
 382     // Show the footprint's FPID.

 383     m_tcLibraryID->SetValue( m_footprint->GetFPID().Format() );
 384 
 385     // Show the footprint's full unique Kicad ID.

 386     m_tcFullUuid->SetValue( m_footprint->GetPath().AsString() );
 387 
 388     for( int col = 0; col < m_itemsGrid->GetNumberCols(); col++ )
 389     {
 390         m_itemsGrid->SetColMinimalWidth( col, m_itemsGrid->GetVisibleWidth( col, true, false,
 391                                                                             false ) );
 392         // Adjust the column size. The column 6 has a small bitmap, so its width must be

 393         // taken in account

 394         int col_size = m_itemsGrid->GetVisibleWidth( col, true, true, false );
 395 
 396         if( col == 6 )
 397             col_size += 20;
 398 
 399         if( m_itemsGrid->IsColShown( col ) )
 400             m_itemsGrid->SetColSize( col, col_size );
 401     }
 402 
 403     m_itemsGrid->SetRowLabelSize( m_itemsGrid->GetVisibleWidth( -1, false, true, true ) );
 404     m_modelsGrid->SetColSize( 1, m_modelsGrid->GetVisibleWidth( 1, true, false, false ) );
 405 
 406     Layout();
 407     adjustGridColumns( m_itemsGrid->GetRect().GetWidth());
 408 
 409     return true;
 410 }

 439 void DIALOG_FOOTPRINT_BOARD_EDITOR::On3DModelCellChanged( wxGridEvent& aEvent )
 440 {
 441     if( aEvent.GetCol() == 0 )
 442     {
 443         bool               hasAlias = false;
 444         FILENAME_RESOLVER* res = Prj().Get3DCacheManager()->GetResolver();
 445         wxString           filename = m_modelsGrid->GetCellValue( aEvent.GetRow(), 0 );
 446 
 447         filename.Replace( "\n", "" );
 448         filename.Replace( "\r", "" );
 449         filename.Replace( "\t", "" );
 450 
 451         if( filename.empty() || !res->ValidateFileName( filename, hasAlias ) )
 452         {
 453             m_delayedErrorMessage = wxString::Format( _( "Invalid filename: %s" ), filename );
 454             m_delayedFocusGrid = m_modelsGrid;
 455             m_delayedFocusRow = aEvent.GetRow();
 456             m_delayedFocusColumn = aEvent.GetCol();
 457             aEvent.Veto();
 458         }
 459 
 460         // if the user has specified an alias in the name then prepend ':'

 461         if( hasAlias )
 462             filename.insert( 0, wxT( ":" ) );
 463 
 464 #ifdef __WINDOWS__
 465         // In Kicad files, filenames and paths are stored using Unix notation

 466         filename.Replace( wxT( "\\" ), wxT( "/" ) );
 467 #endif
 468 
 469         m_shapes3D_list[ aEvent.GetRow() ].m_Filename = filename;
 470         m_modelsGrid->SetCellValue( aEvent.GetRow(), 0, filename );
 471     }
 472     else if( aEvent.GetCol() == 1 )
 473     {
 474         wxString showValue = m_modelsGrid->GetCellValue( aEvent.GetRow(), 1 );
 475 
 476         m_shapes3D_list[ aEvent.GetRow() ].m_Show = ( showValue == wxT( "1" ) );
 477     }
 478 
 479     m_PreviewPane->UpdateDummyModule();
 480 }

 500 void DIALOG_FOOTPRINT_BOARD_EDITOR::OnAdd3DModel( wxCommandEvent&  )
 501 {
 502     if( !m_modelsGrid->CommitPendingChanges() )
 503         return;
 504 
 505     int selected = m_modelsGrid->GetGridCursorRow();
 506 
 507     PROJECT& prj = Prj();
 508     MODULE_3D_SETTINGS model;
 509 
 510     wxString initialpath = prj.GetRString( PROJECT::VIEWER_3D_PATH );
 511     wxString sidx = prj.GetRString( PROJECT::VIEWER_3D_FILTER_INDEX );
 512     int filter = 0;
 513 
 514     // If the PROJECT::VIEWER_3D_PATH hasn't been set yet, use the KISYS3DMOD environment

 515     // variable and fall back to the project path if necessary.

 516     if( initialpath.IsEmpty() )
 517     {
 518         if( !wxGetEnv( "KISYS3DMOD", &initialpath ) || initialpath.IsEmpty() )
 519             initialpath = prj.GetProjectPath();
 520     }
 521 
 522     if( !sidx.empty() )
 523     {
 524         long tmp;
 525         sidx.ToLong( &tmp );
 526 
 527         if( tmp > 0 && tmp <= INT_MAX )
 528             filter = (int) tmp;
 529     }
 530 
 531     if( !S3D::Select3DModel( this, Prj().Get3DCacheManager(), initialpath, filter, &model )
 532         || model.m_Filename.empty() )
 533     {
 534         select3DModel( selected );
 535         return;
 536     }
 537 
 538     prj.SetRString( PROJECT::VIEWER_3D_PATH, initialpath );
 539     sidx = wxString::Format( wxT( "%i" ), filter );
 540     prj.SetRString( PROJECT::VIEWER_3D_FILTER_INDEX, sidx );
 541     FILENAME_RESOLVER* res = Prj().Get3DCacheManager()->GetResolver();
 542     wxString alias;
 543     wxString shortPath;
 544     wxString filename = model.m_Filename;
 545 
 546     if( res && res->SplitAlias( filename, alias, shortPath ) )
 547         filename = alias + wxT( ":" ) + shortPath;
 548 
 549 #ifdef __WINDOWS__
 550     // In KiCad files, filenames and paths are stored using Unix notation

 551     model.m_Filename.Replace( "\\", "/" );
 552 #endif
 553 
 554     model.m_Show = true;
 555     m_shapes3D_list.push_back( model );
 556 
 557     int idx = m_modelsGrid->GetNumberRows();
 558     m_modelsGrid->AppendRows( 1 );
 559     m_modelsGrid->SetCellValue( idx, 0, filename );
 560     m_modelsGrid->SetCellValue( idx, 1, wxT( "1" ) );
 561 
 562     select3DModel( idx );
 563     m_PreviewPane->UpdateDummyModule();
 564 }

 592 bool DIALOG_FOOTPRINT_BOARD_EDITOR::Validate()
 593 {
 594     if( !m_itemsGrid->CommitPendingChanges() )
 595         return false;
 596 
 597     if( !DIALOG_SHIM::Validate() )
 598         return false;
 599 
 600     // Check for empty texts.

 601     for( size_t i = 2; i < m_texts->size(); ++i )
 602     {
 603         TEXTE_MODULE& text = m_texts->at( i );
 604 
 605         if( text.GetText().IsEmpty() )
 606         {
 607             if( m_NoteBook->GetSelection() != 0 )
 608                 m_NoteBook->SetSelection( 0 );
 609 
 610             m_delayedFocusGrid = m_itemsGrid;
 611             m_delayedErrorMessage = _( "Text items must have some content." );
 612             m_delayedFocusColumn = TMC_TEXT;
 613             m_delayedFocusRow = i;
 614 
 615             return false;
 616         }
 617     }
 618 
 619     if( !m_netClearance.Validate( 0, INT_MAX ) )
 620         return false;
 621 
 622     return true;
 623 }

 626 bool DIALOG_FOOTPRINT_BOARD_EDITOR::TransferDataFromWindow()
 627 {
 628     if( !Validate() )
 629         return false;
 630 
 631     if( !DIALOG_SHIM::TransferDataFromWindow() )
 632         return false;
 633 
 634     if( !m_PanelGeneral->TransferDataFromWindow() )
 635         return false;
 636 
 637     if( !m_Panel3D->TransferDataFromWindow() )
 638         return false;
 639 
 640     auto view = m_frame->GetCanvas()->GetView();
 641     BOARD_COMMIT commit( m_frame );
 642     commit.Modify( m_footprint );
 643 
 644     // copy reference and value

 645     m_footprint->Reference() = m_texts->at( 0 );
 646     m_footprint->Value() = m_texts->at( 1 );
 647 
 648     size_t i = 2;
 649 
 650     for( auto item : m_footprint->GraphicalItems() )
 651     {
 652         TEXTE_MODULE* textModule = dyn_cast<TEXTE_MODULE*>( item );
 653 
 654         if( textModule )
 655         {
 656             // copy grid table entries till we run out, then delete any remaining texts

 657             if( i < m_texts->size() )
 658                 *textModule = m_texts->at( i++ );
 659             else
 660                 textModule->DeleteStructure();
 661         }
 662     }
 663 
 664     // if there are still grid table entries, create new texts for them

 665     while( i < m_texts->size() )
 666     {
 667         auto newText = new TEXTE_MODULE( m_texts->at( i++ ) );
 668         m_footprint->Add( newText, ADD_MODE::APPEND );
 669         view->Add( newText );
 670     }
 671 
 672     // Initialize masks clearances

 673     m_footprint->SetLocalClearance( m_netClearance.GetValue() );
 674     m_footprint->SetLocalSolderMaskMargin( m_solderMask.GetValue() );
 675     m_footprint->SetLocalSolderPasteMargin( m_solderPaste.GetValue() );
 676 
 677     double dtmp = 0.0;
 678     wxString msg = m_SolderPasteMarginRatioCtrl->GetValue();
 679     msg.ToDouble( &dtmp );
 680 
 681     // A -50% margin ratio means no paste on a pad, the ratio must be >= -50%

 682     if( dtmp < -50.0 )
 683         dtmp = -50.0;
 684     // A margin ratio is always <= 0

 685     // 0 means use full pad copper area

 686     if( dtmp > 0.0 )
 687         dtmp = 0.0;
 688 
 689     m_footprint->SetLocalSolderPasteMarginRatio( dtmp / 100 );
 690 
 691     switch( m_ZoneConnectionChoice->GetSelection() )
 692     {
 693     default:
 694     case 0:
 695         m_footprint->SetZoneConnection( ZONE_CONNECTION::INHERITED );
 696         break;
 697     case 1:
 698         m_footprint->SetZoneConnection( ZONE_CONNECTION::FULL );
 699         break;
 700     case 2:
 701         m_footprint->SetZoneConnection( ZONE_CONNECTION::THERMAL );
 702         break;
 703     case 3:
 704         m_footprint->SetZoneConnection( ZONE_CONNECTION::NONE );
 705         break;
 706     }
 707 
 708     // Set Module Position

 709     wxPoint modpos( m_posX.GetValue(), m_posY.GetValue() );
 710     m_footprint->SetPosition( modpos );
 711     m_footprint->SetLocked( m_AutoPlaceCtrl->GetSelection() == 2 );
 712     m_footprint->SetPadsLocked( m_AutoPlaceCtrl->GetSelection() == 1 );
 713 
 714     switch( m_AttributsCtrl->GetSelection() )
 715     {
 716     case 0:  m_footprint->SetAttributes( 0 );           break;
 717     case 1:  m_footprint->SetAttributes( MOD_CMS );     break;
 718     case 2:  m_footprint->SetAttributes( MOD_VIRTUAL ); break;
 719     default: wxFAIL;
 720     }
 721 
 722     m_footprint->SetPlacementCost90( m_CostRot90Ctrl->GetValue() );
 723     m_footprint->SetPlacementCost180( m_CostRot180Ctrl->GetValue() );
 724 
 725     // Now, set orientation.  Must be done after other changes because rotation changes field

 726     // positions on board (so that relative positions are held constant)

 727     m_OrientValidator.TransferFromWindow();
 728 
 729     int orient = KiROUND( m_OrientValue * 10.0 );
 730 
 731     if( m_footprint->GetOrientation() != orient )
 732         m_footprint->Rotate( m_footprint->GetPosition(), orient - m_footprint->GetOrientation() );
 733 
 734     // Set component side, that also have effect on the fields positions on board

 735     bool change_layer = false;
 736     if( m_BoardSideCtrl->GetSelection() == 0 )     // layer req = COMPONENT

 737     {
 738         if( m_footprint->GetLayer() == B_Cu )
 739             change_layer = true;
 740     }
 741     else if( m_footprint->GetLayer() == F_Cu )
 742         change_layer = true;
 743 
 744     if( change_layer )
 745         m_footprint->Flip( m_footprint->GetPosition(), m_frame->Settings().m_FlipLeftRight );
 746 
 747     std::list<MODULE_3D_SETTINGS>* draw3D = &m_footprint->Models();
 748     draw3D->clear();
 749     draw3D->insert( draw3D->end(), m_shapes3D_list.begin(), m_shapes3D_list.end() );
 750 
 751     m_footprint->CalculateBoundingBox();
 752 
 753     // This is a simple edit, we must create an undo entry

 754     if( m_footprint->GetEditFlags() == 0 )    // i.e. not edited, or moved

 755         commit.Push( _( "Modify module properties" ) );
 756 
 757     SetReturnCode( PRM_EDITOR_EDIT_OK );
 758     return true;
 759 }

 762 void DIALOG_FOOTPRINT_BOARD_EDITOR::OnAddField( wxCommandEvent&  )
 763 {
 764     if( !m_itemsGrid->CommitPendingChanges() )
 765         return;
 766 
 767     const BOARD_DESIGN_SETTINGS& dsnSettings = m_frame->GetDesignSettings();
 768     TEXTE_MODULE textMod( m_footprint );
 769 
 770     // Set active layer if legal; otherwise copy layer from previous text item

 771     if( LSET::AllTechMask().test( m_frame->GetActiveLayer() ) )
 772         textMod.SetLayer( m_frame->GetActiveLayer() );
 773     else
 774         textMod.SetLayer( m_texts->at( m_texts->size() - 1 ).GetLayer() );
 775 
 776     textMod.SetTextSize( dsnSettings.GetTextSize( textMod.GetLayer() ) );
 777     textMod.SetTextThickness( dsnSettings.GetTextThickness( textMod.GetLayer() ) );
 778     textMod.SetItalic( dsnSettings.GetTextItalic( textMod.GetLayer() ) );
 779     textMod.SetKeepUpright( dsnSettings.GetTextUpright( textMod.GetLayer() ) );
 780     textMod.SetMirrored( IsBackLayer( textMod.GetLayer() ) );
 781 
 782     m_texts->push_back( textMod );
 783 
 784     // notify the grid

 785     wxGridTableMessage msg( m_texts, wxGRIDTABLE_NOTIFY_ROWS_APPENDED, 1 );
 786     m_itemsGrid->ProcessTableMessage( msg );
 787 
 788     m_itemsGrid->SetFocus();
 789     m_itemsGrid->MakeCellVisible( m_texts->size() - 1, 0 );
 790     m_itemsGrid->SetGridCursor( m_texts->size() - 1, 0 );
 791 
 792     m_itemsGrid->EnableCellEditControl( true );
 793     m_itemsGrid->ShowCellEditControl();
 794 }

 797 void DIALOG_FOOTPRINT_BOARD_EDITOR::OnDeleteField( wxCommandEvent&  )
 798 {
 799     m_itemsGrid->CommitPendingChanges( true /* quiet mode */ );
 800 
 801     int curRow   = m_itemsGrid->GetGridCursorRow();
 802 
 803     if( curRow < 0 )
 804         return;
 805     else if( curRow < 2 )
 806     {
 807         DisplayError( nullptr, _( "Reference and value are mandatory." ) );
 808         return;
 809     }
 810 
 811     m_texts->erase( m_texts->begin() + curRow );
 812 
 813     // notify the grid

 814     wxGridTableMessage msg( m_texts, wxGRIDTABLE_NOTIFY_ROWS_DELETED, curRow, 1 );
 815     m_itemsGrid->ProcessTableMessage( msg );
 816 
 817     if( m_itemsGrid->GetNumberRows() > 0 )
 818     {
 819         m_itemsGrid->MakeCellVisible( std::max( 0, curRow-1 ), m_itemsGrid->GetGridCursorCol() );
 820         m_itemsGrid->SetGridCursor( std::max( 0, curRow-1 ), m_itemsGrid->GetGridCursorCol() );
 821     }
 822 }

 832 void DIALOG_FOOTPRINT_BOARD_EDITOR::adjustGridColumns( int aWidth )
 833 {
 834     // Account for scroll bars

 835     int itemsWidth = aWidth - ( m_itemsGrid->GetSize().x - m_itemsGrid->GetClientSize().x );
 836     int modelsWidth = aWidth - ( m_modelsGrid->GetSize().x - m_modelsGrid->GetClientSize().x );
 837 
 838     itemsWidth -= m_itemsGrid->GetRowLabelSize();
 839 
 840     for( int i = 1; i < m_itemsGrid->GetNumberCols(); i++ )
 841         itemsWidth -= m_itemsGrid->GetColSize( i );
 842 
 843     if( itemsWidth > 0 )
 844     {
 845         m_itemsGrid->SetColSize( 0, std::max( itemsWidth,
 846                 m_itemsGrid->GetVisibleWidth( 0, true, false, false ) ) );
 847     }
 848 
 849     m_modelsGrid->SetColSize( 0, modelsWidth - m_modelsGrid->GetColSize( 1 ) - 6 );
 850 }

 853 void DIALOG_FOOTPRINT_BOARD_EDITOR::OnUpdateUI( wxUpdateUIEvent&  )
 854 {
 855     if( !m_itemsGrid->IsCellEditControlShown() && !m_modelsGrid->IsCellEditControlShown() )
 856         adjustGridColumns( m_itemsGrid->GetRect().GetWidth());
 857 
 858     // Handle a grid error.  This is delayed to OnUpdateUI so that we can change focus

 859     // even when the original validation was triggered from a killFocus event, and so

 860     // that the corresponding notebook page can be shown in the background when triggered

 861     // from an OK.

 862     if( m_delayedFocusRow >= 0 )
 863     {
 864         // We will re-enter this routine if an error dialog is displayed, so make sure we

 865         // zero out our member variables first.

 866         wxGrid*  grid = m_delayedFocusGrid;
 867         int      row = m_delayedFocusRow;
 868         int      col = m_delayedFocusColumn;
 869         wxString msg = m_delayedErrorMessage;
 870 
 871         m_delayedFocusGrid = nullptr;
 872         m_delayedFocusRow = -1;
 873         m_delayedFocusColumn = -1;
 874         m_delayedErrorMessage = wxEmptyString;
 875 
 876         if( !msg.IsEmpty() )
 877         {
 878             // Do not use DisplayErrorMessage(); it screws up window order on Mac

 879             DisplayError( nullptr, msg );
 880         }
 881 
 882         grid->SetFocus();
 883         grid->MakeCellVisible( row, col );
 884 
 885         // Selecting the first grid item only makes sense for the

 886         // items grid

 887         if( !m_initialFocus || grid == m_itemsGrid )
 888         {
 889             grid->SetGridCursor( row, col );
 890             grid->EnableCellEditControl( true );
 891             grid->ShowCellEditControl();
 892 
 893             if( grid == m_itemsGrid && row == 0 && col == 0 )
 894             {
 895                 auto referenceEditor = grid->GetCellEditor( 0, 0 );
 896 
 897                 if( auto textEntry = dynamic_cast<wxTextEntry*>( referenceEditor->GetControl() ) )
 898                     SelectReferenceNumber( textEntry );
 899 
 900                 referenceEditor->DecRef();
 901             }
 902         }
 903         m_initialFocus = false;
 904     }
 905 
 906     m_buttonRemove->Enable( m_modelsGrid->GetNumberRows() > 0 );
 907 }
