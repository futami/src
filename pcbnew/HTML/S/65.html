  44 class EDGE_MODULE : public DRAWSEGMENT
  45 {
  46 public:
  47     EDGE_MODULE( MODULE* parent, STROKE_T aShape = S_SEGMENT );
  48 
  49     // Do not create a copy constructor & operator=.

  50     // The ones generated by the compiler are adequate.

  51 
  52     ~EDGE_MODULE();
  53 
  54     static inline bool ClassOf( const EDA_ITEM* aItem )
  55     {
  56         return aItem && PCB_MODULE_EDGE_T == aItem->Type();
  57     }
  58 
  59     bool IsType( const KICAD_T aScanTypes[] ) const override
  60     {
  61         if( BOARD_ITEM::IsType( aScanTypes ) )
  62             return true;
  63 
  64         for( const KICAD_T* p = aScanTypes; *p != EOT; ++p )
  65         {
  66             if( *p == PCB_LOCATE_GRAPHIC_T )
  67                 return true;
  68             else if( *p == PCB_LOCATE_BOARD_EDGE_T )
  69                 return m_Layer == Edge_Cuts;
  70         }
  71 
  72         return false;
  73     }
  74 
  75     /**

  76      * Move an edge of the footprint.

  77      * This is a footprint shape modification.

  78      * (should be only called by a footprint editing function)

  79      */
  80     void Move( const wxPoint& aMoveVector ) override;
  81 
  82     /**

  83      * Mirror an edge of the footprint.

  84      * Do not change the layer

  85      * This is a footprint shape modification.

  86      * (should be only called by a footprint editing function)

  87      */
  88     void Mirror( const wxPoint& aCentre, bool aMirrorAroundXAxis );
  89 
  90     /**

  91      * Rotate an edge of the footprint.

  92      * This is a footprint shape modification.

  93      * (should be only called by a footprint editing function )

  94      */
  95     void Rotate( const wxPoint& aRotCentre, double aAngle ) override;
  96 
  97     /**

  98      * Flip entity relative to aCentre.

  99      * The item is mirrored, and layer changed to the paired corresponding layer if it is on

 100      * a paired layer.

 101      * This function should be called only from MODULE::Flip because it is not usual to flip

 102      * an item alone, without flipping the parent footprint (consider Mirror() instead).

 103      */
 104     void Flip( const wxPoint& aCentre, bool aFlipLeftRight ) override;
 105 
 106     bool IsParentFlipped() const;
 107 
 108     void SetStart0( const wxPoint& aPoint )     { m_Start0 = aPoint; }
 109     const wxPoint& GetStart0() const            { return m_Start0; }
 110 
 111     void SetEnd0( const wxPoint& aPoint )       { m_End0 = aPoint; }
 112     const wxPoint& GetEnd0() const              { return m_End0; }
 113 
 114     void SetBezier0_C1( const wxPoint& aPoint ) { m_Bezier0_C1 = aPoint; }
 115     const wxPoint& GetBezier0_C1() const        { return m_Bezier0_C1; }
 116 
 117     void SetBezier0_C2( const wxPoint& aPoint ) { m_Bezier0_C2 = aPoint; }
 118     const wxPoint& GetBezier0_C2() const        { return m_Bezier0_C2; }
 119 
 120     /**

 121      * Set relative coordinates from draw coordinates.

 122      * Call in only when the geometry or the footprint is modified and therefore the relative

 123      * coordinates have to be updated from the draw coordinates.

 125     void SetLocalCoord();

 128      * Set draw coordinates (absolute values ) from relative coordinates.

 130      */
