  61 /**

  62  * ZONE_SETTINGS

  63  * handles zones parameters.

  64  * Because a zone can be on copper or non copper layers, and can be also

  65  * a keepout area, some parameters are irrelevant depending on the type of zone

  66  */
  67 class ZONE_SETTINGS
  68 {
  69 public:
  70     // the actual zone outline shape can be slightly modified (smoothed):

  71     enum {
  72         SMOOTHING_UNDEFINED = -1,
  73         SMOOTHING_NONE = 0,         // Zone outline is used without change

  74         SMOOTHING_CHAMFER,          // Zone outline is used after chamfering corners

  75         SMOOTHING_FILLET,           // Zone outline is used after rounding corners

  76         SMOOTHING_LAST              // sentinel

  77     };
  78 
  79     ZONE_FILL_MODE  m_FillMode;
  80 
  81     int  m_ZonePriority;                ///< Priority (0 ... N) of the zone

  82 
  83     int  m_ZoneClearance;               ///< Minimal clearance value

  84     int  m_ZoneMinThickness;            ///< Min thickness value in filled areas

  85     int  m_HatchFillTypeThickness;      ///< Grid style shape: thickness of lines (if 0 -> solid shape)

  86     int  m_HatchFillTypeGap;            ///< Grid style shape: clearance between lines (0 -> solid shape)

  87     double m_HatchFillTypeOrientation;  ///< Grid style shape: orientation of grid lines in degrees

  88     int  m_HatchFillTypeSmoothingLevel; ///< Grid pattern smoothing type, similar to corner smoothing type

  89                                         ///< 0 = no smoothing, 1 = fillet, >= 2 = arc

  90     double m_HatchFillTypeSmoothingValue; ///< Grid pattern chamfer distance/fillet value

  91                                         ///< this is the ratio between the gap and the chamfer size

  92 
  93     int  m_NetcodeSelection;            ///< Net code selection for the current zone

  94 
  95     wxString m_Name;                    ///< Unique name for the current zone (can be blank)

  96 
  97     LSET m_Layers;                      ///< Layers that this zone exists on

  98 
  99     /// Option to show the zone area (outlines only, short hatches or full hatches

 100     ZONE_HATCH_STYLE m_Zone_HatchingStyle;
 101 
 102     long m_ThermalReliefGap;            ///< thickness of the gap in thermal reliefs

 103     long m_ThermalReliefCopperBridge;   ///< thickness of the copper bridge in thermal reliefs

 104 
 105     bool m_Zone_45_Only;
 106 
 107 private:
 108     int  m_cornerSmoothingType;           ///< Corner smoothing type

 109     unsigned int  m_cornerRadius;         ///< Corner chamfer distance / fillet radius

 110     ZONE_CONNECTION m_PadConnection;
 111 
 112     /* A zone outline can be a keepout zone.

 113      * It will be never filled, and DRC should test for pads, tracks and vias

 114      */
 115     bool                  m_isKeepout;
 116 
 117     /* For keepout zones only:

 118      * what is not allowed inside the keepout ( pads, tracks and vias )

 119      */
 120     bool m_keepoutDoNotAllowCopperPour;
 121     bool m_keepoutDoNotAllowVias;
 122     bool m_keepoutDoNotAllowTracks;
 123     bool m_keepoutDoNotAllowPads;
 124     bool m_keepoutDoNotAllowFootprints;
 125 
 126     ISLAND_REMOVAL_MODE m_removeIslands;
 127     long long int       m_minIslandArea;
 128 
 129 public:
 130     ZONE_SETTINGS();
 131 
 132     /**

 133      * operator << ( const ZONE_CONTAINER& )

 134      * was Function ImportSetting

 135      * copies settings from a given zone into this object.

 136      * @param aSource: the given zone

 137      */
 138     ZONE_SETTINGS& operator << ( const ZONE_CONTAINER& aSource );
 139 
 140     /**

 141      * A helper routine for the various zone dialogs (copper, non-copper, keepout).

 142      * @param aList the wxDataViewListCtrl to populate

 143      * @param aFrame the parent editor frame

 144      * @param aShowCopper indicates whether copper or technical layers should be shown

 145      * @param aFpEditorMode true to show (when aShowCopper = true) the option: all inner layers

 146      */
 147     void SetupLayersList( wxDataViewListCtrl* aList, PCB_BASE_FRAME* aFrame,
 148                           bool aShowCopper, bool aFpEditorMode = false );
 149 
 150     /**

 151      * Function ExportSetting

 152      * copy settings to a given zone

 153      * @param aTarget: the given zone

 154      * @param aFullExport: if false: some parameters are NOT exported

 155      *   because they must not be  exported when export settings from a zone to others zones

 156      *   Currently:

 157      *      m_NetcodeSelection

 158      */
 159     void ExportSetting( ZONE_CONTAINER& aTarget, bool aFullExport = true ) const;
 160 
 161     void SetCornerSmoothingType( int aType) { m_cornerSmoothingType = aType; }
 162 
 163     int GetCornerSmoothingType() const { return m_cornerSmoothingType; }
 164 
 165     void SetCornerRadius( int aRadius );
 166 
 167     unsigned int GetCornerRadius() const { return m_cornerRadius; }
 168 
 169     ZONE_CONNECTION GetPadConnection() const
 170     {
 171         return m_PadConnection;
 172     }
 173 
 174     void SetPadConnection( ZONE_CONNECTION aPadConnection )
 175     {
 176         m_PadConnection = aPadConnection;
 177     }
 178 
 179     /**

 180      * Accessors to parameters used in Keepout zones:

 181      */
 182     const bool GetIsKeepout() const { return m_isKeepout; }
 183     const bool GetDoNotAllowCopperPour() const { return m_keepoutDoNotAllowCopperPour; }
 184     const bool GetDoNotAllowVias() const { return m_keepoutDoNotAllowVias; }
 185     const bool GetDoNotAllowTracks() const { return m_keepoutDoNotAllowTracks; }
 186     const bool GetDoNotAllowPads() const { return m_keepoutDoNotAllowPads; }
 187     const bool GetDoNotAllowFootprints() const { return m_keepoutDoNotAllowFootprints; }
 188 
 189     void SetIsKeepout( bool aEnable ) { m_isKeepout = aEnable; }
 190     void SetDoNotAllowCopperPour( bool aEnable ) { m_keepoutDoNotAllowCopperPour = aEnable; }
 191     void SetDoNotAllowVias( bool aEnable ) { m_keepoutDoNotAllowVias = aEnable; }
 192     void SetDoNotAllowTracks( bool aEnable ) { m_keepoutDoNotAllowTracks = aEnable; }
 193     void SetDoNotAllowPads( bool aEnable ) { m_keepoutDoNotAllowPads = aEnable; }
 194     void SetDoNotAllowFootprints( bool aEnable ) { m_keepoutDoNotAllowFootprints = aEnable; }
 195 
 196     const ISLAND_REMOVAL_MODE GetIslandRemovalMode() const { return m_removeIslands; }
 197     void SetIslandRemovalMode( ISLAND_REMOVAL_MODE aRemove ) { m_removeIslands = aRemove; }
 198 
 199     long long int GetMinIslandArea() const { return m_minIslandArea; }
 200     void SetMinIslandArea( long long int aArea ) { m_minIslandArea = aArea; }
 201 };
