  65 /**

  66  * Function AskLoadBoardFileName

  67  * puts up a wxFileDialog asking for a BOARD filename to open.

  68  *

  69  * @param aParent is a wxFrame passed to wxFileDialog.

  70  * @param aCtl is where to put the OpenProjectFiles() control bits.

  71  *

  72  * @param aFileName on entry is a probable choice, on return is the chosen filename.

  73  * @param aKicadFilesOnly true to list kiacad pcb files plugins only, false to list import plugins.

  74  *

  75  * @return bool - true if chosen, else false if user aborted.

  76  */
  77 bool AskLoadBoardFileName( wxWindow* aParent, int* aCtl, wxString* aFileName, bool aKicadFilesOnly )
  78 {
  79     // This is a subset of all PLUGINs which are trusted to be able to

  80     // load a BOARD. User may occasionally use the wrong plugin to load a

  81     // *.brd file (since both legacy and eagle use *.brd extension),

  82     // but eventually *.kicad_pcb will be more common than legacy *.brd files.

  83 
  84     // clang-format off

  85     static const struct
  86     {
  87         const wxString&     filter;
  88         IO_MGR::PCB_FILE_T  pluginType;
  89     } loaders[] =
  90     {
  91         { PcbFileWildcard(),                    IO_MGR::KICAD_SEXP },            // Current Kicad board files

  92         { LegacyPcbFileWildcard(),              IO_MGR::LEGACY },                // Old Kicad board files

  93         { EaglePcbFileWildcard(),               IO_MGR::EAGLE },                 // Import Eagle board files

  94         { PCadPcbFileWildcard(),                IO_MGR::PCAD },                  // Import PCAD board files

  95         { AltiumDesignerPcbFileWildcard(),      IO_MGR::ALTIUM_DESIGNER },       // Import Altium Designer board files

  96         { AltiumCircuitStudioPcbFileWildcard(), IO_MGR::ALTIUM_CIRCUIT_STUDIO }, // Import Altium Circuit Studio board files

  97         { AltiumCircuitMakerPcbFileWildcard(),  IO_MGR::ALTIUM_CIRCUIT_MAKER },  // Import Altium Circuit Maker board files

  98     };
  99     // clang-format on

 100 
 101     wxFileName  fileName( *aFileName );
 102     wxString    fileFilters;
 103 
 104     if( aKicadFilesOnly )
 105     {
 106         for( unsigned ii = 0; ii < 2; ++ii )
 107         {
 108             if( !fileFilters.IsEmpty() )
 109                 fileFilters += wxChar( '|' );
 110 
 111             fileFilters += wxGetTranslation( loaders[ii].filter );
 112         }
 113     }
 114     else
 115     {
 116         for( unsigned ii = 2; ii < arrayDim( loaders ); ++ii )
 117         {
 118             if( !fileFilters.IsEmpty() )
 119                 fileFilters += wxChar( '|' );
 120 
 121             fileFilters += wxGetTranslation( loaders[ii].filter );
 122         }
 123     }
 124 
 125     wxString    path;
 126     wxString    name;
 127 
 128     if( fileName.FileExists() )
 129     {
 130         path = fileName.GetPath();
 131         name = fileName.GetFullName();
 132     }
 133     else
 134     {
 135         path = wxStandardPaths::Get().GetDocumentsDir();
 136         // leave name empty

 137     }
 138 
 139     wxFileDialog dlg( aParent,
 140                       aKicadFilesOnly ? _( "Open Board File" ) : _( "Import Non KiCad Board File" ),
 141                       path, name, fileFilters,
 142                       wxFD_OPEN | wxFD_FILE_MUST_EXIST );
 143 
 144     if( dlg.ShowModal() == wxID_OK )
 145     {
 146         // For import option, if Eagle (*.brd files), tell OpenProjectFiles() to use Eagle plugin.

 147         // It's the only special case because of the duplicate use of the *.brd file extension.

 148         // Other cases are clear because of unique file extensions.

 149         *aCtl = aKicadFilesOnly ? 0 : KICTL_EAGLE_BRD;
 150         *aFileName = dlg.GetPath();
 151         return true;
 152     }
 153     else
 154         return false;
 155 }

 158 /**

 159  * Function AskSaveBoardFileName

 160  * puts up a wxFileDialog asking for a BOARD filename to save.

 161  *

 162  * @param aParent is a wxFrame passed to wxFileDialog.

 163  * @param aFileName on entry is a probable choice, on return is the

 164  *  chosen full filename (includes path).

 165  *

 166  * @return bool - true if chosen, else false if user aborted.

 167  */
 168 bool AskSaveBoardFileName( wxWindow* aParent, wxString* aFileName )
 169 {
 170     wxString    wildcard =  PcbFileWildcard();
 171     wxFileName  fn = *aFileName;
 172 
 173     fn.SetExt( KiCadPcbFileExtension );
 174 
 175     wxFileDialog dlg( aParent,
 176             _( "Save Board File As" ),
 177             fn.GetPath(),
 178             fn.GetFullName(),
 179             wildcard,
 180             wxFD_SAVE | wxFD_OVERWRITE_PROMPT
 181             );
 182 
 183     if( dlg.ShowModal() != wxID_OK )
 184         return false;
 185 
 186     fn = dlg.GetPath();
 187 
 188     // always enforce filename extension, user may not have entered it.

 189     fn.SetExt( KiCadPcbFileExtension );
 190 
 191     *aFileName = fn.GetFullPath();
 192 
 193     return true;
 194 }

 197 void PCB_EDIT_FRAME::OnFileHistory( wxCommandEvent& event )
 198 {
 199     wxString fn = GetFileFromHistory( event.GetId(), _( "Printed circuit board" ) );
 200 
 201     if( !!fn )
 202     {
 203         int open_ctl = 0;
 204 
 205         if( !wxFileName::IsFileReadable( fn ) )
 206         {
 207             if( !AskLoadBoardFileName( this, &open_ctl, &fn, true ) )
 208                 return;
 209         }
 210 
 211         OpenProjectFiles( std::vector<wxString>( 1, fn ), open_ctl );
 212     }

 229 bool PCB_EDIT_FRAME::Files_io_from_id( int id )
 230 {
 231     wxString   msg;
 232 
 233     switch( id )
 234     {
 235     case ID_LOAD_FILE:
 236         {
 237             int         open_ctl = 0;
 238             wxString    fileName = Prj().AbsolutePath( GetBoard()->GetFileName() );
 239 
 240             return AskLoadBoardFileName( this, &open_ctl, &fileName, true )
 241                        && OpenProjectFiles( std::vector<wxString>( 1, fileName ), open_ctl );
 242         }
 243 
 244     case ID_IMPORT_NON_KICAD_BOARD:
 245         {
 246             int         open_ctl = 1;
 247             wxString    fileName; // = Prj().AbsolutePath( GetBoard()->GetFileName() );

 248 
 249             return AskLoadBoardFileName( this, &open_ctl, &fileName, false )
 250                        && OpenProjectFiles( std::vector<wxString>( 1, fileName ), open_ctl );
 251         }
 252 
 253     case ID_MENU_RECOVER_BOARD_AUTOSAVE:
 254         {
 255             wxFileName currfn = Prj().AbsolutePath( GetBoard()->GetFileName() );
 256             wxFileName fn = currfn;
 257 
 258             wxString rec_name = GetAutoSaveFilePrefix() + fn.GetName();
 259             fn.SetName( rec_name );
 260 
 261             if( !fn.FileExists() )
 262             {
 263                 msg.Printf( _( "Recovery file \"%s\" not found." ), fn.GetFullPath() );
 264                 DisplayInfoMessage( this, msg );
 265                 return false;
 266             }
 267 
 268             msg.Printf( _( "OK to load recovery file \"%s\"" ), fn.GetFullPath() );
 269 
 270             if( !IsOK( this, msg ) )
 271                 return false;
 272 
 273             GetScreen()->ClrModify();    // do not prompt the user for changes

 274 
 275             if( OpenProjectFiles( std::vector<wxString>( 1, fn.GetFullPath() ) ) )
 276             {
 277                 // Re-set the name since name or extension was changed

 278                 GetBoard()->SetFileName( currfn.GetFullPath() );
 279                 UpdateTitle();
 280                 return true;
 281             }
 282             return false;
 283         }
 284 
 285     case ID_NEW_BOARD:
 286     {
 287         if( IsContentModified() )
 288         {
 289             wxFileName fileName = GetBoard()->GetFileName();
 290             wxString   saveMsg =
 291                     _( "Current board will be closed, save changes to \"%s\" before continuing?" );
 292 
 293             if( !HandleUnsavedChanges( this, wxString::Format( saveMsg, fileName.GetFullName() ),
 294                                        [&]()->bool { return Files_io_from_id( ID_SAVE_BOARD ); } ) )
 295                 return false;
 296         }
 297         else if( !GetBoard()->IsEmpty() )
 298         {
 299             if( !IsOK( this, _( "Current Board will be closed. Continue?" ) ) )
 300                 return false;
 301         }
 302 
 303         SaveProjectSettings();
 304 
 305         GetBoard()->ClearProject();
 306 
 307         SETTINGS_MANAGER* mgr = GetSettingsManager();
 308 
 309         mgr->SaveProject( mgr->Prj().GetProjectFullName() );
 310         mgr->UnloadProject( &mgr->Prj() );
 311 
 312         // Don't set name until the user hits save, so project files are not created

 313         mgr->LoadProject( "" );
 314 
 315         LoadProjectSettings();
 316 
 317         if( !Clear_Pcb( false ) )
 318             return false;
 319 
 320         onBoardLoaded();
 321 
 322         OnModify();
 323         return true;
 324     }
 325 
 326     case ID_SAVE_BOARD:
 327         if( !GetBoard()->GetFileName().IsEmpty() )
 328             return SavePcbFile( Prj().AbsolutePath( GetBoard()->GetFileName() ) );
 329 
 330         KI_FALLTHROUGH;
 331 
 332     case ID_COPY_BOARD_AS:
 333     case ID_SAVE_BOARD_AS:
 334         {
 335             bool addToHistory = false;
 336             wxString orig_name;
 337             wxFileName::SplitPath( GetBoard()->GetFileName(),
 338                     nullptr, nullptr, &orig_name, nullptr );
 339 
 340             if( orig_name.IsEmpty() )
 341             {
 342                 addToHistory = true;
 343                 orig_name = _( "noname" );
 344             }
 345 
 346             wxString    pro_dir = wxPathOnly( Prj().GetProjectFullName() );
 347             wxFileName  fn( pro_dir, orig_name, KiCadPcbFileExtension );
 348             wxString    filename = fn.GetFullPath();
 349 
 350             if( AskSaveBoardFileName( this, &filename ) )
 351             {
 352                 if( id == ID_COPY_BOARD_AS )
 353                     return SavePcbCopy( filename );
 354                 else
 355                     return SavePcbFile( filename, addToHistory, false );
 356             }
 357             return false;
 358         }
 359 
 360     default:
 361         wxLogDebug( wxT( "File_io Internal Error" ) );
 362         return false;
 363     }
 364 }

 367 // The KIWAY_PLAYER::OpenProjectFiles() API knows nothing about plugins, so

 368 // determine how to load the BOARD here, with minor assistance from KICTL_EAGLE_BRD

 369 // bit flag.

 370 IO_MGR::PCB_FILE_T plugin_type( const wxString& aFileName, int aCtl )
 371 {
 372     IO_MGR::PCB_FILE_T  pluginType;
 373 
 374     wxFileName fn = aFileName;
 375 
 376     // Note: file extensions are expected to be in lower case.

 377     // This is not always true, especially when importing files, so the string

 378     // comparisons are case insensitive to try to find the suitable plugin.

 379 
 380     if( fn.GetExt().CmpNoCase( IO_MGR::GetFileExtension( IO_MGR::LEGACY ) ) == 0 )
 381     {
 382         // both legacy and eagle share a common file extension.

 383         pluginType = ( aCtl & KICTL_EAGLE_BRD ) ? IO_MGR::EAGLE : IO_MGR::LEGACY;
 384     }
 385     else if( fn.GetExt().CmpNoCase( IO_MGR::GetFileExtension( IO_MGR::PCAD ) ) == 0 )
 386     {
 387         pluginType = IO_MGR::PCAD;
 388     }
 389     else if( fn.GetExt().CmpNoCase( IO_MGR::GetFileExtension( IO_MGR::ALTIUM_DESIGNER ) ) == 0 )
 390     {
 391         pluginType = IO_MGR::ALTIUM_DESIGNER;
 392     }
 393     else if( fn.GetExt().CmpNoCase( IO_MGR::GetFileExtension( IO_MGR::ALTIUM_CIRCUIT_STUDIO ) ) == 0 )
 394     {
 395         pluginType = IO_MGR::ALTIUM_CIRCUIT_STUDIO;
 396     }
 397     else if( fn.GetExt().CmpNoCase( IO_MGR::GetFileExtension( IO_MGR::ALTIUM_CIRCUIT_MAKER ) ) == 0 )
 398     {
 399         pluginType = IO_MGR::ALTIUM_CIRCUIT_MAKER;
 400     }
 401     else
 402     {
 403         pluginType = IO_MGR::KICAD_SEXP;
 404     }
 405 
 406     return pluginType;
 407 }

 410 int PCB_EDIT_FRAME::inferLegacyEdgeClearance( BOARD* aBoard )
 411 {
 412     PCB_LAYER_COLLECTOR collector;
 413 
 414     collector.SetLayerId( Edge_Cuts );
 415     collector.Collect( aBoard, GENERAL_COLLECTOR::AllBoardItems );
 416 
 417     int  edgeWidth = -1;
 418     bool mixed = false;
 419 
 420     for( int i = 0; i < collector.GetCount(); i++ )
 421     {
 422         if( collector[i]->Type() == PCB_LINE_T )
 423         {
 424             int itemWidth = static_cast<DRAWSEGMENT*>( collector[i] )->GetWidth();
 425 
 426             if( edgeWidth != -1 && edgeWidth != itemWidth )
 427             {
 428                 mixed = true;
 429                 edgeWidth = std::max( edgeWidth, itemWidth );
 430             }
 431             else
 432             {
 433                 edgeWidth = itemWidth;
 434             }
 435         }
 436     }
 437 
 438     if( mixed )
 439     {
 440         // If they had different widths then we can't ensure that fills will be the same.

 441         wxMessageBox( _( "If the zones on this board are refilled the Copper Edge Clearance\n"
 442                          "setting will be used (see Board Setup > Design Rules).  This may\n"
 443                          "result in different fills from previous Kicad versions which used\n"
 444                          "the line thickness of the board boundary on the Edge Cuts layer." ),
 445                       _( "Edge Clearance Warning" ), wxOK|wxICON_WARNING, this );
 446     }
 447 
 448     return std::max( 0, edgeWidth / 2 );
 449 }

 452 bool PCB_EDIT_FRAME::OpenProjectFiles( const std::vector<wxString>& aFileSet, int aCtl )
 453 {
 454     // This is for python:

 455     if( aFileSet.size() != 1 )
 456     {
 457         UTF8 msg = StrPrintf( "Pcbnew:%s() takes only a single filename", __func__ );
 458         DisplayError( this, msg );
 459         return false;
 460     }
 461 
 462     wxString fullFileName( aFileSet[0] );
 463 
 464     // We insist on caller sending us an absolute path, if it does not, we say it's a bug.

 465     wxASSERT_MSG( wxFileName( fullFileName ).IsAbsolute(), wxT( "Path is not absolute!" ) );
 466 
 467     std::unique_ptr<wxSingleInstanceChecker> lockFile = ::LockFile( fullFileName );
 468 
 469     if( !lockFile )
 470     {
 471         wxString msg = wxString::Format( _( "PCB file \"%s\" is already open." ), fullFileName );
 472         DisplayError( this, msg );
 473         return false;
 474     }
 475 
 476     if( IsContentModified() )
 477     {
 478         if( !HandleUnsavedChanges( this, _( "The current PCB has been modified.  Save changes?" ),
 479             [&]()->bool { return SavePcbFile( GetBoard()->GetFileName() ); } ) )
 480         {
 481             return false;
 482         }
 483     }
 484 
 485     // Release the lock file, until the new file is actually loaded

 486     ReleaseFile();
 487 
 488     wxFileName pro = fullFileName;
 489     pro.SetExt( ProjectFileExtension );
 490 
 491     bool is_new = !wxFileName::IsFileReadable( fullFileName );
 492 
 493     // If its a non-existent schematic and caller thinks it exists

 494     if( is_new && !( aCtl & KICTL_CREATE ) )
 495     {
 496         // notify user that fullFileName does not exist, ask if user wants to create it.

 497         wxString ask = wxString::Format( _( "PCB \"%s\" does not exist.  Do you wish to create it?" ),
 498                                          fullFileName );
 499         if( !IsOK( this, ask ) )
 500             return false;
 501     }
 502 
 503     wxWindowUpdateLocker no_update( m_Layers );     // Avoid flicker when rebuilding m_Layers

 504 
 505     // Unlink the old project if needed

 506     GetBoard()->ClearProject();
 507 
 508     // No save prompt (we already prompted above), and only reset to a new blank board if new

 509     Clear_Pcb( false, !is_new );
 510 
 511     IO_MGR::PCB_FILE_T  pluginType = plugin_type( fullFileName, aCtl );
 512 
 513     bool converted =  pluginType != IO_MGR::LEGACY && pluginType != IO_MGR::KICAD_SEXP;
 514 
 515     if( !converted )
 516     {
 517         // Loading a project should only be done under carefully considered circumstances.

 518 
 519         // The calling code should know not to ask me here to change projects unless

 520         // it knows what consequences that will have on other KIFACEs running and using

 521         // this same PROJECT.  It can be very harmful if that calling code is stupid.

 522         SETTINGS_MANAGER* mgr = GetSettingsManager();
 523 
 524         if( pro.GetFullPath() != mgr->Prj().GetProjectFullName() )
 525         {
 526             SaveProjectSettings();
 527 
 528             mgr->SaveProject( mgr->Prj().GetProjectFullName() );
 529             mgr->UnloadProject( &mgr->Prj() );
 530 
 531             mgr->LoadProject( pro.GetFullPath() );
 532         }
 533     }
 534 
 535     if( is_new )
 536     {
 537         OnModify();
 538     }
 539     else
 540     {
 541         BOARD* loadedBoard = 0;   // it will be set to non-NULL if loaded OK

 542 
 543         PLUGIN::RELEASER pi( IO_MGR::PluginFind( pluginType ) );
 544 
 545         // This will rename the file if there is an autosave and the user want to recover

 546     CheckForAutoSaveFile( fullFileName );
 547 
 548         try
 549         {
 550             PROPERTIES  props;
 551             char        xbuf[30];
 552             char        ybuf[30];
 553 
 554             // EAGLE_PLUGIN can use this info to center the BOARD, but it does not yet.

 555             sprintf( xbuf, "%d", GetPageSizeIU().x );
 556             sprintf( ybuf, "%d", GetPageSizeIU().y );
 557 
 558             props["page_width"]  = xbuf;
 559             props["page_height"] = ybuf;
 560 
 561 #if USE_INSTRUMENTATION
 562             // measure the time to load a BOARD.

 563             unsigned startTime = GetRunningMicroSecs();
 564 #endif
 565 
 566             loadedBoard = pi->Load( fullFileName, NULL, &props );
 567 
 568 #if USE_INSTRUMENTATION
 569             unsigned stopTime = GetRunningMicroSecs();
 570             printf( "PLUGIN::Load(): %u usecs\n", stopTime - startTime );
 571 #endif
 572         }
 573         catch( const IO_ERROR& ioe )
 574         {
 575             if( ioe.Problem() != wxT( "CANCEL" ) )
 576             {
 577                 wxString msg = wxString::Format( _( "Error loading board file:\n%s" ), fullFileName );
 578                 DisplayErrorMessage( this, msg, ioe.What() );
 579             }
 580 
 581             // We didn't create a new blank board above, so do that now

 582             Clear_Pcb( false );
 583 
 584             return false;
 585         }
 586 
 587         SetBoard( loadedBoard );
 588 
 589         // On save; design settings will be removed from the board

 590         if( loadedBoard->m_LegacyDesignSettingsLoaded )
 591             loadedBoard->SetModified();
 592 
 593         // Move legacy view settings to local project settings

 594         if( !loadedBoard->m_LegacyVisibleLayers.test( Rescue ) )
 595         {
 596             Prj().GetLocalSettings().m_VisibleLayers = loadedBoard->m_LegacyVisibleLayers;
 597             loadedBoard->SetModified();
 598         }
 599 
 600         if( !loadedBoard->m_LegacyVisibleItems.test( GAL_LAYER_INDEX( GAL_LAYER_ID_BITMASK_END ) ) )
 601         {
 602             Prj().GetLocalSettings().m_VisibleItems = loadedBoard->m_LegacyVisibleItems;
 603             loadedBoard->SetModified();
 604         }
 605 
 606         // we should not ask PLUGINs to do these items:

 607         loadedBoard->BuildListOfNets();
 608         ResolveDRCExclusions();
 609 
 610         if( loadedBoard->IsModified() )
 611             OnModify();
 612         else
 613             GetScreen()->ClrModify();
 614 
 615         if( pluginType == IO_MGR::LEGACY &&
 616             loadedBoard->GetFileFormatVersionAtLoad() < LEGACY_BOARD_FILE_VERSION )
 617         {
 618             DisplayInfoMessage( this,
 619                 _(  "This file was created by an older version of Pcbnew.\n"
 620                     "It will be stored in the new file format when you save this file again." ) );
 621         }
 622     }
 623 
 624     {
 625         wxFileName fn = fullFileName;
 626 
 627         if( converted )
 628             fn.SetExt( PcbFileExtension );
 629 
 630         wxString fname = fn.GetFullPath();
 631 
 632         fname.Replace( WIN_STRING_DIR_SEP, UNIX_STRING_DIR_SEP );
 633 
 634         GetBoard()->SetFileName( fname );
 635     }
 636 
 637     // Lock the file newly opened:

 638     m_file_checker.reset( lockFile.release() );
 639 
 640     if( !converted )
 641         UpdateFileHistory( GetBoard()->GetFileName() );
 642 
 643     // Select netclass Default as current netclass (it always exists)

 644     SetCurrentNetClass( NETCLASS::Default );
 645 
 646     // Rebuild list of nets (full ratsnest rebuild)

 647     GetBoard()->BuildConnectivity();
 648     Compile_Ratsnest( true );
 649 
 650     // Load project settings after setting up board; some of them depend on the nets list

 651     LoadProjectSettings();
 652 
 653     onBoardLoaded();
 654 
 655     // Refresh the 3D view, if any

 656     EDA_3D_VIEWER* draw3DFrame = Get3DViewerFrame();
 657 
 658     if( draw3DFrame )
 659         draw3DFrame->NewDisplay();
 660 
 661 #if 0 && defined(DEBUG)
 662     // Output the board object tree to stdout, but please run from command prompt:

 663     GetBoard()->Show( 0, std::cout );
 664 #endif
 665 
 666     // from EDA_APPL which was first loaded BOARD only:

 667     {
 668         /* For an obscure reason the focus is lost after loading a board file

 669          * when starting up the process.

 670          * (seems due to the recreation of the layer manager after loading the file)

 676         SetFocus();

 684 bool PCB_EDIT_FRAME::SavePcbFile( const wxString& aFileName, bool addToHistory,
 685                                   bool aChangeProject )
 686 {
 687     // please, keep it simple.  prompting goes elsewhere.

 688 
 689     wxFileName pcbFileName = aFileName;
 690 
 691     if( pcbFileName.GetExt() == LegacyPcbFileExtension )
 692         pcbFileName.SetExt( KiCadPcbFileExtension );
 693 
 694     if( !IsWritable( pcbFileName ) )
 695     {
 696         wxString msg = wxString::Format( _(
 697             "No access rights to write to file \"%s\"" ),
 698             pcbFileName.GetFullPath() );
 699 
 700         DisplayError( this, msg );
 701         return false;
 702     }
 703 
 704     // TODO: this will break if we ever go multi-board

 705     wxFileName projectFile( pcbFileName );
 706     projectFile.SetExt( ProjectFileExtension );
 707 
 708     if( aChangeProject && !projectFile.FileExists() )
 709     {
 710         // If this is a new board, project filename won't be set yet

 711         if( projectFile.GetFullPath() != Prj().GetProjectFullName() )
 712         {
 713             GetBoard()->ClearProject();
 714 
 715             SETTINGS_MANAGER* mgr = GetSettingsManager();
 716 
 717             mgr->SaveProject( Prj().GetProjectFullName() );
 718             mgr->UnloadProject( &Prj() );
 719 
 720             mgr->LoadProject( projectFile.GetFullPath() );
 721             GetBoard()->SetProject( &Prj() );
 722         }
 723     }
 724 
 725     wxFileName tempFile( aFileName );
 726     tempFile.SetName( wxT( "." ) + tempFile.GetName() );
 727     tempFile.SetExt( tempFile.GetExt() + wxT( "$" ) );
 728 
 729     GetBoard()->SynchronizeNetsAndNetClasses();
 730 
 731     // Select default Netclass before writing file. Useful to save default values in headers.

 732     SetCurrentNetClass( NETCLASS::Default );
 733 
 734     // Save various DRC parameters, such as violation severities (which may have been

 735     // edited via the DRC dialog as well as the Board Setup dialog), DRC exclusions, etc.

 736     SaveProjectSettings();
 737 
 738     GetSettingsManager()->SaveProject();
 739 
 740     ClearMsgPanel();
 741 
 742     wxString    upperTxt;
 743     wxString    lowerTxt;
 744 
 745     try
 746     {
 747         PLUGIN::RELEASER    pi( IO_MGR::PluginFind( IO_MGR::KICAD_SEXP ) );
 748 
 749         wxASSERT( tempFile.IsAbsolute() );
 750 
 751         pi->Save( tempFile.GetFullPath(), GetBoard(), NULL );
 752     }
 753     catch( const IO_ERROR& ioe )
 754     {
 755         wxString msg = wxString::Format( _(
 756                 "Error saving board file \"%s\".\n%s" ),
 757                 pcbFileName.GetFullPath(), ioe.What()
 758                 );
 759         DisplayError( this, msg );
 760 
 761         lowerTxt.Printf( _( "Failed to create temporary file \"%s\"" ), tempFile.GetFullPath() );
 762 
 763         AppendMsgPanel( upperTxt, lowerTxt, CYAN );
 764 
 765         // In case we started a file but didn't fully write it, clean up

 766         wxRemoveFile( tempFile.GetFullPath() );
 767 
 768         return false;
 769     }
 770 
 771     // If save succeeded, replace the original with what we just wrote

 772     if( !wxRenameFile( tempFile.GetFullPath(), pcbFileName.GetFullPath() ) )
 773     {
 774         wxString msg = wxString::Format( _(
 775                 "Error saving board file \"%s\".\nFailed to rename temporary file \"%s\"" ),
 776                 pcbFileName.GetFullPath(), tempFile.GetFullPath()
 777                 );
 778         DisplayError( this, msg );
 779 
 780         lowerTxt.Printf( _( "Failed to rename temporary file \"%s\"" ), tempFile.GetFullPath() );
 781 
 782         AppendMsgPanel( upperTxt, lowerTxt, CYAN );
 783 
 784         return false;
 785     }
 786 
 787     if( !Kiface().IsSingle() )
 788     {
 789         WX_STRING_REPORTER backupReporter( &upperTxt );
 790 
 791         if( GetSettingsManager()->TriggerBackupIfNeeded( backupReporter ) )
 792             upperTxt.clear();
 793     }
 794 
 795     GetBoard()->SetFileName( pcbFileName.GetFullPath() );
 796     UpdateTitle();
 797 
 798     // Put the saved file in File History if requested

 799     if( addToHistory )
 800         UpdateFileHistory( GetBoard()->GetFileName() );
 801 
 802     // Delete auto save file on successful save.

 803     wxFileName autoSaveFileName = pcbFileName;
 804 
 805     autoSaveFileName.SetName( GetAutoSaveFilePrefix() + pcbFileName.GetName() );
 806 
 807     if( autoSaveFileName.FileExists() )
 808         wxRemoveFile( autoSaveFileName.GetFullPath() );
 809 
 810     lowerTxt.Printf( _( "Wrote board file: \"%s\"" ), pcbFileName.GetFullPath() );
 811 
 812     AppendMsgPanel( upperTxt, lowerTxt, CYAN );
 813 
 814     GetScreen()->ClrModify();
 815     GetScreen()->ClrSave();
 816     return true;
 817 }

 869 bool PCB_EDIT_FRAME::doAutoSave()
 870 {
 871     wxFileName tmpFileName;
 872 
 873     if( GetBoard()->GetFileName().IsEmpty() )
 874     {
 875         tmpFileName = wxFileName( wxStandardPaths::Get().GetDocumentsDir(), wxT( "noname" ),
 876                                   KiCadPcbFileExtension );
 877         GetBoard()->SetFileName( tmpFileName.GetFullPath() );
 878     }
 879     else
 880     {
 881         tmpFileName = Prj().AbsolutePath( GetBoard()->GetFileName() );
 882     }
 883 
 884     wxFileName autoSaveFileName = tmpFileName;
 885 
 886     // Auto save file name is the board file name prepended with autosaveFilePrefix string.

 887     autoSaveFileName.SetName( GetAutoSaveFilePrefix() + autoSaveFileName.GetName() );
 888 
 889     if( !autoSaveFileName.IsOk() )
 890         return false;
 891 
 892     // If the board file path is not writable, try writing to a platform specific temp file

 893     // path.  If that path isn't writabe, give up.

 894     if( !autoSaveFileName.IsDirWritable() )
 895     {
 896         autoSaveFileName.SetPath( wxFileName::GetTempDir() );
 897 
 898         if( !autoSaveFileName.IsOk() || !autoSaveFileName.IsDirWritable() )
 899             return false;
 900     }
 901 
 902     wxLogTrace( traceAutoSave, "Creating auto save file <" + autoSaveFileName.GetFullPath() + ">" );
 903 
 904     if( SavePcbFile( autoSaveFileName.GetFullPath(), false, false ) )
 905     {
 906         GetScreen()->SetModify();
 907         GetBoard()->SetFileName( tmpFileName.GetFullPath() );
 908         UpdateTitle();
 909         m_autoSaveState = false;
 910 
 911         if( !Kiface().IsSingle() &&
 912             GetSettingsManager()->GetCommonSettings()->m_Backup.backup_on_autosave )
 913         {
 914             GetSettingsManager()->TriggerBackupIfNeeded( NULL_REPORTER::GetInstance() );
 915         }
 916 
 917         return true;
 918     }
 919 
 920     GetBoard()->SetFileName( tmpFileName.GetFullPath() );
 921 
 922     return false;
 923 }

 926 bool PCB_EDIT_FRAME::importFile( const wxString& aFileName, int aFileType )
 927 {
 928     switch( (IO_MGR::PCB_FILE_T) aFileType )
 929     {
 930     case IO_MGR::EAGLE:
 931         if( OpenProjectFiles( std::vector<wxString>( 1, aFileName ), KICTL_EAGLE_BRD ) )
 932         {
 933             wxString projectpath = Kiway().Prj().GetProjectPath();
 934             wxFileName newfilename;
 935 
 936             newfilename.SetPath( Prj().GetProjectPath() );
 937             newfilename.SetName( Prj().GetProjectName() );
 938             newfilename.SetExt( KiCadPcbFileExtension );
 939 
 940             GetBoard()->SetFileName( newfilename.GetFullPath() );
 941             UpdateTitle();
 942             OnModify();
 943 
 944             // Extract a footprint library from the design and add it to the fp-lib-table

 945             wxString newLibPath;
 946             ArchiveModulesOnBoard( true, newfilename.GetName(), &newLibPath );
 947 
 948             if( newLibPath.Length() > 0 )
 949             {
 950                 FP_LIB_TABLE* prjlibtable = Prj().PcbFootprintLibs();
 951                 const wxString& project_env = PROJECT_VAR_NAME;
 952                 wxString rel_path, env_path;
 953 
 954                 wxGetEnv( project_env, &env_path );
 955 
 956                 wxString result( newLibPath );
 957                 rel_path =  result.Replace( env_path,
 958                                             wxString( "$(" + project_env + ")" ) ) ? result : "" ;
 959 
 960                 if( !rel_path.IsEmpty() )
 961                     newLibPath = rel_path;
 962 
 963                 FP_LIB_TABLE_ROW* row = new FP_LIB_TABLE_ROW( newfilename.GetName(),
 964                         newLibPath, wxT( "KiCad" ), wxEmptyString );
 965                 prjlibtable->InsertRow( row );
 966             }
 967 
 968             if( !GetBoard()->GetFileName().IsEmpty() )
 969             {
 970                 wxString tblName = Prj().FootprintLibTblName();
 971 
 972                 try
 973                 {
 974                     Prj().PcbFootprintLibs()->Save( tblName );
 975                 }
 976                 catch( const IO_ERROR& ioe )
 977                 {
 978                     wxString msg = wxString::Format( _(
 979                                     "Error occurred saving project specific footprint library "
 980                                     "table:\n\n%s" ), ioe.What() );
 981                     wxMessageBox( msg, _( "File Save Error" ), wxOK | wxICON_ERROR );
 982                 }
 983             }
 984 
 985 
 986             // Update module LIB_IDs to point to the just imported Eagle library

 987             for( MODULE* module : GetBoard()->Modules() )
 988             {
 989                 LIB_ID libId = module->GetFPID();
 990 
 991                 if( libId.GetLibItemName().empty() )
 992                     continue;
 993 
 994                 libId.SetLibNickname( newfilename.GetName() );
 995                 module->SetFPID( libId );
 996             }
 997 
 998 
 999             // Store net names for all pads, to create net remap information

1000             std::unordered_map<D_PAD*, wxString> netMap;
1001 
1002             for( const auto& pad : GetBoard()->GetPads() )
1003             {
1004                 NETINFO_ITEM* netinfo = pad->GetNet();
1005 
1006                 if( netinfo->GetNet() > 0 && !netinfo->GetNetname().IsEmpty() )
1007                     netMap[pad] = netinfo->GetNetname();
1008             }
1009 
1010             // Two stage netlist update:

1011             // - first, assign valid timestamps to footprints (no reannotation)

1012             // - second, perform schematic annotation and update footprint references

1013             //   based on timestamps

1014             NETLIST netlist;
1015             FetchNetlistFromSchematic( netlist, NO_ANNOTATION );
1016             DoUpdatePCBFromNetlist( netlist, false );
1017             FetchNetlistFromSchematic( netlist, QUIET_ANNOTATION );
1018             DoUpdatePCBFromNetlist( netlist, true );
1019 
1020             std::unordered_map<wxString, wxString> netRemap;
1021 
1022             // Compare the old net names with the new net names and create a net map

1023             for( const auto& pad : GetBoard()->GetPads() )
1024             {
1025                 auto it = netMap.find( pad );
1026 
1027                 if( it == netMap.end() )
1028                     continue;
1029 
1030                 NETINFO_ITEM* netinfo = pad->GetNet();
1031 
1032                 // Net name has changed, create a remap entry

1033                 if( netinfo->GetNet() > 0 && netMap[pad] != netinfo->GetNetname() )
1034                     netRemap[netMap[pad]] = netinfo->GetNetname();
1035             }
1036 
1037             if( !netRemap.empty() )
1038                 fixEagleNets( netRemap );
1039 
1040             return true;
1041         }
1042 
1043         return false;
1044 
1045     default:
1046         return false;
1047     }
1048 
1049     return false;
1050 }

1053 bool PCB_EDIT_FRAME::fixEagleNets( const std::unordered_map<wxString, wxString>& aRemap )
1054 {
1055     bool result = true;
1056     BOARD* board = GetBoard();
1057 
1058     // perform netlist matching to prevent orphaned zones.

1059     for( auto zone : board->Zones() )
1060     {
1061         auto it = aRemap.find( zone->GetNet()->GetNetname() );
1062 
1063         if( it != aRemap.end() )
1064         {
1065             NETINFO_ITEM* net = board->FindNet( it->second );
1066 
1067             if( !net )
1068             {
1069                 wxFAIL;
1070                 result = false;
1071                 continue;
1072             }
1073 
1074             zone->SetNet( net );
1075         }
1076     }
1077 
1078 
1079     // perform netlist matching to prevent orphaned tracks/vias.

1080     for( auto track : board->Tracks() )
1081     {
1082         auto it = aRemap.find( track->GetNet()->GetNetname() );
1083 
1084         if( it != aRemap.end() )
1085         {
1086             NETINFO_ITEM* net = board->FindNet( it->second );
1087 
1088             if( !net )
1089             {
1090                 wxFAIL;
1091                 result = false;
1092                 continue;
1093             }
1094 
1095             track->SetNet( net );
1096         }
1097     }
1098 
1099     return result;
