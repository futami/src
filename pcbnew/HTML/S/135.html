 182 {

 184     wxString footprintName = fpID.GetLibItemName();
 185 
 186     m_FootprintNameCtrl->ChangeValue( footprintName );
 187 
 188     m_DocCtrl->SetValue( m_footprint->GetDescription() );
 189     m_KeywordCtrl->SetValue( m_footprint->GetKeywords() );
 190

 192         return false;
 193 
 194     if( !m_PanelGeneral->TransferDataToWindow() )
 195         return false;
 196 
 197     if( !m_Panel3D->TransferDataToWindow() )
 198         return false;
 199 
 200     // Module Texts

 201 
 202     m_texts->push_back( m_footprint->Reference() );
 203     m_texts->push_back( m_footprint->Value() );
 204 
 205     for( auto item : m_footprint->GraphicalItems() )
 206     {
 207         auto textModule = dyn_cast<TEXTE_MODULE*>( item );
 208 
 209         if( textModule )
 210             m_texts->push_back( *textModule );
 211     }
 212 
 213     // notify the grid

 214     wxGridTableMessage tmsg( m_texts, wxGRIDTABLE_NOTIFY_ROWS_APPENDED, m_texts->GetNumberRows() );
 215     m_itemsGrid->ProcessTableMessage( tmsg );
 216 
 217     // Module Properties

 218 
 219     m_AutoPlaceCtrl->SetSelection( (m_footprint->IsLocked()) ? 1 : 0 );
 220     m_AutoPlaceCtrl->SetItemToolTip( 0, _( "Enable hotkey move commands and Auto Placement" ) );
 221     m_AutoPlaceCtrl->SetItemToolTip( 1, _( "Disable hotkey move commands and Auto Placement" ) );
 222 
 223     m_CostRot90Ctrl->SetValue( m_footprint->GetPlacementCost90() );
 224     m_CostRot180Ctrl->SetValue( m_footprint->GetPlacementCost180() );
 225 
 226     m_AttributsCtrl->SetItemToolTip( 0, _( "Use this attribute for most non SMD footprints\n"
 227             "Footprints with this option are not put in the footprint position list file" ) );
 228     m_AttributsCtrl->SetItemToolTip( 1, _( "Use this attribute for SMD footprints.\n"
 229             "Only footprints with this option are put in the footprint position list file" ) );
 230     m_AttributsCtrl->SetItemToolTip( 2, _( "Use this attribute for \"virtual\" footprints drawn on board\n"
 231             "such as an edge connector (old ISA PC bus for instance)" ) );
 232 
 233     switch( m_footprint->GetAttributes() & 255 )
 234     {
 235     case MOD_CMS:     m_AttributsCtrl->SetSelection( 1 ); break;
 236     case MOD_VIRTUAL: m_AttributsCtrl->SetSelection( 2 ); break;
 237     case 0:
 238     default:          m_AttributsCtrl->SetSelection( 0 ); break;
 239     }
 240 
 241     // Local Clearances

 242 
 243     m_netClearance.SetValue( m_footprint->GetLocalClearance() );
 244     m_solderMask.SetValue( m_footprint->GetLocalSolderMaskMargin() );
 245     m_solderPaste.SetValue( m_footprint->GetLocalSolderPasteMargin() );
 246 
 247     // Prefer "-0" to "0" for normally negative values

 248     if( m_footprint->GetLocalSolderPasteMargin() == 0 )
 249         m_SolderPasteMarginCtrl->SetValue( wxT( "-" ) + m_SolderPasteMarginCtrl->GetValue() );
 250 
 251     // Add solder paste margin ratio in percent

 252     // for the usual default value 0.0, display -0.0 (or -0,0 in some countries)

 253     wxString msg;
 254     msg.Printf( wxT( "%f" ), m_footprint->GetLocalSolderPasteMarginRatio() * 100.0 );
 255 
 256     if( m_footprint->GetLocalSolderPasteMarginRatio() == 0.0 &&
 257         msg[0] == '0')  // Sometimes Printf adds a sign if the value is very small (0.0)

 258         m_SolderPasteMarginRatioCtrl->SetValue( wxT("-") + msg );
 259     else
 260         m_SolderPasteMarginRatioCtrl->SetValue( msg );
 261 
 262     switch( m_footprint->GetZoneConnection() )
 263     {
 264     default:
 265     case ZONE_CONNECTION::INHERITED:
 266         m_ZoneConnectionChoice->SetSelection( 0 );
 267         break;
 268     case ZONE_CONNECTION::FULL:
 269         m_ZoneConnectionChoice->SetSelection( 1 );
 270         break;
 271     case ZONE_CONNECTION::THERMAL:
 272         m_ZoneConnectionChoice->SetSelection( 2 );
 273         break;
 274     case ZONE_CONNECTION::NONE:
 275         m_ZoneConnectionChoice->SetSelection( 3 );
 276         break;
 277     }
 278 
 279     // 3D Settings

 280 
 281     wxString default_path;
 282     wxGetEnv( KISYS3DMOD, &default_path );
 283 #ifdef __WINDOWS__
 284     default_path.Replace( wxT( "/" ), wxT( "\\" ) );
 285 #endif
 286 
 287     m_shapes3D_list.clear();
 288     m_modelsGrid->DeleteRows( 0, m_modelsGrid->GetNumberRows() );
 289 
 290     wxString origPath, alias, shortPath;
 291     FILENAME_RESOLVER* res = Prj().Get3DCacheManager()->GetResolver();
 292

 295         m_shapes3D_list.push_back( model );

 302         int row = m_modelsGrid->GetNumberRows() - 1;

 311         // Adjust min size to the column label size

 325 
 326     Layout();
 327     adjustGridColumns( m_itemsGrid->GetRect().GetWidth());
 328 
 329     return true;
 330 }

 359 void DIALOG_FOOTPRINT_FP_EDITOR::On3DModelCellChanged( wxGridEvent& aEvent )
 360 {
 361     if( aEvent.GetCol() == 0 )
 362     {
 363         bool               hasAlias = false;
 364         FILENAME_RESOLVER* res = Prj().Get3DCacheManager()->GetResolver();
 365         wxString           filename = m_modelsGrid->GetCellValue( aEvent.GetRow(), 0 );
 366 
 367         filename.Replace( "\n", "" );
 368         filename.Replace( "\r", "" );
 369         filename.Replace( "\t", "" );
 370 
 371         if( filename.empty() || !res->ValidateFileName( filename, hasAlias ) )
 372         {
 373             m_delayedErrorMessage = wxString::Format( _( "Invalid filename: %s" ), filename );
 374             m_delayedFocusGrid = m_modelsGrid;
 375             m_delayedFocusRow = aEvent.GetRow();
 376             m_delayedFocusColumn = aEvent.GetCol();
 377             m_delayedFocusPage = 2;
 378             aEvent.Veto();
 379         }
 380 
 381         // if the user has specified an alias in the name then prepend ':'

 382         if( hasAlias )
 383             filename.insert( 0, wxT( ":" ) );
 384 
 385 #ifdef __WINDOWS__
 386         // In Kicad files, filenames and paths are stored using Unix notation

 387         filename.Replace( wxT( "\\" ), wxT( "/" ) );
 388 #endif
 389 
 390         m_shapes3D_list[ aEvent.GetRow() ].m_Filename = filename;
 391         m_modelsGrid->SetCellValue( aEvent.GetRow(), 0, filename );
 392     }
 393     else if( aEvent.GetCol() == 1 )
 394     {
 395         wxString showValue = m_modelsGrid->GetCellValue( aEvent.GetRow(), 1 );
 396 
 397         m_shapes3D_list[ aEvent.GetRow() ].m_Show = ( showValue == wxT( "1" ) );
 398     }
 399 
 400     m_PreviewPane->UpdateDummyModule();
 401 }

 422 void DIALOG_FOOTPRINT_FP_EDITOR::OnAdd3DModel( wxCommandEvent&  )
 423 {
 424     if( !m_modelsGrid->CommitPendingChanges() )
 425         return;
 426 
 427     int selected = m_modelsGrid->GetGridCursorRow();
 428 
 429     PROJECT& prj = Prj();
 430     MODULE_3D_SETTINGS model;
 431 
 432     wxString initialpath = prj.GetRString( PROJECT::VIEWER_3D_PATH );
 433     wxString sidx = prj.GetRString( PROJECT::VIEWER_3D_FILTER_INDEX );
 434     int      filter = 0;
 435 
 436     // If the PROJECT::VIEWER_3D_PATH hasn't been set yet, use the KISYS3DMOD environment

 437     // varaible and fall back to the project path if necessary.

 438     if( initialpath.IsEmpty() )
 439     {
 440         if( !wxGetEnv( "KISYS3DMOD", &initialpath ) || initialpath.IsEmpty() )
 441             initialpath = prj.GetProjectPath();
 442     }
 443 
 444     if( !sidx.empty() )
 445     {
 446         long tmp;
 447         sidx.ToLong( &tmp );
 448 
 449         if( tmp > 0 && tmp <= INT_MAX )
 450             filter = (int) tmp;
 451     }
 452 
 453     if( !S3D::Select3DModel( this, Prj().Get3DCacheManager(), initialpath, filter, &model )
 454         || model.m_Filename.empty() )
 455     {
 456         select3DModel( selected );
 457         return;
 458     }
 459 
 460     prj.SetRString( PROJECT::VIEWER_3D_PATH, initialpath );
 461     sidx = wxString::Format( wxT( "%i" ), filter );
 462     prj.SetRString( PROJECT::VIEWER_3D_FILTER_INDEX, sidx );
 463     FILENAME_RESOLVER* res = Prj().Get3DCacheManager()->GetResolver();
 464     wxString alias;
 465     wxString shortPath;
 466     wxString filename = model.m_Filename;
 467 
 468     if( res && res->SplitAlias( filename, alias, shortPath ) )
 469         filename = alias + wxT( ":" ) + shortPath;
 470 
 471 #ifdef __WINDOWS__
 472     // In Kicad files, filenames and paths are stored using Unix notation

 473     model.m_Filename.Replace( "\\", "/" );
 474 #endif
 475 
 476     model.m_Show = true;
 477     m_shapes3D_list.push_back( model );
 478 
 479     int idx = m_modelsGrid->GetNumberRows();
 480     m_modelsGrid->AppendRows( 1 );
 481     m_modelsGrid->SetCellValue( idx, 0, filename );
 482     m_modelsGrid->SetCellValue( idx, 1, wxT( "1" ) );
 483 
 484     select3DModel( idx );
 485     m_PreviewPane->UpdateDummyModule();
 486 }

 532 bool DIALOG_FOOTPRINT_FP_EDITOR::Validate()
 533 {
 534     if( !m_itemsGrid->CommitPendingChanges() )
 535         return false;
 536 
 537     if( !DIALOG_SHIM::Validate() )
 538         return false;
 539 
 540     // First, test for invalid chars in module name

 541     wxString footprintName = m_FootprintNameCtrl->GetValue();
 542 
 543     if( !checkFootprintName( footprintName ) )
 544     {
 545         if( m_NoteBook->GetSelection() != 0 )
 546             m_NoteBook->SetSelection( 0 );
 547 
 548         m_delayedFocusCtrl = m_FootprintNameCtrl;
 549         m_delayedFocusPage = 0;
 550 
 551         return false;
 552     }
 553 
 554     // Check for empty texts.

 555     for( size_t i = 2; i < m_texts->size(); ++i )
 556     {
 557         TEXTE_MODULE& text = m_texts->at( i );
 558 
 559         if( text.GetText().IsEmpty() )
 560         {
 561             if( m_NoteBook->GetSelection() != 0 )
 562                 m_NoteBook->SetSelection( 0 );
 563 
 564             m_delayedErrorMessage = _( "Text items must have some content." );
 565             m_delayedFocusGrid = m_itemsGrid;
 566             m_delayedFocusColumn = TMC_TEXT;
 567             m_delayedFocusRow = i;
 568 
 569             return false;
 570         }
 571     }
 572 
 573     if( !m_netClearance.Validate( 0, INT_MAX ) )
 574         return false;
 575 
 576     return true;
 577 }

 584

 599     fpID.SetLibItemName( m_FootprintNameCtrl->GetValue(), false );
 600     m_footprint->SetFPID( fpID );
 601 
 602     m_footprint->SetDescription( m_DocCtrl->GetValue() );
 603     m_footprint->SetKeywords( m_KeywordCtrl->GetValue() );
 604 
 605     // copy reference and value

 606     m_footprint->Reference() = m_texts->at( 0 );
 607     m_footprint->Value() = m_texts->at( 1 );
 608 
 609     size_t i = 2;
 610     std::vector<TEXTE_MODULE*> items_to_remove;
 611 
 612     for( BOARD_ITEM* item : m_footprint->GraphicalItems() )
 613     {
 614         TEXTE_MODULE* textModule = dynamic_cast<TEXTE_MODULE*>( item );
 615 
 616         if( textModule )
 617         {
 618             // copy grid table entries till we run out, then delete any remaining texts

 619             if( i < m_texts->size() )
 620                 *textModule = m_texts->at( i++ );
 621             else    // store this item to remove and delete it later,

 622                     // after the graphic list is explored:

 623                 items_to_remove.push_back( textModule );
 624         }
 625     }
 626 
 627     // Remove items from m_footprint graphic list:

 628     for( TEXTE_MODULE* item: items_to_remove )
 629         item->DeleteStructure();
 630 
 631     // if there are still grid table entries, create new texts for them

 632     while( i < m_texts->size() )
 633     {
 634         auto newText = new TEXTE_MODULE( m_texts->at( i++ ) );
 635         m_footprint->Add( newText, ADD_MODE::APPEND );
 636         view->Add( newText );
 637     }
 638 
 639     m_footprint->SetLocked( m_AutoPlaceCtrl->GetSelection() == 1 );
 640 
 641     switch( m_AttributsCtrl->GetSelection() )
 642     {
 643     case 0:  m_footprint->SetAttributes( 0 );           break;
 644     case 1:  m_footprint->SetAttributes( MOD_CMS );     break;
 645     case 2:  m_footprint->SetAttributes( MOD_VIRTUAL ); break;
 646     default: wxFAIL;
 647     }
 648 
 649     m_footprint->SetPlacementCost90( m_CostRot90Ctrl->GetValue() );
 650     m_footprint->SetPlacementCost180( m_CostRot180Ctrl->GetValue() );
 651 
 652     // Initialize masks clearances

 653     m_footprint->SetLocalClearance( m_netClearance.GetValue() );
 654     m_footprint->SetLocalSolderMaskMargin( m_solderMask.GetValue() );
 655     m_footprint->SetLocalSolderPasteMargin( m_solderPaste.GetValue() );
 656 
 657     double dtmp = 0.0;
 658     wxString msg = m_SolderPasteMarginRatioCtrl->GetValue();
 659     msg.ToDouble( &dtmp );
 660 
 661     // A -50% margin ratio means no paste on a pad, the ratio must be >= -50%

 662     if( dtmp < -50.0 )
 663         dtmp = -50.0;
 664     // A margin ratio is always <= 0

 665     // 0 means use full pad copper area

 666     if( dtmp > 0.0 )
 667         dtmp = 0.0;
 668 
 669     m_footprint->SetLocalSolderPasteMarginRatio( dtmp / 100 );
 670 
 671     switch( m_ZoneConnectionChoice->GetSelection() )
 672     {
 673     default:
 674     case 0:
 675         m_footprint->SetZoneConnection( ZONE_CONNECTION::INHERITED );
 676         break;
 677     case 1:
 678         m_footprint->SetZoneConnection( ZONE_CONNECTION::FULL );
 679         break;
 680     case 2:
 681         m_footprint->SetZoneConnection( ZONE_CONNECTION::THERMAL );
 682         break;
 683     case 3:
 684         m_footprint->SetZoneConnection( ZONE_CONNECTION::NONE );
 685         break;
 686     }
 687 
 688     std::list<MODULE_3D_SETTINGS>* draw3D  = &m_footprint->Models();
 689     draw3D->clear();
 690     draw3D->insert( draw3D->end(), m_shapes3D_list.begin(), m_shapes3D_list.end() );
 691 
 692     m_footprint->CalculateBoundingBox();
 693 
 694     commit.Push( _( "Modify module properties" ) );
 695 
 696     return true;
 697 }

 721 void DIALOG_FOOTPRINT_FP_EDITOR::OnAddField( wxCommandEvent& event )
 722 {
 723     if( !m_itemsGrid->CommitPendingChanges() )
 724         return;
 725 
 726     const BOARD_DESIGN_SETTINGS& dsnSettings = m_frame->GetDesignSettings();
 727     TEXTE_MODULE textMod( m_footprint );
 728 
 729     // Set active layer if legal; otherwise copy layer from previous text item

 730     if( LSET::AllTechMask().test( m_frame->GetActiveLayer() ) )
 731         textMod.SetLayer( m_frame->GetActiveLayer() );
 732     else
 733         textMod.SetLayer( m_texts->at( m_texts->size() - 1 ).GetLayer() );
 734 
 735     textMod.SetTextSize( dsnSettings.GetTextSize( textMod.GetLayer() ) );
 736     textMod.SetTextThickness( dsnSettings.GetTextThickness( textMod.GetLayer() ) );
 737     textMod.SetItalic( dsnSettings.GetTextItalic( textMod.GetLayer() ) );
 738 
 739     m_texts->push_back( textMod );
 740 
 741     // notify the grid

 742     wxGridTableMessage msg( m_texts, wxGRIDTABLE_NOTIFY_ROWS_APPENDED, 1 );
 743     m_itemsGrid->ProcessTableMessage( msg );
 744 
 745     m_itemsGrid->SetFocus();
 746     m_itemsGrid->MakeCellVisible( m_texts->size() - 1, 0 );
 747     m_itemsGrid->SetGridCursor( m_texts->size() - 1, 0 );
 748 
 749     m_itemsGrid->EnableCellEditControl( true );
 750     m_itemsGrid->ShowCellEditControl();
 751 }

 754 void DIALOG_FOOTPRINT_FP_EDITOR::OnDeleteField( wxCommandEvent& event )
 755 {
 756     if( !m_itemsGrid->CommitPendingChanges() )
 757         return;
 758 
 759     int curRow = m_itemsGrid->GetGridCursorRow();
 760 
 761     if( curRow < 0 )
 762         return;
 763     else if( curRow < 2 )
 764     {
 765         DisplayError( nullptr, _( "Reference and value are mandatory." ) );
 766         return;
 767     }
 768 
 769     m_texts->erase( m_texts->begin() + curRow );
 770 
 771     // notify the grid

 772     wxGridTableMessage msg( m_texts, wxGRIDTABLE_NOTIFY_ROWS_DELETED, curRow, 1 );
 773     m_itemsGrid->ProcessTableMessage( msg );
 774 
 775     if( m_itemsGrid->GetNumberRows() > 0 )
 776     {
 777         m_itemsGrid->MakeCellVisible( std::max( 0, curRow-1 ), m_itemsGrid->GetGridCursorCol() );
 778         m_itemsGrid->SetGridCursor( std::max( 0, curRow-1 ), m_itemsGrid->GetGridCursorCol() );
 779     }
 780 }

 790 void DIALOG_FOOTPRINT_FP_EDITOR::adjustGridColumns( int aWidth )
 791 {
 792     // Account for scroll bars

 793     int itemsWidth = aWidth - ( m_itemsGrid->GetSize().x - m_itemsGrid->GetClientSize().x );
 794     int modelsWidth = aWidth - ( m_modelsGrid->GetSize().x - m_modelsGrid->GetClientSize().x );
 795 
 796     itemsWidth -= m_itemsGrid->GetRowLabelSize();
 797 
 798     for( int i = 1; i < m_itemsGrid->GetNumberCols(); i++ )
 799         itemsWidth -= m_itemsGrid->GetColSize( i );
 800 
 801     if( itemsWidth > 0 )
 802         m_itemsGrid->SetColSize( 0, std::max( itemsWidth,
 803                 m_itemsGrid->GetVisibleWidth( 0, true, false, false ) ) );
 804 
 805     m_modelsGrid->SetColSize( 0, modelsWidth - m_modelsGrid->GetColSize( 1 ) - 5 );
 806 }

 809 void DIALOG_FOOTPRINT_FP_EDITOR::OnUpdateUI( wxUpdateUIEvent& event )
 810 {
 811     if( !m_itemsGrid->IsCellEditControlShown() && !m_modelsGrid->IsCellEditControlShown() )
 812         adjustGridColumns( m_itemsGrid->GetRect().GetWidth());
 813 
 814     if( m_itemsGrid->IsCellEditControlShown() )
 815     {
 816         // Currently: nonthing to do

 817     }
 818 
 819     // Handle a delayed focus.  The delay allows us to:

 820     // a) change focus when the error was triggered from within a killFocus handler

 821     // b) show the correct notebook page in the background before the error dialog comes up

 822     //    when triggered from an OK or a notebook page change

 823 
 824     if( m_delayedFocusPage >= 0 )
 825     {
 826         if( m_NoteBook->GetSelection() != m_delayedFocusPage )
 827             m_NoteBook->SetSelection( (unsigned) m_delayedFocusPage );
 828 
 829         m_delayedFocusPage = -1;
 830     }
 831 
 832     if( !m_delayedErrorMessage.IsEmpty() )
 833     {
 834         // We will re-enter this routine when the error dialog is displayed, so make

 835         // sure we don't keep putting up more dialogs.

 836         wxString msg = m_delayedErrorMessage;
 837         m_delayedErrorMessage = wxEmptyString;
 838 
 839         // Do not use DisplayErrorMessage(); it screws up window order on Mac

 840         DisplayError( nullptr, msg );
 841     }
 842 
 843     if( m_delayedFocusCtrl )
 844     {
 845         m_delayedFocusCtrl->SetFocus();
 846 
 847         if( auto textEntry = dynamic_cast<wxTextEntry*>( m_delayedFocusCtrl ) )
 848             textEntry->SelectAll();
 849 
 850         m_delayedFocusCtrl = nullptr;
 851     }
 852     else if( m_delayedFocusGrid )
 853     {
 854         m_delayedFocusGrid->SetFocus();
 855         m_delayedFocusGrid->MakeCellVisible( m_delayedFocusRow, m_delayedFocusColumn );
 856         m_delayedFocusGrid->SetGridCursor( m_delayedFocusRow, m_delayedFocusColumn );
 857 
 858         m_delayedFocusGrid->EnableCellEditControl( true );
 859         m_delayedFocusGrid->ShowCellEditControl();
 860 
 861         m_delayedFocusGrid = nullptr;
 862         m_delayedFocusRow = -1;
 863         m_delayedFocusColumn = -1;
 864     }
 865 
 866     m_button3DShapeRemove->Enable( m_modelsGrid->GetNumberRows() > 0 );
 867 }
