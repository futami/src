  38 /**

  39  * POINT_EDITOR

  40  *

  41  * Tool that displays edit points allowing to modify items by dragging the points.

  42  */
  43 class POINT_EDITOR : public PCB_TOOL_BASE
  44 {
  45 public:
  46     POINT_EDITOR();
  47 
  48     /// @copydoc TOOL_INTERACTIVE::Reset()

  49     void Reset( RESET_REASON aReason ) override;
  50 
  51     /// @copydoc TOOL_INTERACTIVE::Init()

  52     bool Init() override;
  53 
  54     /**

  55      * Function OnSelected()

  56      *

  57      * Change selection event handler.

  58      */
  59     int OnSelectionChange( const TOOL_EVENT& aEvent );
  60 
  61     /**

  62      * Indicates the cursor is over an edit point.  Used to coordinate cursor shapes with

  63      * other tools.

  64      */
  65     bool HasPoint() { return m_editedPoint != nullptr; }
  66 
  67     ///> Sets up handlers for various events.

  68     void setTransitions() override;
  69 
  70 private:
  71     ///> Selection tool used for obtaining selected items

  72     SELECTION_TOOL* m_selectionTool;
  73 
  74     ///> Currently edited point, NULL if there is none.

  75     EDIT_POINT* m_editedPoint;
  76 
  77     ///> Original position for the current drag point.

  78     EDIT_POINT m_original;
  79 
  80     ///> Currently available edit points.

  81     std::shared_ptr<EDIT_POINTS> m_editPoints;
  82 
  83     // Alternative constraint, enabled while a modifier key is held

  84     std::shared_ptr<EDIT_CONSTRAINT<EDIT_POINT> > m_altConstraint;
  85 
  86     // EDIT_POINT for alternative constraint mode

  87     EDIT_POINT m_altConstrainer;
  88 
  89     // Flag indicating whether the selected zone needs to be refilled

  90     bool m_refill;
  91 
  92     std::unique_ptr<STATUS_TEXT_POPUP> m_statusPopup;
  93 
  94     ///> Updates item's points with edit points.

  95     void updateItem() const;
  96 
  97     ///> Applies the last changes to the edited item.

  98     void finishItem();
  99 
 100     /**

 101      * Validates a polygon and displays a popup warning if invalid.

 102      * @param aModified is the polygon to be checked.

 103      * @return True if polygon is valid.

 104      */
 105     bool validatePolygon( SHAPE_POLY_SET& aModified ) const;
 106 
 107     ///> Updates edit points with item's points.

 108     void updatePoints();
 109 
 110     ///> Updates which point is being edited.

 111     void updateEditedPoint( const TOOL_EVENT& aEvent );
 112 
 113     ///> Sets the current point being edited. NULL means none.

 114     void setEditedPoint( EDIT_POINT* aPoint );
 115 
 116     ///> Returns true if aPoint is the currently modified point.

 117     inline bool isModified( const EDIT_POINT& aPoint ) const
 118     {
 119         return m_editedPoint == &aPoint;
 120     }
 121 
 122     ///> Sets up an alternative constraint (typically enabled upon a modifier key being pressed).

 123     void setAltConstraint( bool aEnabled );
 124 
 125     ///> Returns a point that should be used as a constrainer for 45 degrees mode.

 126     EDIT_POINT get45DegConstrainer() const;
 127 
 128     ///> Condition to display "Create corner" context menu entry.

 129     static bool addCornerCondition( const SELECTION& aSelection );
 130 
 131     ///> Determine if the tool can currently add a corner to the given item

 132     static bool canAddCorner( const EDA_ITEM& aItem );
 133 
 134     ///> Condition to display "Remove corner" context menu entry.

 135     bool removeCornerCondition( const SELECTION& aSelection );
 136 
 137     /// TOOL_ACTION handlers

 138     int addCorner( const TOOL_EVENT& aEvent );
 139     int removeCorner( const TOOL_EVENT& aEvent );
 140     int modifiedSelection( const TOOL_EVENT& aEvent );
 141 };
