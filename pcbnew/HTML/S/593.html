  78 void PCB_RENDER_SETTINGS::LoadColors( const COLOR_SETTINGS* aSettings )
  79 {
  80     SetBackgroundColor( aSettings->GetColor( LAYER_PCB_BACKGROUND ) );
  81 
  82     // Init board layers colors:

  83     for( int i = 0; i < PCB_LAYER_ID_COUNT; i++ )
  84     {
  85         m_layerColors[i] = aSettings->GetColor( i );
  86 
  87         // Guard: if the alpah channel is too small, the layer is not visible.

  88         // clamp it to 0.2

  89         if( m_layerColors[i].a < 0.2 )
  90             m_layerColors[i].a = 0.2;
  91     }
  92 
  93     // Init specific graphic layers colors:

  94     for( int i = GAL_LAYER_ID_START; i < GAL_LAYER_ID_END; i++ )
  95         m_layerColors[i] = aSettings->GetColor( i );
  96 
  97     // Default colors for specific layers (not really board layers).

  98     m_layerColors[LAYER_VIAS_HOLES]         = COLOR4D( 0.5, 0.4, 0.0, 0.8 );
  99     m_layerColors[LAYER_PADS_PLATEDHOLES]   = aSettings->GetColor( LAYER_PCB_BACKGROUND );
 100     m_layerColors[LAYER_VIAS_NETNAMES]      = COLOR4D( 0.2, 0.2, 0.2, 0.9 );
 101     m_layerColors[LAYER_PADS_NETNAMES]      = COLOR4D( 1.0, 1.0, 1.0, 0.9 );
 102     m_layerColors[LAYER_PAD_FR_NETNAMES]    = COLOR4D( 1.0, 1.0, 1.0, 0.9 );
 103     m_layerColors[LAYER_PAD_BK_NETNAMES]    = COLOR4D( 1.0, 1.0, 1.0, 0.9 );
 104 
 105     // LAYER_PADS_TH, LAYER_NON_PLATEDHOLES, LAYER_ANCHOR ,LAYER_RATSNEST,

 106     // LAYER_VIA_THROUGH, LAYER_VIA_BBLIND, LAYER_VIA_MICROVIA

 107     // are initialized from aSettings

 108 
 109     // These colors are not actually used. Set just in case...

 110     m_layerColors[LAYER_MOD_TEXT_FR] = m_layerColors[F_SilkS];
 111     m_layerColors[LAYER_MOD_TEXT_BK] = m_layerColors[B_SilkS];
 112 
 113     // Netnames for copper layers

 114     for( LSEQ cu = LSET::AllCuMask().CuStack();  cu;  ++cu )
 115     {
 116         const COLOR4D lightLabel( 0.8, 0.8, 0.8, 0.7 );
 117         const COLOR4D darkLabel = lightLabel.Inverted();
 118         PCB_LAYER_ID layer = *cu;
 119 
 120         if( m_layerColors[layer].GetBrightness() > 0.5 )
 121             m_layerColors[GetNetnameLayer( layer )] = darkLabel;
 122         else
 123             m_layerColors[GetNetnameLayer( layer )] = lightLabel;
 124     }
 125 
 126     update();
 127 }

 130 void PCB_RENDER_SETTINGS::LoadDisplayOptions( const PCB_DISPLAY_OPTIONS& aOptions,
 131                                               bool aShowPageLimits )
 132 {
 133     m_hiContrastEnabled = ( aOptions.m_ContrastModeDisplay !=
 134                             HIGH_CONTRAST_MODE::NORMAL );
 135     m_padNumbers        = aOptions.m_DisplayPadNum;
 136     m_sketchGraphics    = !aOptions.m_DisplayGraphicsFill;
 137     m_sketchText        = !aOptions.m_DisplayTextFill;
 138     m_curvedRatsnestlines = aOptions.m_DisplayRatsnestLinesCurved;
 139     m_globalRatsnestlines = aOptions.m_ShowGlobalRatsnest;
 140 
 141     // Whether to draw tracks, vias & pads filled or as outlines

 142     m_sketchMode[LAYER_PADS_TH]      = !aOptions.m_DisplayPadFill;
 143     m_sketchMode[LAYER_VIA_THROUGH]  = !aOptions.m_DisplayViaFill;
 144     m_sketchMode[LAYER_VIA_BBLIND]   = !aOptions.m_DisplayViaFill;
 145     m_sketchMode[LAYER_VIA_MICROVIA] = !aOptions.m_DisplayViaFill;
 146     m_sketchMode[LAYER_TRACKS]       = !aOptions.m_DisplayPcbTrackFill;
 147 
 148     // Net names display settings

 149     switch( aOptions.m_DisplayNetNamesMode )
 150     {
 151     case 0:
 152         m_netNamesOnPads = false;
 153         m_netNamesOnTracks = false;
 154         m_netNamesOnVias = false;
 155         break;
 156 
 157     case 1:
 158         m_netNamesOnPads = true;
 159         m_netNamesOnTracks = false;
 160         m_netNamesOnVias = true;        // Follow pads or tracks?  For now we chose pads....

 161         break;
 162 
 163     case 2:
 164         m_netNamesOnPads = false;
 165         m_netNamesOnTracks = true;
 166         m_netNamesOnVias = false;       // Follow pads or tracks?  For now we chose pads....

 167         break;
 168 
 169     case 3:
 170         m_netNamesOnPads = true;
 171         m_netNamesOnTracks = true;
 172         m_netNamesOnVias = true;
 173         break;
 174     }
 175 
 176     // Zone display settings

 177     switch( aOptions.m_DisplayZonesMode )
 178     {
 179     case 0:
 180         m_displayZone = DZ_SHOW_FILLED;
 181         break;
 182 
 183     case 1:
 184         m_displayZone = DZ_HIDE_FILLED;
 185         break;
 186 
 187     case 2:
 188         m_displayZone = DZ_SHOW_OUTLINED;
 189         break;
 190     }
 191 
 192     // Clearance settings

 193     switch( aOptions.m_ShowTrackClearanceMode )
 194     {
 195         case PCB_DISPLAY_OPTIONS::DO_NOT_SHOW_CLEARANCE:
 196             m_clearance = CL_NONE;
 197             break;
 198 
 199         case PCB_DISPLAY_OPTIONS::SHOW_CLEARANCE_NEW_TRACKS:
 200             m_clearance = CL_NEW | CL_TRACKS;
 201             break;
 202 
 203         case PCB_DISPLAY_OPTIONS::SHOW_CLEARANCE_NEW_TRACKS_AND_VIA_AREAS:
 204             m_clearance = CL_NEW | CL_TRACKS | CL_VIAS;
 205             break;
 206 
 207         case PCB_DISPLAY_OPTIONS::SHOW_CLEARANCE_NEW_AND_EDITED_TRACKS_AND_VIA_AREAS:
 208             m_clearance = CL_NEW | CL_EDITED | CL_TRACKS | CL_VIAS;
 209             break;
 210 
 211         case PCB_DISPLAY_OPTIONS::SHOW_CLEARANCE_ALWAYS:
 212             m_clearance = CL_NEW | CL_EDITED | CL_EXISTING | CL_TRACKS | CL_VIAS;
 213             break;
 214     }
 215 
 216     if( aOptions.m_DisplayPadIsol )
 217         m_clearance |= CL_PADS;
 218 
 219     m_contrastModeDisplay = aOptions.m_ContrastModeDisplay;
 220 
 221     m_showPageLimits = aShowPageLimits;
 222 }

 225 void PCB_RENDER_SETTINGS::LoadNetSettings( const NET_SETTINGS& aSettings, const NETINFO_LIST& aList,
 226                                            const std::set<int>& aHiddenNets )
 227 {
 228     m_netColors.clear();
 229 
 230     for( const auto& pair : aSettings.m_PcbNetColors )
 231     {
 232         if( NETINFO_ITEM* net = aList.GetNetItem( pair.first ) )
 233             m_netColors[net->GetNet()] = pair.second;
 234     }
 235 
 236     m_netclassColors.clear();
 237 
 238     for( const auto& pair : aSettings.m_NetClasses )
 239     {
 240         if( pair.second->GetPcbColor() != COLOR4D::UNSPECIFIED )
 241             m_netclassColors[pair.first] = pair.second->GetPcbColor();
 242     }
 243 
 244     m_hiddenNets = aHiddenNets;
 245 }

 248 const COLOR4D& PCB_RENDER_SETTINGS::GetColor( const VIEW_ITEM* aItem, int aLayer ) const
 249 {
 250     int netCode = -1;
 251     const EDA_ITEM* item = dynamic_cast<const EDA_ITEM*>( aItem );
 252     const BOARD_CONNECTED_ITEM* conItem = dynamic_cast<const BOARD_CONNECTED_ITEM*> ( aItem );
 253 
 254     // Make items invisible in "other layers hidden" contrast mode

 255     if( m_contrastModeDisplay == HIGH_CONTRAST_MODE::HIDDEN && m_activeLayers.count( aLayer ) == 0 )
 256         return COLOR4D::CLEAR;
 257 
 258     // Hide net names in "dimmed" contrast mode

 259     if( m_contrastModeDisplay != HIGH_CONTRAST_MODE::NORMAL && IsNetnameLayer( aLayer )
 260         && m_activeLayers.count( aLayer ) == 0 )
 261         return COLOR4D::CLEAR;
 262 
 263     if( item )
 264     {
 265         // Selection disambiguation

 266         if( item->IsBrightened() )
 267         {
 268             return m_selectionCandidateColor;
 269         }
 270 
 271         // Don't let pads that *should* be NPTHs get lost

 272         if( item->Type() == PCB_PAD_T && dyn_cast<const D_PAD*>( item )->PadShouldBeNPTH() )
 273             aLayer = LAYER_MOD_TEXT_INVISIBLE;
 274 
 275         if( item->IsSelected() )
 276             return m_layerColorsSel[aLayer];
 277 
 278         if( item->Type() == PCB_MARKER_T )
 279             return m_layerColors[aLayer];
 280 
 281         // For vias, some layers depend on other layers in high contrast mode

 282         if( m_hiContrastEnabled && item->Type() == PCB_VIA_T &&
 283                 ( aLayer == LAYER_VIAS_HOLES   ||
 284                   aLayer == LAYER_VIA_THROUGH  ||
 285                   aLayer == LAYER_VIA_MICROVIA ||
 286                   aLayer == LAYER_VIA_BBLIND ) )
 287         {
 288             const VIA*   via = static_cast<const VIA*>( item );
 289             const BOARD* pcb = static_cast<const BOARD*>( item->GetParent() );
 290             bool         viaActiveLayer = false;
 291 
 292             for( auto activeLayer : m_activeLayers )
 293             {
 294                 auto lay_id = static_cast<PCB_LAYER_ID>( activeLayer );
 295                 viaActiveLayer |= via->IsOnLayer( lay_id ) && pcb->IsLayerVisible( lay_id );
 296             }
 297 
 298             if( viaActiveLayer )
 299                 return m_layerColors[aLayer];
 300             else
 301                 return m_hiContrastColor[aLayer];
 302         }
 303     }
 304 
 305     // Try to obtain the netcode for the item

 306     if( conItem )
 307         netCode = conItem->GetNetCode();
 308 
 309     // Single net highlight mode

 310     if( m_highlightEnabled && m_highlightNetcodes.count( netCode ) )
 311         return m_layerColorsHi[aLayer];
 312 
 313     // Return grayish color for non-highlighted layers in the dimmed high contrast mode

 314     if( m_contrastModeDisplay == HIGH_CONTRAST_MODE::DIMMED && m_activeLayers.count( aLayer ) == 0 )
 315         return m_hiContrastColor[aLayer];
 316 
 317     // Catch the case when highlight and high-contraste modes are enabled

 318     // and we are drawing a not highlighted track

 319     if( m_highlightEnabled )
 320         return m_layerColorsDark[aLayer];
 321 
 322     // Apply net color overrides

 323     if( conItem && m_netColorMode == NET_COLOR_MODE::ALL )
 324     {
 325         if( m_netColors.count( conItem->GetNetCode() ) )
 326             return m_netColors.at( conItem->GetNetCode() );
 327         else if( m_netclassColors.count( conItem->GetNetClassName() ) )
 328             return m_netclassColors.at( conItem->GetNetClassName() );
 329     }
 330 
 331     // No special modificators enabled

 332     return m_layerColors[aLayer];
 333 }

 372 bool PCB_PAINTER::Draw( const VIEW_ITEM* aItem, int aLayer )
 373 {
 374     const EDA_ITEM* item = dynamic_cast<const EDA_ITEM*>( aItem );
 375 
 376     if( !item )
 377         return false;
 378 
 379     // the "cast" applied in here clarifies which overloaded draw() is called

 380     switch( item->Type() )
 381     {
 382     case PCB_TRACE_T:
 383         draw( static_cast<const TRACK*>( item ), aLayer );
 384         break;
 385 
 386     case PCB_ARC_T:
 387         draw( static_cast<const ARC*>( item ), aLayer );
 388         break;
 389 
 390     case PCB_VIA_T:
 391         draw( static_cast<const VIA*>( item ), aLayer );
 392         break;
 393 
 394     case PCB_PAD_T:
 395         draw( static_cast<const D_PAD*>( item ), aLayer );
 396         break;
 397 
 398     case PCB_LINE_T:
 399     case PCB_MODULE_EDGE_T:
 400         draw( static_cast<const DRAWSEGMENT*>( item ), aLayer );
 401         break;
 402 
 403     case PCB_TEXT_T:
 404         draw( static_cast<const TEXTE_PCB*>( item ), aLayer );
 405         break;
 406 
 407     case PCB_MODULE_TEXT_T:
 408         draw( static_cast<const TEXTE_MODULE*>( item ), aLayer );
 409         break;
 410 
 411     case PCB_MODULE_T:
 412         draw( static_cast<const MODULE*>( item ), aLayer );
 413         break;
 414 
 415     case PCB_ZONE_AREA_T:
 416         draw( static_cast<const ZONE_CONTAINER*>( item ), aLayer );
 417         break;
 418 
 419     case PCB_MODULE_ZONE_AREA_T:
 420         draw( static_cast<const ZONE_CONTAINER*>( item ), aLayer );
 421         break;
 422 
 423     case PCB_DIMENSION_T:
 424         draw( static_cast<const DIMENSION*>( item ), aLayer );
 425         break;
 426 
 427     case PCB_TARGET_T:
 428         draw( static_cast<const PCB_TARGET*>( item ) );
 429         break;
 430 
 431     case PCB_MARKER_T:
 432         draw( static_cast<const MARKER_PCB*>( item ) );
 433         break;
 434 
 435     default:
 436         // Painter does not know how to draw the object

 437         return false;
 438     }
 439 
 440     return true;
 441 }

 444 void PCB_PAINTER::draw( const TRACK* aTrack, int aLayer )
 445 {
 446     VECTOR2D start( aTrack->GetStart() );
 447     VECTOR2D end( aTrack->GetEnd() );
 448     int      width = aTrack->GetWidth();
 449 
 450     if( m_pcbSettings.m_netNamesOnTracks && IsNetnameLayer( aLayer ) )
 451     {
 452         // If there is a net name - display it on the track

 453         if( aTrack->GetNetCode() > NETINFO_LIST::UNCONNECTED )
 454         {
 455             VECTOR2D line = ( end - start );
 456             double length = line.EuclideanNorm();
 457 
 458             // Check if the track is long enough to have a netname displayed

 459             if( length < 10 * width )
 460                 return;
 461 
 462             const wxString& netName = UnescapeString( aTrack->GetShortNetname() );
 463             VECTOR2D textPosition = start + line / 2.0;     // center of the track

 464 
 465             double textOrientation;
 466 
 467             if( end.y == start.y ) // horizontal

 468                 textOrientation = 0;
 469             else if( end.x == start.x ) // vertical

 470                 textOrientation = M_PI / 2;
 471             else
 472                 textOrientation = -atan( line.y / line.x );
 473 
 474             double textSize = width;
 475 
 476             m_gal->SetIsStroke( true );
 477             m_gal->SetIsFill( false );
 478             m_gal->SetStrokeColor( m_pcbSettings.GetColor( NULL, aLayer ) );
 479             m_gal->SetLineWidth( width / 10.0 );
 480             m_gal->SetFontBold( false );
 481             m_gal->SetFontItalic( false );
 482             m_gal->SetTextMirrored( false );
 483             m_gal->SetGlyphSize( VECTOR2D( textSize * 0.7, textSize * 0.7 ) );
 484             m_gal->SetHorizontalJustify( GR_TEXT_HJUSTIFY_CENTER );
 485             m_gal->SetVerticalJustify( GR_TEXT_VJUSTIFY_CENTER );
 486             m_gal->BitmapText( netName, textPosition, textOrientation );
 487         }
 488     }
 489     else if( IsCopperLayer( aLayer ) )
 490     {
 491         // Draw a regular track

 492         const COLOR4D& color = m_pcbSettings.GetColor( aTrack, aLayer );
 493         bool outline_mode = m_pcbSettings.m_sketchMode[LAYER_TRACKS];
 494         m_gal->SetStrokeColor( color );
 495         m_gal->SetFillColor( color );
 496         m_gal->SetIsStroke( outline_mode );
 497         m_gal->SetIsFill( not outline_mode );
 498         m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );
 499 
 500         m_gal->DrawSegment( start, end, width );
 501 
 502         // Clearance lines

 503         constexpr int clearanceFlags = PCB_RENDER_SETTINGS::CL_EXISTING | PCB_RENDER_SETTINGS::CL_TRACKS;
 504 
 505         if( ( m_pcbSettings.m_clearance & clearanceFlags ) == clearanceFlags )
 506         {
 507             m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );
 508             m_gal->SetIsFill( false );
 509             m_gal->SetIsStroke( true );
 510             m_gal->SetStrokeColor( color );
 511             m_gal->DrawSegment( start, end, width + aTrack->GetClearance() * 2 );
 512         }
 513     }
 514 }

 517 void PCB_PAINTER::draw( const ARC* aArc, int aLayer )
 518 {
 519     VECTOR2D center( aArc->GetCenter() );
 520     int      width = aArc->GetWidth();
 521 
 522     if( IsCopperLayer( aLayer ) )
 523     {
 524         // Draw a regular track

 525         const COLOR4D& color = m_pcbSettings.GetColor( aArc, aLayer );
 526         bool outline_mode = m_pcbSettings.m_sketchMode[LAYER_TRACKS];
 527         m_gal->SetStrokeColor( color );
 528         m_gal->SetFillColor( color );
 529         m_gal->SetIsStroke( outline_mode );
 530         m_gal->SetIsFill( not outline_mode );
 531         m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );
 532 
 533         auto radius = aArc->GetRadius();
 534         auto start_angle = DECIDEG2RAD( aArc->GetArcAngleStart() );
 535         auto angle = DECIDEG2RAD( aArc->GetAngle() );
 536 
 537         m_gal->DrawArcSegment( center, radius, start_angle, start_angle + angle, width );
 538 
 539         // Clearance lines

 540         constexpr int clearanceFlags = PCB_RENDER_SETTINGS::CL_EXISTING | PCB_RENDER_SETTINGS::CL_TRACKS;
 541 
 542         if( ( m_pcbSettings.m_clearance & clearanceFlags ) == clearanceFlags )
 543         {
 544             m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );
 545             m_gal->SetIsFill( false );
 546             m_gal->SetIsStroke( true );
 547             m_gal->SetStrokeColor( color );
 548 
 549             m_gal->DrawArcSegment( center, radius, start_angle, start_angle + angle,
 550                     width + aArc->GetClearance() * 2 );
 551         }
 552     }
 553 }

 556 void PCB_PAINTER::draw( const VIA* aVia, int aLayer )
 557 {
 558     VECTOR2D center( aVia->GetStart() );
 559     double   radius = 0.0;
 560 
 561     // Draw description layer

 562     if( IsNetnameLayer( aLayer ) )
 563     {
 564         VECTOR2D position( center );
 565 
 566         // Is anything that we can display enabled?

 567         if( m_pcbSettings.m_netNamesOnVias )
 568         {
 569             bool displayNetname = ( !aVia->GetNetname().empty() );
 570             double maxSize = PCB_RENDER_SETTINGS::MAX_FONT_SIZE;
 571             double size = aVia->GetWidth();
 572 
 573             // Font size limits

 574             if( size > maxSize )
 575                 size = maxSize;
 576 
 577             m_gal->Save();
 578             m_gal->Translate( position );
 579 
 580             // Default font settings

 581             m_gal->ResetTextAttributes();
 582             m_gal->SetStrokeColor( m_pcbSettings.GetColor( NULL, aLayer ) );
 583 
 584             // Set the text position to the pad shape position (the pad position is not the best place)

 585             VECTOR2D textpos( 0.0, 0.0 );
 586 
 587             if( displayNetname )
 588             {
 589                 wxString netname = UnescapeString( aVia->GetShortNetname() );
 590                 // calculate the size of net name text:

 591                 double tsize = 1.5 * size / netname.Length();
 592                 tsize = std::min( tsize, size );
 593                 // Use a smaller text size to handle interline, pen size..

 594                 tsize *= 0.7;
 595                 VECTOR2D namesize( tsize, tsize );
 596 
 597                 m_gal->SetGlyphSize( namesize );
 598                 m_gal->SetLineWidth( namesize.x / 12.0 );
 599                 m_gal->BitmapText( netname, textpos, 0.0 );
 600             }
 601 
 602 
 603             m_gal->Restore();
 604         }
 605         return;
 606     }
 607 
 608     // Choose drawing settings depending on if we are drawing via's pad or hole

 609     if( aLayer == LAYER_VIAS_HOLES )
 610         radius = getDrillSize( aVia ) / 2.0;
 611     else
 612         radius = aVia->GetWidth() / 2.0;
 613 
 614     bool sketchMode = false;
 615     const COLOR4D& color  = m_pcbSettings.GetColor( aVia, aLayer );
 616 
 617     switch( aVia->GetViaType() )
 618     {
 619     case VIATYPE::THROUGH:
 620         sketchMode = m_pcbSettings.m_sketchMode[LAYER_VIA_THROUGH];
 621         break;
 622 
 623     case VIATYPE::BLIND_BURIED:
 624         sketchMode = m_pcbSettings.m_sketchMode[LAYER_VIA_BBLIND];
 625         break;
 626 
 627     case VIATYPE::MICROVIA:
 628         sketchMode = m_pcbSettings.m_sketchMode[LAYER_VIA_MICROVIA];
 629         break;
 630 
 631     default:
 632         wxASSERT( false );
 633         break;
 634     }
 635 
 636     m_gal->SetIsFill( !sketchMode );
 637     m_gal->SetIsStroke( sketchMode );
 638 
 639     if( sketchMode )
 640     {
 641         // Outline mode

 642         m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );
 643         m_gal->SetStrokeColor( color );
 644     }
 645     else
 646     {
 647         // Filled mode

 648         m_gal->SetFillColor( color );
 649     }
 650 
 651     if( ( aVia->GetViaType() == VIATYPE::BLIND_BURIED || aVia->GetViaType() == VIATYPE::MICROVIA )
 652             && aLayer != LAYER_VIAS_HOLES
 653             && !m_pcbSettings.GetDrawIndividualViaLayers() )
 654     {
 655         // Outer circles of blind/buried and micro-vias are drawn in a special way to indicate the

 656         // top and bottom layers

 657         PCB_LAYER_ID layerTop, layerBottom;
 658         aVia->LayerPair( &layerTop, &layerBottom );
 659 
 660         if( !sketchMode )
 661             m_gal->SetLineWidth( ( aVia->GetWidth() - aVia->GetDrillValue() ) / 2.0 );
 662 
 663         if( aLayer == layerTop )
 664             m_gal->DrawArc( center, radius, 0.0, M_PI / 2.0 );
 665 
 666         else if( aLayer == layerBottom )
 667             m_gal->DrawArc( center, radius, M_PI, 3.0 * M_PI / 2.0 );
 668 
 669         else if( aLayer == LAYER_VIA_BBLIND || aLayer == LAYER_VIA_MICROVIA )
 670         {
 671             m_gal->DrawArc( center, radius, M_PI / 2.0, M_PI );
 672             m_gal->DrawArc( center, radius, 3.0 * M_PI / 2.0, 2.0 * M_PI );
 673         }
 674     }
 675     else
 676     {
 677         // Draw the outer circles of normal vias and the holes for all vias

 678         m_gal->DrawCircle( center, radius );
 679     }
 680 
 681     // Clearance lines

 682     constexpr int clearanceFlags = PCB_RENDER_SETTINGS::CL_EXISTING | PCB_RENDER_SETTINGS::CL_VIAS;
 683 
 684     if( ( m_pcbSettings.m_clearance & clearanceFlags ) == clearanceFlags
 685             && aLayer != LAYER_VIAS_HOLES )
 686     {
 687         m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );
 688         m_gal->SetIsFill( false );
 689         m_gal->SetIsStroke( true );
 690         m_gal->SetStrokeColor( color );
 691         m_gal->DrawCircle( center, radius + aVia->GetClearance() );
 692     }
 693 }

 696 void PCB_PAINTER::draw( const D_PAD* aPad, int aLayer )
 697 {
 698     // Draw description layer

 699     if( IsNetnameLayer( aLayer ) )
 700     {
 701         // Is anything that we can display enabled?

 702         if( m_pcbSettings.m_netNamesOnPads || m_pcbSettings.m_padNumbers )
 703         {
 704             bool displayNetname = ( m_pcbSettings.m_netNamesOnPads && !aPad->GetNetname().empty() );
 705             EDA_RECT padBBox = aPad->GetBoundingBox();
 706             VECTOR2D position = padBBox.Centre();
 707             VECTOR2D padsize = VECTOR2D( padBBox.GetSize() );
 708             double maxSize = PCB_RENDER_SETTINGS::MAX_FONT_SIZE;
 709             double size = padsize.y;
 710 
 711             m_gal->Save();
 712             m_gal->Translate( position );
 713 
 714             // Keep the size ratio for the font, but make it smaller

 715             if( padsize.x < padsize.y )
 716             {
 717                 m_gal->Rotate( DECIDEG2RAD( -900.0 ) );
 718                 size = padsize.x;
 719                 std::swap( padsize.x, padsize.y );
 720             }
 721 
 722             // Font size limits

 723             if( size > maxSize )
 724                 size = maxSize;
 725 
 726             // Default font settings

 727             m_gal->SetHorizontalJustify( GR_TEXT_HJUSTIFY_CENTER );
 728             m_gal->SetVerticalJustify( GR_TEXT_VJUSTIFY_CENTER );
 729             m_gal->SetFontBold( false );
 730             m_gal->SetFontItalic( false );
 731             m_gal->SetTextMirrored( false );
 732             m_gal->SetStrokeColor( m_pcbSettings.GetColor( NULL, aLayer ) );
 733             m_gal->SetIsStroke( true );
 734             m_gal->SetIsFill( false );
 735 
 736             // We have already translated the GAL to be centered at the center of the pad's

 737             // bounding box

 738             VECTOR2D textpos( 0.0, 0.0 );
 739 
 740             // Divide the space, to display both pad numbers and netnames and set the Y text

 741             // position to display 2 lines

 742             if( displayNetname && m_pcbSettings.m_padNumbers )
 743             {
 744                 size = size / 2.0;
 745                 textpos.y = size / 2.0;
 746             }
 747 
 748             if( displayNetname )
 749             {
 750                 wxString netname = UnescapeString( aPad->GetShortNetname() );
 751                 // calculate the size of net name text:

 752                 double tsize = 1.5 * padsize.x / netname.Length();
 753                 tsize = std::min( tsize, size );
 754                 // Use a smaller text size to handle interline, pen size..

 755                 tsize *= 0.7;
 756                 VECTOR2D namesize( tsize, tsize );
 757 
 758                 m_gal->SetGlyphSize( namesize );
 759                 m_gal->SetLineWidth( namesize.x / 12.0 );
 760                 m_gal->BitmapText( netname, textpos, 0.0 );
 761             }
 762 
 763             if( m_pcbSettings.m_padNumbers )
 764             {
 765                 const wxString& padName = aPad->GetName();
 766                 textpos.y = -textpos.y;
 767                 double tsize = 1.5 * padsize.x / padName.Length();
 768                 tsize = std::min( tsize, size );
 769                 // Use a smaller text size to handle interline, pen size..

 770                 tsize *= 0.7;
 771                 tsize = std::min( tsize, size );
 772                 VECTOR2D numsize( tsize, tsize );
 773 
 774                 m_gal->SetGlyphSize( numsize );
 775                 m_gal->SetLineWidth( numsize.x / 12.0 );
 776                 m_gal->BitmapText( padName, textpos, 0.0 );
 777             }
 778 
 779             m_gal->Restore();
 780         }
 781         return;
 782     }
 783 
 784     // Pad drawing

 785     COLOR4D color;
 786 
 787     // Pad hole color is pad-type-specific: the background color for plated holes and the

 788     // pad color for NPTHs.  Note the extra check for "should be" NPTHs to keep mis-marked

 789     // holes with no annular ring from getting "lost" in the background.

 790     if( ( aLayer == LAYER_PADS_PLATEDHOLES ) && !aPad->PadShouldBeNPTH() )
 791         color = m_pcbSettings.GetBackgroundColor();
 792     else
 793         color = m_pcbSettings.GetColor( aPad, aLayer );
 794 
 795     if( m_pcbSettings.m_sketchMode[LAYER_PADS_TH] )
 796     {
 797         // Outline mode

 798         m_gal->SetIsFill( false );
 799         m_gal->SetIsStroke( true );
 800         m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );
 801         m_gal->SetStrokeColor( color );
 802     }
 803     else
 804     {
 805         // Filled mode

 806         m_gal->SetIsFill( true );
 807         m_gal->SetIsStroke( false );
 808         m_gal->SetFillColor( color );
 809     }
 810 
 811     // Choose drawing settings depending on if we are drawing a pad itself or a hole

 812     if( aLayer == LAYER_PADS_PLATEDHOLES || aLayer == LAYER_NON_PLATEDHOLES )
 813     {
 814         const SHAPE_SEGMENT* seg = aPad->GetEffectiveHoleShape();
 815         m_gal->DrawSegment( seg->GetSeg().A, seg->GetSeg().B, seg->GetWidth() );
 816     }
 817     else
 818     {
 819         wxSize pad_size = aPad->GetSize();
 820         wxSize margin;
 821 
 822         switch( aLayer )
 823         {
 824         case F_Mask:
 825         case B_Mask:
 826             margin.x = margin.y = aPad->GetSolderMaskMargin();
 827             break;
 828 
 829         case F_Paste:
 830         case B_Paste:
 831             margin = aPad->GetSolderPasteMargin();
 832             break;
 833 
 834         default:
 835             margin.x = margin.y = 0;
 836             break;
 837         }
 838 
 839         if( margin.x != margin.y )
 840         {
 841             const_cast<D_PAD*>( aPad )->SetSize( pad_size + margin + margin );
 842             margin.x = margin.y = 0;
 843         }
 844 
 845         const std::vector<std::shared_ptr<SHAPE>>& shapes = aPad->GetEffectiveShapes();
 846 
 847         if( shapes.size() == 1 && shapes[0]->Type() == SH_SEGMENT )
 848         {
 849             const SHAPE_SEGMENT* seg = (SHAPE_SEGMENT*) shapes[0].get();
 850             m_gal->DrawSegment( seg->GetSeg().A, seg->GetSeg().B, seg->GetWidth() + 2 * margin.x );
 851         }
 852         else if( shapes.size() == 1 && shapes[0]->Type() == SH_CIRCLE )
 853         {
 854             const SHAPE_CIRCLE* circle = (SHAPE_CIRCLE*) shapes[0].get();
 855             m_gal->DrawCircle( circle->GetCenter(), circle->GetRadius() + margin.x );
 856         }
 857         else
 858         {
 859             SHAPE_POLY_SET polySet;
 860             aPad->TransformShapeWithClearanceToPolygon( polySet, margin.x );
 861             m_gal->DrawPolygon( polySet );
 862         }
 863 
 864         if( aPad->GetSize() != pad_size )
 865             const_cast<D_PAD*>( aPad )->SetSize( pad_size );
 866     }
 867 
 868     // Clearance outlines

 869     constexpr int clearanceFlags = PCB_RENDER_SETTINGS::CL_PADS;
 870 
 871     if( ( m_pcbSettings.m_clearance & clearanceFlags ) == clearanceFlags
 872             && ( aLayer == LAYER_PAD_FR
 873                 || aLayer == LAYER_PAD_BK
 874                 || aLayer == LAYER_PADS_TH ) )
 875     {
 876         m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );
 877         m_gal->SetIsStroke( true );
 878         m_gal->SetIsFill( false );
 879         m_gal->SetStrokeColor( color );
 880         int clearance = aPad->GetClearance();
 881 
 882         const std::vector<std::shared_ptr<SHAPE>>& shapes = aPad->GetEffectiveShapes();
 883 
 884         if( shapes.size() == 1 && shapes[0]->Type() == SH_SEGMENT )
 885         {
 886             const SHAPE_SEGMENT* seg = (SHAPE_SEGMENT*) shapes[0].get();
 887             m_gal->DrawSegment( seg->GetSeg().A, seg->GetSeg().B, seg->GetWidth() + 2 * clearance );
 888         }
 889         else if( shapes.size() == 1 && shapes[0]->Type() == SH_CIRCLE )
 890         {
 891             const SHAPE_CIRCLE* circle = (SHAPE_CIRCLE*) shapes[0].get();
 892             m_gal->DrawCircle( circle->GetCenter(), circle->GetRadius() + clearance );
 893         }
 894         else
 895         {
 896             SHAPE_POLY_SET polySet;
 897             aPad->TransformShapeWithClearanceToPolygon( polySet, clearance );
 898             m_gal->DrawPolygon( polySet );
 899         }
 900     }
 901 }

 904 void PCB_PAINTER::draw( const DRAWSEGMENT* aSegment, int aLayer )
 905 {
 906     const COLOR4D& color = m_pcbSettings.GetColor( aSegment, aSegment->GetLayer() );
 907     bool sketch = m_pcbSettings.m_sketchGraphics;
 908     int thickness = getLineThickness( aSegment->GetWidth() );
 909     VECTOR2D start( aSegment->GetStart() );
 910     VECTOR2D end( aSegment->GetEnd() );
 911 
 912     m_gal->SetIsFill( !sketch );
 913     m_gal->SetIsStroke( sketch );
 914     m_gal->SetFillColor( color );
 915     m_gal->SetStrokeColor( color );
 916     m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );
 917 
 918     switch( aSegment->GetShape() )
 919     {
 920     case S_SEGMENT:
 921         m_gal->DrawSegment( start, end, thickness );
 922         break;
 923 
 924     case S_RECT:
 925     {
 926         std::vector<wxPoint> pts;
 927         aSegment->GetRectCorners( &pts );
 928 
 929         if( aSegment->GetWidth() > 0 )
 930         {
 931            m_gal->DrawSegment( pts[0], pts[1], thickness );
 932            m_gal->DrawSegment( pts[1], pts[2], thickness );
 933            m_gal->DrawSegment( pts[2], pts[3], thickness );
 934            m_gal->DrawSegment( pts[3], pts[0], thickness );
 935         }
 936         else
 937         {
 938             SHAPE_POLY_SET poly;
 939             poly.NewOutline();
 940 
 941             for( const wxPoint& pt : pts )
 942                 poly.Append( pt );
 943 
 944             m_gal->DrawPolygon( poly );
 945         }
 946     }
 947         break;
 948 
 949     case S_ARC:
 950         m_gal->DrawArcSegment( start, aSegment->GetRadius(),
 951                         DECIDEG2RAD( aSegment->GetArcAngleStart() ),
 952                         DECIDEG2RAD( aSegment->GetArcAngleStart() + aSegment->GetAngle() ),
 953                         thickness );
 954         break;
 955 
 956     case S_CIRCLE:
 957         if( sketch )
 958         {
 959             m_gal->DrawCircle( start, aSegment->GetRadius() - thickness / 2 );
 960             m_gal->DrawCircle( start, aSegment->GetRadius() + thickness / 2 );
 961         }
 962         else
 963         {
 964             m_gal->SetLineWidth( thickness );
 965             m_gal->SetIsFill( aSegment->GetWidth() == 0 );
 966             m_gal->SetIsStroke( aSegment->GetWidth() > 0 );
 967             m_gal->DrawCircle( start, aSegment->GetRadius() );
 968         }
 969         break;
 970 
 971     case S_POLYGON:
 972     {
 973         SHAPE_POLY_SET& shape = const_cast<DRAWSEGMENT*>( aSegment )->GetPolyShape();
 974 
 975         if( shape.OutlineCount() == 0 )
 976             break;
 977 
 978         // On Opengl, a not convex filled polygon is usually drawn by using triangles as primitives.

 979         // CacheTriangulation() can create basic triangle primitives to draw the polygon solid shape

 980         // on Opengl.

 981         // GLU tesselation is much slower, so currently we are using our tesselation.

 982         if( m_gal->IsOpenGlEngine() && !shape.IsTriangulationUpToDate() )
 983         {
 984             shape.CacheTriangulation();
 985         }
 986 
 987         m_gal->Save();
 988 
 989         if( MODULE* module = aSegment->GetParentModule() )
 990         {
 991             m_gal->Translate( module->GetPosition() );
 992             m_gal->Rotate( -module->GetOrientationRadians() );
 993         }
 994 
 995         m_gal->SetLineWidth( thickness );
 996 
 997         if( sketch )
 998             m_gal->SetIsFill( false );
 999         else
1000             m_gal->SetIsFill( aSegment->IsPolygonFilled() );
1001 
1002         m_gal->SetIsStroke( true );
1003         m_gal->DrawPolygon( shape );
1004 
1005         m_gal->Restore();
1006         break;
1007     }
1008 
1009     case S_CURVE:
1010         m_gal->SetIsFill( false );
1011         m_gal->SetIsStroke( true );
1012         m_gal->SetLineWidth( thickness );
1013         // Use thickness as filter value to convert the curve to polyline

1014         // when the curve is not supported

1015         m_gal->DrawCurve( VECTOR2D( aSegment->GetStart() ),
1016                           VECTOR2D( aSegment->GetBezControl1() ),
1017                           VECTOR2D( aSegment->GetBezControl2() ),
1018                           VECTOR2D( aSegment->GetEnd() ), thickness );
1019         break;
1020 
1021     case S_LAST:
1022         break;
1023     }
1024 }

1113 void PCB_PAINTER::draw( const ZONE_CONTAINER* aZone, int aLayer )
1114 {
1115     PCB_LAYER_ID layer = static_cast<PCB_LAYER_ID>( aLayer );
1116 
1117     if( !aZone->IsOnLayer( layer ) )
1118         return;
1119 
1120     const COLOR4D& color = m_pcbSettings.GetColor( aZone, aLayer );
1121     std::deque<VECTOR2D> corners;
1122     PCB_RENDER_SETTINGS::DISPLAY_ZONE_MODE displayMode = m_pcbSettings.m_displayZone;
1123 
1124     // Draw the outline

1125     const SHAPE_POLY_SET* outline = aZone->Outline();
1126 
1127     if( m_pcbSettings.m_zoneOutlines && outline && outline->OutlineCount() > 0 )
1128     {
1129         m_gal->SetStrokeColor( color );

1131         m_gal->SetIsStroke( true );
1132         m_gal->SetLineWidth( m_pcbSettings.m_outlineWidth );

1137          * m_gal->DrawPolygon( *outline );

1138          * should be enough, but currently does not work to draw holes contours in a complex polygon

1139          * so each contour is draw as a simple polygon

1140          */
1141 
1142         // Draw the main contour

1143         m_gal->DrawPolyline( outline->COutline( 0 ) );
1144

1146         int holes_count = outline->HoleCount( 0 );

1168         m_gal->SetLineWidth( outline_thickness );
1169 
1170         if( displayMode == PCB_RENDER_SETTINGS::DZ_SHOW_FILLED )
1171         {
1172             m_gal->SetIsFill( true );
1173             m_gal->SetIsStroke( outline_thickness > 0 );
1174         }
1175         else if( displayMode == PCB_RENDER_SETTINGS::DZ_SHOW_OUTLINED )
1176         {
1177             m_gal->SetIsFill( false );
1178             m_gal->SetIsStroke( true );
1179         }
1180 
1181         m_gal->DrawPolygon( polySet );
1182     }

1236 void PCB_PAINTER::draw( const PCB_TARGET* aTarget )
1237 {
1238     const COLOR4D& strokeColor = m_pcbSettings.GetColor( aTarget, aTarget->GetLayer() );
1239     VECTOR2D position( aTarget->GetPosition() );
1240     double   size, radius;
1241 
1242     m_gal->SetLineWidth( getLineThickness( aTarget->GetWidth() ) );
1243     m_gal->SetStrokeColor( strokeColor );
1244     m_gal->SetIsFill( false );
1245     m_gal->SetIsStroke( true );
1246 
1247     m_gal->Save();
1248     m_gal->Translate( position );
1249 
1250     if( aTarget->GetShape() )
1251     {
1252         // shape x

1253         m_gal->Rotate( M_PI / 4.0 );
1254         size   = 2.0 * aTarget->GetSize() / 3.0;
1255         radius = aTarget->GetSize() / 2.0;
1256     }
1257     else
1258     {
1259         // shape +

1260         size   = aTarget->GetSize() / 2.0;
1261         radius = aTarget->GetSize() / 3.0;
1262     }
1263 
1264     m_gal->DrawLine( VECTOR2D( -size, 0.0 ), VECTOR2D( size, 0.0 ) );
1265     m_gal->DrawLine( VECTOR2D( 0.0, -size ), VECTOR2D( 0.0,  size ) );
1266     m_gal->DrawCircle( VECTOR2D( 0.0, 0.0 ), radius );
1267 
1268     m_gal->Restore();
1269 }
