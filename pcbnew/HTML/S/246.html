 254 /** A dialog to apply geometry transforms to a shape or set of shapes

 255  * (move, rotate around origin, scaling factor, duplication).

 256  * shapes are scaled, then moved then rotated.

 257  * aList is a list of shapes to transform or duplicate

 258  * if aShowDuplicate == false, the parameter "Duplicate count" is disabled

 259  */
 260 
 261 class DIALOG_PAD_PRIMITIVES_TRANSFORM : public DIALOG_PAD_PRIMITIVES_TRANSFORM_BASE
 262 {
 263 public:
 264     DIALOG_PAD_PRIMITIVES_TRANSFORM( wxWindow* aParent, PCB_BASE_FRAME* aFrame,
 265                                      std::vector<std::shared_ptr<DRAWSEGMENT>>& aList,
 266                                      bool aShowDuplicate );
 267 
 268     /**

 269      * Apply geometric transform (rotation, move, scale) defined in dialog

 270      * aDuplicate = 1 .. n to duplicate the list of shapes

 271      * aDuplicate = 0 to transform the list of shapes

 272      * The duplicated items are transformed, but the initial shpes are not modified.

 273      * The duplicated items are added to aList

 274      */
 275     void Transform( std::vector<std::shared_ptr<DRAWSEGMENT>>* aList = NULL,
 276                     int aDuplicateCount = 0 );
 277 
 278     /**

 279      * @return the number of duplicate, chosen by user

 280      */
 281     int GetDuplicateCount() { return m_spinCtrlDuplicateCount->GetValue(); }
 282 
 283 private:
 284     std::vector<std::shared_ptr<DRAWSEGMENT>>& m_list;
 285 
 286     UNIT_BINDER  m_vectorX;
 287     UNIT_BINDER  m_vectorY;
 288     UNIT_BINDER  m_rotation;
 289 };
