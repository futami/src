  28 #include <tool/tool_interactive.h>

  33 using ALIGNMENT_RECT = std::pair<BOARD_ITEM*, EDA_RECT>;
  34 using ALIGNMENT_RECTS = std::vector<ALIGNMENT_RECT>;
  35 
  36 class SELECTION_TOOL;
  37 
  38 class ALIGN_DISTRIBUTE_TOOL : public TOOL_INTERACTIVE
  39 {
  40 public:
  41     ALIGN_DISTRIBUTE_TOOL();
  42     virtual ~ALIGN_DISTRIBUTE_TOOL();
  43 
  44     /// @copydoc TOOL_INTERACTIVE::Reset()

  45     void Reset( RESET_REASON aReason ) override {}
  46 
  47     /// @copydoc TOOL_INTERACTIVE::Init()

  48     bool Init() override;
  49 
  50     /**

  51      * Function AlignTop()

  52      * Sets Y coordinate of the selected items to the value of the top-most selected item Y coordinate.

  53      */
  54     int AlignTop( const TOOL_EVENT& aEvent );
  55 
  56     /**

  57      * Function AlignBottom()

  58      * Sets Y coordinate of the selected items to the value of the bottom-most selected item Y coordinate.

  59      */
  60     int AlignBottom( const TOOL_EVENT& aEvent );
  61 
  62     /**

  63      * Function AlignLeft()

  64      * Sets X coordinate of the selected items to the value of the left-most selected item X coordinate.

  65      */
  66     int AlignLeft( const TOOL_EVENT& aEvent );
  67 
  68     /**

  69      * Function AlignRight()

  70      * Sets X coordinate of the selected items to the value of the right-most selected item X coordinate.

  71      */
  72     int AlignRight( const TOOL_EVENT& aEvent );
  73 
  74     /**

  75      * Function AlignCenterX()

  76      * Sets the x coordinate of the midpoint of each of the selected items to

  77      * the value of the x coordinate of the center of the middle selected item.

  78      */
  79     int AlignCenterX( const TOOL_EVENT& aEvent );
  80 
  81     /**

  82      * Function AlignCenterX()

  83      * Sets the y coordinate of the midpoint of each of the selected items to

  84      * the value of the y coordinate of the center of the middle selected item.

  85      */
  86     int AlignCenterY( const TOOL_EVENT& aEvent );
  87 
  88     /**

  89      * Function DistributeHorizontally()

  90      * Distributes the selected items along the X axis.

  91      */
  92     int DistributeHorizontally( const TOOL_EVENT& aEvent );
  93 
  94     /**

  95      * Function DistributeVertically()

  96      * Distributes the selected items along the Y axis.

  97      */
  98     int DistributeVertically( const TOOL_EVENT& aEvent );
  99 
 100     ///> Sets up handlers for various events.

 101     void setTransitions() override;
 102 
 103 private:
 104 
 105     /**

 106      * Function GetSelections()

 107      * Populates two vectors with the sorted selection and sorted locked items

 108      * Returns the size of aItems()

 109      */
 110     template< typename T >
 111     size_t GetSelections( ALIGNMENT_RECTS& aItems, ALIGNMENT_RECTS& aLocked, T aCompare );
 112 
 113     template< typename T >
 114     int selectTarget( ALIGNMENT_RECTS& aItems, ALIGNMENT_RECTS& aLocked, T aGetValue );
 115 
 116     /**

 117      * Sets X coordinate of the selected items to the value of the left-most selected item X coordinate.

 118      *

 119      * NOTE: Uses the bounding box of items, which do not get mirrored even when

 120      * the view is mirrored!

 121      */
 122     int doAlignLeft();
 123 
 124     /**

 125      * Aligns selected items using the right edge of their bounding boxes to the right-most item

 126      *

 127      * NOTE: Uses the bounding box of items, which do not get mirrored even when

 128      * the view is mirrored!

 129      */
 130     int doAlignRight();
 131 
 132     SELECTION_TOOL* m_selectionTool;
 133 
 134     ACTION_MENU* m_placementMenu;
 135 
 136     PCB_BASE_FRAME* m_frame;
 137 
 138     /**

 139      * Check a selection to ensure locks are valid for alignment.

 140      *

 141      * This is slightly different from the standard lock checking in that we ignore the lock

 142      * of the first element in the selection as this is meant to be our anchor.

 143      * We also check the lock of a pad's parent as we will not move pads independently of

 144      * the parent module

 145      */
 146     int checkLockedStatus( const PCBNEW_SELECTION &selection ) const;
 147 
 148     /**

 149      * Distributes selected items using an even spacing between the centers of their bounding boxes

 150      *

 151      * NOTE: Using the centers of bounding box of items can give unsatisfactory visual results since

 152      * items of differing widths will be placed with different gaps. Is only used if items overlap

 153      */
 154     void doDistributeCentersHorizontally( ALIGNMENT_RECTS &itemsToDistribute ) const;
 155 
 156     /**

 157      * Distributes selected items using an even spacing between the centers of their bounding boxes

 158      *

 159      * NOTE: Using the centers of bounding box of items can give unsatisfactory visual results since

 160      * items of differing widths will be placed with different gaps. Is only used if items overlap

 161      */
 162     void doDistributeCentersVertically( ALIGNMENT_RECTS &itemsToDistribute ) const;
 163 
 164     /**

 165      * Distributes selected items using an even spacing between their bounding boxes

 166      *

 167      * NOTE: Using the edges of bounding box of items is only possible if there is enough space

 168      * between them. If this is not the case, use the center spacing method

 169      */
 170     void doDistributeGapsHorizontally( ALIGNMENT_RECTS &itemsToDistribute,
 171                                        const BOARD_ITEM *lastItem,
 172                                        int totalGap ) const;
 173 
 174     /**

 175      * Distributes selected items using an even spacing between their bounding boxes

 176      *

 177      * NOTE: Using the edges of bounding box of items is only possible if there is enough space

 178      * between them. If this is not the case, use the center spacing method

 179      */
 180     void doDistributeGapsVertically( ALIGNMENT_RECTS &itemsToDistribute,
 181                                      const BOARD_ITEM *lastItem,
 182                                      int totalGap ) const;
 183 };

 185 #endif /* ALIGN_DISTRIBUTE_TOOL_H_ */
