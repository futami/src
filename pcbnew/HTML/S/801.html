 143 bool SELECTION_TOOL::Init()
 144 {
 145     auto frame = getEditFrame<PCB_BASE_FRAME>();
 146 
 147     if( frame && ( frame->IsType( FRAME_FOOTPRINT_VIEWER )
 148                    || frame->IsType( FRAME_FOOTPRINT_VIEWER_MODAL ) ) )
 149     {
 150         frame->AddStandardSubMenus( m_menu );
 151         return true;
 152     }
 153 
 154     auto selectMenu = std::make_shared<SELECT_MENU>();
 155     selectMenu->SetTool( this );
 156     m_menu.AddSubMenu( selectMenu );
 157 
 158     auto& menu = m_menu.GetMenu();
 159 
 160     auto activeToolCondition =
 161             [ frame ] ( const SELECTION& aSel )
 162             {
 163                 return !frame->ToolStackIsEmpty();
 164             };
 165 
 166     menu.AddMenu( selectMenu.get(), SELECTION_CONDITIONS::NotEmpty );
 167     menu.AddSeparator( 1000 );
 168 
 169     // "Cancel" goes at the top of the context menu when a tool is active

 170     menu.AddItem( ACTIONS::cancelInteractive, activeToolCondition, 1 );
 171     menu.AddSeparator( 1 );
 172 
 173     if( frame )
 174         frame->AddStandardSubMenus( m_menu );
 175 
 176     return true;
 177 }

 180 void SELECTION_TOOL::Reset( RESET_REASON aReason )
 181 {
 182     m_frame = getEditFrame<PCB_BASE_FRAME>();
 183     m_locked = true;
 184 
 185     if( aReason == TOOL_BASE::MODEL_RELOAD )
 186     {
 187         // Deselect any item being currently in edit, to avoid unexpected behavior

 188         // and remove pointers to the selected items from containers

 189         // without changing their properties (as they are already deleted

 190         // while a new board is loaded)

 191         ClearSelection( true );
 192 
 193         getView()->GetPainter()->GetSettings()->SetHighlight( false );
 194     }
 195     else
 196     {
 197         // Restore previous properties of selected items and remove them from containers

 198         ClearSelection( true );
 199     }
 200 
 201     // Reinsert the VIEW_GROUP, in case it was removed from the VIEW

 202     view()->Remove( &m_selection );
 203     view()->Add( &m_selection );
 204 }

 207 int SELECTION_TOOL::Main( const TOOL_EVENT& aEvent )
 208 {
 209     // Main loop: keep receiving events

 210     while( TOOL_EVENT* evt = Wait() )
 211     {
 212         if( m_frame->ToolStackIsEmpty() )
 213             m_frame->GetCanvas()->SetCurrentCursor( wxCURSOR_ARROW );
 214 
 215         bool dragAlwaysSelects = getEditFrame<PCB_BASE_FRAME>()->GetDragSelects();
 216         TRACK_DRAG_ACTION dragAction = getEditFrame<PCB_BASE_FRAME>()->Settings().m_TrackDragAction;
 217         m_additive = m_subtractive = m_exclusive_or = false;
 218 
 219         // OSX uses CTRL for context menu, and SHIFT is exclusive-or

 220 #ifdef __WXOSX_MAC__
 221         if( evt->Modifier( MD_SHIFT ) )
 222             m_exclusive_or = true;
 223 #else
 224         if( evt->Modifier( MD_SHIFT ) && evt->Modifier( MD_CTRL ) )
 225             m_subtractive = true;
 226         else if( evt->Modifier( MD_SHIFT ) )
 227             m_additive = true;
 228         else if( evt->Modifier( MD_CTRL ) )
 229             m_exclusive_or = true;
 230 #endif
 231 
 232         // Is the user requesting that the selection list include all possible

 233         // items without removing less likely selection candidates

 234         m_skip_heuristics = !!evt->Modifier( MD_ALT );
 235 
 236         // Single click? Select single object

 237         if( evt->IsClick( BUT_LEFT ) )
 238         {
 239             m_frame->FocusOnItem( nullptr );
 240 
 241             selectPoint( evt->Position() );
 242         }
 243 
 244         // right click? if there is any object - show the context menu

 245         else if( evt->IsClick( BUT_RIGHT ) )
 246         {
 247             bool selectionCancelled = false;
 248 
 249             if( m_selection.Empty() ||
 250                     !m_selection.GetBoundingBox().Contains( wxPoint( evt->Position() ) ) )
 251             {
 252                 ClearSelection();
 253                 selectPoint( evt->Position(), false, &selectionCancelled );
 254                 m_selection.SetIsHover( true );
 255             }
 256 
 257             if( !selectionCancelled )
 258                 m_menu.ShowContextMenu( m_selection );
 259         }
 260 
 261         // double click? Display the properties window

 262         else if( evt->IsDblClick( BUT_LEFT ) )
 263         {
 264             m_frame->FocusOnItem( nullptr );
 265 
 266             if( m_selection.Empty() )
 267                 selectPoint( evt->Position() );
 268 
 269             m_toolMgr->RunAction( PCB_ACTIONS::properties, true );
 270         }
 271 
 272         // drag with LMB? Select multiple objects (or at least draw a selection box) or drag them

 273         else if( evt->IsDrag( BUT_LEFT ) )
 274         {
 275             m_frame->FocusOnItem( nullptr );
 276 
 277             if( m_additive || m_subtractive || m_exclusive_or || dragAlwaysSelects )
 278             {
 279                 selectMultiple();
 280             }
 281             else
 282             {
 283                 // selection is empty? try to start dragging the item under the point where drag

 284                 // started

 285                 if( m_selection.Empty() && selectCursor() )
 286                     m_selection.SetIsHover( true );
 287 
 288                 // Check if dragging has started within any of selected items bounding box

 289                 if( selectionContains( evt->Position() ) )
 290                 {
 291                     // Yes -> run the move tool and wait till it finishes

 292                     TRACK* track = dynamic_cast<TRACK*>( m_selection.GetItem( 0 ) );
 293 
 294                     if( track && dragAction == TRACK_DRAG_ACTION::DRAG )
 295                         m_toolMgr->RunAction( PCB_ACTIONS::drag45Degree, true );
 296                     else if( track && dragAction == TRACK_DRAG_ACTION::DRAG_FREE_ANGLE )
 297                         m_toolMgr->RunAction( PCB_ACTIONS::dragFreeAngle, true );
 298                     else
 299                         m_toolMgr->RunAction( PCB_ACTIONS::drag, true );
 300                 }
 301                 else
 302                 {
 303                     // No -> drag a selection box

 304                     selectMultiple();
 305                 }
 306             }
 307         }
 308 
 309         else if( evt->IsCancel() )
 310         {
 311             m_frame->FocusOnItem( nullptr );
 312 
 313             ClearSelection();
 314 
 315             if( evt->FirstResponder() == this )
 316                 m_toolMgr->RunAction( PCB_ACTIONS::clearHighlight );
 317         }
 318 
 319         else if( evt->Action() == TA_UNDO_REDO_PRE )
 320         {
 321             ClearSelection();
 322         }
 323 
 324         else
 325             evt->SetPassEvent();
 326     }
 327 
 328     return 0;
 329 }

 338 PCBNEW_SELECTION& SELECTION_TOOL::RequestSelection( CLIENT_SELECTION_FILTER aClientFilter,
 339                                                     std::vector<BOARD_ITEM*>* aFiltered,
 340                                                     bool aConfirmLockedItems )
 341 {
 342     bool selectionEmpty = m_selection.Empty();
 343     m_selection.SetIsHover( selectionEmpty );
 344 
 345     if( selectionEmpty )
 346     {
 347         m_toolMgr->RunAction( PCB_ACTIONS::selectionCursor, true, aClientFilter );
 348         m_selection.ClearReferencePoint();
 349     }
 350 
 351     if ( aConfirmLockedItems && CheckLock() == SELECTION_LOCKED )
 352     {
 353         ClearSelection();
 354         return m_selection;
 355     }
 356 
 357     if( aClientFilter )
 358     {
 359         GENERAL_COLLECTOR collector;
 360 
 361         for( auto item : m_selection )
 362             collector.Append( item );
 363 
 364         aClientFilter( VECTOR2I(), collector );
 365 
 366         /*

 367          * The first step is to find the items that may have been added by the client filter

 368          * This can happen if the locked pads select the module instead

 369          */
 370         std::vector<EDA_ITEM*> new_items;
 371         std::set_difference( collector.begin(), collector.end(),
 372                              m_selection.begin(), m_selection.end(),
 373                              std::back_inserter( new_items ) );
 374 
 375         /**

 376          * The second step is to find the items that were removed by the client filter

 377          */
 378         std::vector<EDA_ITEM*> diff;
 379         std::set_difference( m_selection.begin(), m_selection.end(),
 380                              collector.begin(), collector.end(),
 381                              std::back_inserter( diff ) );
 382 
 383         if( aFiltered )
 384         {
 385             for( auto item : diff )
 386                 aFiltered->push_back( static_cast<BOARD_ITEM*>( item ) );
 387         }
 388 
 389         /**

 390          * Once we find the adjustments to m_selection that are required by the client filter, we

 391          * apply them both

 392          */
 393         for( auto item : diff )
 394             unhighlight( static_cast<BOARD_ITEM*>( item ), SELECTED, &m_selection );
 395 
 396         for( auto item : new_items )
 397             highlight( static_cast<BOARD_ITEM*>( item ), SELECTED, &m_selection );
 398 
 399         m_frame->GetCanvas()->ForceRefresh();
 400     }
 401 
 402     return m_selection;
 403 }

 431 bool SELECTION_TOOL::selectPoint( const VECTOR2I& aWhere, bool aOnDrag,
 432                                   bool* aSelectionCancelledFlag,
 433                                   CLIENT_SELECTION_FILTER aClientFilter )
 434 {
 435     GENERAL_COLLECTORS_GUIDE guide = getCollectorsGuide();
 436     GENERAL_COLLECTOR        collector;
 437     auto&                    displayOpts = m_frame->GetDisplayOptions();
 438 
 439     guide.SetIgnoreZoneFills( displayOpts.m_DisplayZonesMode != 0 );
 440 
 441     collector.Collect( board(),
 442         m_editModules ? GENERAL_COLLECTOR::ModuleItems : GENERAL_COLLECTOR::AllBoardItems,
 443         wxPoint( aWhere.x, aWhere.y ), guide );
 444 
 445     // Remove unselectable items

 446     for( int i = collector.GetCount() - 1; i >= 0; --i )
 447     {
 448         if( !Selectable( collector[ i ] ) || ( aOnDrag && collector[i]->IsLocked() ) )
 449             collector.Remove( i );
 450     }
 451 
 452     m_selection.ClearReferencePoint();
 453 
 454     // Allow the client to do tool- or action-specific filtering to see if we

 455     // can get down to a single item

 456     if( aClientFilter )
 457         aClientFilter( aWhere, collector );
 458 
 459     // Apply the stateful filter

 460     filterCollectedItems( collector );
 461 
 462     // Apply some ugly heuristics to avoid disambiguation menus whenever possible

 463     if( collector.GetCount() > 1 && !m_skip_heuristics )
 464     {
 465         GuessSelectionCandidates( collector, aWhere );
 466     }
 467 
 468     // If still more than one item we're going to have to ask the user.

 469     if( collector.GetCount() > 1 )
 470     {
 471         if( aOnDrag )
 472             Wait( TOOL_EVENT( TC_ANY, TA_MOUSE_UP, BUT_LEFT ) );
 473 
 474         if( !doSelectionMenu( &collector, wxEmptyString ) )
 475         {
 476             if( aSelectionCancelledFlag )
 477                 *aSelectionCancelledFlag = true;
 478 
 479             return false;
 480         }
 481     }
 482 
 483     bool anyAdded      = false;
 484     bool anySubtracted = false;
 485 
 486     if( !m_additive && !m_subtractive && !m_exclusive_or )
 487     {
 488         if( m_selection.GetSize() > 0 )
 489         {
 490             ClearSelection( true /*quiet mode*/ );
 491             anySubtracted = true;
 492         }
 493     }
 494 
 495     if( collector.GetCount() > 0 )
 496     {
 497         for( int i = 0; i < collector.GetCount(); ++i )
 498         {
 499             if( m_subtractive || ( m_exclusive_or && collector[i]->IsSelected() ) )
 500             {
 501                 unselect( collector[i] );
 502                 anySubtracted = true;
 503             }
 504             else
 505             {
 506                 select( collector[i] );
 507                 anyAdded = true;
 508             }
 509         }
 510     }
 511 
 512     if( anyAdded )
 513     {
 514         m_toolMgr->ProcessEvent( EVENTS::SelectedEvent );
 515         return true;
 516     }
 517     else if( anySubtracted )
 518     {
 519         m_toolMgr->ProcessEvent( EVENTS::UnselectedEvent );
 520         return true;
 521     }
 522 
 523     return false;
 524 }

 539 bool SELECTION_TOOL::selectMultiple()
 540 {
 541     bool cancelled = false;     // Was the tool cancelled while it was running?

 542     m_multiple = true;          // Multiple selection mode is active

 543     KIGFX::VIEW* view = getView();
 544 
 545     KIGFX::PREVIEW::SELECTION_AREA area;
 546     view->Add( &area );
 547 
 548     bool anyAdded = false;
 549     bool anySubtracted = false;
 550 
 551     while( TOOL_EVENT* evt = Wait() )
 552     {
 553         if( evt->IsCancelInteractive() || evt->IsActivate() )
 554         {
 555             cancelled = true;
 556             break;
 557         }
 558 
 559         if( evt->IsDrag( BUT_LEFT ) )
 560         {
 561             if( !m_additive && !m_subtractive && !m_exclusive_or )
 562             {
 563                 if( m_selection.GetSize() > 0 )
 564                 {
 565                     anySubtracted = true;
 566                     ClearSelection( true /*quiet mode*/ );
 567                 }
 568             }
 569 
 570             // Start drawing a selection box

 571             area.SetOrigin( evt->DragOrigin() );
 572             area.SetEnd( evt->Position() );
 573             area.SetAdditive( m_additive );
 574             area.SetSubtractive( m_subtractive );
 575             area.SetExclusiveOr( m_exclusive_or );
 576 
 577             view->SetVisible( &area, true );
 578             view->Update( &area );
 579             getViewControls()->SetAutoPan( true );
 580         }
 581 
 582         if( evt->IsMouseUp( BUT_LEFT ) )
 583         {
 584             getViewControls()->SetAutoPan( false );
 585 
 586             // End drawing the selection box

 587             view->SetVisible( &area, false );
 588 
 589             // Mark items within the selection box as selected

 590             std::vector<KIGFX::VIEW::LAYER_ITEM_PAIR> selectedItems;
 591 
 592             // Filter the view items based on the selection box

 593             BOX2I selectionBox = area.ViewBBox();
 594             view->Query( selectionBox, selectedItems );         // Get the list of selected items

 595 
 596             std::vector<KIGFX::VIEW::LAYER_ITEM_PAIR>::iterator it, it_end;
 597 
 598             int width = area.GetEnd().x - area.GetOrigin().x;
 599             int height = area.GetEnd().y - area.GetOrigin().y;
 600 
 601             /* Selection mode depends on direction of drag-selection:

 602              * Left > Right : Select objects that are fully enclosed by selection

 603              * Right > Left : Select objects that are crossed by selection

 604              */
 605             bool windowSelection = width >= 0 ? true : false;
 606 
 607             if( view->IsMirroredX() )
 608                 windowSelection = !windowSelection;
 609 
 610             // Construct an EDA_RECT to determine BOARD_ITEM selection

 611             EDA_RECT selectionRect( (wxPoint) area.GetOrigin(), wxSize( width, height ) );
 612 
 613             selectionRect.Normalize();
 614 
 615             for( it = selectedItems.begin(), it_end = selectedItems.end(); it != it_end; ++it )
 616             {
 617                 BOARD_ITEM* item = static_cast<BOARD_ITEM*>( it->first );
 618 
 619                 if( !item || !Selectable( item ) || !itemPassesFilter( item ) )
 620                     continue;
 621 
 622                 if( item->HitTest( selectionRect, windowSelection ) )
 623                 {
 624                     if( m_subtractive || ( m_exclusive_or && item->IsSelected() ) )
 625                     {
 626                         unselect( item );
 627                         anySubtracted = true;
 628                     }
 629                     else
 630                     {
 631                         select( item );
 632                         anyAdded = true;
 633                     }
 634                 }
 635             }
 636 
 637             m_selection.SetIsHover( false );
 638 
 639             // Inform other potentially interested tools

 640             if( anyAdded )
 641                 m_toolMgr->ProcessEvent( EVENTS::SelectedEvent );
 642             else if( anySubtracted )
 643                 m_toolMgr->ProcessEvent( EVENTS::UnselectedEvent );
 644 
 645             break;  // Stop waiting for events

 646         }
 647     }
 648 
 649     getViewControls()->SetAutoPan( false );
 650 
 651     // Stop drawing the selection box

 652     view->Remove( &area );
 653     m_multiple = false;         // Multiple selection mode is inactive

 654 
 655     if( !cancelled )
 656         m_selection.ClearReferencePoint();
 657 
 658     return cancelled;
 659 }

 662 SELECTION_LOCK_FLAGS SELECTION_TOOL::CheckLock()
 663 {
 664     if( !m_locked || m_editModules )
 665         return SELECTION_UNLOCKED;
 666 
 667     bool containsLocked = false;
 668 
 669     // Check if the selection contains locked items

 670     for( const auto& item : m_selection )
 671     {

 677             break;
 678

 681         case PCB_MODULE_ZONE_AREA_T:
 682             if( static_cast<MODULE*>( item->GetParent() )->IsLocked() )
 683                 containsLocked = true;
 684             break;
 685

 687             break;
 688         }
 689     }
 690 
 691     if( containsLocked )
 692     {
 693         if( IsOK( m_frame, _( "Selection contains locked items. Do you want to continue?" ) ) )
 694         {
 695             m_locked = false;
 696             return SELECTION_LOCK_OVERRIDE;
 697         }
 698         else
 699             return SELECTION_LOCKED;
 700     }
 701 
 702     return SELECTION_UNLOCKED;

 870 void SELECTION_TOOL::selectConnectedTracks( BOARD_CONNECTED_ITEM& aStartItem,
 871                                             STOP_CONDITION aStopCondition )
 872 {
 873     constexpr KICAD_T types[] = { PCB_TRACE_T, PCB_ARC_T, PCB_VIA_T, PCB_PAD_T, EOT };
 874 
 875     auto connectivity = board()->GetConnectivity();
 876     auto connectedItems = connectivity->GetConnectedItems( &aStartItem, types );
 877 
 878     std::map<wxPoint, std::vector<TRACK*>> trackMap;
 879     std::map<wxPoint, VIA*>                viaMap;
 880     std::map<wxPoint, D_PAD*>              padMap;
 881 
 882     // Build maps of connected items

 883     for( BOARD_CONNECTED_ITEM* item : connectedItems )
 884     {
 885         switch( item->Type() )
 886         {
 887         case PCB_TRACE_T:
 888         {
 889             TRACK* track = static_cast<TRACK*>( item );
 890             trackMap[ track->GetStart() ].push_back( track );
 891             trackMap[ track->GetEnd() ].push_back( track );
 892         }
 893             break;
 894 
 895         case PCB_VIA_T:
 896         {
 897             VIA* via = static_cast<VIA*>( item );
 898             viaMap[ via->GetStart() ] = via;
 899         }
 900             break;
 901 
 902         case PCB_PAD_T:
 903         {
 904             D_PAD* pad = static_cast<D_PAD*>( item );
 905             padMap[ pad->GetPosition() ] = pad;
 906         }
 907             break;
 908 
 909         default:
 910             break;
 911         }
 912 
 913         item->SetState( SKIP_STRUCT, false );
 914     }
 915 
 916     std::vector<wxPoint> activePts;
 917 
 918     // Set up the initial active points

 919     switch( aStartItem.Type() )
 920     {
 921     case PCB_TRACE_T:
 922         activePts.push_back( static_cast<TRACK*>( &aStartItem )->GetStart() );
 923         activePts.push_back( static_cast<TRACK*>( &aStartItem )->GetEnd() );
 924         break;
 925 
 926     case PCB_VIA_T:
 927         activePts.push_back( static_cast<TRACK*>( &aStartItem )->GetStart() );
 928         break;
 929 
 930     case PCB_PAD_T:
 931         activePts.push_back( aStartItem.GetPosition() );
 932         break;
 933 
 934     default:
 935         break;
 936     }
 937 
 938     bool expand = true;
 939 
 940     // Iterative push from all active points

 941     while( expand )
 942     {
 943         expand = false;
 944 
 945         for( int i = activePts.size() - 1; i >= 0; --i )
 946         {
 947             wxPoint pt = activePts[i];
 948 
 949             if( trackMap[ pt ].size() > 2 && aStopCondition == STOP_AT_JUNCTION )
 950             {
 951                 activePts.erase( activePts.begin() + i );
 952                 continue;
 953             }
 954 
 955             if( padMap.count( pt ) && aStopCondition != STOP_NEVER )
 956             {
 957                 activePts.erase( activePts.begin() + i );
 958                 continue;
 959             }
 960 
 961             for( TRACK* track : trackMap[ pt ] )
 962             {
 963                 if( track->GetState( SKIP_STRUCT ) )
 964                     continue;
 965 
 966                 track->SetState( SKIP_STRUCT, true );
 967                 select( track );
 968 
 969                 if( track->GetStart() == pt )
 970                     activePts.push_back( track->GetEnd() );
 971                 else
 972                     activePts.push_back( track->GetStart() );
 973 
 974                 expand = true;
 975             }
 976 
 977             if( viaMap.count( pt ) && !viaMap[ pt ]->IsSelected() )
 978                 select( viaMap[ pt ] );
 979 
 980             activePts.erase( activePts.begin() + i );
 981         }
 982     }
 983 }

1020 void SELECTION_TOOL::selectAllItemsOnSheet( wxString& aSheetPath )
1021 {
1022     std::list<MODULE*> modList;
1023 
1024     // store all modules that are on that sheet path

1025     for( MODULE* module : board()->Modules() )
1026     {
1027         if( module == nullptr )
1028             continue;
1029 
1030         wxString footprint_path = module->GetPath().AsString().BeforeLast('/');
1031 
1032         if( aSheetPath.IsEmpty() )
1033             aSheetPath += '/';
1034 
1035         if( footprint_path == aSheetPath )
1036             modList.push_back( module );
1037     }
1038 
1039     //Generate a list of all pads, and of all nets they belong to.

1040     std::list<int> netcodeList;
1041     std::list<D_PAD*> padList;
1042     for( MODULE* mmod : modList )
1043     {
1044         for( D_PAD* pad : mmod->Pads() )
1045         {
1046             if( pad->IsConnected() )
1047             {
1048                 netcodeList.push_back( pad->GetNetCode() );
1049                 padList.push_back( pad );
1050             }
1051         }
1052     }
1053     // remove all duplicates

1054     netcodeList.sort();
1055     netcodeList.unique();
1056 
1057     // auto select trivial connections segments which are launched from the pads

1058     std::list<TRACK*> launchTracks;
1059 
1060     for( D_PAD* pad : padList )
1061         selectConnectedTracks( *pad, STOP_NEVER );
1062 
1063     // now we need to find all modules that are connected to each of these nets

1064     // then we need to determine if these modules are in the list of modules

1065     // belonging to this sheet ( modList )

1066     std::list<int> removeCodeList;
1067     constexpr KICAD_T padType[] = { PCB_PAD_T, EOT };
1068 
1069     for( int netCode : netcodeList )
1070     {
1071         for( BOARD_CONNECTED_ITEM* mitem : board()->GetConnectivity()->GetNetItems( netCode, padType ) )
1072         {
1073             if( mitem->Type() == PCB_PAD_T)
1074             {
1075                 bool found = std::find( modList.begin(), modList.end(), mitem->GetParent() ) != modList.end();
1076 
1077                 if( !found )
1078                 {
1079                     // if we cannot find the module of the pad in the modList

1080                     // then we can assume that that module is not located in the same

1081                     // schematic, therefore invalidate this netcode.

1082                     removeCodeList.push_back( netCode );
1083                     break;
1084                 }
1085             }
1086         }
1087     }
1088 
1089     // remove all duplicates

1090     removeCodeList.sort();
1091     removeCodeList.unique();
1092 
1093     for( int removeCode : removeCodeList )
1094     {
1095         netcodeList.remove( removeCode );
1096     }
1097 
1098     std::list<BOARD_CONNECTED_ITEM*> localConnectionList;
1099     constexpr KICAD_T trackViaType[] = { PCB_TRACE_T, PCB_VIA_T, EOT };
1100 
1101     for( int netCode : netcodeList )
1102     {
1103         for( BOARD_CONNECTED_ITEM* item : board()->GetConnectivity()->GetNetItems( netCode, trackViaType ) )
1104             localConnectionList.push_back( item );
1105     }
1106 
1107     for( BOARD_ITEM* i : modList )
1108     {
1109         if( i != NULL )
1110             select( i );
1111     }
1112 
1113     for( BOARD_CONNECTED_ITEM* i : localConnectionList )
1114     {
1115         if( i != NULL )
1116             select( i );
1117     }
1118 }

1400         if( ( !m_filter.zones && !zone->GetIsKeepout() )
1401             || ( !m_filter.keepouts && zone->GetIsKeepout() ) )
1402         {

1672 bool SELECTION_TOOL::Selectable( const BOARD_ITEM* aItem, bool checkVisibilityOnly ) const
1673 {
1674     const RENDER_SETTINGS* settings = getView()->GetPainter()->GetSettings();
1675 
1676     if( settings->GetHighContrast() )
1677     {
1678         int                    itemLayers[KIGFX::VIEW::VIEW_MAX_LAYERS], layers_count;
1679         std::set<unsigned int> activeLayers = settings->GetActiveLayers();
1680 
1681         aItem->ViewGetLayers( itemLayers, layers_count );
1682 
1683         bool onActive = false;          // Is the item on any of active layers?

1684 
1685         for( int i = 0; i < layers_count; ++i )
1686         {
1687             if( activeLayers.count( itemLayers[i] ) > 0 ) // Item is on at least one of the active layers

1688             {
1689                 onActive = true;
1690                 break;
1691             }
1692         }
1693 
1694         if( !onActive ) // We do not want to select items that are in the background

1695         {
1696             return false;
1697         }
1698     }
1699 
1700     switch( aItem->Type() )
1701     {
1702     case PCB_ZONE_AREA_T:
1703     case PCB_MODULE_ZONE_AREA_T:
1704     {
1705         // Check to see if this keepout is part of a footprint

1706         // If it is, and we are not editing the footprint, it should not be selectable

1707         const bool zoneInFootprint =
1708                 aItem->GetParent() != nullptr && aItem->GetParent()->Type() == PCB_MODULE_T;
1709         if( zoneInFootprint && !m_editModules && !checkVisibilityOnly )
1710             return false;
1711 
1712         // zones can exist on multiple layers!

1713         {
1714             auto* zone       = static_cast<const ZONE_CONTAINER*>( aItem );
1715             auto  zoneLayers = zone->GetLayerSet().Seq();
1716 
1717             for( unsigned int i = 0; i < zoneLayers.size(); i++ )
1718             {
1719                 if( board()->IsLayerVisible( zoneLayers[i] ) )
1720                 {
1721                     return true;
1722                 }
1723             }
1724 
1725             // No active layers selected!

1726             return false;
1727         }
1728     }
1729         break;
1730 
1731     case PCB_TRACE_T:
1732     case PCB_ARC_T:
1733         {
1734             if( !board()->IsElementVisible( LAYER_TRACKS ) )
1735                 return false;
1736         }
1737         break;
1738 
1739     case PCB_VIA_T:
1740         {
1741             const VIA* via = static_cast<const VIA*>( aItem );
1742 
1743             // Check if appropriate element layer is visible

1744             switch( via->GetViaType() )
1745             {
1746             case VIATYPE::THROUGH:
1747                 if( !board()->IsElementVisible( LAYER_VIA_THROUGH ) )
1748                     return false;
1749                 break;
1750 
1751             case VIATYPE::BLIND_BURIED:
1752                 if( !board()->IsElementVisible( LAYER_VIA_BBLIND ) )
1753                     return false;
1754                 break;
1755 
1756             case VIATYPE::MICROVIA:
1757                 if( !board()->IsElementVisible( LAYER_VIA_MICROVIA ) )
1758                     return false;
1759                 break;
1760 
1761             default:
1762                 wxFAIL;
1763                 return false;
1764             }
1765 
1766             // For vias it is enough if only one of its layers is visible

1767             return ( board()->GetVisibleLayers() & via->GetLayerSet() ).any();
1768         }
1769 
1770     case PCB_MODULE_T:
1771     {
1772         // In modedit, we do not want to select the module itself.

1773         if( m_editModules )
1774             return false;
1775 
1776         // Allow selection of footprints if some part of the footprint is visible.

1777 
1778         MODULE* module = const_cast<MODULE*>( static_cast<const MODULE*>( aItem ) );
1779 
1780         for( auto item : module->GraphicalItems() )
1781         {
1782             if( Selectable( item, true ) )
1783                 return true;
1784         }
1785 
1786         for( auto pad : module->Pads() )
1787         {
1788             if( Selectable( pad, true ) )
1789                 return true;
1790         }
1791 
1792         for( auto zone : module->Zones() )
1793         {
1794             if( Selectable( zone, true ) )
1795                 return true;
1796         }
1797 
1798         return false;
1799     }
1800 
1801     case PCB_MODULE_TEXT_T:
1802         // Multiple selection is only allowed in modedit mode.  In pcbnew, you have to select

1803         // module subparts one by one, rather than with a drag selection.  This is so you can

1804         // pick up items under an (unlocked) module without also moving the module's sub-parts.

1805         if( !m_editModules && !checkVisibilityOnly )
1806         {
1807             if( m_multiple && !settings->GetHighContrast() )
1808                 return false;
1809         }
1810 
1811         if( !m_editModules && !view()->IsVisible( aItem ) )
1812             return false;
1813 
1814         break;
1815 
1816     case PCB_MODULE_EDGE_T:
1817     case PCB_PAD_T:
1818     {
1819         // Multiple selection is only allowed in modedit mode.  In pcbnew, you have to select

1820         // module subparts one by one, rather than with a drag selection.  This is so you can

1821         // pick up items under an (unlocked) module without also moving the module's sub-parts.

1822         if( !m_editModules && !checkVisibilityOnly )
1823         {
1824             if( m_multiple )
1825                 return false;
1826         }
1827 
1828         if( aItem->Type() == PCB_PAD_T )
1829         {
1830             auto pad = static_cast<const D_PAD*>( aItem );
1831 
1832             // In pcbnew, locked modules prevent individual pad selection.

1833             // In modedit, we don't enforce this as the module is assumed to be edited by design.

1834             if( !m_editModules && !checkVisibilityOnly )
1835             {
1836                 if( pad->GetParent() && pad->GetParent()->IsLocked() )
1837                     return false;
1838             }
1839 
1840             // Check render mode (from the Items tab) first

1841             switch( pad->GetAttribute() )
1842             {
1843             case PAD_ATTRIB_STANDARD:
1844             case PAD_ATTRIB_HOLE_NOT_PLATED:
1845                 if( !board()->IsElementVisible( LAYER_PADS_TH ) )
1846                     return false;
1847                 break;
1848 
1849             case PAD_ATTRIB_CONN:
1850             case PAD_ATTRIB_SMD:
1851                 if( pad->IsOnLayer( F_Cu ) && !board()->IsElementVisible( LAYER_PAD_FR ) )
1852                     return false;
1853                 else if( pad->IsOnLayer( B_Cu ) && !board()->IsElementVisible( LAYER_PAD_BK ) )
1854                     return false;
1855                 break;
1856             }
1857 
1858             // Otherwise, pads are selectable if any draw layer is visible

1859 
1860             // Shortcut: check copper layer visibility

1861             if( board()->IsLayerVisible( F_Cu ) && pad->IsOnLayer( F_Cu ) )
1862                 return true;
1863 
1864             if( board()->IsLayerVisible( B_Cu ) && pad->IsOnLayer( B_Cu ) )
1865                 return true;
1866 
1867             // Now check the non-copper layers

1868 
1869             bool draw_layer_visible = false;
1870 
1871             int pad_layers[KIGFX::VIEW::VIEW_MAX_LAYERS], pad_layers_count;
1872             pad->ViewGetLayers( pad_layers, pad_layers_count );
1873 
1874             for( int i = 0; i < pad_layers_count; ++i )
1875             {
1876                 // NOTE: Only checking the regular layers (not GAL meta-layers)

1877                 if( ( ( pad_layers[i] < PCB_LAYER_ID_COUNT ) &&
1878                       board()->IsLayerVisible( static_cast<PCB_LAYER_ID>( pad_layers[i] ) ) ) )
1879                 {
1880                     draw_layer_visible = true;
1881                 }
1882             }
1883 
1884             return draw_layer_visible;
1885         }
1886 
1887         break;
1888     }
1889 
1890 
1891     case PCB_MARKER_T:  // Always selectable

1892         return true;
1893 
1894     // These are not selectable

1895     case NOT_USED:
1896     case TYPE_NOT_INIT:
1897         return false;
1898 
1899     default:    // Suppress warnings

1900         break;
1901     }
1902 
1903     // All other items are selected only if the layer on which they exist is visible

1904     return board()->IsLayerVisible( aItem->GetLayer() )
1905             && aItem->ViewGetLOD( aItem->GetLayer(), view() ) < view()->GetScale();
1906 }

2123 // The general idea here is that if the user clicks directly on a small item inside a larger

2124 // one, then they want the small item.  The quintessential case of this is clicking on a pad

2125 // within a footprint, but we also apply it for text within a footprint, footprints within

2126 // larger footprints, and vias within either larger pads or longer tracks.

2128 // These "guesses" presume there is area within the larger item to click in to select it.  If

2129 // an item is mostly covered by smaller items within it, then the guesses are inappropriate as

2130 // there might not be any area left to click to select the larger item.  In this case we must

2131 // leave the items in the collector and bring up a Selection Clarification menu.

2134 // smaller footprints mostly covering a larger footprint.

2136 void SELECTION_TOOL::GuessSelectionCandidates( GENERAL_COLLECTOR& aCollector,
2137                                                const VECTOR2I& aWhere ) const

2141     std::set<BOARD_ITEM*> forced;

2144     // footprints which are below this percentage of the largest footprint will be considered

2145     // for selection; all others will not
