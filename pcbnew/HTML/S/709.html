 266             bds.m_UseConnectedTrackWidth = false;
 267             m_frame.GetToolManager()->RunAction( ACT_CustomTrackWidth, true );
 268         }
 269         else if( id == ID_POPUP_PCB_SELECT_AUTO_WIDTH )
 270         {
 271             bds.UseCustomTrackViaSize( false );
 272             bds.m_UseConnectedTrackWidth = true;
 273         }
 274         else if( id == ID_POPUP_PCB_SELECT_USE_NETCLASS_VALUES )
 275         {
 276             bds.UseCustomTrackViaSize( false );
 277             bds.m_UseConnectedTrackWidth = false;
 278             bds.SetViaSizeIndex( 0 );
 279             bds.SetTrackWidthIndex( 0 );
 280         }

 282         {
 283             bds.UseCustomTrackViaSize( false );
 284             bds.m_UseConnectedTrackWidth = false;
 285             bds.SetViaSizeIndex( id - ID_POPUP_PCB_SELECT_VIASIZE1 );
 286         }
 287         else if( id >= ID_POPUP_PCB_SELECT_WIDTH1 && id <= ID_POPUP_PCB_SELECT_WIDTH16 )
 288         {
 289             bds.UseCustomTrackViaSize( false );
 290             bds.m_UseConnectedTrackWidth = false;
 291             bds.SetTrackWidthIndex( id - ID_POPUP_PCB_SELECT_WIDTH1 );
 292         }
 293 
 294         return OPT_TOOL_EVENT( PCB_ACTIONS::trackViaSizeChanged.MakeEvent() );
 295     }

 557     switch( aFlags & VIA_ACTION_FLAGS::VIA_MASK )

 643 int ROUTER_TOOL::onViaCommand( const TOOL_EVENT& aEvent )
 644 {
 645     if( !IsToolActive() )
 646         return 0;
 647 
 648     // First see if this is one of the switch layer commands

 649     LSEQ layers       = LSET( board()->GetEnabledLayers() & LSET::AllCuMask() ).Seq();
 650     int  currentLayer = m_router->GetCurrentLayer();
 651 
 652     PCB_LAYER_ID targetLayer = UNDEFINED_LAYER;
 653 
 654     if( aEvent.IsAction( &PCB_ACTIONS::layerNext ) )
 655     {
 656         size_t idx = 0;
 657 
 658         for( size_t i = 0; i < layers.size(); i++ )
 659         {
 660             if( layers[i] == currentLayer )
 661             {
 662                 idx = i;
 663                 break;
 664             }
 665         }
 666 
 667         idx         = ( idx + 1 ) % layers.size();
 668         targetLayer = layers[idx];
 669     }
 670     else if( aEvent.IsAction( &PCB_ACTIONS::layerPrev ) )
 671     {
 672         size_t idx = 0;
 673 
 674         for( size_t i = 0; i < layers.size(); i++ )
 675         {
 676             if( layers[i] == currentLayer )
 677             {
 678                 idx = i;
 679                 break;
 680             }
 681         }
 682 
 683         idx         = ( idx > 0 ) ? ( idx - 1 ) : ( layers.size() - 1 );
 684         targetLayer = layers[idx];
 685     }
 686     else
 687     {
 688         targetLayer = getTargetLayerFromEvent( aEvent );
 689 
 690         if( targetLayer != UNDEFINED_LAYER )
 691         {
 692             if( targetLayer == currentLayer )
 693                 return 0;
 694         }
 695     }
 696 
 697     BOARD_DESIGN_SETTINGS& bds        = board()->GetDesignSettings();
 698     const int              layerCount = bds.GetCopperLayerCount();
 699 
 700     PCB_LAYER_ID pairTop    = frame()->GetScreen()->m_Route_Layer_TOP;
 701     PCB_LAYER_ID pairBottom = frame()->GetScreen()->m_Route_Layer_BOTTOM;
 702 
 703     PNS::SIZES_SETTINGS sizes = m_router->Sizes();
 704 
 705     VIATYPE viaType     = VIATYPE::THROUGH;
 706     bool    selectLayer = false;
 707 
 708     // Otherwise it is one of the router-specific via commands

 709     if( targetLayer == UNDEFINED_LAYER )
 710     {
 711         const int actViaFlags = aEvent.Parameter<intptr_t>();
 712         selectLayer           = actViaFlags & VIA_ACTION_FLAGS::SELECT_LAYER;
 713 
 714         viaType = getViaTypeFromFlags( actViaFlags );
 715 
 716         // ask the user for a target layer

 717         if( selectLayer )
 718         {
 719             wxPoint dlgPosition = wxGetMousePosition();
 720 
 721             targetLayer = frame()->SelectLayer(
 722                     static_cast<PCB_LAYER_ID>( currentLayer ), LSET::AllNonCuMask(), dlgPosition );
 723 
 724             // Reset the cursor to the position where the event occured

 725             controls()->SetCursorPosition( aEvent.HasPosition() ? aEvent.Position() : dlgPosition );
 726         }
 727     }
 728 
 729     // fixme: P&S supports more than one fixed layer pair. Update the dialog?

 730     sizes.ClearLayerPairs();
 731 
 732     if( !m_router->IsPlacingVia() )
 733     {
 734         // Cannot place microvias or blind vias if not allowed (obvious)

 735         if( ( viaType == VIATYPE::BLIND_BURIED ) && ( !bds.m_BlindBuriedViaAllowed ) )
 736         {
 737             DisplayError( frame(),
 738                     _( "Blind/buried vias have to be enabled in Board Setup > Design Rules > Constraints." ) );
 739             return false;
 740         }
 741 
 742         if( ( viaType == VIATYPE::MICROVIA ) && ( !bds.m_MicroViasAllowed ) )
 743         {
 744             DisplayError( frame(),
 745                     _( "Microvias have to be enabled in Board Setup > Design Rules > Constraints." ) );
 746             return false;
 747         }
 748 
 749         // Can only place through vias on 2-layer boards

 750         if( ( viaType != VIATYPE::THROUGH ) && ( layerCount <= 2 ) )
 751         {
 752             DisplayError( frame(), _( "Only through vias are allowed on 2 layer boards." ) );
 753             return false;
 754         }
 755 
 756         // Can only place microvias if we're on an outer layer, or directly adjacent to one

 757         if( ( viaType == VIATYPE::MICROVIA ) && ( currentLayer > In1_Cu )
 758                 && ( currentLayer < layerCount - 2 ) )
 759         {
 760             DisplayError( frame(), _( "Microvias can be placed only between the outer layers "
 761                                       "(F.Cu/B.Cu) and the ones directly adjacent to them." ) );
 762             return false;
 763         }
 764     }
 765 
 766     // Convert blind/buried via to a through hole one, if it goes through all layers

 767     if( viaType == VIATYPE::BLIND_BURIED
 768             && ( ( targetLayer == B_Cu && currentLayer == F_Cu )
 769                        || ( targetLayer == F_Cu && currentLayer == B_Cu ) ) )
 770     {
 771         viaType = VIATYPE::THROUGH;
 772     }
 773 
 774     switch( viaType )
 775     {
 776     case VIATYPE::THROUGH:
 777         sizes.SetViaDiameter( bds.GetCurrentViaSize() );
 778         sizes.SetViaDrill( bds.GetCurrentViaDrill() );
 779 
 780         if( targetLayer != UNDEFINED_LAYER )
 781         {
 782             // go from the current layer to the chosen layer

 783             sizes.AddLayerPair( currentLayer, targetLayer );
 784         }
 785         else
 786         {
 787             // use the default layer pair

 788             sizes.AddLayerPair( pairTop, pairBottom );
 789         }
 790         break;
 791 
 792     case VIATYPE::MICROVIA:
 793         sizes.SetViaDiameter( bds.GetCurrentMicroViaSize() );
 794         sizes.SetViaDrill( bds.GetCurrentMicroViaDrill() );
 795 
 796         wxASSERT_MSG( !selectLayer,
 797                 "Unexpected select layer for microvia (microvia layers are implicit)" );
 798 
 799         if( currentLayer == F_Cu || currentLayer == In1_Cu )
 800         {
 801             // front-side microvia

 802             sizes.AddLayerPair( F_Cu, In1_Cu );
 803         }
 804         else if( currentLayer == B_Cu || currentLayer == layerCount - 2 )
 805         {
 806             // back-side microvia

 807             sizes.AddLayerPair( B_Cu, layerCount - 2 );
 808         }
 809         else
 810         {
 811             wxASSERT_MSG( false,
 812                     "Invalid layer pair for microvia (must be on or adjacent to an outer layer)" );
 813         }
 814         break;
 815 
 816     case VIATYPE::BLIND_BURIED:
 817         sizes.SetViaDiameter( bds.GetCurrentViaSize() );
 818         sizes.SetViaDrill( bds.GetCurrentViaDrill() );
 819 
 820         if( targetLayer != UNDEFINED_LAYER )
 821         {
 822             // go directly to the user specified layer

 823             sizes.AddLayerPair( currentLayer, targetLayer );
 824         }
 825         else
 826         {
 827             if( currentLayer == pairTop || currentLayer == pairBottom )
 828             {
 829                 // the current layer is on the defined layer pair,

 830                 // swap to the other side

 831                 sizes.AddLayerPair( pairTop, pairBottom );
 832             }
 833             else
 834             {
 835                 // the current layer is not part of the current layer pair,

 836                 // so fallback and swap to the top layer of the pair by default

 837                 sizes.AddLayerPair( pairTop, currentLayer );
 838             }
 839         }
 840         break;
 841 
 842     default:
 843         wxASSERT( false );
 844         break;
 845     }
 846 
 847     sizes.SetViaType( viaType );
 848 
 849     m_router->UpdateSizes( sizes );
 850     m_router->ToggleViaPlacement();
 851 
 852     if( m_router->RoutingInProgress() )
 853         updateEndItem( aEvent );
 854     else
 855         updateStartItem( aEvent );
 856 
 857     m_router->Move( m_endSnapPoint, m_endItem );        // refresh

 858 
 859     return 0;
 860 }

 931 void ROUTER_TOOL::performRouting()
 932 {
 933     if( !prepareInteractive() )
 934         return;
 935 
 936     while( TOOL_EVENT* evt = Wait() )
 937     {
 938         frame()->GetCanvas()->SetCurrentCursor( wxCURSOR_PENCIL );
 939 
 940         // Don't crash if we missed an operation that cancelled routing.

 941         if( !m_router->RoutingInProgress() )
 942         {
 943             if( evt->IsCancelInteractive() )
 944                 m_cancelled = true;
 945 
 946             break;
 947         }
 948 
 949         handleCommonEvents( *evt );
 950 
 951         if( evt->IsMotion() )
 952         {
 953             m_router->SetOrthoMode( evt->Modifier( MD_CTRL ) );
 954             updateEndItem( *evt );
 955             m_router->Move( m_endSnapPoint, m_endItem );
 956         }
 957         else if( evt->IsAction( &ACT_UndoLastSegment ) )
 958         {
 959             m_router->UndoLastSegment();
 960             updateEndItem( *evt );
 961             m_router->Move( m_endSnapPoint, m_endItem );
 962         }
 963         else if( evt->IsClick( BUT_LEFT ) || evt->IsAction( &PCB_ACTIONS::routeSingleTrack ) )
 964         {
 965             updateEndItem( *evt );
 966             bool needLayerSwitch = m_router->IsPlacingVia();
 967             bool forceFinish = evt->Modifier( MD_SHIFT );
 968 
 969             if( m_router->FixRoute( m_endSnapPoint, m_endItem, forceFinish ) )
 970             {
 971                 break;
 972             }
 973 
 974             if( needLayerSwitch )
 975                 switchLayerOnViaPlacement();
 976 
 977             // Synchronize the indicated layer

 978             frame()->SetActiveLayer( ToLAYER_ID( m_router->GetCurrentLayer() ) );
 979             updateEndItem( *evt );
 980             m_router->Move( m_endSnapPoint, m_endItem );
 981             m_startItem = nullptr;
 982         }
 983         else if( evt->IsAction( &ACT_SwitchRounding ) )
 984         {
 985             m_router->ToggleRounded();
 986             updateEndItem( *evt );
 987             m_router->Move( m_endSnapPoint, m_endItem );        // refresh

 988         }
 989         else if( evt->IsAction( &ACT_SwitchPosture ) )
 990         {
 991             m_router->FlipPosture();
 992             updateEndItem( *evt );
 993             m_router->Move( m_endSnapPoint, m_endItem );        // refresh

 994         }
 995         else if( evt->IsAction( &PCB_ACTIONS::layerChanged ) )
 996         {
 997             m_router->SwitchLayer( frame()->GetActiveLayer() );
 998             updateEndItem( *evt );
 999             m_router->Move( m_endSnapPoint, m_endItem );        // refresh

1000         }
1001         else if( evt->IsAction( &ACT_EndTrack ) || evt->IsDblClick( BUT_LEFT )  )
1002         {
1003             // Stop current routing:

1004             m_router->FixRoute( m_endSnapPoint, m_endItem, true );
1005             break;
1006         }
1007         else if( evt->IsCancelInteractive() || evt->IsActivate()
1008                  || evt->IsUndoRedo()
1009                  || evt->IsAction( &PCB_ACTIONS::routerInlineDrag ) )
1010         {
1011             if( evt->IsCancelInteractive() && !m_router->RoutingInProgress() )
1012                 m_cancelled = true;
1013 
1014             if( evt->IsActivate() && !evt->IsMoveTool() )
1015                 m_cancelled = true;
1016 
1017             break;
1018         }
1019         else if( evt->IsKeyPressed() )
1020         {
1021             // wxWidgets fails to correctly translate shifted keycodes on the wxEVT_CHAR_HOOK

1022             // event so we need to process the wxEVT_CHAR event that will follow as long as we

1023             // pass the event.

1024             evt->SetPassEvent();
1025         }
1026     }
1027 
1028     m_router->CommitRouting();
1029     m_router->StopRouting();
1030 
1031     finishInteractive();
1032 }

1089 int ROUTER_TOOL::MainLoop( const TOOL_EVENT& aEvent )
1090 {
1091     PNS::ROUTER_MODE mode = aEvent.Parameter<PNS::ROUTER_MODE>();
1092     PCB_EDIT_FRAME*  frame = getEditFrame<PCB_EDIT_FRAME>();
1093 
1094     // Deselect all items

1095     m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
1096 
1097     std::string tool = aEvent.GetCommandStr().get();
1098     frame->PushTool( tool );
1099     Activate();
1100 
1101     m_router->SetMode( mode );
1102 
1103     VIEW_CONTROLS* ctls = getViewControls();
1104     ctls->ShowCursor( true );
1105     ctls->ForceCursorPosition( false );
1106     m_cancelled = false;
1107     m_startSnapPoint = ctls->GetCursorPosition();
1108 
1109     std::unique_ptr<ROUTER_TOOL_MENU> ctxMenu( new ROUTER_TOOL_MENU( *frame, mode ) );
1110     SetContextMenu( ctxMenu.get() );
1111 
1112     // Prime the pump

1113     if( aEvent.HasPosition() )
1114         m_toolMgr->PrimeTool( m_startSnapPoint );
1115 
1116     // Main loop: keep receiving events

1117     while( TOOL_EVENT* evt = Wait() )
1118     {
1119         frame->GetCanvas()->SetCurrentCursor( wxCURSOR_PENCIL );
1120 
1121         if( evt->IsCancelInteractive() )
1122         {
1123             frame->PopTool( tool );
1124             break;
1125         }
1126         else if( evt->IsActivate() )
1127         {
1128             if( evt->IsMoveTool() )
1129             {
1130                 // leave ourselves on the stack so we come back after the move

1131                 break;
1132             }
1133             else
1134             {
1135                 frame->PopTool( tool );
1136                 break;
1137             }
1138         }
1139         else if( evt->Action() == TA_UNDO_REDO_PRE )
1140         {
1141             m_router->ClearWorld();
1142         }
1143         else if( evt->Action() == TA_UNDO_REDO_POST || evt->Action() == TA_MODEL_CHANGE )
1144         {
1145             m_router->SyncWorld();
1146         }
1147         else if( evt->IsMotion() )
1148         {
1149             updateStartItem( *evt );
1150         }
1151         else if( evt->IsAction( &PCB_ACTIONS::dragFreeAngle ) )
1152         {
1153             updateStartItem( *evt, true );
1154             performDragging( PNS::DM_ANY | PNS::DM_FREE_ANGLE );
1155         }
1156         else if( evt->IsAction( &PCB_ACTIONS::drag45Degree ) )
1157         {
1158             updateStartItem( *evt, true );
1159             performDragging( PNS::DM_ANY );
1160         }
1161         else if( evt->IsAction( &PCB_ACTIONS::breakTrack ) )
1162         {
1163             updateStartItem( *evt, true );
1164             breakTrack( );
1165         }
1166         else if( evt->IsClick( BUT_LEFT )
1167               || evt->IsAction( &PCB_ACTIONS::routeSingleTrack )
1168               || evt->IsAction( &PCB_ACTIONS::routeDiffPair ) )
1169         {
1170             if( evt->IsAction( &PCB_ACTIONS::routeSingleTrack )
1171              || evt->IsAction( &PCB_ACTIONS::routeDiffPair ) )
1172             {
1173                 mode = evt->Parameter<PNS::ROUTER_MODE>();
1174             }
1175 
1176             updateStartItem( *evt );
1177 
1178             if( evt->HasPosition() )
1179             {
1180                 if( evt->Modifier( MD_CTRL ) )
1181                     performDragging( PNS::DM_ANY );
1182                 else
1183                     performRouting();
1184             }
1185         }
1186         else if( evt->IsAction( &ACT_PlaceThroughVia ) )
1187         {
1188             m_toolMgr->RunAction( PCB_ACTIONS::layerToggle, true );
1189         }
1190         else if( evt->IsAction( &PCB_ACTIONS::layerChanged ) )
1191         {
1192             m_router->SwitchLayer( frame->GetActiveLayer() );
1193             updateStartItem( *evt );
1194         }
1195         else if( evt->IsKeyPressed() )
1196         {
1197             // wxWidgets fails to correctly translate shifted keycodes on the wxEVT_CHAR_HOOK

1198             // event so we need to process the wxEVT_CHAR event that will follow as long as we

1199             // pass the event.

1200             evt->SetPassEvent();
1201         }
1202 
1203         if( m_cancelled )
1204         {
1205             frame->PopTool( tool );
1206             break;
1207         }
1208     }
1209 
1210     SetContextMenu( nullptr );
1211 
1212     // Store routing settings till the next invocation

1213     m_savedSizes = m_router->Sizes();
1214 
1215     return 0;
1216 }

1219 void ROUTER_TOOL::performDragging( int aMode )
1220 {
1221     VIEW_CONTROLS* ctls = getViewControls();
1222 
1223     if( m_startItem && m_startItem->IsLocked() )
1224     {
1225         KIDIALOG dlg( frame(), _( "The selected item is locked." ), _( "Confirmation" ),
1226                       wxOK | wxCANCEL | wxICON_WARNING );
1227         dlg.SetOKLabel( _( "Drag Anyway" ) );
1228         dlg.DoNotShowCheckbox( __FILE__, __LINE__ );
1229 
1230         if( dlg.ShowModal() == wxID_CANCEL )
1231             return;
1232     }
1233 
1234     bool dragStarted = m_router->StartDragging( m_startSnapPoint, m_startItem, aMode );
1235 
1236     if( !dragStarted )
1237         return;
1238 
1239     if( m_startItem && m_startItem->Net() >= 0 )
1240         highlightNet( true, m_startItem->Net() );
1241 
1242     ctls->SetAutoPan( true );
1243     m_gridHelper->SetAuxAxes( true, m_startSnapPoint );
1244     frame()->UndoRedoBlock( true );
1245 
1246     while( TOOL_EVENT* evt = Wait() )
1247     {
1248         ctls->ForceCursorPosition( false );
1249

1287 void ROUTER_TOOL::NeighboringSegmentFilter( const VECTOR2I& aPt, GENERAL_COLLECTOR& aCollector )
1288 {
1289     /*

1290      * If the collection contains a trivial line corner (two connected segments)

1291      * or a non-fanout-via (a via with no more than two connected segments), then

1292      * trim the collection down to a single item (which one won't matter since

1293      * they're all connected).

1294      */
1295 
1296     // First make sure we've got something that *might* match.

1297     int vias = aCollector.CountType( PCB_VIA_T );
1298     int traces = aCollector.CountType( PCB_TRACE_T );
1299     int arcs = aCollector.CountType( PCB_ARC_T );
1300 
1301     if( arcs > 0 || vias > 1 || traces > 2 || vias + traces < 1 )
1302         return;
1303 
1304     // Fetch first TRACK (via or trace) as our reference

1305     TRACK* reference = nullptr;
1306 
1307     for( int i = 0; !reference && i < aCollector.GetCount(); i++ )
1308         reference = dynamic_cast<TRACK*>( aCollector[i] );
1309 
1310     int refNet = reference->GetNetCode();
1311 
1312     wxPoint refPoint( aPt.x, aPt.y );
1313     STATUS_FLAGS flags = reference->IsPointOnEnds( refPoint, -1 );
1314 
1315     if( flags & STARTPOINT )
1316         refPoint = reference->GetStart();
1317     else if( flags & ENDPOINT )
1318         refPoint = reference->GetEnd();
1319 
1320     // Check all items to ensure that any TRACKs are co-terminus with the reference and on

1321     // the same net.

1322     for( int i = 0; i < aCollector.GetCount(); i++ )
1323     {
1324         TRACK* neighbor = dynamic_cast<TRACK*>( aCollector[i] );
1325 
1326         if( neighbor && neighbor != reference )
1327         {
1328             if( neighbor->GetNetCode() != refNet )
1329                 return;
1330 
1331             if( neighbor->GetStart() != refPoint && neighbor->GetEnd() != refPoint )
1332                 return;
1333         }
1334     }
1335 
1336     // Selection meets criteria; trim it to the reference item.

1337     aCollector.Empty();
1338     aCollector.Append( reference );
1339 }

1368 
1369     const BOARD_ITEM* item = static_cast<const BOARD_ITEM*>( selection.Front() );
1370

1372         return 0;

1377     m_router->SyncWorld();
1378     m_startItem = nullptr;
1379

1381     PNS::ITEM_SET itemsToDrag;
1382

1384     {
1385         const auto mod = static_cast<const MODULE*>(item);
1386         for ( const auto p : mod->Pads() )
1387         {
1388             auto solid = m_router->GetWorld()->FindItemByParent( p );
1389 
1390             if( solid )
1391                 itemsToDrag.Add( solid );
1392         }
1393     }
1394     else
1395     {
1396         startItem = m_router->GetWorld()->FindItemByParent( static_cast<const BOARD_CONNECTED_ITEM*>( item ) );
1397

1399             itemsToDrag.Add( startItem );
1400     }
1401 
1402     if( startItem && startItem->IsLocked() )
1403     {
1404         KIDIALOG dlg( frame(), _( "The selected item is locked." ), _( "Confirmation" ),
1405                       wxOK | wxCANCEL | wxICON_WARNING );
1406         dlg.SetOKLabel( _( "Drag Anyway" ) );
1407         dlg.DoNotShowCheckbox( __FILE__, __LINE__ );
1408 
1409         if( dlg.ShowModal() == wxID_CANCEL )
1410             return 0;
1411     }
1412 
1413     VECTOR2I p0 = controls()->GetCursorPosition( false );
1414     VECTOR2I p = p0;
1415 
1416     if( startItem )
1417         p = snapToItem( true, startItem, p0 );
1418 
1419     int dragMode = aEvent.Parameter<int64_t> ();
1420 
1421     bool dragStarted = m_router->StartDragging( p, itemsToDrag, dragMode );
1422 
1423     if( !dragStarted )
1424         return 0;
1425 
1426     m_gridHelper->SetAuxAxes( true, p );
1427     controls()->ShowCursor( true );

1429     controls()->SetAutoPan( true );

1435

1437         {
1438             break;

1440         else if( evt->IsMotion() || evt->IsDrag( BUT_LEFT ) )
1441         {
1442             updateEndItem( *evt );
1443             m_router->Move( m_endSnapPoint, m_endItem );
1444         }
1445         else if( evt->IsMouseUp( BUT_LEFT ) || evt->IsClick( BUT_LEFT ) )
1446         {
1447             updateEndItem( *evt );
1448             m_router->FixRoute( m_endSnapPoint, m_endItem );
1449             break;
1450         }
1451     }
1452 
1453     if( m_router->RoutingInProgress() )
1454         m_router->StopRouting();
1455 
1456     m_gridHelper->SetAuxAxes( false );
1457     controls()->SetAutoPan( false );
1458     controls()->ForceCursorPosition( false );
1459     frame()->UndoRedoBlock( false );
1460 
1461     return 0;
