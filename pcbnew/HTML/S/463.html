  42 /**

  43  * a Tag parser, to extract tagged data in html text.

  44  * this tag handler extract a url link, givent by tag "A"

  45  * like:

  46  * "<a href="/KiCad/Valves.pretty" itemprop="name codeRepository"> Valves.pretty</a>"

  47  * the tag is "a"

  48  * and the link is the parameter given by "href"

  49  */
  50 class LINK_TAGHANDLER : public wxHtmlTagHandler
  51 {
  52     HTML_LINK_PARSER* m_Parser;
  53 
  54 public:
  55     LINK_TAGHANDLER() : m_Parser( NULL )
  56     {
  57     }
  58 
  59     LINK_TAGHANDLER( HTML_LINK_PARSER* aParser ) : m_Parser( aParser )
  60     {
  61     }
  62 
  63     wxString GetSupportedTags() override
  64     {
  65         return "A";
  66     }
  67 
  68     bool HandleTag(const wxHtmlTag& tag) override;
  69 };
  70 
  71 /**

  72  * The engine to parse a html text and extract useful data

  73  * Here, the useful data are url links

  74  */
  75 class HTML_LINK_PARSER : public wxHtmlParser
  76 {
  77     const wxString& m_src;          // the html text to parse

  78     wxArrayString& stringUrls;      // the strings extracted from html text

  79     bool (*m_filter)( const wxString& aData );  // a callback function to filter strings

  80 
  81 public:
  82 
  83     HTML_LINK_PARSER( const wxString& aSrc, wxArrayString& aUrls );
  84 
  85     /**

  86      * Parse the html text and store links in stringUrls

  87      * Stored links can be filtered if aFilter is non NULL

  88      * @param aFilter a filtering function ( bool aFilter( const wxString& aData ) )

  89      * which return true if the text aData must be stored.

  90      */
  91     void ParseLinks( bool (*aFilter)( const wxString& aData ) )
  92     {
  93         m_filter = aFilter;
  94         Parse(m_src);
  95     }
  96 
  97     // virtual pure from wxHtmlParser, do nothing here, but needed.

  98     void AddText( const wxString& aText ) override {}
  99 
 100     // Our "AddText" used to store selected text (the url link)

 101     void AddString( const wxString& aText );
 102 
 103     wxObject* GetProduct() override
 104     {
 105         return NULL;
 106     }
 107 };
