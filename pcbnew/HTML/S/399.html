  55 template<class T=int>
  56 class MINOPTMAX
  57 {
  58 public:
  59     T Min() const { assert( m_hasMin ); return m_min; };
  60     T Max() const { assert( m_hasMax ); return m_max; };
  61     T Opt() const { assert( m_hasOpt ); return m_opt; };
  62 
  63     bool HasMin() const { return m_hasMin; }
  64     bool HasMax() const { return m_hasMax; }
  65     bool HasOpt() const { return m_hasOpt; }
  66 
  67     void SetMin( T v ) { m_min = v; m_hasMin = true; }
  68     void SetMax( T v ) { m_max = v; m_hasMax = true; }
  69     void SetOpt( T v ) { m_opt = v; m_hasOpt = true; }
  70 
  71 private:
  72     T m_min;
  73     T m_opt;
  74     T m_max;
  75     bool m_hasMin = false;
  76     bool m_hasOpt = false;
  77     bool m_hasMax = false;
  78 };
