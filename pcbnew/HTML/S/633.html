 159 void LENGTH_TUNER_TOOL::performTuning()
 160 {
 161     if( m_startItem )
 162     {
 163         frame()->SetActiveLayer( ToLAYER_ID ( m_startItem->Layers().Start() ) );
 164 
 165         if( m_startItem->Net() >= 0 )
 166             highlightNet( true, m_startItem->Net() );
 167     }
 168 
 169     controls()->ForceCursorPosition( false );
 170     controls()->SetAutoPan( true );
 171 
 172     if( !m_router->StartRouting( m_startSnapPoint, m_startItem, 0 ) )
 173     {
 174         wxMessageBox( m_router->FailureReason(), _( "Error" ) );
 175         highlightNet( false );
 176         return;
 177     }
 178 
 179     auto placer = static_cast<PNS::MEANDER_PLACER_BASE*>( m_router->Placer() );
 180 
 181     placer->UpdateSettings( m_savedMeanderSettings );
 182 
 183     VECTOR2I end( m_startSnapPoint );
 184 
 185     // Create an instance of PNS_TUNE_STATUS_POPUP.

 186     PNS_TUNE_STATUS_POPUP statusPopup( frame() );
 187     statusPopup.Popup();
 188 
 189     m_router->Move( end, NULL );
 190     updateStatusPopup( statusPopup );
 191 
 192     while( TOOL_EVENT* evt = Wait() )
 193     {
 194         frame()->GetCanvas()->SetCurrentCursor( wxCURSOR_ARROW );
 195 
 196         if( evt->IsCancelInteractive() || evt->IsActivate() )
 197             break;
 198         else if( evt->IsMotion() )
 199         {
 200             end = evt->Position();
 201             m_router->Move( end, NULL );
 202             updateStatusPopup( statusPopup );
 203         }
 204         else if( evt->IsClick( BUT_LEFT ) )
 205         {
 206             if( m_router->FixRoute( evt->Position(), NULL ) )
 207                 break;
 208         }
 209         else if( evt->IsAction( &ACT_EndTuning ) )
 210         {
 211             if( m_router->FixRoute( end, NULL ) )
 212                 break;
 213         }
 214         else if( evt->IsAction( &ACT_AmplDecrease ) )
 215         {
 216             placer->AmplitudeStep( -1 );
 217             m_router->Move( end, NULL );
 218             updateStatusPopup( statusPopup );
 219         }
 220         else if( evt->IsAction( &ACT_AmplIncrease ) )
 221         {
 222             placer->AmplitudeStep( 1 );
 223             m_router->Move( end, NULL );
 224             updateStatusPopup( statusPopup );
 225         }
 226         else if(evt->IsAction( &ACT_SpacingDecrease ) )
 227         {
 228             placer->SpacingStep( -1 );
 229             m_router->Move( end, NULL );
 230             updateStatusPopup( statusPopup );
 231         }
 232         else if( evt->IsAction( &ACT_SpacingIncrease ) )
 233         {
 234             placer->SpacingStep( 1 );
 235             m_router->Move( end, NULL );
 236             updateStatusPopup( statusPopup );
 237         }
 238         else if( evt->IsAction( &ACT_Settings ) )
 239         {
 240             statusPopup.Hide();
 241             TOOL_EVENT dummy;
 242             meanderSettingsDialog( dummy );
 243             statusPopup.Show();
 244         }
 245     }
 246 
 247     m_router->StopRouting();
 248     controls()->SetAutoPan( false );
 249     controls()->ForceCursorPosition( false );
 250     highlightNet( false );
 251 }

 262 int LENGTH_TUNER_TOOL::MainLoop( const TOOL_EVENT& aEvent )
 263 {
 264     // Deselect all items

 265     m_toolMgr->RunAction( PCB_ACTIONS::selectionClear, true );
 266 
 267     std::string tool = aEvent.GetCommandStr().get();
 268     frame()->PushTool( tool );
 269     Activate();
 270 
 271     m_router->SetMode( aEvent.Parameter<PNS::ROUTER_MODE>() );
 272 
 273     controls()->SetSnapping( true );
 274     controls()->ShowCursor( true );
 275     frame()->UndoRedoBlock( true );
 276 
 277     std::unique_ptr<TUNER_TOOL_MENU> ctxMenu( new TUNER_TOOL_MENU );
 278     SetContextMenu( ctxMenu.get() );
 279 
 280     // Main loop: keep receiving events

 281     while( TOOL_EVENT* evt = Wait() )
 282     {
 283         frame()->GetCanvas()->SetCurrentCursor( wxCURSOR_ARROW );
 284 
 285         if( evt->IsCancelInteractive() || evt->IsActivate() )
 286         {
 287             break; // Finish

 288         }
 289         else if( evt->IsMotion() )
 290         {
 291             updateStartItem( *evt );
 292         }
 293         else if( evt->IsClick( BUT_LEFT ) || evt->IsAction( &ACT_StartTuning ) )
 294         {
 295             updateStartItem( *evt );
 296             performTuning();
 297         }
 298         else if( evt->IsAction( &ACT_Settings ) )
 299         {
 300             TOOL_EVENT dummy;
 301             meanderSettingsDialog( dummy );
 302         }
 303     }
 304 
 305     frame()->UndoRedoBlock( false );
 306 
 307     // Store routing settings till the next invocation

 308     m_savedSizes = m_router->Sizes();
 309 
 310     frame()->PopTool( tool );
 311     return 0;
 312 }

 314 int LENGTH_TUNER_TOOL::meanderSettingsDialog( const TOOL_EVENT& aEvent )
 315 {
 316     PNS::MEANDER_PLACER_BASE* placer = static_cast<PNS::MEANDER_PLACER_BASE*>( m_router->Placer() );
 317 
 318     PNS::MEANDER_SETTINGS settings = placer ? placer->MeanderSettings() : m_savedMeanderSettings;
 319     DIALOG_PNS_LENGTH_TUNING_SETTINGS settingsDlg( frame(), settings, m_router->Mode() );
 320 
 321     if( settingsDlg.ShowModal() )
 322     {
 323         if( placer )
 324             placer->UpdateSettings( settings );
 325 
 326         m_savedMeanderSettings = settings;
 327     }
 328 
 329     return 0;
 330 }
