  77 #define CTL_STD_LAYER_NAMES         (1 << 0)    ///< Use English Standard layer names

  78 #define CTL_OMIT_NETS               (1 << 1)    ///< Omit pads net names (useless in library)

  79 #define CTL_OMIT_TSTAMPS            (1 << 2)    ///< Omit component time stamp (useless in library)

  80 #define CTL_OMIT_INITIAL_COMMENTS   (1 << 3)    ///< omit MODULE initial comments

  81 #define CTL_OMIT_PATH               (1 << 4)    ///< Omit component sheet time stamp (useless in library)

  82 #define CTL_OMIT_AT                 (1 << 5)    ///< Omit position and rotation

 100 /**

 101  * PCB_IO

 102  * is a PLUGIN derivation for saving and loading Pcbnew s-expression formatted files.

 103  *

 104  * @note This class is not thread safe, but it is re-entrant multiple times in sequence.

 105  */
 106 class PCB_IO : public PLUGIN
 107 {
 108     friend class FP_CACHE;
 109 
 110 public:
 111 
 112     //-----<PLUGIN API>---------------------------------------------------------

 113 
 114     const wxString PluginName() const override
 115     {
 116         return wxT( "KiCad" );
 117     }
 118 
 119     const wxString GetFileExtension() const override
 120     {
 121         // Would have used wildcards_and_files_ext.cpp's KiCadPcbFileExtension,

 122         // but to be pure, a plugin should not assume that it will always be linked

 123         // with the core of the pcbnew code. (Might someday be a DLL/DSO.)  Besides,

 124         // file extension policy should be controlled by the plugin.

 125         return wxT( "kicad_pcb" );
 126     }
 127 
 128     virtual void Save( const wxString& aFileName, BOARD* aBoard,
 129                const PROPERTIES* aProperties = NULL ) override;
 130 
 131     BOARD* Load( const wxString& aFileName, BOARD* aAppendToMe,
 132                  const PROPERTIES* aProperties = NULL ) override;
 133 
 134     void FootprintEnumerate( wxArrayString& aFootprintNames, const wxString& aLibraryPath,
 135                              bool aBestEfforts, const PROPERTIES* aProperties = NULL ) override;
 136 
 137     const MODULE* GetEnumeratedFootprint( const wxString& aLibraryPath,
 138                                           const wxString& aFootprintName,
 139                                           const PROPERTIES* aProperties = NULL ) override;
 140 
 141     bool FootprintExists( const wxString& aLibraryPath, const wxString& aFootprintName,
 142                           const PROPERTIES* aProperties = NULL ) override;
 143 
 144     MODULE* FootprintLoad( const wxString& aLibraryPath, const wxString& aFootprintName,
 145                            const PROPERTIES* aProperties = NULL ) override;
 146 
 147     void FootprintSave( const wxString& aLibraryPath, const MODULE* aFootprint,
 148                         const PROPERTIES* aProperties = NULL ) override;
 149 
 150     void FootprintDelete( const wxString& aLibraryPath, const wxString& aFootprintName,
 151                           const PROPERTIES* aProperties = NULL ) override;
 152 
 153     long long GetLibraryTimestamp( const wxString& aLibraryPath ) const override;
 154 
 155     void FootprintLibCreate( const wxString& aLibraryPath, const PROPERTIES* aProperties = NULL) override;
 156 
 157     bool FootprintLibDelete( const wxString& aLibraryPath, const PROPERTIES* aProperties = NULL ) override;
 158 
 159     bool IsFootprintLibWritable( const wxString& aLibraryPath ) override;
 160 
 161     //-----</PLUGIN API>--------------------------------------------------------

 162 
 163     PCB_IO( int aControlFlags = CTL_FOR_BOARD );
 164 
 165     ~PCB_IO();
 166 
 167     /**

 168      * Function Format

 169      * outputs \a aItem to \a aFormatter in s-expression format.

 170      *

 171      * @param aItem A pointer the an #BOARD_ITEM object to format.

 172      * @param aNestLevel The indentation nest level.

 173      * @throw IO_ERROR on write error.

 174      */
 175     void Format( BOARD_ITEM* aItem, int aNestLevel = 0 ) const;
 176 
 177     std::string GetStringOutput( bool doClear )
 178     {
 179         std::string ret = m_sf.GetString();
 180         if( doClear )
 181             m_sf.Clear();
 182 
 183         return ret;
 184     }
 185 
 186     void SetOutputFormatter( OUTPUTFORMATTER* aFormatter ) { m_out = aFormatter; }
 187 
 188     BOARD_ITEM* Parse( const wxString& aClipboardSourceInput );
 189 
 190 protected:
 191 
 192     wxString        m_error;        ///< for throwing exceptions

 193     BOARD*          m_board;        ///< which BOARD, no ownership here

 194 
 195     const
 196     PROPERTIES*     m_props;        ///< passed via Save() or Load(), no ownership, may be NULL.

 197     FP_CACHE*       m_cache;        ///< Footprint library cache.

 198 
 199     LINE_READER*    m_reader;       ///< no ownership here.

 200     wxString        m_filename;     ///< for saves only, name is in m_reader for loads

 201 
 202     int             m_loading_format_version; ///< which #SEXPR_BOARD_FILE_VERSION should be Load()ed?

 203 
 204     STRING_FORMATTER    m_sf;
 205     OUTPUTFORMATTER*    m_out;      ///< output any Format()s to this, no ownership

 206     int                 m_ctl;
 207     PCB_PARSER*         m_parser;
 208     NETINFO_MAPPING*    m_mapping;  ///< mapping for net codes, so only not empty net codes

 209                                     ///< are stored with consecutive integers as net codes

 210 
 211     void validateCache( const wxString& aLibraryPath, bool checkModified = true );
 212 
 213     const MODULE* getFootprint( const wxString& aLibraryPath, const wxString& aFootprintName,
 214                   const PROPERTIES* aProperties, bool checkModified );
 215 
 216     void init( const PROPERTIES* aProperties );
 217 
 218     /// formats the board setup information

 219     void formatSetup( BOARD* aBoard, int aNestLevel = 0 ) const;
 220 
 221     /// formats the General section of the file

 222     void formatGeneral( BOARD* aBoard, int aNestLevel = 0 ) const;
 223 
 224     /// formats the board layer information

 225     void formatBoardLayers( BOARD* aBoard, int aNestLevel = 0 ) const;
 226 
 227     /// formats the Nets and Netclasses

 228     void formatNetInformation( BOARD* aBoard, int aNestLevel = 0 ) const;
 229 
 230     /// writes everything that comes before the board_items, like settings and layers etc

 231     void formatHeader( BOARD* aBoard, int aNestLevel = 0 ) const;
 232 
 233 private:
 234     void format( BOARD* aBoard, int aNestLevel = 0 ) const;
 235 
 236     void format( DIMENSION* aDimension, int aNestLevel = 0 ) const;
 237 
 238     void format( EDGE_MODULE* aModuleDrawing, int aNestLevel = 0 ) const;
 239 
 240     void format( DRAWSEGMENT* aSegment, int aNestLevel = 0 ) const;
 241 
 242     void format( PCB_TARGET* aTarget, int aNestLevel = 0 ) const;
 243 
 244     void format( MODULE* aModule, int aNestLevel = 0 ) const;
 245 
 246     void format( D_PAD* aPad, int aNestLevel = 0 ) const;
 247 
 248     void format( TEXTE_PCB* aText, int aNestLevel = 0 ) const;
 249 
 250     void format( TEXTE_MODULE* aText, int aNestLevel = 0 ) const;
 251 
 252     void format( TRACK* aTrack, int aNestLevel = 0 ) const;
 253 
 254     void format( ZONE_CONTAINER* aZone, int aNestLevel = 0 ) const;
 255 
 256     void formatLayer( const BOARD_ITEM* aItem ) const;
 257 
 258     void formatLayers( LSET aLayerMask, int aNestLevel = 0 ) const;
 259 };
