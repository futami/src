 132 /**

 133  * Traverser implementation that looks to find any and all "folder" libraries by looking for files

 134  * with a specific extension inside folders

 135  */
 136 class LIBRARY_TRAVERSER : public wxDirTraverser
 137 {
 138 public:
 139     LIBRARY_TRAVERSER( wxString aSearchExtension, wxString aInitialDir )
 140         : m_searchExtension( aSearchExtension ),
 141           m_currentDir( aInitialDir )
 142     {
 143     }
 144 
 145 
 146     virtual wxDirTraverseResult OnFile( const wxString& aFileName ) override
 147     {
 148         wxFileName file( aFileName );
 149         if( m_searchExtension.IsSameAs( file.GetExt(), false ) )
 150         {
 151             m_foundDirs.insert( { m_currentDir, 1 } );
 152         }
 153 
 154         return wxDIR_CONTINUE;
 155     }
 156 
 157 
 158     virtual wxDirTraverseResult OnOpenError( const wxString& aOpenErrorName ) override
 159     {
 160         m_failedDirs.insert( { aOpenErrorName, 1 } );
 161         return wxDIR_IGNORE;
 162     }
 163 
 164 
 165     bool HasDirectoryOpenFailures()
 166     {
 167         return m_failedDirs.size() > 0;
 168     }
 169 
 170 
 171     virtual wxDirTraverseResult OnDir( const wxString& aDirName ) override
 172     {
 173         m_currentDir = aDirName;
 174         return wxDIR_CONTINUE;
 175     }
 176 
 177 
 178     void GetPaths( wxArrayString& aPathArray )
 179     {
 180         for( auto foundDirsPair : m_foundDirs )
 181         {
 182             aPathArray.Add( foundDirsPair.first );
 183         }
 184     }
 185 
 186 
 187     void GetFailedPaths( wxArrayString& aPathArray )
 188     {
 189         for( auto failedDirsPair : m_failedDirs )
 190         {
 191             aPathArray.Add( failedDirsPair.first );
 192         }
 193     }
 194 
 195 private:
 196     wxString                          m_searchExtension;
 197     wxString                          m_currentDir;
 198     std::unordered_map<wxString, int> m_foundDirs;
 199     std::unordered_map<wxString, int> m_failedDirs;
 200 };

 309     /// handle specialized clipboard text, with leading "(fp_lib_table", OR

 310     /// spreadsheet formatted text.

 311     void paste_text( const wxString& cb_text ) override
 312     {
 313         FP_LIB_TABLE_GRID* tbl = (FP_LIB_TABLE_GRID*) m_grid->GetTable();
 314         size_t             ndx = cb_text.find( "(fp_lib_table" );
 315 
 316         if( ndx != std::string::npos )
 317         {
 318             // paste the FP_LIB_TABLE_ROWs of s-expression (fp_lib_table), starting

 319             // at column 0 regardless of current cursor column.

 320 
 321             STRING_LINE_READER  slr( TO_UTF8( cb_text ), "Clipboard" );
 322             LIB_TABLE_LEXER     lexer( &slr );
 323             FP_LIB_TABLE        tmp_tbl;
 324             bool                parsed = true;
 325 
 326             try
 327             {
 328                 tmp_tbl.Parse( &lexer );
 329             }
 330             catch( PARSE_ERROR& pe )
 331             {
 332                 DisplayError( m_dialog, pe.What() );
 333                 parsed = false;
 334             }
 335 
 336             if( parsed )
 337             {
 338                 // make sure the table is big enough...

 339                 if( tmp_tbl.GetCount() > (unsigned) tbl->GetNumberRows() )
 340                     tbl->AppendRows( tmp_tbl.GetCount() - tbl->GetNumberRows() );
 341 
 342                 for( unsigned i = 0;  i < tmp_tbl.GetCount();  ++i )
 343                     tbl->rows.replace( i, tmp_tbl.At( i ).clone() );
 344             }
 345 
 346             m_grid->AutoSizeColumns( false );
 347         }
 348         else
 349         {
 350             // paste spreadsheet formatted text.

 351             GRID_TRICKS::paste_text( cb_text );
 352 
 353             m_grid->AutoSizeColumns( false );
 354         }
 355     }

 501 bool PANEL_FP_LIB_TABLE::verifyTables()
 502 {
 503     for( FP_LIB_TABLE_GRID* model : { global_model(), project_model() } )
 504     {
 505         for( int r = 0; r < model->GetNumberRows(); )
 506         {
 507             wxString nick = model->GetValue( r, COL_NICKNAME ).Trim( false ).Trim();
 508             wxString uri  = model->GetValue( r, COL_URI ).Trim( false ).Trim();
 509             unsigned illegalCh = 0;
 510 
 511             if( !nick || !uri )
 512             {
 513                 // Delete the "empty" row, where empty means missing nick or uri.

 514                 // This also updates the UI which could be slow, but there should only be a few

 515                 // rows to delete, unless the user fell asleep on the Add Row

 516                 // button.

 517                 model->DeleteRows( r, 1 );
 518             }
 519             else if( ( illegalCh = LIB_ID::FindIllegalLibNicknameChar( nick, LIB_ID::ID_PCB ) ) )
 520             {
 521                 wxString msg = wxString::Format( _( "Illegal character '%c' in Nickname: \"%s\"" ),
 522                                                  illegalCh,
 523                                                  nick );
 524 
 525                 // show the tabbed panel holding the grid we have flunked:

 526                 if( model != cur_model() )
 527                     m_auinotebook->SetSelection( model == global_model() ? 0 : 1 );
 528 
 529                 m_cur_grid->MakeCellVisible( r, 0 );
 530                 m_cur_grid->SetGridCursor( r, 1 );
 531 
 532                 wxMessageDialog errdlg( this, msg, _( "No Colon in Nicknames" ) );
 533                 errdlg.ShowModal();
 534                 return false;
 535             }
 536             else
 537             {
 538                 // set the trimmed values back into the table so they get saved to disk.

 539                 model->SetValue( r, COL_NICKNAME, nick );
 540                 model->SetValue( r, COL_URI, uri );
 541                 ++r;        // this row was OK.

 542             }
 543         }
 544     }
 545 
 546     // check for duplicate nickNames, separately in each table.

 547     for( FP_LIB_TABLE_GRID* model : { global_model(), project_model() } )
 548     {
 549         for( int r1 = 0; r1 < model->GetNumberRows() - 1; ++r1 )
 550         {
 551             wxString nick1 = model->GetValue( r1, COL_NICKNAME );
 552 
 553             for( int r2 = r1 + 1; r2 < model->GetNumberRows(); ++r2 )
 554             {
 555                 wxString nick2 = model->GetValue( r2, COL_NICKNAME );
 556 
 557                 if( nick1 == nick2 )
 558                 {
 559                     wxString msg = wxString::Format( _( "Duplicate Nicknames \"%s\"." ), nick1 );
 560 
 561                     // show the tabbed panel holding the grid we have flunked:

 562                     if( model != cur_model() )
 563                         m_auinotebook->SetSelection( model == global_model() ? 0 : 1 );
 564 
 565                     // go to the lower of the two rows, it is technically the duplicate:

 566                     m_cur_grid->MakeCellVisible( r2, 0 );
 567                     m_cur_grid->SetGridCursor( r2, 1 );
 568 
 569                     wxMessageDialog errdlg( this, msg, _( "Please Delete or Modify One" ) );
 570                     errdlg.ShowModal();
 571                     return false;
 572                 }
 573             }
 574         }
 575     }
 576 
 577     return true;
 578 }

 608 void PANEL_FP_LIB_TABLE::deleteRowHandler( wxCommandEvent& event )
 609 {
 610     if( !m_cur_grid->CommitPendingChanges() )
 611         return;
 612 
 613     int curRow = m_cur_grid->GetGridCursorRow();
 614     int curCol = m_cur_grid->GetGridCursorCol();
 615 
 616     // In a wxGrid, collect rows that have a selected cell, or are selected

 617     // It is not so easy: it depends on the way the selection was made.

 618     // Here, we collect rows selected by clicking on a row label, and rows that contain any

 619     // previously-selected cells.

 620     // If no candidate, just delete the row with the grid cursor.

 621     wxArrayInt selectedRows = m_cur_grid->GetSelectedRows();
 622     wxGridCellCoordsArray cells = m_cur_grid->GetSelectedCells();
 623     wxGridCellCoordsArray blockTopLeft = m_cur_grid->GetSelectionBlockTopLeft();
 624     wxGridCellCoordsArray blockBotRight = m_cur_grid->GetSelectionBlockBottomRight();
 625 
 626     // Add all row having cell selected to list:

 627     for( unsigned ii = 0; ii < cells.GetCount(); ii++ )
 628         selectedRows.Add( cells[ii].GetRow() );
 629 
 630     // Handle block selection

 631     if( !blockTopLeft.IsEmpty() && !blockBotRight.IsEmpty() )
 632     {
 633         for( int i = blockTopLeft[0].GetRow(); i <= blockBotRight[0].GetRow(); ++i )
 634             selectedRows.Add( i );
 635     }
 636 
 637     // Use the row having the grid cursor only if we have no candidate:

 638     if( selectedRows.size() == 0 && m_cur_grid->GetGridCursorRow() >= 0 )
 639         selectedRows.Add( m_cur_grid->GetGridCursorRow() );
 640 
 641     if( selectedRows.size() == 0 )
 642     {
 643         wxBell();
 644         return;
 645     }
 646 
 647     std::sort( selectedRows.begin(), selectedRows.end() );
 648 
 649     // Remove selected rows (note: a row can be stored more than once in list)

 650     int last_row = -1;
 651 
 652     for( int ii = selectedRows.GetCount()-1; ii >= 0; ii-- )
 653     {
 654         int row = selectedRows[ii];
 655 
 656         if( row != last_row )
 657         {
 658             last_row = row;
 659             m_cur_grid->DeleteRows( row, 1 );
 660         }
 661     }
 662 
 663     if( m_cur_grid->GetNumberRows() > 0 && curRow >= 0 )
 664         m_cur_grid->SetGridCursor( std::min( curRow, m_cur_grid->GetNumberRows() - 1 ), curCol );
 665 }

 728 void PANEL_FP_LIB_TABLE::browseLibrariesHandler( wxCommandEvent& event )
 729 {
 730     if( !m_cur_grid->CommitPendingChanges() )
 731         return;
 732 
 733     std::map<int, supportedFileType>::const_iterator fileTypeIt;
 734 
 735     // We are bound both to the menu and button with this one handler

 736     // So we must set the file type based on it

 737     if( event.GetEventType() == wxEVT_BUTTON )
 738     {
 739         // Let's default to adding a kicad module for just the module

 740         fileTypeIt = fileTypes().find( ID_PANEL_FPLIB_ADD_KICADMOD );
 741     }
 742     else
 743     {
 744         fileTypeIt = fileTypes().find( event.GetId() );
 745     }
 746 
 747     if( fileTypeIt == fileTypes().end() )
 748     {
 749         wxLogWarning( "File type selection event received but could not find the file type in the table" );
 750         return;
 751     }
 752 
 753     supportedFileType fileType = fileTypeIt->second;
 754 
 755     if( m_lastBrowseDir.IsEmpty() )
 756         m_lastBrowseDir = m_projectBasePath;
 757 
 758     wxArrayString files;
 759 
 760     wxString title;
 761 
 762     title.Printf( _( "Select %s Library" ), fileType.m_Description );
 763 
 764     if( fileType.m_IsFile )
 765     {
 766         wxFileDialog dlg( this, title, m_lastBrowseDir, wxEmptyString,
 767                 fileType.m_FileFilter, wxFD_OPEN | wxFD_FILE_MUST_EXIST | wxFD_MULTIPLE );
 768 
 769         int result = dlg.ShowModal();
 770 
 771         if( result == wxID_CANCEL )
 772             return;
 773 
 774         dlg.GetPaths( files );
 775 
 776         m_lastBrowseDir = dlg.GetDirectory();
 777     }
 778     else
 779     {
 780         wxDirDialog dlg( nullptr, title, m_lastBrowseDir,
 781                 wxDD_DEFAULT_STYLE | wxDD_DIR_MUST_EXIST );
 782 
 783         int result = dlg.ShowModal();
 784 
 785         if( result == wxID_CANCEL )
 786             return;
 787 
 788         // is there a file extension configured to hunt out their containing folders?

 789         if( fileType.m_FolderSearchExtension != "" )
 790         {
 791             wxDir rootDir( dlg.GetPath() );
 792 
 793             LIBRARY_TRAVERSER traverser( fileType.m_FolderSearchExtension, rootDir.GetName() );
 794             rootDir.Traverse( traverser );
 795 
 796             traverser.GetPaths( files );
 797 
 798             if( traverser.HasDirectoryOpenFailures() )
 799             {
 800                 wxArrayString failedDirs;
 801                 traverser.GetPaths( failedDirs );
 802                 wxString detailedMsg = _( "The following directories could not be opened: \n" );
 803 
 804                 for( auto& path : failedDirs )
 805                     detailedMsg << path << "\n";
 806 
 807                 DisplayErrorMessage( this, _( "Failed to open directories to look for libraries" ),
 808                         detailedMsg );
 809             }
 810         }
 811         else
 812         {
 813             files.Add( dlg.GetPath() );
 814         }
 815 
 816         m_lastBrowseDir = dlg.GetPath();
 817     }
 818 
 819     // Drop the last directory if the path is a .pretty folder

 820     if( m_lastBrowseDir.EndsWith( KiCadFootprintLibPathExtension ) )
 821         m_lastBrowseDir = m_lastBrowseDir.BeforeLast( wxFileName::GetPathSeparator() );
 822 
 823     const ENV_VAR_MAP& envVars       = Pgm().GetLocalEnvVariables();
 824     bool               addDuplicates = false;
 825     bool               applyToAll    = false;
 826     wxString           warning       = _( "Warning: Duplicate Nickname" );
 827     wxString           msg           = _( "A library nicknamed \"%s\" already exists." );
 828     wxString           detailedMsg   = _( "Please change the library nickname after adding this library." );
 829 
 830     for( const auto& filePath : files )
 831     {
 832         wxFileName fn( filePath );
 833         wxString   nickname = LIB_ID::FixIllegalChars( fn.GetName(), LIB_ID::ID_PCB );
 834         bool       doAdd    = true;
 835 
 836         if( cur_model()->ContainsNickname( nickname ) )
 837         {
 838             if( !applyToAll )
 839             {
 840                 // The cancel button adds the library to the table anyway

 841                 addDuplicates = ( OKOrCancelDialog( this, warning, wxString::Format( msg, nickname ),
 842                         detailedMsg, _( "Skip" ), _( "Add Anyway" ), &applyToAll ) == wxID_CANCEL );
 843             }
 844 
 845             doAdd = addDuplicates;
 846         }
 847 
 848         if( doAdd && m_cur_grid->AppendRows( 1 ) )
 849         {
 850             int last_row = m_cur_grid->GetNumberRows() - 1;
 851 
 852             m_cur_grid->SetCellValue( last_row, COL_NICKNAME, nickname );
 853 
 854             m_cur_grid->SetCellValue( last_row, COL_TYPE, IO_MGR::ShowType( fileType.m_Plugin ) );
 855 
 856             // try to use path normalized to an environmental variable or project path

 857             wxString path = NormalizePath( filePath, &envVars, m_projectBasePath );
 858 
 859             // Do not use the project path in the global library table.  This will almost

 860             // assuredly be wrong for a different project.

 861             if( path.IsEmpty() || ( m_pageNdx == 0 && path.Contains( "${KIPRJMOD}" ) ) )
 862                 path = fn.GetFullPath();
 863 
 864             m_cur_grid->SetCellValue( last_row, COL_URI, path );
 865         }
 866     }
 867 
 868     if( !files.IsEmpty() )
 869     {
 870         int new_row = m_cur_grid->GetNumberRows() - 1;
 871         m_cur_grid->MakeCellVisible( new_row, m_cur_grid->GetGridCursorCol() );
 872         m_cur_grid->SetGridCursor( new_row, m_cur_grid->GetGridCursorCol() );
 873     }
 874 }

 929 /// Populate the readonly environment variable table with names and values

 930 /// by examining all the full_uri columns.

 931 void PANEL_FP_LIB_TABLE::populateEnvironReadOnlyTable()
 932 {
 933     wxRegEx re( ".*?(\\$\\{(.+?)\\})|(\\$\\((.+?)\\)).*?", wxRE_ADVANCED );
 934     wxASSERT( re.IsValid() );   // wxRE_ADVANCED is required.

 935 
 936     std::set< wxString > unique;
 937 
 938     // clear the table

 939     m_path_subs_grid->DeleteRows( 0, m_path_subs_grid->GetNumberRows() );
 940 
 941     for( FP_LIB_TABLE_GRID* tbl : { global_model(), project_model() } )
 942     {
 943         for( int row = 0; row < tbl->GetNumberRows(); ++row )
 944         {
 945             wxString uri = tbl->GetValue( row, COL_URI );
 946 
 947             while( re.Matches( uri ) )
 948             {
 949                 wxString envvar = re.GetMatch( uri, 2 );
 950 
 951                 // if not ${...} form then must be $(...)

 952                 if( envvar.IsEmpty() )
 953                     envvar = re.GetMatch( uri, 4 );
 954 
 955                 // ignore duplicates

 956                 unique.insert( envvar );
 957 
 958                 // delete the last match and search again

 959                 uri.Replace( re.GetMatch( uri, 0 ), wxEmptyString );
 960             }
 961         }
 962     }
 963 
 964     // Make sure this special environment variable shows up even if it was

 965     // not used yet.  It is automatically set by KiCad to the directory holding

 966     // the current project.

 967     unique.insert( PROJECT_VAR_NAME );
 968     unique.insert( FP_LIB_TABLE::GlobalPathEnvVariableName() );
 969     // This special environment variable is used to locate 3d shapes

 970     unique.insert( KISYS3DMOD );
 971 
 972     for( const wxString& evName : unique )
 973     {
 974         int row = m_path_subs_grid->GetNumberRows();
 975         m_path_subs_grid->AppendRows( 1 );
 976 
 977         m_path_subs_grid->SetCellValue( row, 0, wxT( "${" ) + evName + wxT( "}" ) );
 978         m_path_subs_grid->SetCellEditor( row, 0, new GRID_CELL_READONLY_TEXT_EDITOR() );
 979 
 980         wxString evValue;
 981         wxGetEnv( evName, &evValue );
 982         m_path_subs_grid->SetCellValue( row, 1, evValue );
 983         m_path_subs_grid->SetCellEditor( row, 1, new GRID_CELL_READONLY_TEXT_EDITOR() );
 984     }
 985 
 986     // No combobox editors here, but it looks better if its consistent with the other

 987     // grids in the dialog.

 988     m_path_subs_grid->SetDefaultRowSize( m_path_subs_grid->GetDefaultRowSize() + 2 );
 989 
 990     adjustPathSubsGridColumns( m_path_subs_grid->GetRect().GetWidth() );
 991 }
