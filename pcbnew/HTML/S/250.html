  72 void DIALOG_PLOT::init_Dialog()
  73 {
  74     BOARD*      board = m_parent->GetBoard();
  75     wxFileName  fileName;
  76 
  77     auto cfg = m_parent->GetPcbNewSettings();
  78 
  79     m_XScaleAdjust = cfg->m_Plot.fine_scale_x;
  80     m_YScaleAdjust = cfg->m_Plot.fine_scale_y;
  81 
  82     m_zoneFillCheck->SetValue( cfg->m_Plot.check_zones_before_plotting );
  83 
  84     m_browseButton->SetBitmap( KiBitmap( folder_xpm ) );
  85 
  86     // m_PSWidthAdjust is stored in mm in user config

  87     m_PSWidthAdjust = KiROUND( cfg->m_Plot.ps_fine_width_adjust * IU_PER_MM );
  88 
  89     // The reasonable width correction value must be in a range of

  90     // [-(MinTrackWidth-1), +(MinClearanceValue-1)] decimils.

  91     m_widthAdjustMinValue   = -( board->GetDesignSettings().m_TrackMinWidth - 1 );
  92     m_widthAdjustMaxValue   = board->GetDesignSettings().GetSmallestClearanceValue() - 1;
  93 
  94     switch( m_plotOpts.GetFormat() )
  95     {
  96     default:
  97     case PLOT_FORMAT::GERBER: m_plotFormatOpt->SetSelection( 0 ); break;
  98     case PLOT_FORMAT::POST:   m_plotFormatOpt->SetSelection( 1 ); break;
  99     case PLOT_FORMAT::SVG:    m_plotFormatOpt->SetSelection( 2 ); break;
 100     case PLOT_FORMAT::DXF:    m_plotFormatOpt->SetSelection( 3 ); break;
 101     case PLOT_FORMAT::HPGL:   m_plotFormatOpt->SetSelection( 4 ); break;
 102     case PLOT_FORMAT::PDF:    m_plotFormatOpt->SetSelection( 5 ); break;
 103     }
 104 
 105     // Set units and value for HPGL pen size (this param is in mils).

 106     m_defaultPenSize.SetValue( m_plotOpts.GetHPGLPenDiameter() * IU_PER_MILS );
 107 
 108     // Test for a reasonable scale value. Set to 1 if problem

 109     if( m_XScaleAdjust < PLOT_MIN_SCALE || m_YScaleAdjust < PLOT_MIN_SCALE
 110         || m_XScaleAdjust > PLOT_MAX_SCALE || m_YScaleAdjust > PLOT_MAX_SCALE )
 111         m_XScaleAdjust = m_YScaleAdjust = 1.0;
 112 
 113     m_fineAdjustXCtrl->SetValue( StringFromValue( EDA_UNITS::UNSCALED, m_XScaleAdjust ) );
 114     m_fineAdjustYCtrl->SetValue( StringFromValue( EDA_UNITS::UNSCALED, m_YScaleAdjust ) );
 115 
 116     // Test for a reasonable PS width correction value. Set to 0 if problem.

 117     if( m_PSWidthAdjust < m_widthAdjustMinValue || m_PSWidthAdjust > m_widthAdjustMaxValue )
 118         m_PSWidthAdjust = 0.;
 119 
 120     m_trackWidthCorrection.SetValue( m_PSWidthAdjust );
 121 
 122     m_plotPSNegativeOpt->SetValue( m_plotOpts.GetNegative() );
 123     m_forcePSA4OutputOpt->SetValue( m_plotOpts.GetA4Output() );
 124 
 125     // Could devote a PlotOrder() function in place of UIOrder().

 126     m_layerList = board->GetEnabledLayers().UIOrder();
 127 
 128     // Populate the check list box by all enabled layers names

 129     for( LSEQ seq = m_layerList;  seq;  ++seq )
 130     {
 131         PCB_LAYER_ID layer = *seq;
 132 
 133         int checkIndex = m_layerCheckListBox->Append( board->GetLayerName( layer ) );
 134 
 135         if( m_plotOpts.GetLayerSelection()[layer] )
 136             m_layerCheckListBox->Check( checkIndex );
 137     }
 138 
 139     // Option for using proper Gerber extensions. Note also Protel extensions are

 140     // a broken feature. However, for now, we need to handle it.

 141     m_useGerberExtensions->SetValue( m_plotOpts.GetUseGerberProtelExtensions() );
 142 
 143     // Option for including Gerber attributes, from Gerber X2 format, in the output

 144     // In X1 format, they will be added as comments

 145     m_useGerberX2Format->SetValue( m_plotOpts.GetUseGerberX2format() );
 146 
 147     // Option for including Gerber netlist info (from Gerber X2 format) in the output

 148     m_useGerberNetAttributes->SetValue( m_plotOpts.GetIncludeGerberNetlistInfo() );
 149 
 150     // Option to generate a Gerber job file

 151     m_generateGerberJobFile->SetValue( m_plotOpts.GetCreateGerberJobFile() );
 152 
 153     // Gerber precision for coordinates

 154     m_coordFormatCtrl->SetSelection( m_plotOpts.GetGerberPrecision() == 5 ? 0 : 1 );
 155 
 156     // SVG precision and units for coordinates

 157     m_svgPrecsision->SetValue( m_plotOpts.GetSvgPrecision() );
 158     m_svgUnits->SetSelection( m_plotOpts.GetSvgUseInch() );
 159 
 160     // Option for excluding contents of "Edges Pcb" layer

 161     m_excludeEdgeLayerOpt->SetValue( m_plotOpts.GetExcludeEdgeLayer() );
 162 
 163     // Option to exclude pads from silkscreen layers

 164     m_sketchPadsOnFabLayers->SetValue( m_plotOpts.GetSketchPadsOnFabLayers() );
 165 
 166     // Option to tent vias

 167     m_subtractMaskFromSilk->SetValue( m_plotOpts.GetSubtractMaskFromSilk() );
 168 
 169     // Option to use aux origin

 170     m_useAuxOriginCheckBox->SetValue( m_plotOpts.GetUseAuxOrigin() );
 171 
 172     // Option to plot page references:

 173     m_plotSheetRef->SetValue( m_plotOpts.GetPlotFrameRef() );
 174 
 175     // Options to plot texts on footprints

 176     m_plotModuleValueOpt->SetValue( m_plotOpts.GetPlotValue() );
 177     m_plotModuleRefOpt->SetValue( m_plotOpts.GetPlotReference() );
 178     m_plotInvisibleText->SetValue( m_plotOpts.GetPlotInvisibleText() );
 179 
 180     // Options to plot pads and vias holes

 181     m_drillShapeOpt->SetSelection( m_plotOpts.GetDrillMarksType() );
 182 
 183     // Scale option

 184     m_scaleOpt->SetSelection( m_plotOpts.GetScaleSelection() );
 185 
 186     // Plot mode

 187     setPlotModeChoiceSelection( m_plotOpts.GetPlotMode() );
 188 
 189     // DXF outline mode

 190     m_DXF_plotModeOpt->SetValue( m_plotOpts.GetDXFPlotPolygonMode() );
 191 
 192     // DXF text mode

 193     m_DXF_plotTextStrokeFontOpt->SetValue( m_plotOpts.GetTextMode() == PLOT_TEXT_MODE::DEFAULT );
 194 
 195     // DXF units selection

 196     m_DXF_plotUnits->SetSelection( static_cast<int>( m_plotOpts.GetDXFPlotUnits() ) );
 197 
 198     // Plot mirror option

 199     m_plotMirrorOpt->SetValue( m_plotOpts.GetMirror() );
 200 
 201     // Put vias on mask layer

 202     m_plotNoViaOnMaskOpt->SetValue( m_plotOpts.GetPlotViaOnMaskLayer() );
 203 
 204     // Initialize a few other parameters, which can also be modified

 205     // from the drill dialog

 206     reInitDialog();
 207 
 208     // Update options values:

 209     wxCommandEvent cmd_event;
 210     SetPlotFormat( cmd_event );
 211     OnSetScaleOpt( cmd_event );
 212 }

 215 void DIALOG_PLOT::reInitDialog()
 216 {
 217     // after calling the Drill or DRC dialogs some parameters can be modified....

 218 
 219     // Output directory

 220     m_outputDirectoryName->SetValue( m_plotOpts.GetOutputDirectory() );
 221 
 222     // Origin of coordinates:

 223     m_useAuxOriginCheckBox->SetValue( m_plotOpts.GetUseAuxOrigin() );
 224 
 225     int knownViolations = 0;
 226     int exclusions = 0;
 227 
 228     for( MARKER_PCB* marker : m_parent->GetBoard()->Markers() )
 229     {
 230         if( marker->IsExcluded() )
 231             exclusions++;
 232         else
 233             knownViolations++;
 234     }
 235 
 236     if( knownViolations || exclusions )
 237     {
 238         m_DRCExclusionsWarning->SetLabel( wxString::Format( m_DRCWarningTemplate,
 239                                                             knownViolations,
 240                                                             exclusions ) );
 241         m_DRCExclusionsWarning->Show();
 242     }
 243     else
 244         m_DRCExclusionsWarning->Hide();
 245 }

 255 // Select or deselect groups of layers in the layers list:

 256 void DIALOG_PLOT::OnPopUpLayers( wxCommandEvent& event )
 257 {
 258     // Build a list of layers for usual fabrication:

 259     // copper layers + tech layers without courtyard

 260     LSET fab_layer_set = ( LSET::AllCuMask() | LSET::AllTechMask() )
 261                          & ~LSET( 2, B_CrtYd, F_CrtYd );
 262 
 263     switch( event.GetId() )
 264     {
 265     case ID_LAYER_FAB: // Select layers usually needed to build a board

 266         for( unsigned i = 0; i < m_layerList.size(); i++ )
 267         {
 268             LSET layermask( m_layerList[ i ] );
 269 
 270             if( ( layermask & fab_layer_set ).any() )
 271                 m_layerCheckListBox->Check( i, true );
 272             else
 273                 m_layerCheckListBox->Check( i, false );
 274         }
 275         break;
 276 
 277     case ID_SELECT_COPPER_LAYERS:
 278         for( unsigned i = 0; i < m_layerList.size(); i++ )
 279         {
 280             if( IsCopperLayer( m_layerList[i] ) )
 281                 m_layerCheckListBox->Check( i, true );
 282         }
 283         break;
 284 
 285     case ID_DESELECT_COPPER_LAYERS:
 286         for( unsigned i = 0; i < m_layerList.size(); i++ )
 287         {
 288             if( IsCopperLayer( m_layerList[i] ) )
 289                 m_layerCheckListBox->Check( i, false );
 290         }
 291         break;
 292 
 293     case ID_SELECT_ALL_LAYERS:
 294         for( unsigned i = 0; i < m_layerList.size(); i++ )
 295             m_layerCheckListBox->Check( i, true );
 296         break;
 297 
 298     case ID_DESELECT_ALL_LAYERS:
 299         for( unsigned i = 0; i < m_layerList.size(); i++ )
 300             m_layerCheckListBox->Check( i, false );
 301         break;
 302 
 303     default:
 304         break;
 305     }
 306 }

 349 void DIALOG_PLOT::OnOutputDirectoryBrowseClicked( wxCommandEvent& event )
 350 {
 351     // Build the absolute path of current output directory to preselect it in the file browser.

 352     wxString path = ExpandEnvVarSubstitutions( m_outputDirectoryName->GetValue(), &Prj() );
 353     path = Prj().AbsolutePath( path );
 354 
 355     wxDirDialog dirDialog( this, _( "Select Output Directory" ), path );
 356 
 357     if( dirDialog.ShowModal() == wxID_CANCEL )
 358         return;
 359 
 360     wxFileName      dirName = wxFileName::DirName( dirDialog.GetPath() );
 361 
 362     wxFileName fn( Prj().AbsolutePath( m_parent->GetBoard()->GetFileName() ) );
 363     wxString defaultPath = fn.GetPathWithSep();
 364     wxString msg;
 365     msg.Printf( _( "Do you want to use a path relative to\n\"%s\"" ),
 366                    GetChars( defaultPath ) );
 367 
 368     wxMessageDialog dialog( this, msg, _( "Plot Output Directory" ),
 369                             wxYES_NO | wxICON_QUESTION | wxYES_DEFAULT );
 370 
 371     if( dialog.ShowModal() == wxID_YES )
 372     {
 373         if( !dirName.MakeRelativeTo( defaultPath ) )
 374             wxMessageBox( _( "Cannot make path relative (target volume different from file volume)!" ),
 375                           _( "Plot Output Directory" ), wxOK | wxICON_ERROR );
 376     }
 377 
 378     m_outputDirectoryName->SetValue( dirName.GetFullPath() );
 379 }

 556 // A helper function to "clip" aValue between aMin and aMax

 557 // and write result in * aResult

 558 // return false if clipped, true if aValue is just copied into * aResult

 559 static bool setDouble( double* aResult, double aValue, double aMin, double aMax )
 560 {
 561     if( aValue < aMin )
 562     {
 563         *aResult = aMin;
 564         return false;
 565     }
 566     else if( aValue > aMax )
 567     {
 568         *aResult = aMax;
 569         return false;
 570     }
 571 
 572     *aResult = aValue;
 573     return true;
 574 }

 588     }
 589 
 590     *aResult = aValue;
 591     return true;
 592 }

 595 void DIALOG_PLOT::applyPlotSettings()
 596 {
 597     REPORTER&       reporter = m_messagesPanel->Reporter();
 598     int             sel;
 599     PCB_PLOT_PARAMS tempOptions;
 600 
 601     tempOptions.SetExcludeEdgeLayer( m_excludeEdgeLayerOpt->GetValue() );
 602     tempOptions.SetSubtractMaskFromSilk( m_subtractMaskFromSilk->GetValue() );
 603     tempOptions.SetPlotFrameRef( m_plotSheetRef->GetValue() );
 604     tempOptions.SetSketchPadsOnFabLayers( m_sketchPadsOnFabLayers->GetValue());
 605     tempOptions.SetUseAuxOrigin( m_useAuxOriginCheckBox->GetValue() );
 606     tempOptions.SetPlotValue( m_plotModuleValueOpt->GetValue() );
 607     tempOptions.SetPlotReference( m_plotModuleRefOpt->GetValue() );
 608     tempOptions.SetPlotInvisibleText( m_plotInvisibleText->GetValue() );
 609     tempOptions.SetScaleSelection( m_scaleOpt->GetSelection() );
 610 
 611     sel = m_drillShapeOpt->GetSelection();
 612     tempOptions.SetDrillMarksType( static_cast<PCB_PLOT_PARAMS::DrillMarksType>( sel ) );
 613 
 614     tempOptions.SetMirror( m_plotMirrorOpt->GetValue() );
 615     tempOptions.SetPlotMode( m_plotModeOpt->GetSelection() == 1 ? SKETCH : FILLED );
 616     tempOptions.SetDXFPlotPolygonMode( m_DXF_plotModeOpt->GetValue() );
 617 
 618     sel = m_DXF_plotUnits->GetSelection();
 619     tempOptions.SetDXFPlotUnits( static_cast<DXF_PLOTTER::DXF_UNITS>( sel ) );
 620 
 621     tempOptions.SetPlotViaOnMaskLayer( m_plotNoViaOnMaskOpt->GetValue() );
 622 
 623     if( !m_DXF_plotTextStrokeFontOpt->IsEnabled() ) // Currently, only DXF supports this option

 624         tempOptions.SetTextMode( PLOT_TEXT_MODE::DEFAULT );
 625     else
 626         tempOptions.SetTextMode( m_DXF_plotTextStrokeFontOpt->GetValue() ? PLOT_TEXT_MODE::DEFAULT :
 627                                                                            PLOT_TEXT_MODE::NATIVE );
 628 
 629     // Update settings from text fields. Rewrite values back to the fields,

 630     // since the values may have been constrained by the setters.

 631     wxString    msg;
 632 
 633     // read HPLG pen size (this param is stored in mils)

 634     // However, due to issues when converting this value from or to mm

 635     // that can slightly change the value, update this param only if it

 636     // is in use

 637     if( getPlotFormat() == PLOT_FORMAT::HPGL )
 638     {
 639         if( !tempOptions.SetHPGLPenDiameter( m_defaultPenSize.GetValue() / IU_PER_MILS ) )
 640         {
 641             m_defaultPenSize.SetValue( tempOptions.GetHPGLPenDiameter() * IU_PER_MILS );
 642             msg.Printf( _( "HPGL pen size constrained." ) );
 643             reporter.Report( msg, RPT_SEVERITY_INFO );
 644         }
 645     }
 646     else    // keep the last value (initial value if no HPGL plot made)

 647         tempOptions.SetHPGLPenDiameter( m_plotOpts.GetHPGLPenDiameter() );
 648 
 649     // X scale

 650     double tmpDouble;
 651     msg = m_fineAdjustXCtrl->GetValue();
 652     msg.ToDouble( &tmpDouble );
 653 
 654     if( !setDouble( &m_XScaleAdjust, tmpDouble, PLOT_MIN_SCALE, PLOT_MAX_SCALE ) )
 655     {
 656         msg.Printf( wxT( "%f" ), m_XScaleAdjust );
 657         m_fineAdjustXCtrl->SetValue( msg );
 658         msg.Printf( _( "X scale constrained." ) );
 659         reporter.Report( msg, RPT_SEVERITY_INFO );
 660     }
 661 
 662     // Y scale

 663     msg = m_fineAdjustYCtrl->GetValue();
 664     msg.ToDouble( &tmpDouble );
 665 
 666     if( !setDouble( &m_YScaleAdjust, tmpDouble, PLOT_MIN_SCALE, PLOT_MAX_SCALE ) )
 667     {
 668         msg.Printf( wxT( "%f" ), m_YScaleAdjust );
 669         m_fineAdjustYCtrl->SetValue( msg );
 670         msg.Printf( _( "Y scale constrained." ) );
 671         reporter.Report( msg, RPT_SEVERITY_INFO );
 672     }
 673 
 674     auto cfg = m_parent->GetPcbNewSettings();
 675 
 676     cfg->m_Plot.fine_scale_x = m_XScaleAdjust;
 677     cfg->m_Plot.fine_scale_y = m_YScaleAdjust;
 678 
 679     cfg->m_Plot.check_zones_before_plotting = m_zoneFillCheck->GetValue();
 680 
 681     // PS Width correction

 682     if( !setInt( &m_PSWidthAdjust, m_trackWidthCorrection.GetValue(),
 683                  m_widthAdjustMinValue, m_widthAdjustMaxValue ) )
 684     {
 685         m_trackWidthCorrection.SetValue( m_PSWidthAdjust );
 686         msg.Printf( _( "Width correction constrained. "
 687                        "The reasonable width correction value must be in a range of "
 688                        " [%s; %s] (%s) for current design rules." ),
 689                     StringFromValue( GetUserUnits(), m_widthAdjustMinValue, false, true ),
 690                     StringFromValue( GetUserUnits(), m_widthAdjustMaxValue, false, true ),
 691                     GetAbbreviatedUnitsLabel( GetUserUnits(), true ) );
 692         reporter.Report( msg, RPT_SEVERITY_WARNING );
 693     }
 694 
 695     // Store m_PSWidthAdjust in mm in user config

 696     cfg->m_Plot.ps_fine_width_adjust = Iu2Millimeter( m_PSWidthAdjust );
 697 
 698     tempOptions.SetFormat( getPlotFormat() );
 699 
 700     tempOptions.SetUseGerberProtelExtensions( m_useGerberExtensions->GetValue() );
 701     tempOptions.SetUseGerberX2format( m_useGerberX2Format->GetValue() );
 702     tempOptions.SetIncludeGerberNetlistInfo( m_useGerberNetAttributes->GetValue() );
 703     tempOptions.SetCreateGerberJobFile( m_generateGerberJobFile->GetValue() );
 704 
 705     tempOptions.SetGerberPrecision( m_coordFormatCtrl->GetSelection() == 0 ? 5 : 6 );
 706     tempOptions.SetSvgPrecision( m_svgPrecsision->GetValue(), m_svgUnits->GetSelection() );
 707 
 708     LSET selectedLayers;
 709     for( unsigned i = 0; i < m_layerList.size(); i++ )
 710     {
 711         if( m_layerCheckListBox->IsChecked( i ) )
 712             selectedLayers.set( m_layerList[i] );
 713     }
 714     // Get a list of copper layers that aren't being used by inverting enabled layers.

 715     LSET disabledCopperLayers = LSET::AllCuMask() & ~m_parent->GetBoard()->GetEnabledLayers();
 716     // Enable all of the disabled copper layers.

 717     // If someone enables more copper layers they will be selected by default.

 718     selectedLayers = selectedLayers | disabledCopperLayers;
 719     tempOptions.SetLayerSelection( selectedLayers );
 720 
 721     tempOptions.SetNegative( m_plotPSNegativeOpt->GetValue() );
 722     tempOptions.SetA4Output( m_forcePSA4OutputOpt->GetValue() );
 723 
 724     // Set output directory and replace backslashes with forward ones

 725     wxString dirStr;
 726     dirStr = m_outputDirectoryName->GetValue();
 727     dirStr.Replace( wxT( "\\" ), wxT( "/" ) );
 728     tempOptions.SetOutputDirectory( dirStr );
 729 
 730     if( !m_plotOpts.IsSameAs( tempOptions, false ) )
 731     {
 732         // First, mark board as modified only for parameters saved in file

 733         if( !m_plotOpts.IsSameAs( tempOptions, true ) )
 734             m_parent->OnModify();
 735 
 736         // Now, save any change, for the session

 737         m_parent->SetPlotSettings( tempOptions );
 738         m_plotOpts = tempOptions;
 739     }
 740 }

 748 
 749 void DIALOG_PLOT::Plot( wxCommandEvent& event )
 750 {
 751     BOARD* board = m_parent->GetBoard();
 752 
 753     applyPlotSettings();
 754 
 755     SETTINGS_MANAGER& mgr = Pgm().GetSettingsManager();
 756     PCBNEW_SETTINGS*  cfg = mgr.GetAppSettings<PCBNEW_SETTINGS>();
 757 
 758     m_plotOpts.SetColorSettings( mgr.GetColorSettings( cfg->m_ColorTheme ) );
 759 
 760     m_plotOpts.SetSketchPadLineWidth( board->GetDesignSettings().GetLineThickness( F_Fab ) );
 761 
 762     // If no layer selected, we have nothing plotted.

 763     // Prompt user if it happens because he could think there is a bug in Pcbnew.

 764     if( !m_plotOpts.GetLayerSelection().any() )
 765     {
 766         DisplayError( this, _( "No layer selected, Nothing to plot" ) );
 767         return;
 768     }
 769 
 770     // Create output directory if it does not exist (also transform it in

 771     // absolute form). Bail if it fails

 772     wxString    path = ExpandEnvVarSubstitutions( m_plotOpts.GetOutputDirectory(), &Prj() );
 773     wxFileName  outputDir = wxFileName::DirName( path );
 774     wxString    boardFilename = m_parent->GetBoard()->GetFileName();
 775     REPORTER&   reporter = m_messagesPanel->Reporter();
 776 
 777     if( !EnsureFileDirectoryExists( &outputDir, boardFilename, &reporter ) )
 778     {
 779         wxString msg;
 780         msg.Printf( _( "Could not write plot files to folder \"%s\"." ),
 781                     GetChars( outputDir.GetPath() ) );
 782         DisplayError( this, msg );
 783         return;
 784     }
 785 
 786     if( m_zoneFillCheck->GetValue() )
 787         m_parent->GetToolManager()->GetTool<ZONE_FILLER_TOOL>()->CheckAllZones( this );
 788 
 789     m_plotOpts.SetAutoScale( false );
 790 
 791     switch( m_plotOpts.GetScaleSelection() )
 792     {
 793     default: m_plotOpts.SetScale( 1 );        break;
 794     case 0:  m_plotOpts.SetAutoScale( true ); break;
 795     case 2:  m_plotOpts.SetScale( 1.5 );      break;
 796     case 3:  m_plotOpts.SetScale( 2 );        break;
 797     case 4:  m_plotOpts.SetScale( 3 );        break;
 798     }
 799 
 800     /* If the scale factor edit controls are disabled or the scale value

 801      * is 0, don't adjust the base scale factor. This fixes a bug when

 802      * the default scale adjust is initialized to 0 and saved in program

 803      * settings resulting in a divide by zero fault.

 804      */
 805     if( getPlotFormat() == PLOT_FORMAT::POST )
 806     {
 807         if( m_XScaleAdjust != 0.0 )
 808             m_plotOpts.SetFineScaleAdjustX( m_XScaleAdjust );
 809 
 810         if( m_YScaleAdjust != 0.0 )
 811             m_plotOpts.SetFineScaleAdjustY( m_YScaleAdjust );
 812 
 813         m_plotOpts.SetWidthAdjust( m_PSWidthAdjust );
 814     }
 815 
 816     wxString file_ext( GetDefaultPlotExtension( m_plotOpts.GetFormat() ) );
 817 
 818     // Test for a reasonable scale value

 819     // XXX could this actually happen? isn't it constrained in the apply

 820     // function?

 821     if( m_plotOpts.GetScale() < PLOT_MIN_SCALE )
 822         DisplayInfoMessage( this, _( "Warning: Scale option set to a very small value" ) );
 823 
 824     if( m_plotOpts.GetScale() > PLOT_MAX_SCALE )
 825         DisplayInfoMessage( this, _( "Warning: Scale option set to a very large value" ) );
 826 
 827     GERBER_JOBFILE_WRITER jobfile_writer( board, &reporter );
 828 
 829     // Save the current plot options in the board

 830     m_parent->SetPlotSettings( m_plotOpts );
 831 
 832     wxBusyCursor dummy;
 833 
 834     for( LSEQ seq = m_plotOpts.GetLayerSelection().UIOrder();  seq;  ++seq )
 835     {
 836         PCB_LAYER_ID layer = *seq;
 837 
 838         // All copper layers that are disabled are actually selected

 839         // This is due to wonkyness in automatically selecting copper layers

 840         // for plotting when adding more than two layers to a board.

 841         // If plot options become accessible to the layers setup dialog

 842         // please move this functionality there!

 843         // This skips a copper layer if it is actually disabled on the board.

 844         if( ( LSET::AllCuMask() & ~board->GetEnabledLayers() )[layer] )
 845             continue;
 846 
 847         // Pick the basename from the board file

 848         wxFileName fn( boardFilename );
 849 
 850         // Use Gerber Extensions based on layer number

 851         // (See http://en.wikipedia.org/wiki/Gerber_File)

 852         if( m_plotOpts.GetFormat() == PLOT_FORMAT::GERBER && m_useGerberExtensions->GetValue() )
 853             file_ext = GetGerberProtelExtension( layer );
 854 
 855         BuildPlotFileName( &fn, outputDir.GetPath(), board->GetLayerName( layer ), file_ext );
 856         wxString fullname = fn.GetFullName();
 857         jobfile_writer.AddGbrFile( layer, fullname );
 858 
 859         LOCALE_IO toggle;
 860 
 861         PLOTTER*    plotter = StartPlotBoard( board, &m_plotOpts, layer, fn.GetFullPath(), wxEmptyString );
 862 
 863         // Print diags in messages box:

 864         wxString msg;
 865 
 866         if( plotter )
 867         {
 868             PlotOneBoardLayer( board, plotter, layer, m_plotOpts );
 869             plotter->EndPlot();
 870             delete plotter->RenderSettings();
 871             delete plotter;
 872 
 873             msg.Printf( _( "Plot file \"%s\" created." ), fn.GetFullPath() );
 874             reporter.Report( msg, RPT_SEVERITY_ACTION );
 875         }
 876         else
 877         {
 878             msg.Printf( _( "Unable to create file \"%s\"." ), fn.GetFullPath() );
 879             reporter.Report( msg, RPT_SEVERITY_ERROR );
 880         }
 881 
 882         wxSafeYield();      // displays report message.

 883     }
 884 
 885     if( m_plotOpts.GetFormat() == PLOT_FORMAT::GERBER && m_plotOpts.GetCreateGerberJobFile() )
 886     {
 887         // Pick the basename from the board file

 888         wxFileName fn( boardFilename );
 889         // Build gerber job file from basename

 890         BuildPlotFileName( &fn, outputDir.GetPath(), "job", GerberJobFileExtension );
 891         jobfile_writer.CreateJobFile( fn.GetFullPath() );
 892     }
 893 }
