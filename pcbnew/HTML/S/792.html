  68 class PCB_TOOL_BASE : public TOOL_INTERACTIVE
  69 {
  70 public:
  71     /**

  72      * Constructor

  73      *

  74      * Creates a tool with given id & name. The name must be unique. */
  75     PCB_TOOL_BASE( TOOL_ID aId, const std::string& aName ) :
  76         TOOL_INTERACTIVE ( aId, aName ),
  77         m_editModules( false )
  78     {};
  79 
  80     /**

  81      * Constructor

  82      *

  83      * Creates a tool with given name. The name must be unique. */
  84     PCB_TOOL_BASE( const std::string& aName ) :
  85         TOOL_INTERACTIVE ( aName ),
  86         m_editModules( false )
  87     {};
  88 
  89     virtual ~PCB_TOOL_BASE() {};
  90 
  91     virtual bool Init() override;
  92     virtual void Reset( RESET_REASON aReason ) override;
  93 
  94     /**

  95      * Function SetEditModules()

  96      *

  97      * Toggles edit module mode. When enabled, one may select parts of modules individually

  98      * (graphics, pads, etc.), so they can be modified.

  99      * @param aEnabled decides if the mode should be enabled.

 100      */
 101     void SetEditModules( bool aEnabled )
 102     {
 103         m_editModules = aEnabled;
 104     }
 105 
 106     bool EditingModules() const
 107     {
 108         return m_editModules;
 109     }
 110 
 111 protected:
 112 
 113     /**

 114      * Options for placing items interactively.

 115      */
 116     enum INTERACTIVE_PLACEMENT_OPTIONS {
 117         /// Handle the rotate action in the loop by calling the item's rotate method

 118         IPO_ROTATE       = 0x01,
 119 
 120         /// Handle flip action in the loop by calling the item's flip method

 121         IPO_FLIP         = 0x02,
 122 
 123         /// Create an item immediately on placement starting, otherwise show the pencil cursor

 124         /// until the item is created

 125         IPO_SINGLE_CLICK = 0x04,
 126 
 127         /// Allow repeat placement of the item

 128         IPO_REPEAT       = 0x08
 129     };
 130 
 131 
 132     /**

 133      * Helper function for performing a common interactive idiom:

 134      * wait for a left click, place an item there (perhaps with a

 135      * dialog or other user interaction), then have it move with

 136      * the mouse and respond to rotate/flip, etc

 137      *

 138      * More complex interactive processes are not supported here, you

 139      * should implement a customised event loop for those.

 140      *

 141      * @param aItemCreator the callable that will attempt to create the item

 142      * @param aCommitMessage the message used on a successful commit

 143      */
 144     void doInteractiveItemPlacement( const std::string& aTool, INTERACTIVE_PLACER_BASE *aPlacer,
 145                                      const wxString& aCommitMessage,
 146                                      int aOptions = IPO_ROTATE | IPO_FLIP | IPO_REPEAT );
 147 
 148     virtual void setTransitions() override;
 149 
 150 
 151     KIGFX::PCB_VIEW* view() const
 152     {
 153         return static_cast<KIGFX::PCB_VIEW*>( getView() );
 154     }
 155 
 156     KIGFX::VIEW_CONTROLS* controls() const
 157     {
 158         return getViewControls();
 159     }
 160 
 161     PCB_BASE_EDIT_FRAME* frame() const
 162     {
 163         return getEditFrame<PCB_BASE_EDIT_FRAME>();
 164     }
 165 
 166     BOARD* board() const { return getModel<BOARD>(); }
 167 
 168     MODULE* module() const
 169     {
 170         return board()->GetFirstModule();
 171     }
 172 
 173     const PCB_DISPLAY_OPTIONS& displayOptions() const;
 174     PCB_DRAW_PANEL_GAL* canvas() const;
 175     const PCBNEW_SELECTION& selection() const;
 176     PCBNEW_SELECTION& selection();
 177 
 178     bool m_editModules;
 179 };
