  26 #ifndef BOARD_CONNECTED_ITEM_H
  27 #define BOARD_CONNECTED_ITEM_H

  29 #include <class_board_item.h>
  30 #include <netinfo.h>

  33 class NETCLASS;
  34 class TRACK;
  35 class D_PAD;

  37 /**

  38  * BOARD_CONNECTED_ITEM

  39  * is a base class derived from BOARD_ITEM for items that can be connected

  40  * and have a net, a netname, a clearance ...

  41  * mainly: tracks, pads and zones

  42  * Handle connection info

  43  */
  44 class BOARD_CONNECTED_ITEM : public BOARD_ITEM
  45 {
  46 public:
  47     BOARD_CONNECTED_ITEM( BOARD_ITEM* aParent, KICAD_T idtype );
  48 
  49     // Do not create a copy constructor & operator=.

  50     // The ones generated by the compiler are adequate.

  51 
  52     static inline bool ClassOf( const EDA_ITEM* aItem )
  53     {
  54         if( aItem == NULL )
  55             return false;
  56 
  57         switch( aItem->Type() )
  58         {
  59         case PCB_PAD_T:
  60         case PCB_TRACE_T:
  61         case PCB_ARC_T:
  62         case PCB_VIA_T:
  63         case PCB_ZONE_AREA_T:
  64             return true;
  65 
  66         default:
  67             return false;
  68         }
  69     }
  70 
  71     ///> @copydoc BOARD_ITEM::IsConnected()

  72     bool IsConnected() const override
  73     {
  74         return true;
  75     }
  76 
  77     /**

  78      * Function GetNet

  79      * Returns NET_INFO object for a given item.

  80      */
  81     NETINFO_ITEM* GetNet() const
  82     {
  83         return m_netinfo;
  84     }
  85 
  86     /**

  87      * Function SetNet

  88      * Sets a NET_INFO object for the item.

  89      */
  90     void SetNet( NETINFO_ITEM* aNetInfo )
  91     {
  92         wxASSERT( aNetInfo->GetBoard() == GetBoard() || aNetInfo == NETINFO_LIST::OrphanedItem() );
  93         m_netinfo = aNetInfo;
  94     }
  95 
  96     /**

  97      * Function GetNetCode

  98      * @return int - the net code.

  99      */
 100     int GetNetCode() const
 101     {
 102         return m_netinfo ? m_netinfo->GetNet() : -1;
 103     }
 104 
 105     /**

 106      * Sets net using a net code.

 107      * @param aNetCode is a net code for the new net. It has to exist in NETINFO_LIST held by BOARD.

 108      * @param aNoAssert if true, do not assert that the net exists.

 109      * Otherwise, item is assigned to the unconnected net.

 110      * @return true on success, false if the net did not exist

 111      * Note also items (in fact pads) not on copper layers will have

 112      * their net code always set to 0 (not connected)

 113      */
 114     bool SetNetCode( int aNetCode, bool aNoAssert );
 115 
 116     void SetNetCode( int aNetCode )
 117     {
 118         SetNetCode( aNetCode, false );
 119     }
 120 
 121     /**

 122      * Function GetNetname

 123      * @return wxString - the full netname

 124      */
 125     const wxString& GetNetname() const
 126     {
 127         return m_netinfo->GetNetname();
 128     }
 129 
 130     /**

 131      * Function GetNetnameMsg

 132      * @return wxString - the full netname or "<no net>" in square braces, followed by

 133      * "(Not Found)" if the netcode is undefined.

 134      */
 135     wxString GetNetnameMsg() const
 136     {
 137         if( !GetBoard() )
 138             return wxT( "[** NO BOARD DEFINED **]" );
 139 
 140         wxString netname = GetNetname();
 141 
 142         if( !netname.length() )
 143             return wxT( "[<no net>]" );
 144         else if( GetNetCode() < 0 )
 145             return wxT( "[" + UnescapeString( netname ) + "](" + _( "Not Found" ) + ")" );
 146         else
 147             return wxT( "[" + UnescapeString( netname ) + "]" );
 148     }
 149 
 150     /**

 151      * Function GetShortNetname

 152      * @return wxString - the short netname

 153      */
 154     const wxString& GetShortNetname() const
 155     {
 156         return m_netinfo->GetShortNetname();
 157     }
 158 
 159     /**

 160      * Function GetClearance

 161      * returns the clearance in internal units.  If \a aItem is not NULL then the

 162      * returned clearance is the greater of this object's NETCLASS clearance and

 163      * aItem's NETCLASS clearance.  If \a aItem is NULL, then this objects clearance

 164      * is returned.

 165      * @param aItem is an optional BOARD_ITEM

 166      * @param aSource [out] optionally reports the source as a user-readable string

 167      * @return int - the clearance in internal units.

 168      */
 169     virtual int GetClearance( BOARD_ITEM* aItem = nullptr, wxString* aSource = nullptr ) const;
 170 
 171     /**

 172      * Function GetRuleClearance

 173      * returns any rule-based clearance.

 174      * @param aClearance [out] the clearance value in internal units

 175      * @param aSource [out] reports the source as a user-readable string

 176      * @return true if a rule was fired

 177      */
 178     virtual bool GetRuleClearance( BOARD_ITEM* aItem, int* aClearance, wxString* aSource ) const;
 179 
 180     /**

 181      * Function GetLocalClearanceOverrides

 182      * returns any local clearance overrides set in the "classic" (ie: pre-rule) system.

 183      * @param aSource [out] optionally reports the source as a user-readable string

 184      * @return int - the clearance in internal units.

 185      */
 186     virtual int GetLocalClearanceOverrides( wxString* aSource = nullptr ) const { return 0; }
 187 
 188     /**

 189      * Function GetLocalClearance

 190      * returns any local clearances set in the "classic" (ie: pre-rule) system.  These are

 191      * things like zone clearance which are NOT an override.

 192      * @param aSource [out] optionally reports the source as a user-readable string

 193      * @return int - the clearance in internal units.

 194      */
 195     virtual int GetLocalClearance( wxString* aSource = nullptr ) const { return 0; }
 196 
 197     /**

 198      * Function GetNetClassPtr

 199      * returns the NETCLASS for this item.

 200      *

 201      * Note: do NOT return a std::shared_ptr from this.  It is used heavily in DRC, and the

 202      * std::shared_ptr stuff shows up large in performance profiling.

 203      */
 204     NETCLASS* GetNetClass() const;
 205 
 206     /**

 207      * Function GetEffectiveNetclass

 208      * returns the NETCLASS for this item, or the default netclass if none is defined.

 209      *

 210      * Note: do NOT return a std::shared_ptr from this.  It is used heavily in DRC, and the

 211      * std::shared_ptr stuff shows up large in performance profiling.

 212      */
 213     NETCLASS* GetEffectiveNetclass() const;
 214 
 215     /**

 216      * Function GetNetClassName

 217      * returns a pointer to the netclass of the zone.

 218      * If the net is not found (can happen when a netlist is reread,

 219      * and the net name does not exist, return the default net class

 220      * (should not return a null pointer).

 221      * @return the Net Class name of this item

 222      */
 223     wxString GetNetClassName() const;
 224 
 225     void SetLocalRatsnestVisible( bool aVisible )
 226     {
 227         m_localRatsnestVisible = aVisible;
 228     }
 229 
 230     bool GetLocalRatsnestVisible() const
 231     {
 232         return m_localRatsnestVisible;
 233     }
 234 
 235 protected:
 236     /// Stores all informations about the net that item belongs to

 237     NETINFO_ITEM* m_netinfo;
 238 
 239 private:
 240 
 241     bool m_localRatsnestVisible;
 242 };

 244 #endif  // BOARD_CONNECTED_ITEM_H
