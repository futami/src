 388 void PCB_EDIT_FRAME::SetPageSettings( const PAGE_INFO& aPageSettings )
 389 {
 390     PCB_BASE_FRAME::SetPageSettings( aPageSettings );
 391 
 392     // Prepare worksheet template

 393     KIGFX::WS_PROXY_VIEW_ITEM* worksheet;
 394     worksheet = new KIGFX::WS_PROXY_VIEW_ITEM( IU_PER_MILS ,&m_Pcb->GetPageSettings(),
 395                                                m_Pcb->GetProject(), &m_Pcb->GetTitleBlock() );
 396     worksheet->SetSheetName( std::string( GetScreenDesc().mb_str() ) );
 397 
 398     BASE_SCREEN* screen = GetScreen();
 399 
 400     if( screen != NULL )
 401     {
 402         worksheet->SetSheetNumber( screen->m_ScreenNumber );
 403         worksheet->SetSheetCount( screen->m_NumberOfScreens );
 404     }
 405 
 406     if( auto board = GetBoard() )
 407         worksheet->SetFileName(  TO_UTF8( board->GetFileName() ) );
 408 
 409     // PCB_DRAW_PANEL_GAL takes ownership of the worksheet

 410     GetCanvas()->SetWorksheet( worksheet );
 411 }

 468 void PCB_EDIT_FRAME::ReFillLayerWidget()
 469 {
 470     ENUM_MAP<PCB_LAYER_ID>& layerEnum = ENUM_MAP<PCB_LAYER_ID>::Instance();
 471 
 472     layerEnum.Choices().Clear();
 473     layerEnum.Undefined( UNDEFINED_LAYER );
 474 
 475     for( LSEQ seq = LSET::AllLayersMask().Seq(); seq; ++seq )
 476         layerEnum.Map( *seq, GetBoard()->GetLayerName( *seq ) );
 477 
 478     wxWindowUpdateLocker no_update( m_Layers );
 479     m_Layers->ReFill();
 480 
 481     wxAuiPaneInfo& lyrs = m_auimgr.GetPane( m_Layers );
 482 
 483     wxSize bestz = m_Layers->GetBestSize();
 484 
 485     lyrs.MinSize( bestz );
 486     lyrs.BestSize( bestz );
 487     lyrs.FloatingSize( bestz );
 488 
 489     if( lyrs.IsDocked() )
 490         m_auimgr.Update();
 491     else
 492         m_Layers->SetSize( bestz );
 493 }

 506 void PCB_EDIT_FRAME::RecordDRCExclusions()
 507 {
 508     BOARD_DESIGN_SETTINGS& bds = GetBoard()->GetDesignSettings();
 509     bds.m_DrcExclusions.clear();
 510 
 511     for( MARKER_PCB* marker : GetBoard()->Markers() )
 512     {
 513         if( marker->IsExcluded() )
 514             bds.m_DrcExclusions.insert( marker->Serialize() );
 515     }
 516 }

 519 void PCB_EDIT_FRAME::ResolveDRCExclusions()
 520 {
 521     BOARD_DESIGN_SETTINGS& bds = GetBoard()->GetDesignSettings();
 522 
 523     for( MARKER_PCB* marker : GetBoard()->Markers() )
 524     {
 525         auto i = bds.m_DrcExclusions.find( marker->Serialize() );
 526 
 527         if( i != bds.m_DrcExclusions.end() )
 528         {
 529             marker->SetExcluded( true );
 530             bds.m_DrcExclusions.erase( i );
 531         }
 532     }
 533 
 534     BOARD_COMMIT commit( this );
 535 
 536     for( const wxString& exclusionData : bds.m_DrcExclusions )
 537     {
 538         MARKER_PCB* marker = MARKER_PCB::Deserialize( exclusionData );
 539 
 540         if( marker )
 541         {
 542             marker->SetExcluded( true );
 543             commit.Add( marker );
 544         }
 545     }
 546 
 547     bds.m_DrcExclusions.clear();
 548 
 549     commit.Push( wxEmptyString, false, false );
 550 }

 553 void PCB_EDIT_FRAME::OnCloseWindow( wxCloseEvent& aEvent )
 554 {
 555     // Shutdown blocks must be determined and vetoed as early as possible

 556     if( SupportsShutdownBlockReason() && aEvent.GetId() == wxEVT_QUERY_END_SESSION
 557             && IsContentModified() )
 558     {
 559         aEvent.Veto();
 560         return;
 561     }
 562 
 563     // First close the DRC dialog.  For some reason, if the board editor frame is destroyed

 564     // when the DRC dialog currently open, Pcbnew crashes, at least on Windows.

 565     DIALOG_DRC* open_dlg = static_cast<DIALOG_DRC*>(
 566                                         wxWindow::FindWindowByName( DIALOG_DRC_WINDOW_NAME ) );
 567 
 568     if( open_dlg )
 569         open_dlg->Close( true );
 570 
 571     if( IsContentModified() )
 572     {
 573         wxFileName fileName = GetBoard()->GetFileName();
 574         wxString msg = _( "Save changes to \"%s\" before closing?" );
 575 
 576         if( !HandleUnsavedChanges( this, wxString::Format( msg, fileName.GetFullName() ),
 577                                    [&]()->bool { return Files_io_from_id( ID_SAVE_BOARD ); } ) )
 578         {
 579             aEvent.Veto();
 580             return;
 581         }
 582     }
 583 
 584     // On Windows 7 / 32 bits, on OpenGL mode only, Pcbnew crashes

 585     // when closing this frame if a footprint was selected, and the footprint editor called

 586     // to edit this footprint, and when closing pcbnew if this footprint is still selected

 587     // See https://bugs.launchpad.net/kicad/+bug/1655858

 588     // I think this is certainly a OpenGL event fired after frame deletion, so this workaround

 589     // avoid the crash (JPC)

 590     GetCanvas()->SetEvtHandlerEnabled( false );
 591 
 592     GetCanvas()->StopDrawing();
 593 
 594     // Delete the auto save file if it exists.

 595     wxFileName fn = GetBoard()->GetFileName();
 596 
 597     // Auto save file name is the normal file name prefixed with 'GetAutoSaveFilePrefix()'.

 598     fn.SetName( GetAutoSaveFilePrefix() + fn.GetName() );
 599 
 600     // When the auto save feature does not have write access to the board file path, it falls

 601     // back to a platform specific user temporary file path.

 602     if( !fn.IsOk() || !fn.IsDirWritable() )
 603         fn.SetPath( wxFileName::GetTempDir() );
 604 
 605     wxLogTrace( traceAutoSave, "Deleting auto save file <" + fn.GetFullPath() + ">" );
 606 
 607     // Remove the auto save file on a normal close of Pcbnew.

 608     if( fn.FileExists() && !wxRemoveFile( fn.GetFullPath() ) )
 609     {
 610         wxString msg = wxString::Format( _( "The auto save file \"%s\" could not be removed!" ),
 611                                          fn.GetFullPath() );
 612         wxMessageBox( msg, Pgm().App().GetAppName(), wxOK | wxICON_ERROR, this );
 613     }
 614 
 615     // Make sure local settings are persisted

 616     SaveProjectSettings();
 617 
 618     // Do not show the layer manager during closing to avoid flicker

 619     // on some platforms (Windows) that generate useless redraw of items in

 620     // the Layer Manger

 621     if( m_show_layer_manager_tools )
 622         m_auimgr.GetPane( "LayersManager" ).Show( false );
 623 
 624     // Unlink the old project if needed

 625     GetBoard()->ClearProject();
 626 
 627     // Delete board structs and undo/redo lists, to avoid crash on exit

 628     // when deleting some structs (mainly in undo/redo lists) too late

 629     Clear_Pcb( false, true );
 630 
 631     // do not show the window because ScreenPcb will be deleted and we do not

 632     // want any paint event

 633     Show( false );
 634 
 635     // Close frame:

 636     aEvent.Skip();
 637 }

1057 bool PCB_EDIT_FRAME::FetchNetlistFromSchematic( NETLIST& aNetlist, FETCH_NETLIST_MODE aMode )
1058 {
1059     if( Kiface().IsSingle() )
1060     {
1061         DisplayError( this, _( "Cannot update the PCB because Pcbnew is opened in stand-alone "
1062                                "mode. In order to create or update PCBs from schematics, you "
1063                                "must launch the KiCad project manager and create a project." ) );
1064         return false;
1065     }
1066 
1067     // Update PCB requires a netlist. Therefore the schematic editor must be running

1068     // If this is not the case, open the schematic editor

1069     KIWAY_PLAYER* frame = Kiway().Player( FRAME_SCH, true );
1070 
1071     if( !frame->IsShown() )
1072     {
1073         wxFileName fn( Prj().GetProjectPath(), Prj().GetProjectName(),
1074                 KiCadSchematicFileExtension );
1075 
1076         // Maybe the file hasn't been converted to the new s-expression file format so

1077         // see if the legacy schematic file is still in play.

1078         if( !fn.FileExists() )
1079         {
1080             fn.SetExt( LegacySchematicFileExtension );
1081 
1082             if( !fn.FileExists() )
1083             {
1084                 DisplayError( this, _( "The schematic for this board cannot be found." ) );
1085                 return false;
1086             }
1087         }
1088 
1089         frame->OpenProjectFiles( std::vector<wxString>( 1, fn.GetFullPath() ) );
1090 
1091         // we show the schematic editor frame, because do not show is seen as

1092         // a not yet opened schematic by Kicad manager, which is not the case

1093         frame->Show( true );
1094 
1095         // bring ourselves back to the front

1096         Raise();
1097     }
1098 
1099     std::string payload;
1100 
1101     if( aMode == NO_ANNOTATION )
1102         payload = "no-annotate";
1103     else if( aMode == QUIET_ANNOTATION )
1104         payload = "quiet-annotate";
1105 
1106     Kiway().ExpressMail( FRAME_SCH, MAIL_SCH_GET_NETLIST, payload, this );
1107 
1108     try
1109     {
1110         auto lineReader = new STRING_LINE_READER( payload, _( "Eeschema netlist" ) );
1111         KICAD_NETLIST_READER netlistReader( lineReader, &aNetlist );
1112         netlistReader.LoadNetlist();
1113     }
1114     catch( const IO_ERROR& )
1115     {
1116         assert( false ); // should never happen

1117         return false;
1118     }
1119 
1120     return true;
1121 }

1136 void PCB_EDIT_FRAME::RunEeschema()
1137 {
1138     wxString   msg;
1139     wxFileName schfn( Prj().GetProjectPath(), Prj().GetProjectName(),
1140                       KiCadSchematicFileExtension );
1141 
1142     if( !schfn.FileExists() )
1143     {
1144         msg.Printf( _( "Schematic file \"%s\" not found." ), schfn.GetFullPath() );
1145         wxMessageBox( msg, _( "KiCad Error" ), wxOK | wxICON_ERROR, this );
1146         return;
1147     }
1148 
1149     if( Kiface().IsSingle() )
1150     {
1151         wxString filename = wxT( "\"" ) + schfn.GetFullPath( wxPATH_NATIVE ) + wxT( "\"" );
1152         ExecuteFile( this, EESCHEMA_EXE, filename );
1153     }
1154     else
1155     {
1156         KIWAY_PLAYER* frame = Kiway().Player( FRAME_SCH, false );
1157 
1158         // Please: note: DIALOG_EDIT_LIBENTRY_FIELDS_IN_LIB::initBuffers() calls

1159         // Kiway.Player( FRAME_SCH, true )

1160         // therefore, the schematic editor is sometimes running, but the schematic project

1161         // is not loaded, if the library editor was called, and the dialog field editor was used.

1162         // On linux, it happens the first time the schematic editor is launched, if

1163         // library editor was running, and the dialog field editor was open

1164         // On Windows, it happens always after the library editor was called,

1165         // and the dialog field editor was used

1166         if( !frame )
1167         {
1168             try
1169             {
1170                 frame = Kiway().Player( FRAME_SCH, true );
1171             }
1172             catch( const IO_ERROR& err )
1173             {
1174                 wxMessageBox( _( "Eeschema failed to load:\n" ) + err.What(),
1175                               _( "KiCad Error" ), wxOK | wxICON_ERROR, this );
1176                 return;
1177             }
1178         }
1179 
1180         if( !frame->IsShown() ) // the frame exists, (created by the dialog field editor)

1181                                 // but no project loaded.

1182         {
1183             frame->OpenProjectFiles( std::vector<wxString>( 1, schfn.GetFullPath() ) );
1184             frame->Show( true );
1185         }
1186 
1187         // On Windows, Raise() does not bring the window on screen, when iconized or not shown

1188         // On linux, Raise() brings the window on screen, but this code works fine

1189         if( frame->IsIconized() )
1190         {
1191             frame->Iconize( false );
1192             // If an iconized frame was created by Pcbnew, Iconize( false ) is not enough

1193             // to show the frame at its normal size: Maximize should be called.

1194             frame->Maximize( false );
1195         }
1196 
1197         frame->Raise();
1198     }
1199 }

1222 void PCB_EDIT_FRAME::InstallFootprintPropertiesDialog( MODULE* Module )
1223 {
1224     if( Module == NULL )
1225         return;
1226 
1227     DIALOG_FOOTPRINT_BOARD_EDITOR* dlg = new DIALOG_FOOTPRINT_BOARD_EDITOR( this, Module );
1228 
1229     int retvalue = dlg->ShowModal();
1230     /* retvalue =

1231      *  FP_PRM_EDITOR_RETVALUE::PRM_EDITOR_WANT_UPDATE_FP if update footprint

1232      *  FP_PRM_EDITOR_RETVALUE::PRM_EDITOR_WANT_EXCHANGE_FP if change footprint

1233      *  FP_PRM_EDITOR_RETVALUE::PRM_EDITOR_WANT_MODEDIT for a goto editor command

1234      *  FP_PRM_EDITOR_RETVALUE::PRM_EDITOR_EDIT_OK for normal edit

1235      */
1236 
1237     dlg->Close();
1238     dlg->Destroy();
1239 
1240     if( retvalue == DIALOG_FOOTPRINT_BOARD_EDITOR::PRM_EDITOR_EDIT_OK )
1241     {
1242         // If something edited, push a refresh request

1243         GetCanvas()->Refresh();
1244     }
1245     else if( retvalue == DIALOG_FOOTPRINT_BOARD_EDITOR::PRM_EDITOR_EDIT_BOARD_FOOTPRINT )
1246     {
1247         auto editor = (FOOTPRINT_EDIT_FRAME*) Kiway().Player( FRAME_FOOTPRINT_EDITOR, true );
1248 
1249         editor->Load_Module_From_BOARD( Module );
1250 
1251         editor->Show( true );
1252         editor->Raise();        // Iconize( false );

1253     }
1254 
1255     else if( retvalue == DIALOG_FOOTPRINT_BOARD_EDITOR::PRM_EDITOR_EDIT_LIBRARY_FOOTPRINT )
1256     {
1257         auto editor = (FOOTPRINT_EDIT_FRAME*) Kiway().Player( FRAME_FOOTPRINT_EDITOR, true );
1258 
1259         editor->LoadModuleFromLibrary( Module->GetFPID() );
1260 
1261         editor->Show( true );
1262         editor->Raise();        // Iconize( false );

1263     }
1264 
1265     else if( retvalue == DIALOG_FOOTPRINT_BOARD_EDITOR::PRM_EDITOR_WANT_UPDATE_FP )
1266     {
1267         InstallExchangeModuleFrame( Module, true, true );
1268     }
1269 
1270     else if( retvalue == DIALOG_FOOTPRINT_BOARD_EDITOR::PRM_EDITOR_WANT_EXCHANGE_FP )
1271     {
1272         InstallExchangeModuleFrame( Module, false, true );
1273     }
1274 }

1301 {

1303     if( aModule )

1315                 mod->SetLocked( aLocked );

1318         }
