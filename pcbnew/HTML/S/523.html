  47 void D_PAD::AddPrimitivePoly( const SHAPE_POLY_SET& aPoly, int aThickness )
  48 {
  49     std::vector<wxPoint> points;
  50 
  51     // If aPoly has holes, convert it to a polygon with no holes.

  52     SHAPE_POLY_SET poly_no_hole;
  53     poly_no_hole.Append( aPoly );
  54     poly_no_hole.Fracture( SHAPE_POLY_SET::PM_STRICTLY_SIMPLE );
  55 
  56     for( auto iter = poly_no_hole.CIterate(); iter; iter++ )
  57         points.emplace_back( iter->x, iter->y );
  58 
  59     AddPrimitivePoly( points, aThickness );
  60 }

 138 void D_PAD::ReplacePrimitives( const std::vector<std::shared_ptr<DRAWSEGMENT>>& aPrimitivesList )

 178 void D_PAD::addPadPrimitivesToPolygon( SHAPE_POLY_SET* aMergedPolygon, int aError ) const
 179 {
 180     SHAPE_POLY_SET polyset;
 181 
 182     for( const std::shared_ptr<DRAWSEGMENT>& primitive : m_editPrimitives )
 183         primitive->TransformShapeWithClearanceToPolygon( polyset, 0, aError );
 184 
 185     polyset.Simplify( SHAPE_POLY_SET::PM_FAST );
 186 
 187     // Merge all polygons with the initial pad anchor shape

 188     if( polyset.OutlineCount() )
 189     {
 190         aMergedPolygon->BooleanAdd( polyset, SHAPE_POLY_SET::PM_STRICTLY_SIMPLE );
 191         aMergedPolygon->Fracture( SHAPE_POLY_SET::PM_STRICTLY_SIMPLE );
 192     }
 193 }

 195 void D_PAD::MergePrimitivesAsPolygon( SHAPE_POLY_SET* aMergedPolygon ) const
 196 {
 197     auto board = GetBoard();
 198     int maxError = ARC_HIGH_DEF;
 199 
 200     if( board )
 201         maxError = board->GetDesignSettings().m_MaxError;
 202 
 203     aMergedPolygon->RemoveAllContours();
 204 
 205     // Add the anchor pad shape in aMergedPolygon, others in aux_polyset:

 206     // The anchor pad is always at 0,0

 207     switch( GetAnchorPadShape() )
 208     {
 209     case PAD_SHAPE_RECT:
 210     {
 211         SHAPE_RECT rect( -GetSize().x / 2, -GetSize().y / 2, GetSize().x, GetSize().y );
 212         aMergedPolygon->AddOutline( rect.Outline() );
 213     }
 214         break;
 215 
 216     default:
 217     case PAD_SHAPE_CIRCLE:
 218         TransformCircleToPolygon( *aMergedPolygon, wxPoint( 0, 0 ), GetSize().x / 2, maxError );
 219         break;
 220     }
 221 
 222     addPadPrimitivesToPolygon( aMergedPolygon, maxError );
 223 }

 226 bool D_PAD::GetBestAnchorPosition( VECTOR2I& aPos )
 227 {
 228     SHAPE_POLY_SET poly;
 229     addPadPrimitivesToPolygon( &poly, ARC_LOW_DEF );
 230 
 231     if( poly.OutlineCount() > 1 )
 232         return false;
 233 
 234     const int minSteps = 10;
 235     const int maxSteps = 50;
 236 
 237     int stepsX, stepsY;
 238 
 239     auto bbox = poly.BBox();
 240 
 241     if( bbox.GetWidth() < bbox.GetHeight() )
 242     {
 243         stepsX = minSteps;
 244         stepsY = minSteps * (double) bbox.GetHeight() / (double )(bbox.GetWidth() + 1);
 245     }
 246     else
 247     {
 248         stepsY = minSteps;
 249         stepsX = minSteps * (double) bbox.GetWidth() / (double )(bbox.GetHeight() + 1);
 250     }
 251 
 252     stepsX = std::max(minSteps, std::min( maxSteps, stepsX ) );
 253     stepsY = std::max(minSteps, std::min( maxSteps, stepsY ) );
 254 
 255     VECTOR2I center = bbox.Centre();
 256 
 257     int64_t minDist = std::numeric_limits<int64_t>::max();
 258     int64_t minDistEdge;
 259 
 260     if( GetAnchorPadShape() == PAD_SHAPE_CIRCLE )
 261     {
 262         minDistEdge = GetSize().x;
 263     }
 264     else
 265     {
 266         minDistEdge = std::max( GetSize().x, GetSize().y );
 267     }
 268 
 269     OPT<VECTOR2I> bestAnchor( []()->OPT<VECTOR2I> { return NULLOPT; }() );
 270 
 271     for( int y = 0; y < stepsY ; y++ )
 272     {
 273         for( int x = 0; x < stepsX; x++ )
 274         {
 275             VECTOR2I p = bbox.GetPosition();
 276             p.x += rescale( x, bbox.GetWidth(), (stepsX - 1) );
 277             p.y += rescale( y, bbox.GetHeight(), (stepsY - 1) );
 278 
 279             if( poly.Contains(p) )
 280             {
 281 
 282                 int dist = (center - p).EuclideanNorm();
 283                 int distEdge = poly.COutline(0).Distance( p, true );
 284 
 285                 if( distEdge >= minDistEdge )
 286                 {
 287                     if( dist < minDist )
 288                     {
 289                         bestAnchor = p;
 290                         minDist = dist;
 291                     }
 292                 }
 293             }
 294         }
 295     }
 296 
 297     if( bestAnchor )
 298     {
 299         aPos = *bestAnchor;
 300         return true;
 301     }
 302 
 303     return false;
 304 }
