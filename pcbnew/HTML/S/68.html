 390 void MODULE::Add( BOARD_ITEM* aBoardItem, ADD_MODE aMode )
 391 {
 392     switch( aBoardItem->Type() )
 393     {
 394     case PCB_MODULE_TEXT_T:
 395         // Only user text can be added this way.

 396         assert( static_cast<TEXTE_MODULE*>( aBoardItem )->GetType() == TEXTE_MODULE::TEXT_is_DIVERS );
 397         KI_FALLTHROUGH;
 398 
 399     case PCB_MODULE_EDGE_T:
 400         if( aMode == ADD_MODE::APPEND )
 401             m_drawings.push_back( aBoardItem );
 402         else
 403             m_drawings.push_front( aBoardItem );
 404         break;
 405 
 406     case PCB_PAD_T:
 407         if( aMode == ADD_MODE::APPEND )
 408             m_pads.push_back( static_cast<D_PAD*>( aBoardItem ) );
 409         else
 410             m_pads.push_front( static_cast<D_PAD*>( aBoardItem ) );
 411         break;
 412 
 413     case PCB_MODULE_ZONE_AREA_T:
 414         if( aMode == ADD_MODE::APPEND )
 415             m_fp_zones.push_back( static_cast<MODULE_ZONE_CONTAINER*>( aBoardItem ) );
 416         else
 417             m_fp_zones.insert( m_fp_zones.begin(), static_cast<MODULE_ZONE_CONTAINER*>( aBoardItem ) );
 418         break;
 419 
 420     default:
 421     {
 422         wxString msg;
 423         msg.Printf( wxT( "MODULE::Add() needs work: BOARD_ITEM type (%d) not handled" ),
 424                     aBoardItem->Type() );
 425         wxFAIL_MSG( msg );
 426 
 427         return;
 428     }
 429     }
 430 
 431     aBoardItem->ClearEditFlags();
 432     aBoardItem->SetParent( this );
 433 }

 593 /**

 594  * This is a bit hacky right now for performance reasons.

 595  *

 596  * We assume that most footprints will have features aligned to the axes in

 597  * the zero-rotation state.  Therefore, if the footprint is rotated, we

 598  * temporarily rotate back to zero, get the bounding box (excluding reference

 599  * and value text) and then rotate the resulting poly back to the correct

 600  * orientation.

 601  *

 602  * This is more accurate than using the AABB when most footprints are rotated

 603  * off of the axes, but less accurate than computing some kind of bounding hull.

 604  * We should consider doing that instead at some point in the future if we can

 605  * use a performant algorithm and cache the result to avoid extra computing.

 606  */
 607 SHAPE_POLY_SET MODULE::GetBoundingPoly() const
 608 {
 609     SHAPE_POLY_SET poly;
 610 
 611     double orientation = GetOrientationRadians();
 612 
 613     MODULE temp = *this;
 614     temp.SetOrientation( 0.0 );
 615     BOX2I area = temp.GetFootprintRect();
 616 
 617     poly.NewOutline();
 618 
 619     VECTOR2I p = area.GetPosition();
 620     poly.Append( p );
 621     p.x = area.GetRight();
 622     poly.Append( p );
 623     p.y = area.GetBottom();
 624     poly.Append( p );
 625     p.x = area.GetX();
 626     poly.Append( p );
 627 
 628     BOARD* board = GetBoard();
 629     if( board )
 630     {
 631         int biggest_clearance = board->GetDesignSettings().GetBiggestClearanceValue();
 632         poly.Inflate( biggest_clearance, 4 );
 633     }
 634 
 635     poly.Inflate( Millimeter2iu( 0.01 ), 4 );
 636     poly.Rotate( -orientation, m_Pos );
 637 
 638     return poly;
 639 }

 717 bool MODULE::HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy ) const
 718 {
 719     EDA_RECT arect = aRect;
 720     arect.Inflate( aAccuracy );
 721 
 722     if( aContained )
 723         return arect.Contains( m_BoundaryBox );
 724     else
 725     {
 726         // If the rect does not intersect the bounding box, skip any tests

 727         if( !aRect.Intersects( GetBoundingBox() ) )
 728             return false;
 729 
 730         // Determine if any elements in the MODULE intersect the rect

 731         for( D_PAD* pad : m_pads )
 732         {
 733             if( pad->HitTest( arect, false, 0 ) )
 734                 return true;
 735         }

 738         {
 739             if( zone->HitTest( arect, false, 0 ) )
 740                 return true;
 741         }
 742 
 743         for( BOARD_ITEM* item : m_drawings )
 744         {
 745             if( item->HitTest( arect, false, 0 ) )
 746                 return true;
 747         }
 748 
 749         // No items were hit

 750         return false;
 751     }

 867 SEARCH_RESULT MODULE::Visit( INSPECTOR inspector, void* testData, const KICAD_T scanTypes[] )
 868 {

 870     SEARCH_RESULT  result = SEARCH_RESULT::CONTINUE;
 871     const KICAD_T* p    = scanTypes;
 872     bool           done = false;

 874 #if 0 && defined(DEBUG)
 875     std::cout << GetClass().mb_str() << ' ';
 876 #endif
 877 
 878     while( !done )
 879     {
 880         stype = *p;
 881 
 882         switch( stype )
 883         {
 884         case PCB_MODULE_T:
 885             result = inspector( this, testData );  // inspect me

 886             ++p;
 887             break;
 888 
 889         case PCB_PAD_T:
 890             result = IterateForward<D_PAD*>( m_pads, inspector, testData, p );

 892             break;

 895             result = IterateForward<MODULE_ZONE_CONTAINER*>( m_fp_zones, inspector, testData, p );
 896             ++p;
 897             break;

 899         case PCB_MODULE_TEXT_T:
 900             result = inspector( m_Reference, testData );
 901 
 902             if( result == SEARCH_RESULT::QUIT )
 903                 break;
 904 
 905             result = inspector( m_Value, testData );
 906 
 907             if( result == SEARCH_RESULT::QUIT )
 908                 break;
 909 
 910             // Intentionally fall through since m_Drawings can hold TYPETEXTMODULE also

 911             KI_FALLTHROUGH;
 912 
 913         case PCB_MODULE_EDGE_T:
 914             result = IterateForward<BOARD_ITEM*>( m_drawings, inspector, testData, p );
 915 
 916             // skip over any types handled in the above call.

 917             for( ; ; )
 918             {
 919                 switch( stype = *++p )
 920                 {
 921                 case PCB_MODULE_TEXT_T:
 922                 case PCB_MODULE_EDGE_T:
 923                     continue;
 924 
 925                 default:
 926                     ;
 927                 }
 928 
 929                 break;
 930             }
 931 
 932             break;
 933 
 934         default:
 935             done = true;
 936             break;
 937         }
 938 
 939         if( result == SEARCH_RESULT::QUIT )
 940             break;
 941     }
 942

1154 void MODULE::Flip( const wxPoint& aCentre, bool aFlipLeftRight )
1155 {
1156     // Move module to its final position:

1157     wxPoint finalPos = m_Pos;
1158 
1159     // Now Flip the footprint.

1160     // Flipping a footprint is a specific transform:

1161     // it is not mirrored like a text.

1162     // We have to change the side, and ensure the footprint rotation is

1163     // modified accordint to the transform, because this parameter is used

1164     // in pick and place files, and when updating the footprint from library.

1165     // When flipped around the X axis (Y coordinates changed) orientation is negated

1166     // When flipped around the Y axis (X coordinates changed) orientation is 180 - old orient.

1167     // Because it is specfic to a footprint, we flip around the X axis, and after rotate 180 deg

1168 
1169     MIRROR( finalPos.y, aCentre.y );     /// Mirror the Y position (around the X axis)

1170 
1171     SetPosition( finalPos );
1172 
1173     // Flip layer

1174     SetLayer( FlipLayer( GetLayer() ) );
1175 
1176     // Reverse mirror orientation.

1177     m_Orient = -m_Orient;
1178 
1179     NORMALIZE_ANGLE_180( m_Orient );
1180 
1181     // Mirror pads to other side of board.

1182     for( auto pad : m_pads )
1183         pad->Flip( m_Pos, false );
1184 
1185     // Mirror zones to other side of board.

1186     for( auto zone : m_fp_zones )
1187         zone->Flip( m_Pos, aFlipLeftRight );
1188 
1189     // Mirror reference and value.

1190     m_Reference->Flip( m_Pos, false );
1191     m_Value->Flip( m_Pos, false );
1192 
1193     // Reverse mirror module graphics and texts.

1194     for( auto item : m_drawings )
1195     {
1196         switch( item->Type() )
1197         {
1198         case PCB_MODULE_EDGE_T:
1199             static_cast<EDGE_MODULE*>( item )->Flip( m_Pos, false );
1200             break;
1201 
1202         case PCB_MODULE_TEXT_T:
1203             static_cast<TEXTE_MODULE*>( item )->Flip( m_Pos, false );
1204             break;
1205 
1206         default:
1207             wxMessageBox( wxT( "MODULE::Flip() error: Unknown Draw Type" ) );
1208             break;
1209         }
1210     }
1211 
1212     // Now rotate 180 deg if required

1213     if( aFlipLeftRight )
1214         Rotate( aCentre, 1800.0 );
1215 
1216     CalculateBoundingBox();
1217 }

1268      * the footprints elements (pads, outlines, edges .. ) are moved

1270      * - the footprint position is not modified.

1271      * - the relative (local) coordinates of these items are modified

1272      * - Draw coordinates are updated

1273      */

1276     // Update (move) the relative coordinates relative to the new anchor point.

1277     wxPoint moveVector = aMoveVector;

1380     case PCB_MODULE_ZONE_AREA_T:

1385             m_fp_zones.push_back( new_zone );
1386 
1387         new_item = new_zone;
1388         break;
1389     }
1390 
1391     case PCB_MODULE_TEXT_T:
1392     {
1393         TEXTE_MODULE* new_text = new TEXTE_MODULE( *static_cast<const TEXTE_MODULE*>( aItem ) );
1394 
1395         if( new_text->GetType() == TEXTE_MODULE::TEXT_is_REFERENCE )
1396         {
1397             new_text->SetText( wxT( "${REFERENCE}" ) );
1398             new_text->SetType( TEXTE_MODULE::TEXT_is_DIVERS );
1399         }
1400         else if( new_text->GetType() == TEXTE_MODULE::TEXT_is_VALUE )
1401         {
1402             new_text->SetText( wxT( "${VALUE}" ) );
1403             new_text->SetType( TEXTE_MODULE::TEXT_is_DIVERS );
1404         }
1405 
1406         if( aAddToModule )
1407             Add( new_text );
1408 
1409         new_item = new_text;
1410 
1411         break;
1412     }
1413 
1414     case PCB_MODULE_EDGE_T:
1415     {
1416         EDGE_MODULE* new_edge = new EDGE_MODULE( *static_cast<const EDGE_MODULE*>(aItem) );
1417 
1418         if( aAddToModule )
1419             Add( new_edge );
1420 
1421         new_item = new_edge;
1422         break;
1423     }
1424 
1425     case PCB_MODULE_T:
1426         // Ignore the module itself

1427         break;
1428 
1429     default:
1430         // Un-handled item for duplication

1431         wxFAIL_MSG( "Duplication not supported for items of class " + aItem->GetClass() );
1432         break;
1433     }
1434 
1435     return new_item;
1436 }

1473         // Ensure the curr outline is closed, to calculate area

1493 double MODULE::CoverageRatio( const GENERAL_COLLECTOR& aCollector ) const
1494 {
1495     double moduleArea = GetFootprintRect().GetArea();
1496     SHAPE_POLY_SET coveredRegion;
1497     addRect( coveredRegion, GetFootprintRect() );
1498 
1499     // build list of holes (covered areas not available for selection)

1500     SHAPE_POLY_SET holes;
1501 
1502     for( auto pad : m_pads )
1503         addRect( holes, pad->GetBoundingBox() );
1504 
1505     addRect( holes, m_Reference->GetBoundingBox() );
1506     addRect( holes, m_Value->GetBoundingBox() );
1507 
1508     for( int i = 0; i < aCollector.GetCount(); ++i )
1509     {
1510         BOARD_ITEM* item = aCollector[i];
1511 
1512         switch( item->Type() )
1513         {
1514         case PCB_TEXT_T:
1515         case PCB_MODULE_TEXT_T:
1516         case PCB_TRACE_T:
1517         case PCB_ARC_T:
1518         case PCB_VIA_T:
1519             addRect( holes, item->GetBoundingBox() );
1520             break;
1521         default:
1522             break;
1523         }
1524     }
1525 
1526     SHAPE_POLY_SET uncoveredRegion;
1527 
1528     try
1529     {
1530         uncoveredRegion.BooleanSubtract( coveredRegion, holes, SHAPE_POLY_SET::PM_STRICTLY_SIMPLE );
1531         uncoveredRegion.Simplify( SHAPE_POLY_SET::PM_STRICTLY_SIMPLE );
1532         uncoveredRegion.Fracture( SHAPE_POLY_SET::PM_STRICTLY_SIMPLE );
1533     }
1534     catch( ClipperLib::clipperException& )
1535     {
1536         // better to be conservative (this will result in the disambiguate dialog)

1537         return 1.0;
1538     }
1539 
1540     double uncoveredRegionArea = polygonArea( uncoveredRegion );
1541     double coveredArea = moduleArea - uncoveredRegionArea;
1542     double ratio = ( coveredArea / moduleArea );
1543 
1544     return std::min( ratio, 1.0 );
1545 }

1552 bool MODULE::BuildPolyCourtyard()
1553 {
1554     m_poly_courtyard_front.RemoveAllContours();
1555     m_poly_courtyard_back.RemoveAllContours();
1556     // Build the courtyard area from graphic items on the courtyard.

1557     // Only PCB_MODULE_EDGE_T have meaning, graphic texts are ignored.

1558     // Collect items:

1559     std::vector< DRAWSEGMENT* > list_front;
1560     std::vector< DRAWSEGMENT* > list_back;
1561 
1562     for( auto item : GraphicalItems() )
1563     {
1564         if( item->GetLayer() == B_CrtYd && item->Type() == PCB_MODULE_EDGE_T )
1565             list_back.push_back( static_cast< DRAWSEGMENT* > ( item ) );
1566 
1567         if( item->GetLayer() == F_CrtYd && item->Type() == PCB_MODULE_EDGE_T )
1568             list_front.push_back( static_cast< DRAWSEGMENT* > ( item ) );
1569     }
1570 
1571     // Note: if no item found on courtyard layers, return true.

1572     // false is returned only when the shape defined on courtyard layers

1573     // is not convertible to a polygon

1574     if( !list_front.size() && !list_back.size() )
1575         return true;
1576 
1577     wxString error_msg;
1578 
1579     #define ARC_ERROR_MAX 0.02      /* error max in mm to approximate a arc by segments */
1580     bool success = ConvertOutlineToPolygon( list_front, m_poly_courtyard_front,
1581                                             &error_msg,
1582                                             (unsigned) Millimeter2iu( ARC_ERROR_MAX ) );
1583 
1584     if( success )
1585     {
1586         success = ConvertOutlineToPolygon( list_back, m_poly_courtyard_back,
1587                                            &error_msg,
1588                                            (unsigned) Millimeter2iu( ARC_ERROR_MAX ) );
1589     }
1590 
1591     if( !error_msg.IsEmpty() )
1592     {
1593         wxLogMessage( wxString::Format( _( "Processing courtyard of \"%s\": %s" ),
1594                                         GetChars( GetFPID().Format() ),
1595                                         error_msg) );
1596     }
1597 
1598     return success;
1599 }
