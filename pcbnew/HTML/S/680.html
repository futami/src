  43 /**

  44  * COST_ESTIMATOR

  45  *

  46  * Calculates the cost of a given line, taking corner angles and total length into account.

  47  **/
  48 class COST_ESTIMATOR
  49 {
  50 public:
  51     COST_ESTIMATOR() :
  52         m_lengthCost( 0 ),
  53         m_cornerCost( 0 )
  54     {}
  55 
  56     COST_ESTIMATOR( const COST_ESTIMATOR& aB ) :
  57         m_lengthCost( aB.m_lengthCost ),
  58         m_cornerCost( aB.m_cornerCost )
  59     {}
  60 
  61     ~COST_ESTIMATOR() {};
  62 
  63     static int CornerCost( const SEG& aA, const SEG& aB );
  64     static int CornerCost( const SHAPE_LINE_CHAIN& aLine );
  65     static int CornerCost( const LINE& aLine );
  66 
  67     void Add( LINE& aLine );
  68     void Remove( LINE& aLine );
  69     void Replace( LINE& aOldLine, LINE& aNewLine );
  70 
  71     bool IsBetter( COST_ESTIMATOR& aOther, double aLengthTolerance,
  72                    double aCornerTollerace ) const;
  73 
  74     double GetLengthCost() const { return m_lengthCost; }
  75     double GetCornerCost() const { return m_cornerCost; }
  76 
  77 private:
  78     double m_lengthCost;
  79     int m_cornerCost;
  80 };

  84 /**

  85  * OPTIMIZER

  86  *

  87  * Performs various optimizations of the lines being routed, attempting to make the lines shorter

  88  * and less cornery. There are 3 kinds of optimizations so far:

  89  * - Merging obtuse segments (MERGE_OBTUSE): tries to join together as many

  90  *   obtuse segments as possible without causing collisions

  91  * - Rerouting path between pair of line corners with a 2-segment "\__" line and iteratively repeating

  92  *   the procedure as long as the total cost of the line keeps decreasing

  93  * - "Smart Pads" - that is, rerouting pad/via exits to make them look nice (SMART_PADS).

  94  **/
  95 class OPTIMIZER
  96 {
  97 public:
  98     enum OptimizationEffort
  99     {
 100         MERGE_SEGMENTS  = 0x01,
 101         SMART_PADS      = 0x02,

 103         FANOUT_CLEANUP    = 0x08,
 104         KEEP_TOPOLOGY = 0x10,
 105         PRESERVE_VERTEX = 0x20
 106     };
 107 
 108     OPTIMIZER( NODE* aWorld );
 109     ~OPTIMIZER();
 110 
 111     ///> a quick shortcut to optmize a line without creating and setting up an optimizer

 112     static bool Optimize( LINE* aLine, int aEffortLevel, NODE* aWorld, const VECTOR2I aV = VECTOR2I(0, 0) );

 115     bool Optimize( DIFF_PAIR* aPair );
