  24 #ifndef PCBNEW_ARRAY_PAD_NAME_PROVIDER__H
  25 #define PCBNEW_ARRAY_PAD_NAME_PROVIDER__H
  26 
  27 
  28 #include <array_options.h>
  29 #include <class_module.h>
  30 
  31 /**

  32  * Simple class that sequentially provides names from an #ARRAY_OPTIONS

  33  * object, making sure that they do not conflict with names already existing

  34  * in a #MODULE.

  35  */
  36 class ARRAY_PAD_NAME_PROVIDER
  37 {
  38 public:
  39     /**

  40      * @param aMod          the module to gather existing names from (nullptr for no module)

  41      * @oaram aArrayOpts    the array options that provide the candidate names

  42      */
  43     ARRAY_PAD_NAME_PROVIDER( const MODULE* aMod, const ARRAY_OPTIONS& aArrayOpts );
  44 
  45     /**

  46      * Get the next available pad name.

  47      */
  48     wxString GetNextPadName();
  49 
  50 private:
  51     /**

  52      * Get the next name from a given index/list combo

  53      * @param  aIndex    index to start at, will be updated

  54      * @param  aExisting the set of existing names to skip

  55      * @return           the first name found that's not in aExisting

  56      */
  57     wxString getNextName( int& aIndex, const std::set<wxString>& aExisting );
  58 
  59     const ARRAY_OPTIONS& m_arrayOpts;
  60     std::set<wxString>   m_existing_pad_names;
  61     int                  m_current_pad_index;
  62 };
