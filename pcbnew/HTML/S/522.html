  39 /**

  40  * COMPONENT_NET

  41  * is used to store the component pin name to net name (and pin function)

  42  * associations stored in a netlist.

  43  */
  44 class COMPONENT_NET
  45 {
  46     wxString m_pinName;
  47     wxString m_netName;
  48     wxString m_pinFunction;
  49 
  50 public:
  51     COMPONENT_NET() {}
  52 
  53     COMPONENT_NET( const wxString& aPinName, const wxString& aNetName,
  54                    const wxString& aPinFunction ) :
  55         m_pinName( aPinName ), m_netName( aNetName ), m_pinFunction( aPinFunction )
  56     {
  57     }
  58 
  59     const wxString& GetPinName() const { return m_pinName; }
  60     const wxString& GetNetName() const { return m_netName; }
  61     const wxString& GetPinFunction() const { return m_pinFunction; }
  62 
  63     bool IsValid() const { return !m_pinName.IsEmpty(); }
  64 
  65     bool operator <( const COMPONENT_NET& aNet ) const
  66     {
  67         return m_pinName < aNet.m_pinName;
  68     }
  69 
  70     int Format( OUTPUTFORMATTER* aOut, int aNestLevel, int aCtl );
  71 };

  76 /**

  77  * COMPONENT

  78  * is used to store components and all of their related information found in a netlist.

  79  */
  80 class COMPONENT
  81 {
  82     COMPONENT_NETS m_nets;              ///< list of nets shared by the component pins

  83     wxArrayString  m_footprintFilters;  ///< Footprint filters found in netlist.

  84     int            m_pinCount;          ///< Number of pins found in netlist.

  85     wxString       m_reference;         ///< The component reference designator found in netlist.

  86     wxString       m_value;             ///< The component value found in netlist.

  87 
  88     /// A fully specified path to the component: [ sheetUUID, sheetUUID, .., componentUUID ]

  89     KIID_PATH      m_path;
  90 
  91     /// The name of the component in #m_library used when it was placed on the schematic..

  92     wxString       m_name;
  93 
  94     /// The name of the component library where #m_name was found.

  95     wxString       m_library;
  96 
  97     /// The #LIB_ID of the footprint assigned to the component.

  98     LIB_ID         m_fpid;
  99 
 100     /// The alt LIB_ID of the footprint, when there are 2 different assigned footprints,

 101     /// One from the netlist, the other from the .cmp file.

 102     /// this one is a copy of the netlist footprint assignment

 103     LIB_ID         m_altFpid;
 104 
 105     /// The #MODULE loaded for #m_fpid.

 106     std::unique_ptr< MODULE > m_footprint;
 107 
 108     static COMPONENT_NET    m_emptyNet;
 109 
 110 public:
 111     COMPONENT( const LIB_ID&    aFPID,
 112                const wxString&  aReference,
 113                const wxString&  aValue,
 114                const KIID_PATH& aPath )
 115     {
 116         m_fpid             = aFPID;
 117         m_reference        = aReference;
 118         m_value            = aValue;
 119         m_pinCount         = 0;
 120         m_path             = aPath;
 121     }
 122 
 123     virtual ~COMPONENT() { };
 124 
 125     void AddNet( const wxString& aPinName, const wxString& aNetName, const wxString& aPinFunction )
 126     {
 127         m_nets.push_back( COMPONENT_NET( aPinName, aNetName, aPinFunction ) );
 128     }
 129 
 130     unsigned GetNetCount() const { return m_nets.size(); }
 131 
 132     const COMPONENT_NET& GetNet( unsigned aIndex ) const { return m_nets[aIndex]; }
 133 
 134     const COMPONENT_NET& GetNet( const wxString& aPinName ) const;
 135 
 136     void SortPins() { sort( m_nets.begin(), m_nets.end() ); }
 137 
 138     void SetName( const wxString& aName ) { m_name = aName;}
 139     const wxString& GetName() const { return m_name; }
 140 
 141     void SetLibrary( const wxString& aLibrary ) { m_library = aLibrary; }
 142     const wxString& GetLibrary() const { return m_library; }
 143 
 144     const wxString& GetReference() const { return m_reference; }
 145 
 146     const wxString& GetValue() const { return m_value; }
 147 
 148     void SetFPID( const LIB_ID& aFPID ) { m_fpid = aFPID;  }
 149     const LIB_ID& GetFPID() const { return m_fpid; }
 150 
 151     void SetAltFPID( const LIB_ID& aFPID ) { m_altFpid = aFPID; }
 152     const LIB_ID& GetAltFPID() const { return m_altFpid; }
 153 
 154     const KIID_PATH& GetPath() const { return m_path; }
 155 
 156     void SetFootprintFilters( const wxArrayString& aFilters ) { m_footprintFilters = aFilters; }
 157     const wxArrayString& GetFootprintFilters() const { return m_footprintFilters; }
 158 
 159     void SetPinCount( int aPinCount ) { m_pinCount = aPinCount; }
 160     int GetPinCount() const { return m_pinCount; }
 161 
 162     MODULE* GetModule( bool aRelease = false )
 163     {
 164         return ( aRelease ) ? m_footprint.release() : m_footprint.get();
 165     }
 166 
 167     void SetModule( MODULE* aModule );
 168 
 169     bool IsLibSource( const wxString& aLibrary, const wxString& aName ) const
 170     {
 171         return aLibrary == m_library && aName == m_name;
 172     }
 173 
 174     void Format( OUTPUTFORMATTER* aOut, int aNestLevel, int aCtl );
 175 };

 181 /**

 182  * NETLIST

 183  * stores all of information read from a netlist along with the flags used to update

 184  * the NETLIST in the #BOARD.

 185  */
 186 class NETLIST
 187 {
 188     COMPONENTS         m_components;           ///< Components found in the netlist.

 189 
 190     /// Remove footprints from #BOARD not found in netlist when true.

 191     bool               m_deleteExtraFootprints;
 192 
 193     /// Do not actually make any changes.  Only report changes to #BOARD from netlist

 194     /// when true.

 195     bool               m_isDryRun;
 196 
 197     /// Find component by time stamp if true or reference designator if false.

 198     bool               m_findByTimeStamp;
 199 
 200     /// Replace component footprints when they differ from the netlist if true.

 201     bool               m_replaceFootprints;
 202 
 203 public:
 204     NETLIST() :
 205         m_deleteExtraFootprints( false ),
 206         m_isDryRun( false ),
 207         m_findByTimeStamp( false ),
 208         m_replaceFootprints( false )
 209     {
 210     }
 211 
 212     /**

 213      * Function IsEmpty()

 214      * @return true if there are no components in the netlist.

 215      */
 216     bool IsEmpty() const { return m_components.empty(); }
 217 
 218     /**

 219      * Function Clear

 220      * removes all components from the netlist.

 221      */
 222     void Clear() { m_components.clear(); }
 223 
 224     /**

 225      * Function GetCount

 226      * @return the number of components in the netlist.

 227      */
 228     unsigned GetCount() const { return m_components.size(); }
 229 
 230     /**

 231      * Function GetComponent

 232      * returns the #COMPONENT at \a aIndex.

 233      *

 234      * @param aIndex the index in #m_components to fetch.

 235      * @return a pointer to the #COMPONENT at \a Index.

 236      */
 237     COMPONENT* GetComponent( unsigned aIndex ) { return &m_components[ aIndex ]; }
 238 
 239     /**

 240      * Function AddComponent

 241      * adds \a aComponent to the NETLIST.

 242      *

 243      * @note If \a aComponent already exists in the NETLIST, \a aComponent is deleted

 244      *       to prevent memory leaks.  An assertion is raised in debug builds.

 245      *

 246      * @param aComponent is the COMPONENT to save to the NETLIST.

 247      */
 248     void AddComponent( COMPONENT* aComponent );
 249 
 250     /**

 251      * Function GetComponentByReference

 252      * returns a #COMPONENT by \a aReference.

 253      *

 254      * @param aReference is the reference designator the #COMPONENT.

 255      * @return a pointer to the #COMPONENT that matches \a aReference if found.  Otherwise NULL.

 256      */
 257     COMPONENT* GetComponentByReference( const wxString& aReference );
 258 
 259     /**

 260      * Function GetComponentByPath

 261      * returns a #COMPONENT by \a aPath.

 262      *

 263      * @param aPath is the KIID_PATH [ sheetUUID, .., compUUID ] of the #COMPONENT.

 264      * @return a pointer to the #COMPONENT that matches \a aPath if found.  Otherwise NULL.

 265      */
 266     COMPONENT* GetComponentByPath( const KIID_PATH& aPath );
 267 
 268     void SortByFPID();
 269 
 270     void SortByReference();
 271 
 272     void SetDeleteExtraFootprints( bool aDeleteExtraFootprints )
 273     {
 274         m_deleteExtraFootprints = aDeleteExtraFootprints;
 275     }
 276 
 277     void SetFindByTimeStamp( bool aFindByTimeStamp ) { m_findByTimeStamp = aFindByTimeStamp; }
 278 
 279     bool IsFindByTimeStamp() const { return m_findByTimeStamp; }
 280 
 281     void SetReplaceFootprints( bool aReplaceFootprints )
 282     {
 283         m_replaceFootprints = aReplaceFootprints;
 284     }
 285 
 286     bool GetReplaceFootprints() const { return m_replaceFootprints; }
 287 
 288     /**

 289      * Function AnyFootprintsLinked

 290      * @return true if any component with a footprint link is found.

 291      */
 292     bool AnyFootprintsLinked() const;
 293 
 294     void Format( const char* aDocName, OUTPUTFORMATTER* aOut, int aNestLevel, int aCtl = 0 );
 295 
 296 #define CTL_OMIT_EXTRA      (1<<0)
 297 #define CTL_OMIT_NETS       (1<<1)
 298 #define CTL_OMIT_FILTERS    (1<<2)
 299 
 300 #define CTL_FOR_BACKANNO    (CTL_OMIT_NETS | CTL_OMIT_FILTERS | CTL_OMIT_EXTRA)
 301 
 302     void FormatBackAnnotation( OUTPUTFORMATTER* aOut )
 303     {
 304         Format( "back_annotation", aOut, 0, CTL_FOR_BACKANNO );
 305     }
 306 };
