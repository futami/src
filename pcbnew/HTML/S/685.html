  36 ///> Routing modes

  37 enum PNS_MODE
  38 {
  39     RM_MarkObstacles = 0,   ///> Ignore collisions, mark obstacles

  40     RM_Shove,               ///> Only shove

  41     RM_Walkaround,          ///> Only walkaround

  42     RM_Smart                ///> Guess what's better, try to make least mess on the PCB

  43 };
  44 
  45 ///> Optimization effort

  46 enum PNS_OPTIMIZATION_EFFORT
  47 {
  48     OE_LOW = 0,
  49     OE_MEDIUM = 1,
  50     OE_FULL = 2
  51 };

  53 /**

  54  * ROUTING_SETTINGS

  55  *

  56  * Contains all persistent settings of the router, such as the mode, optimization effort, etc.

  57  */
  58 
  59 class ROUTING_SETTINGS : public NESTED_SETTINGS
  60 {
  61 public:
  62     ROUTING_SETTINGS( JSON_SETTINGS* aParent, const std::string& aPath );
  63 
  64     ///> Returns the routing mode.

  65     PNS_MODE Mode() const { return m_routingMode; }
  66 
  67     ///> Sets the routing mode.

  68     void SetMode( PNS_MODE aMode ) { m_routingMode = aMode; }
  69 
  70     ///> Returns the optimizer effort. Bigger means cleaner traces, but slower routing.

  71     PNS_OPTIMIZATION_EFFORT OptimizerEffort() const { return m_optimizerEffort; }
  72 
  73     ///> Sets the optimizer effort. Bigger means cleaner traces, but slower routing.

  74     void SetOptimizerEffort( PNS_OPTIMIZATION_EFFORT aEffort ) { m_optimizerEffort = aEffort; }
  75 
  76     ///> Returns true if shoving vias is enbled.

  77     bool ShoveVias() const { return m_shoveVias; }
  78 
  79     ///> Enables/disables shoving vias.

  80     void SetShoveVias( bool aShoveVias ) { m_shoveVias = aShoveVias; }
  81 
  82     ///> Returns true if loop (redundant track) removal is on.

  83     bool RemoveLoops() const { return m_removeLoops; }
  84 
  85     ///> Enables/disables loop (redundant track) removal.

  86     void SetRemoveLoops( bool aRemoveLoops ) { m_removeLoops = aRemoveLoops; }
  87 
  88     ///> Returns true if suggesting the finish of currently placed track is on.

  89     bool SuggestFinish() { return m_suggestFinish; }
  90 
  91     ///> Enables displaying suggestions for finishing the currently placed track.

  92     void SetSuggestFinish( bool aSuggestFinish ) { m_suggestFinish = aSuggestFinish; }
  93 
  94     ///> Returns true if Smart Pads (optimized connections) is enabled.

  95     bool SmartPads() const { return m_smartPads; }
  96 
  97     ///> Enables/disables Smart Pads (optimized connections).

  98     void SetSmartPads( bool aSmartPads ) { m_smartPads = aSmartPads; }
  99 
 100     ///> Returns true if follow mouse mode is active (permanently on for the moment).

 101     bool FollowMouse() const
 102     {
 103         return m_followMouse && !( Mode() == RM_MarkObstacles );
 104     }
 105 
 106     ///> Returns true if smoothing segments durign dragging is enabled.

 107     bool SmoothDraggedSegments() const { return m_smoothDraggedSegments; }
 108 
 109     ///> Enables/disabled smoothing segments during dragging.

 110     void SetSmoothDraggedSegments( bool aSmooth ) { m_smoothDraggedSegments = aSmooth; }
 111 
 112     ///> Returns true if jumping over unmovable obstacles is on.

 113     bool JumpOverObstacles() const { return m_jumpOverObstacles; }
 114 
 115     ///> Enables/disables jumping over unmovable obstacles.

 116     void SetJumpOverObstacles( bool aJumpOverObstacles ) { m_jumpOverObstacles = aJumpOverObstacles; }
 117 
 118     void SetStartDiagonal( bool aStartDiagonal ) { m_startDiagonal = aStartDiagonal; }
 119 
 120     bool CanViolateDRC() const { return m_canViolateDRC; }
 121     void SetCanViolateDRC( bool aViolate ) { m_canViolateDRC = aViolate; }
 122 
 123     bool GetFreeAngleMode() const { return m_freeAngleMode; }
 124 
 125     void SetFreeAngleMode( bool aEnable ) { m_freeAngleMode = aEnable; }
 126 
 127     const DIRECTION_45 InitialDirection() const;
 128 
 129     int ShoveIterationLimit() const;
 130     TIME_LIMIT ShoveTimeLimit() const;
 131 
 132     int WalkaroundIterationLimit() const { return m_walkaroundIterationLimit; };
 133     TIME_LIMIT WalkaroundTimeLimit() const;
 134 
 135     void SetInlineDragEnabled ( bool aEnable ) { m_inlineDragEnabled = aEnable; }
 136     bool InlineDragEnabled() const { return m_inlineDragEnabled; }
 137 
 138     void SetSnapToTracks( bool aSnap ) { m_snapToTracks = aSnap; }
 139     void SetSnapToPads( bool aSnap ) { m_snapToPads = aSnap; }
 140 
 141     bool GetSnapToTracks() const { return m_snapToTracks; }
 142     bool GetSnapToPads() const { return m_snapToPads; }
 143 
 144     bool GetRounded() const { return m_roundedCorners; }
 145     void SetRounded( bool aRound ) { m_roundedCorners = aRound; }
 146 
 147     bool GetOptimizeDraggedTrack() const { return m_optimizeDraggedTrack; }
 148     void SetOptimizeDraggedTrack( bool aEnable ) { m_optimizeDraggedTrack = aEnable; }
 149 
 150     void SetMinRadius( int aRadius )
 151     {
 152         m_minRadius = aRadius;
 153 
 154         if( m_maxRadius < m_minRadius )
 155             m_maxRadius = m_minRadius;
 156     }
 157 
 158     void SetMaxRadius( int aRadius )
 159     {
 160         m_maxRadius = aRadius;
 161 
 162         if( m_maxRadius < m_minRadius )
 163             m_minRadius = m_maxRadius;
 164     }
 165 
 166     int GetMinRadius() const { return m_minRadius; }
 167     int GetMaxRadius() const { return m_maxRadius; }
 168 
 169 private:
 170     bool m_shoveVias;
 171     bool m_startDiagonal;
 172     bool m_removeLoops;
 173     bool m_smartPads;
 174     bool m_suggestFinish;
 175     bool m_followMouse;
 176     bool m_jumpOverObstacles;
 177     bool m_smoothDraggedSegments;
 178     bool m_canViolateDRC;
 179     bool m_freeAngleMode;
 180     bool m_inlineDragEnabled;
 181     bool m_snapToTracks;
 182     bool m_snapToPads;
 183     bool m_roundedCorners;
 184     bool m_optimizeDraggedTrack;
 185 
 186     int m_minRadius;
 187     int m_maxRadius;
 188 
 189     PNS_MODE m_routingMode;
 190     PNS_OPTIMIZATION_EFFORT m_optimizerEffort;
 191 
 192     int m_walkaroundIterationLimit;
 193     int m_shoveIterationLimit;
 194     TIME_LIMIT m_shoveTimeLimit;
 195     TIME_LIMIT m_walkaroundTimeLimit;
 196 };
