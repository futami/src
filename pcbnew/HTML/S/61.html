  43 /**

  44  * DIMENSION

  45  *

  46  * For better understanding of the points that make a dimension:

  47  *

  48  *            m_featureLineGO  m_featureLineDO

  49  *            |                              |

  50  *            |                              |

  51  *            |                              |

  52  *            |  m_arrowG2F      m_arrowD2F  |

  53  *            | /                          \ |

  54  * m_crossBarO|/____________________________\|m_crossBarF

  55  *            |\           m_Text           /|

  56  *            | \                          / |

  57  *            |  m_arrowG1F      m_arrowD1F  |

  58  *            |                              |

  59  *            m_featureLineGF  m_featureLineDF

  60  */
  61 class DIMENSION : public BOARD_ITEM
  62 {
  63     int         m_Width;        ///< Line width

  64     int         m_Shape;        ///< Currently always 0.

  65     EDA_UNITS   m_Unit;         ///< 0 = inches, 1 = mm

  66     bool        m_UseMils;      ///< If inches, use mils.

  67     int         m_Value;        ///< value of PCB dimensions.

  68     int         m_Height;       ///< length of feature lines

  69     TEXTE_PCB   m_Text;
  70 
  71 public:
  72 // TODO private: These member should be private. they are public only due to legacy code

  73     wxPoint     m_crossBarO, m_crossBarF;
  74     wxPoint     m_featureLineGO, m_featureLineGF;
  75     wxPoint     m_featureLineDO, m_featureLineDF;
  76     wxPoint     m_arrowD1F, m_arrowD2F;
  77     wxPoint     m_arrowG1F, m_arrowG2F;
  78 
  79     DIMENSION( BOARD_ITEM* aParent );
  80 
  81     // Do not create a copy constructor & operator=.

  82     // The ones generated by the compiler are adequate.

  83 
  84     ~DIMENSION();
  85 
  86     static inline bool ClassOf( const EDA_ITEM* aItem )
  87     {
  88         return aItem && PCB_DIMENSION_T == aItem->Type();
  89     }
  90 
  91     bool IsType( const KICAD_T aScanTypes[] ) const override
  92     {
  93         if( BOARD_ITEM::IsType( aScanTypes ) )
  94             return true;
  95 
  96         for( const KICAD_T* p = aScanTypes; *p != EOT; ++p )
  97         {
  98             if( *p == PCB_LOCATE_GRAPHIC_T )
  99                 return true;
 100         }
 101 
 102         return false;
 103     }
 104 
 105     void SetValue( int aValue ) { m_Value = aValue; }
 106 
 107     int GetValue() const { return m_Value; }
 108 
 109     wxPoint GetPosition() const override;
 110 
 111     void SetPosition( const wxPoint& aPos ) override;
 112 
 113     void SetTextSize( const wxSize& aTextSize )
 114     {
 115         m_Text.SetTextSize( aTextSize );
 116     }

 118     void SetLayer( PCB_LAYER_ID aLayer ) override;

 120     void SetShape( int aShape )         { m_Shape = aShape; }
 121     int GetShape() const                { return m_Shape; }
 122 
 123     int GetWidth() const                { return m_Width; }
 124     void SetWidth( int aWidth )         { m_Width = aWidth; }
 125 
 126     /**

 127      * Function SetOrigin

 128      * Sets a new origin of the crossbar line. All remaining lines are adjusted after that.

 129      * @param aOrigin is the new point to be used as the new origin of the crossbar line.

 130      * @param aPrecision number of decimal places for mils (scaled appropriately for other units).

 131      */
 132     void SetOrigin( const wxPoint& aOrigin, int aPrecision );
 133 
 134     /**

 135      * Function GetOrigin

 136      * @return Origin of the crossbar line.

 137      */
 138     const wxPoint& GetOrigin() const
 139     {
 140         return m_featureLineGO;
 141     }
 142 
 143     /**

 144      * Function SetEnd

 145      * Sets a new end of the crossbar line. All remaining lines are adjusted after that.

 146      * @param aEnd is the new point to be used as the new end of the crossbar line.

 147      * @param aPrecision number of decimal places for mils (scaled appropriately for other units).

 148      */
 149     void SetEnd( const wxPoint& aEnd, int aPrecision );
 150 
 151     /**

 152      * Function GetEnd

 153      * @return End of the crossbar line.

 154      */
 155     const wxPoint& GetEnd()
 156     {
 157         return m_featureLineDO;
 158     }
 159 
 160     /**

 161      * Function SetHeight

 162      * Sets the length of feature lines.

 163      * @param aHeight is the new height.

 164      * @param aPrecision number of decimal places for mils (scaled appropriately for other units).

 165      */
 166     void SetHeight( int aHeight, int aPrecision );
 167 
 168     /**

 169      * Function GetHeight

 170      * Returns the length of feature lines.

 171      */
 172     int GetHeight() const
 173     {
 174         return m_Height;
 175     }
 176 
 177     /**

 178      * Function UpdateHeight

 179      * Updates stored height basing on points coordinates.

 180      */
 181     void UpdateHeight();
 182 
 183     /**

 184      * Function GetAngle

 185      * Returns angle of the crossbar.

 186      * @return Angle of the crossbar line expressed in radians.

 187      */
 188     double GetAngle() const
 189     {
 190         wxPoint delta( m_featureLineDO - m_featureLineGO );
 191 
 192         return atan2( (double)delta.y, (double)delta.x );
 193     }
 194 
 195     /**

 196      * Function AdjustDimensionDetails

 197      * Calculate coordinates of segments used to draw the dimension.

 198      * @param aPrecision number of decimal places for mils (scaled appropriately for other units).

 199      */
 200     void AdjustDimensionDetails( int aPrecision );
 201 
 202     void GetUnits( EDA_UNITS& aUnits, bool& aUseMils ) const
 203     {
 204         aUnits = m_Unit;
 205         aUseMils = m_UseMils;
 206     }

 208     void SetUnits( EDA_UNITS aUnits, bool aUseMils )

 212     }
