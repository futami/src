  29 #include "graphics_import_mgr.h"
  30 #include "graphics_import_plugin.h"
  31 
  32 #include <eda_text.h>
  33 #include <math/vector2d.h>
  34 
  35 #include <list>
  36 #include <memory>
  37 #include <vector>

  39 class EDA_ITEM;
  40 
  41 /**

  42  * Interface that creates objects representing shapes for a given data model.

  43  */
  44 class GRAPHICS_IMPORTER
  45 {
  46 public:
  47     GRAPHICS_IMPORTER();
  48 
  49     virtual ~GRAPHICS_IMPORTER()
  50     {
  51     }
  52 
  53     /**

  54      * Set the import plugin used to obtain shapes from a file.

  55      */
  56     void SetPlugin( std::unique_ptr<GRAPHICS_IMPORT_PLUGIN> aPlugin )
  57     {
  58         m_plugin = std::move( aPlugin );
  59 
  60         if( m_plugin )
  61             m_plugin->SetImporter( this );
  62     }
  63 
  64     /**

  65      * Load file and get its basic data

  66      *

  67      */
  68     bool  Load( const wxString &aFileName );
  69 
  70 
  71     /**

  72      * Import shapes from loaded file.

  73      *

  74      * It is important to have the file loaded before importing.

  75      *

  76      * @param aScale allow import graphic items with a non 1:1 import ratio

  77      * aScale = 1.0 to import graphics with their actual size.

  78      */
  79     bool Import( double aScale = 1.0 );
  80 
  81     /**

  82      * Collect warning and error messages after loading/importing.

  83      *

  84      * @return the list of messages in one string. Each message ends by '\n'

  85      */
  86     const std::string& GetMessages() const
  87     {
  88         return m_plugin->GetMessages();
  89     }
  90 
  91     /**

  92      * Get original image Width.

  93      *

  94      * @return Width of the loaded image in mm.

  95      */
  96     double GetImageWidthMM() const
  97     {
  98         return m_originalWidth;
  99     }
 100 
 101     /**

 102      * Get original image Height

 103      *

 104      * @return Height of the loaded image in mm.

 105      */
 106     double GetImageHeightMM() const
 107     {
 108         return m_originalHeight;
 109     }
 110 
 111     /**

 112      * Sets the line width for the imported outlines (in mm).

 113      */
 114     void SetLineWidthMM( double aWidth )
 115     {
 116         m_lineWidth = aWidth;
 117     }
 118 
 119     /**

 120      * Returns the line width used for importing the outlines (in mm).

 121      */
 122     double GetLineWidthMM() const
 123     {
 124         return m_lineWidth;
 125     }
 126 
 127     /**

 128      * @return the scale factor affecting the imported shapes.

 129      */
 130     double GetScale() const
 131     {
 132         return m_scale;
 133     }
 134 
 135     /**

 136      * @return the offset to add to coordinates when importing graphic items.

 137      * The offset is always in mm

 138      */
 139     const VECTOR2D& GetImportOffsetMM() const
 140     {
 141         return m_offsetCoordmm;
 142     }
 143 
 144     /**

 145      * Set the offset to add to coordinates when importing graphic items.

 146      * The offset is always in mm

 147      */
 148     void SetImportOffsetMM( const VECTOR2D& aOffset )
 149     {
 150         m_offsetCoordmm = aOffset;
 151     }
 152 
 153     /**

 154      * Set the scale factor affecting the imported shapes.

 155      * it allows conversion between imported shapes units and mm

 156      */
 157     void SetScale( double aScale )
 158     {
 159         m_scale = aScale;
 160     }
 161 
 162     /**

 163      * @return the conversion factor from mm to internal unit

 164      */
 165     double GetMillimeterToIuFactor()
 166     {
 167         return m_millimeterToIu;
 168     }
 169 
 170     /**

 171      * @return the overall scale factor to convert the imported shapes dimension to mm.

 172      */
 173     double ImportScalingFactor() const
 174     {
 175         return m_scale * m_millimeterToIu;
 176     }
 177 
 178     /**

 179      * Return the list of objects representing the imported shapes.

 180      */
 181     std::list<std::unique_ptr<EDA_ITEM>>& GetItems()
 182     {
 183         return m_items;
 184     }
 185 
 186     ///> Default line thickness (in mm)

 187     static constexpr unsigned int DEFAULT_LINE_WIDTH_DFX = 1;
 188 
 189     // Methods to be implemented by derived graphics importers

 190 
 191     /**

 192      * Create an object representing a line segment.

 193      *

 194      * @param aOrigin is the segment origin point expressed in mm.

 195      * @param aEnd is the segment end point expressed in mm.

 196      * @param aWidth is the segment thickness in mm. Use -1 for default line thickness

 197      */
 198     virtual void AddLine( const VECTOR2D& aOrigin, const VECTOR2D& aEnd, double aWidth ) = 0;
 199 
 200     /**

 201      * Create an object representing a circle.

 202      *

 203      * @param aCenter is the circle center point expressed in mm.

 204      * @param aRadius is the circle radius expressed in mm.

 205      * @param aWidth is the segment thickness in mm. Use -1 for default line thickness

 206      */
 207     virtual void AddCircle( const VECTOR2D& aCenter, double aRadius, double aWidth ) = 0;
 208 
 209     /**

 210      * Create an object representing an arc.

 211      *

 212      * @param aCenter is the arc center point expressed in mm.

 213      * @param aStart is the arc arm end point expressed in mm.

 214      * Its length is the arc radius.

 215      * @param aAngle is the arc angle expressed in degrees.

 216      * @param aWidth is the segment thickness in mm. Use -1 for default line thickness

 217      */
 218     virtual void AddArc( const VECTOR2D& aCenter, const VECTOR2D& aStart, double aAngle,
 219                          double aWidth ) = 0;
 220 
 221     virtual void AddPolygon( const std::vector< VECTOR2D >& aVertices, double aWidth ) = 0;
 222 
 223     /**

 224      * Create an object representing a text.

 225      *

 226      * @param aOrigin is the text position.

 227      * @param aText is the displayed text.

 228      * @param aHeight is the text height expressed in mm.

 229      * @param aWidth is the text width expressed in mm.

 230      * @param aOrientation is the text orientation angle expressed in degrees.

 231      * @param aHJustify is the text horizontal justification.

 232      * @param aVJustify is the text vertical justification.

 233      * @param aWidth is the segment thickness in mm. Use -1 for default line thickness

 234      */
 235     virtual void AddText( const VECTOR2D& aOrigin, const wxString& aText,
 236             double aHeight, double aWidth, double aThickness, double aOrientation,
 237             EDA_TEXT_HJUSTIFY_T aHJustify, EDA_TEXT_VJUSTIFY_T aVJustify ) = 0;
 238 
 239     /**

 240      * Create an object representing an arc.

 241      *

 242      * @param aStart is the curve start point expressed in mm.

 243      * @param aBezierControl1 is the first Bezier control point expressed in mm.

 244      * @param aBezierControl2 is the second Bezier control point expressed in mm.

 245      * @param aEnd is the curve end point expressed in mm.

 246      * @param aWidth is the segment thickness in mm. Use -1 for default line thickness

 247      */
 248     virtual void AddSpline( const VECTOR2D& aStart, const VECTOR2D& aBezierControl1,
 249                             const VECTOR2D& aBezierControl2, const VECTOR2D& aEnd,
 250                             double aWidth ) = 0;
 251 
 252 protected:
 253     ///> Adds an item to the imported shapes list.

 254     void addItem( std::unique_ptr<EDA_ITEM> aItem )
 255     {
 256         m_items.emplace_back( std::move( aItem ) );
 257     }
 258 
 259 private:
 260     ///> List of imported items

 261     std::list<std::unique_ptr<EDA_ITEM>> m_items;
 262 
 263     ///> Plugin used to load a file

 264     std::unique_ptr<GRAPHICS_IMPORT_PLUGIN> m_plugin;
 265 
 266     ///> Total image width

 267     double m_originalWidth;
 268 
 269     ///> Total image Height;

 270     double m_originalHeight;
 271 
 272     /**

 273      * Scale factor applied to the imported graphics.

 274      * 1.0 does not change the size of imported items

 275      * scale < 1.0 reduce the size of imported items

 276      */
 277     double m_scale;
 278 
 279     ///> Default line thickness for the imported graphics

 280     double m_lineWidth;
 281 
 282 protected:
 283     ///> factor to convert millimeters to Internal Units

 284     double m_millimeterToIu;
 285 
 286     ///> Offset (in mm) for imported coordinates

 287     VECTOR2D m_offsetCoordmm;
 288 };
 289 
 290 #endif /* GRAPHICS_IMPORTER_H */
