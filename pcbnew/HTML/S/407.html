  45 /// subset of eagle.drawing.board.designrules in the XML document

  46 struct ERULES
  47 {
  48     int    psElongationLong;    ///< percent over 100%.  0-> not elongated, 100->twice as wide as is tall

  49                                 ///< Goes into making a scaling factor for "long" pads.

  50 
  51     int    psElongationOffset;  ///< the offset of the hole within the "long" pad.

  52 
  53     double mvStopFrame;         ///< solder mask, expressed as percentage of the smaller pad/via dimension

  54     double mvCreamFrame;        ///< solderpaste mask, expressed as percentage of the smaller pad/via dimension

  55     int    mlMinStopFrame;      ///< solder mask, minimum size (Eagle mils, here nanometers)

  56     int    mlMaxStopFrame;      ///< solder mask, maximum size (Eagle mils, here nanometers)

  57     int    mlMinCreamFrame;     ///< solder paste mask, minimum size (Eagle mils, here nanometers)

  58     int    mlMaxCreamFrame;     ///< solder paste mask, maximum size (Eagle mils, here nanometers)

  59 
  60     int    psTop;               ///< Shape of the top pads

  61     int    psBottom;            ///< Shape of the bottom pads

  62     int    psFirst;             ///< Shape of the first pads

  63 
  64     double srRoundness;         ///< corner rounding ratio for SMD pads (percentage)

  65     int    srMinRoundness;      ///< corner rounding radius, minimum size (Eagle mils, here nanometers)

  66     int    srMaxRoundness;      ///< corner rounding radius, maximum size (Eagle mils, here nanometers)

  67 
  68     double rvPadTop;            ///< top pad size as percent of drill size

  69     // double   rvPadBottom;    ///< bottom pad size as percent of drill size

  70 
  71     double rlMinPadTop;         ///< minimum copper annulus on through hole pads

  72     double rlMaxPadTop;         ///< maximum copper annulus on through hole pads

  73 
  74     double rvViaOuter;          ///< copper annulus is this percent of via hole

  75     double rlMinViaOuter;       ///< minimum copper annulus on via

  76     double rlMaxViaOuter;       ///< maximum copper annulus on via

  77     double mdWireWire;          ///< wire to wire spacing I presume.

  78 
  79 
  80     ERULES() :
  81         psElongationLong    ( 100 ),
  82         psElongationOffset  ( 0 ),
  83 
  84         mvStopFrame         ( 1.0 ),
  85         mvCreamFrame        ( 0.0 ),
  86         mlMinStopFrame      ( Mils2iu( 4.0 ) ),
  87         mlMaxStopFrame      ( Mils2iu( 4.0 ) ),
  88         mlMinCreamFrame     ( Mils2iu( 0.0 ) ),
  89         mlMaxCreamFrame     ( Mils2iu( 0.0 ) ),
  90 
  91         psTop               ( EPAD::UNDEF ),
  92         psBottom            ( EPAD::UNDEF ),
  93         psFirst             ( EPAD::UNDEF ),
  94 
  95         srRoundness         ( 0.0 ),
  96         srMinRoundness      ( Mils2iu( 0.0 ) ),
  97         srMaxRoundness      ( Mils2iu( 0.0 ) ),
  98 
  99         rvPadTop            ( 0.25 ),
 100         // rvPadBottom      ( 0.25 ),

 101         rlMinPadTop         ( Mils2iu( 10 ) ),
 102         rlMaxPadTop         ( Mils2iu( 20 ) ),
 103 
 104         rvViaOuter          ( 0.25 ),
 105         rlMinViaOuter       ( Mils2iu( 10 ) ),
 106         rlMaxViaOuter       ( Mils2iu( 20 ) ),
 107         mdWireWire          ( 0 )
 108     {}

 110     void parse( wxXmlNode* aRules );
 111 };
