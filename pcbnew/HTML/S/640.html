  32 namespace PNS
  33 {
  34 class ROUTER;
  35 class OPTIMIZER;
  36 
  37 /**

  38  * COMPONENT_DRAGGER

  39  *

  40  * Implements component dragging algorithm.

  41  */
  42 class COMPONENT_DRAGGER : public DRAG_ALGO
  43 {
  44 public:
  45     COMPONENT_DRAGGER( ROUTER* aRouter );
  46     ~COMPONENT_DRAGGER();
  47 
  48 
  49     /**

  50      * Function Start()

  51      *

  52      * Starts routing a single track at point aP, taking item aStartItem as anchor

  53      * (unless NULL). Returns true if a dragging operation has started.

  54      */
  55     bool Start( const VECTOR2I& aP, ITEM_SET& aPrimitives ) override;
  56 
  57     /**

  58      * Function Drag()

  59      *

  60      * Drags the current segment/corner/via to the point aP.

  61      * @return true, if dragging finished with success.

  62      */
  63     bool Drag( const VECTOR2I& aP ) override;
  64 
  65     /**

  66      * Function FixRoute()

  67      *

  68      * Checks if the result of current dragging operation is correct

  69      * and eventually commits it to the world.

  70      * @return true, if dragging finished with success.

  71      */
  72     bool FixRoute() override;
  73 
  74     /**

  75      * Function CurrentNode()

  76      *

  77      * Returns the most recent world state, including all

  78      * items changed due to dragging operation.

  79      */
  80     NODE* CurrentNode() const override;
  81 
  82     /**

  83      * Function Traces()

  84      *

  85      * Returns the set of dragged items.

  86      */
  87     const ITEM_SET Traces() override;
  88 
  89 private:
  90     struct DRAGGED_CONNECTION
  91     {
  92         LINE     origLine;
  93         SOLID*   attachedPad;
  94         VECTOR2I p_orig, p_next;
  95     };
  96 
  97     std::set<SOLID*>                m_solids;
  98     std::vector<DRAGGED_CONNECTION> m_conns;
  99 
 100     bool     m_dragStatus;
 101     ITEM_SET m_draggedItems;
 102     ITEM_SET m_initialDraggedItems;
 103     NODE*    m_currentNode;
 104     VECTOR2I m_p0;
 105 };
