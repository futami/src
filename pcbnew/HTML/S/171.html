  55 DIALOG_FIND::DIALOG_FIND( PCB_BASE_FRAME* aFrame ) : DIALOG_FIND_BASE( aFrame )
  56 {
  57     m_frame = aFrame;
  58     GetSizer()->SetSizeHints( this );
  59 
  60     m_searchCombo->Append( m_frame->GetFindHistoryList() );
  61 
  62     while( m_searchCombo->GetCount() > 10 )
  63     {
  64         m_frame->GetFindHistoryList().pop_back();
  65         m_searchCombo->Delete( 9 );
  66     }
  67 
  68     if( m_searchCombo->GetCount() )
  69     {
  70         m_searchCombo->SetSelection( 0 );
  71         m_searchCombo->SelectAll();
  72     }
  73 
  74     m_matchCase->SetValue( FindOptionCase );
  75     m_matchWords->SetValue( FindOptionWords );
  76     m_wildcards->SetValue( FindOptionWildcards );
  77     m_wrap->SetValue( FindOptionWrap );
  78 
  79     m_includeTexts->SetValue( FindIncludeTexts );
  80     m_includeValues->SetValue( FindIncludeValues );
  81     m_includeReferences->SetValue( FindIncludeReferences );
  82     m_includeMarkers->SetValue( FindIncludeMarkers );
  83 
  84     m_status->SetLabel( wxEmptyString);
  85     m_upToDate = false;
  86 
  87     m_hitList.clear();
  88     m_it = m_hitList.begin();
  89 
  90     m_findNext->SetDefault();
  91     SetInitialFocus( m_searchCombo );
  92 
  93     Center();
  94 }

 117 void DIALOG_FIND::search( bool aDirection )
 118 {
 119     PCB_SCREEN* screen = m_frame->GetScreen();
 120     int         flags;
 121     int         index;
 122     wxString    msg;
 123     wxString    searchString;
 124 
 125     // Add/move the search string to the top of the list if it isn't already there

 126     searchString = m_searchCombo->GetValue();
 127     index = m_searchCombo->FindString( searchString, true );
 128 
 129     if( index == wxNOT_FOUND )
 130     {
 131         m_searchCombo->Insert( searchString, 0 );
 132         m_searchCombo->SetSelection( 0 );
 133         m_upToDate = false;
 134         m_frame->GetFindHistoryList().Insert( searchString, 0 );
 135 
 136         if( m_searchCombo->GetCount() > 10 )
 137         {
 138             m_frame->GetFindHistoryList().pop_back();
 139             m_searchCombo->Delete( 10 );
 140         }
 141     }
 142     else if( index != 0 )
 143     {
 144         m_searchCombo->Delete( index );
 145         m_searchCombo->Insert( searchString, 0 );
 146         m_searchCombo->SetSelection( 0 );
 147         m_upToDate = false;
 148 
 149         if( m_frame->GetFindHistoryList().Index( searchString ) )
 150             m_frame->GetFindHistoryList().Remove( searchString );
 151 
 152         m_frame->GetFindHistoryList().Insert( searchString, 0 );
 153     }
 154 
 155     // Update search flags

 156     flags = 0;
 157 
 158     if( FindOptionCase != m_matchCase->GetValue() )
 159     {
 160         FindOptionCase = m_matchCase->GetValue();
 161         m_upToDate = false;
 162     }
 163 
 164     if( FindOptionWords != m_matchWords->GetValue() )
 165     {
 166         FindOptionWords = m_matchWords->GetValue();
 167         m_upToDate = false;
 168     }
 169 
 170     if( FindOptionWildcards != m_wildcards->GetValue() )
 171     {
 172         FindOptionWildcards = m_wildcards->GetValue();
 173         m_upToDate = false;
 174     }
 175 
 176     FindOptionWrap = m_wrap->GetValue();
 177 
 178     if( FindIncludeTexts != m_includeTexts->GetValue() )
 179     {
 180         FindIncludeTexts = m_includeTexts->GetValue();
 181         m_upToDate = false;
 182     }
 183 
 184     if( FindIncludeValues != m_includeValues->GetValue() )
 185     {
 186         FindIncludeValues = m_includeValues->GetValue();
 187         m_upToDate = false;
 188     }
 189 
 190     if( FindIncludeReferences != m_includeReferences->GetValue() )
 191     {
 192         FindIncludeReferences = m_includeReferences->GetValue();
 193         m_upToDate = false;
 194     }
 195 
 196     if( FindIncludeMarkers != m_includeMarkers->GetValue() )
 197     {
 198         FindIncludeMarkers = m_includeMarkers->GetValue();
 199         m_upToDate = false;
 200     }
 201 
 202     if( FindOptionCase )
 203         flags |= wxFR_MATCHCASE;
 204 
 205     if( FindOptionWords )
 206         flags |= wxFR_WHOLEWORD;
 207 
 208     if( FindOptionWildcards )
 209         flags |= FR_MATCH_WILDCARD;
 210 
 211     // Search parameters

 212     m_frame->GetFindReplaceData().SetFindString( searchString );
 213     m_frame->GetFindReplaceData().SetFlags( flags );
 214 
 215     m_frame->GetToolManager()->RunAction( PCB_ACTIONS::selectionClear, true );
 216     m_frame->GetCanvas()->GetViewStart( &screen->m_StartVisu.x, &screen->m_StartVisu.y );
 217 
 218     // Refresh the list of results

 219     if( !m_upToDate )
 220     {
 221         m_status->SetLabel( _( "Searching..." ) );
 222         m_hitList.clear();
 223 
 224         if( FindIncludeTexts || FindIncludeValues || FindIncludeReferences )
 225         {
 226             for( MODULE* module : m_frame->GetBoard()->Modules() )
 227             {
 228                 if( ( module->Reference().Matches( m_frame->GetFindReplaceData(), nullptr )
 229                             && FindIncludeReferences )
 230                         || ( module->Value().Matches( m_frame->GetFindReplaceData(), nullptr )
 231                                    && FindIncludeValues ) )
 232                 {
 233                     m_hitList.push_back( module );
 234                 }
 235 
 236                 if( m_includeTexts->GetValue() )
 237                 {
 238                     for( BOARD_ITEM* item : module->GraphicalItems() )
 239                     {
 240                         TEXTE_MODULE* textItem = dynamic_cast<TEXTE_MODULE*>( item );
 241 
 242                         if( textItem
 243                                 && textItem->Matches( m_frame->GetFindReplaceData(), nullptr ) )
 244                         {
 245                             m_hitList.push_back( module );
 246                         }
 247                     }
 248                 }
 249             }
 250 
 251             if( FindIncludeTexts )
 252             {
 253                 for( BOARD_ITEM* item : m_frame->GetBoard()->Drawings() )
 254                 {
 255                     TEXTE_PCB* textItem = dynamic_cast<TEXTE_PCB*>( item );
 256 
 257                     if( textItem && textItem->Matches( m_frame->GetFindReplaceData(), nullptr ) )
 258                     {
 259                         m_hitList.push_back( textItem );
 260                     }
 261                 }
 262             }
 263         }
 264 
 265         if( FindIncludeMarkers )
 266         {
 267             for( MARKER_PCB* marker : m_frame->GetBoard()->Markers() )
 268             {
 269                 if( marker->Matches( m_frame->GetFindReplaceData(), nullptr ) )
 270                     m_hitList.push_back( marker );
 271             }
 272         }
 273 
 274         m_upToDate = true;
 275 
 276         if( aDirection )
 277             m_it = m_hitList.begin();
 278         else
 279             m_it = m_hitList.end();
 280     }
 281 
 282     // Do we want a sorting algorithm ? If so, implement it here.

 283 
 284     // Get the item to display

 285     if( m_hitList.empty() )
 286     {
 287         m_frame->SetStatusText( wxEmptyString );
 288     }
 289     else
 290     {
 291         if( aDirection )
 292         {
 293             m_it++;
 294 
 295             if( m_it == m_hitList.end() )
 296             {
 297                 if( m_wrap->GetValue() )
 298                     m_it = m_hitList.begin();
 299                 else
 300                 {
 301                     m_frame->SetStatusText( wxEmptyString );
 302                     DisplayError( this, _( "No more item to show" ), 10 );
 303                     return;
 304                 }
 305             }
 306         }
 307         else
 308         {
 309             if( m_it == m_hitList.begin() )
 310             {
 311                 if( m_wrap->GetValue() )
 312                 {
 313                     m_it = m_hitList.end();
 314                 }
 315                 else
 316                 {
 317                     m_frame->SetStatusText( wxEmptyString );
 318                     DisplayError( this, _( "No more item to show" ), 10 );
 319                     return;
 320                 }
 321             }
 322 
 323             m_it--;
 324         }
 325     }
 326 
 327     // Display the item

 328     if( m_it != m_hitList.end() )
 329     {
 330         m_frame->GetToolManager()->RunAction( PCB_ACTIONS::selectItem, true, *m_it );
 331         m_frame->FocusOnLocation( ( *m_it )->GetPosition() );
 332 
 333         msg.Printf( _( "\"%s\" found" ), searchString );
 334         m_frame->SetStatusText( msg );
 335 
 336         msg.Printf( _( "Hit(s): %ld / %lu" ), std::distance( m_hitList.begin(), m_it ),
 337                 m_hitList.size() );
 338         m_status->SetLabel( msg );
 339     }
 340     else
 341     {
 342         m_frame->SetStatusText( wxEmptyString );
 343 
 344         msg.Printf( _( "\"%s\" not found" ), searchString );
 345         DisplayError( this, msg, 10 );
 346 
 347         m_status->SetLabel( _( "No hits" ) );
 348     }
 349 
 350     if( m_highlightCallback )
 351         m_highlightCallback( GetItem() );
 352 }

 354 void DIALOG_FIND::onClose( wxCommandEvent& aEvent )
 355 {
 356     FindOptionCase = m_matchCase->GetValue();
 357     FindOptionWords = m_matchWords->GetValue();
 358     FindOptionWildcards = m_wildcards->GetValue();
 359     FindOptionWrap = m_wrap->GetValue();
 360 
 361     FindIncludeTexts = m_includeTexts->GetValue();
 362     FindIncludeValues = m_includeValues->GetValue();
 363     FindIncludeMarkers = m_includeMarkers->GetValue();
 364     FindIncludeReferences = m_includeReferences->GetValue();
 365 
 366     EndModal( 1 );
 367 }
