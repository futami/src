  89     bool IsType( const KICAD_T aScanTypes[] ) const override
  90     {
  91         if( BOARD_CONNECTED_ITEM::IsType( aScanTypes ) )
  92             return true;
  93 
  94         for( const KICAD_T* p = aScanTypes; *p != EOT; ++p )
  95         {
  96             if( m_Drill.x > 0 && m_Drill.y > 0 )
  97             {
  98                 if( *p == PCB_LOCATE_HOLE_T )
  99                     return true;
 100                 else if( *p == PCB_LOCATE_PTH_T && m_Attribute != PAD_ATTRIB_HOLE_NOT_PLATED )
 101                     return true;
 102                 else if( *p == PCB_LOCATE_NPTH_T && m_Attribute == PAD_ATTRIB_HOLE_NOT_PLATED )
 103                     return true;
 104             }
 105         }
 106 
 107         return false;
 108     }
