  46 struct CN_PTR_CMP
  47 {
  48     bool operator()( const CN_ANCHOR_PTR& aItem, const CN_ANCHOR_PTR& bItem ) const
  49     {
  50         if( aItem->Pos().x == bItem->Pos().x )
  51             return aItem->Pos().y < bItem->Pos().y;
  52         else
  53             return aItem->Pos().x < bItem->Pos().x;
  54     }
  55 };

  57 /**

  58  * RN_NET

  59  * Describes ratsnest for a single net.

  60  */
  61 class RN_NET
  62 {
  63 public:
  64     ///> Default constructor.

  65     RN_NET();
  66 
  67     /**

  68      * Function SetVisible()

  69      * Sets state of the visibility flag.

  70      * @param aEnabled is new state. True if ratsnest for a given net is meant to be displayed,

  71      * false otherwise.

  72      */
  73     void SetVisible( bool aEnabled );
  74 
  75     /**

  76      * Function MarkDirty()

  77      * Marks ratsnest for given net as 'dirty', i.e. requiring recomputation.

  78      */
  79     void MarkDirty()
  80     {
  81         m_dirty = true;
  82     }
  83 
  84     /**

  85      * Function IsDirty()

  86      * Returns state of the 'dirty' flag, indicating that ratsnest for a given net is invalid

  87      * and requires an update.

  88      * @return True if ratsnest requires recomputation, false otherwise.

  89      */
  90     bool IsDirty() const
  91     {
  92         return m_dirty;
  93     }
  94 
  95     /**

  96      * Function GetUnconnected()

  97      * Returns pointer to a vector of edges that makes ratsnest for a given net.

  98      * @return Pointer to a vector of edges that makes ratsnest for a given net.

  99      */
 100     const std::vector<CN_EDGE> GetUnconnected() const
 101     {
 102         return m_rnEdges;
 103     }
 104 
 105     /**

 106      * Function Update()

 107      * Recomputes ratsnest for a net.

 108      */
 109     void Update();
 110     void Clear();
 111 
 112     void AddCluster( std::shared_ptr<CN_CLUSTER> aCluster );
 113 
 114     unsigned int GetNodeCount() const
 115     {
 116         return m_nodes.size();
 117     }
 118 
 119     /**

 120      * Function GetNodes()

 121      * Returns list of nodes that are associated with a given item.

 122      * @param aItem is an item for which the list is generated.

 123      * @return List of associated nodes.

 124      */
 125     std::list<CN_ANCHOR_PTR> GetNodes( const BOARD_CONNECTED_ITEM* aItem ) const;
 126 
 127     const std::vector<CN_EDGE>& GetEdges() const
 128     {
 129         return m_rnEdges;
 130     }
 131 
 132     /**

 133      * Function GetAllItems()

 134      * Adds all stored items to a list.

 135      * @param aOutput is the list that will have items added.

 136      * @param aTypes determines the type of added items.

 137      */
 138     void GetAllItems( std::list<BOARD_CONNECTED_ITEM*>& aOutput, const KICAD_T aTypes[] ) const;
 139 
 140     /**

 142      * Returns a single node that lies in the shortest distance from a specific node.
