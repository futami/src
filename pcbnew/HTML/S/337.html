  35 class TEXT_ITEMS_GRID_TABLE : public wxGridTableBase
  36 {
  37     std::vector<TEXT_ITEM_INFO> m_items;
  38 
  39 public:
  40     TEXT_ITEMS_GRID_TABLE()
  41     { }
  42 
  43     int GetNumberRows() override { return m_items.size(); }
  44     int GetNumberCols() override { return 3; }
  45 
  46     wxString GetColLabelValue( int aCol ) override
  47     {
  48         switch( aCol )
  49         {
  50         case 0: return _( "Text Items" );
  51         case 1: return _( "Show" );
  52         case 2: return _( "Layer" );
  53         default: return wxEmptyString;
  54         }
  55     }
  56 
  57     wxString GetRowLabelValue( int aRow ) override
  58     {
  59         switch( aRow )
  60         {
  61         case 0:   return _( "Reference designator" );
  62         case 1:   return _( "Value" );
  63         default:  return wxEmptyString;
  64         }
  65     }
  66 
  67     bool CanGetValueAs( int aRow, int aCol, const wxString& aTypeName ) override
  68     {
  69         switch( aCol )
  70         {
  71         case 0: return aTypeName == wxGRID_VALUE_STRING;
  72         case 1: return aTypeName == wxGRID_VALUE_BOOL;
  73         case 2: return aTypeName == wxGRID_VALUE_NUMBER;
  74         default: wxFAIL; return false;
  75         }
  76     }
  77 
  78     bool CanSetValueAs( int aRow, int aCol, const wxString& aTypeName ) override
  79     {
  80         return CanGetValueAs( aRow, aCol, aTypeName );
  81     }
  82 
  83     wxString GetValue( int row, int col ) override
  84     {
  85         return m_items[row].m_Text;
  86     }
  87     void SetValue( int row, int col, const wxString& value ) override
  88     {
  89         if( col == 0 )
  90             m_items[row].m_Text = value;
  91     }
  92 
  93     bool GetValueAsBool( int row, int col ) override
  94     {
  95         return m_items[row].m_Visible;
  96     }
  97     void SetValueAsBool( int row, int col, bool value ) override
  98     {
  99         if( col == 1 )
 100             m_items[row].m_Visible = value;
 101     }
 102 
 103     long GetValueAsLong( int row, int col ) override
 104     {
 105         return m_items[row].m_Layer;
 106     }
 107     void SetValueAsLong( int row, int col, long value ) override
 108     {
 109         if( col == 2 )
 110             m_items[row].m_Layer = (int) value;
 111     }
 112 
 113     bool AppendRows( size_t aNumRows = 1 ) override
 114     {
 115         for( size_t i = 0; i < aNumRows; ++i )
 116             m_items.emplace_back( wxT( "" ), true, F_SilkS );
 117 
 118         if( GetView() )
 119         {
 120             wxGridTableMessage msg( this,  wxGRIDTABLE_NOTIFY_ROWS_APPENDED, aNumRows );
 121             GetView()->ProcessTableMessage( msg );
 122         }
 123 
 124         return true;
 125     }
 126 
 127     bool DeleteRows( size_t aPos, size_t aNumRows ) override
 128     {
 129         // aPos may be a large positive, e.g. size_t(-1), and the sum of

 130         // aPos+aNumRows may wrap here, so both ends of the range are tested.

 131         if( aPos < m_items.size() && aPos + aNumRows <= m_items.size() )
 132         {
 133             m_items.erase( m_items.begin() + aPos, m_items.begin() + aPos + aNumRows );
 134 
 135             if( GetView() )
 136             {
 137                 wxGridTableMessage msg( this, wxGRIDTABLE_NOTIFY_ROWS_DELETED, aPos, aNumRows );
 138                 GetView()->ProcessTableMessage( msg );
 139             }
 140             return true;
 141         }
 142 
 143         return false;
 144     }
 145 };

 212 bool PANEL_MODEDIT_DEFAULTS::TransferDataToWindow()
 213 {
 214     wxColour disabledColour = wxSystemSettings::GetColour( wxSYS_COLOUR_BACKGROUND );
 215 
 216 #define SET_MILS_CELL( row, col, val ) \
 217     m_layerClassesGrid->SetCellValue( row, col, StringFromValue( m_frame->GetUserUnits(), val, true, true ) )
 218 
 219 #define DISABLE_CELL( row, col ) \
 220     m_layerClassesGrid->SetReadOnly( row, col ); m_layerClassesGrid->SetCellBackgroundColour( row, col, disabledColour );
 221 
 222     for( int i = 0; i < ROW_COUNT; ++i )
 223     {
 224         SET_MILS_CELL( i, COL_LINE_THICKNESS, m_brdSettings.m_LineThickness[ i ] );
 225 
 226         if( i == ROW_EDGES || i == ROW_COURTYARD )
 227         {
 228             DISABLE_CELL( i, COL_TEXT_WIDTH );
 229             DISABLE_CELL( i, COL_TEXT_HEIGHT );
 230             DISABLE_CELL( i, COL_TEXT_THICKNESS );
 231             DISABLE_CELL( i, COL_TEXT_ITALIC );
 232         }
 233         else
 234         {
 235             SET_MILS_CELL( i, COL_TEXT_WIDTH, m_brdSettings.m_TextSize[ i ].x );
 236             SET_MILS_CELL( i, COL_TEXT_HEIGHT, m_brdSettings.m_TextSize[ i ].y );
 237             SET_MILS_CELL( i, COL_TEXT_THICKNESS, m_brdSettings.m_TextThickness[ i ] );
 238             m_layerClassesGrid->SetCellValue( i, COL_TEXT_ITALIC, m_brdSettings.m_TextItalic[ i ] ? "1" : "" );
 239 
 240             auto attr = new wxGridCellAttr;
 241             attr->SetRenderer( new wxGridCellBoolRenderer() );
 242             attr->SetReadOnly();    // not really; we delegate interactivity to GRID_TRICKS

 243             attr->SetAlignment( wxALIGN_CENTER, wxALIGN_BOTTOM );
 244             m_layerClassesGrid->SetAttr( i, COL_TEXT_ITALIC, attr );
 245         }
 246     }
 247 
 248     // Footprint defaults

 249     m_textItemsGrid->GetTable()->AppendRows( m_brdSettings.m_DefaultFPTextItems.size() );
 250 
 251     for( size_t i = 0; i < m_brdSettings.m_DefaultFPTextItems.size(); ++i )
 252     {
 253         TEXT_ITEM_INFO item = m_brdSettings.m_DefaultFPTextItems[i];
 254 
 255         m_textItemsGrid->GetTable()->SetValue( i, 0, item.m_Text );
 256         m_textItemsGrid->GetTable()->SetValueAsBool( i, 1, item.m_Visible );
 257         m_textItemsGrid->GetTable()->SetValueAsLong( i, 2, item.m_Layer );
 258     }
 259 
 260     for( int col = 0; col < m_layerClassesGrid->GetNumberCols(); col++ )
 261     {
 262         // Set the minimal width to the column label size.

 263         m_layerClassesGrid->SetColMinimalWidth( col, m_layerClassesGrid->GetVisibleWidth( col, true, false, false ) );
 264 
 265         // Set the width to see the full contents

 266         if( m_layerClassesGrid->IsColShown( col ) )
 267             m_layerClassesGrid->SetColSize( col, m_layerClassesGrid->GetVisibleWidth( col, true, true, true ) );
 268     }
 269 
 270     m_layerClassesGrid->SetRowLabelSize( m_layerClassesGrid->GetVisibleWidth( -1, true, true, true ) );
 271 
 272     Layout();
 273 
 274     return true;
 275 }

 278 bool PANEL_MODEDIT_DEFAULTS::Show( bool aShow )
 279 {
 280     bool retVal = wxPanel::Show( aShow );
 281 
 282     if( aShow )
 283     {
 284         // These *should* work in the constructor, and indeed they do if this panel is the

 285         // first displayed.  However, on OSX 3.0.5 (at least), if another panel is displayed

 286         // first then the icons will be blank unless they're set here.

 287         m_bpAdd->SetBitmap( KiBitmap( small_plus_xpm ) );
 288         m_bpDelete->SetBitmap( KiBitmap( trash_xpm ) );
 289     }
 290 
 291     if( aShow && m_firstShow )
 292     {
 293         m_layerClassesGrid->SetColSize( 0, m_layerClassesGrid->GetColSize( 0 ) + 1 );
 294         m_firstShow = false;
 295     }
 296 
 297     return retVal;
 298 }

 308 bool PANEL_MODEDIT_DEFAULTS::validateData()
 309 {
 310     if( !m_textItemsGrid->CommitPendingChanges() || !m_layerClassesGrid->CommitPendingChanges() )
 311         return false;
 312 
 313     // Test text parameters.

 314     for( int row : { ROW_SILK, ROW_COPPER, ROW_FAB, ROW_OTHERS } )
 315     {
 316         int textSize = std::min( getGridValue( row, COL_TEXT_WIDTH ),
 317                                  getGridValue( row, COL_TEXT_HEIGHT ) );
 318 
 319         if( getGridValue( row, COL_TEXT_THICKNESS ) > textSize / 4 )
 320         {
 321             wxString msg = _( "Text will not be readable with a thickness greater than\n"
 322                               "1/4 its width or height." );
 323             m_Parent->SetError( msg, this, m_layerClassesGrid, row, COL_TEXT_THICKNESS );
 324             return false;
 325         }
 326     }
 327 
 328     return true;
 329 }

 332 bool PANEL_MODEDIT_DEFAULTS::TransferDataFromWindow()
 333 {
 334     if( !validateData() )
 335         return false;
 336 
 337     for( int i = 0; i < ROW_COUNT; ++i )
 338     {
 339         m_brdSettings.m_LineThickness[ i ] = getGridValue( i, COL_LINE_THICKNESS );
 340 
 341         if( i == ROW_EDGES || i == ROW_COURTYARD )
 342             continue;
 343 
 344         m_brdSettings.m_TextSize[ i ] = wxSize( getGridValue( i, COL_TEXT_WIDTH ),
 345                                                 getGridValue( i, COL_TEXT_HEIGHT ) );
 346         m_brdSettings.m_TextThickness[ i ] = getGridValue( i, COL_TEXT_THICKNESS );
 347 
 348         wxString msg = m_layerClassesGrid->GetCellValue( i, COL_TEXT_ITALIC );
 349         m_brdSettings.m_TextItalic[ i ] = wxGridCellBoolEditor::IsTrueValue( msg );
 350     }
 351 
 352     // Footprint defaults

 353     wxGridTableBase* table = m_textItemsGrid->GetTable();
 354     m_brdSettings.m_DefaultFPTextItems.clear();
 355 
 356     for( int i = 0; i < m_textItemsGrid->GetNumberRows(); ++i )
 357     {
 358         wxString text = table->GetValue( i, 0 );
 359         bool     visible = table->GetValueAsBool( i, 1 );
 360         int      layer = (int) table->GetValueAsLong( i, 2 );
 361 
 362         m_brdSettings.m_DefaultFPTextItems.emplace_back( text, visible, layer );
 363     }
 364 
 365     if( FOOTPRINT_EDITOR_SETTINGS* cfg = m_frame->GetSettings() )
 366         cfg->m_DesignSettings = m_brdSettings;
 367 
 368     return true;
 369 }

 372 void PANEL_MODEDIT_DEFAULTS::OnAddTextItem( wxCommandEvent& event )
 373 {
 374     if( !m_textItemsGrid->CommitPendingChanges() || !m_layerClassesGrid->CommitPendingChanges() )
 375         return;
 376 
 377     wxGridTableBase* table = m_textItemsGrid->GetTable();
 378 
 379     int newRow = m_textItemsGrid->GetNumberRows();
 380     table->AppendRows( 1 );
 381     table->SetValueAsBool( newRow, 1, table->GetValueAsBool( newRow - 1, 1 ) );
 382     table->SetValueAsLong( newRow, 2, table->GetValueAsLong( newRow - 1, 2 ) );
 383 
 384     m_textItemsGrid->MakeCellVisible( newRow, 0 );
 385     m_textItemsGrid->SetGridCursor( newRow, 0 );
 386 
 387     m_textItemsGrid->EnableCellEditControl( true );
 388     m_textItemsGrid->ShowCellEditControl();
 389 }

 392 void PANEL_MODEDIT_DEFAULTS::OnDeleteTextItem( wxCommandEvent& event )
 393 {
 394     if( !m_textItemsGrid->CommitPendingChanges() || !m_layerClassesGrid->CommitPendingChanges() )
 395         return;
 396 
 397     int curRow = m_textItemsGrid->GetGridCursorRow();
 398 
 399     if( curRow < 2 )    // First two rows are required

 400         return;
 401 
 402     m_textItemsGrid->GetTable()->DeleteRows( curRow, 1 );
 403 
 404     curRow = std::max( 0, curRow - 1 );
 405     m_textItemsGrid->MakeCellVisible( curRow, m_textItemsGrid->GetGridCursorCol() );
 406     m_textItemsGrid->SetGridCursor( curRow, m_textItemsGrid->GetGridCursorCol() );
 407 }
