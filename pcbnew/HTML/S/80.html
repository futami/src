 180 bool ZONE_CONTAINER::UnFill()
 181 {
 182     bool change = false;
 183 
 184     for( std::pair<const PCB_LAYER_ID, SHAPE_POLY_SET>& pair : m_FilledPolysList )
 185     {
 186         change |= !pair.second.IsEmpty();
 187         pair.second.RemoveAllContours();
 188     }
 189 
 190     for( std::pair<const PCB_LAYER_ID, ZONE_SEGMENT_FILL>& pair : m_FillSegmList )
 191     {
 192         change |= !pair.second.empty();
 193         pair.second.clear();
 194     }
 195 
 196     m_IsFilled = false;
 197 
 198     return change;
 199 }

 236 void ZONE_CONTAINER::SetLayerSet( LSET aLayerSet )
 237 {
 238     if( GetIsKeepout() )
 239     {
 240         // Keepouts can only exist on copper layers

 241         aLayerSet &= LSET::AllCuMask();
 242     }
 243 
 244     if( aLayerSet.count() == 0 )
 245         return;
 246 
 247     if( m_layerSet != aLayerSet )
 248     {
 249         SetNeedRefill( true );
 250 
 251         UnFill();
 252 
 253         m_FillSegmList.clear();
 254         m_FilledPolysList.clear();
 255         m_RawPolysList.clear();
 256         m_filledPolysHash.clear();
 257         m_insulatedIslands.clear();
 258 
 259         for( PCB_LAYER_ID layer : aLayerSet.Seq() )
 260         {
 261             m_FillSegmList[layer]     = {};
 262             m_FilledPolysList[layer]  = {};
 263             m_RawPolysList[layer]     = {};
 264             m_filledPolysHash[layer]  = {};
 265             m_insulatedIslands[layer] = {};
 266         }
 267     }
 268 
 269     m_layerSet = aLayerSet;
 270 
 271     // Set the single layer parameter.

 272     // For zones that can be on many layers, this parameter does not have

 273     // really meaning and is a bit arbitrary if more than one layer is set.

 274     // But many functions are using it.

 275     // So we need to initialize it to a reasonable value.

 276     // Priority is F_Cu then B_Cu then to the first selected layer

 277     m_Layer = aLayerSet.Seq()[0];
 278 
 279     if( m_Layer != F_Cu && aLayerSet[B_Cu] )
 280         m_Layer = B_Cu;
 281 }

 335 int ZONE_CONTAINER::GetKeepouts( std::map<int, wxString>* aSources ) const
 336 {
 337     wxString source;
 338     int      keepouts = 0;
 339 
 340     auto     setFlag = [&]( int aFlag )
 341                        {
 342                            keepouts |= aFlag;
 343 
 344                            if( aSources )
 345                                (*aSources)[ aFlag ] = source;
 346                        };
 347 
 348     if( m_isKeepout )
 349     {
 350         if( aSources )
 351             source = _( "zone properties" );
 352 
 353         if( m_doNotAllowTracks )
 354             setFlag( DISALLOW_TRACKS );
 355 
 356         if( m_doNotAllowVias )
 357             setFlag( DISALLOW_VIAS );
 358 
 359         if( m_doNotAllowPads )
 360             setFlag( DISALLOW_PADS );
 361 
 362         if( m_doNotAllowFootprints )
 363             setFlag( DISALLOW_FOOTPRINTS );
 364 
 365         if( m_doNotAllowCopperPour )
 366             setFlag( DISALLOW_ZONES );
 367     }
 368 
 369     DRC_RULE* rule = GetRule( this, nullptr, DISALLOW_CONSTRAINT );
 370 
 371     if( rule )
 372     {
 373         if( aSources )
 374             source = wxString::Format( _( "'%s' rule" ), rule->m_Name );
 375 
 376         if( ( rule->m_DisallowFlags & DISALLOW_VIAS ) > 0 )
 377             setFlag( DISALLOW_VIAS );
 378 
 379         if( ( rule->m_DisallowFlags & DISALLOW_MICRO_VIAS ) > 0 )
 380             setFlag( DISALLOW_MICRO_VIAS );
 381 
 382         if( ( rule->m_DisallowFlags & DISALLOW_BB_VIAS ) > 0 )
 383             setFlag( DISALLOW_BB_VIAS );
 384 
 385         if( ( rule->m_DisallowFlags & DISALLOW_TRACKS ) > 0 )
 386             setFlag( DISALLOW_TRACKS );
 387 
 388         if( ( rule->m_DisallowFlags & DISALLOW_PADS ) > 0 )
 389             setFlag( DISALLOW_PADS );
 390 
 391         if( ( rule->m_DisallowFlags & DISALLOW_ZONES ) > 0 )
 392             setFlag( DISALLOW_ZONES );
 393 
 394         if( ( rule->m_DisallowFlags & DISALLOW_TEXTS ) > 0 )
 395             setFlag( DISALLOW_TEXTS );
 396 
 397         if( ( rule->m_DisallowFlags & DISALLOW_GRAPHICS ) > 0 )
 398             setFlag( DISALLOW_GRAPHICS );
 399 
 400         if( ( rule->m_DisallowFlags & DISALLOW_HOLES ) > 0 )
 401             setFlag( DISALLOW_HOLES );
 402 
 403         if( ( rule->m_DisallowFlags & DISALLOW_FOOTPRINTS ) > 0 )
 404             setFlag( DISALLOW_FOOTPRINTS );
 405     }
 406 
 407     return keepouts;
 408 }

 473 bool ZONE_CONTAINER::HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy ) const
 474 {
 475     // Calculate bounding box for zone

 476     EDA_RECT bbox = GetBoundingBox();
 477     bbox.Normalize();
 478 
 479     EDA_RECT arect = aRect;
 480     arect.Normalize();
 481     arect.Inflate( aAccuracy );
 482 
 483     if( aContained )
 484     {
 485          return arect.Contains( bbox );
 486     }
 487     else    // Test for intersection between aBox and the polygon

 488             // For a polygon, using its bounding box has no sense here

 489     {
 490         // Fast test: if aBox is outside the polygon bounding box, rectangles cannot intersect

 491         if( !arect.Intersects( bbox ) )
 492             return false;
 493 
 494         int count = m_Poly->TotalVertices();
 495 
 496         for( int ii = 0; ii < count; ii++ )
 497         {
 498             auto vertex = m_Poly->CVertex( ii );
 499             auto vertexNext = m_Poly->CVertex( ( ii + 1 ) % count );
 500 
 501             // Test if the point is within the rect

 502             if( arect.Contains( ( wxPoint ) vertex ) )
 503             {
 504                 return true;
 505             }
 506 
 507             // Test if this edge intersects the rect

 508             if( arect.Intersects( ( wxPoint ) vertex, ( wxPoint ) vertexNext ) )
 509             {
 510                 return true;
 511             }
 512         }
 513 
 514         return false;
 515     }
 516 }

 531 bool ZONE_CONTAINER::HitTestFilledArea( PCB_LAYER_ID aLayer, const wxPoint& aRefPos ) const
 532 {
 533     // Keepouts have no filled area, but it's generally nice to treat their interior as if it were

 534     // filled so that people don't have to select keepouts by their outline (which is min-width)

 535     if( GetIsKeepout() )
 536         return m_Poly->Contains( VECTOR2I( aRefPos.x, aRefPos.y ) );
 537 
 538     if( !m_FilledPolysList.count( aLayer ) )
 539         return false;
 540 
 541     return m_FilledPolysList.at( aLayer ).Contains( VECTOR2I( aRefPos.x, aRefPos.y ) );
 542 }

 545 bool ZONE_CONTAINER::HitTestCutout( const VECTOR2I& aRefPos, int* aOutlineIdx, int* aHoleIdx ) const
 546 {
 547     // Iterate over each outline polygon in the zone and then iterate over

 548     // each hole it has to see if the point is in it.

 549     for( int i = 0; i < m_Poly->OutlineCount(); i++ )
 550     {
 551         for( int j = 0; j < m_Poly->HoleCount( i ); j++ )
 552         {
 553             if( m_Poly->Hole( i, j ).PointInside( aRefPos ) )
 554             {
 555                 if( aOutlineIdx )
 556                     *aOutlineIdx = i;
 557 
 558                 if( aHoleIdx )
 559                     *aHoleIdx = j;
 560 
 561                 return true;
 562             }
 563         }
 564     }
 565 
 566     return false;
 567 }

 570 void ZONE_CONTAINER::GetMsgPanelInfo( EDA_DRAW_FRAME* aFrame, std::vector<MSG_PANEL_ITEM>& aList )
 571 {
 572     EDA_UNITS units = aFrame->GetUserUnits();
 573     wxString  msg, msg2;
 574 
 575     if( GetIsKeepout() )
 576         msg = _( "Keepout Area" );
 577     else if( IsOnCopperLayer() )
 578         msg = _( "Copper Zone" );
 579     else
 580         msg = _( "Non-copper Zone" );
 581 
 582     // Display Cutout instead of Outline for holes inside a zone

 583     // i.e. when num contour !=0

 584     // Check whether the selected corner is in a hole; i.e., in any contour but the first one.

 585     if( m_CornerSelection != nullptr && m_CornerSelection->m_contour > 0 )
 586         msg << wxT( " " ) << _( "Cutout" );
 587 
 588     aList.emplace_back( _( "Type" ), msg, DARKCYAN );
 589 
 590     if( GetIsKeepout() )
 591     {
 592         msg.Empty();
 593 
 594         if( GetDoNotAllowVias() )
 595             AccumulateDescription( msg, _( "No vias" ) );
 596 
 597         if( GetDoNotAllowTracks() )
 598             AccumulateDescription( msg, _( "No tracks" ) );
 599 
 600         if( GetDoNotAllowPads() )
 601             AccumulateDescription( msg, _( "No pads" ) );
 602 
 603         if( GetDoNotAllowCopperPour() )
 604             AccumulateDescription( msg, _( "No copper zones" ) );
 605 
 606         if( GetDoNotAllowFootprints() )
 607             AccumulateDescription( msg, _( "No footprints" ) );
 608 
 609         aList.emplace_back( MSG_PANEL_ITEM( _( "Keepout" ), msg, RED ) );
 610     }
 611     else if( IsOnCopperLayer() )
 612     {
 613         if( GetNetCode() >= 0 )
 614         {
 615             NETINFO_ITEM* net = GetNet();

 945 // Creates hatch lines inside the outline of the complex polygon

 946 // sort function used in ::Hatch to sort points by descending wxPoint.x values

 952 
 953 void ZONE_CONTAINER::Hatch()
 954 {
 955     UnHatch();
 956 
 957     if( m_hatchStyle == ZONE_HATCH_STYLE::NO_HATCH || m_hatchPitch == 0 || m_Poly->IsEmpty() )
 958         return;
 959 
 960     // define range for hatch lines

 961     int min_x = m_Poly->CVertex( 0 ).x;
 962     int max_x = m_Poly->CVertex( 0 ).x;
 963     int min_y = m_Poly->CVertex( 0 ).y;
 964     int max_y = m_Poly->CVertex( 0 ).y;
 965 
 966     for( auto iterator = m_Poly->IterateWithHoles(); iterator; iterator++ )
 967     {
 968         if( iterator->x < min_x )
 969             min_x = iterator->x;
 970 
 971         if( iterator->x > max_x )
 972             max_x = iterator->x;
 973 
 974         if( iterator->y < min_y )
 975             min_y = iterator->y;
 976 
 977         if( iterator->y > max_y )
 978             max_y = iterator->y;
 979     }
 980 
 981     // Calculate spacing between 2 hatch lines

 982     int spacing;
 983 
 984     if( m_hatchStyle == ZONE_HATCH_STYLE::DIAGONAL_EDGE )
 985         spacing = m_hatchPitch;
 986     else
 987         spacing = m_hatchPitch * 2;
 988 
 989     // set the "length" of hatch lines (the length on horizontal axis)

 990     int  hatch_line_len = m_hatchPitch;
 991 
 992     // To have a better look, give a slope depending on the layer

 993     LAYER_NUM layer = GetLayer();
 994     int     slope_flag = (layer & 1) ? 1 : -1;  // 1 or -1

 995     double  slope = 0.707106 * slope_flag;      // 45 degrees slope

 996     int     max_a, min_a;
 997 
 998     if( slope_flag == 1 )
 999     {
1000         max_a   = KiROUND( max_y - slope * min_x );
1001         min_a   = KiROUND( min_y - slope * max_x );
1002     }
1003     else
1004     {
1005         max_a   = KiROUND( max_y - slope * max_x );
1006         min_a   = KiROUND( min_y - slope * min_x );
1007     }
1008 
1009     min_a = (min_a / spacing) * spacing;
1010 
1011     // calculate an offset depending on layer number,

1012     // for a better look of hatches on a multilayer board

1013     int offset = (layer * 7) / 8;
1014     min_a += offset;
1015 
1016     // loop through hatch lines

1017     #define MAXPTS 200      // Usually we store only few values per one hatch line

1018                             // depending on the complexity of the zone outline

1019 
1020     static std::vector<VECTOR2I> pointbuffer;
1021     pointbuffer.clear();
1022     pointbuffer.reserve( MAXPTS + 2 );
1023 
1024     for( int a = min_a; a < max_a; a += spacing )
1025     {
1026         // get intersection points for this hatch line

1027 
1028         // Note: because we should have an even number of intersections with the

1029         // current hatch line and the zone outline (a closed polygon,

1030         // or a set of closed polygons), if an odd count is found

1031         // we skip this line (should not occur)

1032         pointbuffer.clear();
1033 
1034         // Iterate through all vertices

1035         for( auto iterator = m_Poly->IterateSegmentsWithHoles(); iterator; iterator++ )
1036         {
1037             double  x, y, x2, y2;
1038             int     ok;
1039 
1040             SEG segment = *iterator;
1041 
1042             ok = FindLineSegmentIntersection( a, slope,
1043                                               segment.A.x, segment.A.y,
1044                                               segment.B.x, segment.B.y,
1045                                               &x, &y, &x2, &y2 );
1046 
1047               if( ok )
1048               {
1049                   VECTOR2I point( KiROUND( x ), KiROUND( y ) );
1050                   pointbuffer.push_back( point );
1051               }
1052 
1053               if( ok == 2 )
1054               {
1055                   VECTOR2I point( KiROUND( x2 ), KiROUND( y2 ) );
1056                   pointbuffer.push_back( point );
1057               }
1058 
1059               if( pointbuffer.size() >= MAXPTS )    // overflow

1060               {
1061                   wxASSERT( 0 );
1062                   break;
1063               }
1064         }
1065 
1066         // ensure we have found an even intersection points count

1067         // because intersections are the ends of segments

1068         // inside the polygon(s) and a segment has 2 ends.

1069         // if not, this is a strange case (a bug ?) so skip this hatch

1070         if( pointbuffer.size() % 2 != 0 )
1071             continue;
1072 
1073         // sort points in order of descending x (if more than 2) to

1074         // ensure the starting point and the ending point of the same segment

1075         // are stored one just after the other.

1076         if( pointbuffer.size() > 2 )
1077             sort( pointbuffer.begin(), pointbuffer.end(), sortEndsByDescendingX );
1078 
1079         // creates lines or short segments inside the complex polygon

1080         for( unsigned ip = 0; ip < pointbuffer.size(); ip += 2 )
1081         {
1082             int dx = pointbuffer[ip + 1].x - pointbuffer[ip].x;
1083 
1084             // Push only one line for diagonal hatch,

1085             // or for small lines < twice the line length

1086             // else push 2 small lines

1087             if( m_hatchStyle == ZONE_HATCH_STYLE::DIAGONAL_FULL
1088                     || std::abs( dx ) < 2 * hatch_line_len )
1089             {
1090                 m_HatchLines.emplace_back( SEG( pointbuffer[ip], pointbuffer[ip + 1] ) );
1091             }
1092             else
1093             {
1094                 double dy = pointbuffer[ip + 1].y - pointbuffer[ip].y;
1095                 slope = dy / dx;
1096 
1097                 if( dx > 0 )
1098                     dx = hatch_line_len;
1099                 else
1100                     dx = -hatch_line_len;
1101 
1102                 int x1 = KiROUND( pointbuffer[ip].x + dx );
1103                 int x2 = KiROUND( pointbuffer[ip + 1].x - dx );
1104                 int y1 = KiROUND( pointbuffer[ip].y + dx * slope );
1105                 int y2 = KiROUND( pointbuffer[ip + 1].y - dx * slope );
1106 
1107                 m_HatchLines.emplace_back( SEG( pointbuffer[ip].x, pointbuffer[ip].y, x1, y1 ) );
1108 
1109                 m_HatchLines.emplace_back( SEG( pointbuffer[ip+1].x, pointbuffer[ip+1].y, x2, y2 ) );
1110             }
1111         }
1112     }
1113 }

1155 /*

1156  * Some intersecting zones, despite being on the same layer with the same net, cannot be

1157  * merged due to other parameters such as fillet radius.  The copper pour will end up

1158  * effectively merged though, so we want to keep the corners of such intersections sharp.

1159  */
1160 void ZONE_CONTAINER::GetColinearCorners( BOARD* aBoard, std::set<VECTOR2I>& aCorners )
1161 {
1162     int epsilon = Millimeter2iu( 0.001 );
1163 
1164     // Things get messy when zone of different nets intersect.  To do it right we'd need to

1165     // run our colinear test with the final filled regions rather than the outline regions.

1166     // However, since there's no order dependance the only way to do that is to iterate

1167     // through successive zone fills until the results are no longer changing -- and that's

1168     // not going to happen.  So we punt and ignore any "messy" corners.

1169     std::set<VECTOR2I> colinearCorners;
1170     std::set<VECTOR2I> messyCorners;
1171 
1172     for( ZONE_CONTAINER* candidate : aBoard->Zones() )
1173     {
1174         if( candidate == this )
1175             continue;
1176 
1177         if( candidate->GetLayerSet() != GetLayerSet() )
1178             continue;
1179 
1180         if( candidate->GetIsKeepout() != GetIsKeepout() )
1181             continue;
1182 
1183         for( auto iter = m_Poly->CIterate(); iter; iter++ )
1184         {
1185             if( candidate->m_Poly->Collide( iter.Get(), epsilon ) )
1186             {
1187                 if( candidate->GetNetCode() == GetNetCode() )
1188                     colinearCorners.insert( VECTOR2I( iter.Get() ) );
1189                 else
1190                     messyCorners.insert( VECTOR2I( iter.Get() ) );
1191             }
1192         }
1193     }
1194 
1195     for( VECTOR2I corner : colinearCorners )
1196     {
1197         if( messyCorners.count( corner ) == 0 )
1198             aCorners.insert( corner );
1199     }
1200 }

1203 bool ZONE_CONTAINER::BuildSmoothedPoly( SHAPE_POLY_SET& aSmoothedPoly,
1204                                         std::set<VECTOR2I>* aPreserveCorners ) const
1205 {
1206     if( GetNumCorners() <= 2 )  // malformed zone. polygon calculations do not like it ...

1207         return false;
1208 
1209     // Make a smoothed polygon out of the user-drawn polygon if required

1210     switch( m_cornerSmoothingType )
1211     {
1212     case ZONE_SETTINGS::SMOOTHING_CHAMFER:
1213         aSmoothedPoly = m_Poly->Chamfer( m_cornerRadius, aPreserveCorners );
1214         break;
1215 
1216     case ZONE_SETTINGS::SMOOTHING_FILLET:
1217     {
1218         auto board = GetBoard();
1219         int maxError = ARC_HIGH_DEF;
1220 
1221         if( board )
1222             maxError = board->GetDesignSettings().m_MaxError;
1223 
1224         aSmoothedPoly = m_Poly->Fillet( m_cornerRadius, maxError, aPreserveCorners );
1225         break;
1226     }
1227     default:
1228         // Acute angles between adjacent edges can create issues in calculations,

1229         // in inflate/deflate outlines transforms, especially when the angle is very small.

1230         // We can avoid issues by creating a very small chamfer which remove acute angles,

1231         // or left it without chamfer and use only CPOLYGONS_LIST::InflateOutline to create

1232         // clearance areas

1233         aSmoothedPoly = m_Poly->Chamfer( Millimeter2iu( 0.0 ), aPreserveCorners );
1234         break;
1235     }
1236 
1237     return true;
1238 };

1240 
1241 double ZONE_CONTAINER::CalculateFilledArea()
1242 {
1243     m_area = 0.0;
1244 
1245     // Iterate over each outline polygon in the zone and then iterate over

1246     // each hole it has to compute the total area.

1247     for( std::pair<const PCB_LAYER_ID, SHAPE_POLY_SET>& pair : m_FilledPolysList )
1248     {
1249         SHAPE_POLY_SET& poly = pair.second;
1250 
1251         for( int i = 0; i < poly.OutlineCount(); i++ )
1252         {
1253             m_area += poly.Outline( i ).Area();
1254 
1255             for( int j = 0; j < poly.HoleCount( i ); j++ )
1256             {
1257                 m_area -= poly.Hole( i, j ).Area();
1258             }
1259         }
1260     }
1261 
1262     return m_area;
1263 }

1266 /* Function TransformOutlinesShapeWithClearanceToPolygon

1267  * Convert the zone filled areas polygons to polygons

1268  * inflated (optional) by max( aClearanceValue, the zone clearance)

1269  * and copy them in aCornerBuffer

1270  * @param aClearance       the clearance around outlines

1271  * @param aPreserveCorners an optional set of corners which should not be chamfered/filleted

1272  */
1273 void ZONE_CONTAINER::TransformOutlinesShapeWithClearanceToPolygon( SHAPE_POLY_SET& aCornerBuffer,
1274         int aClearance, std::set<VECTOR2I>* aPreserveCorners ) const
1275 {
1276     // Creates the zone outline polygon (with holes if any)

1277     SHAPE_POLY_SET polybuffer;
1278     BuildSmoothedPoly( polybuffer, aPreserveCorners );
1279 
1280     // Calculate the polygon with clearance

1281     // holes are linked to the main outline, so only one polygon is created.

1282     if( aClearance )
1283     {
1284         BOARD* board = GetBoard();
1285         int maxError = ARC_HIGH_DEF;
1286 
1287         if( board )
1288             maxError = board->GetDesignSettings().m_MaxError;
1289 
1290         int segCount = std::max( GetArcToSegmentCount( aClearance, maxError, 360.0 ), 3 );
1291         polybuffer.Inflate( aClearance, segCount );
1292     }
1293 
1294     polybuffer.Fracture( SHAPE_POLY_SET::PM_FAST );
1295     aCornerBuffer.Append( polybuffer );
1296 }
