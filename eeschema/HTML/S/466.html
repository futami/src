  30 #ifndef _TRANSFORM_H_
  31 #define _TRANSFORM_H_
  32 
  33 #include <wx/gdicmn.h>
  34 
  35 class EDA_RECT;
  36 
  37 /**

  38  * for transforming drawing coordinates for a wxDC device context.

  39  *

  40  * This probably should be a base class with all pure virtual methods and a WXDC_TRANSFORM

  41  * derived class.  Then in the future if some new device context is used, a new transform could

  42  * be derived from the base class and all the drawable objects would have to do is provide

  43  * overloaded draw methods to use the new transorm.

  44  */
  45 class TRANSFORM
  46 {
  47 public:
  48     int x1;
  49     int y1;
  50     int x2;
  51     int y2;
  52 
  53     /**

  54      * The default construct creates a transform that draws object is the normal orientation.

  55      */
  56     TRANSFORM() : x1( 1 ), y1( 0 ), x2( 0 ), y2( -1 ) {}
  57 
  58     TRANSFORM( int ax1, int ay1, int ax2, int ay2 ) : x1( ax1 ), y1( ay1 ), x2( ax2 ), y2( ay2 ) {}
  59 
  60     bool operator==( const TRANSFORM& aTransform ) const;
  61 
  62     bool operator!=( const TRANSFORM& aTransform ) const { return !( *this == aTransform ); }
  63 
  64    /**

  65     * Calculate a new coordinate according to the mirror/rotation transform.

  66     * Useful to calculate actual coordinates of a point

  67     * from coordinates relative to a component

  68     * which are given for a non rotated, non mirrored item

  69     * @param aPoint = The position to transform

  70     * @return The transformed coordinate.

  71     */
  72     wxPoint TransformCoordinate( const wxPoint& aPoint ) const;
  73 
  74    /**

  75     * Calculate a new rect according to the mirror/rotation transform.

  76     * Useful to calculate actual coordinates of a point

  77     * from coordinates relative to a component

  78     * which are given for a non rotated, non mirrored item

  79     * @param aRect = The rectangle to transform

  80     * @return The transformed rectangle.

  81     */
  82     EDA_RECT TransformCoordinate( const EDA_RECT& aRect ) const;
  83 
  84     /**

  85     * Calculate the Inverse mirror/rotation transform.

  86     * Useful to calculate coordinates relative to a component

  87     * which must be for a non rotated, non mirrored item

  88     * from the actual coordinate.

  89     * @return The inverse transform.

  90     */
  91     TRANSFORM InverseTransform( ) const;
  92 
  93    /**

  94     * Calculate new angles according to the transform.

  95     *

  96     * @param aAngle1 = The first angle to transform

  97     * @param aAngle2 = The second angle to transform

  98     * @return True if the angles were swapped during the transform.

  99     */
 100     bool MapAngles( int* aAngle1, int* aAngle2 ) const;
 101 };

 104 #endif    // _TRANSFORM_H_
