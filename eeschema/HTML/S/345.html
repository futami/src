  38 /**

  39  * Object to handle a bitmap image that can be inserted in a schematic.

  40  */
  41 
  42 class SCH_BITMAP : public SCH_ITEM
  43 {
  44     wxPoint      m_pos;                 // XY coordinates of center of the bitmap

  45     BITMAP_BASE* m_image;               // the BITMAP_BASE item

  46 
  47 public:
  48     SCH_BITMAP( const wxPoint& pos = wxPoint( 0, 0 ) );
  49 
  50     SCH_BITMAP( const SCH_BITMAP& aSchBitmap );
  51 
  52     ~SCH_BITMAP()
  53     {
  54         delete m_image;
  55     }
  56 
  57     SCH_BITMAP& operator=( const SCH_ITEM& aItem );
  58 
  59     BITMAP_BASE* GetImage()
  60     {
  61         wxCHECK_MSG( m_image != NULL, NULL, "Invalid SCH_BITMAP init, m_image is NULL." );
  62         return m_image;
  63     }
  64 
  65     /**

  66      * @return the image "zoom" value

  67      *  scale = 1.0 = original size of bitmap.

  68      *  scale < 1.0 = the bitmap is drawn smaller than its original size.

  69      *  scale > 1.0 = the bitmap is drawn bigger than its original size.

  70      */
  71     double GetImageScale() const
  72     {
  73         return m_image->GetScale();
  74     }
  75 
  76     void SetImageScale( double aScale )
  77     {
  78         m_image->SetScale( aScale );
  79     }
  80 
  81     static inline bool ClassOf( const EDA_ITEM* aItem )
  82     {
  83         return aItem && SCH_BITMAP_T == aItem->Type();
  84     }
  85 
  86     wxString GetClass() const override
  87     {
  88         return wxT( "SCH_BITMAP" );
  89     }
  90 
  91     /**

  92      * @return the actual size (in user units, not in pixels) of the image

  93      */
  94     wxSize GetSize() const;
  95 
  96     const EDA_RECT GetBoundingBox() const override;
  97 
  98     void SwapData( SCH_ITEM* aItem ) override;
  99 
 100     void Print( RENDER_SETTINGS* aSettings, const wxPoint& aOffset ) override;
 101 
 102     /// @copydoc VIEW_ITEM::ViewGetLayers()

 103     virtual void ViewGetLayers( int aLayers[], int& aCount ) const override;
 104 
 105    /**

 106      * Reads and stores an image file. Init the bitmap used to draw this item

 107      * format.

 108      *

 109      * @param aFullFilename The full filename of the image file to read.

 110      * @return bool - true if success reading else false.

 111      */
 112     bool ReadImageFile( const wxString& aFullFilename );
 113 
 114     void Move( const wxPoint& aMoveVector ) override
 115     {
 116         m_pos += aMoveVector;
 117     }
 118 
 119     /**

 120      * Virtual function IsMovableFromAnchorPoint

 121      * Return true for items which are moved with the anchor point at mouse cursor

 122      *  and false for items moved with no reference to anchor

 123      * @return false for a bus entry

 124      */
 125     bool IsMovableFromAnchorPoint() override { return false; }
 126 
 127     void MirrorY( int aYaxis_position ) override;
 128     void MirrorX( int aXaxis_position ) override;
 129     void Rotate( wxPoint aPosition ) override;
 130 
 131     wxString GetSelectMenuText( EDA_UNITS aUnits ) const override
 132     {
 133         return wxString( _( "Image" ) );
 134     }
 135 
 136     BITMAP_DEF GetMenuImage() const override;
 137 
 138     void GetMsgPanelInfo( EDA_DRAW_FRAME* aFrame, std::vector<MSG_PANEL_ITEM>& aList ) override;
 139 
 140     wxPoint GetPosition() const override { return m_pos; }
 141     void SetPosition( const wxPoint& aPosition ) override { m_pos = aPosition; }
 142 
 143     bool HitTest( const wxPoint& aPosition, int aAccuracy = 0 ) const override;
 144     bool HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy = 0 ) const override;
 145 
 146     void Plot( PLOTTER* aPlotter ) override;
 147 
 148     EDA_ITEM* Clone() const override;
 149 
 150 #if defined(DEBUG)
 151     void Show( int nestLevel, std::ostream& os ) const override;
 152 #endif
 153 };
