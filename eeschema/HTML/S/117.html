 258 NETLIST_DIALOG::NETLIST_DIALOG( SCH_EDIT_FRAME* parent ) :
 259     NETLIST_DIALOG_BASE( parent )
 260 {
 261     m_Parent = parent;
 262 
 263     SCHEMATIC_SETTINGS& settings = m_Parent->Schematic().Settings();
 264 
 265     m_DefaultNetFmtName = settings.m_NetFormatName;
 266 
 267     for( NETLIST_PAGE_DIALOG*& page : m_PanelNetType)
 268         page = NULL;
 269 
 270     // Add notebook pages:

 271     m_asFormatSelected = false;     // Will be set to true, if a format is selected

 272 
 273     m_PanelNetType[PANELPCBNEW] = new NETLIST_PAGE_DIALOG( m_NoteBook, wxT( "Pcbnew" ),
 274                                                            NET_TYPE_PCBNEW );
 275 
 276     m_PanelNetType[PANELORCADPCB2] = new NETLIST_PAGE_DIALOG( m_NoteBook, wxT( "OrcadPCB2" ),
 277                                                               NET_TYPE_ORCADPCB2 );
 278 
 279     m_PanelNetType[PANELCADSTAR] = new NETLIST_PAGE_DIALOG( m_NoteBook, wxT( "CadStar" ),
 280                                                             NET_TYPE_CADSTAR );
 281 
 282     InstallPageSpice();
 283     InstallCustomPages();
 284 
 285     // Ensure a netlist format is selected:

 286     if( !m_asFormatSelected )
 287     {
 288         m_PanelNetType[PANELPCBNEW]->m_IsCurrentFormat->SetValue( true );
 289         m_NoteBook->SetSelection( PANELPCBNEW );
 290         m_DefaultNetFmtName = m_PanelNetType[PANELPCBNEW]->GetPageNetFmtName();
 291         // call OnNetlistTypeSelection to update some widgets.

 292         // SetSelection() do nothing if the current page is already PANELPCBNEW

 293         wxNotebookEvent event;
 294         OnNetlistTypeSelection( event );
 295     }
 296 
 297     // We use a sdbSizer here to get the order right, which is platform-dependent

 298     m_sdbSizer2OK->SetLabel( _( "Export Netlist" ) );
 299     m_sdbSizer2Cancel->SetLabel( _( "Close" ) );
 300     m_buttonSizer->Layout();
 301 
 302     m_sdbSizer2OK->SetDefault();
 303 
 304     // Now all widgets have the size fixed, call FinishDialogSettings

 305     FinishDialogSettings();
 306 }

 324 void NETLIST_DIALOG::InstallCustomPages()
 325 {
 326     NETLIST_PAGE_DIALOG* currPage;
 327 
 328     auto cfg = dynamic_cast<EESCHEMA_SETTINGS*>( Kiface().KifaceSettings() );
 329     wxASSERT( cfg );
 330 
 331     if( cfg )
 332     {
 333         for( size_t i = 0;
 334              i < CUSTOMPANEL_COUNTMAX && i < cfg->m_NetlistPanel.custom_command_titles.size();
 335              i++ )
 336         {
 337             // pairs of (title, command) are stored

 338             wxString title = cfg->m_NetlistPanel.custom_command_titles[i];
 339 
 340             if( i >= cfg->m_NetlistPanel.custom_command_paths.size() )
 341                 break; // No more panel to install

 342 
 343             wxString command = cfg->m_NetlistPanel.custom_command_paths[i];
 344 
 345             currPage = AddOneCustomPage( title, command,
 346                     static_cast<NETLIST_TYPE_ID>( NET_TYPE_CUSTOM1 + i ) );
 347 
 348             m_PanelNetType[PANELCUSTOMBASE + i] = currPage;
 349         }
 350     }
 351 }

 436 bool NETLIST_DIALOG::TransferDataFromWindow()
 437 {
 438     wxFileName  fn;
 439     wxString    fileWildcard;
 440     wxString    fileExt;
 441     wxString    title = _( "Save Netlist File" );
 442 
 443     NetlistUpdateOpt();
 444 
 445     NETLIST_PAGE_DIALOG* currPage;
 446     currPage = (NETLIST_PAGE_DIALOG*) m_NoteBook->GetCurrentPage();
 447 
 448     unsigned netlist_opt = 0;
 449 
 450     // Calculate the netlist filename

 451     fn = m_Parent->Schematic().GetFileName();
 452     FilenamePrms( currPage->m_IdNetType, &fileExt, &fileWildcard );
 453 
 454     // Set some parameters

 455     switch( currPage->m_IdNetType )
 456     {
 457     case NET_TYPE_SPICE:
 458         // Set spice netlist options:

 459         if( currPage->m_AdjustPassiveValues->GetValue() )
 460             netlist_opt |= NET_ADJUST_PASSIVE_VALS;
 461         break;
 462 
 463     case NET_TYPE_CADSTAR:
 464         break;
 465 
 466     case NET_TYPE_PCBNEW:
 467         break;
 468 
 469     case NET_TYPE_ORCADPCB2:
 470         break;
 471 
 472     default:    // custom, NET_TYPE_CUSTOM1 and greater

 473     {
 474         wxString command = currPage->m_CommandStringCtrl->GetValue();
 475         wxRegEx extRE( wxT( ".*\\.([[:alnum:]][[:alnum:]][[:alnum:]][[:alnum:]]?)\\.xslt?\".*" ) );
 476 
 477         if( extRE.Matches( command ) )
 478             fileExt = extRE.GetMatch( command, 1 );
 479 
 480         title.Printf( _( "%s Export" ), currPage->m_TitleStringCtrl->GetValue().GetData() );
 481     }
 482         break;
 483     }
 484 
 485     fn.SetExt( fileExt );
 486 
 487     if( fn.GetPath().IsEmpty() )
 488        fn.SetPath( wxPathOnly( Prj().GetProjectFullName() ) );
 489 
 490     wxString fullpath = fn.GetFullPath();
 491     wxString fullname = fn.GetFullName();
 492     wxString path     = fn.GetPath();
 493 
 494     // fullname does not and should not include the path, per wx docs.

 495     wxFileDialog dlg( this, title, path, fullname, fileWildcard, wxFD_SAVE );
 496 
 497     if( dlg.ShowModal() == wxID_CANCEL )
 498         return false;
 499 
 500     fullpath = dlg.GetPath();   // directory + filename

 501 
 502     m_Parent->ClearMsgPanel();
 503 
 504     if( currPage->m_CommandStringCtrl )
 505         m_Parent->SetNetListerCommand( currPage->m_CommandStringCtrl->GetValue() );
 506     else
 507         m_Parent->SetNetListerCommand( wxEmptyString );
 508 
 509     if( m_Parent->ReadyToNetlist( false, false ) )
 510         m_Parent->WriteNetListFile( currPage->m_IdNetType, fullpath, netlist_opt, nullptr );
 511     else
 512         wxMessageBox( _( "Schematic netlist not available" ) );
 513 
 514     WriteCurrentNetlistSetup();
 515 
 516     return true;
 517 }

 520 bool NETLIST_DIALOG::FilenamePrms( NETLIST_TYPE_ID aType, wxString * aExt, wxString * aWildCard )
 521 {
 522     wxString fileExt;
 523     wxString fileWildcard;
 524     bool     ret = true;
 525 
 526     switch( aType )
 527     {
 528     case NET_TYPE_SPICE:
 529         fileExt = wxT( "cir" );
 530         fileWildcard = SpiceNetlistFileWildcard();
 531         break;
 532 
 533     case NET_TYPE_CADSTAR:
 534         fileExt = wxT( "frp" );
 535         fileWildcard = CadstarNetlistFileWildcard();
 536         break;
 537 
 538     case NET_TYPE_PCBNEW:
 539     case NET_TYPE_ORCADPCB2:
 540         fileExt = NetlistFileExtension;
 541         fileWildcard = NetlistFileWildcard();
 542         break;
 543 
 544     default:    // custom, NET_TYPE_CUSTOM1 and greater

 545         fileWildcard = AllFilesWildcard();
 546         ret = false;
 547     }
 548 
 549     if( aExt )
 550         *aExt = fileExt;
 551 
 552     if( aWildCard )
 553         *aWildCard = fileWildcard;
 554 
 555     return ret;
 556 }

 559 void NETLIST_DIALOG::WriteCurrentNetlistSetup()
 560 {
 561     wxString  msg;
 562 
 563     NetlistUpdateOpt();
 564 
 565     EESCHEMA_SETTINGS* cfg = dynamic_cast<EESCHEMA_SETTINGS*>( Kiface().KifaceSettings() );
 566     wxASSERT( cfg );
 567 
 568     if( cfg )
 569     {
 570         cfg->m_NetlistPanel.custom_command_titles.clear();
 571         cfg->m_NetlistPanel.custom_command_paths.clear();
 572 
 573         // Update existing custom pages

 574         for( int ii = 0; ii < CUSTOMPANEL_COUNTMAX; ii++ )
 575         {
 576             NETLIST_PAGE_DIALOG* currPage = m_PanelNetType[ii + PANELCUSTOMBASE];
 577 
 578             if( currPage == NULL )
 579                 break;
 580 
 581             wxString title = currPage->m_TitleStringCtrl->GetValue();
 582 
 583             if( title.IsEmpty() )
 584                 continue;
 585 
 586             cfg->m_NetlistPanel.custom_command_titles.emplace_back( title.ToStdString() );
 587             cfg->m_NetlistPanel.custom_command_paths.emplace_back(
 588                     currPage->m_CommandStringCtrl->GetValue().ToStdString() );
 589         }
 590     }
 591 }

 612 void NETLIST_DIALOG::OnAddGenerator( wxCommandEvent& event )
 613 {
 614     NETLIST_DIALOG_ADD_GENERATOR dlg( this );
 615 
 616     if( dlg.ShowModal() != wxID_OK )
 617         return;
 618 
 619     // Creates a new custom plugin page

 620     wxString title = dlg.GetGeneratorTitle();
 621 
 622     // Verify it does not exists

 623     int netTypeId = PANELCUSTOMBASE;    // the first not used type id

 624     NETLIST_PAGE_DIALOG* currPage;
 625 
 626     for( int ii = 0; ii < CUSTOMPANEL_COUNTMAX; ii++ )
 627     {
 628         netTypeId = PANELCUSTOMBASE + ii;
 629         currPage = m_PanelNetType[ii + PANELCUSTOMBASE];
 630 
 631         if( currPage == NULL )
 632             break;
 633 
 634         if( currPage->GetPageNetFmtName() == title )
 635         {
 636             wxMessageBox( _("This plugin already exists.") );
 637             return;
 638         }
 639     }
 640 
 641     wxString cmd = dlg.GetGeneratorTCommandLine();
 642     currPage = AddOneCustomPage( title,cmd, (NETLIST_TYPE_ID)netTypeId );
 643     m_PanelNetType[netTypeId] = currPage;
 644     WriteCurrentNetlistSetup();
 645 
 646     // Close and reopen dialog to rebuild the dialog after changes

 647     EndModal( NET_PLUGIN_CHANGE );
 648 }

 678 void NETLIST_DIALOG_ADD_GENERATOR::OnBrowseGenerators( wxCommandEvent& event )
 679 {
 680     wxString FullFileName, Path;
 681 
 682 #ifndef __WXMAC__
 683     Path = Pgm().GetExecutablePath();
 684 #else
 685     Path = GetOSXKicadDataDir() + wxT( "/plugins" );
 686 #endif
 687     FullFileName = EDA_FILE_SELECTOR( _( "Generator files:" ), Path, FullFileName,
 688                                       wxEmptyString, wxFileSelectorDefaultWildcardStr,
 689                                       this, wxFD_OPEN, true );
 690     if( FullFileName.IsEmpty() )
 691         return;
 692 
 693     // Creates a default command line, suitable for external tool xslproc or python, based on

 694     // the plugin extension ("xsl" or "exe" or "py")

 695     wxString cmdLine;
 696     wxFileName fn( FullFileName );
 697     wxString ext = fn.GetExt();
 698 
 699     if( ext == wxT("xsl" ) )
 700         cmdLine.Printf(wxT("xsltproc -o \"%%O\" \"%s\" \"%%I\""), GetChars(FullFileName) );
 701     else if( ext == wxT("exe" ) || ext.IsEmpty() )
 702         cmdLine.Printf(wxT("\"%s\" > \"%%O\" < \"%%I\""), GetChars(FullFileName) );
 703     else if( ext == wxT("py" ) || ext.IsEmpty() )
 704         cmdLine.Printf(wxT("python \"%s\" \"%%I\" \"%%O\""), GetChars(FullFileName) );
 705     else
 706         cmdLine.Printf(wxT("\"%s\""), GetChars(FullFileName) );
 707 
 708     m_textCtrlCommand->SetValue( cmdLine );
 709 
 710     /* Get a title for this page */
 711     wxString title = m_textCtrlName->GetValue();
 712 
 713     if( title.IsEmpty() )
 714         wxMessageBox( _( "Do not forget to choose a title for this netlist control page" ) );
 715 }
