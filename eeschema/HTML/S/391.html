  46 /**

  47  * Defines the edge of the sheet that the sheet pin is positioned

  48  * SHEET_LEFT_SIDE = 0: pin on left side

  49  * SHEET_RIGHT_SIDE = 1: pin on right side

  50  * SHEET_TOP_SIDE = 2: pin on top side

  51  * SHEET_BOTTOM_SIDE =3: pin on bottom side

  52  *

  53  * For compatibility reasons, this does not follow same values as text orientation.

  54  */

  57     SHEET_LEFT_SIDE = 0,
  58     SHEET_RIGHT_SIDE,
  59     SHEET_TOP_SIDE,
  60     SHEET_BOTTOM_SIDE,
  61     SHEET_UNDEFINED_SIDE

  74 /**

  75  * Define a sheet pin (label) used in sheets to create hierarchical schematics.

  76  *

  77  * A SCH_SHEET_PIN is used to create a hierarchical sheet in the same way a

  78  * pin is used in a component.  It connects the objects in the sheet object

  79  * to the objects in the schematic page to the objects in the page that is

  80  * represented by the sheet.  In a sheet object, a SCH_SHEET_PIN must be

  81  * connected to a wire, bus, or label.  In the schematic page represented by

  82  * the sheet, it corresponds to a hierarchical label.

  83  */
  84 class SCH_SHEET_PIN : public SCH_HIERLABEL
  85 {
  86 private:
  87     int m_number;       ///< Label number use for saving sheet label to file.

  88                         ///< Sheet label numbering begins at 2.

  89                         ///< 0 is reserved for the sheet name.

  90                         ///< 1 is reserve for the sheet file name.

  91 
  92     SHEET_SIDE m_edge;
  93 
  94 public:
  95     SCH_SHEET_PIN( SCH_SHEET* parent,
  96                    const wxPoint& pos = wxPoint( 0, 0 ),
  97                    const wxString& text = wxEmptyString );
  98 
  99     // Do not create a copy constructor.  The one generated by the compiler is adequate.

 100 
 101     ~SCH_SHEET_PIN() { }
 102 
 103     static inline bool ClassOf( const EDA_ITEM* aItem )
 104     {
 105         return aItem && SCH_SHEET_PIN_T == aItem->Type();
 106     }
 107

 214  * Sheet symbol placed in a schematic, and is the entry point for a sub schematic.

 216 class SCH_SHEET : public SCH_ITEM
 217 {
 218     friend class SCH_SHEET_PIN;
 219 
 220 
 221     SCH_SCREEN*   m_screen;     // Screen that contains the physical data for the sheet.  In

 222                                 // complex hierarchies multiple sheets can share a common screen.

 223 
 224     std::vector<SCH_SHEET_PIN*> m_pins;               // The list of sheet connection points.

 225 
 226     std::vector<SCH_FIELD>      m_fields;
 227     FIELDS_AUTOPLACED           m_fieldsAutoplaced;   // Indicates status of field autoplacement.

 228 
 229     wxPoint                     m_pos;                // The position of the sheet.

 230     wxSize                      m_size;               // The size of the sheet.

 231 
 232     int                         m_borderWidth;
 233     KIGFX::COLOR4D              m_borderColor;
 234     KIGFX::COLOR4D              m_backgroundColor;
 235 
 236 public:
 237     SCH_SHEET( EDA_ITEM* aParent = nullptr, const wxPoint& pos = wxPoint( 0, 0 ) );
 238 
 239     /**

 240      * Copy \a aSheet into a new object.  All sheet pins are copied as is except and

 241      * the SCH_SHEET_PIN's #m_Parent pointers are set to the new copied parent object.

 242      */
 243     SCH_SHEET( const SCH_SHEET& aSheet );
 244 
 245     ~SCH_SHEET();

 254         return wxT( "SCH_SHEET" );
 255     }

 259      * and false for items moved with no reference to anchor.

 260      *

 261      * Usually return true for small items (labels, junctions) and false for

 262      * items which can be large (hierarchical sheets, components)

 263      *

 264      * @return false for a hierarchical sheet

 265      */
 266     bool IsMovableFromAnchorPoint() override { return false; }
 267 
 268     std::vector<SCH_FIELD>& GetFields() { return m_fields; }
 269 
 270     /**

 271      * Set multiple schematic fields.

 272      *

 273      * @param aFields are the fields to set in this symbol.

 274      */
 275     void SetFields( const std::vector<SCH_FIELD>& aFields )
 276     {
 277         m_fields = aFields;     // vector copying, length is changed possibly

 278     }
 279 
 280     wxString GetName() const { return m_fields[ SHEETNAME ].GetText(); }
 281 
 282     SCH_SCREEN* GetScreen() const { return m_screen; }
 283 
 284     wxSize GetSize() { return m_size; }
 285     void SetSize( const wxSize& aSize ) { m_size = aSize; }
 286 
 287     int GetBorderWidth() const { return m_borderWidth; }
 288     void SetBorderWidth( int aWidth ) { m_borderWidth = aWidth; }
 289 
 290     KIGFX::COLOR4D GetBorderColor() const { return m_borderColor; }
 291     void SetBorderColor( KIGFX::COLOR4D aColor ) { m_borderColor = aColor; }
 292 
 293     KIGFX::COLOR4D GetBackgroundColor() const { return m_backgroundColor; }
 294     void SetBackgroundColor( KIGFX::COLOR4D aColor ) { m_backgroundColor = aColor; }
 295 
 296     /**

 297      * Test this sheet to see if the default stroke is used to draw the outline.

 298      *

 299      * The default stroke is defined as follows:

 300      * * The outline width is the default line width or 0.

 301      * * The outline style is set to #PLOT_DASH_TYPE::DEFAULT or #PLOT_DASH_TYPE::SOLID.

 302      * * The outline color is set to #COLOR4D::UNSPECIFIED.

 303      *

 304      * @return  True if the outline stroke meets the default criteria.

 305      */
 306     bool UsesDefaultStroke() const;
 307 
 308     /**

 309      * @return true if this sheet is the root sheet.

 310      */
 311     bool IsRootSheet() const;
 312 
 313     /**

 314      * Set the #SCH_SCREEN associated with this sheet to \a aScreen.

 315      *

 316      * The screen reference counting is performed by SetScreen.  If \a aScreen is not

 317      * the same as the current screen, the current screen reference count is decremented

 318      * and \a aScreen becomes the screen for the sheet.  If the current screen reference

 319      * count reaches zero, the current screen is deleted.  NULL is a valid value for

 320      * \a aScreen.

 321      *

 322      * @param aScreen The new screen to associate with the sheet.

 323      */
 324     void SetScreen( SCH_SCREEN* aScreen );
 325 
 326     /**

 327      * Return the number of times the associated screen for the sheet is being used.

 328      *

 329      * If no screen is associated with the sheet, then zero is returned.

 330      */
 331     int GetScreenCount() const;
 332 
 333     /**

 334      * Return the list of system text vars & fields for this sheet.

 335      */
 336     void GetContextualTextVars( wxArrayString* aVars ) const;
 337 
 338     /**

 339      * Resolve any references to system tokens supported by the sheet.

 340      * @param aDepth a counter to limit recursion and circular references.

 341      */
 342     bool ResolveTextVar( wxString* token, int aDepth = 0 ) const;
 343 
 344     void GetMsgPanelInfo( EDA_DRAW_FRAME* aFrame, std::vector<MSG_PANEL_ITEM>& aList ) override;
 345 
 346     /* there is no member for orientation in sch_sheet, to preserve file

 347      * format, we detect orientation based on pin edges

 348      */
 349     bool IsVerticalOrientation() const;
 350 
 351     /**

 352      * Add aSheetPin to the sheet.

 353      *

 354      * Note: Once a sheet pin is added to the sheet, it is owned by the sheet.

 355      *       Do not delete the sheet pin object or you will likely get a segfault

 356      *       when the sheet is destroyed.

 357      *

 358      * @param aSheetPin The sheet pin item to add to the sheet.

 359      */
 360     void AddPin( SCH_SHEET_PIN* aSheetPin );
 361 
 362     std::vector<SCH_SHEET_PIN*>& GetPins() { return m_pins; }
 363 
 364     std::vector<SCH_SHEET_PIN*>& GetPins() const
 365     {
 366         return const_cast< std::vector<SCH_SHEET_PIN*>& >( m_pins );
 367     }
 368 
 369     /**

 370      * Remove \a aSheetPin from the sheet.

 371      *

 372      * @param aSheetPin The sheet pin item to remove from the sheet.

 373      */
 374     void RemovePin( SCH_SHEET_PIN* aSheetPin );
 375 
 376     /**

 377      * Delete sheet label which do not have a corresponding hierarchical label.

 378      *

 379      * Note: Make sure you save a copy of the sheet in the undo list before calling

 380      *       CleanupSheet() otherwise any unreferenced sheet labels will be lost.

 381      */
 382     void CleanupSheet();
 383 
 384     /**

 385      * Return the sheet pin item found at \a aPosition in the sheet.

 386      *

 387      * @param aPosition The position to check for a sheet pin.

 388      *

 389      * @return The sheet pin found at \a aPosition or NULL if no sheet pin is found.

 390      */
 391     SCH_SHEET_PIN* GetPin( const wxPoint& aPosition );
 392 
 393     /**

 394      * Checks if the sheet already has a sheet pin named \a aName.

 395      *

 396      * @param aName Name of the sheet pin to search for.

 397      *

 398      * @return  True if sheet pin with \a aName is found, otherwise false.

 399      */
 400     bool HasPin( const wxString& aName );
 401 
 402     bool HasPins() { return !m_pins.empty(); }
 403 
 404     /**

 405      * Check all sheet labels against schematic for undefined hierarchical labels.

 406      *

 407      * @return True if there are any undefined labels.

 408      */
 409     bool HasUndefinedPins();
 410 
 411     /**

 412      * Return the minimum width of the sheet based on the widths of the sheet pin text.

 413      *

 414      * The minimum sheet width is determined by the width of the bounding box of each

 415      * hierarchical sheet pin.  If two pins are horizontally adjacent ( same Y position )

 416      * to each other, the sum of the bounding box widths is used.  If at some point in

 417      * the future sheet objects can be rotated or pins can be placed in the vertical

 418      * orientation, this function will need to be changed.

 419      *

 420      * @return The minimum width the sheet can be resized.

 421      */
 422     int GetMinWidth() const;
 423 
 424     /**

 425      * Return the minimum height that the sheet can be resized based on the sheet pin positions.

 426      *

 427      * The minimum width of a sheet is determined by the Y axis location of the bottom

 428      * most sheet pin.  If at some point in the future sheet objects can be rotated or

 429      * pins can be placed in the vertical orientation, this function will need to be

 430      * changed.

 431      *

 432      * @return The minimum height the sheet can be resized.

 433      */
 434     int GetMinHeight() const;
 435 
 436     int GetPenWidth() const override;
 437 
 438     void Print( RENDER_SETTINGS* aSettings, const wxPoint& aOffset ) override;
 439 
 440     /**

 444

 451      */

 463     /**

 464      * Search the existing hierarchy for an instance of screen loaded from \a aFileName.

 466      * @param aFilename = the filename to find (MUST be absolute, and in wxPATH_NATIVE encoding)

 467      * @param aScreen = a location to return a pointer to the screen (if found)

 468      * @return bool if found, and a pointer to the screen

 470     bool SearchHierarchy( const wxString& aFilename, SCH_SCREEN** aScreen );

 472     /**

 473      * Search the existing hierarchy for an instance of screen loaded from \a aFileName.
