  88         ACTION_MENU( true ),

 101 protected:
 102     ACTION_MENU* create() const override
 103     {
 104         return new BUS_UNFOLD_MENU();

 107 private:
 108     void update() override
 109     {
 110         SCH_EDIT_FRAME*    frame = (SCH_EDIT_FRAME*) getToolManager()->GetToolHolder();
 111         EE_SELECTION_TOOL* selTool = getToolManager()->GetTool<EE_SELECTION_TOOL>();
 112         KICAD_T            busType[] = { SCH_LINE_LOCATE_BUS_T, EOT };
 113         EE_SELECTION&      selection = selTool->RequestSelection( busType );
 114         SCH_LINE*          bus = (SCH_LINE*) selection.Front();
 115 
 116         Clear();
 117 
 118         // TODO(JE) remove once real-time is enabled

 119         if( !ADVANCED_CFG::GetCfg().m_realTimeConnectivity || !CONNECTION_GRAPH::m_allowRealTime )
 120         {
 121             frame->RecalculateConnections( NO_CLEANUP );
 122 
 123             // Pick up the pointer again because it may have been changed by SchematicCleanUp

 124             selection = selTool->RequestSelection( busType );
 125             bus = (SCH_LINE*) selection.Front();
 126         }
 127 
 128         if( !bus )
 129         {
 130             Append( ID_POPUP_SCH_UNFOLD_BUS, _( "No bus selected" ), wxEmptyString );
 131             Enable( ID_POPUP_SCH_UNFOLD_BUS, false );
 132             return;
 133         }
 134 
 135         SCH_CONNECTION* connection = bus->Connection( frame->GetCurrentSheet() );
 136 
 137         if( !connection || !connection->IsBus() || connection->Members().empty() )
 138         {
 139             Append( ID_POPUP_SCH_UNFOLD_BUS, _( "Bus has no members" ), wxEmptyString );
 140             Enable( ID_POPUP_SCH_UNFOLD_BUS, false );
 141             return;
 142         }
 143 
 144         int idx = 0;
 145 
 146         if( m_showTitle )
 147         {
 148             Append( ID_POPUP_SCH_UNFOLD_BUS, _( "Unfold from Bus" ), wxEmptyString );
 149             Enable( ID_POPUP_SCH_UNFOLD_BUS, false );
 150         }
 151 
 152         for( const auto& member : connection->Members() )
 153         {
 154             int id = ID_POPUP_SCH_UNFOLD_BUS + ( idx++ );
 155             wxString name = SCH_CONNECTION::PrintBusForUI( member->FullLocalName() );
 156 
 157             if( member->Type() == CONNECTION_TYPE::BUS )
 158             {
 159                 ACTION_MENU* submenu = new ACTION_MENU( true );
 160                 submenu->SetTool( m_tool );
 161                 AppendSubMenu( submenu, name );
 162 
 163                 for( const auto& sub_member : member->Members() )
 164                 {
 165                     id = ID_POPUP_SCH_UNFOLD_BUS + ( idx++ );
 166                     name = SCH_CONNECTION::PrintBusForUI( sub_member->FullLocalName() );
 167                     submenu->Append( id, name, wxEmptyString );
 168                 }
 169             }
 170             else
 171             {
 172                 Append( id, name, wxEmptyString );
 173             }
 174         }
 175     }
 176 
 177     bool m_showTitle;
 178 };

 310 int SCH_LINE_WIRE_BUS_TOOL::UnfoldBus( const TOOL_EVENT& aEvent )
 311 {
 312     wxString* netPtr = aEvent.Parameter<wxString*>();
 313     wxString  net;
 314     SCH_LINE* segment = nullptr;
 315 
 316     std::string tool = aEvent.GetCommandStr().get();
 317     m_frame->PushTool( tool );
 318     Activate();
 319 
 320     if( netPtr )
 321     {
 322         net = *netPtr;
 323         delete netPtr;
 324     }
 325     else
 326     {
 327         BUS_UNFOLD_MENU unfoldMenu;
 328         unfoldMenu.SetTool( this );
 329         unfoldMenu.SetShowTitle();
 330 
 331         SetContextMenu( &unfoldMenu, CMENU_NOW );
 332 
 333         while( TOOL_EVENT* evt = Wait() )
 334         {
 335             if( evt->Action() == TA_CHOICE_MENU_CHOICE )
 336             {
 337                 OPT<int> id = evt->GetCommandId();
 338 
 339                 if( id && ( *id > 0 ) )
 340                     net = *evt->Parameter<wxString*>();
 341 
 342                 break;
 343             }
 344             else if( evt->Action() == TA_CHOICE_MENU_CLOSED )
 345             {
 346                 break;
 347             }
 348             else
 349             {
 350                 evt->SetPassEvent();
 351             }
 352         }
 353     }
 354 
 355     // Break a wire for the given net out of the bus

 356     if( !net.IsEmpty() )
 357         segment = doUnfoldBus( net );
 358 
 359     // If we have an unfolded wire to draw, then draw it

 360     if( segment )
 361         return doDrawSegments( tool, LAYER_WIRE );
 362     else
 363     {
 364         m_frame->PopTool( tool );
 365         return 0;
 366     }
 367 }

 416 void SCH_LINE_WIRE_BUS_TOOL::computeBreakPoint( const std::pair<SCH_LINE*, SCH_LINE*>& aSegments,
 417         wxPoint& aPosition )
 418 {
 419     wxCHECK_RET( aSegments.first && aSegments.second,
 420             wxT( "Cannot compute break point of NULL line segment." ) );
 421 
 422     SCH_LINE* segment = aSegments.first;
 423     SCH_LINE* next_segment = aSegments.second;
 424 
 425     wxPoint midPoint;
 426     int iDx = segment->GetEndPoint().x - segment->GetStartPoint().x;
 427     int iDy = segment->GetEndPoint().y - segment->GetStartPoint().y;
 428 
 429     const SCH_SHEET_PIN* connectedPin = getSheetPin( segment->GetStartPoint() );
 430     auto force = connectedPin ? connectedPin->GetEdge() : SHEET_UNDEFINED_SIDE;
 431 
 432     if( force == SHEET_LEFT_SIDE || force == SHEET_RIGHT_SIDE )
 433     {
 434         if( aPosition.x == connectedPin->GetPosition().x )  // push outside sheet boundary

 435         {
 436             int direction = ( force == SHEET_LEFT_SIDE ) ? -1 : 1;
 437             aPosition.x += KiROUND( getView()->GetGAL()->GetGridSize().x * direction );
 438         }
 439 
 440         midPoint.x = aPosition.x;
 441         midPoint.y = segment->GetStartPoint().y;     // force horizontal

 442     }
 443     else if( iDy != 0 )    // keep the first segment orientation (vertical)

 444     {
 445         midPoint.x = segment->GetStartPoint().x;
 446         midPoint.y = aPosition.y;
 447     }
 448     else if( iDx != 0 )    // keep the first segment orientation (horizontal)

 449     {
 450         midPoint.x = aPosition.x;
 451         midPoint.y = segment->GetStartPoint().y;
 452     }
 453     else
 454     {
 455         if( std::abs( aPosition.x - segment->GetStartPoint().x ) <
 456             std::abs( aPosition.y - segment->GetStartPoint().y ) )
 457         {
 458             midPoint.x = segment->GetStartPoint().x;
 459             midPoint.y = aPosition.y;
 460         }
 461         else
 462         {
 463             midPoint.x = aPosition.x;
 464             midPoint.y = segment->GetStartPoint().y;
 465         }
 466     }
 467 
 468     segment->SetEndPoint( midPoint );
 469     next_segment->SetStartPoint( midPoint );
 470     next_segment->SetEndPoint( aPosition );
 471 }

 474 int SCH_LINE_WIRE_BUS_TOOL::doDrawSegments( const std::string& aTool, int aType )
 475 {
 476     SCH_SCREEN*      screen = m_frame->GetScreen();
 477     EE_POINT_EDITOR* pointEditor = m_toolMgr->GetTool<EE_POINT_EDITOR>();
 478     SCH_LINE*        segment = nullptr;
 479 
 480     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 481     getViewControls()->ShowCursor( true );
 482 
 483     Activate();
 484 
 485     // Add the new label to the selection so the rotate command operates on it

 486     if( m_busUnfold.label )
 487         m_selectionTool->AddItemToSel( m_busUnfold.label, true );
 488 
 489     // Continue the existing wires if we've started (usually by immediate action preference)

 490     if( !m_wires.empty() )
 491         segment = m_wires.back();
 492 
 493     // Main loop: keep receiving events

 494     while( TOOL_EVENT* evt = Wait() )
 495     {
 496         if( !pointEditor->HasPoint() )  // Set wxCursor shape when starting the tool

 497             m_frame->GetCanvas()->SetCurrentCursor( wxCURSOR_PENCIL );
 498 
 499         wxPoint cursorPos = (wxPoint) getViewControls()->GetCursorPosition( !evt->Modifier( MD_ALT ) );
 500         bool forceHV = m_frame->eeconfig()->m_Drawing.hv_lines_only;
 501 
 502         //------------------------------------------------------------------------

 503         // Handle cancel:

 504         //

 505         auto cleanup = [&] () {
 506             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 507 
 508             for( auto wire : m_wires )
 509                 delete wire;
 510 
 511             m_wires.clear();
 512             segment = nullptr;
 513 
 514             if( m_busUnfold.entry )
 515                 m_frame->RemoveFromScreen( m_busUnfold.entry, screen );
 516 
 517             if( m_busUnfold.label && !m_busUnfold.label_placed )
 518                 m_selectionTool->RemoveItemFromSel( m_busUnfold.label, true );
 519 
 520             if( m_busUnfold.label && m_busUnfold.label_placed )
 521                 m_frame->RemoveFromScreen( m_busUnfold.label, screen );
 522 
 523             delete m_busUnfold.entry;
 524             delete m_busUnfold.label;
 525             m_busUnfold = {};
 526 
 527             m_view->ClearPreview();
 528             m_view->ShowPreview( false );
 529         };
 530 
 531         if( evt->IsCancelInteractive() )
 532         {
 533             if( segment || m_busUnfold.in_progress )
 534                 cleanup();
 535             else
 536             {
 537                 m_frame->PopTool( aTool );
 538                 break;
 539             }
 540         }
 541         else if( evt->IsActivate() )
 542         {
 543             if( segment || m_busUnfold.in_progress )
 544                 cleanup();
 545 
 546             if( evt->IsMoveTool() )
 547             {
 548                 // leave ourselves on the stack so we come back after the move

 549                 break;
 550             }
 551             else
 552             {
 553                 m_frame->PopTool( aTool );
 554                 break;
 555             }
 556         }
 557         //------------------------------------------------------------------------

 558         // Handle finish:

 559         //

 560         else if( evt->IsAction( &EE_ACTIONS::finishLineWireOrBus )
 561                      || evt->IsAction( &EE_ACTIONS::finishWire )
 562                      || evt->IsAction( &EE_ACTIONS::finishBus )
 563                      || evt->IsAction( &EE_ACTIONS::finishLine ) )
 564         {
 565             if( segment || m_busUnfold.in_progress )
 566             {
 567                 finishSegments();
 568                 segment = nullptr;
 569             }
 570         }
 571         //------------------------------------------------------------------------

 572         // Handle click:

 573         //

 574         else if( evt->IsClick( BUT_LEFT ) || ( segment && evt->IsDblClick( BUT_LEFT ) ) )
 575         {
 576             // First click when unfolding places the label and wire-to-bus entry

 577             if( m_busUnfold.in_progress && !m_busUnfold.label_placed )
 578             {
 579                 wxASSERT( aType == LAYER_WIRE );
 580 
 581                 m_frame->AddToScreen( m_busUnfold.label, screen );
 582                 m_selectionTool->RemoveItemFromSel( m_busUnfold.label, true );
 583                 m_busUnfold.label_placed = true;
 584             }
 585 
 586             if( !segment )
 587             {
 588                 segment = startSegments( aType, VECTOR2D( cursorPos ) );
 589             }
 590             // Create a new segment if we're out of previously-created ones

 591             else if( !segment->IsNull() || ( forceHV && !m_wires[ m_wires.size() - 2 ]->IsNull() ) )
 592             {
 593                 // Terminate the command if the end point is on a pin, junction, or another

 594                 // wire or bus.

 595                 if( !m_busUnfold.in_progress
 596                         && screen->IsTerminalPoint( cursorPos, segment->GetLayer() ) )
 597                 {
 598                     finishSegments();
 599                     segment = nullptr;
 600                 }
 601                 else
 602                 {
 603                     segment->SetEndPoint( cursorPos );
 604 
 605                     // Create a new segment, and chain it after the current segment.

 606                     segment = new SCH_LINE( *segment );
 607                     segment->SetFlags( IS_NEW | IS_MOVED );
 608                     segment->SetStartPoint( cursorPos );
 609                     m_wires.push_back( segment );
 610 
 611                     m_selectionTool->AddItemToSel( segment, true /*quiet mode*/ );
 612                 }
 613             }
 614 
 615             if( evt->IsDblClick( BUT_LEFT ) && segment )
 616             {
 617                 if( forceHV && m_wires.size() >= 2 )
 618                     computeBreakPoint( { m_wires[ m_wires.size() - 2 ], segment }, cursorPos );
 619 
 620                 finishSegments();
 621                 segment = nullptr;
 622             }
 623         }
 624         //------------------------------------------------------------------------

 625         // Handle motion:

 626         //

 627         else if( evt->IsMotion() || evt->IsAction( &ACTIONS::refreshPreview ) )
 628         {
 629             m_view->ClearPreview();
 630 
 631             // Update the bus unfold posture based on the mouse movement

 632             if( m_busUnfold.in_progress && !m_busUnfold.label_placed )
 633             {
 634                 wxPoint cursor_delta = cursorPos - m_busUnfold.origin;
 635                 SCH_BUS_WIRE_ENTRY* entry = m_busUnfold.entry;
 636 
 637                 bool flipX = ( cursor_delta.x < 0 );
 638                 bool flipY = ( cursor_delta.y < 0 );
 639 
 640                 // Erase and redraw if necessary

 641                 if( flipX != m_busUnfold.flipX || flipY != m_busUnfold.flipY )
 642                 {
 643                     wxSize size  = entry->GetSize();
 644                     int    ySign = flipY ? -1 : 1;
 645                     int    xSign = flipX ? -1 : 1;
 646 
 647                     size.x = std::abs( size.x ) * xSign;
 648                     size.y = std::abs( size.y ) * ySign;
 649                     entry->SetSize( size );
 650 
 651                     m_busUnfold.flipY = flipY;
 652                     m_busUnfold.flipX = flipX;
 653 
 654                     m_frame->RefreshItem( entry );
 655                     m_wires.front()->SetStartPoint( entry->m_End() );
 656                 }
 657 
 658                 // Update the label "ghost" position

 659                 m_busUnfold.label->SetPosition( cursorPos );
 660                 m_view->AddToPreview( m_busUnfold.label->Clone() );
 661 
 662                 // Ensure segment is non-null at the start of bus unfold

 663                 if( !segment )
 664                     segment = m_wires.back();
 665             }
 666 
 667             if( segment )
 668             {
 669                 // Coerce the line to vertical or horizontal if necessary

 670                 if( forceHV && m_wires.size() >= 2 )
 671                     computeBreakPoint( { m_wires[ m_wires.size() - 2 ], segment }, cursorPos );
 672                 else
 673                     segment->SetEndPoint( cursorPos );
 674             }
 675 
 676             for( auto wire : m_wires )
 677             {
 678                 if( !wire->IsNull() )
 679                     m_view->AddToPreview( wire->Clone() );
 680             }
 681         }
 682         //------------------------------------------------------------------------

 683         // Handle context menu:

 684         //

 685         else if( evt->IsClick( BUT_RIGHT ) )
 686         {
 687             // Warp after context menu only if dragging...

 688             if( !segment )
 689                 m_toolMgr->VetoContextMenuMouseWarp();
 690 
 691             m_menu.ShowContextMenu( m_selectionTool->GetSelection() );
 692         }
 693         else if( evt->Category() == TC_COMMAND && evt->Action() == TA_CHOICE_MENU_CHOICE )
 694         {
 695             if( evt->GetCommandId().get() >= ID_POPUP_SCH_UNFOLD_BUS
 696                 && evt->GetCommandId().get() <= ID_POPUP_SCH_UNFOLD_BUS_END )
 697             {
 698                 wxASSERT_MSG( !segment, "Bus unfold event received when already drawing!" );
 699 
 700                 aType = LAYER_WIRE;
 701                 wxString net = *evt->Parameter<wxString*>();
 702                 segment = doUnfoldBus( net );
 703             }
 704         }
 705         else
 706             evt->SetPassEvent();
 707 
 708         // Enable autopanning and cursor capture only when there is a segment to be placed

 709         getViewControls()->SetAutoPan( segment != nullptr );
 710         getViewControls()->CaptureCursor( segment != nullptr );
 711     }
 712 
 713     return 0;
 714 }

 717 SCH_LINE* SCH_LINE_WIRE_BUS_TOOL::startSegments( int aType, const VECTOR2D& aPos )
 718 {
 719     SCH_LINE* segment = nullptr;
 720 
 721     switch ( aType )
 722     {
 723     default:
 724         segment = new SCH_LINE( aPos, LAYER_NOTES );
 725         break;
 726     case LAYER_WIRE:
 727         segment = new SCH_LINE( aPos, LAYER_WIRE );
 728         break;
 729     case LAYER_BUS:
 730         segment = new SCH_LINE( aPos, LAYER_BUS );
 731         break;

 733 
 734     segment->SetFlags( IS_NEW | IS_MOVED );
 735     m_wires.push_back( segment );
 736 
 737     m_selectionTool->AddItemToSel( segment, true /*quiet mode*/ );

 740     // horizontal and vertical lines only switch is on.

 741     if( m_frame->eeconfig()->m_Drawing.hv_lines_only )
 742     {
 743         segment = new SCH_LINE( *segment );
 744         segment->SetFlags( IS_NEW | IS_MOVED );
 745         m_wires.push_back( segment );
 746 
 747         m_selectionTool->AddItemToSel( segment, true /*quiet mode*/ );
 748     }
 749 
 750     return segment;

 754 /**

 755  * In a contiguous list of wires, remove wires that backtrack over the previous

 756  * wire. Example:

 757  *

 758  * Wire is added:

 759  * ---------------------------------------->

 760  *

 761  * A second wire backtracks over it:

 762  * -------------------<====================>

 763  *

 764  * simplifyWireList is called:

 765  * ------------------->

 766  */
 767 void SCH_LINE_WIRE_BUS_TOOL::simplifyWireList()
 768 {
 769     for( auto it = m_wires.begin(); it != m_wires.end(); )
 770     {
 771         SCH_LINE* line = *it;
 772 
 773         if( line->IsNull() )
 774         {
 775             delete line;
 776             it = m_wires.erase( it );
 777             continue;
 778         }
 779 
 780         auto next_it = it;
 781         ++next_it;
 782 
 783         if( next_it == m_wires.end() )
 784             break;
 785 
 786         SCH_LINE* next_line = *next_it;
 787 
 788         if( line->IsParallel( next_line ) )
 789         {
 790             if( SCH_LINE* merged = line->MergeOverlap( next_line ) )
 791             {
 792                 delete line;
 793                 delete next_line;
 794                 it = m_wires.erase( it );
 795                 *it = merged;
 796             }
 797         }
 798 
 799         ++it;
 800     }
 801 }

 804 void SCH_LINE_WIRE_BUS_TOOL::finishSegments()
 805 {
 806     // Clear selection when done so that a new wire can be started.

 807     // NOTE: this must be done before simplifyWireList is called or we might end up with

 808     // freed selected items.

 809     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 810 
 811     SCH_SCREEN*       screen = m_frame->GetScreen();
 812     PICKED_ITEMS_LIST itemList;
 813 
 814     // Remove segments backtracking over others

 815     simplifyWireList();
 816 
 817     // Collect the possible connection points for the new lines

 818     std::vector< wxPoint > connections;
 819     std::vector< wxPoint > new_ends;
 820     m_frame->GetSchematicConnections( connections );
 821 
 822     // Check each new segment for possible junctions and add/split if needed

 823     for( auto wire : m_wires )
 824     {
 825         if( wire->HasFlag( SKIP_STRUCT ) )
 826             continue;
 827 
 828         wire->GetConnectionPoints( new_ends );
 829 
 830         for( auto i : connections )
 831         {
 832             if( IsPointOnSegment( wire->GetStartPoint(), wire->GetEndPoint(), i ) )
 833                 new_ends.push_back( i );
 834         }
 835         itemList.PushItem( ITEM_PICKER( screen, wire, UR_NEW ) );
 836     }
 837 
 838     if( m_busUnfold.in_progress && m_busUnfold.label_placed )
 839     {
 840         wxASSERT( m_busUnfold.entry && m_busUnfold.label );
 841 
 842         itemList.PushItem( ITEM_PICKER( screen, m_busUnfold.entry, UR_NEW ) );
 843         itemList.PushItem( ITEM_PICKER( screen, m_busUnfold.label, UR_NEW ) );
 844         m_busUnfold.label->ClearEditFlags();
 845     }
 846 
 847     // Get the last non-null wire (this is the last created segment).

 848     if( !m_wires.empty() )
 849         m_frame->SaveCopyForRepeatItem( m_wires.back() );
 850 
 851     // Add the new wires

 852     for( auto wire : m_wires )
 853     {
 854         wire->ClearFlags( IS_NEW | IS_MOVED );
 855         m_frame->AddToScreen( wire, screen );
 856     }
 857 
 858     m_wires.clear();
 859     m_view->ClearPreview();
 860     m_view->ShowPreview( false );
 861 
 862     getViewControls()->CaptureCursor( false );
 863     getViewControls()->SetAutoPan( false );
 864 
 865     m_frame->SaveCopyInUndoList( itemList, UR_NEW, false );
 866 
 867     // Correct and remove segments that need to be merged.

 868     m_frame->SchematicCleanUp();
 869 
 870     for( auto item : m_frame->GetScreen()->Items().OfType( SCH_COMPONENT_T ) )
 871     {
 872         std::vector< wxPoint > pts;
 873         item->GetConnectionPoints( pts );
 874 
 875         if( pts.size() > 2 )
 876             continue;
 877 
 878         for( auto i = pts.begin(); i != pts.end(); i++ )
 879         {
 880             for( auto j = i + 1; j != pts.end(); j++ )
 881                 m_frame->TrimWire( *i, *j );
 882         }
 883     }
 884 
 885     for( auto i : new_ends )
 886     {
 887         if( m_frame->GetScreen()->IsJunctionNeeded( i, true ) )
 888             m_frame->AddJunction( m_frame->GetScreen(), i, true, false );
 889     }
 890 
 891     if( m_busUnfold.in_progress )
 892         m_busUnfold = {};
 893 
 894     m_frame->TestDanglingEnds();
 895     m_toolMgr->PostEvent( EVENTS::SelectedItemsModified );
 896 
 897     m_frame->OnModify();
 898 }

 901 int SCH_LINE_WIRE_BUS_TOOL::AddJunctionsIfNeeded( const TOOL_EVENT& aEvent )
 902 {
 903     EE_SELECTION* aSelection = aEvent.Parameter<EE_SELECTION*>();
 904 
 905     std::vector<wxPoint> pts;
 906     std::vector<wxPoint> connections;
 907 
 908     m_frame->GetSchematicConnections( connections );
 909 
 910     for( unsigned ii = 0; ii < aSelection->GetSize(); ii++ )
 911     {
 912         SCH_ITEM*            item = dynamic_cast<SCH_ITEM*>( aSelection->GetItem( ii ) );
 913         std::vector<wxPoint> new_pts;
 914 
 915         if( !item || !item->IsConnectable() )
 916             continue;
 917 
 918         item->GetConnectionPoints( new_pts );
 919         pts.insert( pts.end(), new_pts.begin(), new_pts.end() );
 920 
 921         // If the item is a line, we also add any connection points from the rest of the schematic

 922         // that terminate on the line after it is moved.

 923         if( item->Type() == SCH_LINE_T )
 924         {
 925             SCH_LINE* line = (SCH_LINE*) item;
 926             for( auto i : connections )
 927             {
 928                 if( IsPointOnSegment( line->GetStartPoint(), line->GetEndPoint(), i ) )
 929                     pts.push_back( i );
 930             }
 931         }
 932         else
 933         {
 934             // Clean up any wires that short non-wire connections in the list

 935             for( auto point = new_pts.begin(); point != new_pts.end(); point++ )
 936             {
 937                 for( auto second_point = point + 1; second_point != new_pts.end(); second_point++ )
 938                     m_frame->TrimWire( *point, *second_point );
 939             }
 940         }
 941     }
 942 
 943     // We always have some overlapping connection points.  Drop duplicates here

 944     std::sort( pts.begin(), pts.end(), []( const wxPoint& a, const wxPoint& b ) -> bool {
 945         return a.x < b.x || ( a.x == b.x && a.y < b.y );
 946     } );
 947 
 948     pts.erase( unique( pts.begin(), pts.end() ), pts.end() );
 949 
 950     for( auto point : pts )
 951     {
 952         if( m_frame->GetScreen()->IsJunctionNeeded( point, true ) )
 953             m_frame->AddJunction( m_frame->GetScreen(), point, true, false );
 954     }
 955 
 956     return 0;
 957 }
