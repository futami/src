  52 void IncrementLabelMember( wxString& name, int aIncrement )
  53 {
  54     int  ii, nn;
  55     long number = 0;
  56 
  57     ii = name.Len() - 1; nn = 0;
  58 
  59     if( !wxIsdigit( name.GetChar( ii ) ) )
  60         return;
  61 
  62     while( ii >= 0 && wxIsdigit( name.GetChar( ii ) ) )
  63     {
  64         ii--;
  65         nn++;
  66     }
  67 
  68     ii++;   /* digits are starting at ii position */
  69     wxString litt_number = name.Right( nn );
  70 
  71     if( litt_number.ToLong( &number ) )
  72     {
  73         number += aIncrement;
  74         name.Remove( ii ); name << number;
  75     }
  76 }

 158 wxPoint SCH_TEXT::GetSchematicTextOffset( RENDER_SETTINGS* aSettings ) const
 159 {
 160     wxPoint text_offset;
 161 
 162     // add an offset to x (or y) position to aid readability of text on a wire or line

 163     int dist = GetTextOffset( aSettings ) + GetPenWidth();
 164 
 165     switch( GetLabelSpinStyle() )
 166     {
 167     case LABEL_SPIN_STYLE::UP:
 168     case LABEL_SPIN_STYLE::BOTTOM:
 169         text_offset.x = -dist;
 170         break; // Vert Orientation

 171     default:
 172     case LABEL_SPIN_STYLE::LEFT:
 173     case LABEL_SPIN_STYLE::RIGHT:
 174         text_offset.y = -dist;
 175         break; // Horiz Orientation

 176     }
 177 
 178     return text_offset;
 179 }

 200 void SCH_TEXT::Rotate( wxPoint aPosition )
 201 {
 202     wxPoint pt = GetTextPos();
 203     RotatePoint( &pt, aPosition, 900 );
 204     SetTextPos( pt );
 205 
 206     SetLabelSpinStyle( GetLabelSpinStyle().RotateCW() );
 207 
 208     if( this->Type() == SCH_TEXT_T )
 209     {
 210         int dy = 0;
 211 
 212         switch( GetLabelSpinStyle() )
 213         {
 214         case LABEL_SPIN_STYLE::LEFT:
 215         case LABEL_SPIN_STYLE::RIGHT:
 216             dy = GetTextHeight();
 217             break;
 218         case LABEL_SPIN_STYLE::UP:
 219         case LABEL_SPIN_STYLE::BOTTOM:
 220         default:
 221             dy = 0;
 222             break;
 223         }
 224 
 225         SetTextY( GetTextPos().y + dy );
 226     }
 227 }

 230 void SCH_TEXT::SetLabelSpinStyle( LABEL_SPIN_STYLE aSpinStyle )
 231 {
 232     m_spin_style = aSpinStyle;
 233 
 234     // Assume "Right" and Left" mean which side of the anchor the text will be on

 235     // Thus we want to left justify text up agaisnt the anchor if we are on the right

 236     switch( aSpinStyle )
 237     {
 238     default:
 239         wxASSERT_MSG( 1, "Bad spin style" );
 240         break;
 241 
 242     case LABEL_SPIN_STYLE::RIGHT: // Horiz Normal Orientation

 243         //

 244         m_spin_style = LABEL_SPIN_STYLE::RIGHT; // Handle the error spin style by resetting

 245         SetTextAngle( TEXT_ANGLE_HORIZ );
 246         SetHorizJustify( GR_TEXT_HJUSTIFY_LEFT );
 247         SetVertJustify( GR_TEXT_VJUSTIFY_BOTTOM );
 248         break;
 249 
 250     case LABEL_SPIN_STYLE::UP: // Vert Orientation UP

 251         SetTextAngle( TEXT_ANGLE_VERT );
 252         SetHorizJustify( GR_TEXT_HJUSTIFY_LEFT );
 253         SetVertJustify( GR_TEXT_VJUSTIFY_BOTTOM );
 254         break;
 255 
 256     case LABEL_SPIN_STYLE::LEFT: // Horiz Orientation - Right justified

 257         SetTextAngle( TEXT_ANGLE_HORIZ );
 258         SetHorizJustify( GR_TEXT_HJUSTIFY_RIGHT );
 259         SetVertJustify( GR_TEXT_VJUSTIFY_BOTTOM );
 260         break;
 261 
 262     case LABEL_SPIN_STYLE::BOTTOM: //  Vert Orientation BOTTOM

 263         SetTextAngle( TEXT_ANGLE_VERT );
 264         SetHorizJustify( GR_TEXT_HJUSTIFY_RIGHT );
 265         SetVertJustify( GR_TEXT_VJUSTIFY_BOTTOM );
 266         break;
 267     }
 268 }

 286 bool SCH_TEXT::operator<( const SCH_ITEM& aItem ) const
 287 {
 288     if( Type() != aItem.Type() )
 289         return Type() < aItem.Type();
 290 
 291     auto other = static_cast<const SCH_TEXT*>( &aItem );
 292 
 293     if( GetLayer() != other->GetLayer() )
 294             return GetLayer() < other->GetLayer();
 295 
 296     if( GetPosition().x != other->GetPosition().x )
 297         return GetPosition().x < other->GetPosition().x;
 298 
 299     if( GetPosition().y != other->GetPosition().y )
 300         return GetPosition().y < other->GetPosition().y;
 301 
 302     return GetText() < other->GetText();
 303 }

 343 bool SCH_TEXT::UpdateDanglingState( std::vector<DANGLING_END_ITEM>& aItemList,
 344                                     const SCH_SHEET_PATH* aPath )
 345 {
 346     // Normal text labels cannot be tested for dangling ends.

 347     if( Type() == SCH_TEXT_T )
 348         return false;
 349 
 350     bool previousState = m_isDangling;
 351     m_isDangling       = true;
 352     m_connectionType   = CONNECTION_TYPE::NONE;
 353 
 354     for( unsigned ii = 0; ii < aItemList.size(); ii++ )
 355     {
 356         DANGLING_END_ITEM& item = aItemList[ii];
 357 
 358         if( item.GetItem() == this )
 359             continue;
 360 
 361         switch( item.GetType() )
 362         {
 363         case PIN_END:
 364         case LABEL_END:
 365         case SHEET_LABEL_END:
 366         case NO_CONNECT_END:
 367             if( GetTextPos() == item.GetPosition() )
 368             {
 369                 m_isDangling = false;
 370 
 371                 if( aPath && item.GetType() != PIN_END )
 372                     m_connected_items[ *aPath ].insert( static_cast<SCH_ITEM*>( item.GetItem() ) );
 373             }
 374 
 375             break;
 376 
 377 
 378         case BUS_START_END:
 379             m_connectionType = CONNECTION_TYPE::BUS;
 380             KI_FALLTHROUGH;
 381 
 382         case WIRE_START_END:
 383         {
 384             // These schematic items have created 2 DANGLING_END_ITEM one per end.  But being

 385             // a paranoid programmer, I'll check just in case.

 386             ii++;
 387 
 388             wxCHECK_MSG( ii < aItemList.size(), previousState != m_isDangling,
 389                          wxT( "Dangling end type list overflow.  Bad programmer!" ) );
 390 
 391             int accuracy = 1;   // We have rounding issues with an accuracy of 0

 392 
 393             DANGLING_END_ITEM & nextItem = aItemList[ii];
 394             m_isDangling = !TestSegmentHit( GetTextPos(), item.GetPosition(),
 395                                             nextItem.GetPosition(), accuracy );
 396 
 397             if( !m_isDangling )
 398             {
 399                 if( m_connectionType != CONNECTION_TYPE::BUS )
 400                     m_connectionType = CONNECTION_TYPE::NET;
 401 
 402                 // Add the line to the connected items, since it won't be picked

 403                 // up by a search of intersecting connection points

 404                 if( aPath )
 405                 {
 406                     auto sch_item = static_cast<SCH_ITEM*>( item.GetItem() );
 407                     AddConnectionTo( *aPath, sch_item );
 408                     sch_item->AddConnectionTo( *aPath, this );
 409                 }
 410             }
 411         }
 412             break;
 413 
 414         default:
 415             break;
 416         }
 417 
 418         if( !m_isDangling )
 419             break;
 420     }
 421 
 422     if( m_isDangling )
 423         m_connectionType = CONNECTION_TYPE::NONE;
 424 
 425     return previousState != m_isDangling;
 426 }

 439 const EDA_RECT SCH_TEXT::GetBoundingBox() const
 440 {
 441     EDA_RECT rect = GetTextBox();
 442 
 443     if( GetTextAngle() != 0 )      // Rotate rect

 444     {
 445         wxPoint pos = rect.GetOrigin();
 446         wxPoint end = rect.GetEnd();
 447 
 448         RotatePoint( &pos, GetTextPos(), GetTextAngle() );
 449         RotatePoint( &end, GetTextPos(), GetTextAngle() );
 450 
 451         rect.SetOrigin( pos );
 452         rect.SetEnd( end );
 453     }
 454 
 455     rect.Normalize();
 456     return rect;
 457 }

 484 wxString SCH_TEXT::GetShownText( int aDepth ) const
 485 {
 486     std::function<bool( wxString* )> textResolver =
 487             [&]( wxString* token ) -> bool
 488             {
 489                 if( ( Type() == SCH_GLOBAL_LABEL_T
 490                         || Type() == SCH_HIER_LABEL_T
 491                         || Type() == SCH_SHEET_PIN_T )
 492                      && token->IsSameAs( wxT( "CONNECTION_TYPE" ) ) )
 493                 {
 494                     *token = getElectricalTypeLabel( GetShape() );
 495                     return true;
 496                 }
 497 
 498                 if( Type() == SCH_SHEET_PIN_T && m_Parent )
 499                 {
 500                     SCH_SHEET* sheet = static_cast<SCH_SHEET*>( m_Parent );
 501 
 502                     if( sheet->ResolveTextVar( token, aDepth ) )
 503                         return true;
 504                 }
 505 
 506                 if( Type() == SCH_TEXT_T )
 507                 {
 508                     if( token->Contains( ':' ) )
 509                     {
 510                         wxCHECK_MSG( Schematic(), wxEmptyString,
 511                                      "No parent SCHEMATIC set for SCH_TEXT!" );
 512 
 513                         SCH_SHEET_LIST sheetList = Schematic()->GetSheets();
 514                         wxString       remainder;
 515                         wxString       ref = token->BeforeFirst( ':', &remainder );
 516                         SCH_SHEET_PATH dummy;
 517                         SCH_ITEM*      refItem = sheetList.GetItem( KIID( ref ), &dummy );
 518 
 519                         if( refItem && refItem->Type() == SCH_COMPONENT_T )
 520                         {
 521                             SCH_COMPONENT* refComponent = static_cast<SCH_COMPONENT*>( refItem );
 522 
 523                             if( refComponent->ResolveTextVar( &remainder, aDepth + 1 ) )
 524                             {
 525                                 *token = remainder;
 526                                 return true;
 527                             }
 528                         }
 529                         else if( refItem && refItem->Type() == SCH_SHEET_T )
 530                         {
 531                             SCH_SHEET* refSheet = static_cast<SCH_SHEET*>( refItem );
 532 
 533                             if( refSheet->ResolveTextVar( &remainder, aDepth + 1 ) )
 534                             {
 535                                 *token = remainder;
 536                                 return true;
 537                             }
 538                         }
 539                     }
 540                 }
 541 
 542                 return false;
 543             };
 544 
 545     bool     processTextVars = false;
 546     wxString text = EDA_TEXT::GetShownText( &processTextVars );
 547 
 548     if( processTextVars )
 549     {
 550         wxCHECK_MSG( Schematic(), wxEmptyString,
 551                      "No parent SCHEMATIC set for SCH_TEXT!" );
 552 
 553         PROJECT* project = nullptr;
 554 
 555         if( Schematic() )
 556             project = &Schematic()->Prj();
 557 
 558         if( aDepth < 10 )
 559             text = ExpandTextVars( text, &textResolver, project );
 560     }
 561 
 562     return text;
 563 }

 614 bool SCH_TEXT::HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy ) const
 615 {
 616     EDA_RECT bBox = GetBoundingBox();
 617     bBox.Inflate( aAccuracy );
 618 
 619     if( aContained )

 797             wxASSERT( m_connected_items.count( current ) );
 798 
 799             for( SCH_ITEM* connection : m_connected_items.at( current ) )
 800             {
 801                 if( connection->IsType( busTypes ) )
 802                     return true;
 803             }
 804         }
 805     }
 806

 955 void SCH_GLOBALLABEL::CreateGraphicShape( RENDER_SETTINGS* aRenderSettings,
 956                                           std::vector<wxPoint>& aPoints, const wxPoint& Pos )
 957 {
 958     int margin    = GetTextOffset( aRenderSettings );
 959     int halfSize  = ( GetTextHeight() / 2 ) + margin;
 960     int linewidth = GetPenWidth();
 961     int symb_len  = LenSize( GetShownText(), linewidth ) + 2 * margin;
 962 
 963     int x = symb_len + linewidth + 3;
 964     int y = halfSize + linewidth + 3;
 965 
 966     aPoints.clear();
 967 
 968     // Create outline shape : 6 points

 969     aPoints.emplace_back( wxPoint( 0, 0 ) );
 970     aPoints.emplace_back( wxPoint( 0, -y ) );     // Up

 971     aPoints.emplace_back( wxPoint( -x, -y ) );    // left

 972     aPoints.emplace_back( wxPoint( -x, 0 ) );     // Up left

 973     aPoints.emplace_back( wxPoint( -x, y ) );     // left down

 974     aPoints.emplace_back( wxPoint( 0, y ) );      // down

 975 
 976     int x_offset = 0;
 977 
 978     switch( m_shape )
 979     {
 980     case PINSHEETLABEL_SHAPE::PS_INPUT:
 981         x_offset = -halfSize;
 982         aPoints[0].x += halfSize;
 983         break;
 984 
 985     case PINSHEETLABEL_SHAPE::PS_OUTPUT:
 986         aPoints[3].x -= halfSize;
 987         break;
 988 
 989     case PINSHEETLABEL_SHAPE::PS_BIDI:
 990     case PINSHEETLABEL_SHAPE::PS_TRISTATE:
 991         x_offset = -halfSize;
 992         aPoints[0].x += halfSize;
 993         aPoints[3].x -= halfSize;
 994         break;
 995 
 996     case PINSHEETLABEL_SHAPE::PS_UNSPECIFIED:
 997     default:
 998         break;
 999     }
1000 
1001     int angle = 0;
1002 
1003     switch( GetLabelSpinStyle() )
1004     {
1005     default:
1006     case LABEL_SPIN_STYLE::LEFT:                 break;
1007     case LABEL_SPIN_STYLE::UP:     angle = -900; break;
1008     case LABEL_SPIN_STYLE::RIGHT:  angle = 1800; break;
1009     case LABEL_SPIN_STYLE::BOTTOM: angle = 900;  break;
1010     }
1011 
1012     // Rotate outlines and move corners in real position

1013     for( wxPoint& aPoint : aPoints )
1014     {
1015         aPoint.x += x_offset;
1016 
1017         if( angle )
1018             RotatePoint( &aPoint, angle );
1019 
1020         aPoint += Pos;
1021     }
1022 
1023     aPoints.push_back( aPoints[0] ); // closing

1024 }
