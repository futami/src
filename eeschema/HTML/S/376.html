  74 
  75     SCH_RENDER_SETTINGS();
  76 
  77     void LoadColors( const COLOR_SETTINGS* aSettings ) override;
  78 
  79     /// @copydoc RENDER_SETTINGS::GetColor()

  80     virtual const COLOR4D& GetColor( const VIEW_ITEM* aItem, int aLayer ) const override;
  81 
  82     bool IsBackgroundDark() const override
  83     {
  84         auto luma = m_layerColors[ LAYER_SCHEMATIC_BACKGROUND ].GetBrightness();
  85 
  86         return luma < 0.5;
  87     }
  88 
  89     const COLOR4D& GetBackgroundColor() override
  90     {
  91         return m_layerColors[ LAYER_SCHEMATIC_BACKGROUND ];
  92     }
  93 
  94     void SetBackgroundColor( const COLOR4D& aColor ) override
  95     {
  96         m_layerColors[ LAYER_SCHEMATIC_BACKGROUND ] = aColor;
  97     }
  98 
  99     const COLOR4D& GetGridColor() override { return m_layerColors[ LAYER_SCHEMATIC_GRID ]; }
 100 
 101     const COLOR4D& GetCursorColor() override { return m_layerColors[ LAYER_SCHEMATIC_CURSOR ]; }

 103     int    m_ShowUnit;                // Show all units if 0
