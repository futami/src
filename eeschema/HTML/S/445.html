  39 bool LIB_CONTROL::Init()
  40 {
  41     m_frame = getEditFrame<SCH_BASE_FRAME>();
  42     m_selectionTool = m_toolMgr->GetTool<EE_SELECTION_TOOL>();
  43     m_isLibEdit = m_frame->IsType( FRAME_SCH_LIB_EDITOR );
  44 
  45     if( m_isLibEdit )
  46     {
  47         CONDITIONAL_MENU& ctxMenu = m_menu.GetMenu();
  48         LIB_EDIT_FRAME* editFrame = getEditFrame<LIB_EDIT_FRAME>();
  49 
  50         auto libSelectedCondition = [ editFrame ] ( const SELECTION& aSel ) {
  51             LIB_ID sel = editFrame->GetTreeLIBID();
  52             return !sel.GetLibNickname().empty() && sel.GetLibItemName().empty();
  53         };
  54         auto pinnedLibSelectedCondition = [ editFrame ] ( const SELECTION& aSel ) {
  55             LIB_TREE_NODE* current = editFrame->GetCurrentTreeNode();
  56             return current && current->m_Type == LIB_TREE_NODE::LIB && current->m_Pinned;
  57         };
  58         auto unpinnedLibSelectedCondition = [ editFrame ] (const SELECTION& aSel ) {
  59             LIB_TREE_NODE* current = editFrame->GetCurrentTreeNode();
  60             return current && current->m_Type == LIB_TREE_NODE::LIB && !current->m_Pinned;
  61         };
  62         auto symbolSelectedCondition = [ editFrame ] ( const SELECTION& aSel ) {
  63             LIB_ID sel = editFrame->GetTreeLIBID();
  64             return !sel.GetLibNickname().empty() && !sel.GetLibItemName().empty();
  65         };
  66 
  67         ctxMenu.AddItem( ACTIONS::pinLibrary,            unpinnedLibSelectedCondition );
  68         ctxMenu.AddItem( ACTIONS::unpinLibrary,          pinnedLibSelectedCondition );
  69         ctxMenu.AddSeparator();
  70 
  71         ctxMenu.AddItem( ACTIONS::save,                  libSelectedCondition );
  72         ctxMenu.AddItem( ACTIONS::saveAs,                libSelectedCondition );
  73         ctxMenu.AddItem( ACTIONS::revert,                libSelectedCondition );
  74 
  75         ctxMenu.AddSeparator();
  76         ctxMenu.AddItem( EE_ACTIONS::newSymbol,          libSelectedCondition );
  77         ctxMenu.AddItem( EE_ACTIONS::editSymbol,         symbolSelectedCondition );
  78 
  79         ctxMenu.AddSeparator();
  80         ctxMenu.AddItem( ACTIONS::save,                  symbolSelectedCondition );
  81         ctxMenu.AddItem( ACTIONS::saveCopyAs,            symbolSelectedCondition );
  82         ctxMenu.AddItem( EE_ACTIONS::duplicateSymbol,    symbolSelectedCondition );
  83         ctxMenu.AddItem( EE_ACTIONS::deleteSymbol,       symbolSelectedCondition );
  84         ctxMenu.AddItem( ACTIONS::revert,                symbolSelectedCondition );
  85 
  86         ctxMenu.AddSeparator();
  87         ctxMenu.AddItem( EE_ACTIONS::cutSymbol,          symbolSelectedCondition );
  88         ctxMenu.AddItem( EE_ACTIONS::copySymbol,         symbolSelectedCondition );
  89         ctxMenu.AddItem( EE_ACTIONS::pasteSymbol,        libSelectedCondition );
  90 
  91         ctxMenu.AddSeparator();
  92         ctxMenu.AddItem( EE_ACTIONS::importSymbol,       libSelectedCondition );
  93         ctxMenu.AddItem( EE_ACTIONS::exportSymbol,       symbolSelectedCondition );
  94     }
  95 
  96     return true;
  97 }

 207 int LIB_CONTROL::OnDeMorgan( const TOOL_EVENT& aEvent )
 208 {
 209     int convert = aEvent.IsAction( &EE_ACTIONS::showDeMorganStandard ) ?
 210             LIB_ITEM::LIB_CONVERT::BASE : LIB_ITEM::LIB_CONVERT::DEMORGAN;
 211 
 212     if( m_frame->IsType( FRAME_SCH_LIB_EDITOR ) )
 213     {
 214         m_toolMgr->RunAction( ACTIONS::cancelInteractive, true );
 215         m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 216 
 217         LIB_EDIT_FRAME* libEditFrame = static_cast<LIB_EDIT_FRAME*>( m_frame );
 218         libEditFrame->SetConvert( convert );
 219 
 220         m_toolMgr->ResetTools( TOOL_BASE::MODEL_RELOAD );
 221         libEditFrame->RebuildView();
 222     }
 223     else if( m_frame->IsType( FRAME_SCH_VIEWER ) || m_frame->IsType( FRAME_SCH_VIEWER_MODAL ) )
 224     {
 225         LIB_VIEW_FRAME* libViewFrame = static_cast<LIB_VIEW_FRAME*>( m_frame );
 226         libViewFrame->SetUnitAndConvert( libViewFrame->GetUnit(), convert );
 227     }
 228 
 229     return 0;
 230 }

 306 int LIB_CONTROL::ExportView( const TOOL_EVENT& aEvent )
 307 {
 308     if( !m_isLibEdit )
 309         return 0;
 310 
 311     LIB_EDIT_FRAME* editFrame = getEditFrame<LIB_EDIT_FRAME>();
 312     LIB_PART*       part = editFrame->GetCurPart();
 313 
 314     if( !part )
 315     {
 316         wxMessageBox( _( "No symbol to export" ) );
 317         return 0;
 318     }
 319 
 320     wxString   file_ext = wxT( "png" );
 321     wxString   mask = wxT( "*." ) + file_ext;
 322     wxFileName fn( part->GetName() );
 323     fn.SetExt( "png" );
 324 
 325     wxString projectPath = wxPathOnly( m_frame->Prj().GetProjectFullName() );
 326 
 327     wxFileDialog dlg( editFrame, _( "Image File Name" ), projectPath, fn.GetFullName(),
 328                       PngFileWildcard(), wxFD_SAVE | wxFD_OVERWRITE_PROMPT );
 329 
 330     if( dlg.ShowModal() == wxID_OK && !dlg.GetPath().IsEmpty() )
 331     {
 332         // calling wxYield is mandatory under Linux, after closing the file selector dialog

 333         // to refresh the screen before creating the PNG or JPEG image from screen

 334         wxYield();
 335 
 336         if( !SaveCanvasImageToFile( editFrame, dlg.GetPath(), wxBITMAP_TYPE_PNG ) )
 337         {
 338             wxMessageBox( wxString::Format( _( "Can't save file \"%s\"." ), dlg.GetPath() ) );
 339         }
 340     }
 341 
 342     return 0;
 343 }

 346 int LIB_CONTROL::ExportSymbolAsSVG( const TOOL_EVENT& aEvent )
 347 {
 348     if( !m_isLibEdit )
 349         return 0;
 350 
 351     LIB_EDIT_FRAME* editFrame = getEditFrame<LIB_EDIT_FRAME>();
 352     LIB_PART*       part = editFrame->GetCurPart();
 353 
 354     if( !part )
 355     {
 356         wxMessageBox( _( "No symbol to export" ) );
 357         return 0;
 358     }
 359 
 360     wxString   file_ext = wxT( "svg" );
 361     wxString   mask     = wxT( "*." ) + file_ext;
 362     wxFileName fn( part->GetName() );
 363     fn.SetExt( file_ext );
 364 
 365     wxString pro_dir = wxPathOnly( m_frame->Prj().GetProjectFullName() );
 366 
 367     wxString fullFileName = EDA_FILE_SELECTOR( _( "Filename:" ), pro_dir, fn.GetFullName(),
 368                                                file_ext, mask, m_frame, wxFD_SAVE, true );
 369 
 370     if( !fullFileName.IsEmpty() )
 371     {
 372         PAGE_INFO pageSave = editFrame->GetScreen()->GetPageSettings();
 373         PAGE_INFO pageTemp = pageSave;
 374 
 375         wxSize componentSize = part->GetUnitBoundingBox( editFrame->GetUnit(),
 376                                                          editFrame->GetConvert() ).GetSize();
 377 
 378         // Add a small margin to the plot bounding box

 379         pageTemp.SetWidthMils(  int( componentSize.x * 1.2 ) );
 380         pageTemp.SetHeightMils( int( componentSize.y * 1.2 ) );
 381 
 382         editFrame->GetScreen()->SetPageSettings( pageTemp );
 383         editFrame->SVG_PlotComponent( fullFileName );
 384         editFrame->GetScreen()->SetPageSettings( pageSave );
 385     }
 386 
 387     return 0;
 388 }

 391 int LIB_CONTROL::AddSymbolToSchematic( const TOOL_EVENT& aEvent )
 392 {
 393     LIB_PART* part = nullptr;
 394     LIB_ID    libId;
 395     int       unit, convert;
 396 
 397     if( m_isLibEdit )
 398     {
 399         LIB_EDIT_FRAME* editFrame = getEditFrame<LIB_EDIT_FRAME>();
 400 
 401         part = editFrame->GetCurPart();
 402         unit = editFrame->GetUnit();
 403         convert = editFrame->GetConvert();
 404 
 405         if( part )
 406             libId = part->GetLibId();
 407     }
 408     else
 409     {
 410         LIB_VIEW_FRAME* viewFrame = getEditFrame<LIB_VIEW_FRAME>();
 411 
 412         if( viewFrame->IsModal() )
 413         {
 414             // if we're modal then we just need to return the symbol selection; the caller is

 415             // already in a EE_ACTIONS::placeSymbol coroutine.

 416             viewFrame->FinishModal();
 417             return 0;
 418         }
 419         else
 420         {
 421             part    = viewFrame->GetSelectedSymbol();
 422             unit    = viewFrame->GetUnit();
 423             convert = viewFrame->GetConvert();
 424 
 425             if( part )
 426                 libId = part->GetLibId();
 427         }
 428     }
 429 
 430     if( part )
 431     {
 432         SCH_EDIT_FRAME* schframe = (SCH_EDIT_FRAME*) m_frame->Kiway().Player( FRAME_SCH, false );
 433 
 434         if( !schframe )      // happens when the schematic editor is not active (or closed)

 435         {
 436             DisplayErrorMessage( m_frame, _( "No schematic currently open." ) );
 437             return 0;
 438         }
 439 
 440         wxCHECK( part->GetLibId().IsValid(), 0 );
 441 
 442         SCH_COMPONENT* comp =
 443                 new SCH_COMPONENT( *part, libId, &schframe->GetCurrentSheet(), unit, convert );
 444 
 445         comp->SetParent( schframe->GetCurrentSheet().LastScreen() );
 446 
 447         if( schframe->eeconfig()->m_AutoplaceFields.enable )
 448             comp->AutoplaceFields( /* aScreen */ nullptr, /* aManual */ false );
 449 
 450         schframe->Raise();
 451         schframe->GetToolManager()->RunAction( EE_ACTIONS::placeSymbol, true, comp );
 452     }
 453 
 454     return 0;
 455 }
