  67 bool SCH_EDIT_FRAME::SaveEEFile( SCH_SHEET* aSheet, bool aSaveUnderNewName )
  68 {
  69     wxString msg;
  70     wxFileName schematicFileName;
  71     bool success;
  72 
  73     if( aSheet == NULL )
  74         aSheet = GetCurrentSheet().Last();
  75 
  76     SCH_SCREEN* screen = aSheet->GetScreen();
  77 
  78     wxCHECK( screen, false );
  79 
  80     // If no name exists in the window yet - save as new.

  81     if( screen->GetFileName().IsEmpty() )
  82         aSaveUnderNewName = true;
  83 
  84     // Construct the name of the file to be saved

  85     schematicFileName = Prj().AbsolutePath( screen->GetFileName() );
  86 
  87     if( aSaveUnderNewName )
  88     {
  89         wxString wildcards = KiCadSchematicFileWildcard();
  90 
  91         wxFileDialog dlg( this, _( "Schematic Files" ), wxPathOnly( Prj().GetProjectFullName() ),
  92                           schematicFileName.GetFullName(), wildcards,
  93                           wxFD_SAVE | wxFD_OVERWRITE_PROMPT );
  94 
  95         if( dlg.ShowModal() == wxID_CANCEL )
  96             return false;
  97 
  98         schematicFileName = dlg.GetPath();
  99 
 100         if( schematicFileName.GetExt().IsEmpty() )
 101             schematicFileName.SetExt( KiCadSchematicFileExtension );
 102     }
 103 
 104     if( !IsWritable( schematicFileName ) )
 105         return false;
 106 
 107     wxFileName tempFile( schematicFileName );
 108     tempFile.SetName( wxT( "." ) + tempFile.GetName() );
 109     tempFile.SetExt( tempFile.GetExt() + wxT( "$" ) );
 110 
 111     // Save

 112     wxLogTrace( traceAutoSave,
 113                 wxT( "Saving file <" ) + schematicFileName.GetFullPath() + wxT( ">" ) );
 114 
 115     SCH_IO_MGR::SCH_FILE_T pluginType = SCH_IO_MGR::GuessPluginTypeFromSchPath(
 116             schematicFileName.GetFullPath() );
 117     SCH_PLUGIN::SCH_PLUGIN_RELEASER pi( SCH_IO_MGR::FindPlugin( pluginType ) );
 118 
 119     try
 120     {
 121         pi->Save( tempFile.GetFullPath(), aSheet, &Schematic() );
 122         success = true;
 123     }
 124     catch( const IO_ERROR& ioe )
 125     {
 126         msg.Printf( _( "Error saving schematic file \"%s\".\n%s" ),
 127                     schematicFileName.GetFullPath(), ioe.What() );
 128         DisplayError( this, msg );
 129 
 130         msg.Printf( _( "Failed to create temporary file \"%s\"" ), tempFile.GetFullPath() );
 131         AppendMsgPanel( wxEmptyString, msg, CYAN );
 132 
 133         // In case we started a file but didn't fully write it, clean up

 134         wxRemoveFile( tempFile.GetFullPath() );
 135 
 136         success = false;
 137     }
 138 
 139     if( success )
 140     {
 141         // Replace the original with the temporary file we just wrote

 142         success = wxRenameFile( tempFile.GetFullPath(), schematicFileName.GetFullPath() );
 143 
 144         if( !success )
 145         {
 146             msg.Printf( _( "Error saving schematic file \"%s\".\n"
 147                            "Failed to rename temporary file %s" ),
 148                         schematicFileName.GetFullPath(), tempFile.GetFullPath() );
 149             DisplayError( this, msg );
 150 
 151             msg.Printf( _( "Failed to rename temporary file \"%s\"" ), tempFile.GetFullPath() );
 152             AppendMsgPanel( wxEmptyString, msg, CYAN );
 153         }
 154     }
 155 
 156     if( success )
 157     {
 158         // Delete auto save file.

 159         wxFileName autoSaveFileName = schematicFileName;
 160         autoSaveFileName.SetName( GetAutoSaveFilePrefix() + schematicFileName.GetName() );
 161 
 162         if( autoSaveFileName.FileExists() )
 163         {
 164             wxLogTrace( traceAutoSave,
 165                         wxT( "Removing auto save file <" ) + autoSaveFileName.GetFullPath() +
 166                         wxT( ">" ) );
 167 
 168             wxRemoveFile( autoSaveFileName.GetFullPath() );
 169         }
 170 
 171         // Update the screen and frame info and reset the lock file.

 172         if( aSaveUnderNewName )
 173         {
 174             screen->SetFileName( schematicFileName.GetFullPath() );
 175             LockFile( schematicFileName.GetFullPath() );
 176         }
 177 
 178         screen->ClrSave();
 179         screen->ClrModify();
 180 
 181         msg.Printf( _( "File %s saved" ),  screen->GetFileName() );
 182         SetStatusText( msg, 0 );
 183     }
 184     else
 185     {
 186         DisplayError( this, _( "File write operation failed." ) );
 187     }
 188 
 189     return success;
 190 }

 218 bool SCH_EDIT_FRAME::OpenProjectFiles( const std::vector<wxString>& aFileSet, int aCtl )
 219 {
 220     // implement the pseudo code from KIWAY_PLAYER.h:

 221     wxString msg;
 222 
 223     auto cfg = dynamic_cast<EESCHEMA_SETTINGS*>( Kiface().KifaceSettings() );
 224 
 225     // This is for python:

 226     if( aFileSet.size() != 1 )
 227     {
 228         msg.Printf( "Eeschema:%s() takes only a single filename.", __WXFUNCTION__ );
 229         DisplayError( this, msg );
 230         return false;
 231     }
 232 
 233     wxString fullFileName( aFileSet[0] );
 234 
 235     // We insist on caller sending us an absolute path, if it does not, we say it's a bug.

 236     wxASSERT_MSG( wxFileName( fullFileName ).IsAbsolute(), wxT( "Path is not absolute!" ) );
 237 
 238     if( !LockFile( fullFileName ) )
 239     {
 240         msg.Printf( _( "Schematic file \"%s\" is already open." ), fullFileName );
 241         DisplayError( this, msg );
 242         return false;
 243     }
 244 
 245     if( !AskToSaveChanges() )
 246         return false;
 247 
 248     PROF_COUNTER openFiles( "OpenProjectFile" );
 249 
 250     wxFileName pro = fullFileName;
 251     pro.SetExt( ProjectFileExtension );
 252 
 253     bool is_new = !wxFileName::IsFileReadable( fullFileName );
 254 
 255     // If its a non-existent schematic and caller thinks it exists

 256     if( is_new && !( aCtl & KICTL_CREATE ) )
 257     {
 258         // notify user that fullFileName does not exist, ask if user wants to create it.

 259         msg.Printf( _( "Schematic \"%s\" does not exist.  Do you wish to create it?" ),
 260                     fullFileName );
 261 
 262         if( !IsOK( this, msg ) )
 263             return false;
 264     }
 265 
 266     // unload current project file before loading new

 267     {
 268         SetScreen( nullptr );
 269         CreateScreens();
 270     }
 271 
 272     SetStatusText( wxEmptyString );
 273     ClearMsgPanel();
 274 
 275     SCH_IO_MGR::SCH_FILE_T schFileType = SCH_IO_MGR::GuessPluginTypeFromSchPath( fullFileName );
 276 
 277     // PROJECT::SetProjectFullName() is an impactful function.  It should only be

 278     // called under carefully considered circumstances.

 279 
 280     // The calling code should know not to ask me here to change projects unless

 281     // it knows what consequences that will have on other KIFACEs running and using

 282     // this same PROJECT.  It can be very harmful if that calling code is stupid.

 283 
 284     // NOTE: The calling code should never call this in hosted (non-standalone) mode with a

 285     // different project than what has been loaded by the manager frame.  This will crash.

 286 
 287     bool differentProject = pro.GetFullPath() != Prj().GetProjectFullName();
 288 
 289     if( differentProject )
 290     {
 291         GetSettingsManager()->SaveProject();
 292         Schematic().SetProject( nullptr );
 293         GetSettingsManager()->UnloadProject( &Prj() );
 294         GetSettingsManager()->LoadProject( pro.GetFullPath() );
 295         CreateScreens();
 296     }
 297 
 298     if( schFileType == SCH_IO_MGR::SCH_LEGACY )
 299     {
 300         // Don't reload the symbol libraries if we are just launching Eeschema from KiCad again.

 301         // They are already saved in the kiface project object.

 302         if( differentProject || !Prj().GetElem( PROJECT::ELEM_SCH_PART_LIBS ) )
 303         {
 304             // load the libraries here, not in SCH_SCREEN::Draw() which is a context

 305             // that will not tolerate DisplayError() dialog since we're already in an

 306             // event handler in there.

 307             // And when a schematic file is loaded, we need these libs to initialize

 308             // some parameters (links to PART LIB, dangling ends ...)

 309             Prj().SetElem( PROJECT::ELEM_SCH_PART_LIBS, NULL );
 310             Prj().SchLibs();
 311         }
 312     }
 313     else
 314     {
 315         // No legacy symbol libraries including the cache are loaded with the new file format.

 316         Prj().SetElem( PROJECT::ELEM_SCH_PART_LIBS, NULL );
 317     }
 318 
 319     // Load the symbol library table, this will be used forever more.

 320     Prj().SetElem( PROJECT::ELEM_SYMBOL_LIB_TABLE, NULL );
 321     Prj().SchSymbolLibTable();
 322 
 323     // Load project settings after schematic has been set up with the project link, since this will

 324     // update some of the needed schematic settings such as drawing defaults

 325     LoadProjectSettings();
 326 
 327     SetShutdownBlockReason( _( "Schematic file changes are unsaved" ) );
 328 
 329     if( is_new )
 330     {
 331         // mark new, unsaved file as modified.

 332         GetScreen()->SetModify();
 333     }
 334     else
 335     {
 336         SetScreen( nullptr );
 337 
 338         SCH_PLUGIN* plugin = SCH_IO_MGR::FindPlugin( schFileType );
 339         SCH_PLUGIN::SCH_PLUGIN_RELEASER pi( plugin );
 340 
 341         // This will rename the file if there is an autosave and the user want to recover

 342         CheckForAutoSaveFile( fullFileName );
 343 
 344         try
 345         {
 346             Schematic().SetRoot( pi->Load( fullFileName, &Schematic() ) );
 347 
 348             if( !pi->GetError().IsEmpty() )
 349             {
 350                 DisplayErrorMessage( this,
 351                                      _( "The entire schematic could not be loaded.  Errors "
 352                                         "occurred attempting to load \nhierarchical sheet "
 353                                         "schematics." ),
 354                                      pi->GetError() );
 355             }
 356         }
 357         catch( const IO_ERROR& ioe )
 358         {
 359             // Do not leave g_RootSheet == NULL because it is expected to be

 360             // a valid sheet. Therefore create a dummy empty root sheet and screen.

 361             CreateScreens();
 362             m_toolManager->RunAction( ACTIONS::zoomFitScreen, true );
 363 
 364             msg.Printf( _( "Error loading schematic file \"%s\".\n%s" ),
 365                         fullFileName, ioe.What() );
 366             DisplayError( this, msg );
 367 
 368             msg.Printf( _( "Failed to load \"%s\"" ), fullFileName );
 369             AppendMsgPanel( wxEmptyString, msg, CYAN );
 370 
 371             return false;
 372         }
 373 
 374         // It's possible the schematic parser fixed errors due to bugs so warn the user

 375         // that the schematic has been fixed (modified).

 376         SCH_SHEET_LIST sheetList = Schematic().GetSheets();
 377 
 378         if( sheetList.IsModified() )
 379         {
 380             DisplayInfoMessage( this,
 381                                 _( "An error was found when loading the schematic that has "
 382                                    "been automatically fixed.  Please save the schematic to "
 383                                    "repair the broken file or it may not be usable with other "
 384                                    "versions of KiCad." ) );
 385         }
 386 
 387         UpdateFileHistory( fullFileName );
 388 
 389         SCH_SCREENS schematic( Schematic().Root() );
 390 
 391         // LIB_ID checks and symbol rescue only apply to the legacy file formats.

 392         if( schFileType == SCH_IO_MGR::SCH_LEGACY )
 393         {
 394             // Convert old projects over to use symbol library table.

 395             if( schematic.HasNoFullyDefinedLibIds() )
 396             {
 397                 DIALOG_SYMBOL_REMAP dlgRemap( this );
 398 
 399                 dlgRemap.ShowQuasiModal();
 400             }
 401             else
 402             {
 403                 // Double check to ensure no legacy library list entries have been

 404                 // added to the projec file symbol library list.

 405                 wxString paths;
 406                 wxArrayString libNames;
 407 
 408                 PART_LIBS::LibNamesAndPaths( &Prj(), false, &paths, &libNames );
 409 
 410                 if( !libNames.IsEmpty() )
 411                 {
 412                     if( eeconfig()->m_Appearance.show_illegal_symbol_lib_dialog )
 413                     {
 414                         wxRichMessageDialog invalidLibDlg(
 415                                 this,
 416                                 _( "Illegal entry found in project file symbol library list." ),
 417                                 _( "Project Load Warning" ),
 418                                 wxOK | wxCENTER | wxICON_EXCLAMATION );
 419                         invalidLibDlg.ShowDetailedText(
 420                                 _( "Symbol libraries defined in the project file symbol library "
 421                                    "list are no longer supported and will be\nremoved.  This may "
 422                                    "cause broken symbol library links under certain conditions." ) );
 423                         invalidLibDlg.ShowCheckBox( _( "Do not show this dialog again." ) );
 424                         invalidLibDlg.ShowModal();
 425                         eeconfig()->m_Appearance.show_illegal_symbol_lib_dialog =
 426                                 !invalidLibDlg.IsCheckBoxChecked();
 427                     }
 428 
 429                     libNames.Clear();
 430                     paths.Clear();
 431                     PART_LIBS::LibNamesAndPaths( &Prj(), true, &paths, &libNames );
 432                 }
 433 
 434                 if( !cfg || !cfg->m_RescueNeverShow )
 435                 {
 436                     SCH_EDITOR_CONTROL* editor = m_toolManager->GetTool<SCH_EDITOR_CONTROL>();
 437                     editor->RescueSymbolLibTableProject( false );
 438                 }
 439             }
 440 
 441             // Update all symbol library links for all sheets.

 442             schematic.UpdateSymbolLinks();
 443 
 444             if( !cfg || cfg->m_Appearance.show_sexpr_file_convert_warning )
 445             {
 446                 wxRichMessageDialog newFileFormatDlg(
 447                         this,
 448                         _( "The schematic file will be converted to the new file format on save." ),
 449                         _( "Project Load Warning" ),
 450                         wxOK | wxCENTER | wxICON_EXCLAMATION );
 451                 newFileFormatDlg.ShowDetailedText(
 452                         _( "This schematic was saved in the legacy file format which is no "
 453                            "longer supported and will be saved\nusing the new file format.  The "
 454                            "new file format cannot be opened with previous versions of KiCad." ) );
 455                 newFileFormatDlg.ShowCheckBox( _( "Do not show this dialog again." ) );
 456                 newFileFormatDlg.ShowModal();
 457 
 458                 if( cfg )
 459                     cfg->m_Appearance.show_sexpr_file_convert_warning =
 460                             !newFileFormatDlg.IsCheckBoxChecked();
 461             }
 462 
 463             // Legacy schematic can have duplicate time stamps so fix that before converting

 464             // to the s-expression format.

 465             schematic.ReplaceDuplicateTimeStamps();
 466 
 467             // Allow the schematic to be saved to new file format without making any edits.

 468             OnModify();
 469         }
 470         else  // S-expression schematic.

 471         {
 472             for( SCH_SCREEN* screen = schematic.GetFirst(); screen; screen = schematic.GetNext() )
 473                 screen->UpdateLocalLibSymbolLinks();
 474 
 475             // Restore all of the loaded symbol instances from the root sheet screen.

 476             sheetList.UpdateSymbolInstances( Schematic().RootScreen()->m_symbolInstances );
 477         }
 478 
 479         Schematic().ConnectionGraph()->Reset();
 480 
 481         SetScreen( GetCurrentSheet().LastScreen() );
 482 
 483         // Migrate conflicting bus definitions

 484         // TODO(JE) This should only run once based on schematic file version

 485         if( Schematic().ConnectionGraph()->GetBusesNeedingMigration().size() > 0 )
 486         {
 487             DIALOG_MIGRATE_BUSES dlg( this );
 488             dlg.ShowQuasiModal();
 489             RecalculateConnections( NO_CLEANUP );
 490             OnModify();
 491         }
 492 
 493         GetScreen()->TestDanglingEnds();    // Only perform the dangling end test on root sheet.

 494         RecalculateConnections( GLOBAL_CLEANUP );
 495         ClearUndoRedoList();
 496         GetScreen()->m_Initialized = true;
 497     }
 498 
 499     m_toolManager->RunAction( ACTIONS::zoomFitScreen, true );
 500     SetSheetNumberAndCount();
 501 
 502     // re-create junctions if needed. Eeschema optimizes wires by merging

 503     // colinear segments. If a schematic is saved without a valid

 504     // cache library or missing installed libraries, this can cause connectivity errors

 505     // unless junctions are added.

 506     FixupJunctions();
 507 
 508     SyncView();
 509     GetScreen()->ClearDrawingState();
 510 
 511     UpdateTitle();
 512 
 513     wxFileName fn = Prj().AbsolutePath( GetScreen()->GetFileName() );
 514     m_infoBar->Dismiss();
 515 
 516     if( fn.FileExists() && !fn.IsFileWritable() )
 517     {
 518         m_infoBar->RemoveAllButtons();
 519         m_infoBar->AddCloseButton();
 520         m_infoBar->ShowMessage( "Schematic file is read only.", wxICON_WARNING );
 521     }
 522 
 523 #ifdef PROFILE
 524     openFiles.Show();
 525 #endif
 526 
 527     return true;
 528 }

 531 bool SCH_EDIT_FRAME::AppendSchematic()
 532 {
 533     wxString    fullFileName;
 534     SCH_SCREEN* screen = GetScreen();
 535 
 536     if( !screen )
 537     {
 538         wxLogError( wxT( "Document not ready, cannot import" ) );
 539         return false;
 540     }
 541 
 542     // open file chooser dialog

 543     wxString path = wxPathOnly( Prj().GetProjectFullName() );
 544 
 545     wxFileDialog dlg( this, _( "Append Schematic" ), path, wxEmptyString,
 546                       KiCadSchematicFileWildcard(), wxFD_OPEN | wxFD_FILE_MUST_EXIST );
 547 
 548     if( dlg.ShowModal() == wxID_CANCEL )
 549         return false;
 550 
 551     fullFileName = dlg.GetPath();
 552 
 553     if( !LoadSheetFromFile( GetCurrentSheet().Last(), &GetCurrentSheet(), fullFileName ) )
 554         return false;
 555 
 556     SCH_SCREENS screens( GetCurrentSheet().Last() );
 557     screens.TestDanglingEnds();
 558 
 559     m_toolManager->RunAction( ACTIONS::zoomFitScreen, true );
 560     SetSheetNumberAndCount();
 561 
 562     SyncView();
 563     HardRedraw();   // Full reinit of the current screen and the display.

 564     OnModify();
 565 
 566     return true;
 567 }

 585 void SCH_EDIT_FRAME::OnImportProject( wxCommandEvent& aEvent )
 586 {
 587     if( !AskToSaveChanges() )
 588         return;
 589 
 590     // Set the project location if none is set

 591     bool setProject = Prj().GetProjectFullName().IsEmpty();
 592     wxString path = wxPathOnly( Prj().GetProjectFullName() );
 593 
 594     wxFileDialog dlg( this, _( "Import Schematic" ), path, wxEmptyString,
 595                       EagleSchematicFileWildcard(), wxFD_OPEN | wxFD_FILE_MUST_EXIST );
 596 
 597     if( dlg.ShowModal() == wxID_CANCEL )
 598         return;
 599 
 600     if( setProject )
 601     {
 602         GetSettingsManager()->SaveProject();
 603         Schematic().SetProject( nullptr );
 604         GetSettingsManager()->UnloadProject( &Prj() );
 605 
 606         Schematic().Reset();
 607 
 608         wxFileName projectFn( dlg.GetPath() );
 609         projectFn.SetExt( ProjectFileExtension );
 610         GetSettingsManager()->LoadProject( projectFn.GetFullPath() );
 611 
 612         Schematic().SetProject( &Prj() );
 613     }
 614 
 615     // For now there is only one import plugin

 616     importFile( dlg.GetPath(), SCH_IO_MGR::SCH_EAGLE );
 617 }

 620 bool SCH_EDIT_FRAME::SaveProject()
 621 {
 622     wxString msg;
 623     SCH_SCREEN* screen;
 624     SCH_SCREENS screens( Schematic().Root() );
 625     bool success = true;
 626     bool updateFileType = false;
 627 
 628     // I want to see it in the debugger, show me the string!  Can't do that with wxFileName.

 629     wxString    fileName = Prj().AbsolutePath( Schematic().Root().GetFileName() );
 630     wxFileName  fn = fileName;
 631 
 632     if( !fn.IsDirWritable() )
 633     {
 634         msg = wxString::Format( _( "Directory \"%s\" is not writable." ), fn.GetPath() );
 635         DisplayError( this, msg );
 636         return false;
 637     }
 638 
 639     // Warn user on potential file overwrite.  This can happen on shared sheets.

 640     wxArrayString overwrittenFiles;
 641 
 642     for( size_t i = 0; i < screens.GetCount(); i++ )
 643     {
 644         screen = screens.GetScreen( i );
 645 
 646         wxCHECK2( screen, continue );
 647 
 648         // Convert legacy schematics file name extensions for the new format.

 649         wxFileName tmpFn = screen->GetFileName();
 650 
 651         if( tmpFn.GetExt() == KiCadSchematicFileExtension )
 652             continue;
 653 
 654         tmpFn.SetExt( KiCadSchematicFileExtension );
 655 
 656         if( tmpFn.FileExists() )
 657             overwrittenFiles.Add( tmpFn.GetFullPath() );
 658     }
 659 
 660     if( !overwrittenFiles.IsEmpty() )
 661     {
 662         for( auto overwrittenFile : overwrittenFiles )
 663         {
 664             if( msg.IsEmpty() )
 665                 msg = overwrittenFile;
 666             else
 667                 msg += "\n" + overwrittenFile;
 668         }
 669 
 670         wxRichMessageDialog dlg(
 671                 this,
 672                 _( "Saving the project to the new file format will overwrite existing files." ),
 673                 _( "Project Save Warning" ),
 674                 wxOK | wxCANCEL | wxCANCEL_DEFAULT | wxCENTER | wxICON_EXCLAMATION );
 675         dlg.ShowDetailedText( wxString::Format(
 676                               _( "The following files will be overwritten:\n\n%s" ), msg ) );
 677         dlg.SetOKCancelLabels( wxMessageDialog::ButtonLabel( _( "Overwrite Files" ) ),
 678                 wxMessageDialog::ButtonLabel( _( "Abort Project Save" ) ) );
 679 
 680         if( dlg.ShowModal() == wxID_CANCEL )
 681             return false;
 682     }
 683 
 684     for( size_t i = 0; i < screens.GetCount(); i++ )
 685     {
 686         screen = screens.GetScreen( i );
 687 
 688         wxCHECK2( screen, continue );
 689 
 690         // Convert legacy schematics file name extensions for the new format.

 691         wxFileName tmpFn = screen->GetFileName();
 692 
 693         if( tmpFn.GetExt() != KiCadSchematicFileExtension )
 694         {
 695             updateFileType = true;
 696             tmpFn.SetExt( KiCadSchematicFileExtension );
 697 
 698             for( auto item : screen->Items().OfType( SCH_SHEET_T ) )
 699             {
 700                 SCH_SHEET* sheet = static_cast<SCH_SHEET*>( item );
 701                 wxFileName sheetFileName = sheet->GetFileName();
 702 
 703                 if( sheetFileName.GetExt() == KiCadSchematicFileExtension )
 704                     continue;
 705 
 706                 sheetFileName.SetExt( KiCadSchematicFileExtension );
 707                 sheet->SetFileName( sheetFileName.GetFullPath() );
 708                 RefreshItem( sheet );
 709             }
 710 
 711             screen->SetFileName( tmpFn.GetFullPath() );
 712         }
 713 
 714         success &= SaveEEFile( screens.GetSheet( i ) );
 715     }
 716 
 717     if( updateFileType )
 718         UpdateFileHistory( Schematic().RootScreen()->GetFileName() );
 719 
 720     // Save the sheet name map to the project file

 721     std::vector<FILE_INFO_PAIR>& sheets = Prj().GetProjectFile().GetSheets();
 722     sheets.clear();
 723 
 724     for( SCH_SHEET_PATH& sheetPath : Schematic().GetSheets() )
 725     {
 726         SCH_SHEET* sheet = sheetPath.Last();
 727         sheets.emplace_back( std::make_pair( sheet->m_Uuid, sheet->GetName() ) );
 728     }
 729 
 730     Pgm().GetSettingsManager().SaveProject();
 731 
 732     if( !Kiface().IsSingle() )
 733     {
 734         WX_STRING_REPORTER backupReporter( &msg );
 735 
 736         if( !GetSettingsManager()->TriggerBackupIfNeeded( backupReporter ) )
 737             SetStatusText( msg, 0 );
 738     }
 739 
 740     UpdateTitle();
 741 
 742     return success;
 743 }

 746 bool SCH_EDIT_FRAME::doAutoSave()
 747 {
 748     wxFileName  tmpFileName = Schematic().Root().GetFileName();
 749     wxFileName  fn = tmpFileName;
 750     wxFileName  tmp;
 751     SCH_SCREENS screens( Schematic().Root() );
 752 
 753     bool autoSaveOk = true;
 754 
 755     tmp.AssignDir( fn.GetPath() );
 756 
 757     if( !tmp.IsOk() )
 758         return false;
 759 
 760     if( !IsWritable( tmp ) )
 761         return false;
 762 
 763     for( size_t i = 0; i < screens.GetCount(); i++ )
 764     {
 765         // Only create auto save files for the schematics that have been modified.

 766         if( !screens.GetScreen( i )->IsSave() )
 767             continue;
 768 
 769         tmpFileName = fn = screens.GetScreen( i )->GetFileName();
 770 
 771         // Auto save file name is the normal file name prefixed with GetAutoSavePrefix().

 772         fn.SetName( GetAutoSaveFilePrefix() + fn.GetName() );
 773 
 774         screens.GetScreen( i )->SetFileName( fn.GetFullPath() );
 775 
 776         if( SaveEEFile( screens.GetSheet( i ), false ) )
 777             screens.GetScreen( i )->SetModify();
 778         else
 779             autoSaveOk = false;
 780 
 781         screens.GetScreen( i )->SetFileName( tmpFileName.GetFullPath() );
 782     }
 783 
 784     if( autoSaveOk )
 785     {
 786         m_autoSaveState = false;
 787 
 788         if( !Kiface().IsSingle() &&
 789             GetSettingsManager()->GetCommonSettings()->m_Backup.backup_on_autosave )
 790         {
 791             GetSettingsManager()->TriggerBackupIfNeeded( NULL_REPORTER::GetInstance() );
 792         }
 793     }
 794 
 795     return autoSaveOk;
 796 }

 799 bool SCH_EDIT_FRAME::importFile( const wxString& aFileName, int aFileType )
 800 {
 801     wxFileName newfilename;
 802     SCH_SHEET_LIST sheetList = Schematic().GetSheets();
 803 
 804     switch( (SCH_IO_MGR::SCH_FILE_T) aFileType )
 805     {
 806     case SCH_IO_MGR::SCH_EAGLE:
 807         // We insist on caller sending us an absolute path, if it does not, we say it's a bug.

 808         wxASSERT_MSG( wxFileName( aFileName ).IsAbsolute(),
 809                       wxT( "Import eagle schematic caller didn't send full filename" ) );
 810 
 811         if( !LockFile( aFileName ) )
 812         {
 813             wxString msg = wxString::Format( _( "Schematic file \"%s\" is already open." ),
 814                                              aFileName );
 815             DisplayError( this, msg );
 816             return false;
 817         }
 818 
 819         try
 820         {
 821             SCH_PLUGIN::SCH_PLUGIN_RELEASER pi( SCH_IO_MGR::FindPlugin( SCH_IO_MGR::SCH_EAGLE ) );
 822             Schematic().SetRoot( pi->Load( aFileName, &Schematic() ) );
 823 
 824             // Eagle sheets do not use a worksheet frame by default, so set it to an empty one

 825             WS_DATA_MODEL& pglayout = WS_DATA_MODEL::GetTheInstance();
 826             pglayout.SetEmptyLayout();
 827 
 828             BASE_SCREEN::m_PageLayoutDescrFileName = "empty.kicad_wks";
 829             wxFileName layoutfn( Prj().GetProjectPath(), BASE_SCREEN::m_PageLayoutDescrFileName );
 830             wxFile layoutfile;
 831 
 832             if( layoutfile.Create( layoutfn.GetFullPath() ) )
 833             {
 834                 layoutfile.Write( WS_DATA_MODEL::EmptyLayout() );
 835                 layoutfile.Close();
 836             }
 837 
 838             newfilename.SetPath( Prj().GetProjectPath() );
 839             newfilename.SetName( Prj().GetProjectName() );
 840             newfilename.SetExt( LegacySchematicFileExtension );
 841 
 842             SetScreen( GetCurrentSheet().LastScreen() );
 843 
 844             Schematic().Root().SetFileName( newfilename.GetFullPath() );
 845             GetScreen()->SetFileName( newfilename.GetFullPath() );
 846             GetScreen()->SetModify();
 847 
 848             SaveProjectSettings();
 849 
 850             UpdateFileHistory( aFileName );
 851             SCH_SCREENS schematic( Schematic().Root() );
 852             schematic.UpdateSymbolLinks();      // Update all symbol library links for all sheets.

 853 
 854             GetScreen()->m_Initialized = true;
 855 
 856             for( SCH_SCREEN* screen = schematic.GetFirst(); screen; screen = schematic.GetNext() )
 857             {
 858                 for( auto item : screen->Items().OfType( SCH_COMPONENT_T ) )
 859                 {
 860                     std::vector<wxPoint> pts;
 861                     SCH_COMPONENT*       cmp = static_cast<SCH_COMPONENT*>( item );
 862 
 863                     // Update footprint LIB_ID to point to the imported Eagle library

 864                     auto fpField = cmp->GetField( FOOTPRINT );
 865 
 866                     if( !fpField->GetText().IsEmpty() )
 867                     {
 868                         LIB_ID fpId;
 869                         fpId.Parse( fpField->GetText(), LIB_ID::ID_SCH, true );
 870                         fpId.SetLibNickname( newfilename.GetName() );
 871                         fpField->SetText( fpId.Format() );
 872                     }
 873                 }
 874             }
 875 
 876             // Only perform the dangling end test on root sheet.

 877             GetScreen()->TestDanglingEnds();
 878 
 879             ClearUndoRedoList();
 880 
 881             m_toolManager->RunAction( ACTIONS::zoomFitScreen, true );
 882             SetSheetNumberAndCount();
 883             SyncView();
 884             UpdateTitle();
 885         }
 886         catch( const IO_ERROR& ioe )
 887         {
 888             // Do not leave g_RootSheet == NULL because it is expected to be

 889             // a valid sheet. Therefore create a dummy empty root sheet and screen.

 890             CreateScreens();
 891             m_toolManager->RunAction( ACTIONS::zoomFitScreen, true );
 892 
 893             wxString msg;
 894             msg.Printf( _( "Error loading schematic \"%s\".\n%s" ), aFileName, ioe.What() );
 895             DisplayError( this, msg );
 896 
 897             msg.Printf( _( "Failed to load \"%s\"" ), aFileName );
 898             AppendMsgPanel( wxEmptyString, msg, CYAN );
 899 
 900             return false;
 901         }
 902 
 903         return true;
 904 
 905     default:
 906         return false;
 907     }
 908 }
