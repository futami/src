  62 void PARAM_CFG_FIELDNAMES::ReadParam( wxConfigBase* aConfig ) const
  63 {
  64     if( !m_Pt_param || !aConfig )
  65         return;
  66 
  67     wxString templateFieldNames = aConfig->Read( FieldNameTemplatesKey, wxEmptyString );
  68 
  69     if( !templateFieldNames.IsEmpty() )
  70     {
  71         TEMPLATE_FIELDNAMES_LEXER  lexer( TO_UTF8( templateFieldNames ) );
  72 
  73         try
  74         {
  75             m_Pt_param->Parse( &lexer, false );
  76         }
  77         catch( const IO_ERROR& DBG( e ) )
  78         {
  79             // @todo show error msg

  80             DBG( printf( "templatefieldnames parsing error: '%s'\n", TO_UTF8( e.What() ) ); )
  81         }
  82     }
  83 }

  85 void PARAM_CFG_FIELDNAMES::SaveParam( wxConfigBase* aConfig ) const
  86 {
  87     if( !m_Pt_param || !aConfig )
  88         return;
  89 
  90     STRING_FORMATTER sf;
  91     m_Pt_param->Format( &sf, 0, false );

  94     record.Replace( wxT("\n"), wxT(""), true );   // strip all newlines

  95     record.Replace( wxT("  "), wxT(" "), true );  // double space to single

  96 
  97     aConfig->Write( FieldNameTemplatesKey, record );

 101 class PARAM_CFG_SEVERITIES : public PARAM_CFG
 102 {
 103 protected:
 104     ERC_SETTINGS* m_Pt_param;   ///< Pointer to the parameter value

 105 
 106 public:
 107     PARAM_CFG_SEVERITIES( ERC_SETTINGS* ptparam, const wxChar* group = nullptr ) :
 108                           PARAM_CFG( wxEmptyString, PARAM_SEVERITIES, group )
 109     {
 110         m_Pt_param = ptparam;

 113     void ReadParam( wxConfigBase* aConfig ) const override
 114     {
 115         if( !m_Pt_param || !aConfig )
 116             return;
 117 
 118         wxString oldPath = aConfig->GetPath();
 119 
 120         // Read legacy settings first so that modern settings will overwrite them

 121         bool flag;
 122 
 123         if( aConfig->Read( wxT( "ERC_TestSimilarLabels" ), &flag, true ) )
 124         {
 125             if( flag )
 126                 m_Pt_param->m_Severities[ ERCE_SIMILAR_LABELS ] = RPT_SEVERITY_WARNING;
 127             else
 128                 m_Pt_param->m_Severities[ ERCE_SIMILAR_LABELS ] = RPT_SEVERITY_IGNORE;
 129         }
 130 
 131         if( aConfig->Read( wxT( "ERC_CheckUniqueGlobalLabels" ), &flag, true ) )
 132         {
 133             if( flag )
 134                 m_Pt_param->m_Severities[ ERCE_GLOBLABEL ] = RPT_SEVERITY_WARNING;
 135             else
 136                 m_Pt_param->m_Severities[ ERCE_GLOBLABEL ] = RPT_SEVERITY_IGNORE;
 137         }
 138 
 139         if( aConfig->Read( wxT( "ERC_CheckBusDriverConflicts" ), &flag, true ) )
 140         {
 141             if( flag )
 142                 m_Pt_param->m_Severities[ ERCE_DRIVER_CONFLICT ] = RPT_SEVERITY_WARNING;
 143             else
 144                 m_Pt_param->m_Severities[ ERCE_DRIVER_CONFLICT ] = RPT_SEVERITY_IGNORE;
 145         }
 146 
 147         if( aConfig->Read( wxT( "ERC_CheckBusEntryConflicts" ), &flag, true ) )
 148         {
 149             if( flag )
 150                 m_Pt_param->m_Severities[ ERCE_BUS_ENTRY_CONFLICT ] = RPT_SEVERITY_WARNING;
 151             else
 152                 m_Pt_param->m_Severities[ ERCE_BUS_ENTRY_CONFLICT ] = RPT_SEVERITY_IGNORE;
 153         }
 154 
 155         if( aConfig->Read( wxT( "ERC_CheckBusToBusConflicts" ), &flag, true ) )
 156         {
 157             if( flag )
 158                 m_Pt_param->m_Severities[ ERCE_BUS_TO_BUS_CONFLICT ] = RPT_SEVERITY_ERROR;
 159             else
 160                 m_Pt_param->m_Severities[ ERCE_BUS_TO_BUS_CONFLICT ] = RPT_SEVERITY_IGNORE;
 161         }
 162 
 163         if( aConfig->Read( wxT( "ERC_CheckBusToNetConflicts" ), &flag, true ) )
 164         {
 165             if( flag )
 166                 m_Pt_param->m_Severities[ ERCE_BUS_TO_NET_CONFLICT ] = RPT_SEVERITY_ERROR;
 167             else
 168                 m_Pt_param->m_Severities[ ERCE_BUS_TO_NET_CONFLICT ] = RPT_SEVERITY_IGNORE;
 169         }
 170 
 171         // TO DO: figure out what we're going to use as keys here so we can read/write these....

 172 
 173         aConfig->SetPath( oldPath );
 174     }
 175 
 176     void SaveParam( wxConfigBase* aConfig ) const override
 177     {
 178         if( !m_Pt_param || !aConfig )
 179             return;
 180 
 181         wxString oldPath = aConfig->GetPath();
 182 
 183         // TO DO: figure out what we're going to use as keys here so we can read/write these....

 184 
 185         // TO DO: for now just write out the legacy ones so we don't lose them

 186         // TO DO: remove this once the new scheme is in place

 187         aConfig->Write( wxT( "ERC_TestSimilarLabels" ),
 188                         m_Pt_param->IsTestEnabled( ERCE_SIMILAR_LABELS ) );
 189         aConfig->Write( wxT( "ERC_CheckUniqueGlobalLabels" ),
 190                         m_Pt_param->IsTestEnabled( ERCE_GLOBLABEL ) );
 191         aConfig->Write( wxT( "ERC_CheckBusDriverConflicts" ),
 192                         m_Pt_param->IsTestEnabled( ERCE_DRIVER_CONFLICT ) );
 193         aConfig->Write( wxT( "ERC_CheckBusEntryConflicts" ),
 194                         m_Pt_param->IsTestEnabled( ERCE_BUS_ENTRY_CONFLICT ) );

 196                         m_Pt_param->IsTestEnabled( ERCE_BUS_TO_BUS_CONFLICT ) );
 197         aConfig->Write( wxT( "ERC_CheckBusToNetConflicts" ),
 198                         m_Pt_param->IsTestEnabled( ERCE_BUS_TO_NET_CONFLICT ) );
 199 
 200         aConfig->SetPath( oldPath );
 201     }

 235 bool SCH_EDIT_FRAME::LoadProjectSettings()
 236 {
 237     GetRenderSettings()->SetDefaultPenWidth( m_defaults->m_DefaultLineWidth );
 238     GetRenderSettings()->m_DefaultWireThickness = m_defaults->m_DefaultWireThickness;
 239     GetRenderSettings()->m_DefaultBusThickness  = m_defaults->m_DefaultBusThickness;
 240     GetRenderSettings()->m_TextOffsetRatio      = m_defaults->m_TextOffsetRatio;
 241     GetRenderSettings()->m_PinSymbolSize        = m_defaults->m_PinSymbolSize;
 242     GetRenderSettings()->m_JunctionSize         = m_defaults->m_JunctionSize;
 243 
 244     // Verify some values, because the config file can be edited by hand,

 245     // and have bad values:

 246     LIB_PART::SetSubpartIdNotation( LIB_PART::GetSubpartIdSeparator(),
 247                                     LIB_PART::GetSubpartFirstId() );
 248 
 249     // Load the page layout decr file, from the filename stored in

 250     // BASE_SCREEN::m_PageLayoutDescrFileName, read in config project file

 251     // If empty, or not existing, the default descr is loaded

 252     WS_DATA_MODEL& pglayout = WS_DATA_MODEL::GetTheInstance();
 253     wxString filename = WS_DATA_MODEL::MakeFullFileName( BASE_SCREEN::m_PageLayoutDescrFileName,
 254                                                          Prj().GetProjectPath() );
 255 
 256     pglayout.SetPageLayout( filename );
 257 
 258     return true;
 259 }

 326 void SCH_BASE_FRAME::LoadSettings( APP_SETTINGS_BASE* aCfg )
 327 {
 328     wxCHECK_RET( aCfg, "Call to SCH_BASE_FRAME::SaveSettings with null settings" );
 329 
 330     EDA_DRAW_FRAME::LoadSettings( aCfg );
 331 
 332     if( aCfg->m_Window.grid.sizes.empty() )
 333     {
 334         /*

 335          * Do NOT add others values (mainly grid values in mm), because they can break the

 336          * schematic: Because wires and pins are considered as connected when the are to the

 337          * same coordinate we cannot mix coordinates in mils (internal units) and mm (that

 338          * cannot exactly converted in mils in many cases).  In fact schematic must only use

 339          * 50 and 25 mils to place labels, wires and components others values are useful only

 340          * for graphic items (mainly in library editor) so use integer values in mils only.

 341          * The 100 mil grid is added to help conform to the KiCad Library Convention which

 342          * states: "Using a 100mil grid, pin ends and origin must lie on grid nodes IEC-60617"

 343          */
 344         aCfg->m_Window.grid.sizes = { "100 mil",
 345                                       "50 mil",
 346                                       "25 mil",
 347                                       "10 mil",
 348                                       "5 mil",
 349                                       "2 mil",
 350                                       "1 mil" };
 351     }
 352 
 353     if( aCfg->m_Window.zoom_factors.empty() )
 354     {
 355         aCfg->m_Window.zoom_factors = { 0.1,
 356                                         0.15,
 357                                         0.2,
 358                                         0.3,
 359                                         0.5,
 360                                         0.7,
 361                                         1.0,
 362                                         1.5,
 363                                         2.0,
 364                                         3.0,
 365                                         4.5,
 366                                         6.5,
 367                                         10.0,
 368                                         15.0,
 369                                         20.0,
 370                                         30.0,
 371                                         45.0,
 372                                         65.0,
 373                                         100.0,
 374                                         150.0 };
 375     }
 376 
 377     for( double& factor : aCfg->m_Window.zoom_factors )
 378         factor = std::min( factor, MAX_ZOOM_FACTOR );
 379 
 380     EESCHEMA_SETTINGS* cfg = dynamic_cast<EESCHEMA_SETTINGS*>( aCfg );
 381 
 382     if( cfg )
 383     {
 384         wxString templateFieldNames = cfg->m_Drawing.field_names;
 385 
 386         if( !templateFieldNames.IsEmpty() )
 387         {
 388             TEMPLATE_FIELDNAMES_LEXER  lexer( TO_UTF8( templateFieldNames ) );
 389 
 390             try
 391             {
 392                 m_templateFieldNames.Parse( &lexer, true );
 393             }
 394             catch( const IO_ERROR& DBG( e ) )
 395             {
 396                 // @todo show error msg

 397                 DBG( printf( "templatefieldnames parsing error: '%s'\n", TO_UTF8( e.What() ) ); )
 398             }
 399         }
 400     }
 401 }

 439 SYMBOL_LIB_TABLE* PROJECT::SchSymbolLibTable()
 440 {
 441     // This is a lazy loading function, it loads the project specific table when

 442     // that table is asked for, not before.

 443     SYMBOL_LIB_TABLE* tbl = (SYMBOL_LIB_TABLE*) GetElem( ELEM_SYMBOL_LIB_TABLE );
 444 
 445     // its gotta be NULL or a SYMBOL_LIB_TABLE, or a bug.

 446     wxASSERT( !tbl || tbl->Type() == SYMBOL_LIB_TABLE_T );
 447 
 448     if( !tbl )
 449     {
 450         // Stack the project specific SYMBOL_LIB_TABLE overlay on top of the global table.

 451         // ~SYMBOL_LIB_TABLE() will not touch the fallback table, so multiple projects may

 452         // stack this way, all using the same global fallback table.

 453         tbl = new SYMBOL_LIB_TABLE( &SYMBOL_LIB_TABLE::GetGlobalLibTable() );
 454 
 455         SetElem( ELEM_SYMBOL_LIB_TABLE, tbl );
 456 
 457         wxString prjPath;
 458 
 459         wxGetEnv( PROJECT_VAR_NAME, &prjPath );
 460 
 461         if( !prjPath.IsEmpty() )
 462         {
 463             wxFileName fn( prjPath, SYMBOL_LIB_TABLE::GetSymbolLibTableFileName() );
 464 
 465             try
 466             {
 467                 tbl->Load( fn.GetFullPath() );
 468             }
 469             catch( const IO_ERROR& ioe )
 470             {
 471                 wxString msg;
 472                 msg.Printf( _( "An error occurred loading the symbol library table \"%s\"." ),
 473                             fn.GetFullPath() );
 474                 DisplayErrorMessage( NULL, msg, ioe.What() );
 475             }
 476         }
 477     }
 478 
 479     return tbl;
 480 }
