 313     /**

 314      * Get the display symbol orientation.

 315      *

 316      * Because there are different ways to have a given orientation/mirror,

 317      * the orientation/mirror is not necessary what the user does.  For example:

 318      * a mirrorX then a mirrorY returns no mirror but a rotate.  This function finds

 319      * a rotation and a mirror value #CMP_MIRROR_X because this is the first mirror

 320      * option tested.  This can differs from the orientation made by an user.  A

 321      * #CMP_MIRROR_Y is returned as a #CMP_MIRROR_X with an orientation 180 because

 322      * they are equivalent.

 323      *

 324      * @sa COMPONENT_ORIENTATION_T

 325      *

 326      * @return the orientation and mirror of the symbol.

 327      */

 567     // Geometric transforms (used in block operations):

 569     void Move( const wxPoint& aMoveVector ) override
 570     {
 571         if( aMoveVector == wxPoint( 0, 0 ) )
 572             return;
 573 
 574         m_Pos += aMoveVector;
 575 
 576         for( SCH_FIELD& field : m_Fields )
 577             field.Move( aMoveVector );
 578 
 579         SetModified();
 580     }
