  51 class ERC_TESTER
  52 {
  53 public:
  54 
  55     ERC_TESTER( SCHEMATIC* aSchematic ) :
  56             m_schematic( aSchematic )
  57     {
  58     }
  59 
  60     /**

  61      * Perform ERC testing for electrical conflicts between \a NetItemRef and other items

  62      * (mainly pin) on the same net.

  63      * @param aList = a reference to the list of connected objects

  64      * @param aNetItemRef = index in list of the current object

  65      * @param aNetStart = index in list of net objects of the first item

  66      * @param aMinConnexion = a pointer to a variable to store the minimal connection

  67      * found( NOD, DRV, NPI, NET_NC)

  68      */
  69     void TestOthersItems( NETLIST_OBJECT_LIST* aList, unsigned aNetItemRef, unsigned aNetStart,
  70                           int* aMinConnexion );
  71 
  72     /**

  73      * inside a given sheet, one cannot have sheets with duplicate names (file

  74      * names can be duplicated).

  75      * @return the error count

  76      * @param aCreateMarker: true = create error markers in schematic,

  77      *                       false = calculate error count only

  78      */
  79     int TestDuplicateSheetNames( bool aCreateMarker );
  80 
  81     /**

  82      * Checks for any unresolved text variable references.

  83      */
  84     void TestTextVars( KIGFX::WS_PROXY_VIEW_ITEM* aWorksheet );
  85 
  86     /**

  87      * Checks that there are not conflicting bus alias definitions in the schematic

  88      *

  89      * (for example, two hierarchical sub-sheets contain different definitions for

  90      * the same bus alias)

  91      *

  92      * @return the error count

  93      */
  94     int TestConflictingBusAliases();
  95 
  96     /**

  97      * Test if all units of each multiunit component have the same footprint assigned.

  98      * @return The error count.

  99      */
 100     int TestMultiunitFootprints();
 101 
 102     /**

 103      * In KiCad 5 and earlier, you could connect stuff up to pins with NC electrical type.

 104      * In KiCad 6, this no longer results in those pins joining the net, so we need to warn about it

 105      * @return the error count

 106      */
 107     int TestNoConnectPins();
 108 
 109 private:
 110     /**

 111      * Performs ERC testing and creates an ERC marker to show the ERC problem for aNetItemRef

 112      * or between aNetItemRef and aNetItemTst.

 113      *  if MinConn < 0: this is an error on labels

 114      */
 115     void diagnose( NETLIST_OBJECT* NetItemRef, NETLIST_OBJECT* NetItemTst, int MinConnexion,
 116                    PIN_ERROR Diag );
 117 
 118     SCHEMATIC* m_schematic;
 119 };
