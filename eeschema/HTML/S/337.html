  28 class BUS_ALIAS;
  29 class CONNECTION_GRAPH;
  30 class EDA_BASE_FRAME;
  31 class ERC_SETTINGS;
  32 class PROJECT;
  33 class SCH_SCREEN;
  34 class SCH_SHEET;

  36 /**

  37  * Holds all the data relating to one schematic

  38  * A schematic may consist of one or more sheets (and one root sheet)

  39  * Right now, eeschema can have only one schematic open at a time, but this could change.

  40  * Please keep this possibility in mind when adding to this object.

  41  */
  42 class SCHEMATIC : public EDA_ITEM
  43 {
  44     friend class SCH_EDIT_FRAME;
  45 
  46 private:
  47     PROJECT* m_project;
  48 
  49     /// The top-level sheet in this schematic hierarchy (or potentially the only one)

  50     SCH_SHEET* m_rootSheet;
  51 
  52     /**

  53      * The sheet path of the sheet currently being edited or displayed.

  54      * Note that this was moved here from SCH_EDIT_FRAME because currently many places in the code

  55      * want to know the current sheet.  Potentially this can be moved back to the UI code once

  56      * the only places that want to know it are UI-related

  57      */
  58     SCH_SHEET_PATH* m_currentSheet;
  59 
  60     /// Holds and calculates connectivity information of this schematic

  61     CONNECTION_GRAPH* m_connectionGraph;
  62 
  63 public:
  64     SCHEMATIC( PROJECT* aPrj );
  65 
  66     ~SCHEMATIC();
  67 
  68     virtual wxString GetClass() const override
  69     {
  70         return wxT( "SCHEMATIC" );
  71     }
  72 
  73     /// Initializes this schematic to a blank one, unloading anything existing

  74     void Reset();
  75 
  76     /// Return a reference to the project this schematic is part of

  77     PROJECT& Prj() const
  78     {
  79         return *const_cast<PROJECT*>( m_project );
  80     }
  81 
  82     void SetProject( PROJECT* aPrj );
  83 
  84     /// Sets up the template fieldnames link if this project is opened in eeschema

  85     void SetTemplateFieldNames( TEMPLATES* aTemplates );
  86 
  87     /**

  88      * Builds and returns an updated schematic hierarchy

  89      * TODO: can this be cached?

  90      * @return a SCH_SHEET_LIST containing the schematic hierarchy

  91      */
  92     SCH_SHEET_LIST GetSheets() const
  93     {
  94         return SCH_SHEET_LIST( m_rootSheet );
  95     }
  96 
  97     SCH_SHEET& Root() const
  98     {
  99         return *m_rootSheet;
 100     }
 101 
 102     /**

 103      * Initializes the schematic with a new root sheet.

 104      * This is typically done by calling a file loader that returns the new root sheet

 105      * As a side-effect, takes care of some post-load initialization.

 106      * @param aRootSheet is the new root sheet for this schematic.

 107      */
 108     void SetRoot( SCH_SHEET* aRootSheet );
 109 
 110     /// A simple test if the schematic is loaded, not a complete one

 111     bool IsValid() const
 112     {
 113         return m_rootSheet != nullptr;
 114     }
 115 
 116     /// Helper to retreive the screen of the root sheet

 117     SCH_SCREEN* RootScreen() const;
 118 
 119     /// Helper to retrieve the filename from the root sheet screen

 120     wxString GetFileName() const;
 121 
 122     SCH_SHEET_PATH& CurrentSheet() const
 123     {
 124         return *m_currentSheet;
 125     }
 126 
 127     void SetCurrentSheet( const SCH_SHEET_PATH& aPath )
 128     {
 129         *m_currentSheet = aPath;
 130     }
 131 
 132     CONNECTION_GRAPH* ConnectionGraph() const
 133     {
 134         return m_connectionGraph;
 135     }
 136 
 137     SCHEMATIC_SETTINGS& Settings() const;
 138 
 139     ERC_SETTINGS& ErcSettings() const;
 140 
 141     /**

 142      * Returns a pointer to a bus alias object for the given label, or null if one

 143      * doesn't exist.

 144      */
 145     std::shared_ptr<BUS_ALIAS> GetBusAlias( const wxString& aLabel ) const;
 146 
 147     /**

 148      * Returns a list of name candidates for netclass assignment.  The list will include both

 149      * composite names (buses) and atomic net names.  Names are fetched from available labels,

 150      * power pins, etc.

 151      */
 152     std::vector<wxString> GetNetClassAssignmentCandidates();
 153 
 154 #if defined(DEBUG)
 155     void Show( int nestLevel, std::ostream& os ) const override {}
 156 #endif
 157 };
 158 
 159 #endif
