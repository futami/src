  76 int SCH_DRAWING_TOOLS::PlaceComponent(  const TOOL_EVENT& aEvent  )
  77 {
  78     SCH_COMPONENT*                    component = aEvent.Parameter<SCH_COMPONENT*>();
  79     SCHLIB_FILTER                     filter;
  80     std::vector<COMPONENT_SELECTION>* historyList = nullptr;
  81 
  82     if( aEvent.IsAction( &EE_ACTIONS::placeSymbol ) )
  83         historyList = &m_symbolHistoryList;
  84     else if (aEvent.IsAction( &EE_ACTIONS::placePower ) )
  85     {
  86         historyList = &m_powerHistoryList;
  87         filter.FilterPowerParts( true );
  88     }
  89     else
  90         wxFAIL_MSG( "PlaceCompontent(): unexpected request" );
  91 
  92     getViewControls()->ShowCursor( true );
  93 
  94     // If a component was passed in get it ready for placement.

  95     if( component )
  96     {
  97         component->SetFlags( IS_NEW | IS_MOVED );
  98 
  99         m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 100         m_selectionTool->AddItemToSel( component );
 101     }
 102 
 103     std::string tool = aEvent.GetCommandStr().get();
 104     m_frame->PushTool( tool );
 105     Activate();
 106 
 107     // Prime the pump

 108     if( component )
 109     {
 110         getViewControls()->WarpCursor( getViewControls()->GetMousePosition( false ) );
 111         m_toolMgr->RunAction( ACTIONS::refreshPreview );
 112     }
 113     else if( aEvent.HasPosition() )
 114         m_toolMgr->RunAction( EE_ACTIONS::cursorClick );
 115 
 116     // Main loop: keep receiving events

 117     while( TOOL_EVENT* evt = Wait() )
 118     {
 119         m_frame->GetCanvas()->SetCurrentCursor( component ? wxCURSOR_ARROW : wxCURSOR_PENCIL );
 120         VECTOR2I cursorPos = getViewControls()->GetCursorPosition( !evt->Modifier( MD_ALT ) );
 121 
 122         auto cleanup = [&] () {
 123             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 124             m_view->ClearPreview();
 125             delete component;
 126             component = nullptr;
 127         };
 128 
 129         if( evt->IsCancelInteractive() )
 130         {
 131             if( component )
 132                 cleanup();
 133             else
 134             {
 135                 m_frame->PopTool( tool );
 136                 break;
 137             }
 138         }
 139         else if( evt->IsActivate() )
 140         {
 141             if( component )
 142                 cleanup();
 143 
 144             if( evt->IsMoveTool() )
 145             {
 146                 // leave ourselves on the stack so we come back after the move

 147                 break;
 148             }
 149             else
 150             {
 151                 m_frame->PopTool( tool );
 152                 break;
 153             }
 154         }
 155         else if( evt->IsClick( BUT_LEFT ) )
 156         {
 157             if( !component )
 158             {
 159                 m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 160 
 161                 // Pick the module to be placed

 162                 bool footprintPreviews = m_frame->eeconfig()->m_Appearance.footprint_preview;
 163                 COMPONENT_SELECTION sel = m_frame->SelectCompFromLibTree( &filter, *historyList,
 164                                                                           true, 1, 1,
 165                                                                           footprintPreviews );
 166 
 167                 // Restore cursor after dialog

 168                 getViewControls()->WarpCursor( getViewControls()->GetCursorPosition(), true );
 169 
 170                 LIB_PART* part = sel.LibId.IsValid() ? m_frame->GetLibPart( sel.LibId ) : nullptr;
 171 
 172                 if( !part )
 173                     continue;
 174 
 175                 component = new SCH_COMPONENT(
 176                         *part, &m_frame->GetCurrentSheet(), sel, (wxPoint) cursorPos );
 177                 component->SetParent( m_frame->GetCurrentSheet().LastScreen() );
 178                 component->SetFlags( IS_NEW | IS_MOVED );
 179 
 180                 if( m_frame->eeconfig()->m_AutoplaceFields.enable )
 181                     component->AutoplaceFields( /* aScreen */ NULL, /* aManual */ false );
 182 
 183                 m_frame->SaveCopyForRepeatItem( component );
 184 
 185                 m_view->ClearPreview();
 186                 m_view->AddToPreview( component->Clone() );
 187                 m_selectionTool->AddItemToSel( component );
 188             }
 189             else
 190             {
 191                 SCH_COMPONENT* next_comp = nullptr;
 192 
 193                 m_view->ClearPreview();
 194                 m_frame->AddItemToScreenAndUndoList( m_frame->GetScreen(), component, false );
 195 
 196                 EE_SELECTION new_sel;
 197                 new_sel.Add( component );
 198 
 199                 m_toolMgr->RunAction( EE_ACTIONS::addNeededJunctions, true, &new_sel );
 200                 m_frame->OnModify();
 201 
 202                 if( m_frame->eeconfig()->m_SymChooserPanel.place_all_units
 203                         || m_frame->eeconfig()->m_SymChooserPanel.keep_symbol )
 204                 {
 205                     int new_unit = component->GetUnit();
 206 
 207                     if( m_frame->eeconfig()->m_SymChooserPanel.place_all_units
 208                             && component->GetUnit() < component->GetUnitCount() )
 209                         new_unit++;
 210                     else
 211                         new_unit = 1;
 212 
 213                     // We are either stepping to the next unit or next component

 214                     if( m_frame->eeconfig()->m_SymChooserPanel.keep_symbol || new_unit > 1 )
 215                     {
 216                         // Deselect the last placed symbol: obviously we do not want to

 217                         // apply some changes (like rotation, mirror...) to previously placed

 218                         // symbols.

 219                         m_selectionTool->ClearSelection();
 220 
 221                         next_comp = static_cast<SCH_COMPONENT*>( component->Duplicate() );
 222                         next_comp->SetFlags( IS_NEW | IS_MOVED );
 223                         next_comp->SetUnit( new_unit );
 224                         next_comp->SetUnitSelection( &m_frame->GetCurrentSheet(), new_unit );
 225 
 226                         if( m_frame->eeconfig()->m_AutoplaceFields.enable )
 227                             component->AutoplaceFields( /* aScreen */ NULL, /* aManual */ false );
 228 
 229                         m_frame->SaveCopyForRepeatItem( next_comp );
 230                         m_view->AddToPreview( next_comp->Clone() );
 231                         m_selectionTool->AddItemToSel( next_comp );
 232                     }
 233                 }
 234 
 235                 component = next_comp;
 236             }
 237         }
 238         else if( evt->IsClick( BUT_RIGHT ) )
 239         {
 240             // Warp after context menu only if dragging...

 241             if( !component )
 242                 m_toolMgr->VetoContextMenuMouseWarp();
 243 
 244             m_menu.ShowContextMenu( m_selectionTool->GetSelection() );
 245         }
 246         else if( evt->Category() == TC_COMMAND && evt->Action() == TA_CHOICE_MENU_CHOICE )
 247         {
 248             if( evt->GetCommandId().get() >= ID_POPUP_SCH_SELECT_UNIT_CMP
 249                 && evt->GetCommandId().get() <= ID_POPUP_SCH_SELECT_UNIT_CMP_MAX )
 250             {
 251                 int unit = evt->GetCommandId().get() - ID_POPUP_SCH_SELECT_UNIT_CMP;
 252 
 253                 if( component )
 254                 {
 255                     m_frame->SelectUnit( component, unit );
 256                     m_toolMgr->RunAction( ACTIONS::refreshPreview );
 257                 }
 258             }
 259         }
 260         else if( component && ( evt->IsAction( &ACTIONS::refreshPreview ) || evt->IsMotion() ) )
 261         {
 262             component->SetPosition( (wxPoint)cursorPos );
 263             m_view->ClearPreview();
 264             m_view->AddToPreview( component->Clone() );
 265         }
 266         else
 267             evt->SetPassEvent();
 268 
 269         // Enable autopanning and cursor capture only when there is a module to be placed

 270         getViewControls()->SetAutoPan( component != nullptr );
 271         getViewControls()->CaptureCursor( component != nullptr );
 272     }
 273 
 274     return 0;
 275 }

 278 int SCH_DRAWING_TOOLS::PlaceImage( const TOOL_EVENT& aEvent )
 279 {
 280     SCH_BITMAP* image = aEvent.Parameter<SCH_BITMAP*>();
 281     bool        immediateMode = image;
 282     VECTOR2I    cursorPos = getViewControls()->GetCursorPosition();
 283 
 284     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 285     getViewControls()->ShowCursor( true );
 286 
 287     // Add all the drawable parts to preview

 288     if( image )
 289     {
 290         image->SetPosition( (wxPoint)cursorPos );
 291         m_view->ClearPreview();
 292         m_view->AddToPreview( image->Clone() );
 293     }
 294 
 295     std::string tool = aEvent.GetCommandStr().get();
 296     m_frame->PushTool( tool );
 297     Activate();
 298 
 299     // Prime the pump

 300     if( image )
 301         m_toolMgr->RunAction( ACTIONS::refreshPreview );
 302     else if( aEvent.HasPosition() )
 303         m_toolMgr->RunAction( ACTIONS::cursorClick );
 304 
 305     // Main loop: keep receiving events

 306     while( TOOL_EVENT* evt = Wait() )
 307     {
 308         m_frame->GetCanvas()->SetCurrentCursor( image ? wxCURSOR_ARROW : wxCURSOR_PENCIL );
 309         cursorPos = getViewControls()->GetCursorPosition( !evt->Modifier( MD_ALT ) );
 310 
 311         auto cleanup = [&] () {
 312             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 313             m_view->ClearPreview();
 314             delete image;
 315             image = nullptr;
 316         };
 317 
 318         if( evt->IsCancelInteractive() )
 319         {
 320             if( image )
 321                 cleanup();
 322             else
 323             {
 324                 m_frame->PopTool( tool );
 325                 break;
 326             }
 327 
 328             if( immediateMode )
 329             {
 330                 m_frame->PopTool( tool );
 331                 break;
 332             }
 333         }
 334         else if( evt->IsActivate() )
 335         {
 336             if( image )
 337                 cleanup();
 338 
 339             if( evt->IsMoveTool() )
 340             {
 341                 // leave ourselves on the stack so we come back after the move

 342                 break;
 343             }
 344             else
 345             {
 346                 m_frame->PopTool( tool );
 347                 break;
 348             }
 349         }
 350         else if( evt->IsClick( BUT_LEFT ) )
 351         {
 352             if( !image )
 353             {
 354                 m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 355                 wxFileDialog dlg( m_frame, _( "Choose Image" ), wxEmptyString, wxEmptyString,
 356                                   _( "Image Files " ) + wxImage::GetImageExtWildcard(), wxFD_OPEN );
 357 
 358                 if( dlg.ShowModal() != wxID_OK )
 359                     continue;
 360 
 361                 // Restore cursor after dialog

 362                 getViewControls()->WarpCursor( getViewControls()->GetCursorPosition(), true );
 363 
 364                 wxString fullFilename = dlg.GetPath();
 365 
 366                 if( wxFileExists( fullFilename ) )
 367                     image = new SCH_BITMAP( (wxPoint)cursorPos );
 368 
 369                 if( !image || !image->ReadImageFile( fullFilename ) )
 370                 {
 371                     wxMessageBox( _( "Couldn't load image from \"%s\"" ), fullFilename );
 372                     delete image;
 373                     image = nullptr;
 374                     continue;
 375                 }
 376 
 377                 image->SetFlags( IS_NEW | IS_MOVED );
 378 
 379                 m_frame->SaveCopyForRepeatItem( image );
 380 
 381                 m_view->ClearPreview();
 382                 m_view->AddToPreview( image->Clone() );
 383                 m_view->RecacheAllItems();  // Bitmaps are cached in Opengl

 384 
 385                 m_selectionTool->AddItemToSel( image );
 386 
 387                 getViewControls()->SetCursorPosition( cursorPos, false );
 388             }
 389             else
 390             {
 391                 m_frame->AddItemToScreenAndUndoList( m_frame->GetScreen(), image, false );
 392                 image = nullptr;
 393                 m_toolMgr->RunAction( ACTIONS::activatePointEditor );
 394 
 395                 m_view->ClearPreview();
 396 
 397                 if( immediateMode )
 398                 {
 399                     m_frame->PopTool( tool );
 400                     break;
 401                 }
 402             }
 403         }
 404         else if( evt->IsClick( BUT_RIGHT ) )
 405         {
 406             // Warp after context menu only if dragging...

 407             if( !image )
 408                 m_toolMgr->VetoContextMenuMouseWarp();
 409 
 410             m_menu.ShowContextMenu( m_selectionTool->GetSelection() );
 411         }
 412         else if( image && ( evt->IsAction( &ACTIONS::refreshPreview ) || evt->IsMotion() ) )
 413         {
 414             image->SetPosition( (wxPoint)cursorPos );
 415             m_view->ClearPreview();
 416             m_view->AddToPreview( image->Clone() );
 417             m_view->RecacheAllItems();  // Bitmaps are cached in Opengl

 418         }
 419         else
 420             evt->SetPassEvent();
 421 
 422         // Enable autopanning and cursor capture only when there is a module to be placed

 423         getViewControls()->SetAutoPan( image != nullptr );
 424         getViewControls()->CaptureCursor( image != nullptr );
 425     }
 426 
 427     return 0;
 428 }

 431 int SCH_DRAWING_TOOLS::SingleClickPlace( const TOOL_EVENT& aEvent )
 432 {
 433     wxPoint   cursorPos;
 434     KICAD_T   type = aEvent.Parameter<KICAD_T>();
 435 
 436     if( type == SCH_JUNCTION_T && aEvent.HasPosition() )
 437     {
 438         EE_SELECTION& selection = m_selectionTool->GetSelection();
 439         SCH_LINE*     wire = dynamic_cast<SCH_LINE*>( selection.Front() );
 440 
 441         if( wire )
 442         {
 443             SEG seg( wire->GetStartPoint(), wire->GetEndPoint() );
 444             VECTOR2I nearest = seg.NearestPoint( getViewControls()->GetCursorPosition() );
 445             getViewControls()->SetCrossHairCursorPosition( nearest, false );
 446             getViewControls()->WarpCursor( getViewControls()->GetCursorPosition(), true );
 447         }
 448     }
 449 
 450     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 451     getViewControls()->ShowCursor( true );
 452     getViewControls()->SetSnapping( true );
 453 
 454     SCH_ITEM* previewItem;
 455     switch( type )
 456     {
 457     case SCH_NO_CONNECT_T:
 458         previewItem = new SCH_NO_CONNECT( cursorPos );
 459         break;
 460     case SCH_JUNCTION_T:
 461         previewItem = new SCH_JUNCTION( cursorPos );
 462         break;
 463     case SCH_BUS_WIRE_ENTRY_T:
 464         previewItem = new SCH_BUS_WIRE_ENTRY( cursorPos );
 465         break;
 466     default:
 467         wxASSERT_MSG( false, "Unknown item type in SCH_DRAWING_TOOLS::SingleClickPlace" );
 468         return 0;
 469     }
 470 
 471     previewItem->SetParent( m_frame->GetScreen() );
 472 
 473     m_view->ClearPreview();
 474     m_view->AddToPreview( previewItem->Clone() );
 475 
 476     std::string tool = aEvent.GetCommandStr().get();
 477     m_frame->PushTool( tool );
 478     Activate();
 479 
 480     // Prime the pump

 481     if( aEvent.HasPosition() )
 482         m_toolMgr->RunAction( ACTIONS::cursorClick );
 483 
 484     // Main loop: keep receiving events

 485     while( TOOL_EVENT* evt = Wait() )
 486     {
 487         m_frame->GetCanvas()->SetCurrentCursor( wxCURSOR_ARROW );
 488         cursorPos = (wxPoint) getViewControls()->GetCursorPosition( !evt->Modifier( MD_ALT ) );
 489 
 490         if( evt->IsCancelInteractive() )
 491         {
 492             m_frame->PopTool( tool );
 493             break;
 494         }
 495         else if( evt->IsActivate() )
 496         {
 497             if( evt->IsMoveTool() )
 498             {
 499                 // leave ourselves on the stack so we come back after the move

 500                 break;
 501             }
 502             else
 503             {
 504                 m_frame->PopTool( tool );
 505                 break;
 506             }
 507         }
 508         else if( evt->IsClick( BUT_LEFT ) )
 509         {
 510             if( !m_frame->GetScreen()->GetItem( cursorPos, 0, type ) )
 511             {
 512                 if( type == SCH_JUNCTION_T )
 513                     m_frame->AddJunction( m_frame->GetScreen(), cursorPos, false );
 514                 else
 515                 {
 516                     SCH_ITEM* newItem = static_cast<SCH_ITEM*>( previewItem->Clone() );
 517                     newItem->SetPosition( cursorPos );
 518                     newItem->SetFlags( IS_NEW );
 519 
 520                     m_frame->AddItemToScreenAndUndoList( m_frame->GetScreen(), newItem, false );
 521                     m_frame->SaveCopyForRepeatItem( newItem );
 522 
 523                     m_frame->SchematicCleanUp();
 524                     m_frame->OnModify();
 525                 }
 526             }
 527         }
 528         else if( evt->IsClick( BUT_RIGHT ) )
 529         {
 530             m_menu.ShowContextMenu( m_selectionTool->GetSelection() );
 531         }
 532         else if( evt->IsAction( &ACTIONS::refreshPreview ) || evt->IsMotion() )
 533         {
 534             previewItem->SetPosition( (wxPoint)cursorPos );
 535             m_view->ClearPreview();
 536             m_view->AddToPreview( previewItem->Clone() );
 537         }
 538         else if( evt->Category() == TC_COMMAND )
 539         {
 540             if( ( type == SCH_BUS_WIRE_ENTRY_T )
 541                     && (   evt->IsAction( &EE_ACTIONS::rotateCW )
 542                         || evt->IsAction( &EE_ACTIONS::rotateCCW )
 543                         || evt->IsAction( &EE_ACTIONS::mirrorX )
 544                         || evt->IsAction( &EE_ACTIONS::mirrorY ) ) )
 545             {
 546                 SCH_BUS_ENTRY_BASE* busItem = static_cast<SCH_BUS_ENTRY_BASE*>( previewItem );
 547 
 548                 // The bus entries only rotate in one direction

 549                 if( evt->IsAction( &EE_ACTIONS::rotateCW )
 550                         || evt->IsAction( &EE_ACTIONS::rotateCCW ) )
 551                     busItem->Rotate( busItem->GetPosition() );
 552                 else if( evt->IsAction( &EE_ACTIONS::mirrorX ) )
 553                     busItem->MirrorX( busItem->GetPosition().x );
 554                 else if( evt->IsAction( &EE_ACTIONS::mirrorY ) )
 555                     busItem->MirrorY( busItem->GetPosition().y );
 556 
 557                 m_view->ClearPreview();
 558                 m_view->AddToPreview( previewItem->Clone() );
 559             }
 560         }
 561         else
 562             evt->SetPassEvent();
 563     }
 564 
 565     delete previewItem;
 566     m_view->ClearPreview();
 567 
 568     return 0;
 569 }

 586 SCH_TEXT* SCH_DRAWING_TOOLS::createNewText( const VECTOR2I& aPosition, int aType )
 587 {
 588     SCHEMATIC*          schematic = getModel<SCHEMATIC>();
 589     SCHEMATIC_SETTINGS& settings = schematic->Settings();
 590     SCH_TEXT*           textItem = nullptr;
 591 
 592     m_queuedTexts.clear();
 593 
 594     switch( aType )
 595     {
 596     case LAYER_NOTES:
 597         textItem = new SCH_TEXT( (wxPoint) aPosition );
 598         break;
 599 
 600     case LAYER_LOCLABEL:
 601         textItem = new SCH_LABEL( (wxPoint) aPosition );
 602         break;
 603 
 604     case LAYER_HIERLABEL:
 605         textItem = new SCH_HIERLABEL( (wxPoint) aPosition );
 606         textItem->SetShape( m_lastGlobalLabelShape );
 607         break;
 608 
 609     case LAYER_GLOBLABEL:
 610         textItem = new SCH_GLOBALLABEL( (wxPoint) aPosition );
 611         textItem->SetShape( m_lastGlobalLabelShape );
 612         break;
 613 
 614     default:
 615         DisplayError( m_frame, wxT( "SCH_EDIT_FRAME::CreateNewText() Internal error" ) );
 616         return nullptr;
 617     }
 618 
 619     textItem->SetParent( schematic );
 620     textItem->SetBold( m_lastTextBold );
 621     textItem->SetItalic( m_lastTextItalic );
 622     textItem->SetLabelSpinStyle( m_lastTextOrientation );
 623     textItem->SetTextSize( wxSize( settings.m_DefaultTextSize, settings.m_DefaultTextSize ) );
 624     textItem->SetFlags( IS_NEW | IS_MOVED );
 625 
 626     DIALOG_LABEL_EDITOR dlg( m_frame, textItem );
 627 
 628     if( dlg.ShowModal() != wxID_OK || textItem->GetText().IsEmpty() )
 629     {
 630         delete textItem;
 631         return nullptr;
 632     }
 633 
 634     if( aType != LAYER_NOTES )
 635     {
 636         UTF8 text( textItem->GetText() );
 637         int  brace_count = 0;
 638         int  bracket_count = 0;
 639         bool last_space = false;
 640         UTF8 token;
 641 
 642         for( auto chIt = text.ubegin(); chIt != text.uend(); chIt++ )
 643         {
 644             switch( *chIt )
 645             {
 646             case '{':
 647                 brace_count++;
 648                 last_space = false;
 649                 break;
 650 
 651             case '[':
 652                 bracket_count++;
 653                 last_space = false;
 654                 break;
 655 
 656             case '}':
 657                 brace_count = std::max( 0, brace_count - 1 );
 658                 last_space = false;
 659                 break;
 660 
 661             case ']':
 662                 bracket_count = std::max( 0, bracket_count - 1 );
 663                 last_space = false;
 664                 break;
 665 
 666             case ' ':
 667             case '\n':
 668             case '\r':
 669             case '\t':
 670                 if( !token.empty() && bracket_count == 0 && brace_count == 0 )
 671                 {
 672                     std::unique_ptr<SCH_TEXT> nextitem( static_cast<SCH_TEXT*>( textItem->Clone() ) );
 673                     nextitem->SetText( token.wx_str() );
 674                     m_queuedTexts.push_back( std::move( nextitem ) );
 675                     token.clear();
 676                     continue;
 677                 }
 678 
 679                 // Skip leading whitespace

 680                 if( token.empty() || last_space )
 681                     continue;
 682 
 683                 last_space = true;
 684                 break;
 685 
 686             default:
 687                 last_space = false;
 688                 break;
 689             }
 690 
 691             token += *chIt;
 692         }
 693 
 694         if( !token.empty() )
 695         {
 696             std::unique_ptr<SCH_TEXT> nextitem( static_cast<SCH_TEXT*>( textItem->Clone() ) );
 697             nextitem->SetText( token.wx_str() );
 698             m_queuedTexts.push_back( std::move( nextitem ) );
 699         }
 700 
 701         delete textItem;
 702         textItem = getNextNewText();
 703 
 704         if( !textItem )
 705             return nullptr;
 706     }
 707 
 708     m_lastTextBold = textItem->IsBold();
 709     m_lastTextItalic = textItem->IsItalic();
 710     m_lastTextOrientation = textItem->GetLabelSpinStyle();
 711 
 712     if( textItem->Type() == SCH_GLOBAL_LABEL_T || textItem->Type() == SCH_HIER_LABEL_T )
 713         m_lastGlobalLabelShape = textItem->GetShape();
 714 
 715     return textItem;
 716 }

 719 int SCH_DRAWING_TOOLS::TwoClickPlace( const TOOL_EVENT& aEvent )
 720 {
 721     EDA_ITEM* item = nullptr;
 722     bool      importMode = aEvent.IsAction( &EE_ACTIONS::importSheetPin );
 723     KICAD_T   type = aEvent.Parameter<KICAD_T>();
 724 
 725     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 726     getViewControls()->ShowCursor( true );
 727 
 728     std::string tool = aEvent.GetCommandStr().get();
 729     m_frame->PushTool( tool );
 730     Activate();
 731 
 732     // Prime the pump

 733     if( aEvent.HasPosition() )
 734         m_toolMgr->RunAction( ACTIONS::cursorClick );
 735 
 736     // Main loop: keep receiving events

 737     while( TOOL_EVENT* evt = Wait() )
 738     {
 739         m_frame->GetCanvas()->SetCurrentCursor( item ? wxCURSOR_ARROW : wxCURSOR_PENCIL );
 740         VECTOR2I cursorPos = getViewControls()->GetCursorPosition( !evt->Modifier( MD_ALT ) );
 741 
 742         auto cleanup = [&] () {
 743             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 744             m_view->ClearPreview();
 745             delete item;
 746             item = nullptr;
 747         };
 748 
 749         if( evt->IsCancelInteractive() )
 750         {
 751             if( item )
 752                 cleanup();
 753             else
 754             {
 755                 m_frame->PopTool( tool );
 756                 break;
 757             }
 758         }
 759         else if( evt->IsActivate() )
 760         {
 761             if( item )
 762                 cleanup();
 763 
 764             if( evt->IsPointEditor() )
 765             {
 766                 // don't exit (the point editor runs in the background)

 767             }
 768             else if( evt->IsMoveTool() )
 769             {
 770                 // leave ourselves on the stack so we come back after the move

 771                 break;
 772             }
 773             else
 774             {
 775                 m_frame->PopTool( tool );
 776                 break;
 777             }
 778         }
 779         else if( evt->IsClick( BUT_LEFT ) )
 780         {
 781             // First click creates...

 782             if( !item )
 783             {
 784                 m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 785 
 786                 switch( type )
 787                 {
 788                 case SCH_LABEL_T:
 789                     item = createNewText( cursorPos, LAYER_LOCLABEL );
 790                     break;
 791                 case SCH_HIER_LABEL_T:
 792                     item = createNewText( cursorPos, LAYER_HIERLABEL );
 793                     break;
 794                 case SCH_GLOBAL_LABEL_T:
 795                     item = createNewText( cursorPos, LAYER_GLOBLABEL );
 796                     break;
 797                 case SCH_TEXT_T:
 798                     item = createNewText( cursorPos, LAYER_NOTES );
 799                     break;
 800                 case SCH_SHEET_PIN_T:
 801                 {
 802                     SCH_HIERLABEL* label = nullptr;
 803                     SCH_SHEET*     sheet = nullptr;
 804 
 805                     if( m_selectionTool->SelectPoint( cursorPos, EE_COLLECTOR::SheetsOnly, &item ) )
 806                         sheet = dynamic_cast<SCH_SHEET*>( item );
 807 
 808                     item = nullptr;
 809 
 810                     if( !sheet )
 811                     {
 812                         m_statusPopup.reset( new STATUS_TEXT_POPUP( m_frame ) );
 813                         m_statusPopup->SetText( _( "Click over a sheet." ) );
 814                         m_statusPopup->Move( wxGetMousePosition() + wxPoint( 20, 20 ) );
 815                         m_statusPopup->PopupFor( 2000 );
 816                         break;
 817                     }
 818 
 819                     if( importMode )
 820                     {
 821                         label = m_frame->ImportHierLabel( sheet );
 822 
 823                         if( !label )
 824                         {
 825                             m_statusPopup.reset( new STATUS_TEXT_POPUP( m_frame ) );
 826                             m_statusPopup->SetText( _( "No new hierarchical labels found." ) );
 827                             m_statusPopup->Move( wxGetMousePosition() + wxPoint( 20, 20 ) );
 828                             m_statusPopup->PopupFor( 2000 );
 829                             break;
 830                         }
 831                     }
 832 
 833                     item = m_frame->CreateSheetPin( sheet, label );
 834                     break;
 835                 }
 836                 default:
 837                     break;
 838                 }
 839 
 840                 // Restore cursor after dialog

 841                 getViewControls()->WarpCursor( getViewControls()->GetCursorPosition(), true );
 842 
 843                 if( item )
 844                 {
 845                     item->SetFlags( IS_NEW | IS_MOVED );
 846                     m_view->ClearPreview();
 847                     m_view->AddToPreview( item->Clone() );
 848                     m_selectionTool->AddItemToSel( item );
 849                 }
 850 
 851                 getViewControls()->SetCursorPosition( cursorPos, false );
 852             }
 853 
 854             // ... and second click places:

 855             else
 856             {
 857                 item->ClearFlags( IS_MOVED );
 858                 m_frame->AddItemToScreenAndUndoList( m_frame->GetScreen(), (SCH_ITEM*) item, false );
 859                 item = getNextNewText();
 860 
 861                 if( item )
 862                 {
 863                     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 864                     item->SetFlags( IS_NEW | IS_MOVED );
 865                     m_view->ClearPreview();
 866                     m_view->AddToPreview( item->Clone() );
 867                     m_selectionTool->AddItemToSel( item );
 868                 }
 869                 else
 870                 {
 871                     m_view->ClearPreview();
 872                 }
 873             }
 874         }
 875         else if( evt->IsClick( BUT_RIGHT ) )
 876         {
 877             // Warp after context menu only if dragging...

 878             if( !item )
 879                 m_toolMgr->VetoContextMenuMouseWarp();
 880 
 881             m_menu.ShowContextMenu( m_selectionTool->GetSelection() );
 882         }
 883         else if( item && evt->IsSelectionEvent() )
 884         {
 885             // This happens if our text was replaced out from under us by ConvertTextType()

 886             EE_SELECTION& selection = m_selectionTool->GetSelection();
 887 
 888             if( selection.GetSize() == 1 )
 889             {
 890                 item = (SCH_ITEM*) selection.Front();
 891                 m_view->ClearPreview();
 892                 m_view->AddToPreview( item->Clone() );
 893             }
 894             else
 895                 item = nullptr;
 896         }
 897         else if( item && ( evt->IsAction( &ACTIONS::refreshPreview ) || evt->IsMotion() ) )
 898         {
 899             static_cast<SCH_ITEM*>( item )->SetPosition( (wxPoint) cursorPos );
 900             m_view->ClearPreview();
 901             m_view->AddToPreview( item->Clone() );
 902         }
 903         else
 904             evt->SetPassEvent();
 905 
 906         // Enable autopanning and cursor capture only when there is a module to be placed

 907         getViewControls()->SetAutoPan( item != nullptr );
 908         getViewControls()->CaptureCursor( item != nullptr );
 909     }
 910 
 911     return 0;
 912 }

 915 int SCH_DRAWING_TOOLS::DrawSheet( const TOOL_EVENT& aEvent )
 916 {
 917     EE_POINT_EDITOR* pointEditor = m_toolMgr->GetTool<EE_POINT_EDITOR>();
 918     SCH_SHEET*       sheet = nullptr;
 919

 924     m_frame->PushTool( tool );

 929         m_toolMgr->RunAction( ACTIONS::cursorClick );
 930 
 931     // Main loop: keep receiving events

 932     while( TOOL_EVENT* evt = Wait() )
 933     {
 934         if( !pointEditor->HasPoint() )
 935             m_frame->GetCanvas()->SetCurrentCursor( wxCURSOR_PENCIL );
 936 
 937         VECTOR2I cursorPos = getViewControls()->GetCursorPosition( !evt->Modifier( MD_ALT ) );
 938 
 939         auto cleanup = [&] () {
 940             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 941             m_view->ClearPreview();
 942             delete sheet;
 943             sheet = nullptr;
 944         };
 945 
 946         if( evt->IsCancelInteractive() )
 947         {
 948             if( sheet )
 949                 cleanup();
 950             else
 951             {
 952                 m_frame->PopTool( tool );
 953                 break;
 954             }
 955         }
 956         else if( evt->IsActivate() )
 957         {
 958             if( sheet )
 959                 cleanup();
 960 
 961             if( evt->IsPointEditor() )
 962             {
 963                 // don't exit (the point editor runs in the background)

 966             {

 970             else
 971             {
 972                 m_frame->PopTool( tool );
 973                 break;
 974             }
 975         }
 976

 980 
 981             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 982

1013             }
1014             else
1015             {
1016                 delete sheet;
1017             }
1018 
1019             sheet = nullptr;
1020         }
1021 
1022         else if( sheet && ( evt->IsAction( &ACTIONS::refreshPreview ) || evt->IsMotion() ) )
1023         {
1024             sizeSheet( sheet, cursorPos );
1025             m_view->ClearPreview();
1026             m_view->AddToPreview( sheet->Clone() );
1027         }
1028 
1029         else if( evt->IsClick( BUT_RIGHT ) )
1030         {
1031             // Warp after context menu only if dragging...

1032             if( !sheet )
1033                 m_toolMgr->VetoContextMenuMouseWarp();
1034 
1035             m_menu.ShowContextMenu( m_selectionTool->GetSelection() );
1036         }
1037         else
1038             evt->SetPassEvent();
1039 
1040         // Enable autopanning and cursor capture only when there is a sheet to be placed

1041         getViewControls()->SetAutoPan( sheet != nullptr );
1042         getViewControls()->CaptureCursor( sheet != nullptr );
1043     }
1044 
1045     return 0;

1049 void SCH_DRAWING_TOOLS::sizeSheet( SCH_SHEET* aSheet, VECTOR2I aPos )
