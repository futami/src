  43 void SCH_EDIT_FRAME::ReCreateMenuBar()
  44 {
  45     EE_SELECTION_TOOL* selTool = m_toolManager->GetTool<EE_SELECTION_TOOL>();
  46     // wxWidgets handles the Mac Application menu behind the scenes, but that means

  47     // we always have to start from scratch with a new wxMenuBar.

  48     wxMenuBar*  oldMenuBar = GetMenuBar();
  49     WX_MENUBAR* menuBar    = new WX_MENUBAR();
  50 
  51     auto modifiedDocumentCondition = [&]( const SELECTION& sel )
  52                                      {
  53                                          return Schematic().GetSheets().IsModified();
  54                                      };
  55 
  56     //-- File menu -----------------------------------------------------------

  57     //

  58     CONDITIONAL_MENU*   fileMenu = new CONDITIONAL_MENU( false, selTool );
  59     static ACTION_MENU* openRecentMenu;
  60 
  61     if( Kiface().IsSingle() )   // When not under a project mgr

  62     {
  63         FILE_HISTORY& fileHistory = GetFileHistory();
  64 
  65         // Add this menu to the list of menus managed by the file history

  66         // (the file history will be updated when adding/removing files in history)

  67         if( !openRecentMenu )
  68         {
  69             openRecentMenu = new ACTION_MENU( false );
  70             openRecentMenu->SetTool( selTool );
  71             openRecentMenu->SetTitle( _( "Open Recent" ) );
  72             openRecentMenu->SetIcon( recent_xpm );
  73 
  74             fileHistory.UseMenu( openRecentMenu );
  75             fileHistory.AddFilesToMenu( openRecentMenu );
  76         }
  77 
  78         fileMenu->AddItem( ACTIONS::doNew,         EE_CONDITIONS::ShowAlways );
  79         fileMenu->AddItem( ACTIONS::open,          EE_CONDITIONS::ShowAlways );
  80         fileMenu->AddMenu( openRecentMenu,         FILE_HISTORY::FileHistoryNotEmpty( fileHistory ) );
  81         fileMenu->AddSeparator();
  82     }
  83 
  84     fileMenu->AddItem( ACTIONS::save,              modifiedDocumentCondition );
  85     fileMenu->AddItem( ACTIONS::saveAs,            EE_CONDITIONS::ShowAlways );
  86 
  87     fileMenu->AddSeparator();
  88 
  89     fileMenu->AddItem( ID_APPEND_PROJECT, _( "Append Schematic Sheet Content..." ),
  90                        _( "Append schematic sheet content from another project to the current sheet" ),
  91                        add_document_xpm,           EE_CONDITIONS::ShowAlways );
  92 
  93     fileMenu->AddSeparator();
  94 
  95     // Import submenu

  96     ACTION_MENU* submenuImport = new ACTION_MENU( false );
  97     submenuImport->SetTool( selTool );
  98     submenuImport->SetTitle( _( "Import" ) );
  99     submenuImport->SetIcon( import_xpm );
 100     submenuImport->Add( _( "Import Non KiCad Schematic..." ),
 101                         _( "Replace current schematic sheet with one imported from another application" ),
 102                         ID_IMPORT_NON_KICAD_SCH, import_document_xpm );
 103 
 104     submenuImport->Add( EE_ACTIONS::importFPAssignments );
 105     fileMenu->AddMenu( submenuImport,              EE_CONDITIONS::ShowAlways );
 106 
 107 
 108     // Export submenu

 109     ACTION_MENU* submenuExport = new ACTION_MENU( false );
 110     submenuExport->SetTool( selTool );
 111     submenuExport->SetTitle( _( "Export" ) );
 112     submenuExport->SetIcon( export_xpm );
 113     submenuExport->Add( EE_ACTIONS::drawSheetOnClipboard );
 114     submenuExport->Add( EE_ACTIONS::exportNetlist );
 115     fileMenu->AddMenu( submenuExport,              EE_CONDITIONS::ShowAlways );
 116 
 117     fileMenu->AddSeparator();
 118     fileMenu->AddItem( EE_ACTIONS::schematicSetup, EE_CONDITIONS::ShowAlways );
 119 
 120     fileMenu->AddSeparator();
 121     fileMenu->AddItem( ACTIONS::pageSettings,      EE_CONDITIONS::ShowAlways );
 122     fileMenu->AddItem( ACTIONS::print,             EE_CONDITIONS::ShowAlways );
 123     fileMenu->AddItem( ACTIONS::plot,              EE_CONDITIONS::ShowAlways );
 124 
 125     fileMenu->AddSeparator();
 126     fileMenu->AddQuitOrClose( &Kiface(), _( "Eeschema" ) );
 127 
 128     fileMenu->Resolve();
 129 
 130     //-- Edit menu -----------------------------------------------------------

 131     //

 132     CONDITIONAL_MENU* editMenu = new CONDITIONAL_MENU( false, selTool );
 133 
 134     auto enableUndoCondition = [ this ] ( const SELECTION& sel ) {
 135         return GetUndoCommandCount() > 0;
 136     };
 137     auto enableRedoCondition = [ this ] ( const SELECTION& sel ) {
 138         return GetRedoCommandCount() > 0;
 139     };
 140 
 141     editMenu->AddItem( ACTIONS::undo,                       enableUndoCondition );
 142     editMenu->AddItem( ACTIONS::redo,                       enableRedoCondition );
 143 
 144     editMenu->AddSeparator();
 145     editMenu->AddItem( ACTIONS::cut,                        EE_CONDITIONS::NotEmpty );
 146     editMenu->AddItem( ACTIONS::copy,                       EE_CONDITIONS::NotEmpty );
 147     editMenu->AddItem( ACTIONS::paste,                      EE_CONDITIONS::Idle );
 148     editMenu->AddItem( ACTIONS::pasteSpecial,               EE_CONDITIONS::Idle );
 149     editMenu->AddItem( ACTIONS::doDelete,                   EE_CONDITIONS::NotEmpty );
 150     editMenu->AddItem( ACTIONS::duplicate,                  EE_CONDITIONS::NotEmpty );
 151 
 152     editMenu->AddSeparator();
 153     editMenu->AddItem( ACTIONS::find,                       EE_CONDITIONS::ShowAlways );
 154     editMenu->AddItem( ACTIONS::findAndReplace,             EE_CONDITIONS::ShowAlways );
 155 
 156     editMenu->AddSeparator();
 157     editMenu->AddItem( ACTIONS::deleteTool,                 EE_CONDITIONS::ShowAlways );
 158     editMenu->AddItem( EE_ACTIONS::editTextAndGraphics,     EE_CONDITIONS::ShowAlways );
 159     editMenu->AddItem( EE_ACTIONS::updateFieldsFromLibrary, EE_CONDITIONS::ShowAlways );
 160 
 161     editMenu->Resolve();
 162 
 163     //-- View menu -----------------------------------------------------------

 164     //

 165     CONDITIONAL_MENU* viewMenu = new CONDITIONAL_MENU( false, selTool );
 166 
 167     auto belowRootSheetCondition =
 168             [this]( const SELECTION& aSel )
 169             {
 170                 return GetCurrentSheet().Last() != &Schematic().Root();
 171             };
 172 
 173     auto gridShownCondition =
 174             [this]( const SELECTION& aSel )
 175             {
 176                 return IsGridVisible();
 177             };
 178 
 179     auto imperialUnitsCondition =
 180             [this]( const SELECTION& aSel )
 181             {
 182                 return GetUserUnits() == EDA_UNITS::INCHES;
 183             };
 184 
 185     auto metricUnitsCondition =
 186             [this]( const SELECTION& aSel )
 187             {
 188                 return GetUserUnits() == EDA_UNITS::MILLIMETRES;
 189             };
 190 
 191     auto fullCrosshairCondition =
 192             [this]( const SELECTION& aSel )
 193             {
 194                 return GetGalDisplayOptions().m_fullscreenCursor;
 195             };
 196     auto hiddenPinsCondition =
 197             [this]( const SELECTION& aSel )
 198             {
 199                 return GetShowAllPins();
 200             };
 201 
 202     viewMenu->AddItem( ACTIONS::showSymbolBrowser,        EE_CONDITIONS::ShowAlways );
 203     viewMenu->AddItem( EE_ACTIONS::navigateHierarchy,     EE_CONDITIONS::ShowAlways );
 204     viewMenu->AddItem( EE_ACTIONS::leaveSheet,            belowRootSheetCondition );
 205 
 206     viewMenu->AddSeparator();
 207     viewMenu->AddItem( ACTIONS::zoomInCenter,             EE_CONDITIONS::ShowAlways );
 208     viewMenu->AddItem( ACTIONS::zoomOutCenter,            EE_CONDITIONS::ShowAlways );
 209     viewMenu->AddItem( ACTIONS::zoomFitScreen,            EE_CONDITIONS::ShowAlways );
 210     viewMenu->AddItem( ACTIONS::zoomTool,                 EE_CONDITIONS::ShowAlways );
 211     viewMenu->AddItem( ACTIONS::zoomRedraw,               EE_CONDITIONS::ShowAlways );
 212 
 213     viewMenu->AddSeparator();
 214     viewMenu->AddCheckItem( ACTIONS::toggleGrid,          gridShownCondition );
 215     viewMenu->AddItem( ACTIONS::gridProperties,           EE_CONDITIONS::ShowAlways );
 216 
 217     // Units submenu

 218     CONDITIONAL_MENU* unitsSubMenu = new CONDITIONAL_MENU( false, selTool );
 219     unitsSubMenu->SetTitle( _( "&Units" ) );
 220     unitsSubMenu->SetIcon( unit_mm_xpm );
 221     unitsSubMenu->AddCheckItem( ACTIONS::imperialUnits,   imperialUnitsCondition );
 222     unitsSubMenu->AddCheckItem( ACTIONS::metricUnits,     metricUnitsCondition );
 223     viewMenu->AddMenu( unitsSubMenu );
 224 
 225     viewMenu->AddCheckItem( ACTIONS::toggleCursorStyle,   fullCrosshairCondition );
 226 
 227     viewMenu->AddSeparator();
 228     viewMenu->AddCheckItem( EE_ACTIONS::toggleHiddenPins, hiddenPinsCondition );
 229 
 230 #ifdef __APPLE__
 231     viewMenu->AddSeparator();
 232 #endif
 233 
 234     viewMenu->Resolve();
 235 
 236     //-- Place menu -----------------------------------------------------------

 237     //

 238     CONDITIONAL_MENU* placeMenu = new CONDITIONAL_MENU( false, selTool );
 239 
 240     placeMenu->AddItem( EE_ACTIONS::placeSymbol,            EE_CONDITIONS::ShowAlways );
 241     placeMenu->AddItem( EE_ACTIONS::placePower,             EE_CONDITIONS::ShowAlways );
 242     placeMenu->AddItem( EE_ACTIONS::drawWire,               EE_CONDITIONS::ShowAlways );
 243     placeMenu->AddItem( EE_ACTIONS::drawBus,                EE_CONDITIONS::ShowAlways );
 244     placeMenu->AddItem( EE_ACTIONS::placeBusWireEntry,      EE_CONDITIONS::ShowAlways );
 245     placeMenu->AddItem( EE_ACTIONS::placeNoConnect,         EE_CONDITIONS::ShowAlways );
 246     placeMenu->AddItem( EE_ACTIONS::placeJunction,          EE_CONDITIONS::ShowAlways );
 247     placeMenu->AddItem( EE_ACTIONS::placeLabel,             EE_CONDITIONS::ShowAlways );
 248     placeMenu->AddItem( EE_ACTIONS::placeGlobalLabel,       EE_CONDITIONS::ShowAlways );
 249 
 250     placeMenu->AddSeparator();
 251     placeMenu->AddItem( EE_ACTIONS::placeHierLabel, EE_CONDITIONS::ShowAlways );
 252     placeMenu->AddItem( EE_ACTIONS::drawSheet,              EE_CONDITIONS::ShowAlways );
 253     placeMenu->AddItem( EE_ACTIONS::importSheetPin,         EE_CONDITIONS::ShowAlways );
 254 
 255     placeMenu->AddSeparator();
 256     placeMenu->AddItem( EE_ACTIONS::drawLines,              EE_CONDITIONS::ShowAlways );
 257     placeMenu->AddItem( EE_ACTIONS::placeSchematicText,     EE_CONDITIONS::ShowAlways );
 258     placeMenu->AddItem( EE_ACTIONS::placeImage,             EE_CONDITIONS::ShowAlways );
 259 
 260     placeMenu->Resolve();
 261 
 262     //-- Inspect menu -----------------------------------------------

 263     //

 264     CONDITIONAL_MENU* inspectMenu = new CONDITIONAL_MENU( false, selTool );
 265 
 266     inspectMenu->AddItem( EE_ACTIONS::runERC,               EE_CONDITIONS::ShowAlways );
 267 #ifdef KICAD_SPICE
 268     inspectMenu->AddItem( EE_ACTIONS::runSimulation,        EE_CONDITIONS::ShowAlways );
 269 #endif
 270 
 271     inspectMenu->Resolve();
 272 
 273     //-- Tools menu -----------------------------------------------

 274     //

 275     CONDITIONAL_MENU* toolsMenu = new CONDITIONAL_MENU( false, selTool );
 276 
 277     auto remapSymbolsCondition =
 278             [&]( const SELECTION& aSel )
 279             {
 280                 SCH_SCREENS schematic( Schematic().Root() );
 281 
 282                 // The remapping can only be performed on legacy projects.

 283                 return schematic.HasNoFullyDefinedLibIds();
 284             };
 285 
 286     toolsMenu->AddItem( ACTIONS::updatePcbFromSchematic,    EE_CONDITIONS::ShowAlways );
 287     toolsMenu->AddItem( ACTIONS::updateSchematicFromPcb,    EE_CONDITIONS::ShowAlways );
 288     toolsMenu->AddItem( EE_ACTIONS::showPcbNew,             EE_CONDITIONS::ShowAlways );
 289 
 290     toolsMenu->AddSeparator();
 291     toolsMenu->AddItem( ACTIONS::showSymbolEditor,          EE_CONDITIONS::ShowAlways );
 292     toolsMenu->AddItem( EE_ACTIONS::rescueSymbols,          EE_CONDITIONS::ShowAlways );
 293     toolsMenu->AddItem( EE_ACTIONS::remapSymbols,           remapSymbolsCondition );
 294 
 295     toolsMenu->AddSeparator();
 296     toolsMenu->AddItem( EE_ACTIONS::editSymbolFields,       EE_CONDITIONS::ShowAlways );
 297     toolsMenu->AddItem( EE_ACTIONS::editSymbolLibraryLinks, EE_CONDITIONS::ShowAlways );
 298 
 299     toolsMenu->AddSeparator();
 300     toolsMenu->AddItem( EE_ACTIONS::annotate,               EE_CONDITIONS::ShowAlways );
 301     toolsMenu->AddItem( EE_ACTIONS::showBusManager,         EE_CONDITIONS::ShowAlways );
 302 
 303     toolsMenu->AddSeparator();
 304     toolsMenu->AddItem( EE_ACTIONS::assignFootprints,       EE_CONDITIONS::ShowAlways );
 305     toolsMenu->AddItem( EE_ACTIONS::generateBOM,            EE_CONDITIONS::ShowAlways );
 306 
 307     toolsMenu->Resolve();
 308 
 309     //-- Preferences menu -----------------------------------------------

 310     //

 311     CONDITIONAL_MENU* prefsMenu = new CONDITIONAL_MENU( false, selTool );
 312 
 313     auto acceleratedGraphicsCondition = [ this ] ( const SELECTION& aSel ) {
 314         return GetCanvas()->GetBackend() == EDA_DRAW_PANEL_GAL::GAL_TYPE_OPENGL;
 315     };
 316     auto standardGraphicsCondition = [ this ] ( const SELECTION& aSel ) {
 317         return GetCanvas()->GetBackend() == EDA_DRAW_PANEL_GAL::GAL_TYPE_CAIRO;
 318     };
 319 
 320     prefsMenu->AddItem( ACTIONS::configurePaths,           EE_CONDITIONS::ShowAlways );
 321     prefsMenu->AddItem( ACTIONS::showSymbolLibTable,       EE_CONDITIONS::ShowAlways );
 322     prefsMenu->AddItem( wxID_PREFERENCES,
 323                         _( "Preferences...\tCTRL+," ),
 324                         _( "Show preferences for all open tools" ),
 325                         preference_xpm,                    EE_CONDITIONS::ShowAlways );
 326 
 327     prefsMenu->AddSeparator();
 328     AddMenuLanguageList( prefsMenu, selTool );
 329 
 330     prefsMenu->AddSeparator();
 331     prefsMenu->AddCheckItem( ACTIONS::acceleratedGraphics, acceleratedGraphicsCondition );
 332     prefsMenu->AddCheckItem( ACTIONS::standardGraphics,    standardGraphicsCondition );
 333 
 334     prefsMenu->Resolve();
 335 
 336     //-- Menubar -------------------------------------------------------------

 337     //

 338     menuBar->Append( fileMenu, _( "&File" ) );
 339     menuBar->Append( editMenu, _( "&Edit" ) );
 340     menuBar->Append( viewMenu, _( "&View" ) );
 341     menuBar->Append( placeMenu, _( "&Place" ) );
 342     menuBar->Append( inspectMenu, _( "&Inspect" ) );
 343     menuBar->Append( toolsMenu, _( "&Tools" ) );
 344     menuBar->Append( prefsMenu, _( "P&references" ) );
 345     AddStandardHelpMenu( menuBar );
 346 
 347     SetMenuBar( menuBar );
 348     delete oldMenuBar;
 349 }
