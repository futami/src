 112 const EDA_RECT SCH_BUS_ENTRY_BASE::GetBoundingBox() const
 113 {
 114     EDA_RECT box;
 115 
 116     box.SetOrigin( m_pos );
 117     box.SetEnd( m_End() );
 118 
 119     box.Normalize();
 120     box.Inflate( ( GetPenWidth() / 2 ) + 1 );
 121 
 122     return box;
 123 }

 230 bool SCH_BUS_WIRE_ENTRY::UpdateDanglingState( std::vector<DANGLING_END_ITEM>& aItemList,
 231                                               const SCH_SHEET_PATH* aPath )
 232 {
 233     bool previousStateStart = m_isDanglingStart;
 234     bool previousStateEnd = m_isDanglingEnd;
 235 
 236     m_isDanglingStart = m_isDanglingEnd = true;
 237 
 238     // Wires and buses are stored in the list as a pair, start and end. This

 239     // variable holds the start position from one iteration so it can be used

 240     // when the end position is found.

 241     wxPoint seg_start;
 242 
 243     // Store the connection type and state for the start (0) and end (1)

 244     bool has_wire[2] = { false };
 245     bool has_bus[2] = { false };
 246 
 247     for( DANGLING_END_ITEM& each_item : aItemList )
 248     {
 249         if( each_item.GetItem() == this )
 250             continue;
 251 
 252         switch( each_item.GetType() )
 253         {
 254         case WIRE_START_END:
 255         case WIRE_END_END:
 256             if( m_pos == each_item.GetPosition() )
 257                 has_wire[0] = true;
 258             else if( m_End() == each_item.GetPosition() )
 259                 has_wire[1] = true;
 260 
 261             break;
 262 
 263         case BUS_START_END:
 264             seg_start = each_item.GetPosition();
 265             break;
 266 
 267         case BUS_END_END:
 268             if( IsPointOnSegment( seg_start, each_item.GetPosition(), m_pos ) )
 269                 has_bus[0] = true;
 270             else if( IsPointOnSegment( seg_start, each_item.GetPosition(), m_End() ) )
 271                 has_bus[1] = true;
 272 
 273             break;
 274 
 275         default:
 276             break;
 277         }
 278     }
 279 
 280     // A bus-wire entry is connected at both ends if it has a bus and a wire on its

 281     // ends.  Otherwise, we connect only one end (in the case of a wire-wire or bus-bus)

 282     if( ( has_wire[0] && has_bus[1] ) || ( has_wire[1] && has_bus[0] ) )
 283         m_isDanglingEnd = m_isDanglingStart = false;
 284     else if( has_wire[0] || has_bus[0] )
 285         m_isDanglingStart = false;
 286     else if( has_wire[1] || has_bus[1] )
 287         m_isDanglingEnd = false;
 288 
 289     return (previousStateStart != m_isDanglingStart) || (previousStateEnd != m_isDanglingEnd);
 290 }

 293 bool SCH_BUS_BUS_ENTRY::UpdateDanglingState( std::vector<DANGLING_END_ITEM>& aItemList,
 294                                              const SCH_SHEET_PATH* aPath )
 295 {
 296     bool previousStateStart = m_isDanglingStart;
 297     bool previousStateEnd = m_isDanglingEnd;
 298 
 299     m_isDanglingStart = m_isDanglingEnd = true;
 300 
 301     // Wires and buses are stored in the list as a pair, start and end. This

 302     // variable holds the start position from one iteration so it can be used

 303     // when the end position is found.

 304     wxPoint seg_start;
 305 
 306     for( DANGLING_END_ITEM& each_item : aItemList )
 307     {
 308         if( each_item.GetItem() == this )
 309             continue;
 310 
 311         switch( each_item.GetType() )
 312         {
 313         case BUS_START_END:
 314             seg_start = each_item.GetPosition();
 315             break;
 316         case BUS_END_END:
 317             if( IsPointOnSegment( seg_start, each_item.GetPosition(), m_pos ) )
 318                 m_isDanglingStart = false;
 319             if( IsPointOnSegment( seg_start, each_item.GetPosition(), m_End() ) )
 320                 m_isDanglingEnd = false;
 321             break;
 322         default:
 323             break;
 324         }
 325     }
 326 
 327     return (previousStateStart != m_isDanglingStart) || (previousStateEnd != m_isDanglingEnd);
 328 }

 368 bool SCH_BUS_ENTRY_BASE::HitTest( const wxPoint& aPosition, int aAccuracy ) const
 369 {
 370     // Insure minimum accuracy

 371     if( aAccuracy == 0 )
 372         aAccuracy = ( GetPenWidth() / 2 ) + 4;
 373 
 374     return TestSegmentHit( aPosition, m_pos, m_End(), aAccuracy );
 375 }

 378 bool SCH_BUS_ENTRY_BASE::HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy ) const
 379 {
 380     EDA_RECT rect = aRect;
 381 
 382     rect.Inflate( aAccuracy );
 383 
 384     if( aContained )
 385         return rect.Contains( GetBoundingBox() );
 386 
 387     return rect.Intersects( GetBoundingBox() );
 388 }

 391 void SCH_BUS_ENTRY_BASE::Plot( PLOTTER* aPlotter )
 392 {
 393     auto* settings = static_cast<KIGFX::SCH_RENDER_SETTINGS*>( aPlotter->RenderSettings() );
 394 
 395     COLOR4D color = ( GetStrokeColor() == COLOR4D::UNSPECIFIED ) ?
 396             settings->GetLayerColor( m_Layer ) : GetStrokeColor();
 397     int     penWidth = ( GetPenWidth() == 0 ) ? settings->GetDefaultPenWidth() : GetPenWidth();
 398 
 399     aPlotter->SetCurrentLineWidth( penWidth );
 400     aPlotter->SetColor( color );
 401     aPlotter->SetDash( GetStrokeStyle() );
 402     aPlotter->MoveTo( m_pos );
 403     aPlotter->FinishTo( m_End() );
 404 }

 407 void SCH_BUS_ENTRY_BASE::GetMsgPanelInfo( EDA_DRAW_FRAME* aFrame, MSG_PANEL_ITEMS& aList )
 408 {

 422     if( !frame )

 424 
 425     if( auto conn = Connection( frame->GetCurrentSheet() ) )
 426         conn->AppendInfoToMsgPanel( aList );

 430 bool SCH_BUS_ENTRY_BASE::operator <( const SCH_ITEM& aItem ) const
 431 {
 432     if( Type() != aItem.Type() )
 433         return Type() < aItem.Type();
 434 
 435     auto component = static_cast<const SCH_BUS_ENTRY_BASE*>( &aItem );
 436 
 437     if( GetLayer() != component->GetLayer() )
 438         return GetLayer() < component->GetLayer();
 439 
 440     if( GetPosition().x != component->GetPosition().x )
 441         return GetPosition().x < component->GetPosition().x;
 442 
 443     if( GetPosition().y != component->GetPosition().y )
 444         return GetPosition().y < component->GetPosition().y;
 445 
 446     if( m_End().x != component->m_End().x )
 447         return m_End().x < component->m_End().x;
 448 
 449     return m_End().y < component->m_End().y;
 450 }

 453 bool SCH_BUS_WIRE_ENTRY::ConnectionPropagatesTo( const EDA_ITEM* aItem ) const
 454 {
 455     // Don't generate connections between bus entries and buses, since there is

 456     // a connectivity change at that point (e.g. A[7..0] to A7)

 457     if( ( aItem->Type() == SCH_LINE_T ) &&
 458         ( static_cast<const SCH_LINE*>( aItem )->GetLayer() == LAYER_BUS ) )
 459     {
 460         return false;
 461     }
 462 
 463     // Don't generate connections between bus entries and bus labels that happen

 464     // to land at the same point on the bus wire as this bus entry

 465     if( ( aItem->Type() == SCH_LABEL_T ) &&
 466         SCH_CONNECTION::IsBusLabel( static_cast<const SCH_LABEL*>( aItem )->GetText() ) )
 467     {
 468         return false;
 469     }
 470 
 471     // Don't generate connections between two bus-wire entries

 472     if( aItem->Type() == SCH_BUS_WIRE_ENTRY_T )
 473         return false;
 474 
 475     return true;
 476 }
