  63     /// @copydoc TOOL_BASE::Init()

  72      * Function Main()

  74      * The main loop.

  75      */
  76     int Main( const TOOL_EVENT& aEvent );
  77 
  78     /**

  79      * Function GetSelection()

  80      *

  81      * Returns the set of currently selected items.

  82      */
  83     EE_SELECTION& GetSelection();
  84 
  85     /**

  86      * Function RequestSelection()

  87      *

  88      * Returns either an existing selection (filtered), or the selection at the current

  89      * cursor if the existing selection is empty.

  90      */
  91     EE_SELECTION& RequestSelection( const KICAD_T* aFilterList = EE_COLLECTOR::AllItems );
  92 
  93     /**

  94      * Function SelectPoint()

  95      * Selects one or all items pointed by the parameter aWhere.

  96      * If there is more than one item at that place,

  97      * there is a menu displayed that allows one to choose the item or all of them.

  98      *

  99      * @param aWhere is the place where the item should be selected.

 100      * @param aItem is set to the newly selected item if only one was selected, otherwise is unchanged.

 101      * @param aSelectionCancelledFlag allows the function to inform its caller that a selection

 102      * was cancelled (for instance, by clicking outside of the disambiguation menu).

 103      * @param aCheckLocked indicates if locked items should be excluded.

 104      */
 105     bool SelectPoint( const VECTOR2I& aWhere, const KICAD_T* aFilterList = EE_COLLECTOR::AllItems,
 106                       EDA_ITEM** aItem = nullptr, bool* aSelectionCancelledFlag = nullptr,
 107                       bool aCheckLocked = false, bool aAdd = false, bool aSubtract = false,
 108                       bool aExclusiveOr = false );
 109 
 110     int AddItemToSel( const TOOL_EVENT& aEvent );
 111     void AddItemToSel( EDA_ITEM* aItem, bool aQuietMode = false );
 112     int AddItemsToSel( const TOOL_EVENT& aEvent );
 113     void AddItemsToSel( EDA_ITEMS* aList, bool aQuietMode = false );
 114 
 115     int RemoveItemFromSel( const TOOL_EVENT& aEvent );
 116     void RemoveItemFromSel( EDA_ITEM* aItem, bool aQuietMode = false );
 117     int RemoveItemsFromSel( const TOOL_EVENT& aEvent );
 118     void RemoveItemsFromSel( EDA_ITEMS* aList, bool aQuietMode = false );
 119 
 120     void BrightenItem( EDA_ITEM* aItem );
 121     void UnbrightenItem( EDA_ITEM* aItem );
 122 
 123     void SelectHighlightItem( EDA_ITEM* aItem ) { highlight( aItem, SELECTED ); }
 124 
 125     ///> Find (but don't select) node under cursor

 129     int SelectNode( const TOOL_EVENT& aEvent );

 137     void ClearSelection();

 140      * Function Selectable()

 141      * Checks conditions for an item to be selected.

 142      * @return True if the item fulfills conditions to be selected.

 143      */
 144     bool Selectable( const EDA_ITEM* aItem, bool checkVisibilityOnly = false ) const;
 145 
 146     /**

 147      * Apply heuristics to try and determine a single object when multiple are found under the

 148      * cursor.

 149      */
 150     void GuessSelectionCandidates( EE_COLLECTOR& collector, const VECTOR2I& aPos );
 151 
 152     /**

 153      * Function SelectionMenu()

 154      * Shows a popup menu to trim the COLLECTOR passed as aEvent's parameter down to a single

 155      * item.

 156      *

 157      * NOTE: this routine DOES NOT modify the selection.

 158      */
 159     int SelectionMenu( const TOOL_EVENT& aEvent );
 160 
 161     /**

 162      * Rebuilds the selection from the EDA_ITEMs' selection flags.  Commonly called after

 163      * rolling back an undo state to make sure there aren't any stale pointers.

 164      */
 165     void RebuildSelection();
 166 
 167 private:
 168     /**

 169      * Function selectMultiple()

 170      * Handles drawing a selection box that allows one to select many items at

 171      * the same time.

 172      *

 173      * @return true if the function was cancelled (i.e. CancelEvent was received).

 174      */
 175     bool selectMultiple();
 176 
 177     /**

 178      * Allows the selection of a single item from a list via pop-up menu.  The items are

 179      * highlighted on the canvas when hovered in the menu.  The collector is trimmed to

 180      * the picked item.

 181      * @return true if an item was picked

 182      */
 183     bool doSelectionMenu( EE_COLLECTOR* aItems );
 184 
 185     /**

 186      * Function select()

 187      * Takes necessary action mark an item as selected.

 188      *

 189      * @param aItem is an item to be selected.

 190      */
 191     void select( EDA_ITEM* aItem );
 192 
 193     /**

 194      * Function unselect()

 195      * Takes necessary action mark an item as unselected.

 196      *

 197      * @param aItem is an item to be unselected.

 198      */
 199     void unselect( EDA_ITEM* aItem );
 200 
 201     /**

 202      * Function highlight()

 203      * Highlights the item visually.

 204      * @param aItem is an item to be be highlighted.

 205      * @param aHighlightMode should be either SELECTED or BRIGHTENED

 206      * @param aGroup is the group to add the item to in the BRIGHTENED mode.

 207      */
 208     void highlight( EDA_ITEM* aItem, int aHighlightMode, EE_SELECTION* aGroup = nullptr );
 209 
 210     /**

 211      * Function unhighlight()

 212      * Unhighlights the item visually.

 213      * @param aItem is an item to be be highlighted.

 214      * @param aHighlightMode should be either SELECTED or BRIGHTENED

 215      * @param aGroup is the group to remove the item from.

 216      */
 217     void unhighlight( EDA_ITEM* aItem, int aHighlightMode, EE_SELECTION* aGroup = nullptr );
 218 
 219     /**

 220      * Sets the reference point to the anchor of the top-left item.

 221      */
 222     void updateReferencePoint();

 224     /**

 225      * Function selectionContains()

 226      * @return True if the given point is contained in any of selected items' bounding box.

 227      */
 228     bool selectionContains( const VECTOR2I& aPoint ) const;

 238     bool m_subtractive;          // Items should be removed from selection

 239     bool m_exclusive_or;         // Items' selection state should be toggled

 240     bool m_multiple;             // Multiple selection mode is active

 241     bool m_skip_heuristics;      // Heuristics are not allowed when choosing item under cursor

 243     bool m_isLibEdit;            // True when libedit is the parent frame

 246     int  m_convert;              // Fixed DeMorgan filter (for symbol editor)

 247 };
