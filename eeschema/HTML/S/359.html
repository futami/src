  40 /**

  41  * SCH_FIELD

  42  * instances are attached to a component and provide a place for the component's value,

  43  * reference designator, footprint, and user definable name-value pairs of arbitrary purpose.

  44  *

  45  * <ul> <li>Field 0 is reserved for the component reference.</li>

  46  * <li>Field 1 is reserved for the component value.</li>

  47  * <li>Field 2 is reserved for the component footprint.</li>

  48  * <li>Field 3 is reserved for the component data sheet file.</li>

  49  * <li>Field 4 and higher are user defineable.</li></ul>

  50  */
  51 
  52 class SCH_FIELD : public SCH_ITEM, public EDA_TEXT
  53 {
  54     int      m_id;         ///< Field index, @see enum NumFieldType

  55 
  56     wxString m_name;
  57 
  58 public:
  59     SCH_FIELD( const wxPoint& aPos, int aFieldId, SCH_ITEM* aParent,
  60                const wxString& aName = wxEmptyString );
  61 
  62     // Do not create a copy constructor.  The one generated by the compiler is adequate.

  63 
  64     ~SCH_FIELD();
  65 
  66     static inline bool ClassOf( const EDA_ITEM* aItem )
  67     {
  68         return aItem && SCH_FIELD_T == aItem->Type();
  69     }
  70 
  71     wxString GetClass() const override
  72     {
  73         return wxT( "SCH_FIELD" );
  74     }
  75 
  76     bool IsType( const KICAD_T aScanTypes[] ) const override
  77     {
  78         if( SCH_ITEM::IsType( aScanTypes ) )
  79             return true;
  80 
  81         for( const KICAD_T* p = aScanTypes; *p != EOT; ++p )
  82         {
  83             if( *p == SCH_FIELD_LOCATE_REFERENCE_T && m_id == REFERENCE )
  84                 return true;
  85             else if ( *p == SCH_FIELD_LOCATE_VALUE_T && m_id == VALUE )
  86                 return true;
  87             else if ( *p == SCH_FIELD_LOCATE_FOOTPRINT_T && m_id == FOOTPRINT )
  88                 return true;
  89             else if ( *p == SCH_FIELD_LOCATE_DATASHEET_T && m_id == DATASHEET )
  90                 return true;
  91         }
  92 
  93         return false;
  94     }
