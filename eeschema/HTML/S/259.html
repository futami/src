 294 void LIB_EDIT_FRAME::RebuildSymbolUnitsList()
 295 {
 296     if( !m_unitSelectBox )
 297         return;
 298 
 299     if( m_unitSelectBox->GetCount() != 0 )
 300         m_unitSelectBox->Clear();
 301 
 302     if( !m_my_part || m_my_part->GetUnitCount() <= 1 )
 303     {
 304         m_unit = 1;
 305         m_unitSelectBox->Append( wxEmptyString );
 306     }
 307     else
 308     {
 309         for( int i = 0; i < m_my_part->GetUnitCount(); i++ )
 310         {
 311             wxString sub  = LIB_PART::SubReference( i+1, false );
 312             wxString unit = wxString::Format( _( "Unit %s" ), GetChars( sub ) );
 313             m_unitSelectBox->Append( unit );
 314         }
 315     }
 316 
 317     // Ensure the selected unit is compatible with the number of units of the current part:

 318     if( m_my_part && m_my_part->GetUnitCount() < m_unit )
 319         m_unit = 1;
 320 
 321     m_unitSelectBox->SetSelection(( m_unit > 0 ) ? m_unit - 1 : 0 );
 322 }

 368 void LIB_EDIT_FRAME::OnSelectUnit( wxCommandEvent& event )
 369 {
 370     int i = event.GetSelection();
 371 
 372     if( ( i == wxNOT_FOUND ) || ( ( i + 1 ) == m_unit ) )
 373         return;
 374 
 375     m_toolManager->RunAction( ACTIONS::cancelInteractive, true );
 376     m_toolManager->RunAction( EE_ACTIONS::clearSelection, true );
 377 
 378     m_unit = i + 1;
 379 
 380     m_toolManager->ResetTools( TOOL_BASE::MODEL_RELOAD );
 381     RebuildView();
 382 }

 419 void LIB_EDIT_FRAME::SetCurPart( LIB_PART* aPart )
 420 {
 421     m_toolManager->RunAction( EE_ACTIONS::clearSelection, true );
 422 
 423     delete m_my_part;
 424     m_my_part = aPart;
 425 
 426     // select the current component in the tree widget

 427     if( m_my_part )
 428         m_treePane->GetLibTree()->SelectLibId( m_my_part->GetLibId() );
 429 
 430     wxString partName = m_my_part ? m_my_part->GetName() : wxString();
 431     m_libMgr->SetCurrentPart( partName );
 432 
 433     // retain in case this wxFrame is re-opened later on the same PROJECT

 434     Prj().SetRString( PROJECT::SCH_LIBEDIT_CUR_PART, partName );

 437     m_SyncPinEdit = aPart && aPart->IsRoot() && aPart->IsMulti() && !aPart->UnitsLocked();
 438 
 439     m_toolManager->ResetTools( TOOL_BASE::MODEL_RELOAD );
 440     RebuildView();
 441 }

 491 bool LIB_EDIT_FRAME::AddLibraryFile( bool aCreateNew )
 492 {
 493     wxFileName fn = m_libMgr->GetUniqueLibraryName();
 494 
 495     if( !LibraryFileBrowser( !aCreateNew, fn, KiCadSymbolLibFileWildcard(),
 496                              KiCadSymbolLibFileExtension, false ) )
 497     {
 498         return false;
 499     }
 500 
 501     wxString libName = fn.GetName();
 502 
 503     if( libName.IsEmpty() )
 504         return false;
 505 
 506     if( m_libMgr->LibraryExists( libName ) )
 507     {
 508         DisplayError( this, wxString::Format( _( "Library \"%s\" already exists" ), libName ) );
 509         return false;
 510     }
 511 
 512     // Select the target library table (global/project)

 513     SYMBOL_LIB_TABLE* libTable = selectSymLibTable();
 514 
 515     if( !libTable )
 516         return false;
 517 
 518     if( aCreateNew )
 519     {
 520         if( !m_libMgr->CreateLibrary( fn.GetFullPath(), libTable ) )
 521         {
 522             DisplayError( this, wxString::Format( _( "Could not create the library file '%s'.\n"
 523                                                      "Check write permission." ),
 524                                                   fn.GetFullPath() ) );
 525             return false;
 526         }
 527     }
 528     else
 529     {
 530         if( !m_libMgr->AddLibrary( fn.GetFullPath(), libTable ) )
 531         {
 532             DisplayError( this, _( "Could not open the library file." ) );
 533             return false;
 534         }
 535     }
 536 
 537     bool globalTable = ( libTable == &SYMBOL_LIB_TABLE::GetGlobalLibTable() );
 538     saveSymbolLibTables( globalTable, !globalTable );
 539 
 540     return true;
 541 }

 587 void LIB_EDIT_FRAME::SyncLibraries( bool aShowProgress )
 588 {
 589     LIB_ID selected;
 590 
 591     if( m_treePane )
 592         selected = m_treePane->GetLibTree()->GetSelectedLibId();
 593 
 594     if( aShowProgress )
 595     {
 596         wxProgressDialog progressDlg( _( "Loading Symbol Libraries" ), wxEmptyString,
 597                                       m_libMgr->GetAdapter()->GetLibrariesCount(), this );
 598 
 599         m_libMgr->Sync( true, [&]( int progress, int max, const wxString& libName )
 600         {
 601             progressDlg.Update( progress, wxString::Format( _( "Loading library \"%s\"" ),
 602                                                             libName ) );
 603         } );
 604     }
 605     else
 606     {
 607         m_libMgr->Sync( true );
 608     }
 609 
 610     if( m_treePane )
 611     {
 612         wxDataViewItem found;
 613 
 614         if( selected.IsValid() )
 615         {
 616             // Check if the previously selected item is still valid,

 617             // if not - it has to be unselected to prevent crash

 618             found = m_libMgr->GetAdapter()->FindItem( selected );
 619 
 620             if( !found )
 621                 m_treePane->GetLibTree()->Unselect();
 622         }
 623 
 624         m_treePane->GetLibTree()->Regenerate( true );
 625 
 626         // Try to select the parent library, in case the part is not found

 627         if( !found && selected.IsValid() )
 628         {
 629             selected.SetLibItemName( "" );
 630             found = m_libMgr->GetAdapter()->FindItem( selected );
 631 
 632             if( found )
 633                 m_treePane->GetLibTree()->SelectLibId( selected );
 634         }
 635 
 636         // If no selection, see if there's a current part to centre

 637         if( !selected.IsValid() && m_my_part )
 638         {
 639             LIB_ID current( GetCurLib(), m_my_part->GetName() );
 640             m_treePane->GetLibTree()->CenterLibId( current );
 641         }
 642     }
 643 }

 684 bool LIB_EDIT_FRAME::backupFile( const wxFileName& aOriginalFile, const wxString& aBackupExt )
 685 {
 686     if( aOriginalFile.FileExists() )
 687     {
 688         wxFileName backupFileName( aOriginalFile );
 689         backupFileName.SetExt( aBackupExt );
 690 
 691         if( backupFileName.FileExists() )
 692             wxRemoveFile( backupFileName.GetFullPath() );
 693 
 694         if( !wxCopyFile( aOriginalFile.GetFullPath(), backupFileName.GetFullPath() ) )
 695         {
 696             DisplayError( this, wxString::Format( _( "Failed to save backup to \"%s\"" ),
 697                                                   backupFileName.GetFullPath() ) );
 698             return false;
 699         }
 700     }
 701 
 702     return true;
 703 }

 820 void LIB_EDIT_FRAME::KiwayMailIn( KIWAY_EXPRESS& mail )
 821 {
 822     const std::string& payload = mail.GetPayload();
 823 
 824     switch( mail.Command() )
 825     {
 826     case MAIL_LIB_EDIT:
 827         if( !payload.empty() )
 828         {
 829             wxString libFileName( payload );
 830             wxString libNickname;
 831             wxString msg;
 832 
 833             SYMBOL_LIB_TABLE*    libTable = Prj().SchSymbolLibTable();
 834             const LIB_TABLE_ROW* libTableRow = libTable->FindRowByURI( libFileName );
 835 
 836             if( !libTableRow )
 837             {
 838                 msg.Printf( _( "The current configuration does not include the symbol library\n"
 839                                "\"%s\".\nUse Manage Symbol Libraries to edit the configuration." ),
 840                             libFileName );
 841                 DisplayErrorMessage( this, _( "Library not found in symbol library table." ), msg );
 842                 break;
 843             }
 844 
 845             libNickname = libTableRow->GetNickName();
 846 
 847             if( !libTable->HasLibrary( libNickname, true ) )
 848             {
 849                 msg.Printf( _( "The library with the nickname \"%s\" is not enabled\n"
 850                                "in the current configuration.  Use Manage Symbol Libraries to\n"
 851                                "edit the configuration." ), libNickname );
 852                 DisplayErrorMessage( this, _( "Symbol library not enabled." ), msg );
 853                 break;
 854             }
 855 
 856             SetCurLib( libNickname );
 857 
 858             if( m_treePane )
 859             {
 860                 LIB_ID id( libNickname, wxEmptyString );
 861                 m_treePane->GetLibTree()->ExpandLibId( id );
 862                 m_treePane->GetLibTree()->CenterLibId( id );
 863             }
 864         }
 865 
 866         break;
 867 
 868     default:
 869         ;
 870     }
 871 }

 893 bool LIB_EDIT_FRAME::IsContentModified()
 894 {
 895     wxCHECK( m_libMgr, false );
 896 
 897     // Test if the currently edited part is modified

 898     if( GetScreen() && GetScreen()->IsModify() && GetCurPart() )
 899         return true;
 900 
 901     // Test if any library has been modified

 902     for( const auto& libNickname : m_libMgr->GetLibraryNames() )
 903     {
 904         if( m_libMgr->IsLibraryModified( libNickname )
 905           && !m_libMgr->IsLibraryReadOnly( libNickname ) )
 906             return true;
 907     }
 908 
 909     return false;
 910 }
