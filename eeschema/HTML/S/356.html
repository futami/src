  91 SEARCH_STACK* PROJECT::SchSearchS()
  92 {
  93     SEARCH_STACK* ss = (SEARCH_STACK*) GetElem( PROJECT::ELEM_SCH_SEARCH_STACK );
  94 
  95     wxASSERT( !ss || dynamic_cast<SEARCH_STACK*>( GetElem( PROJECT::ELEM_SCH_SEARCH_STACK ) ) );
  96 
  97     if( !ss )
  98     {
  99         ss = new SEARCH_STACK();
 100 
 101         // Make PROJECT the new SEARCH_STACK owner.

 102         SetElem( PROJECT::ELEM_SCH_SEARCH_STACK, ss );
 103 
 104         // to the empty SEARCH_STACK for SchSearchS(), add project dir as first

 105         ss->AddPaths( m_project_name.GetPath() );
 106 
 107         // next add the paths found in *.pro, variable "LibDir"

 108         wxString        libDir;
 109 
 110         try
 111         {
 112             PART_LIBS::LibNamesAndPaths( this, false, &libDir );
 113         }
 114         catch( const IO_ERROR& DBG( ioe ) )
 115         {
 116             DBG(printf( "%s: %s\n", __func__, TO_UTF8( ioe.What() ) );)
 117         }
 118 
 119         if( !!libDir )
 120         {
 121             wxArrayString   paths;
 122 
 123             SEARCH_STACK::Split( &paths, libDir );
 124 
 125             for( unsigned i =0; i<paths.GetCount();  ++i )
 126             {
 127                 wxString path = AbsolutePath( paths[i] );
 128 
 129                 ss->AddPaths( path );     // at the end

 130             }
 131         }
 132 
 133         // append all paths from aSList

 134         add_search_paths( ss, Kiface().KifaceSearch(), -1 );
 135     }
 136 
 137     return ss;
 138 }

 141 PART_LIBS* PROJECT::SchLibs()
 142 {
 143     PART_LIBS* libs = (PART_LIBS*) GetElem( PROJECT::ELEM_SCH_PART_LIBS );
 144 
 145     wxASSERT( !libs || libs->Type() == PART_LIBS_T );
 146 
 147     if( !libs )
 148     {
 149         libs = new PART_LIBS();
 150 
 151         // Make PROJECT the new PART_LIBS owner.

 152         SetElem( PROJECT::ELEM_SCH_PART_LIBS, libs );
 153 
 154         try
 155         {
 156             libs->LoadAllLibraries( this );
 157         }
 158         catch( const PARSE_ERROR& pe )
 159         {
 160             wxString    lib_list = UTF8( pe.inputLine );
 161             wxWindow*   parent = Pgm().App().GetTopWindow();
 162 
 163             // parent of this dialog cannot be NULL since that breaks the Kiway() chain.

 164             HTML_MESSAGE_BOX dlg( parent, _( "Not Found" ) );
 165 
 166             dlg.MessageSet( _( "The following libraries were not found:" ) );
 167 
 168             dlg.ListSet( lib_list );
 169 
 170             dlg.Layout();
 171 
 172             dlg.ShowModal();
 173         }
 174         catch( const IO_ERROR& ioe )
 175         {
 176             wxWindow* parent = Pgm().App().GetTopWindow();
 177 
 178             DisplayError( parent, ioe.What() );
 179         }
 180     }
 181 
 182     return libs;

 344 void SCH_EDIT_FRAME::SaveCopyForRepeatItem( SCH_ITEM* aItem )
 345 {
 346     // we cannot store a pointer to an item in the display list here since

 347     // that item may be deleted, such as part of a line concatenation or other.

 348     // So simply always keep a copy of the object which is to be repeated.

 350     if( aItem )
 351     {
 352         delete m_item_to_repeat;
 353 
 354         m_item_to_repeat = (SCH_ITEM*) aItem->Clone();

 357     }

 367 void SCH_EDIT_FRAME::SetSheetNumberAndCount()
 368 {
 369     SCH_SCREEN* screen;
 370     SCH_SCREENS s_list( Schematic().Root() );
 371 
 372     // Set the sheet count, and the sheet number (1 for root sheet)

 373     int              sheet_count       = Schematic().Root().CountSheets();
 374     int              sheet_number      = 1;
 375     const KIID_PATH& current_sheetpath = GetCurrentSheet().Path();
 376 
 377     // Examine all sheets path to find the current sheets path,

 378     // and count them from root to the current sheet path:

 379     for( const SCH_SHEET_PATH& sheet : Schematic().GetSheets() )
 380     {
 381         if( sheet.Path() == current_sheetpath )   // Current sheet path found

 382             break;
 383 
 384         sheet_number++;                         // Not found, increment before this current path

 385     }
 386 
 387     GetCurrentSheet().SetPageNumber( sheet_number );
 388 
 389     for( screen = s_list.GetFirst(); screen != NULL; screen = s_list.GetNext() )
 390         screen->m_NumberOfScreens = sheet_count;
 391 
 392     GetScreen()->m_ScreenNumber = sheet_number;
 393 }

 467 void SCH_EDIT_FRAME::OnCloseWindow( wxCloseEvent& aEvent )
 468 {
 469     // Shutdown blocks must be determined and vetoed as early as possible

 470     if( SupportsShutdownBlockReason() && aEvent.GetId() == wxEVT_QUERY_END_SESSION
 471             && Schematic().GetSheets().IsModified() )
 472     {
 473         aEvent.Veto();
 474         return;
 475     }
 476 
 477     if( Kiface().IsSingle() )
 478     {
 479         LIB_EDIT_FRAME* libeditFrame = (LIB_EDIT_FRAME*) Kiway().Player( FRAME_SCH_LIB_EDITOR, false );
 480         if( libeditFrame && !libeditFrame->Close() )   // Can close component editor?

 481             return;
 482 
 483         LIB_VIEW_FRAME* viewlibFrame = (LIB_VIEW_FRAME*) Kiway().Player( FRAME_SCH_VIEWER, false );
 484         if( viewlibFrame && !viewlibFrame->Close() )   // Can close component viewer?

 485             return;
 486 
 487         viewlibFrame = (LIB_VIEW_FRAME*) Kiway().Player( FRAME_SCH_VIEWER_MODAL, false );
 488 
 489         if( viewlibFrame && !viewlibFrame->Close() )   // Can close modal component viewer?

 490             return;
 491     }
 492 
 493     SIM_PLOT_FRAME* simFrame = (SIM_PLOT_FRAME*) Kiway().Player( FRAME_SIMULATOR, false );
 494 
 495     if( simFrame && !simFrame->Close() )   // Can close the simulator?

 496         return;
 497 
 498     SCH_SHEET_LIST sheetlist = Schematic().GetSheets();
 499 
 500     if( sheetlist.IsModified() )
 501     {
 502         wxFileName fileName = Schematic().RootScreen()->GetFileName();
 503         wxString msg = _( "Save changes to \"%s\" before closing?" );
 504 
 505         if( !HandleUnsavedChanges( this, wxString::Format( msg, fileName.GetFullName() ),
 506                                    [&]()->bool { return SaveProject(); } ) )
 507         {
 508             aEvent.Veto();
 509             return;
 510         }
 511     }
 512 
 513     //

 514     // OK, we're really closing now.  No more returns after this.

 515     //

 516 
 517     // Shutdown all running tools ( and commit any pending change )

 518     if( m_toolManager )
 519         m_toolManager->ShutdownAllTools();
 520 
 521     // Close the find dialog and preserve it's setting if it is displayed.

 522     if( m_findReplaceDialog )
 523     {
 524         m_findStringHistoryList = m_findReplaceDialog->GetFindEntries();
 525         m_replaceStringHistoryList = m_findReplaceDialog->GetReplaceEntries();
 526 
 527         m_findReplaceDialog->Destroy();
 528         m_findReplaceDialog = nullptr;
 529     }
 530 
 531     if( FindHierarchyNavigator() )
 532         FindHierarchyNavigator()->Close( true );
 533 
 534     SCH_SCREENS screens( Schematic().Root() );
 535     wxFileName fn;
 536 
 537     for( SCH_SCREEN* screen = screens.GetFirst(); screen != NULL; screen = screens.GetNext() )
 538     {
 539         fn = Prj().AbsolutePath( screen->GetFileName() );
 540 
 541         // Auto save file name is the normal file name prepended with GetAutoSaveFilePrefix().

 542         fn.SetName( GetAutoSaveFilePrefix() + fn.GetName() );
 543 
 544         if( fn.FileExists() && fn.IsFileWritable() )
 545             wxRemoveFile( fn.GetFullPath() );
 546     }
 547 
 548     sheetlist.ClearModifyStatus();
 549 
 550     wxString fileName = Prj().AbsolutePath( Schematic().RootScreen()->GetFileName() );
 551 
 552     if( !Schematic().GetFileName().IsEmpty() && !Schematic().RootScreen()->IsEmpty() )
 553         UpdateFileHistory( fileName );
 554 
 555     Schematic().RootScreen()->Clear();
 556 
 557     // all sub sheets are deleted, only the main sheet is usable

 558     GetCurrentSheet().clear();
 559 
 560     GetSettingsManager()->SaveProject();
 561     Schematic().SetTemplateFieldNames( nullptr );
 562     Schematic().Reset();
 563 
 564     Destroy();
 565 }

 568 wxString SCH_EDIT_FRAME::GetUniqueFilenameForCurrentSheet()
 569 {
 570     // Filename is rootSheetName-sheetName-...-sheetName

 571     // Note that we need to fetch the rootSheetName out of its filename, as the root SCH_SHEET's

 572     // name is just a timestamp.

 573 
 574     wxFileName rootFn( GetCurrentSheet().at( 0 )->GetFileName() );
 575     wxString   filename = rootFn.GetName();
 576 
 577     for( unsigned i = 1; i < GetCurrentSheet().size(); i++ )
 578         filename += wxT( "-" ) + GetCurrentSheet().at( i )->GetName();
 579 
 580     return filename;
 581 }

 601 void SCH_EDIT_FRAME::OnUpdatePCB( wxCommandEvent& event )
 602 {
 603     if( Kiface().IsSingle() )
 604     {
 605         DisplayError( this,  _( "Cannot update the PCB, because the Schematic Editor is opened"
 606                                 " in stand-alone mode. In order to create/update PCBs from"
 607                                 " schematics, launch the Kicad shell and create a project." ) );
 608         return;
 609     }
 610 
 611     KIWAY_PLAYER* frame = Kiway().Player( FRAME_PCB_EDITOR, false );
 612 
 613     if( !frame )
 614     {
 615         wxFileName fn = Prj().GetProjectFullName();
 616         fn.SetExt( PcbFileExtension );
 617 
 618         frame = Kiway().Player( FRAME_PCB_EDITOR, true );
 619         frame->OpenProjectFiles( std::vector<wxString>( 1, fn.GetFullPath() ) );
 620     }
 621 
 622     if( !frame->IsVisible() )
 623         frame->Show( true );
 624 
 625     // On Windows, Raise() does not bring the window on screen, when iconized

 626     if( frame->IsIconized() )
 627         frame->Iconize( false );
 628 
 629     frame->Raise();
 630 
 631     std::string payload;
 632     Kiway().ExpressMail( FRAME_PCB_EDITOR, MAIL_PCB_UPDATE, payload, this );
 633 }

 892 void SCH_EDIT_FRAME::AddItemToScreenAndUndoList( SCH_SCREEN* aScreen, SCH_ITEM* aItem,
 893                                                  bool aUndoAppend )
 894 {
 895     wxCHECK_RET( aItem != NULL, wxT( "Cannot add null item to list." ) );
 896 
 897     SCH_SHEET*     parentSheet = nullptr;
 898     SCH_COMPONENT* parentComponent = nullptr;
 899     SCH_ITEM*      undoItem = aItem;
 900 
 901     if( aItem->Type() == SCH_SHEET_PIN_T )
 902     {
 903         parentSheet = (SCH_SHEET*) aItem->GetParent();
 904 
 905         wxCHECK_RET( parentSheet && parentSheet->Type() == SCH_SHEET_T,
 906                      wxT( "Cannot place sheet pin in invalid schematic sheet object." ) );
 907 
 908         undoItem = parentSheet;
 909     }
 910 
 911     else if( aItem->Type() == SCH_FIELD_T )
 912     {
 913         parentComponent = (SCH_COMPONENT*) aItem->GetParent();
 914 
 915         wxCHECK_RET( parentComponent && parentComponent->Type() == SCH_COMPONENT_T,
 916                      wxT( "Cannot place field in invalid schematic component object." ) );
 917 
 918         undoItem = parentComponent;
 919     }
 920 
 921     if( aItem->IsNew() )
 922     {
 923         if( aItem->Type() == SCH_SHEET_PIN_T )
 924         {
 925             // Sheet pins are owned by their parent sheet.

 926             SaveCopyInUndoList( aScreen, undoItem, UR_CHANGED, aUndoAppend );
 927 
 928             parentSheet->AddPin( (SCH_SHEET_PIN*) aItem );
 929         }
 930         else if( aItem->Type() == SCH_FIELD_T )
 931         {
 932             // Component fields are also owned by their parent, but new component fields

 933             // are handled elsewhere.

 934             wxLogMessage( wxT( "addCurrentItemToScreen: unexpected new SCH_FIELD" ) );
 935         }
 936         else
 937         {
 938             if( !aScreen->CheckIfOnDrawList( aItem ) )  // don't want a loop!

 939                 AddToScreen( aItem, aScreen );
 940 
 941             SaveCopyForRepeatItem( aItem );
 942             SaveCopyInUndoList( aScreen, undoItem, UR_NEW, aUndoAppend );
 943         }
 944 
 945         // Update connectivity info for new item

 946         if( !aItem->IsMoving() )
 947             RecalculateConnections( LOCAL_CLEANUP );
 948     }
 949 
 950     aItem->ClearFlags( IS_NEW );
 951 
 952     aScreen->SetModify();
 953     RefreshItem( aItem );
 954 
 955     if( !aItem->IsMoving() && aItem->IsConnectable() )
 956     {
 957         std::vector< wxPoint > pts;
 958         aItem->GetConnectionPoints( pts );
 959 
 960         for( auto i = pts.begin(); i != pts.end(); i++ )
 961         {
 962             for( auto j = i + 1; j != pts.end(); j++ )
 963                 TrimWire( *i, *j );
 964 
 965             if( aScreen->IsJunctionNeeded( *i, true ) )
 966                 AddJunction( aScreen, *i, true, false );
 967         }
 968 
 969         TestDanglingEnds();
 970 
 971         for( SCH_ITEM* item : aItem->ConnectedItems( GetCurrentSheet() ) )
 972             RefreshItem( item );
 973     }
 974 
 975     aItem->ClearEditFlags();
 976     GetCanvas()->Refresh();
 977 }

1090 void SCH_EDIT_FRAME::FixupJunctions()
1091 {
1092     // Save the current sheet, to retrieve it later

1093     SCH_SHEET_PATH oldsheetpath = GetCurrentSheet();
1094 
1095     bool modified = false;
1096 
1097     SCH_SHEET_LIST sheetList = Schematic().GetSheets();
1098 
1099     for( const SCH_SHEET_PATH& sheet : sheetList )
1100     {
1101         // We require a set here to avoid adding multiple junctions to the same spot

1102         std::set<wxPoint> junctions;
1103 
1104         SetCurrentSheet( sheet );
1105         GetCurrentSheet().UpdateAllScreenReferences();
1106 
1107         SCH_SCREEN* screen = GetCurrentSheet().LastScreen();
1108 
1109         for( auto aItem : screen->Items().OfType( SCH_COMPONENT_T ) )
1110         {
1111             auto cmp = static_cast<SCH_COMPONENT*>( aItem );
1112 
1113             for( const SCH_PIN* pin : cmp->GetSchPins( &sheet ) )
1114             {
1115                 auto pos = pin->GetPosition();
1116 
1117                 // Test if a _new_ junction is needed, and add it if missing

1118                 if( screen->IsJunctionNeeded( pos, true ) )
1119                     junctions.insert( pos );
1120             }
1121         }
1122 
1123         for( const wxPoint& pos : junctions )
1124             AddJunction( screen, pos, false, false );
1125 
1126         if( junctions.size() )
1127             modified = true;
1128     }
1129 
1130     if( modified )
1131         OnModify();
1132 
1133     // Reselect the initial sheet:

1134     SetCurrentSheet( oldsheetpath );
1135     GetCurrentSheet().UpdateAllScreenReferences();
1136     SetScreen( GetCurrentSheet().LastScreen() );
1137 }
