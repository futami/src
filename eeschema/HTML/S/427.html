  65 void SYMBOL_TREE_SYNCHRONIZING_ADAPTER::Sync( bool aForce,
  66         std::function<void( int, int, const wxString& )> aProgressCallback )
  67 {
  68     wxLongLong nextUpdate = wxGetUTCTimeMillis() + (PROGRESS_INTERVAL_MILLIS / 2);
  69 
  70     int libMgrHash = m_libMgr->GetHash();
  71 
  72     if( !aForce && m_lastSyncHash == libMgrHash )
  73         return;
  74 
  75     m_lastSyncHash = libMgrHash;
  76     int i = 0, max = GetLibrariesCount();
  77 
  78     // Process already stored libraries

  79     for( auto it = m_tree.m_Children.begin(); it != m_tree.m_Children.end(); /* iteration inside */ )
  80     {
  81         const wxString& name = it->get()->m_Name;
  82 
  83         if( wxGetUTCTimeMillis() > nextUpdate )
  84         {
  85             aProgressCallback( i, max, name );
  86             nextUpdate = wxGetUTCTimeMillis() + PROGRESS_INTERVAL_MILLIS;
  87         }
  88 
  89         // There is a bug in LIB_MANAGER::LibraryExists() that uses the buffered modified

  90         // libraries before the symbol library table which prevents the library from being

  91         // removed from the tree control.

  92         if( !m_libMgr->LibraryExists( name, true )
  93           || !m_frame->Prj().SchSymbolLibTable()->HasLibrary( name, true ) )
  94         {
  95             it = deleteLibrary( it );
  96             continue;
  97         }
  98         else if( aForce || m_libMgr->GetLibraryHash( name ) != m_libHashes[name] )
  99         {
 100             updateLibrary( *(LIB_TREE_NODE_LIB*) it->get() );
 101         }
 102 
 103         ++it;
 104         ++i;
 105     }
 106 
 107     // Look for new libraries

 108     for( const wxString& libName : m_libMgr->GetLibraryNames() )
 109     {
 110         if( m_libHashes.count( libName ) == 0 )
 111         {
 112             if( wxGetUTCTimeMillis() > nextUpdate )
 113             {
 114                 aProgressCallback( i++, max, libName );
 115                 nextUpdate = wxGetUTCTimeMillis() + PROGRESS_INTERVAL_MILLIS;
 116             }
 117 
 118             SYMBOL_LIB_TABLE_ROW* library = m_libMgr->GetLibrary( libName );
 119             LIB_TREE_NODE_LIB&    lib_node = DoAddLibraryNode( libName, library->GetDescr() );
 120 
 121             updateLibrary( lib_node );
 122         }
 123     }
 124 
 125     m_tree.AssignIntrinsicRanks();
 126 }

 129 int SYMBOL_TREE_SYNCHRONIZING_ADAPTER::GetLibrariesCount() const
 130 {
 131     int count = LIB_TREE_MODEL_ADAPTER::GetLibrariesCount();
 132 
 133     for( const wxString& libName : m_libMgr->GetLibraryNames() )
 134     {
 135         if( m_libHashes.count( libName ) == 0 )
 136             ++count;
 137     }
 138 
 139     return count;

 143 void SYMBOL_TREE_SYNCHRONIZING_ADAPTER::updateLibrary( LIB_TREE_NODE_LIB& aLibNode )
 144 {
 145     auto hashIt = m_libHashes.find( aLibNode.m_Name );
 146 
 147     if( hashIt == m_libHashes.end() )
 148     {
 149         // add a new library

 150         for( LIB_PART* alias : m_libMgr->GetAliases( aLibNode.m_Name ) )
 151             aLibNode.AddItem( alias );
 152     }
 153     else if( hashIt->second != m_libMgr->GetLibraryHash( aLibNode.m_Name ) )
 154     {
 155         // update an existing library

 156         std::list<LIB_PART*> aliases = m_libMgr->GetAliases( aLibNode.m_Name );
 157 
 158         // remove the common part from the aliases list

 159         for( auto nodeIt = aLibNode.m_Children.begin(); nodeIt != aLibNode.m_Children.end(); /**/ )
 160         {
 161             auto aliasIt = std::find_if( aliases.begin(), aliases.end(),
 162                                          [&] ( const LIB_PART* a )
 163                                          {
 164                                              return a->GetName() == (*nodeIt)->m_Name;
 165                                          } );
 166 
 167             if( aliasIt != aliases.end() )
 168             {
 169                 // alias exists both in the component tree and the library manager,

 170                 // update only the node data

 171                 static_cast<LIB_TREE_NODE_LIB_ID*>( nodeIt->get() )->Update( *aliasIt );
 172                 aliases.erase( aliasIt );
 173                 ++nodeIt;
 174             }
 175             else
 176             {
 177                 // node does not exist in the library manager, remove the corresponding node

 178                 nodeIt = aLibNode.m_Children.erase( nodeIt );
 179             }
 180         }
 181 
 182         // now the aliases list contains only new aliases that need to be added to the tree

 183         for( LIB_PART* alias : aliases )
 184             aLibNode.AddItem( alias );
 185     }
 186 
 187     aLibNode.AssignIntrinsicRanks();
 188     m_libHashes[aLibNode.m_Name] = m_libMgr->GetLibraryHash( aLibNode.m_Name );
 189 }

 192 LIB_TREE_NODE::PTR_VECTOR::iterator SYMBOL_TREE_SYNCHRONIZING_ADAPTER::deleteLibrary(
 193             LIB_TREE_NODE::PTR_VECTOR::iterator& aLibNodeIt )
 194 {

 196     m_libHashes.erase( node->m_Name );

 202 void SYMBOL_TREE_SYNCHRONIZING_ADAPTER::GetValue( wxVariant& aVariant, wxDataViewItem const& aItem,
 203                                                   unsigned int aCol ) const
 204 {
 205     if( IsFrozen() )
 206     {
 207         aVariant = wxEmptyString;
 208         return;
 209     }
 210 
 211     LIB_TREE_NODE* node = ToNode( aItem );
 212     wxASSERT( node );
 213 
 214     switch( aCol )
 215     {
 216     case 0:
 217         if( m_frame->GetCurPart() && m_frame->GetCurPart()->GetLibId() == node->m_LibId )
 218             node->m_Name = m_frame->GetCurPart()->GetLibId().GetLibItemName();
 219 
 220         if( node->m_Pinned )
 221             aVariant = GetPinningSymbol() + node->m_Name;
 222         else
 223             aVariant = node->m_Name;
 224 
 225         // mark modified items with an asterisk

 226         if( node->m_Type == LIB_TREE_NODE::LIB )
 227         {
 228             if( m_libMgr->IsLibraryModified( node->m_Name ) )
 229                 aVariant = aVariant.GetString() + " *";
 230         }
 231         else if( node->m_Type == LIB_TREE_NODE::LIBID )
 232         {
 233             if( m_libMgr->IsPartModified( node->m_Name, node->m_Parent->m_Name ) )
 234                 aVariant = aVariant.GetString() + " *";
 235         }
 236 
 237         break;
 238 
 239     case 1:
 240         if( m_frame->GetCurPart() && m_frame->GetCurPart()->GetLibId() == node->m_LibId )
 241             node->m_Desc = m_frame->GetCurPart()->GetDescription();
 242 
 243         aVariant = node->m_Desc;
 244         break;
 245 
 246     default:    // column == -1 is used for default Compare function

 247         aVariant = node->m_Name;
 248         break;
 249     }
 250 }

 253 bool SYMBOL_TREE_SYNCHRONIZING_ADAPTER::GetAttr( wxDataViewItem const& aItem, unsigned int aCol,
 254                                                  wxDataViewItemAttr& aAttr ) const
 255 {
 256     if( IsFrozen() )
 257         return false;
 258 
 259     // change attributes only for the name field

 260     if( aCol != 0 )
 261         return false;
 262 
 263     LIB_TREE_NODE* node = ToNode( aItem );
 264     wxCHECK( node, false );
 265 
 266     switch( node->m_Type )
 267     {
 268     case LIB_TREE_NODE::LIB:
 269         // mark modified libs with bold font

 270         aAttr.SetBold( m_libMgr->IsLibraryModified( node->m_Name ) );
 271 
 272 #ifdef __WXGTK__
 273         // The native wxGTK+ impl ignores background colour, so set the text colour instead.

 274         // This works reasonably well in dark themes, and quite poorly in light ones....

 275         if( node->m_Name == m_libMgr->GetCurrentLib() )
 276             aAttr.SetColour( wxSystemSettings::GetColour( wxSYS_COLOUR_HIGHLIGHT ) );
 277 #else
 278         // mark the current library with background color

 279         if( node->m_Name == m_libMgr->GetCurrentLib() )
 280         {
 281             aAttr.SetBackgroundColour( wxSystemSettings::GetColour( wxSYS_COLOUR_HIGHLIGHT  ) );
 282             aAttr.SetColour( wxSystemSettings::GetColour( wxSYS_COLOUR_LISTBOXHIGHLIGHTTEXT  ) );
 283         }
 284 #endif
 285         break;
 286 
 287     case LIB_TREE_NODE::LIBID:
 288         // mark modified part with bold font

 289         aAttr.SetBold( m_libMgr->IsPartModified( node->m_Name, node->m_Parent->m_Name ) );
 290 
 291         // mark aliases with italic font

 292         aAttr.SetItalic( !node->m_IsRoot );
 293 
 294 #ifdef __WXGTK__
 295         // The native wxGTK+ impl ignores background colour, so set the text colour instead.

 296         // This works reasonably well in dark themes, and quite poorly in light ones....

 297         if( node->m_LibId == m_libMgr->GetCurrentLibId() )
 298             aAttr.SetColour( wxSystemSettings::GetColour( wxSYS_COLOUR_HIGHLIGHT ) );
 299 #else
 300         // mark the current part with background color

 301         if( node->m_LibId == m_libMgr->GetCurrentLibId() )
 302         {
 303             aAttr.SetBackgroundColour( wxSystemSettings::GetColour( wxSYS_COLOUR_HIGHLIGHT ) );
 304             aAttr.SetColour( wxSystemSettings::GetColour( wxSYS_COLOUR_LISTBOXHIGHLIGHTTEXT  ) );
 305         }
 306 #endif
 307         break;
 308 
 309     default:
 310         return false;
 311     }
 312 
 313     return true;
 314 }
