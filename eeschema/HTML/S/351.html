  40 enum class CONNECTION_TYPE
  41 {
  42     NONE,      ///< No connection to this item

  43     NET,       ///< This item represents a net

  44     BUS,       ///< This item represents a bus vector

  45     BUS_GROUP, ///< This item represents a bus group

  46 };

  48 /**

  49  * Each graphical item can have a SCH_CONNECTION describing its logical

  50  * connection (to a bus or net).  These are generated when netlisting, or when

  51  * editing operations that can change the netlist are performed.

  52  *

  53  * In hierarchical schematics, a single SCH_ITEM object can refer to multiple

  54  * distinct parts of a design (in the case of a sub-sheet that is instanced

  55  * more than once in a higher level sheet).  Because of this, a single item may

  56  * contain more than one SCH_CONNECTION -- each is specific to a sheet.

  57  *

  58  * Components contain connections for each of their pins (and for each sheet

  59  * they exist on) but don't use their own connection object.

  60  */
  61 class SCH_CONNECTION
  62 {
  63 public:
  64     SCH_CONNECTION( SCH_ITEM* aParent = nullptr, SCH_SHEET_PATH aPath = SCH_SHEET_PATH() );
  65

  68     ~SCH_CONNECTION()
  69     {}

  73      * properties, not the ownership / sheet location!

  77     bool operator!=( const SCH_CONNECTION& aOther ) const;

  85      * Configures the connection given a label.

  86      * For CONNECTION_NET, this just sets the name.

  87      * For CONNECTION_BUS, this will deduce the correct BUS_TYPE and also

  88      * generate a correct list of members.

  89      */
  90     void ConfigureFromLabel( const wxString& aLabel );

 324     /**

 325      * For bus members, we want to keep track of the "local" name of a member, that is,

 326      * the name it takes on from its parent bus name.  This is because we always want to use

 327      * the local name for bus unfolding, matching within buses, etc.  The actual resolved name

 328      * of this bus member might change, for example if it's connected elsewhere to some other

 329      * item with higher priority.

 330      */
 331     wxString m_local_name;
 332 
 333     /// Prefix if connection is member of a labeled bus group (or "" if not)

 338 
 339     wxString m_suffix;      ///< Name suffix (used only for disambiguation)

 344 
 345     int m_subgraph_code;    ///< Groups directly-connected items

 346 
 347     long m_vector_index;    ///< Index of bus vector member nets

 348 
 349     long m_vector_start;    ///< Highest member of a vector bus

 350 
 351     long m_vector_end;      ///< Lowest member of a vector bus

 352 
 353     ///< Prefix name of the vector, if m_type == CONNECTION_BUS (or "" if not)

 354     wxString m_vector_prefix;
 355 
 356     /**

 357      * For bus connections, store a list of member connections

 358      *

 359      * NOTE: All connections that Clone() others share the list of member

 360      * pointers.  This seems fine at the moment.

 361      */
 362     std::vector< std::shared_ptr< SCH_CONNECTION > > m_members;
 363 
 364     /**

 367      */
 368     CONNECTION_GRAPH* m_graph;
