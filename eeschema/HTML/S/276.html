  58 /**

  59  * The base class for drawable items used by schematic library components.

  60  */
  61 class LIB_ITEM : public EDA_ITEM
  62 {
  63     /**

  64      * Print the item to \a aDC.

  65      *

  66      * @param aOffset A reference to a wxPoint object containing the offset where to draw

  67      *                from the object's current position.

  68      * @param aData A pointer to any object specific data required to perform the draw.

  69      * @param aTransform A reference to a #TRANSFORM object containing drawing transform.

  70      */
  71     virtual void print( RENDER_SETTINGS* aSettings, const wxPoint& aOffset, void* aData,
  72                         const TRANSFORM& aTransform ) = 0;
  73 
  74     friend class LIB_PART;
  75 
  76 protected:
  77     /**

  78      * Unit identification for multiple parts per package.  Set to 0 if the item is common

  79      * to all units.

  80      */
  81     int      m_Unit;
  82 
  83     /**

  84      * Shape identification for alternate body styles.  Set 0 if the item is common to all

  85      * body styles.  This is typially used for representing DeMorgan variants in KiCad.

  86      */
  87     int      m_Convert;
  88 
  89     /**

  90      * The body fill type.  This has meaning only for some items.  For a list of fill types

  91      * see #FILL_T.

  92      */
  93     FILL_T   m_Fill;
  94     bool     m_isFillable;
  95 
  96 public:
  97 
  98     LIB_ITEM( KICAD_T aType, LIB_PART* aComponent = NULL, int aUnit = 0, int aConvert = 0,
  99               FILL_T aFillType = NO_FILL );
 100 
 101     // Do not create a copy constructor.  The one generated by the compiler is adequate.

 102 
 103     virtual ~LIB_ITEM() { }
 104 
 105 
 106     // Define the enums for basic

 107     enum LIB_CONVERT : int  { BASE = 1, DEMORGAN = 2 };
 108 
 109     /**

 110      * The list of flags used by the #compare function.

 111      *

 112      * - NORMAL This compares everthing between two #LIB_ITEM objects.

 113      * - UNIT This compare flag ignores unit and convert and pin number information when

 114      *        comparing #LIB_ITEM objects for unit comparison.

 115      */
 116     enum COMPARE_FLAGS : int { NORMAL = 0x00, UNIT = 0x01 };
 117 
 118     /**

 119      * Provide a user-consumable name of the object type.  Perform localization when

 120      * called so that run-time language selection works.

 121      */
 122     virtual wxString GetTypeName() = 0;
 123 
 124     /**

 125      * Begin drawing a component library draw item at \a aPosition.

 126      *

 127      * It typically would be called on a left click when a draw tool is selected in

 128      * the component library editor and one of the graphics tools is selected.

 129      *

 130      * @param aPosition The position in drawing coordinates where the drawing was started.

 131      *                  May or may not be required depending on the item being drawn.

 132      */
 133     virtual void BeginEdit( const wxPoint aPosition ) {}
 134 
 135     /**

 136      * Continue an edit in progress at \a aPosition.

 137      *

 138      * This is used to perform the next action while drawing an item.  This would be

 139      * called for each additional left click when the mouse is captured while the item

 140      * is being drawn.

 141      *

 142      * @param aPosition The position of the mouse left click in drawing coordinates.

 143      * @return True if additional mouse clicks are required to complete the edit in progress.

 144      */
 145     virtual bool ContinueEdit( const wxPoint aPosition ) { return false; }
 146 
 147     /**

 148      * End an object editing action.

 149      *

 150      * This is used to end or abort an edit action in progress initiated by BeginEdit().

 151      */
 152     virtual void EndEdit() {}
 153 
 154     /**

 155      * Calculates the attributes of an item at \a aPosition when it is being edited.

 156      *

 157      * This method gets called by the Draw() method when the item is being edited.  This

 158      * probably should be a pure virtual method but bezier curves are not yet editable in

 159      * the component library editor.  Therefore, the default method does nothing.

 160      *

 161      * @param aPosition The current mouse position in drawing coordinates.

 162      */
 163     virtual void CalcEdit( const wxPoint& aPosition ) {}
 164 
 165 
 166     /**

 167      * Draw an item

 168      *

 169      * @param aDC Device Context (can be null)

 170      * @param aOffset Offset to draw

 171      * @param aData Value or pointer used to pass others parameters, depending on body items.

 172      *              Used for some items to force to force no fill mode ( has meaning only for

 173      *              items what can be filled ). used in printing or moving objects mode or to

 174      *              pass reference to the lib component for pins.

 175      * @param aTransform Transform Matrix (rotation, mirror ..)

 176      */
 177     virtual void Print( RENDER_SETTINGS* aSettings, const wxPoint &aOffset, void* aData,
 178                         const TRANSFORM& aTransform );
 179 
 180     virtual int GetPenWidth() const = 0;
 181 
 182     LIB_PART* GetParent() const
 183     {
 184         return (LIB_PART*) m_Parent;
 185     }
 186 
 187     void ViewGetLayers( int aLayers[], int& aCount ) const override;
 188 
 189     bool HitTest( const wxPoint& aPosition, int aAccuracy = 0 ) const override
 190     {
 191         // This is just here to prevent annoying compiler warnings about hidden overloaded

 192         // virtual functions

 193         return EDA_ITEM::HitTest( aPosition, aAccuracy );
 194     }
 195 
 196     bool HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy = 0 ) const override;
 197 
 198     /**

 199      * @return the boundary box for this, in library coordinates

 200      */
 201     const EDA_RECT GetBoundingBox() const override { return EDA_ITEM::GetBoundingBox(); }
 202 
 203     /**

 204      * Display basic info (type, part and convert) about the current item in message panel.

 205      * <p>

 206      * This base function is used to display the information common to the

 207      * all library items.  Call the base class from the derived class or the

 208      * common information will not be updated in the message panel.

 209      * </p>

 210      * @param aList is the list to populate.

 211      */
 212     void GetMsgPanelInfo( EDA_DRAW_FRAME* aFrame, std::vector<MSG_PANEL_ITEM>& aList ) override;
 213 
 214     /**

 215      * Test LIB_ITEM objects for equivalence.

 216      *

 217      * @param aOther Object to test against.

 218      * @return True if object is identical to this object.

 219      */
 220     bool operator==( const LIB_ITEM& aOther ) const;
 221     bool operator==( const LIB_ITEM* aOther ) const
 222     {
 223         return *this == *aOther;
 224     }
 225 
 226     /**

 227      * Test if another draw item is less than this draw object.

 228      *

 229      * @param aOther - Draw item to compare against.

 230      * @return - True if object is less than this object.

 231      */
 232     bool operator<( const LIB_ITEM& aOther) const;
 233 
 234     /**

 235      * Set the drawing object by \a aOffset from the current position.

 236      *

 237      * @param aOffset Coordinates to offset the item position.

 238      */
 239     virtual void Offset( const wxPoint& aOffset ) = 0;
 240 
 241     /**

 242      * Move a draw object to \a aPosition.

 243      *

 244      * @param aPosition Position to move draw item to.

 245      */
 246     virtual void MoveTo( const wxPoint& aPosition ) = 0;
 247 
 248     void SetPosition( const wxPoint& aPosition ) { MoveTo( aPosition ); }
 249 
 250     /**

 251      * Mirror the draw object along the horizontal (X) axis about \a aCenter point.

 252      *

 253      * @param aCenter Point to mirror around.

 254      */
 255     virtual void MirrorHorizontal( const wxPoint& aCenter ) = 0;
 256 
 257     /**

 258      * Mirror the draw object along the MirrorVertical (Y) axis about \a aCenter point.

 259      *

 260      * @param aCenter Point to mirror around.

 261      */
 262     virtual void MirrorVertical( const wxPoint& aCenter ) = 0;
 263 
 264     /**

 265      * Rotate the object about \a aCenter point.

 266      *

 267      * @param aCenter Point to rotate around.

 268      * @param aRotateCCW True to rotate counter clockwise.  False to rotate clockwise.

 269      */
 270     virtual void Rotate( const wxPoint& aCenter, bool aRotateCCW = true ) = 0;
 271 
 272     /**

 273      * Plot the draw item using the plot object.

 274      *

 275      * @param aPlotter The plot object to plot to.

 276      * @param aOffset Plot offset position.

 277      * @param aFill Flag to indicate whether or not the object is filled.

 278      * @param aTransform The plot transform.

 279      */
 280     virtual void Plot( PLOTTER* aPlotter, const wxPoint& aOffset, bool aFill,
 281                        const TRANSFORM& aTransform ) = 0;
 282 
 283     virtual int GetWidth() const = 0;
 284     virtual void SetWidth( int aWidth ) = 0;
 285 
 286     /**

 287      * Check if draw object can be filled.

 288      *

 289      * The default setting is false.  If the derived object support filling, set the

 290      * m_isFillable member to true.

 291      */
 292     bool IsFillable() const { return m_isFillable; }
 293 
 294     void SetUnit( int aUnit ) { m_Unit = aUnit; }
 295     int GetUnit() const { return m_Unit; }
 296 
 297     void SetConvert( int aConvert ) { m_Convert = aConvert; }
 298     int GetConvert() const { return m_Convert; }
 299 
 300     void SetFillMode( FILL_T aFillMode ) { m_Fill = aFillMode; }
 301     FILL_T GetFillMode() const { return m_Fill; }
 302 
 303 #if defined(DEBUG)
 304     void Show( int nestLevel, std::ostream& os ) const override { ShowDummy( os ); }
 305 #endif
 306 
 307 protected:
 308 
 309     /**

 310      * Provide the draw object specific comparison called by the == and < operators.

 311      *

 312      * The base object sort order which always proceeds the derived object sort order

 313      * is as follows:

 314      *      - Component alternate part (DeMorgan) number.

 315      *      - Component part number.

 316      *      - KICAD_T enum value.

 317      *      - Result of derived classes comparison.

 318      *

 319      * @note Make sure you call down to #LIB_ITEM::compare before doing any derived object

 320      *       comparisons or you will break the sorting using the symbol library file format.

 321      *

 322      * @param aOther A reference to the other #LIB_ITEM to compare the arc against.

 323      * @param aCompareFlags The flags used to perform the comparison.

 324      *

 325      * @return An integer value less than 0 if the object is less than \a aOther ojbect,

 326      *         zero if the object is equal to \a aOther object, or greater than 0 if the

 327      *         object is greater than \a aOther object.

 328      */
 329     virtual int compare( const LIB_ITEM& aOther,
 330             LIB_ITEM::COMPARE_FLAGS aCompareFlags = LIB_ITEM::COMPARE_FLAGS::NORMAL ) const;
 331 };
