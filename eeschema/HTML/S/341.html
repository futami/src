  36 /* Functions to undo and redo edit commands.

  37  *

  38  *  m_UndoList and m_RedoList handle a std::vector of PICKED_ITEMS_LIST

  39  *  Each PICKED_ITEMS_LIST handle a std::vector of pickers (class ITEM_PICKER),

  40  *  that store the list of schematic items that are concerned by the command to

  41  *  undo or redo and is created for each command to undo (handle also a command

  42  *  to redo). each picker has a pointer pointing to an item to undo or redo (in

  43  *  fact: deleted, added or modified), and has a pointer to a copy of this item,

  44  *  when this item has been modified (the old values of parameters are

  45  *  therefore saved)

  46  *

  47  *  there are 3 cases:

  48  *  - delete item(s) command

  49  *  - change item(s) command

  50  *  - add item(s) command

  51  *  and 2 cases for block:

  52  *  - move list of items

  53  *  - mirror (Y) list of items

  54  *

  55  *  Undo command

  56  *  - delete item(s) command:

  57  *       =>  deleted items are moved in undo list

  58  *

  59  *  - change item(s) command

  60  *      => A copy of item(s) is made (a DrawPickedStruct list of wrappers)

  61  *      the .m_Link member of each wrapper points the modified item.

  62  *      the .m_Item member of each wrapper points the old copy of this item.

  63  *

  64  *  - add item(s) command

  65  *      =>A list of item(s) is made. The .m_Item member of each wrapper points

  66  *        the new item.

  67  *

  68  *  Redo command

  69  *  - delete item(s) old command:

  70  *      => deleted items are moved into m_tree

  71  *

  72  *  - change item(s) command

  73  *      => the copy of item(s) is moved in Undo list

  74  *

  75  *  - add item(s) command

  76  *      => The list of item(s) is used to create a deleted list in undo

  77  *         list(same as a delete command)

  78  *

  79  *   Some block operations that change items can be undone without memorized

  80  *   items, just the coordinates of the transform: move list of items (undo/

  81  *   redo is made by moving with the opposite move vector) mirror (Y) and flip

  82  *   list of items (undo/redo is made by mirror or flip items) so they are

  83  *    handled specifically.

  84  *

  85  *  A problem is the hierarchical sheet handling.

  86  *  the data associated (sub-hierarchy, undo/redo list) is deleted only

  87  *  when the sheet is really deleted (i.e. when deleted from undo or redo list)

  88  *  This is handled by its destructor.

  89  */

  93  * swap data between Item and its copy, pointed by its picked item link member

  97 void SCH_EDIT_FRAME::SaveCopyInUndoList( SCH_SCREEN*    aScreen,
  98                                          SCH_ITEM*      aItem,
  99                                          UNDO_REDO_T    aCommandType,
 100                                          bool           aAppend,
 101                                          const wxPoint& aTransformPoint )
 102 {
 103     PICKED_ITEMS_LIST* commandToUndo = nullptr;
 104 
 105     if( !aItem )
 106         return;
 107 
 108     // Connectivity may change

 109     aItem->SetConnectivityDirty();
 110 
 111     if( aAppend )
 112         commandToUndo = PopCommandFromUndoList();
 113 
 114     if( !commandToUndo )
 115     {
 116         commandToUndo = new PICKED_ITEMS_LIST();
 117         commandToUndo->m_TransformPoint = aTransformPoint;
 118     }
 119 
 120     ITEM_PICKER itemWrapper( aScreen, aItem, aCommandType );
 121     itemWrapper.SetFlags( aItem->GetFlags() );
 122 
 123     switch( aCommandType )
 124     {
 125     case UR_CHANGED:            /* Create a copy of item */
 126         itemWrapper.SetLink( aItem->Duplicate( true ) );
 127         commandToUndo->PushItem( itemWrapper );
 128         break;
 129 
 130     case UR_NEW:
 131     case UR_DELETED:
 132     case UR_ROTATED:
 133     case UR_MOVED:
 134         commandToUndo->PushItem( itemWrapper );
 135         break;
 136 
 137     default:
 138         wxFAIL_MSG( wxString::Format( wxT( "SaveCopyInUndoList() error (unknown code %X)" ),
 139                                       aCommandType ) );
 140         break;
 141     }
 142 
 143     if( commandToUndo->GetCount() )
 144     {
 145         /* Save the copy in undo list */
 146         PushCommandToUndoList( commandToUndo );
 147 
 148         /* Clear redo list, because after new save there is no redo to do */
 149         ClearUndoORRedoList( REDO_LIST );
 150     }
 151     else
 152     {
 153         delete commandToUndo;
 154     }
 155 }

 158 void SCH_EDIT_FRAME::SaveCopyInUndoList( const PICKED_ITEMS_LIST& aItemsList,
 159                                          UNDO_REDO_T              aTypeCommand,
 160                                          bool                     aAppend,
 161                                          const wxPoint&           aTransformPoint )
 162 {
 163     PICKED_ITEMS_LIST* commandToUndo = nullptr;
 164 
 165     if( !aItemsList.GetCount() )
 166         return;
 167 
 168     // Can't append a WIRE IMAGE, so fail to a new undo point

 169     if( aAppend )
 170         commandToUndo = PopCommandFromUndoList();
 171 
 172     if( !commandToUndo )
 173     {
 174         commandToUndo = new PICKED_ITEMS_LIST();
 175         commandToUndo->m_TransformPoint = aTransformPoint;
 176         commandToUndo->m_Status = aTypeCommand;
 177     }
 178 
 179     // Copy picker list:

 180     if( !commandToUndo->GetCount() )
 181         commandToUndo->CopyList( aItemsList );
 182     else
 183     {
 184         // Unless we are appending, in which case, get the picker items

 185         for( unsigned ii = 0; ii < aItemsList.GetCount(); ii++ )
 186             commandToUndo->PushItem( aItemsList.GetItemWrapper( ii) );
 187     }
 188 
 189     // Verify list, and creates data if needed

 190     for( unsigned ii = 0; ii < commandToUndo->GetCount(); ii++ )
 191     {
 192         SCH_ITEM* sch_item = dynamic_cast<SCH_ITEM*>( commandToUndo->GetPickedItem( ii ) );
 193 
 194         // Common items implemented in EDA_DRAW_FRAME will not be SCH_ITEMs.

 195         if( !sch_item )
 196             continue;
 197 
 198         // Connectivity may change

 199         sch_item->SetConnectivityDirty();
 200 
 201         UNDO_REDO_T command = commandToUndo->GetPickedItemStatus( ii );
 202 
 203         if( command == UR_UNSPECIFIED )
 204         {
 205             command = aTypeCommand;
 206             commandToUndo->SetPickedItemStatus( command, ii );
 207         }
 208 
 209         switch( command )
 210         {
 211         case UR_CHANGED:
 212 
 213             /* If needed, create a copy of item, and put in undo list

 214              * in the picker, as link

 215              * If this link is not null, the copy is already done

 216              */
 217             if( commandToUndo->GetPickedItemLink( ii ) == nullptr )
 218                 commandToUndo->SetPickedItemLink( sch_item->Duplicate( true ), ii );
 219 
 220             wxASSERT( commandToUndo->GetPickedItemLink( ii ) );
 221             break;
 222 
 223         case UR_MOVED:
 224         case UR_MIRRORED_Y:
 225         case UR_MIRRORED_X:
 226         case UR_ROTATED:
 227         case UR_NEW:
 228         case UR_DELETED:
 229         case UR_EXCHANGE_T:
 230         case UR_PAGESETTINGS:
 231             break;
 232 
 233         default:
 234             wxFAIL_MSG( wxString::Format( wxT( "Unknown undo/redo command %d" ), command ) );
 235             break;
 236         }
 237     }
 238 
 239     if( commandToUndo->GetCount() )
 240     {
 241         /* Save the copy in undo list */
 242         PushCommandToUndoList( commandToUndo );
 243 
 244         /* Clear redo list, because after new save there is no redo to do */
 245         ClearUndoORRedoList( REDO_LIST );
 246     }
 247     else    // Should not occur

 248     {
 249         delete commandToUndo;
 250     }
 251 }

 254 void SCH_EDIT_FRAME::PutDataInPreviousState( PICKED_ITEMS_LIST* aList, bool aRedoCommand )
 255 {
 256     // Undo in the reverse order of list creation: (this can allow stacked changes like the

 257     // same item can be changed and deleted in the same complex command).

 258     for( int ii = aList->GetCount() - 1; ii >= 0; ii-- )
 259     {
 260         UNDO_REDO_T status = aList->GetPickedItemStatus((unsigned) ii );
 261         EDA_ITEM*   eda_item = aList->GetPickedItem( (unsigned) ii );
 262 
 263         eda_item->SetFlags( aList->GetPickerFlags( (unsigned) ii ) );
 264         eda_item->ClearEditFlags();
 265         eda_item->ClearTempFlags();
 266 
 267         if( status == UR_NEW )
 268         {
 269             // new items are deleted on undo

 270             RemoveFromScreen( eda_item, (SCH_SCREEN*) aList->GetScreenForItem( (unsigned) ii ) );
 271             aList->SetPickedItemStatus( UR_DELETED, (unsigned) ii );
 272         }
 273         else if( status == UR_DELETED )
 274         {
 275             // deleted items are re-inserted on undo

 276             AddToScreen( eda_item, (SCH_SCREEN*) aList->GetScreenForItem( (unsigned) ii ) );
 277             aList->SetPickedItemStatus( UR_NEW, (unsigned) ii );
 278         }
 279         else if( status == UR_PAGESETTINGS )
 280         {
 281             // swap current settings with stored settings

 282             WS_PROXY_UNDO_ITEM  alt_item( this );
 283             WS_PROXY_UNDO_ITEM* item = (WS_PROXY_UNDO_ITEM*) eda_item;
 284             item->Restore( this );
 285             *item = alt_item;
 286             GetToolManager()->RunAction( ACTIONS::zoomFitScreen, true );
 287         }
 288         else if( dynamic_cast<SCH_ITEM*>( eda_item ) )
 289         {
 290             // everthing else is modified in place

 291 
 292             SCH_ITEM* item = (SCH_ITEM*) eda_item;
 293             SCH_ITEM* alt_item = (SCH_ITEM*) aList->GetPickedItemLink( (unsigned) ii );
 294             RemoveFromScreen( item, (SCH_SCREEN*) aList->GetScreenForItem( (unsigned) ii ) );
 295 
 296             switch( status )
 297             {
 298             case UR_CHANGED:
 299                 item->SwapData( alt_item );
 300                 break;
 301 
 302             case UR_MOVED:
 303                 item->Move( aRedoCommand ? aList->m_TransformPoint : -aList->m_TransformPoint );
 304                 break;
 305 
 306             case UR_MIRRORED_Y:
 307                 item->MirrorY( aList->m_TransformPoint.x );
 308                 break;
 309 
 310             case UR_MIRRORED_X:
 311                 item->MirrorX( aList->m_TransformPoint.y );
 312                 break;
 313 
 314             case UR_ROTATED:
 315                 if( aRedoCommand )
 316                     item->Rotate( aList->m_TransformPoint );
 317                 else
 318                 {
 319                     // Rotate 270 deg to undo 90-deg rotate

 320                     item->Rotate( aList->m_TransformPoint );
 321                     item->Rotate( aList->m_TransformPoint );
 322                     item->Rotate( aList->m_TransformPoint );
 323                 }
 324                 break;
 325 
 326             case UR_EXCHANGE_T:
 327                 aList->SetPickedItem( alt_item, (unsigned) ii );
 328                 aList->SetPickedItemLink( item, (unsigned) ii );
 329                 item = alt_item;
 330                 break;
 331 
 332             default:
 333                 wxFAIL_MSG( wxString::Format( wxT( "Unknown undo/redo command %d" ),
 334                                               aList->GetPickedItemStatus( (unsigned) ii ) ) );
 335                 break;
 336             }
 337 
 338             AddToScreen( item, (SCH_SCREEN*) aList->GetScreenForItem( (unsigned) ii ) );
 339         }
 340     }
 341 
 342     EE_SELECTION_TOOL* selTool = m_toolManager->GetTool<EE_SELECTION_TOOL>();
 343     selTool->RebuildSelection();
 344 
 345     // Bitmaps are cached in Opengl: clear the cache, because

 346     // the cache data can be invalid

 347     GetCanvas()->GetView()->RecacheAllItems();
 348     GetCanvas()->GetView()->ClearHiddenFlags();
 349 }

 352 void SCH_EDIT_FRAME::RollbackSchematicFromUndo()
 353 {
 354     PICKED_ITEMS_LIST* undo = PopCommandFromUndoList();
 355 
 356     if( undo )
 357     {
 358         PutDataInPreviousState( undo, false );
 359         undo->ClearListAndDeleteItems();
 360         delete undo;
 361 
 362         SetSheetNumberAndCount();
 363 
 364         TestDanglingEnds();
 365     }
 366 
 367     SyncView();
 368     GetCanvas()->Refresh();
 369 }

 372 void SCH_EDIT_FRAME::ClearUndoORRedoList( UNDO_REDO_LIST whichList, int aItemCount )
 373 {
 374     if( aItemCount == 0 )
 375         return;
 376 
 377     UNDO_REDO_CONTAINER& list = whichList == UNDO_LIST ? m_undoList : m_redoList;
 378 
 379     for( PICKED_ITEMS_LIST* command : list.m_CommandsList )
 380     {
 381         command->ClearListAndDeleteItems();
 382         delete command;
 383     }
 384 
 385     list.m_CommandsList.clear();
 386 }
