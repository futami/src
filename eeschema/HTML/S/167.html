  38 class DIALOG_SPICE_MODEL : public DIALOG_SPICE_MODEL_BASE
  39 {
  40 public:
  41     DIALOG_SPICE_MODEL( wxWindow* aParent, SCH_COMPONENT& aComponent, SCH_FIELDS* aSchFields );
  42     DIALOG_SPICE_MODEL( wxWindow* aParent, SCH_COMPONENT& aComponent, LIB_FIELDS* aLibFields );
  43 
  44 private:
  45     /**

  46      * Parse a string describing a power source, so appropriate settings are checked in the dialog

  47      * @param aModel contains the string to be parse (e.g. sin(0 1 10k))

  48      * @return True if the input string was parsed without errors.

  49      */
  50     bool parsePowerSource( const wxString& aModel );
  51 
  52     /**

  53      * Generates a string to describe power source parameters, basing on the current selection.

  54      * If there are missing fields, it will not modify the target string.

  55      * @param aTarget is the destination for the generated string.

  56      * @return True if the string was saved successfully.

  57      */
  58     bool generatePowerSource( wxString& aTarget ) const;
  59 
  60     /**

  61      * Loads a list of components (.model and .subckt) from a spice library

  62      * file and adds them to a combo box.

  63      * @param aComboBox is the target combo box

  64      * @param aFilePath is path to the library file

  65      */
  66     void loadLibrary( const wxString& aFilePath );
  67 
  68     /**

  69      * Returns or creates a field in the edited schematic fields vector.

  70      * @param aFieldType is an SPICE_FIELD enum value.

  71      * @return Requested field.

  72      */
  73     SCH_FIELD& getSchField( int aFieldType );
  74     LIB_FIELD& getLibField( int aFieldType );
  75 
  76     /**

  77      * Adds a value to the PWL values list.

  78      * @param aTime is the time value.

  79      * @param aValue is the source value at the given time.

  80      * @return True if request has completed successfully, false if the data is invalid.

  81      */
  82     bool addPwlValue( const wxString& aTime, const wxString& aValue );
  83 
  84     virtual bool TransferDataFromWindow() override;
  85     virtual bool TransferDataToWindow() override;
  86 
  87     // The default dialog Validate() calls the validators of all widgets.

  88     // This is not what we want; We want only validators of the selected page

  89     // of the notbooks. So disable the wxDialog::Validate(), and let our

  90     // TransferDataFromWindow doing the job.

  91     virtual bool Validate() override
  92     {
  93         return true;
  94     }
  95 
  96     virtual void onInitDlg( wxInitDialogEvent& event ) override
  97     {
  98         // Call the default wxDialog handler of a wxInitDialogEvent

  99         TransferDataToWindow();
 100 
 101         // Now all widgets have the size fixed, call FinishDialogSettings

 102         FinishDialogSettings();
 103     }

 108     void Init();

 113     void showPinOrderNote( int aModelType );

 117     void onModelSelected( wxCommandEvent& event ) override;
 118     void onPwlAdd( wxCommandEvent& event ) override;
 119     void onPwlRemove( wxCommandEvent& event ) override;
 120     void onRandomSourceType( wxCommandEvent& event ) override;
 121     void onTypeSelected( wxCommandEvent& event ) override;

 124     SCH_COMPONENT& m_component;

 129     bool m_useSchFields;
