 101 bool DIALOG_LIB_EDIT_PIN::TransferDataToWindow()
 102 {
 103     if( !DIALOG_SHIM::TransferDataToWindow() )
 104         return false;
 105 
 106     m_origPos = m_pin->GetPosition();
 107 
 108     m_choiceOrientation->SetSelection( LIB_PIN::GetOrientationIndex( m_pin->GetOrientation() ) );
 109     m_choiceStyle->SetSelection( m_pin->GetShape() );
 110     m_choiceElectricalType->SetSelection( m_pin->GetType() );
 111     m_textPinName->SetValue( m_pin->GetName() );
 112     m_nameSize.SetValue( m_pin->GetNameTextSize() );
 113     m_posX.SetValue( m_origPos.x );
 114     m_posY.SetValue( -m_origPos.y );
 115     m_textPinNumber->SetValue( m_pin->GetNumber() );
 116     m_numberSize.SetValue( m_pin->GetNumberTextSize() );
 117     m_pinLength.SetValue( m_pin->GetLength() );
 118     m_checkApplyToAllParts->SetValue( m_pin->GetUnit() == 0 );
 119     m_checkApplyToAllConversions->SetValue( m_pin->GetConvert() == 0 );
 120     m_checkShow->SetValue( m_pin->IsVisible() );
 121 
 122     m_dummyPin->SetVisible( m_pin->IsVisible() );
 123 
 124     return true;
 125 }

 128 bool DIALOG_LIB_EDIT_PIN::TransferDataFromWindow()
 129 {
 130     if( !DIALOG_SHIM::TransferDataFromWindow() )
 131         return false;
 132 
 133     wxPoint newPos( m_posX.GetValue(), -m_posY.GetValue() );
 134 
 135     const int acceptable_mingrid = 50;
 136 
 137     // Only show the warning if the position has been changed

 138     if( ( m_origPos != newPos )
 139         && ( ( m_posX.GetValue() % acceptable_mingrid ) || ( m_posY.GetValue() % acceptable_mingrid ) ) )
 140     {
 141         auto msg = wxString::Format( _( "This pin is not on a %d mils grid which will make it\n"
 142                                         "difficult to connect to in the schematic.\n"
 143                                         "Do you want to continue?" ),
 144                                      acceptable_mingrid );
 145         if( !IsOK( this, msg ) )
 146             return false;
 147     }
 148 
 149     if( m_pin->GetEditFlags() == 0 )
 150         m_frame->SaveCopyInUndoList( m_pin->GetParent() );
 151 
 152     m_pin->SetName( m_textPinName->GetValue() );
 153     m_pin->SetNumber( m_textPinNumber->GetValue() );
 154     m_pin->SetNameTextSize( m_nameSize.GetValue() );
 155     m_pin->SetNumberTextSize( m_numberSize.GetValue() );
 156     m_pin->SetOrientation( LIB_PIN::GetOrientationCode( m_choiceOrientation->GetSelection() ) );
 157     m_pin->SetLength( m_pinLength.GetValue() );
 158     m_pin->SetPinPosition( newPos );
 159     m_pin->SetType( m_choiceElectricalType->GetPinTypeSelection() );
 160     m_pin->SetShape( m_choiceStyle->GetPinShapeSelection() );
 161     m_pin->SetConversion( m_checkApplyToAllConversions->GetValue() ? 0 : m_frame->GetConvert() );
 162     m_pin->SetPartNumber( m_checkApplyToAllParts->GetValue() ? 0 : m_frame->GetUnit() );
 163     m_pin->SetVisible( m_checkShow->GetValue() );
 164 
 165     return true;
 166 }

 169 /*

 170  * Draw (on m_panelShowPin) the pin according to current settings in dialog

 171  */
 172 void DIALOG_LIB_EDIT_PIN::OnPaintShowPanel( wxPaintEvent& event )
 173 {
 174     wxPaintDC dc( m_panelShowPin );
 175     wxSize    dc_size = dc.GetSize();
 176     dc.SetDeviceOrigin( dc_size.x / 2, dc_size.y / 2 );
 177 
 178     // Give a parent to m_dummyPin only from draw purpose.

 179     // In fact m_dummyPin should not have a parent, but draw functions need a parent

 180     // to know some options, about pin texts

 181     LIB_EDIT_FRAME* libframe = (LIB_EDIT_FRAME*) GetParent();
 182     m_dummyPin->SetParent( libframe->GetCurPart() );
 183 
 184     // Calculate a suitable scale to fit the available draw area

 185     EDA_RECT bBox = m_dummyPin->GetBoundingBox( true );
 186     double   xscale = (double) dc_size.x / bBox.GetWidth();
 187     double   yscale = (double) dc_size.y / bBox.GetHeight();
 188     double   scale = std::min( xscale, yscale );
 189 
 190     // Give a 10% margin and limit to no more than 100% zoom

 191     scale = std::min( scale * 0.9, 1.0 );
 192     dc.SetUserScale( scale, scale );
 193     GRResetPenAndBrush( &dc );
 194 
 195     PART_DRAW_OPTIONS opts;
 196     opts.draw_hidden_fields = true;
 197 
 198     RENDER_SETTINGS* renderSettings = libframe->GetRenderSettings();
 199     renderSettings->SetPrintDC( &dc );
 200 
 201     m_dummyPin->Print( renderSettings, -bBox.Centre(), (void*) &opts, DefaultTransform );
 202 
 203     m_dummyPin->SetParent( nullptr );
 204 
 205     event.Skip();
 206 }

 209 void DIALOG_LIB_EDIT_PIN::OnPropertiesChange( wxCommandEvent& event )
 210 {
 211     if( !IsShown() )   // do nothing at init time

 212         return;
 213 
 214     m_dummyPin->SetName( m_textPinName->GetValue() );
 215     m_dummyPin->SetNumber( m_textPinNumber->GetValue() );
 216     m_dummyPin->SetNameTextSize( m_nameSize.GetValue() );
 217     m_dummyPin->SetNumberTextSize( m_numberSize.GetValue() );
 218     m_dummyPin->SetOrientation( LIB_PIN::GetOrientationCode( m_choiceOrientation->GetSelection() ) );
 219     m_dummyPin->SetLength( m_pinLength.GetValue() );
 220     m_dummyPin->SetType( m_choiceElectricalType->GetPinTypeSelection() );
 221     m_dummyPin->SetShape( m_choiceStyle->GetPinShapeSelection() );
 222     m_dummyPin->SetVisible( m_checkShow->GetValue() );
 223 
 224     m_panelShowPin->Refresh();
 225 }
