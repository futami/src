  29 class SCH_SCREEN;
  30 
  31 
  32 class BUS_ALIAS
  33 {
  34 public:
  35     BUS_ALIAS( SCH_SCREEN* aParent = NULL );
  36 
  37     ~BUS_ALIAS();
  38 
  39     std::shared_ptr< BUS_ALIAS > Clone() const
  40     {
  41         return std::make_shared< BUS_ALIAS >( *this );
  42     }
  43 
  44     wxString GetName()
  45     {
  46         return m_name;
  47     }
  48 
  49     void SetName( const wxString& aName )
  50     {
  51         m_name = aName;
  52     }
  53 
  54     void ClearMembers()
  55     {
  56         m_members.clear();
  57     }
  58 
  59     void AddMember( const wxString& aName )
  60     {
  61         m_members.push_back( aName );
  62     }
  63 
  64     int GetMemberCount()
  65     {
  66         return m_members.size();
  67     }
  68 
  69     wxArrayString& Members()
  70     {
  71         return m_members;
  72     }
  73 
  74     bool Contains( const wxString& aName );
  75 
  76     SCH_SCREEN* GetParent()
  77     {
  78         return m_parent;
  79     }
  80 
  81     void SetParent( SCH_SCREEN* aParent )
  82     {
  83         m_parent = aParent;
  84     }
  85 
  86 protected:
  87 
  88     wxString m_name;
  89 
  90     wxArrayString m_members;
  91 
  92     /**

  93      * The bus alias editor dialog can edit aliases from all open sheets.

  94      * This means we have to store a reference back to our parent so that

  95      * the dialog can update the parent if aliases are changed or removed.

  96      */
  97     SCH_SCREEN* m_parent;
  98 };
