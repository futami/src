 138 {
 139     KIGFX::SCH_RENDER_SETTINGS* settings = static_cast<KIGFX::SCH_RENDER_SETTINGS*>( aSettings );

 188 void LIB_PIN::SetName( const wxString& aName, bool aTestOtherPins )
 189 {
 190     wxString tmp = ( aName.IsEmpty() ) ? wxT( "~" ) : aName;
 191 
 192     // pin name string does not support spaces

 193     tmp.Replace( wxT( " " ), wxT( "_" ) );
 194 
 195     if( m_name != tmp )
 196     {
 197         m_name = tmp;
 198         SetModified();
 199     }
 200 
 201     if( !aTestOtherPins )
 202         return;
 203 
 204     if( GetParent() == NULL )
 205         return;
 206 
 207     LIB_PINS pinList;
 208     GetParent()->GetPins( pinList );
 209 
 210     for( size_t i = 0; i < pinList.size(); i++ )
 211     {
 212         if( ( pinList[i]->m_Flags & IS_LINKED ) == 0 || pinList[i]->m_name == m_name )
 213             continue;
 214 
 215         pinList[i]->m_name = m_name;
 216         SetModified();
 217     }
 218 }

 437 void LIB_PIN::SetPartNumber( int part )
 438 {
 439     if( m_Unit == part )
 440         return;
 441 
 442     m_Unit = part;
 443     SetModified();
 444 
 445     if( m_Unit == 0 )
 446     {
 447         LIB_PIN* pin;
 448         LIB_PIN* tmp = GetParent()->GetNextPin();
 449 
 450         while( tmp != NULL )
 451         {
 452             pin = tmp;
 453             tmp = GetParent()->GetNextPin( pin );
 454 
 455             if( pin->m_Flags == 0 || pin == this
 456                || ( m_Convert && ( m_Convert != pin->m_Convert ) )
 457                || ( m_position != pin->m_position )
 458                || ( pin->m_orientation != m_orientation ) )
 459                 continue;
 460 
 461             GetParent()->RemoveDrawItem( (LIB_ITEM*) pin );
 462         }
 463     }

 531 void LIB_PIN::EnableEditMode( bool aEnable, bool aEditPinByPin )
 532 {
 533     LIB_PINS pinList;
 534 
 535     if( GetParent() == NULL )
 536         return;
 537 
 538     GetParent()->GetPins( pinList );
 539 
 540     for( size_t i = 0; i < pinList.size(); i++ )

 542         if( pinList[i] == this )
 543             continue;
 544 
 545         if( ( pinList[i]->m_position == m_position )

 547             && !IsNew() && !aEditPinByPin && aEnable )
 548         {
 549             pinList[i]->SetFlags( IS_LINKED | IN_EDIT );
 550         }
 551         else
 552             pinList[i]->ClearFlags( IS_LINKED | IN_EDIT );
 553     }
 554 }

 616 void LIB_PIN::PrintPinSymbol( RENDER_SETTINGS* aSettings, const wxPoint& aPos, int aOrient )
 617 {
 618     wxDC*   DC = aSettings->GetPrintDC();
 619     int     MapX1, MapY1, x1, y1;
 620     int     width = std::max( GetPenWidth(), aSettings->GetDefaultPenWidth() );
 621     int     posX = aPos.x, posY = aPos.y, len = m_length;
 622     COLOR4D color = aSettings->GetLayerColor( IsVisible() ? LAYER_PIN : LAYER_HIDDEN );
 623 
 624     MapX1 = MapY1 = 0;
 625     x1    = posX;
 626     y1    = posY;
 627 
 628     switch( aOrient )
 629     {
 630     case PIN_UP:     y1 = posY - len;  MapY1 = 1;   break;
 631     case PIN_DOWN:   y1 = posY + len;  MapY1 = -1;  break;
 632     case PIN_LEFT:   x1 = posX - len;  MapX1 = 1;   break;
 633     case PIN_RIGHT:  x1 = posX + len;  MapX1 = -1;  break;
 634     }
 635 
 636     if( m_shape == GRAPHIC_PINSHAPE::INVERTED || m_shape == GRAPHIC_PINSHAPE::INVERTED_CLOCK )
 637     {
 638         const int radius = externalPinDecoSize( aSettings, *this );
 639         GRCircle( nullptr, DC, MapX1 * radius + x1, MapY1 * radius + y1, radius, width, color );
 640 
 641         GRMoveTo( MapX1 * radius * 2 + x1, MapY1 * radius * 2 + y1 );
 642         GRLineTo( nullptr, DC, posX, posY, width, color );
 643     }
 644     else
 645     {
 646         GRMoveTo( x1, y1 );
 647         GRLineTo( nullptr, DC, posX, posY, width, color );
 648     }
 649 
 650     // Draw the clock shape (>)inside the symbol

 651     if( m_shape == GRAPHIC_PINSHAPE::CLOCK || m_shape == GRAPHIC_PINSHAPE::INVERTED_CLOCK
 652             || m_shape == GRAPHIC_PINSHAPE::FALLING_EDGE_CLOCK
 653             || m_shape == GRAPHIC_PINSHAPE::CLOCK_LOW )
 654     {
 655         const int clock_size = internalPinDecoSize( aSettings, *this );
 656         if( MapY1 == 0 ) /* MapX1 = +- 1 */
 657         {
 658             GRMoveTo( x1, y1 + clock_size );
 659             GRLineTo( nullptr, DC, x1 - MapX1 * clock_size * 2, y1, width, color );
 660             GRLineTo( nullptr, DC, x1, y1 - clock_size, width, color );
 661         }
 662         else    /* MapX1 = 0 */
 663         {
 664             GRMoveTo( x1 + clock_size, y1 );
 665             GRLineTo( nullptr, DC, x1, y1 - MapY1 * clock_size * 2, width, color );
 666             GRLineTo( nullptr, DC, x1 - clock_size, y1, width, color );
 667         }
 668     }
 669 
 670     // Draw the active low (or H to L active transition)

 671     if( m_shape == GRAPHIC_PINSHAPE::INPUT_LOW || m_shape == GRAPHIC_PINSHAPE::FALLING_EDGE_CLOCK
 672             || m_shape == GRAPHIC_PINSHAPE::CLOCK_LOW )
 673     {
 674         const int deco_size = externalPinDecoSize( aSettings, *this );
 675         if( MapY1 == 0 )            /* MapX1 = +- 1 */
 676         {
 677             GRMoveTo( x1 + MapX1 * deco_size * 2, y1 );
 678             GRLineTo( nullptr, DC, x1 + MapX1 * deco_size * 2, y1 - deco_size * 2, width, color );
 679             GRLineTo( nullptr, DC, x1, y1, width, color );
 680         }
 681         else    /* MapX1 = 0 */
 682         {

1250 {

1252     wxPoint end;   // position of pin end starting at 0,0 according to its orientation, length = 1

1255     {

1268         if( end.y > 0 )
1269             orient = PIN_DOWN;

1271     else
1272     {
1273         orient = PIN_RIGHT;
1274 
1275         if( end.x < 0 )

1289 int LIB_PIN::compare( const LIB_ITEM& aOther, LIB_ITEM::COMPARE_FLAGS aCompareFlags ) const
1290 {
1291     wxASSERT( aOther.Type() == LIB_PIN_T );
1292 
1293     int retv = LIB_ITEM::compare( aOther, aCompareFlags );
1294 
1295     if( retv )
1296         return retv;
1297 
1298     const LIB_PIN* tmp = (LIB_PIN*) &aOther;
1299 
1300     // When comparing units, we do not compare the part numbers.  If everything else is

1301     // identical, then we can just renumber the parts for the inherited symbol.

1302     if( !( aCompareFlags & COMPARE_FLAGS::UNIT ) && m_number != tmp->m_number )
1303         return m_number.Cmp( tmp->m_number );
1304 
1305     int result = m_name.CmpNoCase( tmp->m_name );
1306 
1307     if( result )
1308         return result;
1309 
1310     if( m_position.x != tmp->m_position.x )
1311         return m_position.x - tmp->m_position.x;
1312 
1313     if( m_position.y != tmp->m_position.y )
1314         return m_position.y - tmp->m_position.y;
1315 
1316     if( m_length != tmp->m_length )
1317         return m_length - tmp->m_length;
1318 
1319     if( m_orientation != tmp->m_orientation )
1320         return m_orientation - tmp->m_orientation;
1321 
1322     if( m_shape != tmp->m_shape )
1323         return static_cast<int>( m_shape ) - static_cast<int>( tmp->m_shape );
1324 
1325     if( m_type != tmp->m_type )
1326         return static_cast<int>( m_type ) - static_cast<int>( tmp->m_type );
1327 
1328     if( m_attributes != tmp->m_attributes )
1329         return m_attributes - tmp->m_attributes;
1330 
1331     if( m_width != tmp->m_width )
1332         return m_width - tmp->m_width;
1333 
1334     if( m_numTextSize != tmp->m_numTextSize )
1335         return m_numTextSize - tmp->m_numTextSize;
1336 
1337     if( m_nameTextSize != tmp->m_nameTextSize )
1338         return m_nameTextSize - tmp->m_nameTextSize;
1339 
1340     return 0;
1341 }

1518 const EDA_RECT LIB_PIN::GetBoundingBox( bool aIncludeInvisibles, bool aPinOnly ) const
1519 {
1520     EDA_RECT       bbox;
1521     wxPoint        begin;
1522     wxPoint        end;
1523     int            nameTextOffset = 0;
1524     bool           showName = !m_name.IsEmpty() && (m_name != wxT( "~" ));
1525     bool           showNum = !m_number.IsEmpty();
1526     int            minsizeV = TARGET_PIN_RADIUS;
1527 
1528     if( !aIncludeInvisibles && !IsVisible() )
1529         showName = false;
1530 
1531     if( GetParent() )
1532     {
1533         if( GetParent()->ShowPinNames() )
1534             nameTextOffset = GetParent()->GetPinNameOffset();
1535         else
1536             showName = false;
1537 
1538         if( !GetParent()->ShowPinNumbers() )
1539             showNum = false;
1540     }
1541 
1542     if( aPinOnly )
1543     {
1544         showName = false;
1545         showNum = false;
1546     }
1547 
1548     // First, calculate boundary box corners position

1549     int numberTextLength = showNum ? m_numTextSize * m_number.Len() : 0;
1550 
1551     // Actual text height is bigger than text size

1552     int numberTextHeight  = showNum ? KiROUND( m_numTextSize * 1.1 ) : 0;
1553 
1554     if( m_shape == GRAPHIC_PINSHAPE::INVERTED || m_shape == GRAPHIC_PINSHAPE::INVERTED_CLOCK )
1555         minsizeV = std::max( TARGET_PIN_RADIUS, externalPinDecoSize( nullptr, *this ) );
1556 
1557     // calculate top left corner position

1558     // for the default pin orientation (PIN_RIGHT)

1559     begin.y = std::max( minsizeV, numberTextHeight + Mils2iu( PIN_TEXT_MARGIN ) );
1560     begin.x = std::min( -TARGET_PIN_RADIUS, m_length - (numberTextLength / 2) );
1561 
1562     // calculate bottom right corner position and adjust top left corner position

1563     int nameTextLength = 0;
1564     int nameTextHeight = 0;
1565 
1566     if( showName )
1567     {
1568         int length = m_name.Len();
1569 
1570         // Don't count the line over text symbol.

1571         if( m_name.Left( 1 ) == wxT( "~" ) )
1572             length -= 1;
1573 
1574         nameTextLength = ( m_nameTextSize * length ) + nameTextOffset;
1575 
1576         // Actual text height are bigger than text size

1577         nameTextHeight = KiROUND( m_nameTextSize * 1.1 ) + Mils2iu( PIN_TEXT_MARGIN );
1578     }
1579 
1580     if( nameTextOffset )        // for values > 0, pin name is inside the body

1581     {
1582         end.x = m_length + nameTextLength + TARGET_PIN_RADIUS;
1583         end.y = std::min( -minsizeV, -nameTextHeight / 2 );
1584     }
1585     else        // if value == 0:

1586                 // pin name is outside the body, and above the pin line

1587                 // pin num is below the pin line

1588     {
1589         end.x   = std::max( m_length + TARGET_PIN_RADIUS, nameTextLength );
1590         end.y   = -begin.y;
1591         begin.y = std::max( minsizeV, nameTextHeight );
1592     }
1593 
1594     // Now, calculate boundary box corners position for the actual pin orientation

1595     int orient = PinDrawOrient( DefaultTransform );
1596 
1597     /* Calculate the pin position */
1598     switch( orient )
1599     {
1600     case PIN_UP:
1601         // Pin is rotated and texts positions are mirrored

1602         RotatePoint( &begin, wxPoint( 0, 0 ), -900 );
1603         RotatePoint( &end, wxPoint( 0, 0 ), -900 );
1604         break;
1605 
1606     case PIN_DOWN:
1607         RotatePoint( &begin, wxPoint( 0, 0 ), 900 );
1608         RotatePoint( &end, wxPoint( 0, 0 ), 900 );
1609         begin.x = -begin.x;
1610         end.x = -end.x;
1611         break;
1612 
1613     case PIN_LEFT:
1614         begin.x = -begin.x;
1615         end.x = -end.x;
1616         break;
1617 
1618     case PIN_RIGHT:
1619         break;
1620     }
1621 
1622     begin += m_position;
1623     end += m_position;
1624 
1625     bbox.SetOrigin( begin );
1626     bbox.SetEnd( end );
1627     bbox.Normalize();
1628     bbox.Inflate( ( GetPenWidth() / 2 ) + 1 );
1629 
1630     // Draw Y axis is reversed in schematic:

1631     bbox.RevertYAxis();
1632 
1633     return bbox;
1634 }
