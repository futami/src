  83 DIALOG_ANNOTATE::DIALOG_ANNOTATE( SCH_EDIT_FRAME* parent, const wxString& message )
  84     : DIALOG_ANNOTATE_BASE( parent )
  85 {
  86     SetName( DLG_WINDOW_NAME );
  87     m_Parent = parent;
  88 
  89     if( !message.IsEmpty() )
  90     {
  91         m_userMessage->SetLabelText( message );
  92         m_userMessage->Show( true );
  93 
  94         m_rbScope->Enable( false );
  95     }
  96 
  97     m_MessageWindow->SetLabel( _( "Annotation Messages:" ) );
  98 
  99     // We use a sdbSizer to get platform-dependent ordering of the action buttons, but

 100     // that requires us to correct the button labels here.

 101     m_sdbSizer1OK->SetLabel( _( "Annotate" ) );
 102     m_sdbSizer1Cancel->SetLabel( _( "Close" ) );
 103     m_sdbSizer1->Layout();
 104 
 105     m_sdbSizer1OK->SetDefault();
 106 
 107     InitValues();
 108     Layout();
 109 
 110     // When all widgets have the size fixed, call FinishDialogSettings

 111     FinishDialogSettings();

 125 void DIALOG_ANNOTATE::InitValues()
 126 {
 127     EESCHEMA_SETTINGS* cfg = static_cast<EESCHEMA_SETTINGS*>( Kiface().KifaceSettings() );
 128     int option;
 129 
 130     // These are always reset to attempt to keep the user out of trouble...

 131     m_rbScope->SetSelection( 0 );
 132     m_rbOptions->SetSelection( 0 );
 133 
 134     option = cfg->m_AnnotatePanel.sort_order;
 135 
 136     switch( option )
 137     {
 138     default:
 139     case 0: m_rbSortBy_X_Position->SetValue( true ); break;
 140     case 1: m_rbSortBy_Y_Position->SetValue( true ); break;
 141     }
 142 
 143     option = cfg->m_AnnotatePanel.method;
 144 
 145     switch( option )
 146     {
 147     default:
 148     case 0: m_rbFirstFree->SetValue( true );  break;
 149     case 1: m_rbSheetX100->SetValue( true );  break;
 150     case 2: m_rbSheetX1000->SetValue( true ); break;
 151     }
 152 
 153     m_textNumberAfter->SetValue( wxT( "0" ) );
 154 
 155     annotate_down_right_bitmap->SetBitmap( KiBitmap( annotate_down_right_xpm ) );
 156     annotate_right_down_bitmap->SetBitmap( KiBitmap( annotate_right_down_xpm ) );
 157 
 158     m_MessageWindow->SetVisibleSeverities( cfg->m_AnnotatePanel.messages_filter );
 159 
 160     m_MessageWindow->MsgPanelSetMinSize( wxSize( -1, 160 ) );
 161 }

 177 void DIALOG_ANNOTATE::OnApplyClick( wxCommandEvent& event )
 178 {
 179     m_MessageWindow->Clear();
 180     REPORTER& reporter = m_MessageWindow->Reporter();
 181     m_MessageWindow->SetLazyUpdate( true );     // Don't update after each message

 182 
 183     m_Parent->AnnotateComponents( GetLevel(), (ANNOTATE_ORDER_T) GetSortOrder(),
 184                                   (ANNOTATE_OPTION_T) GetAnnotateAlgo(), GetStartNumber(),
 185                                   GetResetItems() , true, GetLockUnits(), reporter );
 186 
 187     m_MessageWindow->Flush( true );             // Now update to show all messages

 188 
 189     m_Parent->GetCanvas()->Refresh();
 190 
 191     m_btnClear->Enable();
 192 
 193     // Don't close dialog if there are things the user needs to address

 194     if( reporter.HasMessage() )
 195         return;
 196 
 197     if( m_userMessage->IsShown() )
 198     {
 199         // Close the dialog by calling the default handler for a wxID_OK event

 200         event.SetId( wxID_OK );
 201         event.Skip();
 202     }
 203 }

 242 int DIALOG_ANNOTATE::GetAnnotateAlgo()
 243 {
 244     if( m_rbSheetX100->GetValue() )
 245         return 1;
 246     else if( m_rbSheetX1000->GetValue() )
 247         return 2;
 248     else
 249         return 0;

 259 void SCH_EDIT_FRAME::OnAnnotate( wxCommandEvent& event )
 260 {
 261     DIALOG_ANNOTATE* dlg = static_cast<DIALOG_ANNOTATE*> ( wxWindow::FindWindowByName( DLG_WINDOW_NAME ) );
 262 
 263     if( !dlg )
 264     {
 265         dlg = new DIALOG_ANNOTATE( this, wxEmptyString );
 266         dlg->Show( true );
 267     }
 268     else    // The dialog is already opened, perhaps not visible

 269     {
 270         dlg->Show( true );
 271     }
 272 }
