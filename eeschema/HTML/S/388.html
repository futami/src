 423 SCH_SHEET* SCH_SEXPR_PLUGIN::Load( const wxString& aFileName, SCHEMATIC* aSchematic,
 424                                    SCH_SHEET* aAppendToMe, const PROPERTIES* aProperties )
 425 {
 426     wxASSERT( !aFileName || aSchematic != nullptr );
 427 
 428     LOCALE_IO   toggle;     // toggles on, then off, the C locale.

 429     SCH_SHEET*  sheet;
 430 
 431     wxFileName fn = aFileName;
 432 
 433     // Unfortunately child sheet file names the legacy schematic file format are not fully

 434     // qualified and are always appended to the project path.  The aFileName attribute must

 435     // always be an absolute path so the project path can be used for load child sheet files.

 436     wxASSERT( fn.IsAbsolute() );
 437 
 438     if( aAppendToMe )
 439     {
 440         wxLogTrace( traceSchLegacyPlugin, "Append \"%s\" to sheet \"%s\".",
 441                     aFileName, aAppendToMe->GetFileName() );
 442 
 443         wxFileName normedFn = aAppendToMe->GetFileName();
 444 
 445         if( !normedFn.IsAbsolute() )
 446         {
 447             if( aFileName.Right( normedFn.GetFullPath().Length() ) == normedFn.GetFullPath() )
 448                 m_path = aFileName.Left( aFileName.Length() - normedFn.GetFullPath().Length() );
 449         }
 450 
 451         if( m_path.IsEmpty() )
 452             m_path = aSchematic->Prj().GetProjectPath();
 453 
 454         wxLogTrace( traceSchLegacyPlugin, "Normalized append path \"%s\".", m_path );
 455     }
 456     else
 457     {
 458         m_path = aSchematic->Prj().GetProjectPath();
 459     }
 460 
 461     m_currentPath.push( m_path );
 462     init( aSchematic, aProperties );
 463 
 464     if( aAppendToMe == NULL )
 465     {
 466         // Clean up any allocated memory if an exception occurs loading the schematic.

 467         std::unique_ptr< SCH_SHEET > newSheet( new SCH_SHEET( aSchematic ) );
 468         newSheet->SetFileName( aFileName );
 469         m_rootSheet = newSheet.get();
 470         loadHierarchy( newSheet.get() );
 471 
 472         // If we got here, the schematic loaded successfully.

 473         sheet = newSheet.release();
 474         m_rootSheet = nullptr;         // Quiet Coverity warning.

 475     }
 476     else
 477     {
 478         wxCHECK_MSG( aSchematic->IsValid(), nullptr, "Can't append to a schematic with no root!" );
 479         m_rootSheet = &aSchematic->Root();
 480         sheet = aAppendToMe;
 481         loadHierarchy( sheet );
 482     }
 483 
 484     wxASSERT( m_currentPath.size() == 1 );  // only the project path should remain

 485 
 486     return sheet;
 487 }

 492 void SCH_SEXPR_PLUGIN::loadHierarchy( SCH_SHEET* aSheet )
 493 {
 494     SCH_SCREEN* screen = NULL;
 495 
 496     if( !aSheet->GetScreen() )
 497     {
 498         // SCH_SCREEN objects store the full path and file name where the SCH_SHEET object only

 499         // stores the file name and extension.  Add the project path to the file name and

 500         // extension to compare when calling SCH_SHEET::SearchHierarchy().

 501         wxFileName fileName = aSheet->GetFileName();
 502 
 503         if( !fileName.IsAbsolute() )
 504             fileName.MakeAbsolute( m_currentPath.top() );
 505 
 506         // Save the current path so that it gets restored when decending and ascending the

 507         // sheet hierarchy which allows for sheet schematic files to be nested in folders

 508         // relative to the last path a schematic was loaded from.

 509         wxLogTrace( traceSchLegacyPlugin, "Saving path    \"%s\"", m_currentPath.top() );
 510         m_currentPath.push( fileName.GetPath() );
 511         wxLogTrace( traceSchLegacyPlugin, "Current path   \"%s\"", m_currentPath.top() );
 512         wxLogTrace( traceSchLegacyPlugin, "Loading        \"%s\"", fileName.GetFullPath() );
 513 
 514         m_rootSheet->SearchHierarchy( fileName.GetFullPath(), &screen );
 515 
 516         if( screen )
 517         {
 518             aSheet->SetScreen( screen );
 519             aSheet->GetScreen()->SetParent( m_schematic );
 520             // Do not need to load the sub-sheets - this has already been done.

 521         }
 522         else
 523         {
 524             aSheet->SetScreen( new SCH_SCREEN( m_schematic ) );
 525             aSheet->GetScreen()->SetFileName( fileName.GetFullPath() );
 526 
 527             try
 528             {
 529                 loadFile( fileName.GetFullPath(), aSheet );
 530             }
 531             catch( const IO_ERROR& ioe )
 532             {
 533                 // If there is a problem loading the root sheet, there is no recovery.

 534                 if( aSheet == m_rootSheet )
 535                     throw( ioe );
 536 
 537                 // For all subsheets, queue up the error message for the caller.

 538                 if( !m_error.IsEmpty() )
 539                     m_error += "\n";
 540 
 541                 m_error += ioe.What();
 542             }
 543 
 544             // This was moved out of the try{} block so that any sheets definitionsthat

 545             // the plugin fully parsed before the exception was raised will be loaded.

 546             for( auto aItem : aSheet->GetScreen()->Items().OfType( SCH_SHEET_T ) )
 547             {
 548                 wxCHECK2( aItem->Type() == SCH_SHEET_T, /* do nothing */ );
 549                 auto sheet = static_cast<SCH_SHEET*>( aItem );
 550 
 551                 // Recursion starts here.

 552                 loadHierarchy( sheet );
 553             }
 554         }
 555 
 556         m_currentPath.pop();
 557         wxLogTrace( traceSchLegacyPlugin, "Restoring path \"%s\"", m_currentPath.top() );
 558     }
 559 }

1327 LIB_PART* SCH_SEXPR_PLUGIN_CACHE::removeSymbol( LIB_PART* aPart )
1328 {
1329     wxCHECK_MSG( aPart != NULL, NULL, "NULL pointer cannot be removed from library." );
1330 
1331     LIB_PART* firstChild = NULL;
1332     LIB_PART_MAP::iterator it = m_symbols.find( aPart->GetName() );
1333 
1334     if( it == m_symbols.end() )
1335         return NULL;
1336 
1337     // If the entry pointer doesn't match the name it is mapped to in the library, we

1338     // have done something terribly wrong.

1339     wxCHECK_MSG( *it->second == aPart, NULL,
1340                  "Pointer mismatch while attempting to remove alias entry <" + aPart->GetName() +
1341                  "> from library cache <" + m_libFileName.GetName() + ">." );
1342 
1343     // If the symbol is a root symbol used by other symbols find the first alias that uses

1344     // the root part and make it the new root.

1345     if( aPart->IsRoot() )
1346     {
1347         for( auto entry : m_symbols )
1348         {
1349             if( entry.second->IsAlias()
1350               && entry.second->GetParent().lock() == aPart->SharedPtr() )
1351             {
1352                 firstChild = entry.second;
1353                 break;
1354             }
1355         }
1356 
1357         if( firstChild )
1358         {
1359             for( LIB_ITEM& drawItem : aPart->GetDrawItems() )
1360             {
1361                 if( drawItem.Type() == LIB_FIELD_T )
1362                 {
1363                     LIB_FIELD& field = static_cast<LIB_FIELD&>( drawItem );
1364 
1365                     if( firstChild->FindField( field.GetCanonicalName() ) )
1366                         continue;
1367                 }
1368 
1369                 LIB_ITEM* newItem = (LIB_ITEM*) drawItem.Clone();
1370                 drawItem.SetParent( firstChild );
1371                 firstChild->AddDrawItem( newItem );
1372             }
1373 
1374             // Reparent the remaining aliases.

1375             for( auto entry : m_symbols )
1376             {
1377                 if( entry.second->IsAlias()
1378                   && entry.second->GetParent().lock() == aPart->SharedPtr() )
1379                     entry.second->SetParent( firstChild );
1380             }
1381         }
1382     }
1383 
1384     m_symbols.erase( it );
1385     delete aPart;
1386     m_isModified = true;
1387     ++m_modHash;
1388     return firstChild;
1389 }

1936 void SCH_SEXPR_PLUGIN_CACHE::DeleteSymbol( const wxString& aSymbolName )
1937 {
1938     LIB_PART_MAP::iterator it = m_symbols.find( aSymbolName );
1939 
1940     if( it == m_symbols.end() )
1941         THROW_IO_ERROR( wxString::Format( _( "library %s does not contain a symbol named %s" ),
1942                                           m_libFileName.GetFullName(), aSymbolName ) );
1943 
1944     LIB_PART* part = it->second;
1945 
1946     if( part->IsRoot() )
1947     {
1948         LIB_PART* rootPart = part;
1949 
1950         // Remove the root symbol and all it's children.

1951         m_symbols.erase( it );
1952 
1953         LIB_PART_MAP::iterator it1 = m_symbols.begin();
1954 
1955         while( it1 != m_symbols.end() )
1956         {
1957             if( it1->second->IsAlias() && it1->second->GetParent().lock() == rootPart->SharedPtr() )
1958             {
1959                 delete it1->second;
1960                 it1 = m_symbols.erase( it1 );
1961             }
1962             else
1963             {
1964                 it1++;
1965             }
1966         }
1967 
1968         delete rootPart;
1969     }
1970     else
1971     {
1972         // Just remove the alias.

1973         m_symbols.erase( it );
1974         delete part;
1975     }
1976 
1977     ++m_modHash;
1978     m_isModified = true;
1979 }
