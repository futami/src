 141 bool EE_SELECTION_TOOL::Init()
 142 {
 143     m_frame = getEditFrame<SCH_BASE_FRAME>();
 144 
 145     LIB_VIEW_FRAME* libViewFrame = dynamic_cast<LIB_VIEW_FRAME*>( m_frame );
 146     LIB_EDIT_FRAME* libEditFrame = dynamic_cast<LIB_EDIT_FRAME*>( m_frame );
 147 
 148     if( libEditFrame )
 149     {
 150         m_isLibEdit = true;
 151         m_unit = libEditFrame->GetUnit();
 152         m_convert = libEditFrame->GetConvert();
 153     }
 154     else
 155         m_isLibView = libViewFrame != nullptr;
 156 
 157 
 158     static KICAD_T wireOrBusTypes[] = { SCH_LINE_LOCATE_WIRE_T, SCH_LINE_LOCATE_BUS_T, EOT };
 159     static KICAD_T connectedTypes[] = { SCH_LINE_LOCATE_WIRE_T, SCH_LINE_LOCATE_BUS_T,
 160                                         SCH_GLOBAL_LABEL_T, SCH_HIER_LABEL_T, SCH_LABEL_T,
 161                                         SCH_SHEET_PIN_T, SCH_PIN_T, EOT };
 162 
 163     auto wireSelection =      E_C::MoreThan( 0 ) && E_C::OnlyType( SCH_LINE_LOCATE_WIRE_T );
 164     auto busSelection =       E_C::MoreThan( 0 ) && E_C::OnlyType( SCH_LINE_LOCATE_BUS_T );
 165     auto wireOrBusSelection = E_C::MoreThan( 0 ) && E_C::OnlyTypes( wireOrBusTypes );
 166     auto connectedSelection = E_C::MoreThan( 0 ) && E_C::OnlyTypes( connectedTypes );
 167     auto sheetSelection =     E_C::Count( 1 )    && E_C::OnlyType( SCH_SHEET_T );
 168 
 169     auto schEditCondition =
 170             [this] ( const SELECTION& aSel )
 171             {
 172 
 173                 return !m_isLibEdit && !m_isLibView;
 174             };
 175 
 176     auto belowRootSheetCondition =
 177             [&]( const SELECTION& aSel )
 178             {
 179                 SCH_EDIT_FRAME* schEditFrame = dynamic_cast<SCH_EDIT_FRAME*>( m_frame );
 180 
 181                 return ( schEditFrame&&
 182                          schEditFrame->GetCurrentSheet().Last() !=
 183                          &schEditFrame->Schematic().Root() );
 184             };
 185 
 186     auto havePartCondition =
 187             [&]( const SELECTION& sel )
 188             {
 189                 return m_isLibEdit && static_cast<LIB_EDIT_FRAME*>( m_frame )->GetCurPart();
 190             };
 191 
 192     auto& menu = m_menu.GetMenu();
 193 
 194     menu.AddItem( EE_ACTIONS::enterSheet,         sheetSelection && EE_CONDITIONS::Idle, 1 );
 195     menu.AddItem( EE_ACTIONS::explicitCrossProbe, sheetSelection && EE_CONDITIONS::Idle, 1 );
 196     menu.AddItem( EE_ACTIONS::leaveSheet,         belowRootSheetCondition, 1 );
 197 
 198     menu.AddSeparator( 100 );
 199     menu.AddItem( EE_ACTIONS::drawWire,           schEditCondition && EE_CONDITIONS::Empty, 100 );
 200     menu.AddItem( EE_ACTIONS::drawBus,            schEditCondition && EE_CONDITIONS::Empty, 100 );
 201 
 202     menu.AddSeparator( 100 );
 203     menu.AddItem( EE_ACTIONS::finishWire,         SCH_LINE_WIRE_BUS_TOOL::IsDrawingWire, 100 );
 204 
 205     menu.AddSeparator( 100 );
 206     menu.AddItem( EE_ACTIONS::finishBus,          SCH_LINE_WIRE_BUS_TOOL::IsDrawingBus, 100 );
 207 
 208     menu.AddSeparator( 200 );
 209     menu.AddItem( EE_ACTIONS::selectConnection,   wireOrBusSelection && EE_CONDITIONS::Idle, 250 );
 210     menu.AddItem( EE_ACTIONS::placeJunction,      wireOrBusSelection && EE_CONDITIONS::Idle, 250 );
 211     menu.AddItem( EE_ACTIONS::placeLabel,         wireOrBusSelection && EE_CONDITIONS::Idle, 250 );
 212     menu.AddItem( EE_ACTIONS::placeGlobalLabel,   wireOrBusSelection && EE_CONDITIONS::Idle, 250 );
 213     menu.AddItem( EE_ACTIONS::placeHierLabel,     wireOrBusSelection && EE_CONDITIONS::Idle, 250 );
 214     menu.AddItem( EE_ACTIONS::breakWire,          wireSelection && EE_CONDITIONS::Idle, 250 );
 215     menu.AddItem( EE_ACTIONS::breakBus,           busSelection && EE_CONDITIONS::Idle, 250 );
 216     menu.AddItem( EE_ACTIONS::importSheetPin,     sheetSelection && EE_CONDITIONS::Idle, 250 );
 217     menu.AddItem( EE_ACTIONS::assignNetclass,     connectedSelection && EE_CONDITIONS::Idle, 250 );
 218 
 219     menu.AddSeparator( 400 );
 220     menu.AddItem( EE_ACTIONS::symbolProperties,   havePartCondition && EE_CONDITIONS::Empty, 400 );
 221     menu.AddItem( EE_ACTIONS::pinTable,           havePartCondition && EE_CONDITIONS::Empty, 400 );
 222 
 223     menu.AddSeparator( 1000 );
 224     m_frame->AddStandardSubMenus( m_menu );
 225 
 226     return true;
 227 }

 230 void EE_SELECTION_TOOL::Reset( RESET_REASON aReason )
 231 {
 232     m_frame = getEditFrame<SCH_BASE_FRAME>();
 233 
 234     if( aReason == TOOL_BASE::MODEL_RELOAD )
 235     {
 236         // Remove pointers to the selected items from containers without changing their

 237         // properties (as they are already deleted while a new sheet is loaded)

 238         m_selection.Clear();
 239         getView()->GetPainter()->GetSettings()->SetHighlight( false );
 240 
 241         LIB_EDIT_FRAME* libEditFrame = dynamic_cast<LIB_EDIT_FRAME*>( m_frame );
 242         LIB_VIEW_FRAME* libViewFrame = dynamic_cast<LIB_VIEW_FRAME*>( m_frame );
 243 
 244         if( libEditFrame )
 245         {
 246             m_isLibEdit = true;
 247             m_unit = libEditFrame->GetUnit();
 248             m_convert = libEditFrame->GetConvert();
 249         }
 250         else
 251             m_isLibView = libViewFrame != nullptr;
 252     }
 253     else
 254         // Restore previous properties of selected items and remove them from containers

 255         ClearSelection();
 256 
 257     // Reinsert the VIEW_GROUP, in case it was removed from the VIEW

 258     getView()->Remove( &m_selection );
 259     getView()->Add( &m_selection );
 260 }

 313 int EE_SELECTION_TOOL::Main( const TOOL_EVENT& aEvent )
 314 {
 315 
 316     // Main loop: keep receiving events

 317     while( TOOL_EVENT* evt = Wait() )
 318     {
 319         if( m_frame->ToolStackIsEmpty() )
 320             m_frame->GetCanvas()->SetCurrentCursor( wxCURSOR_ARROW );
 321 
 322         m_additive = m_subtractive = m_exclusive_or = false;
 323 
 324         if( evt->Modifier( MD_SHIFT ) && evt->Modifier( MD_CTRL ) )
 325             m_subtractive = true;
 326         else if( evt->Modifier( MD_SHIFT ) )
 327             m_additive = true;
 328         else if( evt->Modifier( MD_CTRL ) )
 329             m_exclusive_or = true;
 330 
 331         // Is the user requesting that the selection list include all possible

 332         // items without removing less likely selection candidates

 333         m_skip_heuristics = !!evt->Modifier( MD_ALT );
 334 
 335         // Single click? Select single object

 336         if( evt->IsClick( BUT_LEFT ) )
 337         {
 338             if( auto schframe = dynamic_cast<SCH_EDIT_FRAME*>( m_frame ) )
 339                 schframe->FocusOnItem( nullptr );
 340 
 341             SelectPoint( evt->Position(), EE_COLLECTOR::AllItems, nullptr, nullptr, false,
 342                          m_additive, m_subtractive, m_exclusive_or );
 343         }
 344 
 345         // right click? if there is any object - show the context menu

 346         else if( evt->IsClick( BUT_RIGHT ) )
 347         {
 348             bool selectionCancelled = false;
 349 
 350             if( m_selection.Empty() ||
 351                     !m_selection.GetBoundingBox().Contains( wxPoint( evt->Position() ) ) )
 352             {
 353                 ClearSelection();
 354                 SelectPoint( evt->Position(), EE_COLLECTOR::AllItems, nullptr,
 355                              &selectionCancelled );
 356                 m_selection.SetIsHover( true );
 357             }
 358 
 359             if( !selectionCancelled )
 360                 m_menu.ShowContextMenu( m_selection );
 361         }
 362 
 363         // double click? Display the properties window

 364         else if( evt->IsDblClick( BUT_LEFT ) )
 365         {
 366             if( auto schframe = dynamic_cast<SCH_EDIT_FRAME*>( m_frame ) )
 367                 schframe->FocusOnItem( nullptr );
 368 
 369             if( m_selection.Empty() )
 370                 SelectPoint( evt->Position());
 371 
 372             EDA_ITEM* item = m_selection.Front();
 373 
 374             if( item && item->Type() == SCH_SHEET_T )
 375                 m_toolMgr->RunAction( EE_ACTIONS::enterSheet );
 376             else
 377                 m_toolMgr->RunAction( EE_ACTIONS::properties );
 378         }
 379 
 380         // drag with LMB? Select multiple objects (or at least draw a selection box) or drag them

 381         else if( evt->IsDrag( BUT_LEFT ) )
 382         {
 383             if( auto schframe = dynamic_cast<SCH_EDIT_FRAME*>( m_frame ) )
 384                 schframe->FocusOnItem( nullptr );
 385 
 386             if( m_additive || m_subtractive || m_exclusive_or || m_frame->GetDragSelects() )
 387             {
 388                 selectMultiple();
 389             }
 390             else
 391             {
 392                 // selection is empty? try to start dragging the item under the point where drag

 393                 // started

 394                 if( m_isLibEdit && m_selection.Empty() )
 395                     m_selection = RequestSelection( movableSymbolItems );
 396                 else if( m_selection.Empty() )
 397                     m_selection = RequestSelection( movableSchematicItems );
 398 
 399                 // Check if dragging has started within any of selected items bounding box

 400                 if( selectionContains( evt->Position() ) )
 401                 {
 402                     // Yes -> run the move tool and wait till it finishes

 403                     if( m_isLibEdit )
 404                         m_toolMgr->InvokeTool( "eeschema.SymbolMoveTool" );
 405                     else
 406                         m_toolMgr->InvokeTool( "eeschema.InteractiveMove" );
 407                 }
 408                 else
 409                 {
 410                     // No -> drag a selection box

 411                     selectMultiple();
 412                 }
 413             }
 414         }
 415 
 416         // context sub-menu selection?  Handle unit selection or bus unfolding

 417         else if( evt->Category() == TC_COMMAND && evt->Action() == TA_CHOICE_MENU_CHOICE )
 418         {
 419             if( evt->GetCommandId().get() >= ID_POPUP_SCH_SELECT_UNIT_CMP
 420                 && evt->GetCommandId().get() <= ID_POPUP_SCH_SELECT_UNIT_CMP_MAX )
 421             {
 422                 SCH_COMPONENT* component = dynamic_cast<SCH_COMPONENT*>( m_selection.Front() );
 423                 int unit = evt->GetCommandId().get() - ID_POPUP_SCH_SELECT_UNIT_CMP;
 424 
 425                 if( component )
 426                     static_cast<SCH_EDIT_FRAME*>( m_frame )->SelectUnit( component, unit );
 427             }
 428             else if( evt->GetCommandId().get() >= ID_POPUP_SCH_UNFOLD_BUS
 429                      && evt->GetCommandId().get() <= ID_POPUP_SCH_UNFOLD_BUS_END )
 430             {
 431                 wxString* net = new wxString( *evt->Parameter<wxString*>() );
 432                 m_toolMgr->RunAction( EE_ACTIONS::unfoldBus, true, net );
 433             }
 434 
 435         }
 436 
 437         else if( evt->IsCancelInteractive() )
 438         {
 439             if( auto schframe = dynamic_cast<SCH_EDIT_FRAME*>( m_frame ) )
 440                 schframe->FocusOnItem( nullptr );
 441 
 442             ClearSelection();
 443         }
 444 
 445         else if( evt->Action() == TA_UNDO_REDO_PRE )
 446         {
 447             if( auto schframe = dynamic_cast<SCH_EDIT_FRAME*>( m_frame ) )
 448                 schframe->FocusOnItem( nullptr );
 449 
 450             ClearSelection();
 451         }
 452 
 453         else
 454             evt->SetPassEvent();
 455     }
 456 
 457     return 0;
 458 }

 467 bool EE_SELECTION_TOOL::SelectPoint( const VECTOR2I& aWhere, const KICAD_T* aFilterList,
 468         EDA_ITEM** aItem, bool* aSelectionCancelledFlag, bool aCheckLocked, bool aAdd,
 469         bool aSubtract, bool aExclusiveOr )
 470 {
 471     EE_COLLECTOR collector;
 472 
 473     collector.m_Threshold = KiROUND( getView()->ToWorld( HITTEST_THRESHOLD_PIXELS ) );
 474 
 475     if( m_isLibEdit )
 476     {
 477         auto part = static_cast<LIB_EDIT_FRAME*>( m_frame )->GetCurPart();
 478 
 479         if( !part )
 480             return false;
 481 
 482         collector.Collect( part->GetDrawItems(), aFilterList, (wxPoint) aWhere, m_unit, m_convert );
 483     }
 484     else
 485         collector.Collect( m_frame->GetScreen(), aFilterList, (wxPoint) aWhere, m_unit, m_convert );
 486 
 487     // Post-process collected items

 488     for( int i = collector.GetCount() - 1; i >= 0; --i )
 489     {
 490         if( !Selectable( collector[ i ] ) )
 491         {
 492             collector.Remove( i );
 493             continue;
 494         }
 495 
 496         if( aCheckLocked && collector[ i ]->IsLocked() )
 497         {
 498             collector.Remove( i );
 499             continue;
 500         }
 501 
 502         // SelectPoint, unlike other selection routines, can select line ends

 503         if( collector[ i ]->Type() == SCH_LINE_T )
 504         {
 505             SCH_LINE* line = (SCH_LINE*) collector[ i ];
 506             line->ClearFlags( STARTPOINT | ENDPOINT );
 507 
 508             if( HitTestPoints( line->GetStartPoint(), (wxPoint) aWhere, collector.m_Threshold ) )
 509                 line->SetFlags( STARTPOINT );
 510             else if (HitTestPoints( line->GetEndPoint(), (wxPoint) aWhere, collector.m_Threshold ) )
 511                 line->SetFlags( ENDPOINT );
 512             else
 513                 line->SetFlags( STARTPOINT | ENDPOINT );
 514         }
 515     }
 516 
 517     m_selection.ClearReferencePoint();
 518 
 519     // Apply some ugly heuristics to avoid disambiguation menus whenever possible

 520     if( collector.GetCount() > 1 && !m_skip_heuristics )
 521     {
 522         GuessSelectionCandidates( collector, aWhere );
 523     }
 524 
 525     // If still more than one item we're going to have to ask the user.

 526     if( collector.GetCount() > 1 )
 527     {
 528         collector.m_MenuTitle = wxEmptyString;
 529         // Must call selectionMenu via RunAction() to avoid event-loop contention

 530         m_toolMgr->RunAction( EE_ACTIONS::selectionMenu, true, &collector );
 531 
 532         if( collector.m_MenuCancelled )
 533         {
 534             if( aSelectionCancelledFlag )
 535                 *aSelectionCancelledFlag = true;
 536 
 537             return false;
 538         }
 539     }
 540 
 541     if( !aAdd && !aSubtract && !aExclusiveOr )
 542         ClearSelection();
 543 
 544     bool anyAdded      = false;
 545     bool anySubtracted = false;
 546 
 547     if( collector.GetCount() > 0 )
 548     {
 549         for( int i = 0; i < collector.GetCount(); ++i )
 550         {
 551             if( aSubtract || ( aExclusiveOr && collector[i]->IsSelected() ) )
 552             {
 553                 unselect( collector[i] );
 554                 anySubtracted = true;
 555             }
 556             else
 557             {
 558                 select( collector[i] );
 559                 anyAdded = true;
 560             }
 561         }
 562     }
 563 
 564     if( anyAdded )
 565     {
 566         m_toolMgr->ProcessEvent( EVENTS::SelectedEvent );
 567 
 568         if( aItem && collector.GetCount() == 1 )
 569             *aItem = collector[0];
 570 
 571         return true;
 572     }
 573     else if( anySubtracted )
 574     {
 575         m_toolMgr->ProcessEvent( EVENTS::UnselectedEvent );
 576         return true;
 577     }
 578 
 579     return false;
 580 }

 583 void EE_SELECTION_TOOL::GuessSelectionCandidates( EE_COLLECTOR& collector, const VECTOR2I& aPos )
 584 {
 585     // There are certain parent/child and enclosure combinations that can be handled

 586     // automatically.

 587 
 588     // Prefer exact hits to sloppy ones

 589     int exactHits = 0;
 590 
 591     for( int i = collector.GetCount() - 1; i >= 0; --i )
 592     {
 593         EDA_ITEM* item = collector[ i ];
 594 
 595         if( item->HitTest( (wxPoint) aPos, 0 ) )
 596             exactHits++;
 597     }
 598 
 599     if( exactHits > 0 && exactHits < collector.GetCount() )
 600     {
 601         for( int i = collector.GetCount() - 1; i >= 0; --i )
 602         {
 603             EDA_ITEM* item = collector[ i ];
 604 
 605             if( !item->HitTest( (wxPoint) aPos, 0 ) )
 606                 collector.Transfer( item );
 607         }
 608     }
 609 
 610     // Prefer a non-sheet to a sheet

 611     for( int i = 0; collector.GetCount() == 2 && i < 2; ++i )
 612     {
 613         EDA_ITEM* item = collector[ i ];
 614         EDA_ITEM* other = collector[ ( i + 1 ) % 2 ];
 615 
 616         if( item->Type() != SCH_SHEET_T && other->Type() == SCH_SHEET_T )
 617             collector.Transfer( other );
 618     }
 619 
 620     // Prefer a symbol to a pin or the opposite, when both a symbol and a pin are selected

 621     // We need to be able to select only a pin:

 622     // - to display its characteristics (especially if an ERC is attached to the pin)

 623     // - for cross probing, to select the corresponding pad.

 624     // Note also the case happens only in schematic editor. In symbol editor, the symbol

 625     // itself is never selected

 626     for( int i = 0; collector.GetCount() == 2 && i < 2; ++i )
 627     {
 628         EDA_ITEM* item = collector[ i ];
 629         EDA_ITEM* other = collector[ ( i + 1 ) % 2 ];
 630 
 631         if( item->Type() == SCH_COMPONENT_T && other->Type() == SCH_PIN_T )
 632         {
 633             if( !m_isLibEdit && m_frame->eeconfig()->m_Selection.select_pin_selects_symbol )
 634                 collector.Transfer( other );
 635             else
 636                 collector.Transfer( item );
 637         }
 638     }
 639 
 640     // Prefer a field to a symbol

 641     for( int i = 0; collector.GetCount() == 2 && i < 2; ++i )
 642     {
 643         EDA_ITEM* item = collector[ i ];
 644         EDA_ITEM* other = collector[ ( i + 1 ) % 2 ];
 645 
 646         if( item->Type() == SCH_FIELD_T && other->Type() == SCH_COMPONENT_T )
 647             collector.Transfer( other );
 648     }
 649 
 650     // No need for multiple wires at a single point; if there's a junction select that;

 651     // otherwise any of the wires will do

 652     bool junction = false;
 653     bool wiresOnly = true;
 654 
 655     for( EDA_ITEM* item : collector )
 656     {
 657         if( item->Type() == SCH_JUNCTION_T )
 658             junction = true;
 659         else if( item->Type() != SCH_LINE_T )
 660             wiresOnly = false;
 661     }
 662 
 663     if( wiresOnly )
 664     {
 665         for( int j = collector.GetCount() - 1; j >= 0; --j )
 666         {
 667             if( junction && collector[ j ]->Type() != SCH_JUNCTION_T )
 668                 collector.Transfer( j );
 669             else if( !junction && j > 0 )
 670                 collector.Transfer( j );
 671         }
 672     }
 673 }

 723 bool EE_SELECTION_TOOL::selectMultiple()
 724 {
 725     bool cancelled = false;     // Was the tool cancelled while it was running?

 726     m_multiple = true;          // Multiple selection mode is active

 727     KIGFX::VIEW* view = getView();
 728 
 729     KIGFX::PREVIEW::SELECTION_AREA area;
 730     view->Add( &area );
 731 
 732     while( TOOL_EVENT* evt = Wait() )
 733     {
 734         if( evt->IsCancelInteractive() || evt->IsActivate() )
 735         {
 736             cancelled = true;
 737             break;
 738         }
 739 
 740         if( evt->IsDrag( BUT_LEFT ) )
 741         {
 742             if( !m_additive && !m_subtractive && !m_exclusive_or )
 743                 ClearSelection();
 744 
 745             // Start drawing a selection box

 746             area.SetOrigin( evt->DragOrigin() );
 747             area.SetEnd( evt->Position() );
 748             area.SetAdditive( m_additive );
 749             area.SetSubtractive( m_subtractive );
 750             area.SetExclusiveOr( m_exclusive_or );
 751 
 752             view->SetVisible( &area, true );
 753             view->Update( &area );
 754             getViewControls()->SetAutoPan( true );
 755         }
 756 
 757         if( evt->IsMouseUp( BUT_LEFT ) )
 758         {
 759             getViewControls()->SetAutoPan( false );
 760 
 761             // End drawing the selection box

 762             view->SetVisible( &area, false );
 763 
 764             // Mark items within the selection box as selected

 765             std::vector<KIGFX::VIEW::LAYER_ITEM_PAIR> selectedItems;
 766             std::vector<KIGFX::VIEW::LAYER_ITEM_PAIR> sheetPins;
 767 
 768             // Filter the view items based on the selection box

 769             BOX2I selectionBox = area.ViewBBox();
 770             view->Query( selectionBox, selectedItems );         // Get the list of selected items

 771 
 772             // Sheet pins aren't in the view; add them by hand

 773             for( KIGFX::VIEW::LAYER_ITEM_PAIR& pair : selectedItems )
 774             {
 775                 SCH_SHEET* sheet = dynamic_cast<SCH_SHEET*>( pair.first );
 776 
 777                 if( sheet )
 778                 {
 779                     int layer = pair.second;
 780 
 781                     for( SCH_SHEET_PIN* pin : sheet->GetPins() )
 782                         sheetPins.emplace_back( KIGFX::VIEW::LAYER_ITEM_PAIR( pin, layer ) );
 783                 }
 784             }
 785 
 786             selectedItems.insert( selectedItems.end(), sheetPins.begin(), sheetPins.end() );
 787 
 788             int width = area.GetEnd().x - area.GetOrigin().x;
 789             int height = area.GetEnd().y - area.GetOrigin().y;
 790 
 791             /* Selection mode depends on direction of drag-selection:

 792              * Left > Right : Select objects that are fully enclosed by selection

 793              * Right > Left : Select objects that are crossed by selection

 794              */
 795             bool windowSelection = width >= 0;
 796             bool anyAdded = false;
 797             bool anySubtracted = false;
 798 
 799             if( view->IsMirroredX() )
 800                 windowSelection = !windowSelection;
 801 
 802             // Construct an EDA_RECT to determine EDA_ITEM selection

 803             EDA_RECT selectionRect( (wxPoint) area.GetOrigin(), wxSize( width, height ) );
 804 
 805             selectionRect.Normalize();
 806 
 807             for( KIGFX::VIEW::LAYER_ITEM_PAIR& pair : selectedItems )
 808             {
 809                 EDA_ITEM* item = dynamic_cast<EDA_ITEM*>( pair.first );
 810 
 811                 if( item && Selectable( item ) && item->HitTest( selectionRect, windowSelection ) )
 812                 {
 813                     if( m_subtractive || ( m_exclusive_or && item->IsSelected() ) )
 814                     {
 815                         unselect( item );
 816                         anySubtracted = true;
 817                     }
 818                     else
 819                     {
 820                         select( item );
 821                         item->SetFlags( STARTPOINT | ENDPOINT );
 822                         anyAdded = true;
 823                     }
 824                 }
 825             }
 826 
 827             m_selection.SetIsHover( false );
 828 
 829             // Inform other potentially interested tools

 830             if( anyAdded )
 831                 m_toolMgr->ProcessEvent( EVENTS::SelectedEvent );
 832 
 833             if( anySubtracted )
 834                 m_toolMgr->ProcessEvent( EVENTS::UnselectedEvent );
 835 
 836             break;  // Stop waiting for events

 837         }
 838     }
 839 
 840     getViewControls()->SetAutoPan( false );
 841 
 842     // Stop drawing the selection box

 843     view->Remove( &area );
 844     m_multiple = false;         // Multiple selection mode is inactive

 845 
 846     if( !cancelled )
 847         m_selection.ClearReferencePoint();
 848 
 849     return cancelled;
 850 }

1102 bool EE_SELECTION_TOOL::doSelectionMenu( EE_COLLECTOR* aCollector )
1103 {
1104     EDA_ITEM*   current = nullptr;
1105     bool        selectAll = false;
1106     bool        expandSelection = false;
1107 
1108     do
1109     {
1110         /// The user has requested the full, non-limited list of selection items

1111         if( expandSelection )
1112             aCollector->Combine();
1113 
1114         expandSelection = false;
1115 
1116         int         limit = std::min( 9, aCollector->GetCount() );
1117         ACTION_MENU menu( true );
1118 
1119         for( int i = 0; i < limit; ++i )
1120         {
1121             wxString  text;
1122             EDA_ITEM* item = ( *aCollector )[i];
1123             text           = item->GetSelectMenuText( m_frame->GetUserUnits() );
1124 
1125             wxString menuText = wxString::Format( "&%d. %s\t%d", i + 1, text, i + 1 );
1126             menu.Add( menuText, i + 1, item->GetMenuImage() );
1127         }
1128 
1129         menu.AppendSeparator();
1130         menu.Add( _( "Select &All\tA" ), limit + 1, net_highlight_schematic_xpm );
1131 
1132         if( !expandSelection && aCollector->HasAdditionalItems() )
1133             menu.Add( _( "&Expand Selection\tE" ), limit + 2, nullptr );
1134 
1135         if( aCollector->m_MenuTitle.Length() )
1136         {
1137             menu.SetTitle( aCollector->m_MenuTitle );
1138             menu.SetIcon( info_xpm );
1139             menu.DisplayTitle( true );
1140         }
1141         else
1142         {
1143             menu.DisplayTitle( false );
1144         }
1145 
1146         SetContextMenu( &menu, CMENU_NOW );
1147 
1148         while( TOOL_EVENT* evt = Wait() )
1149         {
1150             if( evt->Action() == TA_CHOICE_MENU_UPDATE )
1151             {
1152                 if( selectAll )
1153                 {
1154                     for( int i = 0; i < aCollector->GetCount(); ++i )
1155                         unhighlight( ( *aCollector )[i], BRIGHTENED );
1156                 }
1157                 else if( current )
1158                 {
1159                     unhighlight( current, BRIGHTENED );
1160                 }
1161 
1162                 int id = *evt->GetCommandId();
1163 
1164                 // User has pointed an item, so show it in a different way

1165                 if( id > 0 && id <= limit )
1166                 {
1167                     current = ( *aCollector )[id - 1];
1168                     highlight( current, BRIGHTENED );
1169                 }
1170                 else
1171                 {
1172                     current = nullptr;
1173                 }
1174 
1175                 // User has pointed on the "Select All" option

1176                 if( id == limit + 1 )
1177                 {
1178                     for( int i = 0; i < aCollector->GetCount(); ++i )
1179                         highlight( ( *aCollector )[i], BRIGHTENED );
1180                     selectAll = true;
1181                 }
1182                 else
1183                 {
1184                     selectAll = false;
1185                 }
1186             }
1187             else if( evt->Action() == TA_CHOICE_MENU_CHOICE )
1188             {
1189                 if( selectAll )
1190                 {
1191                     for( int i = 0; i < aCollector->GetCount(); ++i )
1192                         unhighlight( ( *aCollector )[i], BRIGHTENED );
1193                 }
1194                 else if( current )
1195                     unhighlight( current, BRIGHTENED );
1196 
1197                 OPT<int> id = evt->GetCommandId();
1198 
1199                 // User has selected the "Select All" option

1200                 if( id == limit + 1 )
1201                 {
1202                     selectAll = true;
1203                     current   = nullptr;
1204                 }
1205                 // User has selected an item, so this one will be returned

1206                 else if( id && ( *id > 0 ) && ( *id <= limit ) )
1207                 {
1208                     selectAll = false;
1209                     current   = ( *aCollector )[*id - 1];
1210                 }
1211                 else
1212                 {
1213                     selectAll = false;
1214                     current   = nullptr;
1215                 }
1216             }
1217             else if( evt->Action() == TA_CHOICE_MENU_CLOSED )
1218             {
1219                 break;
1220             }
1221 
1222             getView()->UpdateItems();
1223             m_frame->GetCanvas()->Refresh();
1224         }
1225     } while( expandSelection );
1226 
1227     if( selectAll )
1228         return true;
1229     else if( current )
1230     {
1231         unhighlight( current, BRIGHTENED );
1232 
1233         getView()->UpdateItems();
1234         m_frame->GetCanvas()->Refresh();
1235 
1236         aCollector->Empty();
1237         aCollector->Append( current );
1238         return true;
1239     }
1240 
1241     return false;
1242 }

1245 bool EE_SELECTION_TOOL::Selectable( const EDA_ITEM* aItem, bool checkVisibilityOnly ) const
1246 {
1247     // NOTE: in the future this is where eeschema layer/itemtype visibility will be handled

1248     LIB_EDIT_FRAME* symbeditFrame = dynamic_cast< LIB_EDIT_FRAME* >( m_frame );
1249 
1250     switch( aItem->Type() )
1251     {
1252     case SCH_PIN_T:
1253         if( !static_cast<const SCH_PIN*>( aItem )->IsVisible() && !m_frame->GetShowAllPins() )
1254             return false;
1255         break;
1256 
1257     case LIB_PART_T:    // In libedit we do not want to select the symbol itself.

1258         return false;
1259 
1260     case LIB_FIELD_T:   // LIB_FIELD object can always be edited.

1261         break;
1262 
1263     case LIB_ARC_T:
1264     case LIB_CIRCLE_T:
1265     case LIB_TEXT_T:
1266     case LIB_RECTANGLE_T:
1267     case LIB_POLYLINE_T:
1268     case LIB_BEZIER_T:
1269     case LIB_PIN_T:
1270     {
1271         if( symbeditFrame )
1272         {
1273             LIB_ITEM* lib_item = (LIB_ITEM*) aItem;
1274 
1275             if( lib_item->GetUnit() && lib_item->GetUnit() != symbeditFrame->GetUnit() )
1276                 return false;
1277 
1278             if( lib_item->GetConvert() && lib_item->GetConvert() != symbeditFrame->GetConvert() )
1279                 return false;
1280         }
1281 
1282         break;
1283     }
1284 
1285     case SCH_MARKER_T:  // Always selectable

1286         return true;
1287 
1288     default:            // Suppress warnings

1289         break;
1290     }
1291 
1292     return true;
1293 }

1326 void EE_SELECTION_TOOL::highlight( EDA_ITEM* aItem, int aMode, EE_SELECTION* aGroup )
1327 {
1328     KICAD_T itemType = aItem->Type();
1329 
1330     if( aMode == SELECTED )
1331         aItem->SetSelected();
1332     else if( aMode == BRIGHTENED )
1333         aItem->SetBrightened();
1334 
1335     if( aGroup )
1336         aGroup->Add( aItem );
1337 
1338     // Highlight pins and fields.  (All the other component children are currently only

1339     // represented in the LIB_PART and will inherit the settings of the parent component.)

1340     if( itemType == SCH_COMPONENT_T )
1341     {
1342         if( auto schframe = dynamic_cast<SCH_EDIT_FRAME*>( m_frame ) )
1343         {
1344             SCH_PIN_PTRS pins = static_cast<SCH_COMPONENT*>( aItem )->GetSchPins(
1345                     &schframe->GetCurrentSheet() );
1346 
1347             for( SCH_PIN* pin : pins )
1348             {
1349                 if( aMode == SELECTED )
1350                     pin->SetSelected();
1351                 else if( aMode == BRIGHTENED )
1352                     pin->SetBrightened();
1353             }
1354         }
1355 
1356         for( SCH_FIELD& field : static_cast<SCH_COMPONENT*>( aItem )->GetFields() )
1357         {
1358             if( aMode == SELECTED )
1359                 field.SetSelected();
1360             else if( aMode == BRIGHTENED )
1361                 field.SetBrightened();
1362         }
1363     }
1364     else if( itemType == SCH_SHEET_T )
1365     {
1366         for( SCH_FIELD& field : static_cast<SCH_SHEET*>( aItem )->GetFields() )
1367         {

1454 bool EE_SELECTION_TOOL::selectionContains( const VECTOR2I& aPoint ) const
1455 {
1456     const unsigned GRIP_MARGIN = 20;
1457     VECTOR2I margin = getView()->ToWorld( VECTOR2I( GRIP_MARGIN, GRIP_MARGIN ), false );
1458 
1459     // Check if the point is located within any of the currently selected items bounding boxes

1460     for( auto item : m_selection )
1461     {
1462         BOX2I itemBox = item->ViewBBox();
1463         itemBox.Inflate( margin.x, margin.y );    // Give some margin for gripping an item

1464 
1465         if( itemBox.Contains( aPoint ) )
1466             return true;
1467     }
1468 
1469     return false;
1470 }
