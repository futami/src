  38 /**

  39  * Hold a record identifying a symbol library accessed by the appropriate symbol library

  40  * #SCH_PLUGIN object in the #SYMBOL_LIB_TABLE.

  41  */
  42 class SYMBOL_LIB_TABLE_ROW : public LIB_TABLE_ROW
  43 {
  44     friend class SYMBOL_LIB_TABLE;
  45 
  46 public:
  47     typedef SCH_IO_MGR::SCH_FILE_T LIB_T;
  48 
  49     SYMBOL_LIB_TABLE_ROW( const wxString& aNick, const wxString& aURI, const wxString& aType,
  50                           const wxString& aOptions = wxEmptyString,
  51                           const wxString& aDescr = wxEmptyString ) :
  52         LIB_TABLE_ROW( aNick, aURI, aOptions, aDescr )
  53     {
  54         SetType( aType );
  55         SetEnabled( true );
  56     }
  57 
  58     SYMBOL_LIB_TABLE_ROW() :
  59         type( SCH_IO_MGR::SCH_KICAD )
  60     {
  61         SetEnabled( true );
  62     }
  63 
  64     bool operator==( const SYMBOL_LIB_TABLE_ROW& aRow ) const;
  65 
  66     bool operator!=( const SYMBOL_LIB_TABLE_ROW& aRow ) const   { return !( *this == aRow ); }
  67 
  68     /**

  69      * Return the type of symbol library table represented by this row.

  70      */
  71     const wxString GetType() const override         { return SCH_IO_MGR::ShowType( type ); }
  72 
  73     /**

  74      * Change the schematic plugin type represented by this row.

  75      */
  76     void SetType( const wxString& aType ) override;
  77 
  78     /**

  79      * Attempt to reload the library.

  80      * @return true if a reload was required.

  81      * @throws IO_ERROR if the reload was unsuccessful.

  82      */
  83     bool Refresh();
  84 
  85 protected:
  86     SYMBOL_LIB_TABLE_ROW( const SYMBOL_LIB_TABLE_ROW& aRow ) :
  87         LIB_TABLE_ROW( aRow ),
  88         type( aRow.type )
  89     {
  90         SetEnabled( aRow.GetIsEnabled() );
  91     }
  92 
  93 private:
  94 
  95     virtual LIB_TABLE_ROW* do_clone() const override
  96     {
  97         return new SYMBOL_LIB_TABLE_ROW( *this );
  98     }
  99 
 100     void setPlugin( SCH_PLUGIN* aPlugin )
 101     {
 102         plugin.set( aPlugin );
 103     }
 104 
 105     SCH_PLUGIN::SCH_PLUGIN_RELEASER  plugin;
 106     LIB_T                            type;
 107 };

 110 class SYMBOL_LIB_TABLE : public LIB_TABLE
 111 {
 112     friend class SYMBOL_LIB_TABLE_GRID;
 113     friend class PANEL_SYM_LIB_TABLE;
 114 
 115     static int m_modifyHash;     ///< helper for GetModifyHash()

 116 
 117 public:
 118     KICAD_T Type() override { return SYMBOL_LIB_TABLE_T; }
 119 
 120     static const char* PropPowerSymsOnly;
 121     static const char* PropNonPowerSymsOnly;
 122 
 123     virtual void Parse( LIB_TABLE_LEXER* aLexer ) override;
 124 
 125     virtual void Format( OUTPUTFORMATTER* aOutput, int aIndentLevel ) const override;
 126 
 127     /**

 128      * Build a symbol library table by pre-pending this table fragment in front of

 129      * @a aFallBackTable.  Loading of this table fragment is done by using Parse().

 130      *

 131      * @param aFallBackTable is another SYMBOL_LIB_TABLE which is searched only when

 132      *                       a row is not found in this table.  No ownership is

 133      *                       taken of aFallBackTable.

 134      */
 135     SYMBOL_LIB_TABLE( SYMBOL_LIB_TABLE* aFallBackTable = NULL );
 136 
 137     /**

 138      * Return an SYMBOL_LIB_TABLE_ROW if \a aNickName is found in this table or in any chained

 139      * fallBack table fragment.

 140      *

 141      * The #SCH_PLUGIN is loaded and attached to the "plugin" fieldf the #SYMBOL_LIB_TABLE_ROW if

 142      * not already loaded.

 143      *

 144      * @param aNickName is the name of the row to find.

 145      *

 146      * @return the row found or NULL if \a aNickName was not found.

 147      *

 148      * @throw IO_ERROR if \a aNickName cannot be found.

 149      */
 150     SYMBOL_LIB_TABLE_ROW* FindRow( const wxString& aNickName );
 151 
 152     int GetModifyHash();
 153 
 154     //-----<PLUGIN API SUBSET, REBASED ON aNickname>---------------------------

 155 
 156     /**

 157      * Return a list of symbol alias names contained within the library given by @a aNickname.

 158      *

 159      * @param aNickname is a locator for the "library", it is a "name" in LIB_TABLE_ROW.

 160      * @param aAliasNames is a reference to an array for the alias names.

 161      * @param aPowerSymbolsOnly is a flag to enumerate only power symbols.

 162      *

 163      * @throw IO_ERROR if the library cannot be found or loaded.

 164      */
 165     void EnumerateSymbolLib( const wxString& aNickname, wxArrayString& aAliasNames,
 166                              bool aPowerSymbolsOnly = false );
 167 
 168     void LoadSymbolLib( std::vector<LIB_PART*>& aAliasList, const wxString& aNickname,
 169                         bool aPowerSymbolsOnly = false );
 170 
 171     /**

 172      * Load a #LIB_PART having @a aName from the library given by @a aNickname.

 173      *

 174      * @param aNickname is a locator for the "library", it is a "name" in #LIB_TABLE_ROW

 175      * @param aName is the name of the #LIB_PART to load.

 176      * @param aFlatten set to true to flatten derived parts.

 177      *

 178      * @return the symbol alias if found or NULL if not found.

 179      *

 180      * @throw IO_ERROR if the library cannot be found or read.  No exception

 181      *                 is thrown in the case where \a aNickname cannot be found.

 182      */
 183     LIB_PART* LoadSymbol( const wxString& aNickname, const wxString& aName );
 184 
 185     LIB_PART* LoadSymbol( const LIB_ID& aLibId )
 186     {
 187         return LoadSymbol( aLibId.GetLibNickname(), aLibId.GetLibItemName() );
 188     }
 189 
 190     /**

 191      * The set of return values from SaveSymbol() below.

 192      */
 193     enum SAVE_T
 194     {
 195         SAVE_OK,
 196         SAVE_SKIPPED,
 197     };
 198 
 199     /**

 200      * Write @a aSymbol to an existing library given by @a aNickname.

 201      *

 202      * If a #LIB_PART by the same name already exists or there are any conflicting alias

 203      * names, the new #LIB_PART will silently overwrite any existing aliases and/or part

 204      * becaue libraries cannot have duplicate alias names.  It is the responsibility of

 205      * the caller to check the library for conflicts before saving.

 206      *

 207      * @param aNickname is a locator for the "library", it is a "name" in LIB_TABLE_ROW

 208      * @param aSymbol is what to store in the library. The library owns the symbol after this

 209      *                call.

 210      * @param aOverwrite when true means overwrite any existing symbol by the same name,

 211      *                   else if false means skip the write and return SAVE_SKIPPED.

 212      *

 213      * @return SAVE_T - SAVE_OK or SAVE_SKIPPED.  If error saving, then IO_ERROR is thrown.

 214      *

 215      * @throw IO_ERROR if there is a problem saving the symbol.

 216      */
 217     SAVE_T SaveSymbol( const wxString& aNickname, const LIB_PART* aSymbol,
 218                        bool aOverwrite = true );
 219 
 220     /**

 221      * Deletes the @a aSymbolName from the library given by @a aNickname.

 222      *

 223      * @param aNickname is a locator for the "library", it is a "name" in LIB_TABLE_ROW.

 224      *

 225      * @param aSymbolName is the name of a symbol to delete from the specified library.

 226      *

 227      * @throw IO_ERROR if there is a problem finding the footprint or the library, or deleting it.

 228      */
 229     void DeleteSymbol( const wxString& aNickname, const wxString& aSymbolName );
 230 
 231     /**

 232      * Return true if the library given by @a aNickname is writable.

 233      *

 234      * It is possible that some symbols libraries are read only because of where they are

 235      * installed.

 236      *

 237      * @param aNickname is the library nickname in the symbol library table.

 238      *

 239      * @throw IO_ERROR if no library at @a aNickname exists.

 240      */
 241     bool IsSymbolLibWritable( const wxString& aNickname );
 242 
 243     void DeleteSymbolLib( const wxString& aNickname );
 244 
 245     void CreateSymbolLib( const wxString& aNickname );
 246 
 247     //-----</PLUGIN API SUBSET, REBASED ON aNickname>---------------------------

 248 
 249     /**

 250      * Load a #LIB_PART having @a aFootprintId with possibly an empty library nickname.

 251      *

 252      * @param aId the library nickname and name of the symbol to load.

 253      *

 254      * @return  the library symbol if found (the library owns it) or NULL if not found.

 255      *

 256      * @throw IO_ERROR if the library cannot be found or read.  No exception

 257      *                 is thrown in the case where aId cannot be found.

 258      * @throw PARSE_ERROR if @a aId is not parsed OK.

 259      */
 260     LIB_PART* LoadSymbolWithOptionalNickname( const LIB_ID& aId );
 261 
 262     /**

 263      * Load the global symbol library table into \a aTable.

 264      *

 265      * This probably should be move into the application object when KiCad is changed

 266      * to a single process application.  This is the least painful solution for the

 267      * time being.

 268      *

 269      * @param aTable the #SYMBOL_LIB_TABLE object to load.

 270      *

 271      * @return true if the global library table exists and is loaded properly.

 272      *

 273      * @throw IO_ERROR if an error occurs attempting to load the symbol library table.

 274      */
 275     static bool LoadGlobalTable( SYMBOL_LIB_TABLE& aTable );
 276 
 277     /**

 278      *

 279      * Fetch the global symbol library table file name.

 280      *

 281      * @return the platform specific global symbol library path and file name.

 282      */
 283     static wxString GetGlobalTableFileName();
 284 
 285     /**

 286      * Return the name of the environment variable used to hold the directory of locally

 287      * installed "KiCad sponsored" system symbol libraries.

 288      *

 289      * These can be either legacy or sweet format.  The only thing special about this

 290      * particular environment variable is that it is set automatically by KiCad on

 291      * program start up, <b>if</b> it is not set already in the environment.

 292      */
 293     static const wxString GlobalPathEnvVariableName();
 294 
 295     static SYMBOL_LIB_TABLE& GetGlobalLibTable();
 296 
 297     static const wxString& GetSymbolLibTableFileName();
 298 };
