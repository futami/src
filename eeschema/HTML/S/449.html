 118 int LIB_EDIT_TOOL::Rotate( const TOOL_EVENT& aEvent )
 119 {
 120     EE_SELECTION& selection = m_selectionTool->RequestSelection();
 121 
 122     if( selection.GetSize() == 0 )
 123         return 0;
 124 
 125     wxPoint   rotPoint;
 126     bool      ccw = ( aEvent.Matches( EE_ACTIONS::rotateCCW.MakeEvent() ) );
 127     LIB_ITEM* item = static_cast<LIB_ITEM*>( selection.Front() );
 128 
 129     if( !item->IsMoving() )
 130         saveCopyInUndoList( m_frame->GetCurPart(), UR_LIBEDIT );
 131 
 132     if( selection.GetSize() == 1 )
 133         rotPoint = item->GetPosition();
 134     else
 135         rotPoint = m_frame->GetNearestGridPosition( mapCoords( selection.GetCenter() ) );
 136 
 137     for( unsigned ii = 0; ii < selection.GetSize(); ii++ )
 138     {
 139         item = static_cast<LIB_ITEM*>( selection.GetItem( ii ) );
 140         item->Rotate( rotPoint, ccw );
 141         m_frame->RefreshItem( item );
 142     }
 143 
 144     if( item->IsMoving() )
 145         m_toolMgr->RunAction( ACTIONS::refreshPreview, true );
 146     else
 147     {
 148         m_toolMgr->PostEvent( EVENTS::SelectedItemsModified );
 149 
 150         if( selection.IsHover() )
 151             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 152 
 153         m_frame->OnModify();
 154     }
 155 
 156     return 0;
 157 }

 160 int LIB_EDIT_TOOL::Mirror( const TOOL_EVENT& aEvent )
 161 {
 162     EE_SELECTION& selection = m_selectionTool->RequestSelection();
 163 
 164     if( selection.GetSize() == 0 )
 165         return 0;
 166 
 167     wxPoint   mirrorPoint;
 168     bool      xAxis = ( aEvent.Matches( EE_ACTIONS::mirrorX.MakeEvent() ) );
 169     LIB_ITEM* item = static_cast<LIB_ITEM*>( selection.Front() );
 170 
 171     if( !item->IsMoving() )
 172         saveCopyInUndoList( m_frame->GetCurPart(), UR_LIBEDIT );
 173 
 174     if( selection.GetSize() == 1 )
 175         mirrorPoint = item->GetPosition();
 176     else
 177         mirrorPoint = m_frame->GetNearestGridPosition( mapCoords( selection.GetCenter() ) );
 178 
 179     for( unsigned ii = 0; ii < selection.GetSize(); ii++ )
 180     {
 181         item = static_cast<LIB_ITEM*>( selection.GetItem( ii ) );
 182 
 183         if( xAxis )
 184             item->MirrorVertical( mirrorPoint );
 185         else
 186             item->MirrorHorizontal( mirrorPoint );
 187 
 188         m_frame->RefreshItem( item );
 189     }
 190 
 191     m_toolMgr->PostEvent( EVENTS::SelectedItemsModified );
 192 
 193     if( item->IsMoving() )
 194         m_toolMgr->RunAction( ACTIONS::refreshPreview, true );
 195     else
 196     {
 197         if( selection.IsHover() )
 198             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 199 
 200         m_frame->OnModify();
 201     }
 202 
 203     return 0;
 204 }

 221 int LIB_EDIT_TOOL::DoDelete( const TOOL_EVENT& aEvent )
 222 {
 223     LIB_PART* part = m_frame->GetCurPart();
 224     auto      items = m_selectionTool->RequestSelection( nonFields ).GetItems();
 225 
 226     if( items.empty() )
 227         return 0;
 228 
 229     // Don't leave a freed pointer in the selection

 230     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 231 
 232     saveCopyInUndoList( part, UR_LIBEDIT );
 233 
 234     std::set<LIB_ITEM *> toDelete;
 235 
 236     for( EDA_ITEM* item : items )
 237     {
 238         if( item->Type() == LIB_PIN_T )
 239         {
 240             LIB_PIN*  pin = static_cast<LIB_PIN*>( item );
 241             wxPoint   pos = pin->GetPosition();
 242 
 243             toDelete.insert( pin );
 244 
 245             // when pin editing is synchronized, all pins of the same body style are removed:

 246             if( m_frame->SynchronizePins() )
 247             {
 248                 int curr_convert = pin->GetConvert();
 249                 LIB_PIN* next_pin = part->GetNextPin();
 250 
 251                 while( next_pin != NULL )
 252                 {
 253                     pin = next_pin;
 254                     next_pin = part->GetNextPin( pin );
 255 
 256                     if( pin->GetPosition() != pos )
 257                         continue;
 258 
 259                     if( pin->GetConvert() != curr_convert )
 260                         continue;
 261 
 262                     toDelete.insert( pin );
 263                 }
 264             }
 265         }
 266         else
 267         {
 268             toDelete.insert( (LIB_ITEM*) item );
 269         }
 270     }
 271 
 272     for( auto item : toDelete )
 273     {
 274         part->RemoveDrawItem( item );
 275     }
 276 
 277     m_frame->RebuildView();
 278     m_frame->OnModify();
 279 
 280     return 0;
 281 }

 287 int LIB_EDIT_TOOL::DeleteItemCursor( const TOOL_EVENT& aEvent )
 288 {
 289     std::string  tool = aEvent.GetCommandStr().get();
 290     PICKER_TOOL* picker = m_toolMgr->GetTool<PICKER_TOOL>();
 291 
 292     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 293     m_pickerItem = nullptr;
 294 
 295     // Deactivate other tools; particularly important if another PICKER is currently running

 296     Activate();
 297 
 298     picker->SetCursor( wxStockCursor( wxCURSOR_BULLSEYE ) );
 299 
 300     picker->SetClickHandler(
 301         [this] ( const VECTOR2D& aPosition ) -> bool
 302         {
 303             if( m_pickerItem )
 304             {
 305                 EE_SELECTION_TOOL* selectionTool = m_toolMgr->GetTool<EE_SELECTION_TOOL>();
 306                 selectionTool->UnbrightenItem( m_pickerItem );
 307                 selectionTool->AddItemToSel( m_pickerItem, true /*quiet mode*/ );
 308                 m_toolMgr->RunAction( ACTIONS::doDelete, true );
 309                 m_pickerItem = nullptr;
 310             }
 311 
 312             return true;
 313         } );
 314 
 315     picker->SetMotionHandler(
 316         [this] ( const VECTOR2D& aPos )
 317         {
 318             EE_SELECTION_TOOL* selectionTool = m_toolMgr->GetTool<EE_SELECTION_TOOL>();
 319             EE_COLLECTOR collector;
 320             collector.m_Threshold = KiROUND( getView()->ToWorld( HITTEST_THRESHOLD_PIXELS ) );
 321             collector.Collect( m_frame->GetScreen(), nonFields, (wxPoint) aPos, m_frame->GetUnit(),
 322                     m_frame->GetConvert() );
 323 
 324             // Remove unselectable items

 325             for( int i = collector.GetCount() - 1; i >= 0; --i )
 326             {
 327                 if( !selectionTool->Selectable( collector[ i ] ) )
 328                     collector.Remove( i );
 329             }
 330 
 331             if( collector.GetCount() > 1 )
 332                 selectionTool->GuessSelectionCandidates( collector, aPos );
 333 
 334             EDA_ITEM* item = collector.GetCount() == 1 ? collector[ 0 ] : nullptr;
 335 
 336             if( m_pickerItem != item )
 337             {
 338                 if( m_pickerItem )
 339                     selectionTool->UnbrightenItem( m_pickerItem );
 340 
 341                 m_pickerItem = item;
 342 
 343                 if( m_pickerItem )
 344                     selectionTool->BrightenItem( m_pickerItem );
 345             }
 346         } );
 347 
 348     picker->SetFinalizeHandler(
 349         [this] ( const int& aFinalState )
 350         {
 351             if( m_pickerItem )
 352                 m_toolMgr->GetTool<EE_SELECTION_TOOL>()->UnbrightenItem( m_pickerItem );
 353         } );
 354 
 355     m_toolMgr->RunAction( ACTIONS::pickerTool, true, &tool );
 356 
 357     return 0;
 358 }

 361 int LIB_EDIT_TOOL::Properties( const TOOL_EVENT& aEvent )
 362 {
 363     EE_SELECTION& selection = m_selectionTool->RequestSelection();
 364 
 365     if( selection.Empty() || aEvent.IsAction( &EE_ACTIONS::symbolProperties ) )
 366     {
 367         if( m_frame->GetCurPart() )
 368             editSymbolProperties();
 369     }
 370     else if( selection.Size() == 1 )
 371     {
 372         LIB_ITEM* item = (LIB_ITEM*) selection.Front();
 373 
 374         // Save copy for undo if not in edit (edit command already handle the save copy)

 375         if( item->GetEditFlags() == 0 )
 376             saveCopyInUndoList( item->GetParent(), UR_LIBEDIT );
 377 
 378         switch( item->Type() )
 379         {
 380         case LIB_PIN_T:
 381         {
 382             LIB_PIN_TOOL* pinTool = m_toolMgr->GetTool<LIB_PIN_TOOL>();
 383 
 384             if( pinTool )
 385                 pinTool->EditPinProperties( (LIB_PIN*) item );
 386 
 387             break;
 388         }
 389         case LIB_ARC_T:
 390         case LIB_CIRCLE_T:
 391         case LIB_RECTANGLE_T:
 392         case LIB_POLYLINE_T:
 393             editGraphicProperties( item );
 394             break;
 395 
 396         case LIB_TEXT_T:
 397             editTextProperties( item );
 398             break;
 399 
 400         case LIB_FIELD_T:
 401             editFieldProperties( (LIB_FIELD*) item );
 402             break;
 403 
 404         default:
 405             wxFAIL_MSG( wxT( "Unhandled item <" ) + item->GetClass() + wxT( ">" ) );
 406             break;
 407         }
 408     }
 409 
 410     m_toolMgr->PostEvent( EVENTS::SelectedItemsModified );
 411 
 412     return 0;
 413 }

 416 void LIB_EDIT_TOOL::editGraphicProperties( LIB_ITEM* aItem )
 417 {
 418     if( aItem == NULL )
 419         return;
 420 
 421     DIALOG_LIB_EDIT_DRAW_ITEM dialog( m_frame, aItem );
 422 
 423     if( dialog.ShowModal() != wxID_OK )
 424         return;
 425 
 426     if( aItem->IsFillable() )
 427         aItem->SetFillMode( (FILL_T) dialog.GetFillStyle() );
 428 
 429     aItem->SetWidth( dialog.GetWidth() );
 430 
 431     if( dialog.GetApplyToAllConversions() )
 432         aItem->SetConvert( 0 );
 433     else
 434         aItem->SetConvert( m_frame->GetConvert() );
 435 
 436     if( dialog.GetApplyToAllUnits() )
 437         aItem->SetUnit( 0 );
 438     else
 439         aItem->SetUnit( m_frame->GetUnit() );
 440 
 441     updateView( aItem );
 442     m_frame->GetCanvas()->Refresh();
 443     m_frame->OnModify( );
 444 
 445     LIB_DRAWING_TOOLS* drawingTools = m_toolMgr->GetTool<LIB_DRAWING_TOOLS>();
 446     drawingTools->SetDrawSpecificConvert( !dialog.GetApplyToAllConversions() );
 447     drawingTools->SetDrawSpecificUnit( !dialog.GetApplyToAllUnits() );
 448 
 449     MSG_PANEL_ITEMS items;
 450     aItem->GetMsgPanelInfo( m_frame, items );
 451     m_frame->SetMsgPanel( items );
 452 }

 471 void LIB_EDIT_TOOL::editFieldProperties( LIB_FIELD* aField )
 472 {
 473     if( aField == NULL )
 474         return;
 475 
 476     wxString  caption;
 477     LIB_PART* parent = aField->GetParent();
 478     wxCHECK( parent, /* void */ );
 479 
 480     // Editing the component value field is equivalent to creating a new component based

 481     // on the current component.  Set the dialog message to inform the user.

 482     if( aField->GetId() == VALUE )
 483         caption = _( "Edit Component Name" );
 484     else
 485         caption.Printf( _( "Edit %s Field" ), aField->GetName() );
 486 
 487     DIALOG_LIB_EDIT_ONE_FIELD dlg( m_frame, caption, aField );
 488 
 489     // The dialog may invoke a kiway player for footprint fields

 490     // so we must use a quasimodal dialog.

 491     if( dlg.ShowQuasiModal() != wxID_OK )
 492         return;
 493 
 494     wxString newFieldValue = LIB_ID::FixIllegalChars( dlg.GetText(), LIB_ID::ID_SCH );
 495     wxString oldFieldValue = aField->GetFullText( m_frame->GetUnit() );
 496     bool     renamed = aField->GetId() == VALUE && newFieldValue != oldFieldValue;
 497 
 498     if( renamed )
 499         saveCopyInUndoList( parent, UR_LIB_RENAME );
 500     else
 501         saveCopyInUndoList( parent, UR_LIBEDIT );
 502 
 503     dlg.UpdateField( aField );
 504 
 505     if( renamed )
 506     {
 507         parent->SetName( newFieldValue );
 508         m_frame->UpdateAfterSymbolProperties( &oldFieldValue );
 509     }
 510     else
 511     {
 512         updateView( aField );
 513         m_frame->GetCanvas()->Refresh();
 514         m_frame->OnModify();
 515         m_frame->DisplayCmpDoc();
 516     }
 517 }

 520 void LIB_EDIT_TOOL::editSymbolProperties()
 521 {
 522     LIB_PART*     part = m_frame->GetCurPart();
 523     bool          partLocked = part->UnitsLocked();
 524     wxString      oldName = part->GetName();
 525 
 526     m_toolMgr->RunAction( ACTIONS::cancelInteractive, true );
 527     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 528 
 529     DIALOG_EDIT_COMPONENT_IN_LIBRARY dlg( m_frame, part );
 530 
 531     // This dialog itself subsequently can invoke a KIWAY_PLAYER as a quasimodal

 532     // frame. Therefore this dialog as a modal frame parent, MUST be run under

 533     // quasimodal mode for the quasimodal frame support to work.  So don't use

 534     // the QUASIMODAL macros here.

 535     if( dlg.ShowQuasiModal() != wxID_OK )
 536         return;
 537 
 538     m_frame->OnModify();
 539 
 540     // if m_UnitSelectionLocked has changed, set some edit options or defaults

 541     // to the best value

 542     if( partLocked != part->UnitsLocked() )
 543     {
 544         LIB_DRAWING_TOOLS* tools = m_toolMgr->GetTool<LIB_DRAWING_TOOLS>();
 545 
 546         // Enable synchronized pin edit mode for symbols with interchangeable units

 547         m_frame->m_SyncPinEdit = !part->UnitsLocked();
 548 
 549         // also set default edit options to the better value

 550         // Usually if units are locked, graphic items are specific to each unit

 551         // and if units are interchangeable, graphic items are common to units

 552         tools->SetDrawSpecificUnit( part->UnitsLocked() );
 553     }
 554 }

 613 int LIB_EDIT_TOOL::Copy( const TOOL_EVENT& aEvent )
 614 {
 615     LIB_PART*     part = m_frame->GetCurPart();
 616     EE_SELECTION& selection = m_selectionTool->RequestSelection( nonFields );
 617 
 618     if( !part || !selection.GetSize() )
 619         return 0;
 620 
 621     for( LIB_ITEM& item : part->GetDrawItems() )
 622     {
 623         if( item.Type() == LIB_FIELD_T )
 624             continue;
 625 
 626         wxASSERT( !item.HasFlag( STRUCT_DELETED ) );
 627 
 628         if( !item.IsSelected() )
 629             item.SetFlags( STRUCT_DELETED );
 630     }
 631 
 632     LIB_PART* partCopy = new LIB_PART( *part );
 633 
 634     STRING_FORMATTER  formatter;
 635     SCH_SEXPR_PLUGIN::FormatPart( partCopy, formatter );
 636 
 637     delete partCopy;
 638 
 639     for( LIB_ITEM& item : part->GetDrawItems() )
 640         item.ClearFlags( STRUCT_DELETED );
 641 
 642     if( m_toolMgr->SaveClipboard( formatter.GetString() ) )
 643         return 0;
 644     else
 645         return -1;
 646 }

 649 int LIB_EDIT_TOOL::Paste( const TOOL_EVENT& aEvent )
 650 {
 651     LIB_PART*           part = m_frame->GetCurPart();
 652 
 653     if( !part )
 654         return 0;
 655 
 656     std::string         text = m_toolMgr->GetClipboard();
 657     STRING_LINE_READER  reader( text, "Clipboard" );
 658     LIB_PART*           newPart;
 659 
 660     try
 661     {
 662         newPart = SCH_SEXPR_PLUGIN::ParsePart( reader );
 663     }
 664     catch( IO_ERROR& )
 665     {
 666         // If it's not a part then paste as text

 667         newPart = new LIB_PART( "dummy_part" );
 668         LIB_TEXT* newText = new LIB_TEXT( newPart );
 669         newText->SetText( text );
 670         newPart->AddDrawItem( newText );
 671     }
 672 
 673     if( !newPart )
 674         return -1;
 675 
 676     m_frame->SaveCopyInUndoList( part );
 677     m_selectionTool->ClearSelection();
 678 
 679     for( LIB_ITEM& item : part->GetDrawItems() )
 680         item.ClearFlags( IS_NEW | IS_PASTED | SELECTED );
 681 
 682     for( LIB_ITEM& item : newPart->GetDrawItems() )
 683     {
 684         if( item.Type() == LIB_FIELD_T )
 685             continue;
 686 
 687         LIB_ITEM* newItem = (LIB_ITEM*) item.Clone();
 688         newItem->SetParent( part );
 689         newItem->SetFlags( IS_NEW | IS_PASTED | SELECTED );
 690 
 691         newItem->SetUnit( newItem->GetUnit() ? m_frame->GetUnit() : 0 );
 692         newItem->SetConvert( newItem->GetConvert() ? m_frame->GetConvert() : 0 );
 693 
 694         part->GetDrawItems().push_back( newItem );
 695         getView()->Add( newItem );
 696     }
 697 
 698     delete newPart;
 699 
 700     m_selectionTool->RebuildSelection();
 701 
 702     EE_SELECTION& selection = m_selectionTool->GetSelection();
 703 
 704     if( !selection.Empty() )
 705     {
 706         selection.SetReferencePoint( getViewControls()->GetCursorPosition( true ) );
 707         m_toolMgr->RunAction( EE_ACTIONS::move, false );
 708     }
 709 
 710     return 0;
 711 }

 714 int LIB_EDIT_TOOL::Duplicate( const TOOL_EVENT& aEvent )
 715 {
 716     LIB_PART*     part = m_frame->GetCurPart();
 717     EE_SELECTION& selection = m_selectionTool->RequestSelection( nonFields );
 718 
 719     if( selection.GetSize() == 0 )
 720         return 0;
 721 
 722     // Doing a duplicate of a new object doesn't really make any sense; we'd just end

 723     // up dragging around a stack of objects...

 724     if( selection.Front()->IsNew() )
 725         return 0;
 726 
 727     if( !selection.Front()->IsMoving() )
 728         saveCopyInUndoList( m_frame->GetCurPart(), UR_LIBEDIT );
 729 
 730     EDA_ITEMS newItems;
 731 
 732     for( unsigned ii = 0; ii < selection.GetSize(); ++ii )
 733     {
 734         LIB_ITEM* oldItem = static_cast<LIB_ITEM*>( selection.GetItem( ii ) );
 735         LIB_ITEM* newItem = (LIB_ITEM*) oldItem->Clone();
 736         oldItem->ClearFlags( IS_NEW | IS_PASTED | SELECTED );
 737         newItem->SetFlags( IS_NEW | IS_PASTED | SELECTED );
 738         newItem->SetParent( part );
 739         newItems.push_back( newItem );
 740 
 741         part->GetDrawItems().push_back( newItem );
 742         getView()->Add( newItem );
 743     }
 744 
 745     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 746     m_toolMgr->RunAction( EE_ACTIONS::addItemsToSel, true, &newItems );
 747 
 748     selection.SetReferencePoint( mapCoords( getViewControls()->GetCursorPosition( true ) ) );
 749     m_toolMgr->RunAction( EE_ACTIONS::move, false );
 750 
 751     return 0;
 752 }
