  73 ///> Trace descriptor class

  74 class TRACE_DESC
  75 {
  76 public:
  77     TRACE_DESC( const NETLIST_EXPORTER_PSPICE_SIM& aExporter, const wxString& aName,
  78             SIM_PLOT_TYPE aType, const wxString& aParam );
  79 
  80     ///> Modifies an existing TRACE_DESC simulation type

  81     TRACE_DESC( const NETLIST_EXPORTER_PSPICE_SIM& aExporter,
  82             const TRACE_DESC& aDescription, SIM_PLOT_TYPE aNewType )
  83         : TRACE_DESC( aExporter, aDescription.GetName(), aNewType, aDescription.GetParam() )
  84     {
  85     }
  86 
  87     const wxString& GetTitle() const
  88     {
  89         return m_title;
  90     }
  91 
  92     const wxString& GetName() const
  93     {
  94         return m_name;
  95     }
  96 
  97     const wxString& GetParam() const
  98     {
  99         return m_param;
 100     }
 101 
 102     SIM_PLOT_TYPE GetType() const
 103     {
 104         return m_type;
 105     }
 106 
 107 private:
 108     // Three basic parameters

 109     ///> Name of the measured net/device

 110     wxString m_name;
 111 
 112     ///> Type of the signal

 113     SIM_PLOT_TYPE m_type;
 114 
 115     ///> Name of the signal parameter

 116     wxString m_param;
 117 
 118     // Generated data

 119     ///> Title displayed in the signal list/plot legend

 120     wxString m_title;
 121 };

 124 /** Implementing SIM_PLOT_FRAME_BASE */

 136     /**

 137      * @brief Creates a new plot panel for a given simulation type and adds it to the main

 138      * notebook.

 139      * @param aSimType is requested simulation type.

 140      * @return The new plot panel.

 141      */
 142     SIM_PANEL_BASE* NewPlotPanel( SIM_TYPE aSimType );
 143 
 144     /**

 145      * @brief Adds a voltage plot for a given net name.

 146      * @param aNetName is the net name for which a voltage plot should be created.

 147      */
 148     void AddVoltagePlot( const wxString& aNetName );
 149 
 150     /**

 151      * @brief Adds a current plot for a particular device.

 152      * @param aDeviceName is the device name (e.g. R1, C1).

 153      * @param aParam is the current type (e.g. I, Ic, Id).

 154      */
 155     void AddCurrentPlot( const wxString& aDeviceName, const wxString& aParam );
 156 
 157     /**

 158      * @brief Adds a tuner for a component.

 159      */
 160     void AddTuner( SCH_COMPONENT* aComponent );
 161 
 162     /**

 163      * @brief Removes an existing tuner.

 164      * @param aTuner is the tuner to be removed.

 165      * @param aErase decides whether the tuner should be also removed from the tuners list.

 166      * Otherwise it is removed only from the SIM_PLOT_FRAME pane.

 167      */
 168     void RemoveTuner( TUNER_SLIDER* aTuner, bool aErase = true );
 169 
 170     /**

 171      * @brief Returns the currently opened plot panel (or NULL if there is none).

 172      */
 173     SIM_PLOT_PANEL* CurrentPlot() const;
 174 
 175     /**

 176      * Returns the netlist exporter object used for simulations.

 177      */
 178     const NETLIST_EXPORTER_PSPICE_SIM* GetExporter() const;
 179 
 180     /**

 181      * @return the current background option for plotting.

 182      * false for drak bg, true for clear bg

 183      */
 184     bool GetPlotBgOpt() const { return m_plotUseWhiteBg; }
 185 
 186     /**

 187      * @return the wxColor selected in color list.

 188      * @param aColorId is the index in color list

 189      */
 190     wxColor GetPlotColor( int aColorId );
 191 
 192     /**

 193      * @return the count of colors in color list

 194      */
 195     int GetPlotColorCount() { return m_colorList.size(); }
 196

 226     /**

 227      * @brief Adds a new plot to the current panel.

 228      * @param aName is the device/net name.

 229      * @param aType describes the type of plot.

 230      * @param aParam is the parameter for the device/net (e.g. I, Id, V).

 231      */
 232     void addPlot( const wxString& aName, SIM_PLOT_TYPE aType, const wxString& aParam );
 233 
 234     /**

 235      * @brief Removes a plot with a specific title.

 236      * @param aPlotName is the full plot title (e.g. I(Net-C1-Pad1)).

 237      * @param aErase decides if plot should be removed from corresponding TRACE_MAP (see m_plots).

 238      */

 240 
 241     /**

 245 
 246     /**

 247      * @brief Updates plot in a particular SIM_PLOT_PANEL. If the panel does not contain

 250      * @param aPanel is the panel that should receive the update.

 251      * @return True if a plot was successfully added/updated.

 252      */
 253     bool updatePlot( const TRACE_DESC& aDescriptor, SIM_PLOT_PANEL* aPanel );
