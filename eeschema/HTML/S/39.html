 140 bool DIALOG_EDIT_COMPONENT_IN_LIBRARY::TransferDataToWindow()
 141 {
 142     if( !wxDialog::TransferDataToWindow() )
 143         return false;
 144 
 145     // Push a copy of each field into m_updateFields

 146     m_libEntry->GetFields( *m_fields );
 147 
 148     // The Y axis for components in lib is from bottom to top while the screen axis is top

 149     // to bottom: we must change the y coord sign for editing

 150     for( size_t i = 0; i < m_fields->size(); ++i )
 151     {
 152         wxPoint pos = m_fields->at( i ).GetPosition();
 153         pos.y = -pos.y;
 154         m_fields->at( i ).SetPosition( pos );
 155     }
 156 
 157     // notify the grid

 158     wxGridTableMessage msg( m_fields, wxGRIDTABLE_NOTIFY_ROWS_APPENDED, m_fields->GetNumberRows() );
 159     m_grid->ProcessTableMessage( msg );
 160     adjustGridColumns( m_grid->GetRect().GetWidth() );
 161 
 162     m_SymbolNameCtrl->SetValue( m_libEntry->GetName() );
 163 
 164     m_DescCtrl->SetValue( m_libEntry->GetDescription() );
 165     m_KeywordCtrl->SetValue( m_libEntry->GetKeyWords() );
 166     m_SelNumberOfUnits->SetValue( m_libEntry->GetUnitCount() );
 167     m_OptionPartsLocked->SetValue( m_libEntry->UnitsLocked() && m_libEntry->GetUnitCount() > 1 );
 168     m_AsConvertButt->SetValue( m_libEntry->HasConversion() );
 169     m_OptionPower->SetValue( m_libEntry->IsPower() );
 170 
 171     m_ShowPinNumButt->SetValue( m_libEntry->ShowPinNumbers() );
 172     m_ShowPinNameButt->SetValue( m_libEntry->ShowPinNames() );
 173     m_PinsNameInsideButt->SetValue( m_libEntry->GetPinNameOffset() != 0 );
 174     m_pinNameOffset.SetValue( m_libEntry->GetPinNameOffset() );
 175 
 176     wxArrayString tmp = m_libEntry->GetFootprints();
 177     m_FootprintFilterListBox->Append( tmp );
 178 
 179     // Populate the list of root parts for inherited objects.

 180     if( m_libEntry->IsAlias() )
 181     {
 182         wxArrayString rootSymbolNames;
 183         wxString libName = m_Parent->GetCurLib();
 184 
 185         // Someone forgot to set the current library in the editor frame window.

 186         wxCHECK( !libName.empty(), false );
 187 
 188         m_Parent->GetLibManager().GetRootSymbolNames( libName, rootSymbolNames );
 189         m_inheritanceSelectCombo->Append( rootSymbolNames );
 190 
 191         PART_SPTR rootPart = m_libEntry->GetParent().lock();
 192 
 193         wxCHECK( rootPart, false );
 194 
 195         wxString parentName = rootPart->GetName();
 196         int selection = m_inheritanceSelectCombo->FindString( parentName );
 197 
 198         wxCHECK( selection != wxNOT_FOUND, false );
 199         m_inheritanceSelectCombo->SetSelection( selection );
 200 
 201         m_lastOpenedPage = 0;
 202     }
 203 
 204     m_NoteBook->SetSelection( (unsigned) m_lastOpenedPage );
 205 
 206     return true;
 207 }

 210 bool DIALOG_EDIT_COMPONENT_IN_LIBRARY::Validate()
 211 {
 212     if( !m_grid->CommitPendingChanges() )
 213         return false;
 214 
 215     // Alias symbol reference can be empty because it inherits from the parent symbol.

 216     if( m_libEntry->IsRoot() &&
 217         !SCH_COMPONENT::IsReferenceStringValid( m_fields->at( REFERENCE ).GetText() ) )
 218     {
 219         if( m_NoteBook->GetSelection() != 0 )
 220             m_NoteBook->SetSelection( 0 );
 221 
 222         m_delayedErrorMessage = _( "References must start with a letter." );
 223         m_delayedFocusGrid = m_grid;
 224         m_delayedFocusColumn = FDC_VALUE;
 225         m_delayedFocusRow = REFERENCE;
 226         m_delayedFocusPage = 0;
 227 
 228         return false;
 229     }
 230 
 231     // Check for missing field names.

 232     for( size_t i = MANDATORY_FIELDS;  i < m_fields->size(); ++i )
 233     {
 234         LIB_FIELD& field = m_fields->at( i );
 235         wxString   fieldName = field.GetName( false );
 236 
 237         if( fieldName.IsEmpty() )
 238         {
 239             if( m_NoteBook->GetSelection() != 0 )
 240                 m_NoteBook->SetSelection( 0 );
 241 
 242             m_delayedErrorMessage = _( "Fields must have a name." );
 243             m_delayedFocusGrid = m_grid;
 244             m_delayedFocusColumn = FDC_NAME;
 245             m_delayedFocusRow = i;
 246             m_delayedFocusPage = 0;
 247 
 248             return false;
 249         }
 250     }
 251 
 252     if( m_SelNumberOfUnits->GetValue() < m_libEntry->GetUnitCount() )
 253     {
 254         if( !IsOK( this, _( "Delete extra units from symbol?" ) ) )
 255             return false;
 256     }
 257 
 258     if( m_AsConvertButt->GetValue() && !m_libEntry->HasConversion() )
 259     {
 260         if( !IsOK( this, _( "Add new pins for alternate body style (DeMorgan) to symbol?" ) ) )
 261             return false;
 262     }
 263     else if( !m_AsConvertButt->GetValue() && m_libEntry->HasConversion() )
 264     {
 265         if( !IsOK( this, _( "Delete alternate body style (DeMorgan) draw items from symbol?" ) ) )
 266             return false;
 267     }
 268 
 269     return true;
 270 }

 273 bool DIALOG_EDIT_COMPONENT_IN_LIBRARY::TransferDataFromWindow()
 274 {
 275     if( !wxDialog::TransferDataFromWindow() )
 276         return false;
 277 
 278     // We need to keep the name and the value the same at the moment!

 279     wxString   newName = m_fields->at( VALUE ).GetText();
 280     wxString   oldName = m_libEntry->GetName();
 281 
 282     if( oldName != newName )
 283     {
 284         wxString libName = m_Parent->GetCurLib();
 285 
 286         if( m_Parent->GetLibManager().PartExists( newName, libName ) )
 287         {
 288             wxString msg;
 289 
 290             msg.Printf( _( "The name '%s' conflicts with an existing entry in the library '%s'." ),
 291                         newName, libName );
 292             DisplayErrorMessage( this, msg );
 293             return false;
 294         }
 295 
 296         m_Parent->SaveCopyInUndoList( m_libEntry, UR_LIB_RENAME );
 297     }
 298     else
 299     {
 300         m_Parent->SaveCopyInUndoList( m_libEntry );
 301     }
 302 
 303     // The Y axis for components in lib is from bottom to top while the screen axis is top

 304     // to bottom: we must change the y coord sign when writing back to the library

 305     for( size_t i = 0; i < m_fields->size(); ++i )
 306     {
 307         wxPoint pos = m_fields->at( i ).GetPosition();
 308         pos.y = -pos.y;
 309         m_fields->at( i ).SetPosition( pos );
 310     }
 311 
 312     m_libEntry->SetFields( *m_fields );
 313 
 314     // We need to keep the name and the value the same at the moment!

 315     m_libEntry->SetName( newName );
 316     m_libEntry->SetDescription( m_DescCtrl->GetValue() );
 317     m_libEntry->SetKeyWords( m_KeywordCtrl->GetValue() );
 318     m_libEntry->SetUnitCount( m_SelNumberOfUnits->GetValue() );
 319     m_libEntry->LockUnits( m_libEntry->GetUnitCount() > 1 && m_OptionPartsLocked->GetValue() );
 320     m_libEntry->SetConversion( m_AsConvertButt->GetValue() );
 321 
 322     if( m_OptionPower->GetValue() )
 323         m_libEntry->SetPower();
 324     else
 325         m_libEntry->SetNormal();
 326 
 327     m_libEntry->SetShowPinNumbers( m_ShowPinNumButt->GetValue() );
 328     m_libEntry->SetShowPinNames( m_ShowPinNameButt->GetValue() );
 329 
 330     if( m_PinsNameInsideButt->GetValue() )
 331     {
 332         int offset = KiROUND( (double) m_pinNameOffset.GetValue() );
 333 
 334         // We interpret an offset of 0 as "outside", so make sure it's non-zero

 335         m_libEntry->SetPinNameOffset( offset == 0 ? 20 : offset );
 336     }
 337     else
 338     {
 339         m_libEntry->SetPinNameOffset( 0 );   // pin text outside the body (name is on the pin)

 340     }
 341 
 342     m_libEntry->SetFootprintFilters( m_FootprintFilterListBox->GetStrings() );
 343 
 344     m_Parent->UpdateAfterSymbolProperties( &oldName );
 345 
 346     // It's possible that the symbol being edited has no pins, in which case there may be no

 347     // alternate body style objects causing #LIB_PART::HasCoversion() to always return false.

 348     // This allows the user to edit the alternate body style just in case this condition occurs.

 349     m_Parent->SetShowDeMorgan( m_AsConvertButt->GetValue() );
 350 
 351     return true;
 352 }

 355 void DIALOG_EDIT_COMPONENT_IN_LIBRARY::OnGridCellChanging( wxGridEvent& event )
 356 {
 357     wxGridCellEditor* editor = m_grid->GetCellEditor( event.GetRow(), event.GetCol() );
 358     wxControl* control = editor->GetControl();
 359 
 360     if( control && control->GetValidator() && !control->GetValidator()->Validate( control ) )
 361     {
 362         event.Veto();
 363 
 364         m_delayedFocusGrid = m_grid;
 365         m_delayedFocusRow = event.GetRow();
 366         m_delayedFocusColumn = event.GetCol();
 367         m_delayedFocusPage = 0;
 368     }
 369     else if( event.GetCol() == FDC_NAME )
 370     {
 371         wxString newName = event.GetString();
 372 
 373         for( int i = 0; i < m_grid->GetNumberRows(); ++i )
 374         {
 375             if( i == event.GetRow() )
 376                 continue;
 377 
 378             if( newName.CmpNoCase( m_grid->GetCellValue( i, FDC_NAME ) ) == 0 )
 379             {
 380                 DisplayError( this, wxString::Format( _( "The name '%s' is already in use." ),
 381                                                       newName ) );
 382                 event.Veto();
 383                 m_delayedFocusRow = event.GetRow();
 384                 m_delayedFocusColumn = event.GetCol();
 385             }
 386         }
 387     }
 388     else if( event.GetRow() == VALUE && event.GetCol() == FDC_VALUE )
 389         m_SymbolNameCtrl->ChangeValue( event.GetString() );
 390 
 391     editor->DecRef();
 392 }

 439 void DIALOG_EDIT_COMPONENT_IN_LIBRARY::OnDeleteField( wxCommandEvent& event )
 440 {
 441     int curRow = m_grid->GetGridCursorRow();
 442 
 443     if( curRow < 0 )
 444         return;
 445     else if( curRow < MANDATORY_FIELDS )
 446     {
 447         DisplayError( this, wxString::Format( _( "The first %d fields are mandatory." ),
 448                                               MANDATORY_FIELDS ) );
 449         return;
 450     }
 451 
 452     m_grid->CommitPendingChanges( true /* quiet mode */ );
 453 
 454     m_fields->erase( m_fields->begin() + curRow );
 455 
 456     // notify the grid

 457     wxGridTableMessage msg( m_fields, wxGRIDTABLE_NOTIFY_ROWS_DELETED, curRow, 1 );
 458     m_grid->ProcessTableMessage( msg );
 459 
 460     if( m_grid->GetNumberRows() > 0 )
 461     {
 462         m_grid->MakeCellVisible( std::max( 0, curRow-1 ), m_grid->GetGridCursorCol() );
 463         m_grid->SetGridCursor( std::max( 0, curRow-1 ), m_grid->GetGridCursorCol() );
 464     }
 465 }

 468 void DIALOG_EDIT_COMPONENT_IN_LIBRARY::OnMoveUp( wxCommandEvent& event )
 469 {
 470     if( !m_grid->CommitPendingChanges() )
 471         return;
 472 
 473     int i = m_grid->GetGridCursorRow();
 474 
 475     if( i > MANDATORY_FIELDS )
 476     {
 477         LIB_FIELD tmp = m_fields->at( (unsigned) i );
 478         m_fields->erase( m_fields->begin() + i, m_fields->begin() + i + 1 );
 479         m_fields->insert( m_fields->begin() + i - 1, tmp );
 480         m_grid->ForceRefresh();
 481 
 482         m_grid->SetGridCursor( i - 1, m_grid->GetGridCursorCol() );
 483         m_grid->MakeCellVisible( m_grid->GetGridCursorRow(), m_grid->GetGridCursorCol() );
 484     }
 485     else
 486         wxBell();
 487 }

 505         m_grid->MakeCellVisible( m_grid->GetGridCursorRow(), m_grid->GetGridCursorCol() );
 506     }

 512 void DIALOG_EDIT_COMPONENT_IN_LIBRARY::OnEditSpiceModel( wxCommandEvent& event )
 513 {
 514 #ifdef KICAD_SPICE
 515     int diff = m_fields->size();
 516     auto cmp = SCH_COMPONENT( *m_libEntry, m_libEntry->GetLibId(), nullptr );
 517 
 518     DIALOG_SPICE_MODEL dialog( this, cmp, m_fields );
 519 
 520     if( dialog.ShowModal() != wxID_OK )
 521         return;
 522 
 523     diff = (int) m_fields->size() - diff;
 524 
 525     if( diff > 0 )
 526     {
 527         wxGridTableMessage msg( m_fields, wxGRIDTABLE_NOTIFY_ROWS_APPENDED, diff );
 528         m_grid->ProcessTableMessage( msg );
 529     }
 530     else if( diff < 0 )
 531     {
 532         wxGridTableMessage msg( m_fields, wxGRIDTABLE_NOTIFY_ROWS_DELETED, 0, -diff );
 533         m_grid->ProcessTableMessage( msg );
 534     }
 535 
 536     m_grid->ForceRefresh();
 537 #endif /* KICAD_SPICE */
 538 }

 605         if( dlg.ShowModal() == wxID_OK && !dlg.GetValue().IsEmpty() )
 606             m_FootprintFilterListBox->SetString( (unsigned) idx, dlg.GetValue() );
 607     }

 611 void DIALOG_EDIT_COMPONENT_IN_LIBRARY::adjustGridColumns( int aWidth )
 612 {
 613     m_width = aWidth;
 614 
 615     // Account for scroll bars

 616     aWidth -= ( m_grid->GetSize().x - m_grid->GetClientSize().x );
 617 
 618     m_grid->AutoSizeColumn( FDC_NAME );
 619 
 620     int fixedColsWidth = m_grid->GetColSize( FDC_NAME );
 621 
 622     for( int i = 2; i < m_grid->GetNumberCols(); i++ )
 623         fixedColsWidth += m_grid->GetColSize( i );
 624 
 625     m_grid->SetColSize( FDC_VALUE, aWidth - fixedColsWidth );
 626 }

 629 void DIALOG_EDIT_COMPONENT_IN_LIBRARY::OnUpdateUI( wxUpdateUIEvent& event )
 630 {
 631     m_OptionPartsLocked->Enable( m_SelNumberOfUnits->GetValue() > 1 );
 632     m_pinNameOffset.Enable( m_PinsNameInsideButt->GetValue() );
 633 
 634     if( m_grid->IsCellEditControlShown() )
 635     {
 636         int row = m_grid->GetGridCursorRow();
 637         int col = m_grid->GetGridCursorCol();
 638 
 639         if( row == VALUE && col == FDC_VALUE )
 640         {
 641             wxGridCellEditor* editor = m_grid->GetCellEditor( row, col );
 642             m_SymbolNameCtrl->ChangeValue( editor->GetValue() );
 643             editor->DecRef();
 644         }
 645     }
 646 
 647     // Handle shown columns changes

 648     wxString shownColumns = m_grid->GetShownColumns();
 649 
 650     if( shownColumns != m_shownColumns )
 651     {
 652         m_shownColumns = shownColumns;
 653 
 654         if( !m_grid->IsCellEditControlShown() )
 655             adjustGridColumns( m_grid->GetRect().GetWidth() );
 656     }
 657 
 658     // Handle a delayed focus.  The delay allows us to:

 659     // a) change focus when the error was triggered from within a killFocus handler

 660     // b) show the correct notebook page in the background before the error dialog comes up

 661     //    when triggered from an OK or a notebook page change

 662 
 663     if( m_delayedFocusPage >= 0 && m_NoteBook->GetSelection() != m_delayedFocusPage )
 664     {
 665         m_NoteBook->SetSelection( (unsigned) m_delayedFocusPage );
 666         m_delayedFocusPage = -1;
 667     }
 668 
 669     if( !m_delayedErrorMessage.IsEmpty() )
 670     {
 671         // We will re-enter this routine when the error dialog is displayed, so make

 672         // sure we don't keep putting up more dialogs.

 673         wxString msg = m_delayedErrorMessage;
 674         m_delayedErrorMessage = wxEmptyString;
 675 
 676         // Do not use DisplayErrorMessage(); it screws up window order on Mac

 677         DisplayError( nullptr, msg );
 678     }
 679 
 680     if( m_delayedFocusCtrl )
 681     {
 682         m_delayedFocusCtrl->SetFocus();
 683 
 684         if( auto textEntry = dynamic_cast<wxTextEntry*>( m_delayedFocusCtrl ) )
 685             textEntry->SelectAll();
 686 
 687         m_delayedFocusCtrl = nullptr;
 688     }
 689     else if( m_delayedFocusGrid )
 690     {
 691         m_delayedFocusGrid->SetFocus();
 692         m_delayedFocusGrid->MakeCellVisible( m_delayedFocusRow, m_delayedFocusColumn );
 693         m_delayedFocusGrid->SetGridCursor( m_delayedFocusRow, m_delayedFocusColumn );
 694 
 695         m_delayedFocusGrid->EnableCellEditControl( true );
 696         m_delayedFocusGrid->ShowCellEditControl();
 697 
 698         m_delayedFocusGrid = nullptr;
 699         m_delayedFocusRow = -1;
 700         m_delayedFocusColumn = -1;
 701     }
 702 }
