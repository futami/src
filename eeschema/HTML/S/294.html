  41 /**

  42  * A set of bits which control the totality of the tree built by makeRoot()

  43  */
  44 enum GNL_T
  45 {
  46     GNL_LIBRARIES   = 1 << 0,
  47     GNL_COMPONENTS  = 1 << 1,
  48     GNL_PARTS       = 1 << 2,
  49     GNL_HEADER      = 1 << 3,
  50     GNL_NETS        = 1 << 4,
  51     GNL_OPT_KICAD   = 1 << 5,
  52     GNL_OPT_BOM     = 1 << 6,
  53 };
  54 
  55 
  56 /**

  57  * Generate a generic XML based netlist file.

  58  *

  59  * This allows using XSLT or other methods to transform the XML to other netlist formats

  60  * outside of the C++ codebase.

  61  */
  62 class NETLIST_EXPORTER_GENERIC : public NETLIST_EXPORTER
  63 {
  64 private:
  65     std::set< wxString >  m_libraries;    ///< Set of library nicknames.

  66 
  67 public:
  68     NETLIST_EXPORTER_GENERIC( SCHEMATIC* aSchematic ) :
  69         NETLIST_EXPORTER( aSchematic )
  70     {}
  71 
  72     /**

  73      * Write generic netlist to \a aOutFileName.

  74      *

  75      * @param aOutFileName is the file name to write.

  76      * @param aNetlistOptions are the options used to control the netlist output.

  77      *

  78      * @return true if the netlist was written successfully.

  79      */
  80     bool WriteNetlist( const wxString& aOutFileName, unsigned aNetlistOptions ) override;
  81 
  82 #define GNL_ALL     ( GNL_LIBRARIES | GNL_COMPONENTS | GNL_PARTS | GNL_HEADER | GNL_NETS )
  83 
  84 protected:
  85    /**

  86      * A convenience function that creates a new XNODE with an optional textual child.

  87      * It also provides some insulation from a possible change in XML library.

  88      *

  89      * @param aName is the name to associate with a new node of type wxXML_ELEMENT_NODE.

  90      * @param aTextualContent is optional, and if given is the text to include in a child

  91      *   of the returned node, and has type wxXML_TEXT_NODE.

  92      */
  93     XNODE* node( const wxString& aName, const wxString& aTextualContent = wxEmptyString );
  94 
  95     /**

  96      * Build the entire document tree for the generic export.  This is factored

  97      * out here so we can write the tree in either S-expression file format

  98      * or in XML if we put the tree built here into a wxXmlDocument.

  99      * @param aCtl - a bitset or-ed together from GNL_ENUM values

 100      * @return XNODE* - the root nodes

 101      */
 102     XNODE* makeRoot( int aCtl = GNL_ALL );
 103 
 104     /**

 105      * @return XNODE* - returns a sub-tree holding all the schematic components.

 106      */
 107     XNODE* makeComponents( unsigned aCtl );
 108 
 109     /**

 110      * Fills out a project "design" header into an XML node.

 111      * @return XNODE* - the design header

 112      */
 113     XNODE* makeDesignHeader();
 114 
 115     /**

 116      * Fill out an XML node with the unique library parts and returns it.

 117      * @return XNODE* - the library parts nodes

 118      */
 119     XNODE* makeLibParts();
 120 
 121     /**

 122      * Fill out an XML node with a list of nets and returns it.

 123      * @return XNODE* - the list of nets nodes

 124      */
 125     XNODE* makeListOfNets();
 126 
 127     /**

 128      * Fill out an XML node with a list of used libraries and returns it.

 129      * Must have called makeGenericLibParts() before this function.

 130      * @return XNODE* - the library nodes

 131      */
 132     XNODE* makeLibraries();
 133 
 134     void addComponentFields(  XNODE* xcomp, SCH_COMPONENT* comp, SCH_SHEET_PATH* aSheet );
 135 };
