  69 SIM_NOPLOT_PANEL::SIM_NOPLOT_PANEL( SIM_TYPE aType, wxWindow* parent, wxWindowID id,
  70         const wxPoint& pos, const wxSize& size, long style, const wxString& name )
  71         : SIM_PANEL_BASE( aType, parent, id, pos, size, style, name )
  72 {
  73     m_sizer = new wxBoxSizer( wxVERTICAL );
  74     m_sizer->Add( 0, 1, 1, wxEXPAND, 5 );
  75 
  76     m_textInfo = new wxStaticText( dynamic_cast<wxWindow*>( this ), wxID_ANY, "", wxDefaultPosition,
  77             wxDefaultSize, wxALL | wxEXPAND | wxALIGN_CENTER_HORIZONTAL );
  78     m_textInfo->SetFont( wxFont( wxNORMAL_FONT->GetPointSize(), wxFONTFAMILY_DEFAULT,
  79             wxFONTSTYLE_NORMAL, wxFONTWEIGHT_BOLD, false, wxEmptyString ) );
  80     m_textInfo->SetForegroundColour( wxSystemSettings::GetColour( wxSYS_COLOUR_GRAYTEXT ) );
  81 
  82     //ST_UNKNOWN serves purpose of a welcome panel

  83     m_textInfo->SetLabel(
  84             ( aType == ST_UNKNOWN ) ?
  85                     _( "Start the simulation by clicking the Run Simulation button" ) :
  86                     _( "This simulation provide no plots. Please refer to console window for results" ) );
  87 
  88     m_sizer->Add( m_textInfo, 1, wxALL | wxEXPAND, 5 );
  89     m_sizer->Add( 0, 1, 1, wxEXPAND, 5 );
  90 
  91     dynamic_cast<wxWindow*>( this )->SetSizer( m_sizer );
  92 }
