  40 /**

  41  * Base class for a bus or wire entry.

  42  */
  43 class SCH_BUS_ENTRY_BASE : public SCH_ITEM
  44 {
  45 protected:
  46     wxPoint m_pos;
  47     wxSize  m_size;
  48     bool m_isDanglingStart;
  49     bool m_isDanglingEnd;
  50     STROKE_PARAMS m_stroke;
  51 
  52 public:
  53     SCH_BUS_ENTRY_BASE( KICAD_T aType, const wxPoint& pos = wxPoint( 0, 0 ), bool aFlipY = false );
  54 
  55     bool IsDanglingStart() const { return m_isDanglingStart; }
  56     bool IsDanglingEnd() const { return m_isDanglingEnd; }
  57 
  58     // Do not create a copy constructor.  The one generated by the compiler is adequate.

  59 
  60     ~SCH_BUS_ENTRY_BASE() { }
  61 
  62     /**

  63      * Return true for items which are moved with the anchor point at mouse cursor

  64      *  and false for items moved with no reference to anchor

  65      * @return false for a bus entry

  66      */
  67     bool IsMovableFromAnchorPoint() override { return false; }
  68 
  69     wxPoint m_End() const;
  70 
  71     wxSize GetSize() const { return m_size; }
  72 
  73     void SetSize( const wxSize& aSize ) { m_size = aSize; }
  74 
  75     void SetPenWidth( int aWidth ) { m_stroke.SetWidth( aWidth ); }
  76 
  77     virtual bool HasLineStroke() const override { return true; }
  78     virtual STROKE_PARAMS GetStroke() const override { return m_stroke; }
  79     virtual void SetStroke( const STROKE_PARAMS& aStroke ) override { m_stroke = aStroke; }
  80 
  81     PLOT_DASH_TYPE GetStrokeStyle() const;

  83 
  84     COLOR4D GetStrokeColor() const;

  86 
  87     void SwapData( SCH_ITEM* aItem ) override;
  88 
  89     void ViewGetLayers( int aLayers[], int& aCount ) const override;
  90 
  91     void Print( RENDER_SETTINGS* aSettings, const wxPoint& aOffset ) override;
  92 
  93     const EDA_RECT GetBoundingBox() const override;
  94 
  95     void Move( const wxPoint& aMoveVector ) override
  96     {
  97         m_pos += aMoveVector;
  98     }
  99 
 100     void MirrorY( int aYaxis_position ) override;
 101     void MirrorX( int aXaxis_position ) override;
 102     void Rotate( wxPoint aPosition ) override;
 103 
 104     bool IsDangling() const override;
 105 
 106     bool IsConnectable() const override { return true; }
 107 
 108     void GetConnectionPoints( std::vector< wxPoint >& aPoints ) const override;

 128 };

 130 /**

 131  * Class for a wire to bus entry.

 132  */
 133 class SCH_BUS_WIRE_ENTRY : public SCH_BUS_ENTRY_BASE
 134 {
 135 public:
 136     SCH_BUS_WIRE_ENTRY( const wxPoint& pos = wxPoint( 0, 0 ), bool aFlipY = false );
 137 
 138     ~SCH_BUS_WIRE_ENTRY() { }
 139 
 140     static inline bool ClassOf( const EDA_ITEM* aItem )
 141     {
 142         return aItem && SCH_BUS_WIRE_ENTRY_T == aItem->Type();
 143     }
 144 
 145     wxString GetClass() const override
 146     {
 147         return wxT( "SCH_BUS_WIRE_ENTRY" );
 148     }
 149 
 150     int GetPenWidth() const override;
 151 
 152     void GetEndPoints( std::vector< DANGLING_END_ITEM >& aItemList ) override;
 153 
 154     bool CanConnect( const SCH_ITEM* aItem ) const override
 155     {
 156         return aItem->Type() == SCH_LINE_T &&
 157                 ( aItem->GetLayer() == LAYER_WIRE || aItem->GetLayer() == LAYER_BUS );
 158     }
 159 
 160     wxString GetSelectMenuText( EDA_UNITS aUnits ) const override;
 161 
 162     EDA_ITEM* Clone() const override;
 163 
 164     virtual bool ConnectionPropagatesTo( const EDA_ITEM* aItem ) const override;
 165 
 166     BITMAP_DEF GetMenuImage() const override;
 167 
 168     bool UpdateDanglingState( std::vector<DANGLING_END_ITEM>& aItemList,
 169                               const SCH_SHEET_PATH* aPath = nullptr ) override;
 170 
 171     /**

 172      * Pointer to the bus item (usually a bus wire) connected to this bus-wire

 173      * entry, if it is connected to one.

 174      */
 175     SCH_ITEM* m_connected_bus_item;
 176 };
