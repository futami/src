 139 int SCH_SHEET_PATH::Cmp( const SCH_SHEET_PATH& aSheetPathToTest ) const
 140 {
 141     if( size() > aSheetPathToTest.size() )
 142         return 1;
 143 
 144     if( size() < aSheetPathToTest.size() )
 145         return -1;
 146 
 147     //otherwise, same number of sheets.

 148     for( unsigned i = 0; i < size(); i++ )
 149     {
 150         if( at( i )->m_Uuid < aSheetPathToTest.at( i )->m_Uuid )
 151             return -1;
 152 
 153         if( at( i )->m_Uuid != aSheetPathToTest.at( i )->m_Uuid )
 154             return 1;
 155     }
 156 
 157     return 0;
 158 }

 192 wxString SCH_SHEET_PATH::PathAsString() const
 193 {
 194     wxString s;
 195 
 196     s = wxT( "/" );     // This is the root path

 197 
 198     // Start at 1 to avoid the root sheet, which does not need to be added to the path.

 199     // It's timestamp changes anyway.

 200     for( unsigned i = 1; i < size(); i++ )
 201         s += at( i )->m_Uuid.AsString() + "/";
 202 
 203     return s;

 257 void SCH_SHEET_PATH::GetComponents( SCH_REFERENCE_LIST& aReferences, bool aIncludePowerSymbols,
 258                                     bool aForceIncludeOrphanComponents ) const
 259 {
 260     for( auto item : LastScreen()->Items().OfType( SCH_COMPONENT_T ) )
 261     {
 262         auto component = static_cast<SCH_COMPONENT*>( item );
 263 
 264         // Skip pseudo components, which have a reference starting with #.  This mainly

 265         // affects power symbols.

 266         if( aIncludePowerSymbols || component->GetRef( this )[0] != wxT( '#' ) )
 267         {
 268             LIB_PART* part = component->GetPartRef().get();
 269 
 270             if( part || aForceIncludeOrphanComponents )
 271             {
 272                 SCH_REFERENCE schReference( component, part, *this );
 273 
 274                 schReference.SetSheetNumber( m_pageNumber );
 275                 aReferences.AddItem( schReference );
 276             }
 277         }
 278     }
 279 }

 282 void SCH_SHEET_PATH::GetMultiUnitComponents( SCH_MULTI_UNIT_REFERENCE_MAP& aRefList,
 283                                              bool aIncludePowerSymbols ) const
 284 {
 285     for( auto item : LastScreen()->Items().OfType( SCH_COMPONENT_T ) )
 286     {
 287         auto component = static_cast<SCH_COMPONENT*>( item );
 288 
 289         // Skip pseudo components, which have a reference starting with #.  This mainly

 290         // affects power symbols.

 291         if( !aIncludePowerSymbols && component->GetRef( this )[0] == wxT( '#' ) )
 292             continue;
 293 
 294         LIB_PART* part = component->GetPartRef().get();
 295 
 296         if( part && part->GetUnitCount() > 1 )
 297         {
 298             SCH_REFERENCE schReference = SCH_REFERENCE( component, part, *this );
 299             schReference.SetSheetNumber( m_pageNumber );
 300             wxString reference_str = schReference.GetRef();
 301 
 302             // Never lock unassigned references

 303             if( reference_str[reference_str.Len() - 1] == '?' )
 304                 continue;
 305 
 306             aRefList[reference_str].AddItem( schReference );
 307         }
 308     }
 309 }

 330 bool SCH_SHEET_PATH::TestForRecursion( const wxString& aSrcFileName, const wxString& aDestFileName )
 331 {
 332     auto pair = std::make_pair( aSrcFileName, aDestFileName );
 333 
 334     if( m_recursion_test_cache.count( pair ) )
 335         return m_recursion_test_cache.at( pair );
 336 
 337     SCHEMATIC* sch = LastScreen()->Schematic();
 338 
 339     wxCHECK_MSG( sch, false, "No SCHEMATIC found in SCH_SHEET_PATH::TestForRecursion!" );
 340 
 341     wxFileName rootFn = sch->GetFileName();
 342     wxFileName srcFn = aSrcFileName;
 343     wxFileName destFn = aDestFileName;
 344 
 345     if( srcFn.IsRelative() )
 346         srcFn.MakeAbsolute( rootFn.GetPath() );
 347 
 348     if( destFn.IsRelative() )
 349         destFn.MakeAbsolute( rootFn.GetPath() );
 350 
 351 
 352     // The source and destination sheet file names cannot be the same.

 353     if( srcFn == destFn )
 354     {
 355         m_recursion_test_cache[pair] = true;
 356         return true;
 357     }
 358 
 359     /// @todo Store sheet file names with full path, either relative to project path

 360     ///       or absolute path.  The current design always assumes subsheet files are

 361     ///       located in the project folder which may or may not be desirable.

 362     unsigned i = 0;
 363 
 364     while( i < size() )
 365     {
 366         wxFileName cmpFn = at( i )->GetFileName();
 367 
 368         if( cmpFn.IsRelative() )
 369             cmpFn.MakeAbsolute( rootFn.GetPath() );
 370 
 371         // Test if the file name of the destination sheet is in anywhere in this sheet path.

 372         if( cmpFn == destFn )
 373             break;
 374 
 375         i++;
 376     }
 377 
 378     // The destination sheet file name was not found in the sheet path or the destination

 379     // sheet file name is the root sheet so no recursion is possible.

 380     if( i >= size() || i == 0 )
 381     {
 382         m_recursion_test_cache[pair] = false;
 383         return false;
 384     }
 385 
 386     // Walk back up to the root sheet to see if the source file name is already a parent in

 387     // the sheet path.  If so, recursion will occur.

 388     do
 389     {
 390         i -= 1;
 391 
 392         wxFileName cmpFn = at( i )->GetFileName();
 393 
 394         if( cmpFn.IsRelative() )
 395             cmpFn.MakeAbsolute( rootFn.GetPath() );
 396 
 397         if( cmpFn == srcFn )
 398         {
 399             m_recursion_test_cache[pair] = true;
 400             return true;
 401         }
 402 
 403     } while( i != 0 );
 404 
 405     // The source sheet file name is not a parent of the destination sheet file name.

 406     m_recursion_test_cache[pair] = false;
 407     return false;
 408 }

 418 void SCH_SHEET_LIST::BuildSheetList( SCH_SHEET* aSheet )
 419 {
 420     wxCHECK_RET( aSheet != NULL, wxT( "Cannot build sheet list from undefined sheet." ) );
 421 
 422     std::vector<SCH_SHEET*> badSheets;
 423 
 424     m_currentSheetPath.push_back( aSheet );
 425 
 426     /**

 427      * @todo:  Schematic page number is currently a left over relic and is generated as

 428      *         SCH_SHEET_PATH object is pushed to the list.  This only has meaning when

 429      *         entire hierarchy is created from the root sheet down.

 430      */
 431     m_currentSheetPath.SetPageNumber( size() + 1 );
 432     push_back( m_currentSheetPath );
 433 
 434     if( m_currentSheetPath.LastScreen() )
 435     {
 436         std::vector<SCH_ITEM*> childSheets;
 437         m_currentSheetPath.LastScreen()->GetSheets( &childSheets );
 438 
 439         for( SCH_ITEM* item : childSheets )
 440         {
 441             auto sheet = static_cast<SCH_SHEET*>( item );
 442 
 443             if( !m_currentSheetPath.TestForRecursion(
 444                         sheet->GetFileName(), aSheet->GetFileName() ) )
 445                 BuildSheetList( sheet );
 446             else
 447                 badSheets.push_back( sheet );
 448         }
 449     }
 450 
 451     for( auto sheet : badSheets )
 452     {
 453         m_currentSheetPath.LastScreen()->Remove( sheet );
 454         m_currentSheetPath.LastScreen()->SetModify();
 455     }
 456 
 457     m_currentSheetPath.pop_back();
 458 }

 495 SCH_ITEM* SCH_SHEET_LIST::GetItem( const KIID& aID, SCH_SHEET_PATH* aPathOut )
 496 {
 497     for( const SCH_SHEET_PATH& sheet : *this )
 498     {
 499         SCH_SCREEN* screen = sheet.LastScreen();
 500 
 501         for( SCH_ITEM* aItem : screen->Items() )
 502         {
 503             if( aItem->m_Uuid == aID )
 504             {
 505                 if( aPathOut )
 506                     *aPathOut = sheet;
 507 
 508                 return aItem;
 509             }
 510             else if( aItem->Type() == SCH_COMPONENT_T )
 511             {
 512                 SCH_COMPONENT* comp = static_cast<SCH_COMPONENT*>( aItem );
 513 
 514                 for( SCH_FIELD& field : comp->GetFields() )
 515                 {
 516                     if( field.m_Uuid == aID )
 517                     {
 518                         if( aPathOut )
 519                             *aPathOut = sheet;
 520 
 521                         return &field;
 522                     }
 523                 }
 524 
 525                 for( SCH_PIN* pin : comp->GetSchPins() )
 526                 {
 527                     if( pin->m_Uuid == aID )
 528                     {
 529                         if( aPathOut )
 530                             *aPathOut = sheet;
 531 
 532                         return pin;
 533                     }
 534                 }
 535             }
 536             else if( aItem->Type() == SCH_SHEET_T )
 537             {
 538                 SCH_SHEET* sch_sheet = static_cast<SCH_SHEET*>( aItem );
 539 
 540                 for( SCH_FIELD& field : sch_sheet->GetFields() )
 541                 {
 542                     if( field.m_Uuid == aID )
 543                     {
 544                         if( aPathOut )
 545                             *aPathOut = sheet;
 546                         return &field;
 547                     }
 548                 }
 549 
 550                 for( SCH_SHEET_PIN* pin : sch_sheet->GetPins() )
 551                 {
 552                     if( pin->m_Uuid == aID )
 553                     {
 554                         if( aPathOut )
 555                             *aPathOut = sheet;
 556 
 557                         return pin;
 558                     }
 559                 }
 560             }
 561         }
 562     }
 563 
 564     // Not found; weak reference has been deleted.

 565     return DELETED_SHEET_ITEM::GetInstance();
 566 }

 569 void SCH_SHEET_LIST::FillItemMap( std::map<KIID, EDA_ITEM*>& aMap )
 570 {
 571     for( const SCH_SHEET_PATH& sheet : *this )
 572     {
 573         SCH_SCREEN* screen = sheet.LastScreen();
 574 
 575         for( SCH_ITEM* aItem : screen->Items() )
 576         {
 577             aMap[ aItem->m_Uuid ] = aItem;
 578 
 579             if( aItem->Type() == SCH_COMPONENT_T )
 580             {
 581                 SCH_COMPONENT* comp = static_cast<SCH_COMPONENT*>( aItem );
 582 
 583                 for( SCH_FIELD& field : comp->GetFields() )
 584                     aMap[ field.m_Uuid ] = &field;
 585 
 586                 for( SCH_PIN* pin : comp->GetSchPins() )
 587                     aMap[ pin->m_Uuid ] = pin;
 588             }
 589             else if( aItem->Type() == SCH_SHEET_T )
 590             {
 591                 SCH_SHEET* sch_sheet = static_cast<SCH_SHEET*>( aItem );
 592 
 593                 for( SCH_FIELD& field : sch_sheet->GetFields() )
 594                     aMap[ field.m_Uuid ] = &field;
 595 
 596                 for( SCH_SHEET_PIN* pin : sch_sheet->GetPins() )
 597                     aMap[ pin->m_Uuid ] = pin;
 598             }
 599         }
 600     }
 601 }

 604 void SCH_SHEET_LIST::AnnotatePowerSymbols()
 605 {
 606     // List of reference for power symbols

 607     SCH_REFERENCE_LIST references;
 608 
 609     // Map of locked components (not used, but needed by Annotate()

 610     SCH_MULTI_UNIT_REFERENCE_MAP lockedComponents;
 611 
 612     // Build the list of power components:

 613     for( SCH_SHEET_PATH& sheet : *this )
 614     {
 615         for( auto item : sheet.LastScreen()->Items().OfType( SCH_COMPONENT_T ) )
 616         {
 617             auto      component = static_cast<SCH_COMPONENT*>( item );
 618             LIB_PART* part = component->GetPartRef().get();
 619 
 620             if( !part || !part->IsPower() )
 621                 continue;
 622 
 623             if( part )
 624             {
 625                 SCH_REFERENCE schReference( component, part, sheet );
 626                 references.AddItem( schReference );
 627             }
 628         }
 629     }
 630 
 631     // Find duplicate, and silently clear annotation of duplicate

 632     std::map<wxString, int> ref_list;   // stores the existing references

 633 
 634     for( unsigned ii = 0; ii< references.GetCount(); ++ii )
 635     {
 636         wxString curr_ref = references[ii].GetRef();
 637 
 638         if( ref_list.find( curr_ref ) == ref_list.end() )
 639         {
 640             ref_list[curr_ref] = ii;
 641             continue;
 642         }
 643 
 644         // Possible duplicate, if the ref ends by a number:

 645         if( curr_ref.Last() < '0' && curr_ref.Last() > '9' )
 646             continue;   // not annotated

 647 
 648         // Duplicate: clear annotation by removing the number ending the ref

 649         while( curr_ref.Last() >= '0' && curr_ref.Last() <= '9' )
 650             curr_ref.RemoveLast();
 651 
 652         references[ii].SetRef( curr_ref );
 653     }
 654 
 655 
 656     // Break full components reference in name (prefix) and number:

 657     // example: IC1 become IC, and 1

 658     references.SplitReferences();
 659 
 660     // Ensure all power symbols have the reference starting by '#'

 661     // (No sure this is really useful)

 662     for( unsigned ii = 0; ii< references.GetCount(); ++ii )
 663     {
 664         if( references[ii].GetRef()[0] != '#' )
 665         {
 666             wxString new_ref = "#" + references[ii].GetRef();
 667             references[ii].SetRef( new_ref );
 668         }
 669     }
 670 
 671     // Recalculate and update reference numbers in schematic

 672     references.Annotate( false, 0, 100, lockedComponents );
 673     references.UpdateAnnotation();
 674 }

 687 {
 688     for( SCH_SHEET_PATHS::const_iterator it = begin(); it != end(); ++it )

 707 bool SCH_SHEET_LIST::SetComponentFootprint( const wxString& aReference,
 708                                             const wxString& aFootPrint, bool aSetVisible )
 709 {
 710     bool found = false;
 711 
 712     for( SCH_SHEET_PATH& sheet : *this )
 713         found = sheet.SetComponentFootprint( aReference, aFootPrint, aSetVisible );
 714 
 715     return found;
 716 }

 719 bool SCH_SHEET_LIST::TestForRecursion( const SCH_SHEET_LIST& aSrcSheetHierarchy,
 720                                        const wxString& aDestFileName )
 721 {
 722     if( empty() )
 723         return false;
 724 
 725     SCHEMATIC* sch = at( 0 ).LastScreen()->Schematic();
 726 
 727     wxCHECK_MSG( sch, false, "No SCHEMATIC found in SCH_SHEET_LIST::TestForRecursion!" );
 728 
 729     wxFileName rootFn = sch->GetFileName();
 730     wxFileName destFn = aDestFileName;
 731 
 732     if( destFn.IsRelative() )
 733         destFn.MakeAbsolute( rootFn.GetPath() );
 734 
 735     // Test each SCH_SHEET_PATH in this SCH_SHEET_LIST for potential recursion.

 736     for( unsigned i = 0; i < size(); i++ )
 737     {
 738         // Test each SCH_SHEET_PATH in the source sheet.

 739         for( unsigned j = 0; j < aSrcSheetHierarchy.size(); j++ )
 740         {
 741             const SCH_SHEET_PATH* sheetPath = &aSrcSheetHierarchy[j];
 742 
 743             for( unsigned k = 0; k < sheetPath->size(); k++ )
 744             {
 745                 if( at( i ).TestForRecursion( sheetPath->GetSheet( k )->GetFileName(),
 746                                               aDestFileName ) )
 747                     return true;
 748             }
 749         }
 750     }
 751 
 752     // The source sheet file can safely be added to the destination sheet file.

 753     return false;
 754 }

 769 void SCH_SHEET_LIST::UpdateSymbolInstances(
 770         const std::vector<COMPONENT_INSTANCE_REFERENCE>& aSymbolInstances )
 771 {
 772     SCH_REFERENCE_LIST symbolInstances;
 773 
 774     GetComponents( symbolInstances, true, true );
 775 
 776     std::map<KIID_PATH, wxString> pathNameCache;
 777 
 778     // Calculating the name of a path is somewhat expensive; on large designs with many components

 779     // this can blow up to a serious amount of time when loading the schematic

 780     auto getName =
 781             [&pathNameCache]( const KIID_PATH& aPath ) -> const wxString&
 782             {
 783                 if( pathNameCache.count( aPath ) )
 784                     return pathNameCache.at( aPath );
 785 
 786                 pathNameCache[aPath] = aPath.AsString();
 787                 return pathNameCache[aPath];
 788             };
 789 
 790     for( size_t i = 0; i < symbolInstances.GetCount(); i++ )
 791     {
 792         // The instance paths are stored in the file sans root path so the comparison

 793         // should not include the root path.

 794         wxString path = symbolInstances[i].GetPath();
 795 
 796         auto it = std::find_if( aSymbolInstances.begin(), aSymbolInstances.end(),
 797                     [ path, &getName ]( const COMPONENT_INSTANCE_REFERENCE& r ) -> bool
 798                     {
 799                         return path == getName( r.m_Path );
 800                     }
 801                 );
 802 
 803         if( it == aSymbolInstances.end() )
 804         {
 805             wxLogTrace( traceSchSheetPaths, "No symbol instance found for path \"%s\"", path );
 806             continue;
 807         }
 808 
 809         SCH_COMPONENT* symbol = symbolInstances[i].GetComp();
 810 
 811         wxCHECK2( symbol, continue );
 812 
 813         // Symbol instance paths are stored and looked up in memory with the root path so use

 814         // the full path here.

 815         symbol->AddHierarchicalReference( symbolInstances[i].GetSheetPath().Path(),
 816                 it->m_Reference, it->m_Unit );
 817         symbol->GetField( REFERENCE )->SetText( it->m_Reference );
 818     }
 819 }

 833 void SCH_SHEET_LIST::ReplaceLegacySheetPaths( const std::vector<KIID_PATH>& aOldSheetPaths )
 834 {
 835     wxCHECK( size() == aOldSheetPaths.size(), /* void */ );
 836 
 837     for( size_t i = 0;  i < size(); i++ )
 838     {
 839         const KIID_PATH oldSheetPath = aOldSheetPaths.at( i );
 840         const KIID_PATH newSheetPath = at( i ).Path();
 841         SCH_SCREEN* screen = at(i).LastScreen();

 843         wxCHECK( screen, /* void */ );
 844 
 845         for( auto symbol : screen->Items().OfType( SCH_COMPONENT_T ) )
 846         {
 847             static_cast<SCH_COMPONENT*>( symbol )->ReplaceInstanceSheetPath( oldSheetPath,
 848                     newSheetPath );
 849         }
 850     }
