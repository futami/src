  81 /**

  82  * Helper class used to store the state of schematic items that can be connected to

  83  * other schematic items.

  84  */
  85 class DANGLING_END_ITEM
  86 {
  87 private:
  88     /// A pointer to the connectable object.

  89     EDA_ITEM*       m_item;
  90 
  91     /// The position of the connection point.

  92     wxPoint         m_pos;
  93 
  94     /// The type of connection of #m_item.

  95     DANGLING_END_T  m_type;
  96 
  97     /// A pointer to the parent object (in the case of pins)

  98     const EDA_ITEM* m_parent;
  99 
 100 public:
 101     DANGLING_END_ITEM( DANGLING_END_T aType, EDA_ITEM* aItem, const wxPoint& aPosition )
 102     {
 103         m_item = aItem;
 104         m_type = aType;
 105         m_pos = aPosition;
 106         m_parent = aItem;
 107     }
 108 
 109     DANGLING_END_ITEM( DANGLING_END_T aType, EDA_ITEM* aItem,
 110             const wxPoint& aPosition, const EDA_ITEM* aParent )
 111     {
 112         m_item = aItem;
 113         m_type = aType;
 114         m_pos = aPosition;
 115         m_parent = aParent;
 116     }
 117 
 118     bool operator==( const DANGLING_END_ITEM& aB )
 119     {
 120         return GetItem() == aB.GetItem()
 121             && GetPosition() == aB.GetPosition()
 122             && GetType() == aB.GetType()
 123             && GetParent() == aB.GetParent();
 124     }
 125 
 126     bool operator!=( const DANGLING_END_ITEM& aB )
 127     {
 128         return GetItem() != aB.GetItem()
 129                 || GetPosition() != aB.GetPosition()
 130                 || GetType() != aB.GetType()
 131                 || GetParent() != aB.GetParent();;
 132     }
 133 
 134     bool operator<( const DANGLING_END_ITEM& rhs ) const
 135     {
 136         return( m_pos.x < rhs.m_pos.x || ( m_pos.x == rhs.m_pos.x && m_pos.y < rhs.m_pos.y )
 137                 || ( m_pos == rhs.m_pos && m_item < rhs.m_item ) );
 138     }
 139 
 140     wxPoint GetPosition() const { return m_pos; }
 141     EDA_ITEM* GetItem() const { return m_item; }
 142     const EDA_ITEM* GetParent() const { return m_parent; }
 143     DANGLING_END_T GetType() const { return m_type; }
 144 };

 180 /**

 181  * Base class for any item which can be embedded within the #SCHEMATIC container class,

 182  * and therefore instances of derived classes should only be found in EESCHEMA or other

 183  * programs that use class SCHEMATIC and its contents.

 184  *

 185  * The corresponding class in Pcbnew is #BOARD_ITEM.

 186  */
 187 class SCH_ITEM : public EDA_ITEM
 188 {
 189     friend class CONNECTION_GRAPH;
 190 
 191 protected:
 192     SCH_LAYER_ID      m_Layer;

 194     FIELDS_AUTOPLACED m_fieldsAutoplaced; // indicates status of field autoplacement

 195     wxPoint           m_storedPos;        // a temporary variable used in some move commands

 196                                           // to store a initial pos of the item or mouse cursor

 197 
 198     /// Stores pointers to other items that are connected to this one, per sheet

 199     std::unordered_map<SCH_SHEET_PATH, ITEM_SET> m_connected_items;
 200 
 201     /// Stores connectivity information, per sheet

 202     std::unordered_map<SCH_SHEET_PATH, SCH_CONNECTION*> m_connection_map;
 203 
 204     /// True if connectivity info might be out of date

 205     bool m_connectivity_dirty;
 206 
 207 public:
 208     SCH_ITEM( EDA_ITEM* aParent, KICAD_T aType );
 209 
 210     SCH_ITEM( const SCH_ITEM& aItem );
 211 
 212     ~SCH_ITEM();
 213 
 214     virtual wxString GetClass() const override
 215     {
 216         return wxT( "SCH_ITEM" );
 217     }

 226     /**

 227      * Routine to create a new copy of given item.

 228      * The new object is not put in draw list (not linked).

 229      *

 230      * @param doClone (default = false) indicates unique values (such as timestamp and

 231      *     sheet name) should be duplicated.  Use only for undo/redo operations.

 232      */
 233     SCH_ITEM* Duplicate( bool doClone = false ) const;
 234 
 235     /**

 236      * @return true for items which are moved with the anchor point at mouse cursor

 237      *  and false for items moved with no reference to anchor

 238      * Usually return true for small items (labels, junctions) and false for

 239      * items which can be large (hierarchical sheets, compoments)

 240      */
 241     virtual bool IsMovableFromAnchorPoint() { return true; }
 242 
 243     wxPoint& GetStoredPos() { return m_storedPos; }
 244     void     SetStoredPos( wxPoint aPos ) { m_storedPos = aPos; }

 246     /**

 247      * Searches the item hierarchy to find a SCHEMATIC

 248      *

 253      * Note that this hierarchy is not the same as the sheet hierarchy!

 254      *

 255      * @return the parent schematic this item lives on, or nullptr

 256      */

 260      * @return bool - true if the object is locked, else false

 261      */

 266      */
 267     virtual void SetLocked( bool aLocked ) {}

 269     /**

 270      * Return the layer this item is on.

 271      */
 272     SCH_LAYER_ID GetLayer() const { return m_Layer; }

 275      * Set the layer this item is on.

 329     /**

 330      * Add the schematic item end points to \a aItemList if the item has end points.

 332      * The default version doesn't do anything since many of the schematic object cannot

 341     /**

 342      * Test the schematic item to \a aItemList to check if it's dangling state has changed.

 344      * Note that the return value only true when the state of the test has changed.  Use

 354      * @param aSheet - Sheet path to update connections for

 355      * @return True if the dangling state has changed from it's current setting.

 356      */
 357     virtual bool UpdateDanglingState( std::vector<DANGLING_END_ITEM>& aItemList,
 358                                       const SCH_SHEET_PATH* aPath = nullptr )
 359     {
 360         return false;
 361     }

 397     /**

 401      */
 402     SCH_CONNECTION* Connection( const SCH_SHEET_PATH& aPath ) const;
 403 
 404     /**

 405      * Retrieves the set of items connected to this item on the given sheet

 406      */
 407     ITEM_SET& ConnectedItems( const SCH_SHEET_PATH& aPath );
 408 
 409     /**

 410      * Adds a connection link between this item and another

 412     void AddConnectionTo( const SCH_SHEET_PATH& aPath, SCH_ITEM* aItem );
 413 
 414     /**

 415      * Creates a new connection object associated with this object

 417      * @param aPath is the sheet path to initialize

 419     SCH_CONNECTION* InitializeConnection( const SCH_SHEET_PATH& aPath, CONNECTION_GRAPH* aGraph );
 420 
 421     /**

 422      * Returns true if this item should propagate connection info to aItem

 423      */
 424     virtual bool ConnectionPropagatesTo( const EDA_ITEM* aItem ) const { return true; }
 425 
 426     bool IsConnectivityDirty() { return m_connectivity_dirty; }
 427 
 428     void SetConnectivityDirty( bool aDirty = true ) { m_connectivity_dirty = aDirty; }

 499 private:

 512      */
 513     virtual bool doIsConnected( const wxPoint& aPosition ) const { return false; }
 514 };
