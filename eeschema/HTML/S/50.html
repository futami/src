  44     void SetTitle( const wxString& aTitle ) override
  45     {
  46         // This class is shared for numerous tasks: a couple of single line labels and

  47         // multi-line text fields.  Since the desired size of the multi-line text field editor

  48         // is often larger, we retain separate sizes based on the dialog titles.

  49         switch( m_CurrentText->Type() )
  50         {
  51         case SCH_GLOBAL_LABEL_T:
  52         case SCH_HIER_LABEL_T:
  53         case SCH_LABEL_T:
  54             // labels can share retained settings probably.

  55             break;
  56 
  57         default:
  58             m_hash_key = TO_UTF8( aTitle );
  59             m_hash_key += typeid(*this).name();
  60         }
  61 
  62         DIALOG_LABEL_EDITOR_BASE::SetTitle( aTitle );
  63     }
