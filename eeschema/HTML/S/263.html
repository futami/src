  47 class LIB_LOGGER : public wxLogGui
  48 {
  49 public:
  50     LIB_LOGGER() :
  51             m_previousLogger( nullptr ),
  52             m_activated( false )
  53     { }
  54 
  55     ~LIB_LOGGER() override
  56     {
  57         Deactivate();
  58     }
  59 
  60     void Activate()
  61     {
  62         if( !m_activated )
  63         {
  64             m_previousLogger = wxLog::GetActiveTarget();
  65             wxLog::SetActiveTarget( this );
  66             m_activated = true;
  67         }
  68     }
  69 
  70     void Deactivate()
  71     {
  72         if( m_activated )
  73         {
  74             Flush();
  75             m_activated = false;
  76             wxLog::SetActiveTarget( m_previousLogger );
  77         }
  78     }
  79 
  80     void Flush() override
  81     {
  82         if( m_bHasMessages )
  83         {
  84             wxLogMessage( _( "Not all symbol libraries could be loaded.  Use the Manage Symbol\n"
  85                              "Libraries dialog to adjust paths and add or remove libraries." ) );
  86             wxLogGui::Flush();
  87         }
  88     }
  89 
  90 private:
  91     wxLog* m_previousLogger;
  92     bool   m_activated;
  93 };

 353     ///> Class to store a working copy of a library

 354     class LIB_BUFFER
 355     {
 356     public:
 357         LIB_BUFFER( const wxString& aLibrary ) :
 358                 m_libName( aLibrary ),
 359                 m_hash( 1 )
 360         { }
 361 
 362         bool IsModified() const
 363         {
 364             if( !m_deleted.empty() )
 365                 return true;
 366 
 367             for( const auto& partBuf : m_parts )
 368             {
 369                 if( partBuf->IsModified() )
 370                     return true;
 371             }
 372 
 373             return false;
 374         }
 375 
 376         int GetHash() const { return m_hash; }
 377 
 378         ///> Returns the working copy of a LIB_PART root object with specified alias.

 379         LIB_PART* GetPart( const wxString& aAlias ) const;
 380 
 381         ///> Creates a new buffer to store a part. LIB_BUFFER takes ownership of aCopy.

 382         bool CreateBuffer( LIB_PART* aCopy, SCH_SCREEN* aScreen );
 383 
 384         ///> Updates the buffered part with the contents of \a aCopy.

 385         bool UpdateBuffer( PART_BUFFER::PTR aPartBuf, LIB_PART* aCopy );
 386 
 387         bool DeleteBuffer( PART_BUFFER::PTR aPartBuf );
 388 
 389         void ClearDeletedBuffer()
 390         {
 391             m_deleted.clear();
 392         }
 393 
 394         ///> Saves stored modifications to Symbol Lib Table. It may result in saving the symbol

 395         ///> to disk as well, depending on the row properties.

 396         bool SaveBuffer( PART_BUFFER::PTR aPartBuf, SYMBOL_LIB_TABLE* aLibTable );
 397 
 398         ///> Saves stored modifications using a plugin. aBuffer decides whether the changes

 399         ///> should be cached or stored directly to the disk (for SCH_LEGACY_PLUGIN).

 400         bool SaveBuffer( PART_BUFFER::PTR aPartBuf, SCH_PLUGIN* aPlugin, bool aBuffer );
 401 
 402         ///> Returns a part buffer with LIB_PART holding a particular alias

 403         PART_BUFFER::PTR GetBuffer( const wxString& aAlias ) const;
 404 
 405         ///> Returns all buffered parts

 406         const std::deque<PART_BUFFER::PTR>& GetBuffers() const { return m_parts; }
 407 
 408         /**

 409          * Checks to see any parts in the buffer are derived from a parent named \a aParentName.

 410          *

 411          * @param aParentName is the name of the parent to test.

 412          *

 413          * @return true if any symbols are found derived from a symbol named \a aParent, otherwise

 414          *         false.

 415          */
 416         bool HasDerivedSymbols( const wxString& aParentName ) const;
 417 
 418         /**

 419          * Fetchs a list of root symbols names from the library buffer.

 420          *

 421          * @param aRootSymbolNames is a reference to a list to populate with root symbol names.

 422          */
 423         void GetRootSymbolNames( wxArrayString& aRootSymbolNames );
 424 
 425         /**

 426          * Fetch all of the symbols derived from a \a aSymbolName into \a aList.

 427          *

 428          * @param aSymbolName is the name of the symbol to search for derived parts in this

 429          *                    buffer.

 430          * @param aList is the list of symbols names derived from \a aSymbolName.

 431          *

 432          * @return a size_t count of the number of symbols derived from \a aSymbolName.

 433          */
 434         size_t GetDerivedSymbolNames( const wxString& aSymbolName, wxArrayString& aList );
 435 
 436     private:
 437         /**

 438          * Remove all symbols derived from \a aParent from the library buffer.

 439          *

 440          * @param aParent is the #PART_BUFFER to check against.

 441          *

 442          * @return the count of #PART_BUFFER objects removed from the library.

 443          */
 444         int removeChildSymbols( PART_BUFFER::PTR aPartBuf );
 445 
 446         std::deque<PART_BUFFER::PTR> m_parts;
 447         std::deque<PART_BUFFER::PTR> m_deleted;  // Buffer for deleted parts until library is saved
