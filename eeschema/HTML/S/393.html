  51 /** Info about complex hierarchies handling:

  52  * A hierarchical schematic uses sheets (hierarchical sheets) included in a

  53  * given sheet.  Each sheet corresponds to a schematic drawing handled by a

  54  * SCH_SCREEN structure.  A SCH_SCREEN structure contains drawings, and have

  55  * a filename to write it's data.  Also a SCH_SCREEN display a sheet number

  56  * and the name of the sheet.

  57  *

  58  * In simple (and flat) hierarchies a sheet is linked to a SCH_SCREEN,

  59  * and a SCH_SCREEN is used by only one hierarchical sheet.

  60  *

  61  * In complex hierarchies the same SCH_SCREEN (and its data) is shared between

  62  * more than one sheet.  Therefore subsheets (like subsheets in a SCH_SCREEN

  63  * shared by many sheets) can be also shared.  So the same SCH_SCREEN must

  64  * handle different components references and parts selection depending on

  65  * which sheet is currently selected, and how a given subsheet is selected.

  66  * 2 sheets share the same SCH_SCREEN (the same drawings) if they have the

  67  * same filename.

  68  *

  69  * In KiCad each component and sheet receives (when created) an unique

  70  * identification called Time Stamp.  So each sheet has 2 ids: its time stamp

  71  * (that cannot change) and its name ( that can be edited and therefore is

  72  * not reliable for strong identification).  KiCad uses Time Stamp ( a unique

  73  * 32 bit id), to identify sheets in hierarchies.

  74  * A given sheet in a hierarchy is fully labeled by its path (or sheet path)

  75  * that is the list of time stamp found to access it through the hierarchy

  76  * the root sheet is /.  All  other sheets have a path like /1234ABCD or

  77  * /4567FEDC/AA2233DD/.  This path can be displayed as human readable sheet

  78  * name like: / or /sheet1/include_sheet/ or /sheet2/include_sheet/

  79  *

  80  * So to know for a given SCH_SCREEN (a given schematic drawings) we must:

  81  *   1) Handle all references possibilities.

  82  *   2) When acceded by a given selected sheet, display (update) the

  83  *      corresponding references and sheet path

  84  *

  85  * The class SCH_SHEET_PATH handles paths used to access a sheet.  The class

  86  * SCH_SHEET_LIST allows one to handle the full (or partial) list of sheets and

  87  * their paths in a complex hierarchy.  The class EDA_ScreenList allows one

  88  * to handle the list of SCH_SCREEN. It is useful to clear or save data,

  89  * but is not suitable to handle the full complex hierarchy possibilities

  90  * (usable in flat and simple hierarchies).

  91  */

 108 /**

 109  * SCH_SHEET_PATH

 110  *

 111  * handles access to a stack of flattened #SCH_SHEET objects by way of a path for

 112  * creating a flattened schematic hierarchy.

 113  *

 114  * <p>

 115  * The #SCH_SHEET objects are stored in a list from first (usually the root sheet) to a

 116  * given sheet in last position.  The _last_ sheet is usually the sheet we want to select

 117  * or reach (which is what the function Last() returns).   Others sheets constitute the

 118  * "path" from the first to the last sheet.

 119  * </p>

 120  */
 121 class SCH_SHEET_PATH
 122 {
 123 protected:
 124     std::vector< SCH_SHEET* > m_sheets;
 125 
 126     size_t m_current_hash;
 127 
 128     int m_pageNumber;              /// Page numbers are maintained by the sheet load order.

 129 
 130     std::map<std::pair<wxString, wxString>, bool> m_recursion_test_cache;
 131 
 132 public:
 133     SCH_SHEET_PATH();
 134 
 135     SCH_SHEET_PATH( const SCH_SHEET_PATH& aOther );
 136 
 137     SCH_SHEET_PATH& operator=( const SCH_SHEET_PATH& aOther );
 138 
 139     /// Forwarded method from std::vector

 140     SCH_SHEET* at( size_t aIndex ) const { return m_sheets.at( aIndex ); }
 141 
 142     /// Forwarded method from std::vector

 143     void clear()
 144     {
 145         m_sheets.clear();
 146         Rehash();
 147     }
 148 
 149     /// Forwarded method from std::vector

 150     bool empty() const { return m_sheets.empty(); }
 151 
 152     /// Forwarded method from std::vector

 153     void pop_back()
 154     {
 155         m_sheets.pop_back();
 156         Rehash();
 157     }
 158 
 159     /// Forwarded method from std::vector

 160     void push_back( SCH_SHEET* aSheet )
 161     {
 162         m_sheets.push_back( aSheet );
 163         Rehash();
 164     }
 165 
 166     /// Forwarded method from std::vector

 167     size_t size() const { return m_sheets.size(); }
 168 
 169     void Rehash();
 170 
 171     size_t GetCurrentHash() const { return m_current_hash; }
 172 
 173     void SetPageNumber( int aPageNumber ) { m_pageNumber = aPageNumber; }
 174 
 175     int GetPageNumber() const { return m_pageNumber; }
 176 
 177     const SCH_SHEET* GetSheet( unsigned aIndex ) const
 178     {
 179         SCH_SHEET* retv = NULL;
 180 
 181         if( aIndex < size() )
 182             retv = at( aIndex );
 183 
 184         return const_cast< SCH_SHEET* >( retv );
 185     }
 186 
 187     /**

 188      * Function Cmp

 189      * Compare if this is the same sheet path as aSheetPathToTest

 190      * @param aSheetPathToTest = sheet path to compare

 191      * @return 1 if this sheet path has more sheets than aSheetPathToTest,

 192      *   -1 if this sheet path has fewer sheets than aSheetPathToTest,

 193      *   or 0 if same

 194      */
 195     int Cmp( const SCH_SHEET_PATH& aSheetPathToTest ) const;
 196 
 197     /**

 198      * Function Last

 199      * returns a pointer to the last sheet of the list

 200      * One can see the others sheet as the "path" to reach this last sheet

 201      */
 202     SCH_SHEET* Last() const;
 203 
 204     /**

 205      * Function LastScreen

 206      * @return the SCH_SCREEN relative to the last sheet in list

 207      */
 208     SCH_SCREEN* LastScreen();
 209 
 210 
 211     ///> @copydoc SCH_SHEET_PATH::LastScreen()

 212     SCH_SCREEN* LastScreen() const;
 213 
 214     /**

 215      * Function PathAsString

 216      * the path uses the time stamps which do not changes even when editing

 217      * sheet parameters

 218      * a path is something like / (root) or /34005677 or /34005677/00AE4523

 219      */
 220     wxString PathAsString() const;
 221 
 222     /**

 223      * Get the sheet path as an KIID_PATH.

 224      * @return

 225      */
 226     KIID_PATH Path() const;
 227 
 228     /**

 229      * Function PathHumanReadable

 230      * returns the sheet path in a human readable form, i.e. as a path made

 231      * from sheet names.  The the "normal" path instead uses the time

 232      * stamps in the path.  (Time stamps do not change even when editing

 233      * sheet parameters).

 234      */
 235     wxString PathHumanReadable() const;
 236 
 237     /**

 238      * @return a PathName for the root sheet (like "/" or "<root>"

 239      * @param aUseShortName: true to return "/", false to return a longer name

 240      */
 241     static wxString GetRootPathName( bool aUseShortName = true );
 242 
 243     /**

 244      * Function UpdateAllScreenReferences

 245      * updates the reference and the m_Multi parameter (part selection) for all

 246      * components on a screen depending on the actual sheet path.

 247      * Mandatory in complex hierarchies because sheets use the same screen

 248      * (basic schematic)

 249      * but with different references and part selections according to the

 250      * displayed sheet

 251      */
 252     void UpdateAllScreenReferences();
 253 
 254     /**

 255      * Function GetComponents

 256      * adds a SCH_REFERENCE() object to \a aReferences for each component in the sheet.

 257      *

 258      * @param aReferences List of references to populate.

 259      * @param aIncludePowerSymbols : false to only get normal components.

 260      * @param aForceIncludeOrphanComponents : true to include components having no symbol found in lib.

 261      * ( orphan components)

 262      * The normal option is false, and set to true only to build the full list of components.

 263      */
 264     void GetComponents( SCH_REFERENCE_LIST& aReferences, bool aIncludePowerSymbols = true,
 265                         bool aForceIncludeOrphanComponents = false ) const;
 266 
 267     /**

 268      * Function GetMultiUnitComponents

 269      * adds a SCH_REFERENCE_LIST object to \a aRefList for each same-reference set of

 270      * multi-unit parts in the sheet. The map key for each element will be the

 271      * reference designator.

 272      *

 273      * @param aRefList Map of reference designators to reference lists

 274      * @param aIncludePowerSymbols : false to only get normal components.

 275      */
 276     void GetMultiUnitComponents( SCH_MULTI_UNIT_REFERENCE_MAP &aRefList,
 277                                  bool aIncludePowerSymbols = true ) const;
 278 
 279     /**

 280      * Function SetFootprintField

 281      * searches last sheet in the path for a component with \a aReference and set the footprint

 282      * field to \a aFootPrint if found.

 283      *

 284      * @param aReference The reference designator of the component.

 285      * @param aFootPrint The value to set the footprint field.

 286      * @param aSetVisible The value to set the field visibility flag.

 287      * @return true if \a aReference was found otherwise false.

 288      */
 289     bool SetComponentFootprint( const wxString& aReference, const wxString& aFootPrint,
 290                                 bool aSetVisible );
 291 
 292     /**

 293      * Function TestForRecursion

 294      *

 295      * test the SCH_SHEET_PATH file names to check adding the sheet stored in the file

 296      * \a aSrcFileName to the sheet stored in file \a aDestFileName  will cause a sheet

 297      * path recursion.

 298      *

 299      * @param aSrcFileName is the source file name of the sheet add to \a aDestFileName.

 300      * @param aDestFileName is the file name of the destination sheet for \a aSrcFileName.

 301      * @return true if \a aFileName will cause recursion in the sheet path.  Otherwise false.

 302      */
 303     bool TestForRecursion( const wxString& aSrcFileName, const wxString& aDestFileName );
 304 
 305     bool operator==( const SCH_SHEET_PATH& d1 ) const;
 306 
 307     bool operator!=( const SCH_SHEET_PATH& d1 ) const { return !( *this == d1 ) ; }
 308 
 309     bool operator<( const SCH_SHEET_PATH& d1 ) const { return m_sheets < d1.m_sheets; }
 310 
 311 private:
 312 
 313     void initFromOther( const SCH_SHEET_PATH& aOther );
 314 };

 330 /**

 331  * SCH_SHEET_LIST

 332  *

 333  * handles a list of #SCH_SHEET_PATH objects in a flattened hierarchy.

 334  *

 335  * #SCH_SHEET objects are not unique, there can be many sheets with the same filename and

 336  * that share the same #SCH_SCREEN reference.   Each The schematic file (#SCH_SCREEN) may

 337  * be shared between these sheets and component references are specific to a sheet path.

 338  * When a sheet is entered, component references and sheet page number are updated.

 339  */
 340 class SCH_SHEET_LIST : public SCH_SHEET_PATHS
 341 {
 342 private:
 343     SCH_SHEET_PATH  m_currentSheetPath;
 344 
 345 public:
 346 
 347     /**

 348      * Constructor

 349      * build a flattened list of SCH_SHEET_PATH objects from \a aSheet.

 350      *

 351      * If aSheet == NULL, then this is an empty hierarchy which the user can populate.

 352      */
 353     SCH_SHEET_LIST( SCH_SHEET* aSheet = NULL );
 354 
 355     ~SCH_SHEET_LIST() {}
 356 
 357     /**

 358      * Function IsModified

 359      * checks the entire hierarchy for any modifications.

 360      * @return True if the hierarchy is modified otherwise false.

 361      */
 362     bool IsModified();
 363 
 364     void ClearModifyStatus();
 365 
 366     /**

 367      * Fetch a SCH_ITEM by ID.  Also returns the sheet the item was found on in \a aPathOut.

 368      */
 369     SCH_ITEM* GetItem( const KIID& aID, SCH_SHEET_PATH* aPathOut = nullptr );
 370 
 371     /**

 372      * Fill an item cache for temporary use when many items need to be fetched.

 373      */
 374     void FillItemMap( std::map<KIID, EDA_ITEM*>& aMap );
 375 
 376     /**

 377      * Function AnnotatePowerSymbols

 378      * Silently annotates the not yet annotated power symbols of the entire hierarchy

 379      * of the sheet path list.

 380      * It is called before creating a netlist, to annotate power symbols, without prompting

 381      * the user about not annotated or duplicate for these symbols, if only these symbols

 382      * need annotation ( a very frequent case ).

 383      */
 384     void AnnotatePowerSymbols();
 385 
 386     /**

 387      * Function GetComponents

 388      * adds a SCH_REFERENCE() object to \a aReferences for each component in the list

 389      * of sheets.

 390      *

 391      * @param aReferences List of references to populate.

 392      * @param aIncludePowerSymbols Set to false to only get normal components.

 393      * @param aForceIncludeOrphanComponents : true to include components having no symbol found in lib.

 394      * ( orphan components)

 395      * The normal option is false, and set to true only to build the full list of components.

 396      */
 397     void GetComponents( SCH_REFERENCE_LIST& aReferences, bool aIncludePowerSymbols = true,
 398                         bool aForceIncludeOrphanComponents = false ) const;
 399 
 400     /**

 401      * Function GetMultiUnitComponents

 402      * adds a SCH_REFERENCE_LIST object to \a aRefList for each same-reference set of

 403      * multi-unit parts in the list of sheets. The map key for each element will be the

 404      * reference designator.

 405      *

 406      * @param aRefList Map of reference designators to reference lists

 407      * @param aIncludePowerSymbols Set to false to only get normal components.

 408      */
 409     void GetMultiUnitComponents( SCH_MULTI_UNIT_REFERENCE_MAP &aRefList,
 410                                  bool aIncludePowerSymbols = true ) const;
 411 
 412     /**

 413      * Function SetFootprintField

 414      * searches all the sheets for a component with \a aReference and set the footprint

 415      * field to \a aFootPrint if found.

 416      *

 417      * @param aReference The reference designator of the component.

 418      * @param aFootPrint The value to set the footprint field.

 419      * @param aSetVisible The value to set the field visibility flag.

 420      * @return True if \a aReference was found otherwise false.

 421      */
 422     bool SetComponentFootprint( const wxString& aReference, const wxString& aFootPrint,
 423                                 bool aSetVisible );
 424 
 425     /**

 426      * Function TestForRecursion

 427      *

 428      * test every SCH_SHEET_PATH in the SCH_SHEET_LIST to verify if adding the sheets stored

 429      * in \a aSrcSheetHierarchy to the sheet stored in \a aDestFileName  will cause recursion.

 430      *

 431      * @param aSrcSheetHierarchy is the SCH_SHEET_LIST of the source sheet add to \a aDestFileName.

 432      * @param aDestFileName is the file name of the destination sheet for \a aSrcFileName.

 433      * @return true if \a aFileName will cause recursion in the sheet path.  Otherwise false.

 434      */
 435     bool TestForRecursion( const SCH_SHEET_LIST& aSrcSheetHierarchy,
 436                            const wxString& aDestFileName );
 437 
 438     /**

 439      * Function FindSheetForScreen

 440      *

 441      * returns the first sheetPath (not necessarily the only one) using a particular screen

 442      */
 443     SCH_SHEET_PATH* FindSheetForScreen( SCH_SCREEN* aScreen );
 444 
 445     /**

 446      * Function BuildSheetList

 447      * builds the list of sheets and their sheet path from \a aSheet.

 448      * If \a aSheet is the root sheet, the full sheet path and sheet list are built.

 449      *

 450      * @param aSheet is the starting sheet from which the list is built, or NULL

 451      *               indicating that g_RootSheet should be used.

 452      * @throw std::bad_alloc if the memory for the sheet path list could not be allocated.

 453      */
 454     void BuildSheetList( SCH_SHEET* aSheet );
 455 
 456     bool NameExists( const wxString& aSheetName );
 457 
 458     /**

 459      * Update all of the symbol instance information using \a aSymbolInstances.

 460      * WARNING: Do not call this on anything other than the full hierarchy.

 461      * @param aSymbolInstances is the symbol path information loaded from the root schematic.

 462      */
 463     void UpdateSymbolInstances( const std::vector<COMPONENT_INSTANCE_REFERENCE>& aSymbolInstances );
 464 
 465     std::vector<KIID_PATH> GetPaths() const;
 466 
 467     /**

 468      * Update all of the symbol sheet paths to the sheet paths defined in \a aOldSheetPaths.

 469      *

 470      * @note The list of old sheet paths must be the exact same size and order as the existing

 471      *       sheet paths.  This should not be an issue if no new sheets where added between the

 472      *       creation of this sheet list and \a aOldSheetPaths.  This should only be called

 473      *       when updating legacy schematics to the new schematic file format.  Once this

 474      *       happens, the schematic cannot be save to the legacy file format because the

 475      *       time stamp part of UUIDs are no longer guaranteed to be unique.

 476      *

 477      * @param aOldSheetPaths is the #SHEET_PATH_LIST to update from.

 478      */
 479     void ReplaceLegacySheetPaths( const std::vector<KIID_PATH>& aOldSheetPaths );
 480 };
