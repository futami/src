  76 wxString LIB_EDIT_FRAME::SelectLibraryFromList()
  77 {
  78     PROJECT& prj = Prj();
  79 
  80     if( prj.SchSymbolLibTable()->IsEmpty() )
  81     {
  82         DisplayError( this, _( "No symbol libraries are loaded." ) );
  83         return wxEmptyString;
  84     }
  85 
  86     wxArrayString headers;
  87 
  88     headers.Add( _( "Library" ) );
  89 
  90     std::vector< wxArrayString > itemsToDisplay;
  91     std::vector< wxString > libNicknames = prj.SchSymbolLibTable()->GetLogicalLibs();
  92 
  93     // Conversion from wxArrayString to vector of ArrayString

  94     for( const wxString& name : libNicknames )
  95     {
  96         wxArrayString item;
  97 
  98         item.Add( name );
  99         itemsToDisplay.push_back( item );
 100     }
 101 
 102     wxString oldLibName = prj.GetRString( PROJECT::SCH_LIB_SELECT );
 103 
 104     EDA_LIST_DIALOG dlg( this, _( "Select Symbol Library" ), headers, itemsToDisplay, oldLibName );
 105 
 106     if( dlg.ShowModal() != wxID_OK )
 107         return wxEmptyString;
 108 
 109     wxString libName = dlg.GetTextSelection();
 110 
 111     if( !libName.empty() )
 112     {
 113         if( prj.SchSymbolLibTable()->HasLibrary( libName ) )
 114             prj.SetRString( PROJECT::SCH_LIB_SELECT, libName );
 115         else
 116             libName = wxEmptyString;
 117     }
 118 
 119     return libName;

 165 bool LIB_EDIT_FRAME::LoadComponentFromCurrentLib( const wxString& aAliasName, int aUnit,
 166                                                   int aConvert )
 167 {
 168     LIB_PART* alias = nullptr;
 169 
 170     try
 171     {
 172         alias = Prj().SchSymbolLibTable()->LoadSymbol( GetCurLib(), aAliasName );
 173     }
 174     catch( const IO_ERROR& ioe )
 175     {
 176         wxString msg;
 177 
 178         msg.Printf( _( "Error occurred loading symbol \"%s\" from library \"%s\"." ),
 179                     aAliasName, GetCurLib() );
 180         DisplayErrorMessage( this, msg, ioe.What() );
 181         return false;
 182     }
 183 
 184     if( !alias || !LoadOneLibraryPartAux( alias, GetCurLib(), aUnit, aConvert ) )
 185         return false;
 186 
 187     // Enable synchronized pin edit mode for symbols with interchangeable units

 188     m_SyncPinEdit = !GetCurPart()->UnitsLocked();
 189 
 190     ClearUndoRedoList();
 191     m_toolManager->RunAction( ACTIONS::zoomFitScreen, true );
 192     SetShowDeMorgan( GetCurPart()->Flatten()->HasConversion() );
 193 
 194     if( aUnit > 0 )
 195         RebuildSymbolUnitsList();
 196 
 197     return true;
 198 }

 200 bool LIB_EDIT_FRAME::LoadOneLibraryPartAux( LIB_PART* aEntry, const wxString& aLibrary,
 201                                             int aUnit, int aConvert )
 202 {
 203     wxString msg, rootName;
 204 
 205     if( !aEntry || aLibrary.empty() )
 206         return false;
 207 
 208     if( aEntry->GetName().IsEmpty() )
 209     {
 210         wxLogWarning( "Symbol in library \"%s\" has empty name field.", aLibrary );
 211         return false;
 212     }
 213 
 214     LIB_PART* lib_part = m_libMgr->GetBufferedPart( aEntry->GetName(), aLibrary );
 215     wxCHECK( lib_part, false );
 216 
 217     m_unit = aUnit > 0 ? aUnit : 1;
 218     m_convert = aConvert > 0 ? aConvert : 1;
 219 
 220     // The buffered screen for the part

 221     SCH_SCREEN* part_screen = m_libMgr->GetScreen( lib_part->GetName(), aLibrary );
 222 
 223     SetScreen( part_screen );
 224     SetCurPart( new LIB_PART( *lib_part ) );
 225     SetCurLib( aLibrary );
 226 
 227     m_toolManager->RunAction( ACTIONS::zoomFitScreen, true );
 228     updateTitle();
 229     RebuildSymbolUnitsList();
 230     SetShowDeMorgan( GetCurPart()->HasConversion() );
 231     SyncToolbars();
 232 
 233     // Display the document information based on the entry selected just in

 234     // case the entry is an alias.

 235     DisplayCmpDoc();
 236     Refresh();
 237 
 238     return true;

 250 void LIB_EDIT_FRAME::CreateNewPart()
 251 {
 252     m_toolManager->RunAction( ACTIONS::cancelInteractive, true );
 253 
 254     wxArrayString rootSymbols;
 255     wxString lib = getTargetLib();
 256 
 257     if( !m_libMgr->LibraryExists( lib ) )
 258     {
 259         lib = SelectLibraryFromList();
 260 
 261         if( !m_libMgr->LibraryExists( lib ) )
 262             return;
 263     }
 264 
 265     m_libMgr->GetRootSymbolNames( lib, rootSymbols );
 266 
 267     DIALOG_LIB_NEW_COMPONENT dlg( this, &rootSymbols );
 268     dlg.SetMinSize( dlg.GetSize() );
 269 
 270     if( dlg.ShowModal() == wxID_CANCEL )
 271         return;
 272 
 273     if( dlg.GetName().IsEmpty() )
 274     {
 275         wxMessageBox( _( "This new symbol has no name and cannot be created." ) );
 276         return;
 277     }
 278 
 279     wxString name = dlg.GetName();
 280     // Currently, symbol names cannot include a space, that breaks libraries:

 281     name.Replace( " ", "_" );
 282 
 283     // Test if there is a component with this name already.

 284     if( !lib.empty() && m_libMgr->PartExists( name, lib ) )
 285     {
 286         wxString msg = wxString::Format( _( "Symbol \"%s\" already exists in library \"%s\"" ),
 287                                          name, lib );
 288         DisplayError( this, msg );
 289         return;
 290     }
 291 
 292     LIB_PART new_part( name );      // do not create part on the heap, it will be buffered soon

 293 
 294     wxString parentSymbolName = dlg.GetParentSymbolName();
 295 
 296     if( parentSymbolName.IsEmpty() )
 297     {
 298         new_part.GetReferenceField().SetText( dlg.GetReference() );
 299         new_part.SetUnitCount( dlg.GetUnitCount() );
 300 
 301         // Initialize new_part.m_TextInside member:

 302         // if 0, pin text is outside the body (on the pin)

 303         // if > 0, pin text is inside the body

 304 
 305         if( dlg.GetPinNameInside() )
 306         {
 307             new_part.SetPinNameOffset( dlg.GetPinTextPosition() );
 308 
 309             if( new_part.GetPinNameOffset() == 0 )
 310                 new_part.SetPinNameOffset( 1 );
 311         }
 312         else
 313         {
 314             new_part.SetPinNameOffset( 0 );
 315         }
 316 
 317         ( dlg.GetPowerSymbol() ) ? new_part.SetPower() : new_part.SetNormal();
 318         new_part.SetShowPinNumbers( dlg.GetShowPinNumber() );
 319         new_part.SetShowPinNames( dlg.GetShowPinName() );
 320         new_part.LockUnits( dlg.GetLockItems() );
 321 
 322         if( dlg.GetUnitCount() < 2 )
 323             new_part.LockUnits( false );
 324 
 325         new_part.SetConversion( dlg.GetAlternateBodyStyle() );
 326         // must be called after loadPart, that calls SetShowDeMorgan, but

 327         // because the symbol is empty,it looks like it has no alternate body

 328         SetShowDeMorgan( dlg.GetAlternateBodyStyle() );
 329     }
 330     else
 331     {
 332         LIB_PART* parent = m_libMgr->GetAlias( parentSymbolName, lib );
 333         wxCHECK( parent, /* void */ );
 334         new_part.SetParent( parent );
 335 
 336         // Inherit the parent mandatory field attributes.

 337         for( int id=0;  id<MANDATORY_FIELDS;  ++id )
 338         {
 339             LIB_FIELD* field = new_part.GetField( id );
 340 
 341             // the MANDATORY_FIELDS are exactly that in RAM.

 342             wxCHECK( field, /* void */ );
 343 
 344             LIB_FIELD* parentField = parent->GetField( id );
 345 
 346             wxCHECK( parentField, /* void */ );
 347 
 348             *field = *parentField;
 349 
 350             if( id == VALUE )
 351                 field->SetText( name );
 352 
 353             field->SetParent( &new_part );
 354         }
 355     }
 356 
 357     m_libMgr->UpdatePart( &new_part, lib );
 358     SyncLibraries( false );
 359     LoadPart( name, lib, 1 );
 360 }

 363 void LIB_EDIT_FRAME::Save()
 364 {
 365     LIB_ID libId = getTargetLibId();
 366     const wxString& libName = libId.GetLibNickname();
 367     const wxString& partName = libId.GetLibItemName();
 368 
 369     if( partName.IsEmpty() )
 370     {
 371         saveLibrary( libName, false );
 372     }
 373     else if( !libName.IsEmpty() && m_libMgr->IsLibraryReadOnly( libName ) )
 374     {
 375         // Force a "Save As..." if the modified library is read only.

 376         saveLibrary( libName, true );
 377     }
 378     else
 379     {
 380         // Save a single library.

 381         if( m_libMgr->FlushPart( partName, libName ) )
 382             m_libMgr->ClearPartModified( partName, libName );
 383     }
 384 
 385     m_treePane->GetLibTree()->RefreshLibTree();
 386     refreshSchematic();
 387 }

 406 void LIB_EDIT_FRAME::savePartAs()
 407 {
 408     LIB_ID old_lib_id = getTargetLibId();
 409     wxString old_name = old_lib_id.GetLibItemName();
 410     wxString old_lib = old_lib_id.GetLibNickname();
 411     LIB_PART* part = m_libMgr->GetBufferedPart( old_name, old_lib );
 412 
 413     if( part )
 414     {
 415         SYMBOL_LIB_TABLE* tbl = Prj().SchSymbolLibTable();
 416         wxArrayString headers;
 417         std::vector< wxArrayString > itemsToDisplay;
 418         std::vector< wxString > libNicknames = tbl->GetLogicalLibs();
 419 
 420         headers.Add( _( "Nickname" ) );
 421         headers.Add( _( "Description" ) );
 422 
 423         for( const auto& name : libNicknames )
 424         {
 425             wxArrayString item;
 426             item.Add( name );
 427             item.Add( tbl->GetDescription( name ) );
 428             itemsToDisplay.push_back( item );
 429         }
 430 
 431         EDA_LIST_DIALOG dlg( this, _( "Save Copy of Symbol" ), headers, itemsToDisplay, old_lib );
 432         dlg.SetListLabel( _( "Save in library:" ) );
 433         dlg.SetOKLabel( _( "Save" ) );
 434 
 435         wxBoxSizer* bNameSizer = new wxBoxSizer( wxHORIZONTAL );
 436 
 437         wxStaticText* label = new wxStaticText( &dlg, wxID_ANY, _( "Name:" ),
 438                                                 wxDefaultPosition, wxDefaultSize, 0 );
 439         bNameSizer->Add( label, 0, wxALIGN_CENTER_VERTICAL|wxTOP|wxBOTTOM|wxLEFT, 5 );
 440 
 441         wxTextCtrl* nameTextCtrl = new wxTextCtrl( &dlg, wxID_ANY, old_name,
 442                                                    wxDefaultPosition, wxDefaultSize, 0 );
 443         bNameSizer->Add( nameTextCtrl, 1, wxALIGN_CENTER_VERTICAL|wxALL, 5 );
 444 
 445         wxSizer* mainSizer = dlg.GetSizer();
 446         mainSizer->Prepend( bNameSizer, 0, wxEXPAND|wxTOP|wxLEFT|wxRIGHT, 5 );
 447 
 448         // Move nameTextCtrl to the head of the tab-order

 449         if( dlg.GetChildren().DeleteObject( nameTextCtrl ) )
 450             dlg.GetChildren().Insert( nameTextCtrl );
 451 
 452         dlg.SetInitialFocus( nameTextCtrl );
 453 
 454         dlg.Layout();
 455         mainSizer->Fit( &dlg );
 456 
 457         if( dlg.ShowModal() != wxID_OK )
 458             return;                   // canceled by user

 459 
 460         wxString new_lib = dlg.GetTextSelection();
 461 
 462         if( new_lib.IsEmpty() )
 463         {
 464             DisplayError( this, _( "No library specified.  Symbol could not be saved." ) );
 465             return;
 466         }
 467 
 468         // @todo Either check the selecteced library to see if the parent symbol name is in

 469         //       the new library and/or copy the parent symbol as well.  This is the lazy

 470         //       solution to ensure derived parts do not get orphaned.

 471         if( part->IsAlias() && new_lib != old_lib )
 472         {
 473             DisplayError( this, _( "Derived symbols must be save in the same library\n"
 474                                    "that the parent symbol exists." ) );
 475             return;
 476         }
 477 
 478         wxString new_name = nameTextCtrl->GetValue();
 479         new_name.Trim( true );
 480         new_name.Trim( false );
 481         new_name.Replace( " ", "_" );
 482 
 483         if( new_name.IsEmpty() )
 484         {
 485             DisplayError( this, _( "No symbol name specified.  Symbol could not be saved." ) );
 486             return;
 487         }
 488 
 489         // Test if there is a component with this name already.

 490         if( m_libMgr->PartExists( new_name, new_lib ) )
 491         {
 492             wxString msg = wxString::Format( _( "Symbol \"%s\" already exists in library \"%s\"" ),
 493                                              new_name, new_lib );
 494             DisplayError( this, msg );
 495             return;
 496         }
 497 
 498         LIB_PART new_part( *part );
 499         new_part.SetName( new_name );
 500 
 501         m_libMgr->UpdatePart( &new_part, new_lib );
 502         SyncLibraries( false );
 503         m_treePane->GetLibTree()->SelectLibId( LIB_ID( new_lib, new_part.GetName() ) );
 504         LoadPart( new_name, new_lib, m_unit );
 505     }
 506 }

 509 void LIB_EDIT_FRAME::UpdateAfterSymbolProperties( wxString* aOldName )
 510 {
 511     wxCHECK( m_my_part, /* void */ );
 512 
 513     wxString  msg;
 514     wxString  lib = GetCurLib();
 515 
 516     if( aOldName && *aOldName != m_my_part->GetName() )
 517     {
 518         // Test the current library for name conflicts

 519         if( !lib.empty() && m_libMgr->PartExists( m_my_part->GetName(), lib ) )
 520         {
 521             msg.Printf( _( "The name '%s' conflicts with an existing entry in the library '%s'." ),
 522                         m_my_part->GetName(),
 523                         lib );
 524 
 525             DisplayErrorMessage( this, msg );
 526             m_my_part->SetName( *aOldName );
 527         }
 528         else
 529         {
 530             m_libMgr->UpdatePartAfterRename( m_my_part, *aOldName, lib );
 531         }
 532 
 533         // Reselect the renamed part

 534         m_treePane->GetLibTree()->SelectLibId( LIB_ID( lib, m_my_part->GetName() ) );
 535     }
 536 
 537     RebuildSymbolUnitsList();
 538     SetShowDeMorgan( GetCurPart()->Flatten()->HasConversion() );
 539     updateTitle();
 540     DisplayCmpDoc();
 541 
 542     RebuildView();
 543     OnModify();
 544 }

 547 void LIB_EDIT_FRAME::DeletePartFromLibrary()
 548 {
 549     LIB_ID libId = getTargetLibId();
 550 
 551     if( m_libMgr->IsPartModified( libId.GetLibItemName(), libId.GetLibNickname() )
 552         && !IsOK( this, _( wxString::Format( "The symbol \"%s\" has been modified\n"
 553                                              "Do you want to remove it from the library?",
 554                                              libId.GetUniStringLibItemName() ) ) ) )
 555     {
 556         return;
 557     }
 558 
 559     if( m_libMgr->HasDerivedSymbols( libId.GetLibItemName(), libId.GetLibNickname() ) )
 560     {
 561         wxString msg;
 562 
 563         msg.Printf( _( "The symbol \"%s\" is used to derive other symbols.\n"
 564                        "Deleting this symbol will delete all of the symbols derived from it.\n\n"
 565                        "Do you wish to delete this symbol and all of it's derivatives?" ),
 566                     libId.GetLibItemName().wx_str() );
 567 
 568         wxMessageDialog::ButtonLabel yesButtonLabel( _( "Delete Symbol" ) );
 569         wxMessageDialog::ButtonLabel noButtonLabel( _( "Keep Symbol" ) );
 570 
 571         wxMessageDialog dlg( this, msg, _( "Warning" ),
 572                              wxYES_NO | wxYES_DEFAULT | wxICON_QUESTION | wxCENTER );
 573         dlg.SetYesNoLabels( yesButtonLabel, noButtonLabel );
 574 
 575         if( dlg.ShowModal() == wxID_NO )
 576             return;
 577     }
 578 
 579     if( isCurrentPart( libId ) )
 580         emptyScreen();
 581 
 582     m_libMgr->RemovePart( libId.GetLibItemName(), libId.GetLibNickname() );
 583 
 584     m_treePane->GetLibTree()->RefreshLibTree();
 585     refreshSchematic();
 586 }

 589 void LIB_EDIT_FRAME::CopyPartToClipboard()
 590 {
 591     int dummyUnit;
 592     LIB_ID libId = m_treePane->GetLibTree()->GetSelectedLibId( &dummyUnit );
 593     LIB_PART* part = m_libMgr->GetBufferedPart( libId.GetLibItemName(), libId.GetLibNickname() );
 594 
 595     if( !part )
 596         return;
 597 
 598     std::unique_ptr< LIB_PART> tmp = part->Flatten();
 599     STRING_FORMATTER formatter;
 600     SCH_SEXPR_PLUGIN::FormatPart( tmp.get(), formatter );
 601 
 602     auto clipboard = wxTheClipboard;
 603     wxClipboardLocker clipboardLock( clipboard );
 604 
 605     if( !clipboardLock || !clipboard->IsOpened() )
 606         return;
 607 
 608     auto data = new wxTextDataObject( wxString( formatter.GetString().c_str(), wxConvUTF8 ) );
 609     clipboard->SetData( data );
 610 
 611     clipboard->Flush();
 612 }

 615 void LIB_EDIT_FRAME::DuplicatePart( bool aFromClipboard )
 616 {
 617     int dummyUnit;
 618     LIB_ID libId = m_treePane->GetLibTree()->GetSelectedLibId( &dummyUnit );
 619     wxString lib = libId.GetLibNickname();
 620 
 621     if( !m_libMgr->LibraryExists( lib ) )
 622         return;
 623 
 624     LIB_PART* srcPart = nullptr;
 625     LIB_PART* newPart = nullptr;
 626 
 627     if( aFromClipboard )
 628     {
 629         auto clipboard = wxTheClipboard;
 630         wxClipboardLocker clipboardLock( clipboard );
 631 
 632         if( !clipboardLock || ! clipboard->IsSupported( wxDF_TEXT ) )
 633             return;
 634 
 635         wxTextDataObject data;
 636         clipboard->GetData( data );
 637         wxString partSource = data.GetText();
 638 
 639         STRING_LINE_READER reader( TO_UTF8( partSource ), "Clipboard" );
 640 
 641         try
 642         {
 643             newPart = SCH_SEXPR_PLUGIN::ParsePart( reader );
 644         }
 645         catch( IO_ERROR& e )
 646         {
 647             wxLogMessage( "Can not paste: %s", GetChars( e.Problem() ) );
 648             return;
 649         }
 650     }
 651     else
 652     {
 653         srcPart = m_libMgr->GetBufferedPart( libId.GetLibItemName(), lib );
 654 
 655         wxCHECK( srcPart, /* void */ );
 656 
 657         newPart = new LIB_PART( *srcPart );
 658 
 659         // Derive from same parent.

 660         if( srcPart->IsAlias() )
 661         {
 662             std::shared_ptr< LIB_PART > srcParent = srcPart->GetParent().lock();
 663 
 664             wxCHECK( srcParent, /* void */ );
 665 
 666             newPart->SetParent( srcParent.get() );
 667         }
 668     }
 669 
 670     if( !newPart )
 671         return;
 672 
 673     ensureUniqueName( newPart, lib );
 674     m_libMgr->UpdatePart( newPart, lib );
 675     SyncLibraries( false );
 676     m_treePane->GetLibTree()->SelectLibId( LIB_ID( lib, newPart->GetName() ) );
 677 
 678     delete newPart;
 679 }

 682 void LIB_EDIT_FRAME::ensureUniqueName( LIB_PART* aPart, const wxString& aLibrary )
 683 {
 684     wxCHECK( aPart, /* void */ );
 685 
 686     int      i = 1;
 687     wxString newName = aPart->GetName();
 688 
 689     // Append a number to the name until the name is unique in the library.

 690     while( m_libMgr->PartExists( newName, aLibrary ) )
 691         newName.Printf( "%s_%d", aPart->GetName(), i++ );
 692 
 693     aPart->SetName( newName );
 694 }

 697 void LIB_EDIT_FRAME::Revert( bool aConfirm )
 698 {
 699     LIB_ID libId = getTargetLibId();
 700     const wxString& libName = libId.GetLibNickname();
 701 
 702     // Empty if this is the library itself that is selected.

 703     const wxString& partName = libId.GetLibItemName();
 704 
 705     wxString msg = wxString::Format( _( "Revert \"%s\" to last version saved?" ),
 706                                      partName.IsEmpty() ? libName : partName );
 707 
 708     if( aConfirm && !ConfirmRevertDialog( this, msg ) )
 709         return;
 710 
 711     bool reload_currentPart = false;
 712     wxString curr_partName = partName;
 713 
 714     if( GetCurPart() )
 715     {
 716         // the library itself is reverted: the current part will be reloaded only if it is

 717         // owned by this library

 718         if( partName.IsEmpty() )
 719         {
 720             LIB_ID curr_libId = GetCurPart()->GetLibId();
 721             reload_currentPart = libName == curr_libId.GetLibNickname();
 722 
 723             if( reload_currentPart )
 724                 curr_partName = curr_libId.GetLibItemName();
 725         }
 726         else
 727         {
 728             reload_currentPart = isCurrentPart( libId );
 729         }
 730     }
 731 
 732     int unit = m_unit;
 733 
 734     if( reload_currentPart )
 735         emptyScreen();
 736 
 737     if( partName.IsEmpty() )
 738     {
 739         m_libMgr->RevertLibrary( libName );
 740     }
 741     else
 742     {
 743         libId = m_libMgr->RevertPart( libId.GetLibItemName(), libId.GetLibNickname() );
 744 
 745         m_treePane->GetLibTree()->SelectLibId( libId );
 746         m_libMgr->ClearPartModified( libId.GetLibItemName(), libId.GetLibNickname() );
 747     }
 748 
 749     if( reload_currentPart && m_libMgr->PartExists( curr_partName, libName ) )
 750         LoadPart( curr_partName, libName, unit );
 751 
 752     m_treePane->Refresh();
 753     refreshSchematic();
 754 }

 789 bool LIB_EDIT_FRAME::saveLibrary( const wxString& aLibrary, bool aNewFile )
 790 {
 791     wxFileName fn;
 792     wxString   msg;
 793     SCH_IO_MGR::SCH_FILE_T fileType = SCH_IO_MGR::SCH_FILE_T::SCH_KICAD;
 794     PROJECT&   prj = Prj();
 795 
 796     m_toolManager->RunAction( ACTIONS::cancelInteractive, true );
 797 
 798     if( !aNewFile && ( aLibrary.empty() || !prj.SchSymbolLibTable()->HasLibrary( aLibrary ) ) )
 799     {
 800         DisplayError( this, _( "No library specified." ) );
 801         return false;
 802     }
 803 
 804     if( aNewFile )
 805     {
 806         SEARCH_STACK* search = prj.SchSearchS();
 807 
 808         // Get a new name for the library

 809         wxString default_path = prj.GetRString( PROJECT::SCH_LIB_PATH );
 810 
 811         if( !default_path )
 812             default_path = search->LastVisitedPath();
 813 
 814         fn.SetName( aLibrary );
 815         fn.SetExt( KiCadSymbolLibFileExtension );
 816 
 817         wxString wildcards = KiCadSymbolLibFileWildcard();
 818 
 819         wxFileDialog dlg( this, wxString::Format( _( "Save Library \"%s\" As..." ), aLibrary ),
 820                           default_path, fn.GetFullName(), wildcards,
 821                           wxFD_SAVE | wxFD_OVERWRITE_PROMPT );
 822 
 823         if( dlg.ShowModal() == wxID_CANCEL )
 824             return false;
 825 
 826         fn = dlg.GetPath();
 827 
 828         prj.SetRString( PROJECT::SCH_LIB_PATH, fn.GetPath() );
 829 
 830         if( fn.GetExt().IsEmpty() )
 831             fn.SetExt( KiCadSymbolLibFileExtension );
 832     }
 833     else
 834     {
 835         fn = prj.SchSymbolLibTable()->GetFullURI( aLibrary );
 836         fileType = SCH_IO_MGR::GuessPluginTypeFromLibPath( fn.GetFullPath() );
 837     }
 838 
 839     // Verify the user has write privileges before attempting to save the library file.

 840     if( !aNewFile && m_libMgr->IsLibraryReadOnly( aLibrary ) )
 841         return false;
 842 
 843     ClearMsgPanel();
 844 
 845     // Copy .kicad_symb file to .bak.

 846     if( !backupFile( fn, "bak" ) )
 847         return false;
 848 
 849     if( !m_libMgr->SaveLibrary( aLibrary, fn.GetFullPath(), fileType ) )
 850     {
 851         msg.Printf( _( "Failed to save changes to symbol library file \"%s\"" ),
 852                     fn.GetFullPath() );
 853         DisplayErrorMessage( this, _( "Error saving library" ), msg );
 854         return false;
 855     }
 856 
 857     if( !aNewFile )
 858         m_libMgr->ClearLibraryModified( aLibrary );
 859 
 860     ClearMsgPanel();
 861     msg.Printf( _( "Symbol library file \"%s\" saved" ), fn.GetFullPath() );
 862     RebuildSymbolUnitsList();
 863 
 864     return true;
 865 }

 868 bool LIB_EDIT_FRAME::saveAllLibraries( bool aRequireConfirmation )
 869 {
 870     bool doSave = true;
 871     int dirtyCount = 0;
 872     bool applyToAll = false;
 873 
 874     for( const auto& libNickname : m_libMgr->GetLibraryNames() )
 875     {
 876         if( m_libMgr->IsLibraryModified( libNickname ) )
 877             dirtyCount++;
 878     }
 879 
 880     for( const auto& libNickname : m_libMgr->GetLibraryNames() )
 881     {
 882         if( m_libMgr->IsLibraryModified( libNickname ) )
 883         {
 884             if( aRequireConfirmation && !applyToAll )
 885             {
 886                 wxString msg = wxString::Format( _( "Save changes to \"%s\" before closing?" ),
 887                                                  libNickname );
 888 
 889                 switch( UnsavedChangesDialog( this, msg, dirtyCount > 1 ? &applyToAll : nullptr ) )
 890                 {
 891                 case wxID_YES: doSave = true;  break;
 892                 case wxID_NO:  doSave = false; break;
 893                 default:
 894                 case wxID_CANCEL: return false;
 895                 }
 896             }
 897 
 898             if( doSave )
 899             {
 900                 // If saving under existing name fails then do a Save As..., and if that

 901                 // fails then cancel close action.

 902                 if( !m_libMgr->IsLibraryReadOnly( libNickname )
 903                   && saveLibrary( libNickname, false ) )
 904                     continue;
 905 
 906                 if( !saveLibrary( libNickname, true ) )
 907                     return false;
 908             }
 909         }
 910     }
 911 
 912     return true;
 913 }
