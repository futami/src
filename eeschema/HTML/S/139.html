  31 DIALOG_SCH_FIND::DIALOG_SCH_FIND( SCH_EDIT_FRAME* aParent, wxFindReplaceData* aData,
  32                                   const wxPoint& aPosition, const wxSize& aSize, int aStyle ) :
  33     DIALOG_SCH_FIND_BASE( aParent, wxID_ANY, _( "Find" ), aPosition, aSize,
  34                           wxDEFAULT_DIALOG_STYLE | wxRESIZE_BORDER | aStyle ),
  35     m_frame( aParent ),
  36     m_editorControl( m_frame->GetToolManager()->GetTool<SCH_EDITOR_CONTROL>() ),
  37     m_findReplaceData( aData ),
  38     m_findDirty( true )
  39 {
  40     wxASSERT_MSG( m_findReplaceData, wxT( "can't create find dialog without data" ) );
  41 
  42     if( aStyle & wxFR_REPLACEDIALOG )
  43     {
  44         SetTitle( _( "Find and Replace" ) );
  45         m_buttonReplace->Show( true );
  46         m_buttonReplaceAll->Show( true );
  47         m_staticReplace->Show( true );
  48         m_comboReplace->Show( true );
  49         m_checkReplaceReferences->Show( true );
  50         m_checkWildcardMatch->Show( false );  // Wildcard replace is not implemented.

  51     }
  52 
  53     int flags = m_findReplaceData->GetFlags();
  54     m_radioForward->SetValue( flags & wxFR_DOWN );
  55     m_radioBackward->SetValue( ( flags & wxFR_DOWN ) == 0 );
  56     m_checkMatchCase->SetValue( flags & wxFR_MATCHCASE );
  57     m_checkWholeWord->SetValue( flags & wxFR_WHOLEWORD );
  58 
  59     /* Whole word and wild card searches are mutually exclusive. */
  60     if( !( flags & wxFR_WHOLEWORD ) )
  61         m_checkWildcardMatch->SetValue( flags & FR_MATCH_WILDCARD );
  62 
  63     m_checkAllFields->SetValue( flags & FR_SEARCH_ALL_FIELDS );
  64     m_checkReplaceReferences->SetValue( flags & FR_REPLACE_REFERENCES );
  65     m_checkAllPins->SetValue( flags & FR_SEARCH_ALL_PINS );
  66     m_checkCurrentSheetOnly->SetValue( flags & FR_CURRENT_SHEET_ONLY );
  67 
  68     m_buttonFind->SetDefault();
  69     SetInitialFocus( m_comboFind );
  70 
  71     SetPosition( aPosition );
  72 
  73     // Adjust the height of the dialog to prevent controls from being hidden when

  74     // switching between the find and find/replace modes of the dialog.  This ignores

  75     // the users preferred height if any of the controls would be hidden.

  76     GetSizer()->SetSizeHints( this );
  77     wxSize size = aSize;
  78 
  79     if( aSize != wxDefaultSize )
  80     {
  81         wxSize bestSize = GetBestSize();
  82 
  83         if( size.GetHeight() != bestSize.GetHeight() )
  84             size.SetHeight( bestSize.GetHeight() );
  85     }
  86 
  87     SetSize( size );
  88 
  89     GetSizer()->Fit( this ); // Needed on Ubuntu/Unity to display the dialog

  90 
  91     Connect( wxEVT_CHAR, wxKeyEventHandler( DIALOG_SCH_FIND::OnChar ), nullptr, this );
  92 }

 152 void DIALOG_SCH_FIND::OnSearchForSelect( wxCommandEvent& aEvent )
 153 {
 154     m_findReplaceData->SetFindString( m_comboFind->GetValue() );
 155 
 156     // Move the search string to the top of the list if it isn't already there.

 157     if( aEvent.GetSelection() != 0 )
 158     {
 159         wxString tmp = m_comboFind->GetValue();
 160         m_comboFind->Delete( aEvent.GetSelection() );
 161         m_comboFind->Insert( tmp, 0 );
 162         m_comboFind->SetSelection( 0 );
 163     }
 164 
 165     m_editorControl->UpdateFind( ACTIONS::updateFind.MakeEvent() );
 166 }

 202 void DIALOG_SCH_FIND::OnOptions( wxCommandEvent& aEvent )
 203 {
 204     int flags = 0;
 205 
 206     if( m_radioForward->GetValue() )
 207         flags |= wxFR_DOWN;
 208 
 209     if( m_checkMatchCase->GetValue() )
 210         flags |= wxFR_MATCHCASE;
 211 
 212     if( m_checkWholeWord->GetValue() )
 213         flags |= wxFR_WHOLEWORD;
 214 
 215     if( m_checkWildcardMatch->IsShown() && m_checkWildcardMatch->GetValue() )
 216         flags |= FR_MATCH_WILDCARD;
 217 
 218     if( m_checkAllFields->GetValue() )
 219         flags |= FR_SEARCH_ALL_FIELDS;
 220 
 221     if( m_checkAllPins->GetValue() )
 222         flags |= FR_SEARCH_ALL_PINS;
 223 
 224     if( m_checkCurrentSheetOnly->GetValue() )
 225         flags |= FR_CURRENT_SHEET_ONLY;
 226 
 227     if( m_checkReplaceReferences->GetValue() )
 228         flags |= FR_REPLACE_REFERENCES;
 229 
 230     m_findReplaceData->SetFlags( flags );
 231     m_findDirty = true;
 232 }

 235 void DIALOG_SCH_FIND::OnFind( wxCommandEvent& aEvent )
 236 {
 237     int index = m_comboFind->FindString( m_comboFind->GetValue(), true );
 238 
 239     if( index == wxNOT_FOUND )
 240     {
 241         m_comboFind->Insert( m_comboFind->GetValue(), 0 );
 242     }
 243     else if( index != 0 )
 244     {
 245         /* Move the search string to the top of the list if it isn't already there. */
 246         wxString tmp = m_comboFind->GetValue();
 247         m_comboFind->Delete( index );
 248         m_comboFind->Insert( tmp, 0 );
 249         m_comboFind->SetSelection( 0 );
 250     }
 251 
 252     m_editorControl->FindNext( ACTIONS::findNext.MakeEvent());
 253 }

 256 void DIALOG_SCH_FIND::OnReplace( wxCommandEvent& aEvent )
 257 {
 258     int index = m_comboReplace->FindString( m_comboReplace->GetValue(), true );
 259 
 260     if( index == wxNOT_FOUND )
 261     {
 262         m_comboReplace->Insert( m_comboReplace->GetValue(), 0 );
 263     }
 264     else if( index != 0 )
 265     {
 266         /* Move the search string to the top of the list if it isn't already there. */
 267         wxString tmp = m_comboReplace->GetValue();
 268         m_comboReplace->Delete( index );
 269         m_comboReplace->Insert( tmp, 0 );
 270         m_comboReplace->SetSelection( 0 );
 271     }
 272 
 273     if( aEvent.GetId() == wxID_REPLACE )
 274         m_editorControl->ReplaceAndFindNext( ACTIONS::replaceAndFindNext.MakeEvent());
 275     else if( aEvent.GetId() == wxID_REPLACE_ALL )
 276         m_editorControl->ReplaceAll( ACTIONS::replaceAll.MakeEvent());
 277 }
