  42 class EE_COLLECTOR : public COLLECTOR
  43 {
  44 public:
  45     static const KICAD_T AllItems[];
  46     static const KICAD_T EditableItems[];
  47     static const KICAD_T ComponentsOnly[];
  48     static const KICAD_T SheetsOnly[];
  49 
  50     EE_COLLECTOR( const KICAD_T* aScanTypes = EE_COLLECTOR::AllItems ) :
  51         m_Unit( 0 ),
  52         m_Convert( 0 )
  53     {
  54         SetScanTypes( aScanTypes );
  55     }
  56 
  57     /**

  58      * Overload COLLECTOR::operator[](int) to return a SCH_ITEM instead of an EDA_ITEM.

  59      *

  60      * @param aIndex The index into the list.

  61      * @return SCH_ITEM* at \a aIndex or NULL.

  62      */
  63     SCH_ITEM* operator[]( int aIndex ) const override
  64     {
  65         if( (unsigned)aIndex < (unsigned)GetCount() )
  66             return (SCH_ITEM*) m_List[ aIndex ];
  67 
  68         return NULL;
  69     }
  70 
  71     SEARCH_RESULT Inspect( EDA_ITEM* aItem, void* aTestData ) override;
  72 
  73     /**

  74      * Function Collect

  75      * scans a EDA_ITEM using this class's Inspector method, which does the collection.

  76      * @param aScreen The eeschema screen to use for scanning

  77      * @param aFilterList A list of #KICAD_T types with a terminating #EOT, that determines

  78      *                    what is to be collected and the priority order of the resulting

  79      *                    collection.

  80      * @param aPos A wxPoint to use in hit-testing.

  81      * @param aUnit A symbol unit filter (for symbol editor)

  82      * @param aConvert A DeMorgan filter (for symbol editor)

  83      */
  84     void Collect( SCH_SCREEN* aScreen, const KICAD_T aFilterList[], const wxPoint& aPos,
  85             int aUnit = 0, int aConvert = 0 );
  86 
  87     /**

  88      * Function Collect

  89      * scans a EDA_ITEM using this class's Inspector method, which does the collection.

  90      * @param aItems The LIB_PART multivector holding the part components

  91      * @param aFilterList A list of #KICAD_T types with a terminating #EOT, that determines

  92      *                    what is to be collected and the priority order of the resulting

  93      *                    collection.

  94      * @param aPos A wxPoint to use in hit-testing.

  95      * @param aUnit A symbol unit filter (for symbol editor)

  96      * @param aConvert A DeMorgan filter (for symbol editor)

  97      */
  98     void Collect( LIB_ITEMS_CONTAINER& aItems, const KICAD_T aFilterList[], const wxPoint& aPos,
  99             int aUnit = 0, int aConvert = 0 );
 100 
 101     /**

 102      * Function IsCorner

 103      * tests if the collected items forms as corner of two line segments.

 104      * @return True if the collected items form a corner of two line segments.

 105      */
 106     bool IsCorner() const;

 109     int      m_Unit;            // Fixed symbol unit filter (for symbol editor)

 111 };
