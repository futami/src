  52 /**

  53  * Dialog class to select a component from the libraries. This is the master

  54  * View class in a Model-View-Adapter (mediated MVC) architecture. The other

  55  * pieces are in:

  56  *

  57  * - Adapter: CMP_TREE_MODEL_ADAPTER in common/cmp_tree_model_adapter.h

  58  * - Model: CMP_TREE_NODE and descendants in common/cmp_tree_model.h

  59  *

  60  * Because everything is tied together in the adapter class, see that file

  61  * for thorough documentation. A simple example usage follows:

  62  *

  63  *     // Create the adapter class

  64  *     auto adapter( SYMBOL_TREE_MODEL_ADAPTER::Create( Prj().SchSymbolLibTable() ) );

  65  *

  66  *     // Perform any configuration of adapter properties here

  67  *     adapter->SetPreselectNode( "LIB_NICKNAME", "SYMBO_NAME", 2 );

  68  *

  69  *     // Initialize model from #SYMBOL_LIB_TABLE

  70  *     libNicknames = libs->GetLogicalLibs();

  71  *

  72  *     for( auto nickname : libNicknames )

  73  *         adapter->AddLibrary( nickname );

  74  *

  75  *     // Create and display dialog

  76  *     DIALOG_CHOOSE_COMPONENT dlg( this, title, adapter, 1 );

  77  *     bool selected = ( dlg.ShowModal() != wxID_CANCEL );

  78  *

  79  *     // Receive part

  80  *     if( selected )

  81  *     {

  82  *         int unit;

  83  *         #LIB_ID id = dlg.GetSelectedAlias( &unit );

  84  *         do_something( id, unit );

  85  *     }

  86  *

  87  */
  88 class DIALOG_CHOOSE_COMPONENT : public DIALOG_SHIM
  89 {
  90 public:
  91     /**

  92      * Create dialog to choose component.

  93      *

  94      * @param aParent   a SCH_BASE_FRAME parent window.

  95      * @param aTitle    Dialog title.

  96      * @param aAdapter  SYMBOL_TREE_MODEL_ADAPTER::PTR. See CMP_TREE_MODEL_ADAPTER

  97      *                  for documentation.

  98      * @param aDeMorganConvert  preferred deMorgan conversion

  99      *                          (TODO: should happen in dialog)

 100      * @param aAllowFieldEdits  if false, all functions that allow the user to edit fields

 101      *                          (currently just footprint selection) will not be available.

 102      * @param aShowFootprints   if false, all footprint preview and selection features are

 103      *                          disabled. This forces aAllowFieldEdits false too.

 104      * @param aAllowBrowser     show a Select with Browser button

 105      */
 106     DIALOG_CHOOSE_COMPONENT( SCH_BASE_FRAME* aParent, const wxString& aTitle,
 107                              SYMBOL_TREE_MODEL_ADAPTER::PTR& aAdapter, int aDeMorganConvert,
 108                              bool aAllowFieldEdits, bool aShowFootprints, bool aAllowBrowser );
 109 
 110     ~DIALOG_CHOOSE_COMPONENT();
 111 
 112     /**

 113      * To be called after this dialog returns from ShowModal().

 114      *

 115      * For multi-unit components, if the user selects the component itself

 116      * rather than picking an individual unit, 0 will be returned in aUnit.

 117      * Beware that this is an invalid unit number - this should be replaced

 118      * with whatever default is desired (usually 1).

 119      *

 120      * @param aUnit if not NULL, the selected unit is filled in here.

 121      * @return the #LIB_ID of the symbol that has been selected.

 122      */
 123     LIB_ID GetSelectedLibId( int* aUnit = nullptr ) const;
 124 
 125     /**

 126      * To be called after this dialog returns from ShowModal()

 127      *

 128      * In the case of multi-unit symbols, this preferences asks to iterate through

 129      * all units of the symbol, one per click

 130      * @return The value of the dialog preference checkbox

 131      */
 132     bool GetUseAllUnits() const { return m_useUnits->GetValue(); }
 133 
 134     /**

 135      * To be called after this dialog returns from ShowModal()

 136      *

 137      * Keeps a new copy of the symbol on the mouse cursor, allowing the user to rapidly

 138      * place multiple copies of the same symbol on their schematic

 139      *

 140      * @return The value of the keep symbol preference checkbox

 141      */
 142     bool GetKeepSymbol() const { return m_keepSymbol->GetValue(); }
 143 
 144     /**

 145      * Get a list of fields edited by the user.

 146      * @return vector of pairs; each.first = field ID, each.second = new value

 147      */
 148     std::vector<std::pair<int, wxString>> GetFields() const
 149     {
 150         return m_field_edits;
 151     }
 152 
 153     /** Function IsExternalBrowserSelected

 154      *

 155      * @return true, iff the user pressed the thumbnail view of the component to

 156      *               launch the component browser.

 157      */
 158     bool IsExternalBrowserSelected() const
 159     {
 160         return m_external_browser_requested;
 161     }

 182      * is a component, the component is picked.

 224 };
