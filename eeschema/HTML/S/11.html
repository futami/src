  77 // Helper class to filter a list of libraries, and/or a list of PART_LIB

  78 // in dialogs

  79 class SCHLIB_FILTER
  80 {
  81     wxArrayString m_allowedLibs;        ///< a list of lib names to list some libraries

  82                                         ///< if empty: no filter

  83     bool          m_filterPowerParts;   ///< true to filter (show only) power parts

  84     bool          m_forceLoad;          // When true, load a part lib from the lib

  85                                         // which is given in m_allowedLibs[0]

  86 
  87 public:
  88     SCHLIB_FILTER()
  89     {
  90         m_filterPowerParts = false;
  91         m_forceLoad = false;
  92     }
  93 
  94     /**

  95      * add a lib name to the allowed libraries

  96      */
  97     void AddLib( const wxString& aLibName )
  98     {
  99         m_allowedLibs.Add( aLibName );
 100         m_forceLoad = false;
 101     }
 102 
 103 
 104     /**

 105      * add a lib name to the allowed libraries

 106      */
 107     void LoadFrom( const wxString& aLibName )
 108     {
 109         m_allowedLibs.Clear();
 110         m_allowedLibs.Add( aLibName );
 111         m_forceLoad = true;
 112     }
 113 
 114     /**

 115      * Clear the allowed libraries list (allows all libs)

 116      */
 117     void ClearLibList()
 118     {
 119         m_allowedLibs.Clear();
 120         m_forceLoad = false;
 121     }
 122 
 123     /**

 124      * set the filtering of power parts

 125      */
 126     void FilterPowerParts( bool aFilterEnable )
 127     {
 128         m_filterPowerParts = aFilterEnable;
 129     }
 130 
 131     // Accessors

 132 
 133     /**

 134      * @return true if the filtering of power parts is on

 135      */
 136     bool GetFilterPowerParts() const { return m_filterPowerParts; }
 137 
 138 
 139     /**

 140      * @return am wxArrayString of the names of allowed libs

 141      */
 142     const wxArrayString& GetAllowedLibList() const { return m_allowedLibs; }
 143 
 144     /**

 145      * @return the name of the lib to use to load a part, or an a empty string

 146      * Useful to load (in lib editor or lib viewer) a part from a given library

 147      */
 148     const wxString& GetLibSource() const
 149     {
 150         static wxString dummy;
 151 
 152         if( m_forceLoad && m_allowedLibs.GetCount() > 0 )
 153             return m_allowedLibs[0];
 154         else
 155             return dummy;
 156     }
 157 };

 312     int             m_mod_hash;     ///< incremented each time library is changed.

 313 
 314     SCH_IO_MGR::SCH_FILE_T        m_pluginType;
 315     std::unique_ptr< SCH_PLUGIN > m_plugin;
 316     std::unique_ptr< PROPERTIES > m_properties;   ///< Library properties

 370     /**

 371      * Find #LIB_PART by \a aName.

 376     LIB_PART* FindPart( const wxString& aName ) const;
 377 
 378     LIB_PART* FindPart( const LIB_ID& aLibId ) const;

 380     /**

 381      * Add \a aPart entry to library.

 382      *

 383      * @note A #LIB_PART can have an alias list so these alias will be added in library.

 384      *       and the any existing duplicate aliases will be removed from the library.

 385      *

 386      * @param aPart - Part to add, caller retains ownership, a clone is added.

 387      */
 388     void AddPart( LIB_PART* aPart );
 389 
 390     /**

 391      * Safely remove \a aEntry from the library and return the next entry.

 392      *

 393      * The next entry returned depends on the entry being removed.  If the entry being

 394      * remove also removes the part, then the next entry from the list is returned.

 395      * If the entry being used only removes an alias from a part, then the next alias

 396      * of the part is returned.

 397      *

 398      * @param aEntry - Entry to remove from library.

 399      * @return The next entry in the library or NULL if the library is empty.

 400      */
 401     LIB_PART* RemovePart( LIB_PART* aEntry );
 402 
 403     /**

 404      * Replace an existing part entry in the library.

 405      *

 406      * Note a part can have an alias list,

 407      * so these alias will be added in library (and previously existing alias removed)

 408      * @param aOldPart - The part to replace.

 409      * @param aNewPart - The new part.

 410      */
 411     LIB_PART* ReplacePart( LIB_PART* aOldPart, LIB_PART* aNewPart );
 412 
 413     /**

 414      * Return the file name without path or extension.

 415      *

 416      * @return Name of library file.

 417      */
 418     const wxString GetName() const            { return fileName.GetName(); }
 419 
 420     /**

 421      * Return the full file library name with path and extension.

 422      *

 423      * @return wxString - Full library file name with path and extension.

 424      */
 425     wxString GetFullFileName() const          { return fileName.GetFullPath(); }
 426 
 427     /**

 428      * Return the logical name of the library.

 429      *

 430      * @return wxString - The logical name of this library.

 431      */
 432     const wxString GetLogicalName() const
 433     {
 434         /*  for now is the filename without path or extension.

 435 

 436             Technically the library should not know its logical name!

 437             This will eventually come out of a pair of lookup tables using a

 438             reverse lookup using the full name or library pointer as a key.

 439             Search will be by project lookup table and then user lookup table if

 440             not found.

 441         */
 442         return fileName.GetName();
 443     }

 452      */
 453     static PART_LIB* LoadLibrary( const wxString& aFileName );
 454 };
