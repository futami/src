 101 int ERC_TESTER::TestDuplicateSheetNames( bool aCreateMarker )
 102 {
 103     SCH_SCREEN* screen;
 104     int         err_count = 0;
 105 
 106     SCH_SCREENS screenList( m_schematic->Root() );
 107 
 108     for( screen = screenList.GetFirst(); screen != nullptr; screen = screenList.GetNext() )
 109     {
 110         std::vector<SCH_SHEET*> list;
 111 
 112         for( SCH_ITEM* item : screen->Items().OfType( SCH_SHEET_T ) )
 113             list.push_back( static_cast<SCH_SHEET*>( item ) );
 114 
 115         for( size_t i = 0; i < list.size(); i++ )
 116         {
 117             SCH_SHEET* sheet = list[i];
 118 
 119             for( size_t j = i + 1; j < list.size(); j++ )
 120             {
 121                 SCH_SHEET* test_item = list[j];
 122 
 123                 // We have found a second sheet: compare names

 124                 // we are using case insensitive comparison to avoid mistakes between

 125                 // similar names like Mysheet and mysheet

 126                 if( sheet->GetName().CmpNoCase( test_item->GetName() ) == 0 )
 127                 {
 128                     if( aCreateMarker )
 129                     {
 130                         ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_DUPLICATE_SHEET_NAME );
 131                         ercItem->SetItems( sheet, test_item );
 132 
 133                         SCH_MARKER* marker = new SCH_MARKER( ercItem, sheet->GetPosition() );
 134                         screen->Append( marker );
 135                     }
 136 
 137                     err_count++;
 138                 }
 139             }
 140         }
 141     }
 142 
 143     return err_count;
 144 }

 147 void ERC_TESTER::TestTextVars( KIGFX::WS_PROXY_VIEW_ITEM* aWorksheet )
 148 {
 149     WS_DRAW_ITEM_LIST wsItems;
 150 
 151     if( aWorksheet )
 152     {
 153         wsItems.SetMilsToIUfactor( IU_PER_MILS );
 154         wsItems.BuildWorkSheetGraphicList( aWorksheet->GetPageInfo(), aWorksheet->GetTitleBlock() );
 155     }
 156 
 157     SCH_SCREENS screens( m_schematic->Root() );
 158 
 159     for( SCH_SCREEN* screen = screens.GetFirst(); screen != NULL; screen = screens.GetNext() )
 160     {
 161         for( SCH_ITEM* item : screen->Items().OfType( SCH_LOCATE_ANY_T ) )
 162         {
 163             if( item->Type() == SCH_COMPONENT_T )
 164             {
 165                 SCH_COMPONENT* component = static_cast<SCH_COMPONENT*>( item );
 166 
 167                 for( SCH_FIELD& field : component->GetFields() )
 168                 {
 169                     if( field.GetShownText().Matches( wxT( "*${*}*" ) ) )
 170                     {
 171                         wxPoint pos = field.GetPosition() - component->GetPosition();
 172                         pos = component->GetTransform().TransformCoordinate( pos );
 173                         pos += component->GetPosition();
 174 
 175                         ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_UNRESOLVED_VARIABLE );
 176                         ercItem->SetItems( &field );
 177 
 178                         SCH_MARKER* marker = new SCH_MARKER( ercItem, pos );
 179                         screen->Append( marker );
 180                     }
 181                 }
 182             }
 183             else if( item->Type() == SCH_SHEET_T )
 184             {
 185                 SCH_SHEET* sheet = static_cast<SCH_SHEET*>( item );
 186 
 187                 for( SCH_FIELD& field : sheet->GetFields() )
 188                 {
 189                     if( field.GetShownText().Matches( wxT( "*${*}*" ) ) )
 190                     {
 191                         ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_UNRESOLVED_VARIABLE );
 192                         ercItem->SetItems( &field );
 193 
 194                         SCH_MARKER* marker = new SCH_MARKER( ercItem, field.GetPosition() );
 195                         screen->Append( marker );
 196                     }
 197                 }
 198 
 199                 for( SCH_SHEET_PIN* pin : static_cast<SCH_SHEET*>( item )->GetPins() )
 200                 {
 201                     if( pin->GetShownText().Matches( wxT( "*${*}*" ) ) )
 202                     {
 203                         ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_UNRESOLVED_VARIABLE );
 204                         ercItem->SetItems( pin );
 205 
 206                         SCH_MARKER* marker = new SCH_MARKER( ercItem, pin->GetPosition() );
 207                         screen->Append( marker );
 208                     }
 209                 }
 210             }
 211             else if( SCH_TEXT* text = dynamic_cast<SCH_TEXT*>( item ) )
 212             {
 213                 if( text->GetShownText().Matches( wxT( "*${*}*" ) ) )
 214                 {
 215                     ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_UNRESOLVED_VARIABLE );
 216                     ercItem->SetItems( text );
 217 
 218                     SCH_MARKER* marker = new SCH_MARKER( ercItem, text->GetPosition() );
 219                     screen->Append( marker );
 220                 }
 221             }
 222         }
 223 
 224         for( WS_DRAW_ITEM_BASE* item = wsItems.GetFirst(); item; item = wsItems.GetNext() )
 225         {
 226             if( WS_DRAW_ITEM_TEXT* text = dynamic_cast<WS_DRAW_ITEM_TEXT*>( item ) )
 227             {
 228                 if( text->GetShownText().Matches( wxT( "*${*}*" ) ) )
 229                 {
 230                     ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_UNRESOLVED_VARIABLE );
 231                     ercItem->SetErrorMessage( _( "Unresolved text variable in worksheet." ) );
 232 
 233                     SCH_MARKER* marker = new SCH_MARKER( ercItem, text->GetPosition() );
 234                     screen->Append( marker );
 235                 }
 236             }
 237         }
 238     }
 239 }

 242 int ERC_TESTER::TestConflictingBusAliases()
 243 {
 244     wxString    msg;
 245     int         err_count = 0;
 246 
 247     SCH_SCREENS screens( m_schematic->Root() );
 248     std::vector< std::shared_ptr<BUS_ALIAS> > aliases;
 249 
 250     for( SCH_SCREEN* screen = screens.GetFirst(); screen != NULL; screen = screens.GetNext() )
 251     {
 252         std::unordered_set< std::shared_ptr<BUS_ALIAS> > screen_aliases = screen->GetBusAliases();
 253 
 254         for( const std::shared_ptr<BUS_ALIAS>& alias : screen_aliases )
 255         {
 256             for( const std::shared_ptr<BUS_ALIAS>& test : aliases )
 257             {
 258                 if( alias->GetName() == test->GetName() && alias->Members() != test->Members() )
 259                 {
 260                     msg.Printf( _( "Bus alias %s has conflicting definitions on %s and %s" ),
 261                                 alias->GetName(),
 262                                 alias->GetParent()->GetFileName(),
 263                                 test->GetParent()->GetFileName() );
 264 
 265                     ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_BUS_ALIAS_CONFLICT );
 266                     ercItem->SetErrorMessage( msg );
 267 
 268                     SCH_MARKER* marker = new SCH_MARKER( ercItem, wxPoint() );
 269                     test->GetParent()->Append( marker );
 270 
 271                     ++err_count;
 272                 }
 273             }
 274         }
 275 
 276         aliases.insert( aliases.end(), screen_aliases.begin(), screen_aliases.end() );
 277     }
 278 
 279     return err_count;
 280 }

 283 int ERC_TESTER::TestMultiunitFootprints()
 284 {
 285     SCH_SHEET_LIST sheets = m_schematic->GetSheets();
 286 
 287     int errors = 0;
 288     std::map<wxString, LIB_ID> footprints;
 289     SCH_MULTI_UNIT_REFERENCE_MAP refMap;
 290     sheets.GetMultiUnitComponents( refMap, true );
 291 
 292     for( auto& component : refMap )
 293     {
 294         auto& refList = component.second;
 295 
 296         if( refList.GetCount() == 0 )
 297         {
 298             wxFAIL;   // it should not happen

 299             continue;
 300         }
 301 
 302         // Reference footprint

 303         SCH_COMPONENT* unit = nullptr;
 304         wxString       unitName;
 305         wxString       unitFP;
 306 
 307         for( unsigned i = 0; i < component.second.GetCount(); ++i )
 308         {
 309             SCH_SHEET_PATH sheetPath = refList.GetItem( i ).GetSheetPath();
 310             unitFP = refList.GetItem( i ).GetComp()->GetField( FOOTPRINT )->GetText();
 311 
 312             if( !unitFP.IsEmpty() )
 313             {
 314                 unit = refList.GetItem( i ).GetComp();
 315                 unitName = unit->GetRef( &sheetPath, true );
 316                 break;
 317             }
 318         }
 319 
 320         for( unsigned i = 0; i < component.second.GetCount(); ++i )
 321         {
 322             SCH_REFERENCE& secondRef = refList.GetItem( i );
 323             SCH_COMPONENT* secondUnit = secondRef.GetComp();
 324             wxString       secondName = secondUnit->GetRef( &secondRef.GetSheetPath(), true );
 325             const wxString secondFp = secondUnit->GetField( FOOTPRINT )->GetText();
 326             wxString       msg;
 327 
 328             if( !secondFp.IsEmpty() && unitFP != secondFp )
 329             {
 330                 msg.Printf( _( "Different footprints assigned to %s and %s" ),
 331                             unitName, secondName );
 332 
 333                 ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_DIFFERENT_UNIT_FP );
 334                 ercItem->SetErrorMessage( msg );
 335                 ercItem->SetItems( unit, secondUnit );
 336 
 337                 SCH_MARKER* marker = new SCH_MARKER( ercItem, secondUnit->GetPosition() );
 338                 secondRef.GetSheetPath().LastScreen()->Append( marker );
 339 
 340                 ++errors;
 341             }
 342         }
 343     }
 344 
 345     return errors;
 346 }

 349 void ERC_TESTER::diagnose( NETLIST_OBJECT* aNetItemRef, NETLIST_OBJECT* aNetItemTst, int aMinConn,
 350                            PIN_ERROR aDiag )
 351 {
 352     if( aDiag == PIN_ERROR::OK || aMinConn < 1 || aNetItemRef->m_Type != NETLIST_ITEM::PIN )
 353         return;
 354 
 355     ERC_SETTINGS& settings = m_schematic->ErcSettings();
 356 
 357     SCH_PIN* pin = static_cast<SCH_PIN*>( aNetItemRef->m_Comp );
 358 
 359     if( aNetItemTst == NULL)
 360     {
 361         if( aMinConn == NOD )    /* Nothing driving the net. */
 362         {
 363             if( settings.GetSeverity( ERCE_PIN_NOT_DRIVEN ) != RPT_SEVERITY_IGNORE )
 364             {
 365                 ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_PIN_NOT_DRIVEN );
 366                 ercItem->SetItems( pin );
 367 
 368                 SCH_MARKER* marker = new SCH_MARKER( ercItem, aNetItemRef->m_Start );
 369                 aNetItemRef->m_SheetPath.LastScreen()->Append( marker );
 370             }
 371             return;
 372         }
 373     }
 374 
 375     if( aNetItemTst && aNetItemTst->m_Type == NETLIST_ITEM::PIN )  /* Error between 2 pins */
 376     {
 377         if( settings.GetSeverity( ERCE_PIN_TO_PIN_WARNING ) != RPT_SEVERITY_IGNORE )
 378         {
 379             ERC_ITEM* ercItem = ERC_ITEM::Create(
 380                     aDiag == PIN_ERROR::PP_ERROR ? ERCE_PIN_TO_PIN_ERROR : ERCE_PIN_TO_PIN_WARNING );
 381             ercItem->SetItems( pin, static_cast<SCH_PIN*>( aNetItemTst->m_Comp ) );
 382 
 383             SCH_MARKER* marker = new SCH_MARKER( ercItem, aNetItemRef->m_Start );
 384             aNetItemRef->m_SheetPath.LastScreen()->Append( marker );
 385         }
 386     }
 387 }

 390 void ERC_TESTER::TestOthersItems( NETLIST_OBJECT_LIST* aList, unsigned aNetItemRef,
 391                                   unsigned aNetStart, int* aMinConnexion )
 392 {
 393     ERC_SETTINGS& settings = m_schematic->ErcSettings();
 394 
 395     unsigned netItemTst = aNetStart;
 396     ELECTRICAL_PINTYPE jj;
 397     PIN_ERROR erc = PIN_ERROR::OK;
 398 
 399     /* Analysis of the table of connections. */
 400     ELECTRICAL_PINTYPE ref_elect_type = aList->GetItem( aNetItemRef )->m_ElectricalPinType;
 401     int local_minconn = NOC;
 402 
 403     if( ref_elect_type == ELECTRICAL_PINTYPE::PT_NC )
 404         local_minconn = NPI;
 405 
 406     /* Test pins connected to NetItemRef */
 407     for( ; ; netItemTst++ )
 408     {
 409         if( aNetItemRef == netItemTst )
 410             continue;
 411 
 412         // We examine only a given net. We stop the search if the net changes

 413         if( ( netItemTst >= aList->size() ) // End of list

 414             || ( aList->GetItemNet( aNetItemRef ) !=
 415                  aList->GetItemNet( netItemTst ) ) ) // End of net

 416         {
 417             /* End net code found: minimum connection test. */
 418             if( ( *aMinConnexion < NET_NC ) && ( local_minconn < NET_NC ) )
 419             {
 420                 /* Not connected or not driven pin. */
 421                 bool seterr = true;
 422 
 423                 if( local_minconn == NOC && aList->GetItemType( aNetItemRef ) == NETLIST_ITEM::PIN )
 424                 {
 425                     /* This pin is not connected: for multiple part per

 426                      * package, and duplicated pin,

 427                      * search for another instance of this pin

 428                      * this will be flagged only if all instances of this pin

 429                      * are not connected

 430                      * TODO test also if instances connected are connected to

 431                      * the same net

 432                      */
 433                     for( unsigned duplicate = 0; duplicate < aList->size(); duplicate++ )
 434                     {
 435                         if( aList->GetItemType( duplicate ) != NETLIST_ITEM::PIN )
 436                             continue;
 437 
 438                         if( duplicate == aNetItemRef )
 439                             continue;
 440 
 441                         if( aList->GetItem( aNetItemRef )->m_PinNum !=
 442                             aList->GetItem( duplicate )->m_PinNum )
 443                             continue;
 444 
 445                         if( ( (SCH_COMPONENT*) aList->GetItem( aNetItemRef )->
 446                              m_Link )->GetRef( &aList->GetItem( aNetItemRef )-> m_SheetPath ) !=
 447                             ( (SCH_COMPONENT*) aList->GetItem( duplicate )->m_Link )
 448                            ->GetRef( &aList->GetItem( duplicate )->m_SheetPath ) )
 449                             continue;
 450 
 451                         // Same component and same pin. Do dot create error for this pin

 452                         // if the other pin is connected (i.e. if duplicate net has another

 453                         // item)

 454                         if( (duplicate > 0)
 455                           && ( aList->GetItemNet( duplicate ) ==
 456                                aList->GetItemNet( duplicate - 1 ) ) )
 457                             seterr = false;
 458 
 459                         if( (duplicate < aList->size() - 1)
 460                           && ( aList->GetItemNet( duplicate ) ==
 461                                aList->GetItemNet( duplicate + 1 ) ) )
 462                             seterr = false;
 463                     }
 464                 }
 465 
 466                 if( seterr )
 467                 {
 468                     diagnose( aList->GetItem( aNetItemRef ), nullptr, local_minconn,
 469                             PIN_ERROR::WARNING );
 470                 }
 471 
 472                 *aMinConnexion = DRV;   // inhibiting other messages of this

 473                                        // type for the net.

 474             }
 475             return;
 476         }
 477 
 478         switch( aList->GetItemType( netItemTst ) )
 479         {
 480         case NETLIST_ITEM::ITEM_UNSPECIFIED:
 481         case NETLIST_ITEM::SEGMENT:
 482         case NETLIST_ITEM::BUS:
 483         case NETLIST_ITEM::JUNCTION:
 484         case NETLIST_ITEM::LABEL:
 485         case NETLIST_ITEM::HIERLABEL:
 486         case NETLIST_ITEM::BUSLABELMEMBER:
 487         case NETLIST_ITEM::HIERBUSLABELMEMBER:
 488         case NETLIST_ITEM::SHEETBUSLABELMEMBER:
 489         case NETLIST_ITEM::SHEETLABEL:
 490         case NETLIST_ITEM::GLOBLABEL:
 491         case NETLIST_ITEM::GLOBBUSLABELMEMBER:
 492         case NETLIST_ITEM::PINLABEL:
 493             break;
 494 
 495         case NETLIST_ITEM::NOCONNECT:
 496             local_minconn = std::max( NET_NC, local_minconn );
 497             break;
 498 
 499         case NETLIST_ITEM::PIN:
 500             jj            = aList->GetItem( netItemTst )->m_ElectricalPinType;
 501             local_minconn = std::max( settings.GetPinMinDrive( ref_elect_type, jj ),
 502                                       local_minconn );
 503 
 504             if( netItemTst <= aNetItemRef )
 505                 break;
 506 
 507             if( erc == PIN_ERROR::OK )
 508             {
 509                 erc = settings.GetPinMapValue( ref_elect_type, jj );
 510 
 511                 if( erc != PIN_ERROR::OK )
 512                 {
 513                     if( aList->GetConnectionType( netItemTst ) == NET_CONNECTION::UNCONNECTED )
 514                     {
 515                         aList->SetConnectionType( netItemTst,
 516                                                   NET_CONNECTION::NOCONNECT_SYMBOL_PRESENT );
 517                     }
 518 
 519                     diagnose( aList->GetItem( aNetItemRef ), aList->GetItem( netItemTst ), 1, erc );
 520                 }
 521             }
 522 
 523             break;
 524         }
 525     }
 526 }

 529 int ERC_TESTER::TestNoConnectPins()
 530 {
 531     int err_count = 0;
 532 
 533     for( const SCH_SHEET_PATH& sheet : m_schematic->GetSheets() )
 534     {
 535         std::map<wxPoint, std::vector<SCH_PIN*>> pinMap;
 536 
 537         for( SCH_ITEM* item : sheet.LastScreen()->Items().OfType( SCH_COMPONENT_T ) )
 538         {
 539             SCH_COMPONENT* comp = static_cast<SCH_COMPONENT*>( item );
 540 
 541             for( SCH_PIN* pin : comp->GetSchPins( &sheet ) )
 542             {
 543                 if( pin->GetLibPin()->GetType() == ELECTRICAL_PINTYPE::PT_NC )
 544                     pinMap[pin->GetPosition()].emplace_back( pin );
 545             }
 546         }
 547 
 548         for( auto& pair : pinMap )
 549         {
 550             if( pair.second.size() > 1 )
 551             {
 552                 err_count++;
 553 
 554                 ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_NOCONNECT_CONNECTED );
 555 
 556                 ercItem->SetItems( pair.second[0], pair.second[1],
 557                                    pair.second.size() > 2 ? pair.second[2] : nullptr,
 558                                    pair.second.size() > 3 ? pair.second[3] : nullptr );
 559                 ercItem->SetErrorMessage( _( "Pins with \"no connection\" type are connected" ) );
 560 
 561                 SCH_MARKER* marker = new SCH_MARKER( ercItem, pair.first );
 562                 sheet.LastScreen()->Append( marker );
 563             }
 564         }
 565     }
 566 
 567     return err_count;
 568 }

 610 void NETLIST_OBJECT_LIST::TestforSimilarLabels()
 611 {
 612     // Similar labels which are different when using case sensitive comparisons

 613     // but are equal when using case insensitive comparisons

 614 
 615     // list of all labels (used the better item to build diag messages)

 616     std::vector<NETLIST_OBJECT*> fullLabelList;
 617     // list of all labels , each label appears only once (used to to detect similar labels)

 618     std::set<NETLIST_OBJECT*, compare_labels> uniqueLabelList;
 619     wxString msg;
 620 
 621     // Build a list of differents labels. If inside a given sheet there are

 622     // more than one given label, only one label is stored.

 623     // not also the sheet labels are not taken in account for 2 reasons:

 624     //  * they are in the root sheet but they are seen only from the child sheet

 625     //  * any mismatch between child sheet hierarchical labels and the sheet label

 626     //    already detected by ERC

 627     for( unsigned netItem = 0; netItem < size(); ++netItem )
 628     {
 629         switch( GetItemType( netItem ) )
 630         {
 631         case NETLIST_ITEM::LABEL:
 632         case NETLIST_ITEM::BUSLABELMEMBER:
 633         case NETLIST_ITEM::PINLABEL:
 634         case NETLIST_ITEM::GLOBBUSLABELMEMBER:
 635         case NETLIST_ITEM::HIERLABEL:
 636         case NETLIST_ITEM::HIERBUSLABELMEMBER:
 637         case NETLIST_ITEM::GLOBLABEL:
 638             // add this label in lists

 639             uniqueLabelList.insert( GetItem( netItem ) );
 640             fullLabelList.push_back( GetItem( netItem ) );
 641             break;
 642 
 643         case NETLIST_ITEM::SHEETLABEL:
 644         case NETLIST_ITEM::SHEETBUSLABELMEMBER:
 645         default:
 646             break;
 647         }
 648     }
 649 
 650     // build global labels and compare

 651     std::set<NETLIST_OBJECT*, compare_label_names> loc_labelList;
 652 
 653     for( NETLIST_OBJECT* label : uniqueLabelList )
 654     {
 655         if( label->IsLabelGlobal() )
 656             loc_labelList.insert( label );
 657     }
 658 
 659     // compare global labels (same label names appears only once in list)

 660     for( auto it = loc_labelList.begin(); it != loc_labelList.end(); ++it )
 661     {
 662         auto it_aux = it;
 663 
 664         for( ++it_aux; it_aux != loc_labelList.end(); ++it_aux )
 665         {
 666             if( (*it)->m_Label.CmpNoCase( (*it_aux)->m_Label ) == 0 )
 667             {
 668                 // Create new marker for ERC.

 669                 int cntA = countIndenticalLabels( fullLabelList, *it );
 670                 int cntB = countIndenticalLabels( fullLabelList, *it_aux );
 671 
 672                 if( cntA <= cntB )
 673                     SimilarLabelsDiagnose( (*it), (*it_aux) );
 674                 else
 675                     SimilarLabelsDiagnose( (*it_aux), (*it) );
 676             }
 677         }
 678     }
 679 
 680     // Build paths list

 681     std::set<NETLIST_OBJECT*, compare_paths> pathsList;
 682 
 683     for( NETLIST_OBJECT* label : uniqueLabelList )
 684         pathsList.insert( label );
 685 
 686     // Examine each label inside a sheet path:

 687     for( NETLIST_OBJECT* candidate : pathsList )
 688     {
 689         loc_labelList.clear();
 690 
 691         for( NETLIST_OBJECT* uniqueLabel : uniqueLabelList)
 692         {
 693             if( candidate->m_SheetPath.Path() == uniqueLabel->m_SheetPath.Path() )
 694                 loc_labelList.insert( uniqueLabel );
 695         }
 696 
 697         // at this point, loc_labelList contains labels of the current sheet path.

 698         // Detect similar labels (same label names appears only once in list)

 699 
 700         for( auto ref_it = loc_labelList.begin(); ref_it != loc_labelList.end(); ++ref_it )
 701         {
 702             NETLIST_OBJECT* ref_item = *ref_it;
 703             auto it_aux = ref_it;
 704 
 705             for( ++it_aux; it_aux != loc_labelList.end(); ++it_aux )
 706             {
 707                 // global label versus global label was already examined.

 708                 // here, at least one label must be local

 709                 if( ref_item->IsLabelGlobal() && ( *it_aux )->IsLabelGlobal() )
 710                     continue;
 711 
 712                 if( ref_item->m_Label.CmpNoCase( ( *it_aux )->m_Label ) == 0 )
 713                 {
 714                     // Create new marker for ERC.

 715                     int cntA = countIndenticalLabels( fullLabelList, ref_item );
 716                     int cntB = countIndenticalLabels( fullLabelList, *it_aux );
 717 
 718                     if( cntA <= cntB )
 719                         SimilarLabelsDiagnose( ref_item, ( *it_aux ) );
 720                     else
 721                         SimilarLabelsDiagnose( ( *it_aux ), ref_item );
 722                 }
 723             }
 724         }
 725     }
 726 }

 729 // Helper function: count the number of labels identical to aLabel

 730 //  for global label: global labels in the full project

 731 //  for local label: all labels in the current sheet

 732 static int countIndenticalLabels( std::vector<NETLIST_OBJECT*>& aList, NETLIST_OBJECT* aRef )
 733 {
 734     int count = 0;
 735 
 736     if( aRef->IsLabelGlobal() )
 737     {
 738         for( NETLIST_OBJECT* i : aList )
 739         {
 740             if( i->IsLabelGlobal() && i->m_Label == aRef->m_Label )
 741                 count++;
 742         }
 743     }
 744     else
 745     {
 746         for( NETLIST_OBJECT* i : aList )
 747         {
 748             if( i->m_Label == aRef->m_Label && i->m_SheetPath.Path() == aRef->m_SheetPath.Path() )
 749                 count++;
 750         }
 751     }
 752 
 753     return count;
 754 }

 757 // Helper function: creates a marker for similar labels ERC warning

 758 static void SimilarLabelsDiagnose( NETLIST_OBJECT* aItemA, NETLIST_OBJECT* aItemB )
 759 {
 760     ERC_ITEM* ercItem = ERC_ITEM::Create( ERCE_SIMILAR_LABELS );
 761     ercItem->SetItems( aItemA->m_Comp, aItemB->m_Comp );
 762 
 763     SCH_MARKER* marker = new SCH_MARKER( ercItem, aItemA->m_Start );
 764     aItemA->m_SheetPath.LastScreen()->Append( marker );
 765 }
