  25 #ifndef LIB_BEZIER_H
  26 #define LIB_BEZIER_H
  27 
  28 #include <lib_item.h>
  29 
  30 
  31 /**

  32  * Define a bezier curve graphic body item.

  33  */
  34 class LIB_BEZIER : public LIB_ITEM
  35 {
  36     int m_Width;                           // Line width

  37     std::vector<wxPoint> m_BezierPoints;   // list of parameter (3|4)

  38     std::vector<wxPoint> m_PolyPoints;     // list of points (>= 2)

  39 
  40 public:
  41     LIB_BEZIER( LIB_PART * aParent );
  42 
  43     // Do not create a copy constructor.  The one generated by the compiler is adequate.

  44 
  45     ~LIB_BEZIER() { }
  46 
  47     wxString GetClass() const override
  48     {
  49         return wxT( "LIB_BEZIER" );
  50     }
  51 
  52     wxString GetTypeName() override
  53     {
  54         return _( "Bezier" );
  55     }
  56 
  57     void Reserve( size_t aCount ) { m_BezierPoints.reserve( aCount ); }
  58     void AddPoint( const wxPoint& aPoint ) { m_BezierPoints.push_back( aPoint ); }
  59 
  60     void Offset( const wxPoint& aOffset ) override;
  61     const wxPoint GetOffset() const;
  62 
  63     /**

  64      * @return the number of corners

  65      */
  66     unsigned GetCornerCount() const { return m_PolyPoints.size(); }
  67 
  68     const std::vector< wxPoint >& GetPoints() const { return m_BezierPoints; }
  69 
  70     bool HitTest( const wxPoint& aPosition, int aAccuracy = 0 ) const override;
  71     bool HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy = 0 ) const override;
  72 
  73     const EDA_RECT GetBoundingBox() const override;
  74 
  75     void MoveTo( const wxPoint& aPosition ) override;
  76 
  77     wxPoint GetPosition() const override;
  78 
  79     void MirrorHorizontal( const wxPoint& aCenter ) override;
  80     void MirrorVertical( const wxPoint& aCenter ) override;
  81     void Rotate( const wxPoint& aCenter, bool aRotateCCW = true ) override;
  82 
  83     void Plot( PLOTTER* aPlotter, const wxPoint& aOffset, bool aFill,
  84                const TRANSFORM& aTransform ) override;
  85 
  86     int GetWidth() const override { return m_Width; }
  87     void SetWidth( int aWidth ) override { m_Width = aWidth; }
  88 
  89     int GetPenWidth() const override;
  90 
  91     void GetMsgPanelInfo( EDA_DRAW_FRAME* aFrame, std::vector<MSG_PANEL_ITEM>& aList ) override;
  92 
  93     EDA_ITEM* Clone() const override;
  94 
  95 private:
  96     /**

  97      * @copydoc LIB_ITEM::compare()

  98      *

  99      * The bezier curve specific sort order for each curve segment point is as follows:

 100      *      - Bezier horizontal (X) point position.

 101      *      - Bezier vertical (Y) point position.

 102      */
 103     int compare( const LIB_ITEM& aOther,
 104             LIB_ITEM::COMPARE_FLAGS aCompareFlags = LIB_ITEM::COMPARE_FLAGS::NORMAL ) const override;
 105 
 106     void print( RENDER_SETTINGS* aSettings, const wxPoint& aOffset, void* aData,
 107                 const TRANSFORM& aTransform ) override;
 108 };

 111 #endif     // LIB_BEZIER_H
