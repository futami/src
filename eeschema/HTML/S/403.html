  37 struct SPICE_DC_PARAMS
  38 {
  39     wxString m_source;
  40     SPICE_VALUE m_vstart;
  41     SPICE_VALUE m_vend;
  42     SPICE_VALUE m_vincrement;
  43 };

  45 /// Special netlist exporter flavor that allows one to override simulation commands

  46 class NETLIST_EXPORTER_PSPICE_SIM : public NETLIST_EXPORTER_PSPICE
  47 {
  48 public:
  49     NETLIST_EXPORTER_PSPICE_SIM( SCHEMATIC* aSchematic ) :
  50             NETLIST_EXPORTER_PSPICE( aSchematic )
  51     {
  52     }
  53 
  54     /**

  55      * @brief Returns name of Spice dataset for a specific plot.

  56      * @param aName is name of the measured net or device

  57      * @param aType describes the type of expected plot

  58      * @param aParam is an optional parameter for devices, if absent it will return current (only

  59      * for passive devices).

  60      * @return Empty string if query is invalid, otherwise a plot name that

  61      * can be requested from the simulator.

  62      */
  63     wxString ComponentToVector( const wxString& aName, SIM_PLOT_TYPE aType,
  64             const wxString& aParam = wxEmptyString ) const;
  65 
  66     /**

  67      * @brief Returns name of Spice dataset for a specific plot.

  68      * @param aVector is name of the vector produced by ngspice

  69      * @param [out] aSignal is output in form: V(R1), Ib(Q2), I(L8)

  70      * @return [SPT_VOLTAGE, SPT_CURRENT]. Otherwise SPT_UNKNOWN if vector is

  71      *         of different, unsupported type.

  72      */
  73     SIM_PLOT_TYPE VectorToSignal( const std::string& aVector, wxString& aSignal ) const;
  74 
  75     /**

  76      * @brief Returns a list of currents that can be probed in a Spice primitive.

  77      */
  78     static const std::vector<wxString>& GetCurrents( SPICE_PRIMITIVE aPrimitive );
  79 
  80     /**

  81      * @brief Overrides the simulation command directive.

  82      */
  83     void SetSimCommand( const wxString& aCmd )
  84     {
  85         m_simCommand = aCmd;
  86     }
  87 
  88     /**

  89      * @brief Returns the simulation command directive.

  90      */
  91     const wxString& GetSimCommand() const
  92     {
  93         return m_simCommand;
  94     }
  95 
  96     /**

  97      * @brief Clears the simulation command directive.

  98      */
  99     void ClearSimCommand()
 100     {
 101         m_simCommand.Clear();
 102     }
 103 
 104     /**

 105      * Returns the command directive that is in use (either from the sheet or from m_simCommand

 106      * @return

 107      */
 108     wxString GetUsedSimCommand();
 109 
 110     /**

 111      * @brief Returns simulation type basing on the simulation command directives.

 112      * Simulation directives set using SetSimCommand() have priority over the ones placed in

 113      * schematic sheets.

 114      */
 115     SIM_TYPE GetSimType();
 116 
 117     /**

 118      * @brief Returns simulation command directives placed in schematic sheets (if any).

 119      */
 120     wxString GetSheetSimCommand();
 121 
 122     /**

 123      * Parses a two-source .dc command directive into its components

 124      *

 125      * @param aCmd is the input command string

 126      * @return true if the command was parsed successfully

 127      */
 128     bool ParseDCCommand( const wxString& aCmd, SPICE_DC_PARAMS* aSource1,
 129                          SPICE_DC_PARAMS* aSource2 );
 130 
 131     /**

 132      * @brief Determines if a directive is a simulation command.

 133      */
 134     static bool IsSimCommand( const wxString& aCmd )
 135     {
 136         return CommandToSimType( aCmd ) != ST_UNKNOWN;
 137     }
 138 
 139     /**

 140      * @brief Returns simulation type basing on a simulation command directive.

 141      */
 142     static SIM_TYPE CommandToSimType( const wxString& aCmd );
 143 
 144 protected:
 145     void writeDirectives( OUTPUTFORMATTER* aFormatter, unsigned aCtl ) const override;
 146 
 147 private:
 148 
 149     ///> Custom simulation command (has priority over the schematic sheet simulation commands)

 150     wxString m_simCommand;
 151 };
 152 
 153 #endif /* NETLIST_EXPORTER_PSPICE_SIM_H */
