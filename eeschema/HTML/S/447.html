  73 int LIB_DRAWING_TOOLS::TwoClickPlace( const TOOL_EVENT& aEvent )
  74 {
  75     KICAD_T           type = aEvent.Parameter<KICAD_T>();
  76     LIBEDIT_SETTINGS* settings = Pgm().GetSettingsManager().GetAppSettings<LIBEDIT_SETTINGS>();
  77     LIB_PIN_TOOL*     pinTool = type == LIB_PIN_T ? m_toolMgr->GetTool<LIB_PIN_TOOL>() : nullptr;
  78     VECTOR2I          cursorPos;
  79     EDA_ITEM*         item = nullptr;
  80 
  81     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
  82     getViewControls()->ShowCursor( true );
  83 
  84     std::string tool = aEvent.GetCommandStr().get();
  85     m_frame->PushTool( tool );
  86     Activate();
  87 
  88     // Prime the pump

  89     if( aEvent.HasPosition() )
  90         m_toolMgr->RunAction( ACTIONS::cursorClick );
  91 
  92     // Main loop: keep receiving events

  93     while( TOOL_EVENT* evt = Wait() )
  94     {
  95         m_frame->GetCanvas()->SetCurrentCursor( wxCURSOR_PENCIL );
  96         cursorPos = getViewControls()->GetCursorPosition( !evt->Modifier( MD_ALT ) );
  97 
  98         auto cleanup = [&] () {
  99             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 100             m_view->ClearPreview();
 101             delete item;
 102             item = nullptr;
 103         };
 104 
 105         if( evt->IsCancelInteractive() )
 106         {
 107             if( item )
 108                 cleanup();
 109             else
 110             {
 111                 m_frame->PopTool( tool );
 112                 break;
 113             }
 114         }
 115         else if( evt->IsActivate() )
 116         {
 117             if( item )
 118                 cleanup();
 119 
 120             if( evt->IsMoveTool() )
 121             {
 122                 // leave ourselves on the stack so we come back after the move

 123                 break;
 124             }
 125             else
 126             {
 127                 m_frame->PopTool( tool );
 128                 break;
 129             }
 130         }
 131 
 132         else if( evt->IsClick( BUT_LEFT ) )
 133         {
 134             LIB_PART* part = m_frame->GetCurPart();
 135 
 136             if( !part )
 137                 continue;
 138 
 139             // First click creates...

 140             if( !item )
 141             {
 142                 m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 143 
 144                 switch( type )
 145                 {
 146                 case LIB_PIN_T:
 147                 {
 148                     item = pinTool->CreatePin( wxPoint( cursorPos.x, -cursorPos.y ), part );
 149                     g_lastPinWeakPtr = item;
 150                     break;
 151                 }
 152                 case LIB_TEXT_T:
 153                 {
 154                     LIB_TEXT* text = new LIB_TEXT( part );
 155 
 156                     text->SetPosition( wxPoint( cursorPos.x, -cursorPos.y ) );
 157                     text->SetTextSize( wxSize( Mils2iu( settings->m_Defaults.text_size ),
 158                                                Mils2iu( settings->m_Defaults.text_size ) ) );
 159                     text->SetTextAngle( m_lastTextAngle );
 160 
 161                     DIALOG_LIB_EDIT_TEXT dlg( m_frame, text );
 162 
 163                     if( dlg.ShowModal() != wxID_OK )
 164                         delete text;
 165                     else
 166                         item = text;
 167 
 168                     break;
 169                 }
 170                 default:
 171                     wxFAIL_MSG( "TwoClickPlace(): unknown type" );
 172                 }
 173 
 174                 // Restore cursor after dialog

 175                 getViewControls()->WarpCursor( getViewControls()->GetCursorPosition(), true );
 176 
 177                 if( item )
 178                 {
 179                     item->SetFlags( IS_NEW | IS_MOVED );
 180                     m_view->ClearPreview();
 181                     m_view->AddToPreview( item->Clone() );
 182                     m_selectionTool->AddItemToSel( item );
 183                 }
 184 
 185                 getViewControls()->SetCursorPosition( cursorPos, false );
 186             }
 187 
 188             // ... and second click places:

 189             else
 190             {
 191                 m_frame->SaveCopyInUndoList( part );
 192 
 193                 switch( item->Type() )
 194                 {
 195                 case LIB_PIN_T:
 196                     pinTool->PlacePin( (LIB_PIN*) item );
 197                     break;
 198                 case LIB_TEXT_T:
 199                     part->AddDrawItem( (LIB_TEXT*) item );
 200                     break;
 201                 default:
 202                     wxFAIL_MSG( "TwoClickPlace(): unknown type" );
 203                 }
 204 
 205                 item->ClearEditFlags();
 206                 item = nullptr;
 207                 m_view->ClearPreview();
 208 
 209                 m_frame->RebuildView();
 210                 m_frame->OnModify();
 211             }
 212         }
 213         else if( evt->IsClick( BUT_RIGHT ) )
 214         {
 215             // Warp after context menu only if dragging...

 216             if( !item )
 217                 m_toolMgr->VetoContextMenuMouseWarp();
 218 
 219             m_menu.ShowContextMenu( m_selectionTool->GetSelection() );
 220         }
 221 
 222         else if( item && ( evt->IsAction( &ACTIONS::refreshPreview ) || evt->IsMotion() ) )
 223         {
 224             static_cast<LIB_ITEM*>( item )->SetPosition( wxPoint( cursorPos.x, -cursorPos.y ) );
 225             m_view->ClearPreview();
 226             m_view->AddToPreview( item->Clone() );
 227         }
 228 
 229         else
 230             evt->SetPassEvent();
 231 
 232         // Enable autopanning and cursor capture only when there is an item to be placed

 233         getViewControls()->SetAutoPan( item != nullptr );
 234         getViewControls()->CaptureCursor( item != nullptr );
 235     }
 236 
 237     return 0;
 238 }

 241 int LIB_DRAWING_TOOLS::DrawShape( const TOOL_EVENT& aEvent )
 242 {
 243     LIBEDIT_SETTINGS* settings = Pgm().GetSettingsManager().GetAppSettings<LIBEDIT_SETTINGS>();
 244     KICAD_T           type = aEvent.Parameter<KICAD_T>();
 245     EE_POINT_EDITOR*  pointEditor = m_toolMgr->GetTool<EE_POINT_EDITOR>();
 246 
 247     // We might be running as the same shape in another co-routine.  Make sure that one

 248     // gets whacked.

 249     m_toolMgr->DeactivateTool();
 250 
 251     m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 252     getViewControls()->ShowCursor( true );
 253 
 254     std::string tool = aEvent.GetCommandStr().get();
 255     m_frame->PushTool( tool );
 256     Activate();
 257 
 258     LIB_PART* part = m_frame->GetCurPart();
 259     LIB_ITEM* item = nullptr;
 260 
 261     // Prime the pump

 262     if( aEvent.HasPosition() )
 263         m_toolMgr->RunAction( ACTIONS::cursorClick );
 264 
 265     // Main loop: keep receiving events

 266     while( TOOL_EVENT* evt = Wait() )
 267     {
 268         if( !pointEditor->HasPoint() )
 269             m_frame->GetCanvas()->SetCurrentCursor( wxCURSOR_PENCIL );
 270 
 271         VECTOR2I cursorPos = getViewControls()->GetCursorPosition( !evt->Modifier( MD_ALT ) );
 272 
 273         auto cleanup = [&] () {
 274             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 275             m_view->ClearPreview();
 276             delete item;
 277             item = nullptr;
 278         };
 279 
 280         if( evt->IsCancelInteractive() )
 281         {
 282             if( item )
 283                 cleanup();
 284             else
 285             {
 286                 m_frame->PopTool( tool );
 287                 break;
 288             }
 289         }
 290 
 291         else if( evt->IsActivate() )
 292         {
 293             if( item )
 294                 cleanup();
 295 
 296             if( evt->IsPointEditor() )
 297             {
 298                 // don't exit (the point editor runs in the background)

 299             }
 300             else if( evt->IsMoveTool() )
 301             {
 302                 // leave ourselves on the stack so we come back after the move

 303                 break;
 304             }
 305             else
 306             {
 307                 m_frame->PopTool( tool );
 308                 break;
 309             }
 310         }
 311 
 312         else if( evt->IsClick( BUT_LEFT ) && !item )
 313         {
 314             if( !part )
 315                 continue;
 316 
 317             m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 318 
 319             switch( type )
 320             {
 321             case LIB_ARC_T:       item = new LIB_ARC( part );       break;
 322             case LIB_CIRCLE_T:    item = new LIB_CIRCLE( part );    break;
 323             case LIB_POLYLINE_T:  item = new LIB_POLYLINE( part );  break;
 324             case LIB_RECTANGLE_T: item = new LIB_RECTANGLE( part ); break;
 325             default: break;     // keep compiler quiet

 326             }
 327 
 328             wxASSERT( item );
 329 
 330             item->SetWidth( settings->m_Defaults.line_width );
 331             item->SetFillMode( m_lastFillStyle );
 332             item->SetFlags( IS_NEW );
 333             item->BeginEdit( wxPoint( cursorPos.x, -cursorPos.y ) );
 334 
 335             if( m_drawSpecificUnit )
 336                 item->SetUnit( m_frame->GetUnit() );
 337 
 338             if( m_drawSpecificConvert )
 339                 item->SetConvert( m_frame->GetConvert() );
 340 
 341             m_selectionTool->AddItemToSel( item );
 342         }
 343 
 344         else if( item && ( evt->IsClick( BUT_LEFT ) || evt->IsDblClick( BUT_LEFT )
 345                         || evt->IsAction( &EE_ACTIONS::finishDrawing ) ) )
 346         {
 347             if( evt->IsDblClick( BUT_LEFT ) || evt->IsAction( &EE_ACTIONS::finishDrawing )
 348                     || !item->ContinueEdit( wxPoint( cursorPos.x, -cursorPos.y ) ) )
 349             {
 350                 item->EndEdit();
 351                 item->ClearEditFlags();
 352                 m_view->ClearPreview();
 353 
 354                 m_frame->SaveCopyInUndoList( part );
 355                 part->AddDrawItem( item );
 356                 item = nullptr;
 357 
 358                 m_frame->RebuildView();
 359                 m_frame->OnModify();
 360                 m_toolMgr->RunAction( ACTIONS::activatePointEditor );
 361             }
 362         }
 363 
 364         else if( item && ( evt->IsAction( &ACTIONS::refreshPreview ) || evt->IsMotion() ) )
 365         {
 366             item->CalcEdit( wxPoint( cursorPos.x, -cursorPos.y) );
 367             m_view->ClearPreview();
 368             m_view->AddToPreview( item->Clone() );
 369         }
 370 
 371         else if( evt->IsDblClick( BUT_LEFT ) && !item )
 372         {
 373             m_toolMgr->RunAction( EE_ACTIONS::properties, true );
 374         }
 375 
 376         else if( evt->IsClick( BUT_RIGHT ) )
 377         {
 378             // Warp after context menu only if dragging...

 379             if( !item )
 380                 m_toolMgr->VetoContextMenuMouseWarp();
 381 
 382             m_menu.ShowContextMenu( m_selectionTool->GetSelection() );
 383         }
 384 
 385         else
 386             evt->SetPassEvent();
 387 
 388         // Enable autopanning and cursor capture only when there is a shape being drawn

 389         getViewControls()->SetAutoPan( item != nullptr );
 390         getViewControls()->CaptureCursor( item != nullptr );
 391     }
 392 
 393     return 0;
 394 }

 397 int LIB_DRAWING_TOOLS::PlaceAnchor( const TOOL_EVENT& aEvent )
 398 {
 399     getViewControls()->ShowCursor( true );
 400     getViewControls()->SetSnapping( true );
 401 
 402     std::string tool = aEvent.GetCommandStr().get();
 403     m_frame->PushTool( tool );
 404     Activate();
 405 
 406     // Main loop: keep receiving events

 407     while( TOOL_EVENT* evt = Wait() )
 408     {
 409         m_frame->GetCanvas()->SetCurrentCursor( wxCURSOR_BULLSEYE );
 410 
 411         if( evt->IsCancelInteractive() )
 412         {
 413             m_frame->PopTool( tool );
 414             break;
 415         }
 416         else if( evt->IsActivate() )
 417         {
 418             m_frame->PopTool( tool );
 419             break;
 420         }
 421         else if( evt->IsClick( BUT_LEFT ) )
 422         {
 423             LIB_PART* part = m_frame->GetCurPart();
 424 
 425             if( !part )
 426                 continue;
 427 
 428             VECTOR2I cursorPos = getViewControls()->GetCursorPosition( !evt->Modifier( MD_ALT ) );
 429             wxPoint  offset( -cursorPos.x, cursorPos.y );
 430 
 431             part->SetOffset( offset );
 432 
 433             // Refresh the view without changing the viewport

 434             auto center = m_view->GetCenter();
 435             center.x += offset.x;
 436             center.y -= offset.y;
 437             m_view->SetCenter( center );
 438             m_view->RecacheAllItems();
 439             m_frame->OnModify();
 440         }
 441         else if( evt->IsClick( BUT_RIGHT ) )
 442         {
 443             m_menu.ShowContextMenu( m_selectionTool->GetSelection() );
 444         }
 445         else
 446             evt->SetPassEvent();
 447     }
 448 
 449     return 0;
 450 }

 453 int LIB_DRAWING_TOOLS::RepeatDrawItem( const TOOL_EVENT& aEvent )
 454 {
 455     LIB_PIN_TOOL* pinTool = m_toolMgr->GetTool<LIB_PIN_TOOL>();
 456     LIB_PART*     part = m_frame->GetCurPart();
 457     LIB_PIN*      sourcePin = nullptr;
 458 
 459     if( !part )
 460         return 0;
 461 
 462     // See if we have a pin matching our weak ptr

 463     for( LIB_PIN* test = part->GetNextPin();  test;  test = part->GetNextPin( test ) )
 464     {
 465         if( (void*) test == g_lastPinWeakPtr )
 466             sourcePin = test;
 467     }
 468 
 469     if( sourcePin )
 470     {
 471         LIB_PIN* pin = pinTool->RepeatPin( sourcePin );
 472         g_lastPinWeakPtr = pin;
 473 
 474         m_toolMgr->RunAction( EE_ACTIONS::clearSelection, true );
 475 
 476         if( pin )
 477             m_toolMgr->RunAction( EE_ACTIONS::addItemToSel, true, pin );
 478     }
 479 
 480     return 0;
 481 }
