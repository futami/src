  36 SYMBOL_PREVIEW_WIDGET::SYMBOL_PREVIEW_WIDGET( wxWindow* aParent, KIWAY& aKiway,
  37                                               EDA_DRAW_PANEL_GAL::GAL_TYPE aCanvasType ) :
  38     wxPanel( aParent, wxID_ANY, wxDefaultPosition, wxDefaultSize, 0 ),
  39     m_kiway( aKiway ),
  40     m_preview( nullptr ), m_status( nullptr ), m_statusSizer( nullptr ), m_previewItem( nullptr )
  41 {
  42     auto common_settings = Pgm().GetCommonSettings();
  43     auto app_settings = Pgm().GetSettingsManager().GetAppSettings<EESCHEMA_SETTINGS>();
  44 
  45     m_galDisplayOptions.ReadConfig( *common_settings, app_settings->m_Window, this );
  46     m_galDisplayOptions.m_forceDisplayCursor = false;
  47 
  48     EDA_DRAW_PANEL_GAL::GAL_TYPE canvasType = aCanvasType;
  49 
  50     // Allows only a CAIRO or OPENGL canvas:

  51     if( canvasType != EDA_DRAW_PANEL_GAL::GAL_TYPE_OPENGL &&
  52         canvasType != EDA_DRAW_PANEL_GAL::GAL_TYPE_CAIRO )
  53         canvasType = EDA_DRAW_PANEL_GAL::GAL_TYPE_OPENGL;
  54 
  55     m_preview = new SCH_PREVIEW_PANEL( aParent, wxID_ANY, wxDefaultPosition, wxSize( -1, -1 ),
  56                                        m_galDisplayOptions, canvasType );
  57     m_preview->SetStealsFocus( false );
  58 
  59     // Do not display the grid: the look is not good for a small canvas area.

  60     // But mainly, due to some strange bug I (JPC) was unable to fix, the grid creates

  61     // strange artifacts on Windows when eeschema is run from Kicad manager (but not in

  62     // stand alone...).

  63     m_preview->GetGAL()->SetGridVisibility( false );
  64 
  65     // Early initialization of the canvas background color,

  66     // before any OnPaint event is fired for the canvas using a wrong bg color

  67     KIGFX::VIEW* view = m_preview->GetView();
  68     auto settings = static_cast<KIGFX::SCH_RENDER_SETTINGS*>( view->GetPainter()->GetSettings() );
  69 
  70     if( auto* theme = Pgm().GetSettingsManager().GetColorSettings( app_settings->m_ColorTheme ) )
  71         settings->LoadColors( theme );
  72 
  73     m_preview->GetGAL()->SetClearColor( settings->GetBackgroundColor() );
  74 
  75     SetBackgroundColour( *wxWHITE );
  76     SetForegroundColour( *wxBLACK );
  77 
  78     m_status = new wxStaticText( this, wxID_ANY, wxEmptyString );
  79     m_statusSizer = new wxBoxSizer( wxVERTICAL );
  80     m_statusSizer->Add( 0, 0, 1 );  // add a spacer

  81     m_statusSizer->Add( m_status, 0, wxALIGN_CENTER );
  82     m_statusSizer->Add( 0, 0, 1 );  // add a spacer

  83 
  84     auto outer_sizer = new wxBoxSizer( wxVERTICAL );
  85     outer_sizer->Add( m_preview, 1, wxTOP | wxEXPAND, 5 );
  86     outer_sizer->Add( m_statusSizer, 1, wxALIGN_CENTER );
  87 
  88     m_statusSizer->ShowItems( false );
  89 
  90     SetSizer( outer_sizer );
  91 
  92     Connect( wxEVT_SIZE, wxSizeEventHandler( SYMBOL_PREVIEW_WIDGET::onSize ), NULL, this );
  93 }

 126 void SYMBOL_PREVIEW_WIDGET::fitOnDrawArea()
 127 {
 128     if( !m_previewItem )
 129         return;
 130 
 131     // set the view scale to fit the item on screen

 132     KIGFX::VIEW* view = m_preview->GetView();
 133 
 134     // Calculate the drawing area size, in internal units, for a scaling factor = 1.0

 135     view->SetScale( 1.0 );
 136     VECTOR2D  clientSize = view->ToWorld( m_preview->GetClientSize(), false );
 137     // Calculate the draw scale to fit the drawing area

 138     double    scale = std::min( fabs( clientSize.x / m_itemBBox.GetWidth() ),
 139                                 fabs( clientSize.y / m_itemBBox.GetHeight() ) );
 140 
 141     // Above calculation will yield an exact fit; add a bit of whitespace around symbol

 142     scale /= 1.2;
 143 
 144     // Now fix the best scale

 145     view->SetScale( scale );
 146     view->SetCenter( m_itemBBox.Centre() );
 147 }

 150 void SYMBOL_PREVIEW_WIDGET::DisplaySymbol( const LIB_ID& aSymbolID, int aUnit, int aConvert )
 151 {
 152     KIGFX::VIEW* view = m_preview->GetView();
 153     auto settings = static_cast<KIGFX::SCH_RENDER_SETTINGS*>( view->GetPainter()->GetSettings() );
 154     std::unique_ptr< LIB_PART > symbol;
 155 
 156     try
 157     {
 158         LIB_PART* tmp = m_kiway.Prj().SchSymbolLibTable()->LoadSymbol( aSymbolID );
 159 
 160         if( tmp )
 161             symbol = tmp->Flatten();
 162     }
 163     catch( const IO_ERROR& ioe )
 164     {
 165         wxLogError( wxString::Format( _( "Error loading symbol %s from library %s.\n\n%s" ),
 166                                       aSymbolID.GetLibItemName().wx_str(),
 167                                       aSymbolID.GetLibNickname().wx_str(),
 168                                       ioe.What() ) );
 169     }
 170 
 171     if( m_previewItem )
 172     {
 173         view->Remove( m_previewItem );
 174         delete m_previewItem;
 175         m_previewItem = nullptr;
 176     }
 177 
 178     if( symbol )
 179     {
 180         // This will flatten derived parts so that the correct final symbol can be shown.

 181         m_previewItem = symbol.release();
 182 
 183         // If unit isn't specified for a multi-unit part, pick the first.  (Otherwise we'll

 184         // draw all of them.)

 185         settings->m_ShowUnit = ( m_previewItem->IsMulti() && aUnit == 0 ) ? 1 : aUnit;
 186 
 187         // For symbols having a De Morgan body style, use the first style

 188         settings->m_ShowConvert =
 189                 ( m_previewItem->HasConversion() && aConvert == 0 ) ? 1 : aConvert;
 190 
 191         view->Add( m_previewItem );
 192 
 193         // Get the symbole size, in internal units

 194         m_itemBBox = m_previewItem->GetUnitBoundingBox( settings->m_ShowUnit,
 195                                                         settings->m_ShowConvert );
 196 
 197         if( !m_preview->IsShown() )
 198         {
 199             m_preview->Show();
 200             m_statusSizer->ShowItems( false );
 201             Layout();   // Ensure panel size is up to date.

 202         }
 203 
 204         // Calculate the draw scale to fit the drawing area

 205         fitOnDrawArea();
 206     }
 207 
 208     m_preview->ForceRefresh();
 209 }

 212 void SYMBOL_PREVIEW_WIDGET::DisplayPart( LIB_PART* aPart, int aUnit, int aConvert )
 213 {
 214     KIGFX::VIEW* view = m_preview->GetView();
 215 
 216     if( m_previewItem )
 217     {
 218         view->Remove( m_previewItem );
 219         delete m_previewItem;
 220         m_previewItem = nullptr;
 221     }
 222 
 223     if( aPart )
 224     {
 225         m_previewItem = new LIB_PART( *aPart );
 226 
 227         // For symbols having a De Morgan body style, use the first style

 228         auto settings = static_cast<KIGFX::SCH_RENDER_SETTINGS*>( view->GetPainter()->GetSettings() );
 229         // If unit isn't specified for a multi-unit part, pick the first.  (Otherwise we'll

 230         // draw all of them.)

 231         settings->m_ShowUnit = ( m_previewItem->IsMulti() && aUnit == 0 ) ? 1 : aUnit;
 232 
 233         settings->m_ShowConvert =
 234                 ( m_previewItem->HasConversion() && aConvert == 0 ) ? 1 : aConvert;
 235 
 236         view->Add( m_previewItem );
 237 
 238         // Get the symbole size, in internal units

 239         m_itemBBox = aPart->GetUnitBoundingBox( settings->m_ShowUnit, settings->m_ShowConvert );
 240 
 241         // Calculate the draw scale to fit the drawing area

 242         fitOnDrawArea();
 243     }
 244 
 245     m_preview->ForceRefresh();
 246     m_preview->Show();
 247     m_statusSizer->ShowItems( false );
 248 }
