 108  */
 109 void SCH_EDIT_FRAME::ReCreateVToolbar()
 110 {
 111     if( m_drawToolBar )
 112         m_drawToolBar->ClearToolbar();
 113     else
 114         m_drawToolBar = new ACTION_TOOLBAR( this, ID_V_TOOLBAR, wxDefaultPosition, wxDefaultSize,
 115                                             KICAD_AUI_TB_STYLE | wxAUI_TB_VERTICAL );

 118     m_drawToolBar->Add( ACTIONS::selectionTool,             ACTION_TOOLBAR::TOGGLE );

 126     m_drawToolBar->Add( EE_ACTIONS::placeBusWireEntry,      ACTION_TOOLBAR::TOGGLE );
 127     m_drawToolBar->Add( EE_ACTIONS::placeNoConnect,         ACTION_TOOLBAR::TOGGLE );
 128     m_drawToolBar->Add( EE_ACTIONS::placeJunction,          ACTION_TOOLBAR::TOGGLE );
 129     m_drawToolBar->Add( EE_ACTIONS::placeLabel,             ACTION_TOOLBAR::TOGGLE );
 130     m_drawToolBar->Add( EE_ACTIONS::placeGlobalLabel,       ACTION_TOOLBAR::TOGGLE );
 131     m_drawToolBar->Add( EE_ACTIONS::placeHierLabel, ACTION_TOOLBAR::TOGGLE );
 132     m_drawToolBar->Add( EE_ACTIONS::drawSheet,              ACTION_TOOLBAR::TOGGLE );

 145 /* Create Vertical Left Toolbar (Option Toolbar)

 146  */
 147 void SCH_EDIT_FRAME::ReCreateOptToolbar()
 148 {
 149     if( m_optionsToolBar )
 150         m_optionsToolBar->ClearToolbar();
 151     else
 152         m_optionsToolBar = new ACTION_TOOLBAR( this, ID_OPT_TOOLBAR,
 153                                                wxDefaultPosition, wxDefaultSize,
 154                                                KICAD_AUI_TB_STYLE | wxAUI_TB_VERTICAL );
 155 
 156     m_optionsToolBar->Add( ACTIONS::toggleGrid,          ACTION_TOOLBAR::TOGGLE );
 157     m_optionsToolBar->Add( ACTIONS::imperialUnits,       ACTION_TOOLBAR::TOGGLE );
 158     m_optionsToolBar->Add( ACTIONS::metricUnits,         ACTION_TOOLBAR::TOGGLE );
 159     m_optionsToolBar->Add( ACTIONS::toggleCursorStyle,   ACTION_TOOLBAR::TOGGLE );
 160     m_optionsToolBar->Add( EE_ACTIONS::toggleHiddenPins, ACTION_TOOLBAR::TOGGLE );
 161     m_optionsToolBar->Add( EE_ACTIONS::toggleForceHV,    ACTION_TOOLBAR::TOGGLE );
 162 
 163     m_optionsToolBar->Realize();
 164 }

 167 void SCH_EDIT_FRAME::SyncToolbars()
 168 {
 169 #define TOGGLE_TOOL( toolbar, tool ) toolbar->Toggle( tool, IsCurrentTool( tool ) )
 170 
 171     KIGFX::GAL_DISPLAY_OPTIONS& galOpts = GetGalDisplayOptions();
 172 
 173 
 174     m_mainToolBar->Toggle( ACTIONS::save, IsContentModified() );
 175     m_mainToolBar->Toggle( ACTIONS::undo, GetUndoCommandCount() > 0 );
 176     m_mainToolBar->Toggle( ACTIONS::redo, GetRedoCommandCount() > 0 );
 177     TOGGLE_TOOL( m_mainToolBar, ACTIONS::zoomTool );
 178     m_mainToolBar->Refresh();
 179 
 180     m_optionsToolBar->Toggle( ACTIONS::toggleGrid,          IsGridVisible() );
 181     m_optionsToolBar->Toggle( ACTIONS::metricUnits,         GetUserUnits() != EDA_UNITS::INCHES );
 182     m_optionsToolBar->Toggle( ACTIONS::imperialUnits,       GetUserUnits() == EDA_UNITS::INCHES );
 183     m_optionsToolBar->Toggle( ACTIONS::toggleCursorStyle,   galOpts.m_fullscreenCursor );
 184     m_optionsToolBar->Toggle( EE_ACTIONS::toggleHiddenPins, GetShowAllPins() );
 185     m_optionsToolBar->Toggle( EE_ACTIONS::toggleForceHV,    eeconfig()->m_Drawing.hv_lines_only );
 186     m_optionsToolBar->Refresh();
 187 
 188     TOGGLE_TOOL( m_drawToolBar, ACTIONS::selectionTool );
 189     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::highlightNetTool );
 190     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placeSymbol );
 191     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placePower );
 192     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::drawWire );
 193     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::drawBus );
 194     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placeBusWireEntry );
 195     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placeNoConnect );
 196     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placeJunction );
 197     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placeLabel );
 198     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placeGlobalLabel );
 199     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placeHierLabel );
 200     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::drawSheet );
 201     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::importSheetPin );
 202     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::drawLines );
 203     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placeSchematicText );
 204     TOGGLE_TOOL( m_drawToolBar, EE_ACTIONS::placeImage );
 205     TOGGLE_TOOL( m_drawToolBar, ACTIONS::deleteTool );
 206     m_drawToolBar->Refresh();
 207 }
