  25 #ifndef EE_POINT_EDITOR_H
  26 #define EE_POINT_EDITOR_H
  27 
  28 #include <tools/ee_tool_base.h>
  29 #include <tool/edit_points.h>
  30 #include <tool/selection.h>
  31 
  32 class EE_SELECTION_TOOL;
  33 class SCH_BASE_FRAME;
  34 
  35 /**

  36  * EE_POINT_EDITOR

  37  *

  38  * Tool that displays edit points allowing to modify items by dragging the points.

  39  */
  40 class EE_POINT_EDITOR : public EE_TOOL_BASE<SCH_BASE_FRAME>
  41 {
  42 public:
  43     EE_POINT_EDITOR();
  44 
  45     /// @copydoc TOOL_INTERACTIVE::Reset()

  46     void Reset( RESET_REASON aReason ) override;
  47 
  48     /// @copydoc TOOL_INTERACTIVE::Init()

  49     bool Init() override;
  50 
  51     int Main( const TOOL_EVENT& aEvent );
  52 
  53     /**

  54      * Indicates the cursor is over an edit point.  Used to coordinate cursor shapes with

  55      * other tools.

  56      */
  57     bool HasPoint() { return m_editedPoint != nullptr; }
  58 
  59     ///> Sets up handlers for various events.

  60     void setTransitions() override;
  61 
  62 private:
  63     ///> Updates item's points with edit points.

  64     void updateItem() const;
  65 
  66     ///> Updates edit points with item's points.

  67     void updatePoints();
  68 
  69     ///> Updates which point is being edited.

  70     void updateEditedPoint( const TOOL_EVENT& aEvent );
  71 
  72     ///> Sets the current point being edited. NULL means none.

  73     void setEditedPoint( EDIT_POINT* aPoint );
  74 
  75     ///> Returns true if aPoint is the currently modified point.

  76     inline bool isModified( const EDIT_POINT& aPoint ) const
  77     {
  78         return m_editedPoint == &aPoint;
  79     }
  80 
  81     inline int getEditedPointIndex() const
  82     {
  83         for( unsigned i = 0; i < m_editPoints->PointsSize(); ++i )
  84         {
  85             if( m_editedPoint == &m_editPoints->Point( i ) )
  86                 return i;
  87         }
  88 
  89         return wxNOT_FOUND;
  90     }
  91 
  92     bool addCornerCondition( const SELECTION& aSelection );
  93     bool removeCornerCondition( const SELECTION& aSelection );
  94 
  95     /// TOOL_ACTION handlers

  96     int addCorner( const TOOL_EVENT& aEvent );
  97     int removeCorner( const TOOL_EVENT& aEvent );
  98     int modifiedSelection( const TOOL_EVENT& aEvent );
  99 
 100     void saveItemsToUndo();
 101     void rollbackFromUndo();
 102 
 103 private:
 104     ///> Currently edited point, NULL if there is none.

 105     EDIT_POINT* m_editedPoint;
 106 
 107     ///> Currently available edit points.

 108     std::shared_ptr<EDIT_POINTS> m_editPoints;
 109 };
 110 
 111 #endif  // EE_POINT_EDITOR_H
