  46 bool SCH_EDIT_FRAME::CheckSheetForRecursion( SCH_SHEET* aSheet, SCH_SHEET_PATH* aHierarchy )
  47 {
  48     wxASSERT( aSheet && aHierarchy );
  49 
  50     wxString msg;
  51     SCH_SHEET_LIST hierarchy = Schematic().GetSheets();  // The full schematic sheet hierarchy.

  52     SCH_SHEET_LIST sheetHierarchy( aSheet );  // This is the hierarchy of the loaded file.

  53 
  54     wxFileName destFile = aHierarchy->LastScreen()->GetFileName();
  55 
  56     // SCH_SCREEN object file paths are expected to be absolute.  If this assert fires,

  57     // something is seriously broken.

  58     wxASSERT( destFile.IsAbsolute() );
  59 
  60     if( hierarchy.TestForRecursion( sheetHierarchy, destFile.GetFullPath() ) )
  61     {
  62         msg.Printf( _( "The sheet changes cannot be made because the destination sheet already "
  63                        "has the sheet \"%s\" or one of it's subsheets as a parent somewhere in "
  64                        "the schematic hierarchy." ),
  65                     destFile.GetFullPath() );
  66         DisplayError( this, msg );
  67         return true;
  68     }
  69 
  70     return false;
  71 }

 103 bool SCH_EDIT_FRAME::LoadSheetFromFile( SCH_SHEET* aSheet, SCH_SHEET_PATH* aHierarchy,
 104                                         const wxString& aFileName )
 105 {
 106     wxASSERT( aSheet && aHierarchy );
 107 
 108     wxString    msg;
 109     wxString    topLevelSheetPath;
 110     wxFileName  tmp;
 111     wxFileName  currentSheetFileName;
 112     bool        libTableChanged = false;
 113     SCH_SCREEN* currentScreen = aHierarchy->LastScreen();
 114     SCH_IO_MGR::SCH_FILE_T schFileType = SCH_IO_MGR::GuessPluginTypeFromSchPath( aFileName );
 115     SCH_PLUGIN::SCH_PLUGIN_RELEASER pi( SCH_IO_MGR::FindPlugin( schFileType ) );
 116     std::unique_ptr< SCH_SHEET> newSheet( new SCH_SHEET( &Schematic() ) );
 117 
 118     // This will cause the sheet UUID to be set to the loaded schematic UUID.  This is required

 119     // to ensure all of the sheet paths in any subsheets are correctly generated.

 120     const_cast<KIID&>( newSheet->m_Uuid ) = KIID( 0 );
 121 
 122     wxFileName fileName( aFileName );
 123 
 124     if( !fileName.IsAbsolute() && !fileName.MakeAbsolute() )
 125     {
 126         wxFAIL_MSG( wxString::Format( "Cannot make file name \"%s\" path absolute.", aFileName ) );
 127         return false;
 128     }
 129 
 130     wxString fullFilename = fileName.GetFullPath();
 131 
 132     try
 133     {
 134         if( aSheet->GetScreen() != nullptr )
 135         {
 136             newSheet.reset( pi->Load( fullFilename, &Schematic() ) );
 137         }
 138         else
 139         {
 140             newSheet->SetFileName( fullFilename );
 141             pi->Load( fullFilename, &Schematic(), newSheet.get() );
 142         }
 143 
 144         if( !pi->GetError().IsEmpty() )
 145         {
 146             msg = _( "The entire schematic could not be loaded.  Errors occurred attempting "
 147                      "to load hierarchical sheet schematics." );
 148 
 149             wxMessageDialog msgDlg1( this, msg, _( "Schematic Load Error" ),
 150                                      wxOK | wxCANCEL | wxCANCEL_DEFAULT |
 151                                      wxCENTER | wxICON_QUESTION );
 152             msgDlg1.SetOKLabel( wxMessageDialog::ButtonLabel( _( "Use partial schematic" ) ) );
 153             msgDlg1.SetExtendedMessage( pi->GetError() );
 154 
 155             if( msgDlg1.ShowModal() == wxID_CANCEL )
 156                 return false;
 157         }
 158     }
 159     catch( const IO_ERROR& ioe )
 160     {
 161         msg.Printf( _( "Error occurred loading schematic file \"%s\"." ), fullFilename );
 162         DisplayErrorMessage( this, msg, ioe.What() );
 163 
 164         msg.Printf( _( "Failed to load schematic \"%s\"" ), fullFilename );
 165         AppendMsgPanel( wxEmptyString, msg, CYAN );
 166 
 167         return false;
 168     }
 169 
 170     tmp = fileName;
 171 
 172     // If the loaded schematic is in a different folder from the current project and

 173     // it contains hierarchical sheets, the hierarchical sheet paths need to be updated.

 174     if( fileName.GetPath( wxPATH_GET_SEPARATOR ) != Prj().GetProjectPath()
 175       && newSheet->CountSheets() )
 176     {
 177         // Give the user the option to choose relative path if possible.

 178         if( tmp.MakeRelativeTo( Prj().GetProjectPath() ) )
 179         {
 180             wxMessageDialog msgDlg2(
 181                     this,
 182                     "Do you want to use a relative path to the loaded "
 183                     "schematic?", "Select Path Type",
 184                     wxYES_NO | wxCANCEL | wxYES_DEFAULT | wxICON_QUESTION | wxCENTER );
 185             msgDlg2.SetYesNoLabels( wxMessageDialog::ButtonLabel( "Use Relative Path" ),
 186                                     wxMessageDialog::ButtonLabel( "Use Absolute Path" ) );
 187             int rsp = msgDlg2.ShowModal();
 188 
 189             if( rsp == wxID_CANCEL )
 190             {
 191                 return false;
 192             }
 193             else if( rsp == wxID_NO )
 194             {
 195                 topLevelSheetPath = fileName.GetPathWithSep();
 196             }
 197             else
 198             {
 199                 topLevelSheetPath = tmp.GetPathWithSep();
 200             }
 201         }
 202         else
 203         {
 204             topLevelSheetPath = tmp.GetPathWithSep();
 205         }
 206 
 207         if( wxFileName::GetPathSeparator() == '\\' )
 208             topLevelSheetPath.Replace( "\\", "/" );
 209     }
 210 
 211     // Make sure any new sheet changes do not cause any recursion issues.

 212     SCH_SHEET_LIST hierarchy = Schematic().GetSheets(); // This is the schematic sheet hierarchy.

 213     SCH_SHEET_LIST sheetHierarchy( newSheet.get() );    // This is the hierarchy of the loaded file.

 214 
 215     if( CheckSheetForRecursion( newSheet.get(), aHierarchy )
 216       || checkForNoFullyDefinedLibIds( newSheet.get() ) )
 217         return false;
 218 
 219     // Make a valiant attempt to warn the user of all possible scenarios where there could

 220     // be broken symbol library links.

 221     wxArrayString    names;
 222     wxArrayString    newLibNames;
 223     SCH_SCREENS      newScreens( newSheet.get() );   // All screens associated with the import.

 224     SCH_SCREENS      prjScreens( &Schematic().Root() );
 225 
 226     newScreens.GetLibNicknames( names );
 227 
 228     wxMessageDialog::ButtonLabel okButtonLabel( _( "Continue Load" ) );
 229     wxMessageDialog::ButtonLabel cancelButtonLabel( _( "Cancel Load" ) );
 230 
 231     if( fileName.GetPath( wxPATH_GET_SEPARATOR ) == Prj().GetProjectPath()
 232       && !prjScreens.HasSchematic( fullFilename ) )
 233     {
 234         // A schematic in the current project path that isn't part of the current project.

 235         // It's possible the user copied this schematic from another project so the library

 236         // links may not be avaible.  Even this is check is no guarantee that all symbol

 237         // library links are valid but it's better than nothing.

 238         for( const auto& name : names )
 239         {
 240             if( !Prj().SchSymbolLibTable()->HasLibrary( name ) )
 241                 newLibNames.Add( name );
 242         }
 243 
 244         if( !newLibNames.IsEmpty() )
 245         {
 246             msg = _( "There are library names in the loaded schematic that are missing "
 247                      "from the project library table.  This may result in broken symbol "
 248                      "library links for the loaded schematic.  Do you wish to continue?" );
 249             wxMessageDialog msgDlg3( this, msg, _( "Continue Load Schematic" ),
 250                                      wxOK | wxCANCEL | wxCANCEL_DEFAULT |
 251                                      wxCENTER | wxICON_QUESTION );
 252             msgDlg3.SetOKCancelLabels( okButtonLabel, cancelButtonLabel );
 253 
 254             if( msgDlg3.ShowModal() == wxID_CANCEL )
 255                 return false;
 256         }
 257     }
 258     else if( fileName.GetPath( wxPATH_GET_SEPARATOR ) != Prj().GetProjectPath() )
 259     {
 260         // A schematic loaded from a path other than the current project path.

 261 
 262         // If there are symbol libraries in the imported schematic that are not in the

 263         // symbol library table of this project, there could be a lot of broken symbol

 264         // library links.  Attempt to add the missing libraries to the project symbol

 265         // library table.

 266         wxArrayString    duplicateLibNames;
 267 
 268         for( const auto& name : names )
 269         {
 270             if( !Prj().SchSymbolLibTable()->HasLibrary( name ) )
 271                 newLibNames.Add( name );
 272             else
 273                 duplicateLibNames.Add( name );
 274         }
 275 
 276         SYMBOL_LIB_TABLE table;
 277         wxFileName symLibTableFn( fileName.GetPath(),
 278                                   SYMBOL_LIB_TABLE::GetSymbolLibTableFileName() );
 279 
 280         // If there are any new or duplicate libraries, check to see if it's possible that

 281         // there could be any missing libraries that would cause broken symbol library links.

 282         if( !newLibNames.IsEmpty() || !duplicateLibNames.IsEmpty() )
 283         {
 284             if( !symLibTableFn.Exists() || !symLibTableFn.IsFileReadable() )
 285             {
 286                 msg.Printf( _( "The project library table \"%s\" does not exist or cannot "
 287                                "be read.  This may result in broken symbol links for the "
 288                                "schematic.  Do you wish to continue?" ),
 289                             fileName.GetFullPath() );
 290                 wxMessageDialog msgDlg4( this, msg, _( "Continue Load Schematic" ),
 291                                          wxOK | wxCANCEL | wxCANCEL_DEFAULT |
 292                                          wxCENTER | wxICON_QUESTION );
 293                 msgDlg4.SetOKCancelLabels( okButtonLabel, cancelButtonLabel );
 294 
 295                 if( msgDlg4.ShowModal() == wxID_CANCEL )
 296                     return false;
 297             }
 298             else
 299             {
 300                 try
 301                 {
 302                     table.Load( symLibTableFn.GetFullPath() );
 303                 }
 304                 catch( const IO_ERROR& ioe )
 305                 {
 306                     msg.Printf( _( "An error occurred loading the symbol library table "
 307                                    "\"%s\"." ),
 308                                 symLibTableFn.GetFullPath() );
 309                     DisplayErrorMessage( NULL, msg, ioe.What() );
 310                     return false;
 311                 }
 312             }
 313         }
 314 
 315         // Check to see if any of the symbol libraries found in the appended schematic do

 316         // not exist in the current project are missing from the appended project symbol

 317         // library table.

 318         if( !newLibNames.IsEmpty() )
 319         {
 320             bool missingLibNames = table.IsEmpty();
 321 
 322             if( !missingLibNames )
 323             {
 324                 for( const auto& newLibName : newLibNames )
 325                 {
 326                     if( !table.HasLibrary( newLibName ) )
 327                     {
 328                         missingLibNames = true;
 329                         break;
 330                     }
 331                 }
 332             }
 333 
 334             if( missingLibNames )
 335             {
 336                 msg = _( "There are library names in the loaded schematic that are missing "
 337                          "from the loaded schematic project library table.  This may result "
 338                          "in broken symbol library links for the schematic.  "
 339                          "Do you wish to continue?" );
 340                 wxMessageDialog msgDlg5( this, msg, _( "Continue Load Schematic" ),
 341                                          wxOK | wxCANCEL | wxCANCEL_DEFAULT |
 342                                          wxCENTER | wxICON_QUESTION );
 343                 msgDlg5.SetOKCancelLabels( okButtonLabel, cancelButtonLabel );
 344 
 345                 if( msgDlg5.ShowModal() == wxID_CANCEL )
 346                     return false;
 347             }
 348         }
 349 
 350         // The library name already exists in the current project.  Check to see if the

 351         // duplicate name is the same library in the current project.  If it's not, it's

 352         // most likely that the symbol library links will be broken.

 353         if( !duplicateLibNames.IsEmpty() && !table.IsEmpty() )
 354         {
 355             bool libNameConflict = false;
 356 
 357             for( const auto& duplicateLibName : duplicateLibNames )
 358             {
 359                 const SYMBOL_LIB_TABLE_ROW* thisRow = nullptr;
 360                 const SYMBOL_LIB_TABLE_ROW* otherRow = nullptr;
 361 
 362                 if( Prj().SchSymbolLibTable()->HasLibrary( duplicateLibName ) )
 363                     thisRow = Prj().SchSymbolLibTable()->FindRow( duplicateLibName );
 364 
 365                 if( table.HasLibrary( duplicateLibName ) )
 366                     otherRow = table.FindRow( duplicateLibName );
 367 
 368                 // It's in the global library table so there is no conflict.

 369                 if( thisRow && !otherRow )
 370                     continue;
 371 
 372                 if( !thisRow || !otherRow )
 373                     continue;
 374 
 375                 wxFileName otherUriFileName;
 376                 wxString thisURI = thisRow->GetFullURI( true );
 377                 wxString otherURI = otherRow->GetFullURI( false);
 378 
 379                 if( otherURI.Contains( "${KIPRJMOD}" ) || otherURI.Contains( "$(KIPRJMOD)" ) )
 380                 {
 381                     // Cannot use relative paths here, "${KIPRJMOD}../path-to-cache-lib" does

 382                     // not expand to a valid symbol library path.

 383                     otherUriFileName.SetPath( fileName.GetPath() );
 384                     otherUriFileName.SetFullName( otherURI.AfterLast( '}' ) );
 385                     otherURI = otherUriFileName.GetFullPath();
 386                 }
 387 
 388                 if( thisURI != otherURI )
 389                 {
 390                     libNameConflict = true;
 391                     break;
 392                 }
 393             }
 394 
 395             if( libNameConflict )
 396             {
 397                 msg = _( "A duplicate library name that references a different library exists "
 398                          "in the current library table.  This conflict cannot be resolved and "
 399                          "may result in broken symbol library links for the schematic.  "
 400                          "Do you wish to continue?" );
 401                 wxMessageDialog msgDlg6( this, msg, _( "Continue Load Schematic" ),
 402                                          wxOK | wxCANCEL | wxCANCEL_DEFAULT |
 403                                          wxCENTER | wxICON_QUESTION );
 404                 msgDlg6.SetOKCancelLabels( okButtonLabel, cancelButtonLabel );
 405 
 406                 if( msgDlg6.ShowModal() == wxID_CANCEL )
 407                     return false;
 408             }
 409         }
 410 
 411         // All (most?) of the possible broken symbol library link cases are covered.  Map the

 412         // new appended schematic project symbol library table entries to the current project

 413         // symbol library table.

 414         if( !newLibNames.IsEmpty() && !table.IsEmpty() )
 415         {
 416             for( const auto& libName : newLibNames )
 417             {
 418                 if( !table.HasLibrary( libName )
 419                   || Prj().SchSymbolLibTable()->HasLibrary( libName ) )
 420                     continue;
 421 
 422                 // Don't expand environment variable because KIPRJMOD will not be correct

 423                 // for a different project.

 424                 wxString uri = table.GetFullURI( libName, false );
 425                 wxFileName newLib;
 426 
 427                 if( uri.Contains( "${KIPRJMOD}" ) || uri.Contains( "$(KIPRJMOD)" ) )
 428                 {
 429                     // Cannot use relative paths here, "${KIPRJMOD}../path-to-cache-lib" does

 430                     // not expand to a valid symbol library path.

 431                     newLib.SetPath( fileName.GetPath() );
 432                     newLib.SetFullName( uri.AfterLast( '}' ) );
 433                     uri = newLib.GetFullPath();
 434                 }
 435                 else
 436                 {
 437                     uri = table.GetFullURI( libName );
 438                 }
 439 
 440                 // Add the library from the imported project to the current project

 441                 // symbol library table.

 442                 const SYMBOL_LIB_TABLE_ROW* row = table.FindRow( libName );
 443 
 444                 auto newRow = new SYMBOL_LIB_TABLE_ROW( libName, uri, row->GetType(),
 445                                                         row->GetOptions(), row->GetDescr() );
 446 
 447                 Prj().SchSymbolLibTable()->InsertRow( newRow );
 448                 libTableChanged = true;
 449             }
 450         }
 451     }
 452 
 453     SCH_SCREEN* newScreen = newSheet->GetScreen();
 454     wxCHECK_MSG( newScreen, false, "No screen defined for sheet." );
 455 
 456     // Set all sheets loaded into the correct sheet file paths.

 457 
 458     for( auto aItem : currentScreen->Items().OfType( SCH_SHEET_T ) )
 459     {
 460         auto sheet = static_cast<SCH_SHEET*>( aItem );
 461         sheet->SetFileName( topLevelSheetPath + sheet->GetFileName() );
 462     }
 463 
 464     if( libTableChanged )
 465         Prj().SchSymbolLibTable()->Save( Prj().GetProjectPath() +
 466                                          SYMBOL_LIB_TABLE::GetSymbolLibTableFileName() );
 467 
 468     // It is finally safe to add or append the imported schematic.

 469     if( aSheet->GetScreen() == nullptr )
 470         aSheet->SetScreen( newScreen );
 471     else
 472         aSheet->GetScreen()->Append( newScreen );
 473 
 474     SCH_SCREENS allScreens( Schematic().Root() );
 475     allScreens.ReplaceDuplicateTimeStamps();
 476 
 477     return true;
 478 }

 500 SCH_SHEET_PIN* SCH_EDIT_FRAME::CreateSheetPin( SCH_SHEET* aSheet, SCH_HIERLABEL* aLabel )
 501 {
 502     SCHEMATIC_SETTINGS& settings = aSheet->Schematic()->Settings();
 503     wxString            text;
 504     SCH_SHEET_PIN*      sheetPin;
 505 
 506     if( aLabel )
 507     {
 508         text = aLabel->GetText();
 509         m_lastSheetPinType = aLabel->GetShape();
 510     }
 511 
 512     sheetPin = new SCH_SHEET_PIN( aSheet, wxPoint( 0, 0 ), text );
 513     sheetPin->SetFlags( IS_NEW );
 514     sheetPin->SetTextSize( wxSize( settings.m_DefaultTextSize, settings.m_DefaultTextSize ) );
 515     sheetPin->SetShape( m_lastSheetPinType );
 516 
 517     if( !aLabel )
 518     {
 519         DIALOG_EDIT_SHEET_PIN dlg( this, sheetPin );
 520 
 521         if( dlg.ShowModal() != wxID_OK || sheetPin->GetText().IsEmpty()  )
 522         {
 523             delete sheetPin;
 524             return nullptr;
 525         }
 526     }
 527 
 528     m_lastSheetPinType = sheetPin->GetShape();
 529 
 530     sheetPin->SetPosition( (wxPoint) GetCanvas()->GetViewControls()->GetCursorPosition() );
 531 
 532     return sheetPin;
 533 }

 536 SCH_HIERLABEL* SCH_EDIT_FRAME::ImportHierLabel( SCH_SHEET* aSheet )
 537 {
 538     if( !aSheet->GetScreen() )
 539         return nullptr;
 540 
 541     for( auto item : aSheet->GetScreen()->Items().OfType( SCH_HIER_LABEL_T ) )

 550     return nullptr;

 554 void SCH_EDIT_FRAME::DrawCurrentSheetToClipboard()
 555 {
 556     wxRect  DrawArea;
 557     BASE_SCREEN* screen = GetScreen();
 558 
 559     DrawArea.SetSize( GetPageSizeIU() );
 560 
 561     // Calculate a reasonable dc size, in pixels, and the dc scale to fit

 562     // the drawings into the dc size

 563     // scale is the ratio resolution (in PPI) / internal units

 564     double ppi = 300;   // Use 300 pixels per inch to create bitmap images on start

 565     double inch2Iu = 1000.0 * IU_PER_MILS;
 566     double scale = ppi / inch2Iu;
 567 
 568     wxSize dcsize = DrawArea.GetSize();
 569 
 570     int maxdim = std::max( dcsize.x, dcsize.y );
 571 
 572     // the max size in pixels of the bitmap used to build the sheet copy

 573     const int maxbitmapsize = 5600;
 574 
 575     while( int( maxdim * scale ) > maxbitmapsize )
 576     {
 577         ppi = ppi / 1.5;
 578         scale = ppi / inch2Iu;
 579     }
 580 
 581     dcsize.x *= scale;
 582     dcsize.y *= scale;
 583 
 584     // Set draw offset, zoom... to values needed to draw in the memory DC

 585     // after saving initial values:

 586     wxPoint tmp_startvisu = screen->m_StartVisu;
 587     wxPoint old_org       = screen->m_DrawOrg;
 588     screen->m_DrawOrg.x   = screen->m_DrawOrg.y = 0;
 589     screen->m_StartVisu.x = screen->m_StartVisu.y = 0;
 590 
 591     wxMemoryDC dc;
 592     wxBitmap image( dcsize );
 593     dc.SelectObject( image );
 594     dc.Clear();
 595 
 596     GRResetPenAndBrush( &dc );
 597     GRForceBlackPen( false );
 598     dc.SetUserScale( scale, scale );
 599 
 600     GetRenderSettings()->SetPrintDC( &dc );
 601 
 602     PrintPage( GetRenderSettings() );
 603 
 604     if( wxTheClipboard->Open() )
 605     {
 606         // This data objects are held by the clipboard, so do not delete them in the app.

 607         wxBitmapDataObject* clipbrd_data = new wxBitmapDataObject( image );
 608         wxTheClipboard->SetData( clipbrd_data );
 609         wxTheClipboard->Close();
 610     }
 611 
 612     // Deselect Bitmap from DC in order to delete the MemoryDC

 613     dc.SelectObject( wxNullBitmap );
 614 
 615     GRForceBlackPen( false );
 616 
 617     screen->m_StartVisu = tmp_startvisu;
 618     screen->m_DrawOrg   = old_org;
 619 }

 622 bool SCH_EDIT_FRAME::AllowCaseSensitiveFileNameClashes( const wxString& aSchematicFileName )
 623 {
 624     wxString msg;
 625     SCH_SCREENS screens( Schematic().Root() );
 626     wxFileName fn = aSchematicFileName;
 627 
 628     wxCHECK( fn.IsAbsolute(), false );
 629 
 630     if( eeconfig()->m_Appearance.show_sheet_filename_case_sensitivity_dialog
 631       && screens.CanCauseCaseSensitivityIssue( aSchematicFileName ) )
 632     {
 633         msg.Printf( _( "The file name \"%s\" can cause issues with an existing file name\n"
 634                        "already defined in the schematic on systems that support case\n"
 635                        "insensitive file names.  This will cause issues if you copy this\n"
 636                        "project to an operating system that supports case insensitive file\n"
 637                        "names.\n\nDo you wish to continue?" ),
 638                     fn.GetName() );
 639
