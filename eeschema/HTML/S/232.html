  31 /**

  32  * The maximum number of items in the clarify selection context menu.  While in

  33  * most cases it is highly unlikely that there would ever be more than 10 items

  34  * at the current cursor, there are some exceptions  (a bunch of pins created at

  35  * once, for instance).  The current setting of 200 is arbitrary.

  36  */
  37 #define MAX_SELECT_ITEM_IDS 200
  38 
  39 /**

  40  * The maximum number of units per package.

  41  * While counts approaching 100 start to make the unit-selection popup menu

  42  * difficult to use, the limit is currently 'ZZ' (26 * 26).

  43  */
  44 #define MAX_UNIT_COUNT_PER_PACKAGE 676
