 103 void SYMBOL_LIB_TABLE::Parse( LIB_TABLE_LEXER* in )
 104 {
 105     T        tok;
 106     wxString errMsg;    // to collect error messages

 107 
 108     // This table may be nested within a larger s-expression, or not.

 109     // Allow for parser of that optional containing s-epression to have looked ahead.

 110     if( in->CurTok() != T_sym_lib_table )
 111     {
 112         in->NeedLEFT();
 113         if( ( tok = in->NextTok() ) != T_sym_lib_table )
 114             in->Expecting( T_sym_lib_table );
 115     }
 116 
 117     while( ( tok = in->NextTok() ) != T_RIGHT )
 118     {
 119         std::unique_ptr< SYMBOL_LIB_TABLE_ROW > row( new SYMBOL_LIB_TABLE_ROW );
 120 
 121         if( tok == T_EOF )
 122             in->Expecting( T_RIGHT );
 123 
 124         if( tok != T_LEFT )
 125             in->Expecting( T_LEFT );
 126 
 127         // in case there is a "row integrity" error, tell where later.

 128         int lineNum = in->CurLineNumber();
 129 
 130         if( ( tok = in->NextTok() ) != T_lib )
 131             in->Expecting( T_lib );
 132 
 133         // (name NICKNAME)

 134         in->NeedLEFT();
 135 
 136         if( ( tok = in->NextTok() ) != T_name )
 137             in->Expecting( T_name );
 138 
 139         in->NeedSYMBOLorNUMBER();
 140 
 141         row->SetNickName( in->FromUTF8() );
 142 
 143         in->NeedRIGHT();
 144 
 145         // After (name), remaining (lib) elements are order independent, and in

 146         // some cases optional.

 147         bool    sawType     = false;
 148         bool    sawOpts     = false;
 149         bool    sawDesc     = false;
 150         bool    sawUri      = false;
 151         bool    sawDisabled = false;
 152 
 153         while( ( tok = in->NextTok() ) != T_RIGHT )
 154         {
 155             if( tok == T_EOF )
 156                 in->Unexpected( T_EOF );
 157 
 158             if( tok != T_LEFT )
 159                 in->Expecting( T_LEFT );
 160 
 161             tok = in->NeedSYMBOLorNUMBER();
 162 
 163             switch( tok )
 164             {
 165             case T_uri:
 166                 if( sawUri )
 167                     in->Duplicate( tok );
 168                 sawUri = true;
 169                 in->NeedSYMBOLorNUMBER();
 170                 row->SetFullURI( in->FromUTF8() );
 171                 break;
 172 
 173             case T_type:
 174                 if( sawType )
 175                     in->Duplicate( tok );
 176                 sawType = true;
 177                 in->NeedSYMBOLorNUMBER();
 178                 row->SetType( in->FromUTF8() );
 179                 break;
 180 
 181             case T_options:
 182                 if( sawOpts )
 183                     in->Duplicate( tok );
 184                 sawOpts = true;
 185                 in->NeedSYMBOLorNUMBER();
 186                 row->SetOptions( in->FromUTF8() );
 187                 break;
 188 
 189             case T_descr:
 190                 if( sawDesc )
 191                     in->Duplicate( tok );
 192                 sawDesc = true;
 193                 in->NeedSYMBOLorNUMBER();
 194                 row->SetDescr( in->FromUTF8() );
 195                 break;
 196 
 197             case T_disabled:
 198                 if( sawDisabled )
 199                     in->Duplicate( tok );
 200                 sawDisabled = true;
 201                 row->SetEnabled( false );
 202                 break;
 203 
 204             default:
 205                 in->Unexpected( tok );
 206             }
 207 
 208             in->NeedRIGHT();
 209         }
 210 
 211         if( !sawType )
 212             in->Expecting( T_type );
 213 
 214         if( !sawUri )
 215             in->Expecting( T_uri );
 216 
 217         // all nickNames within this table fragment must be unique, so we do not

 218         // use doReplace in InsertRow().  (However a fallBack table can have a

 219         // conflicting nickName and ours will supercede that one since in

 220         // FindLib() we search this table before any fall back.)

 221         wxString nickname = row->GetNickName(); // store it to be able to used it

 222                                                 // after row deletion if an error occurs

 223         LIB_TABLE_ROW* tmp = row.release();
 224 
 225         if( !InsertRow( tmp ) )
 226         {
 227             delete tmp;     // The table did not take ownership of the row.

 228 
 229             wxString msg = wxString::Format(
 230                                 _( "Duplicate library nickname \"%s\" found in symbol library "
 231                                    "table file line %d" ), GetChars( nickname ), lineNum );
 232 
 233             if( !errMsg.IsEmpty() )
 234                 errMsg << '\n';
 235 
 236             errMsg << msg;
 237         }
 238     }
 239 
 240     if( !errMsg.IsEmpty() )
 241         THROW_IO_ERROR( errMsg );
 242 }

 258 int SYMBOL_LIB_TABLE::GetModifyHash()
 259 {
 260     int                     hash = 0;
 261     std::vector< wxString > libNames = GetLogicalLibs();
 262 
 263     for( const auto& libName : libNames )
 264     {
 265         const SYMBOL_LIB_TABLE_ROW* row = FindRow( libName );
 266 
 267         if( !row || !row->plugin )
 268         {
 269             wxFAIL;
 270             continue;
 271         }
 272 
 273         hash += row->plugin->GetModifyHash();
 274     }
 275 
 276     hash += m_modifyHash;
 277 
 278     return hash;
 279 }

 300 SYMBOL_LIB_TABLE_ROW* SYMBOL_LIB_TABLE::FindRow( const wxString& aNickname )
 301 
 302 {
 303     SYMBOL_LIB_TABLE_ROW* row = dynamic_cast< SYMBOL_LIB_TABLE_ROW* >( findRow( aNickname ) );
 304 
 305     if( !row )
 306         return nullptr;
 307 
 308     // We've been 'lazy' up until now, but it cannot be deferred any longer,

 309     // instantiate a PLUGIN of the proper kind if it is not already in this

 310     // SYMBOL_LIB_TABLE_ROW.

 311     if( !row->plugin )
 312         row->setPlugin( SCH_IO_MGR::FindPlugin( row->type ) );
 313 
 314     return row;
 315 }

 318 void SYMBOL_LIB_TABLE::LoadSymbolLib( std::vector<LIB_PART*>& aSymbolList,
 319                                       const wxString& aNickname, bool aPowerSymbolsOnly )
 320 {
 321     SYMBOL_LIB_TABLE_ROW* row = FindRow( aNickname );
 322     wxCHECK( row && row->plugin, /* void */  );
 323 
 324     wxString options = row->GetOptions();
 325 
 326     if( aPowerSymbolsOnly )
 327         row->SetOptions( row->GetOptions() + " " + PropPowerSymsOnly );
 328 
 329     row->plugin->EnumerateSymbolLib( aSymbolList, row->GetFullURI( true ), row->GetProperties() );
 330 
 331     if( aPowerSymbolsOnly )
 332         row->SetOptions( options );
 333 
 334     // The library cannot know its own name, because it might have been renamed or moved.

 335     // Therefore footprints cannot know their own library nickname when residing in

 336     // a symbol library.

 337     // Only at this API layer can we tell the symbol about its actual library nickname.

 338     for( LIB_PART* part : aSymbolList )
 339     {
 340         LIB_ID id = part->GetLibId();
 341 
 342         id.SetLibNickname( row->GetNickName() );
 343         part->SetLibId( id );
 344     }
 345 }

 348 LIB_PART* SYMBOL_LIB_TABLE::LoadSymbol( const wxString& aNickname, const wxString& aSymbolName )
 349 {
 350     SYMBOL_LIB_TABLE_ROW* row = FindRow( aNickname );
 351 
 352     if( !row || !row->plugin )
 353         return nullptr;
 354 
 355     LIB_PART* part = row->plugin->LoadSymbol( row->GetFullURI( true ), aSymbolName,
 356                                               row->GetProperties() );
 357 
 358     if( part == nullptr )
 359         return part;
 360 
 361     // The library cannot know its own name, because it might have been renamed or moved.

 362     // Therefore footprints cannot know their own library nickname when residing in

 363     // a symbol library.

 364     // Only at this API layer can we tell the symbol about its actual library nickname.

 365     if( part )
 366     {
 367         LIB_ID id = part->GetLibId();
 368 
 369         id.SetLibNickname( row->GetNickName() );
 370         part->SetLibId( id );
 371     }
 372 
 373     return part;
 374 }

 377 SYMBOL_LIB_TABLE::SAVE_T SYMBOL_LIB_TABLE::SaveSymbol( const wxString& aNickname,
 378                                                        const LIB_PART* aSymbol, bool aOverwrite )
 379 {
 380     const SYMBOL_LIB_TABLE_ROW* row = FindRow( aNickname );
 381     wxCHECK( row && row->plugin, SAVE_SKIPPED );
 382 
 383     if( !aOverwrite )
 384     {
 385         // Try loading the footprint to see if it already exists, caller wants overwrite

 386         // protection, which is atypical, not the default.

 387 
 388         wxString name = aSymbol->GetLibId().GetLibItemName();
 389 
 390         std::unique_ptr< LIB_PART > symbol( row->plugin->LoadSymbol( row->GetFullURI( true ),
 391                                                                      name,
 392                                                                      row->GetProperties() ) );
 393 
 394         if( symbol.get() )
 395             return SAVE_SKIPPED;
 396     }
 397 
 398     row->plugin->SaveSymbol( row->GetFullURI( true ), aSymbol, row->GetProperties() );
 399 
 400     return SAVE_OK;
 401 }

 437 LIB_PART* SYMBOL_LIB_TABLE::LoadSymbolWithOptionalNickname( const LIB_ID& aLibId )
 438 {
 439     wxString   nickname = aLibId.GetLibNickname();
 440     wxString   name     = aLibId.GetLibItemName();
 441 
 442     if( nickname.size() )
 443     {
 444         return LoadSymbol( nickname, name );
 445     }
 446 
 447     // nickname is empty, sequentially search (alphabetically) all libs/nicks for first match:

 448     else
 449     {
 450         std::vector<wxString> nicks = GetLogicalLibs();
 451 
 452         // Search each library going through libraries alphabetically.

 453         for( unsigned i = 0;  i < nicks.size();  ++i )
 454         {
 455             // FootprintLoad() returns NULL on not found, does not throw exception

 456             // unless there's an IO_ERROR.

 457             LIB_PART* ret = LoadSymbol( nicks[i], name );
 458 
 459             if( ret )
 460                 return ret;
 461         }
 462 
 463         return nullptr;
 464     }
 465 }

 474 bool SYMBOL_LIB_TABLE::LoadGlobalTable( SYMBOL_LIB_TABLE& aTable )
 475 {
 476     bool        tableExists = true;
 477     wxFileName  fn = GetGlobalTableFileName();
 478 
 479     if( !fn.FileExists() )
 480     {
 481         tableExists = false;
 482 
 483         if( !fn.DirExists() && !fn.Mkdir( 0x777, wxPATH_MKDIR_FULL ) )
 484         {
 485             THROW_IO_ERROR( wxString::Format( _( "Cannot create global library table path \"%s\"." ),
 486                                               GetChars( fn.GetPath() ) ) );
 487         }
 488 
 489         // Attempt to copy the default global file table from the KiCad

 490         // template folder to the user's home configuration path.

 491         SEARCH_STACK ss;
 492 
 493         SystemDirsAppend( &ss );
 494 
 495         wxString templatePath =
 496             Pgm().GetLocalEnvVariables().at( wxT( "KICAD_TEMPLATE_DIR" ) ).GetValue();
 497 
 498         if( !templatePath.IsEmpty() )
 499             ss.AddPaths( templatePath, 0 );
 500 
 501         wxString fileName = ss.FindValidPath( global_tbl_name );
 502 
 503         // The fallback is to create an empty global symbol table for the user to populate.

 504         if( fileName.IsEmpty() || !::wxCopyFile( fileName, fn.GetFullPath(), false ) )
 505         {
 506             SYMBOL_LIB_TABLE    emptyTable;
 507 
 508             emptyTable.Save( fn.GetFullPath() );
 509         }
 510     }
 511 
 512     aTable.Load( fn.GetFullPath() );
 513 
 514     return tableExists;
 515 }
