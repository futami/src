  47 bool DIALOG_GLOBAL_SYM_LIB_TABLE_CONFIG::TransferDataFromWindow()
  48 {
  49     // Create an empty table if requested by the user.

  50     if( m_emptyRb->GetValue() )
  51     {
  52         SYMBOL_LIB_TABLE    emptyTable;
  53 
  54         try
  55         {
  56             emptyTable.Save( SYMBOL_LIB_TABLE::GetGlobalTableFileName() );
  57         }
  58         catch( const IO_ERROR& ioe )
  59         {
  60             DisplayError( this,
  61                           wxString::Format( _( "Error occurred writing empty symbol library table "
  62                                                "file.\n\n%s" ),
  63                                             SYMBOL_LIB_TABLE::GetGlobalTableFileName(),
  64                                             ioe.What() ) );
  65             return false;
  66         }
  67 
  68         return true;
  69     }
  70 
  71     wxString fileName = m_filePicker1->GetPath();
  72 
  73     if( fileName.IsEmpty() )
  74     {
  75         DisplayError( this, _( "Please select a symbol library table file." ) );
  76         return false;
  77     }
  78 
  79     wxFileName fn = fileName;
  80 
  81     // Make sure the symbol library table to copy actually exists.

  82     if( !fn.FileExists() )
  83     {
  84         DisplayError( this,
  85                       wxString::Format( _( "File \"%s\" not found." ), fn.GetFullPath() ) );
  86         return false;
  87     }
  88 
  89     // Make sure the symbol library table to copy is a valid symbol library table file.

  90     SYMBOL_LIB_TABLE tmpTable;
  91 
  92     try
  93     {
  94         tmpTable.Load( fn.GetFullPath() );
  95     }
  96     catch( const IO_ERROR& ioe )
  97     {
  98         DisplayError( this,
  99                       wxString::Format( _( "File \"%s\" is not a valid symbol library table "
 100                                            "file.\n\n%s" ), fn.GetFullPath(), ioe.What() ) );
 101         return false;
 102     }
 103 
 104     // Create the config path if it doesn't already exist.

 105     wxFileName symTableFileName = SYMBOL_LIB_TABLE::GetGlobalTableFileName();
 106 
 107     if( !symTableFileName.DirExists() && !symTableFileName.Mkdir( 0x777, wxPATH_MKDIR_FULL ) )
 108     {
 109         DisplayError( this,
 110                       wxString::Format( _( "Cannot create global library table path \"%s\"." ),
 111                                             symTableFileName.GetPath() ) );
 112         return false;
 113     }
 114 
 115     // Copy the global symbol library table file to the user config.

 116     if( !::wxCopyFile( fn.GetFullPath(), symTableFileName.GetFullPath() ) )
 117     {
 118         DisplayError( this,
 119                       wxString::Format( _( "Cannot copy global symbol library table "
 120                                            "file:\n\n \"%s\"\n\n:to:\n\n\"%s\"." ),
 121                                         fn.GetFullPath(), symTableFileName.GetFullPath() ) );
 122         return false;
 123     }
 124 
 125     // Load the successfully copied symbol library table file.  This should not fail

 126     // since the file was tested above.  Check for failure anyway to keep the compiler

 127     // from complaining.

 128     try
 129     {
 130         if( !SYMBOL_LIB_TABLE::LoadGlobalTable( SYMBOL_LIB_TABLE::GetGlobalLibTable() ) )
 131             return false;
 132     }
 133     catch( const IO_ERROR& ioe )
 134     {
 135         DisplayError( this,
 136                       wxString::Format( _( "Error occurred loading global symbol library table:"
 137                                            "\n\n%s" ), ioe.What() ) );
 138         return false;
 139     }
 140 
 141     return true;
 142 }
