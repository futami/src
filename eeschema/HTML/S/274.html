  39 /**

  40  * Field object used in symbol libraries.  At least MANDATORY_FIELDS are always present

  41  * in a ram resident library symbol.  All constructors must ensure this because

  42  * the component property editor assumes it.

  43  * <p>

  44  * A field is a string linked to a component.  Unlike purely graphical text, fields can

  45  * be used in netlist generation and other tools (BOM).

  46  *

  47  *  The first 4 fields have a special meaning:

  48  *

  49  *  0 = REFERENCE

  50  *  1 = VALUE

  51  *  2 = FOOTPRINT (default Footprint)

  52  *  3 = DATASHEET (user doc link)

  53  *

  54  *  others = free fields

  55  * </p>

  56  *

  57  * @see enum NumFieldType

  58  */
  59 class LIB_FIELD : public LIB_ITEM, public EDA_TEXT
  60 {
  61     int      m_id;           ///< @see enum NumFieldType

  62     wxString m_name;         ///< Name (not the field text value itself, that is .m_Text)

  63 
  64     /**

  65      * Print the field.

  66      * <p>

  67      * If \a aData not NULL, \a aData must point a wxString which is used instead of

  68      * the m_Text

  69      * </p>

  70      */
  71     void print( RENDER_SETTINGS* aSettings, const wxPoint& aOffset, void* aData,
  72                 const TRANSFORM& aTransform ) override;
  73 
  74     /**

  75      * Calculate the new circle at \a aPosition when editing.

  76      *

  77      * @param aPosition - The position to edit the circle in drawing coordinates.

  78      */
  79     void CalcEdit( const wxPoint& aPosition ) override;
  80 
  81     friend class SCH_LEGACY_PLUGIN_CACHE;   // Required to access m_name.

  82 
  83 public:
  84 
  85     LIB_FIELD( int idfield = 2 );
  86 
  87     LIB_FIELD( int aID, const wxString& aName );
  88 
  89     LIB_FIELD( LIB_PART * aParent, int idfield = 2 );
  90 
  91     // Do not create a copy constructor.  The one generated by the compiler is adequate.

  92 
  93     ~LIB_FIELD();
  94 
  95     wxString GetClass() const override
  96     {
  97         return wxT( "LIB_FIELD" );
  98     }
  99 
 100     wxString GetTypeName() override
 101     {
 102         return _( "Field" );
 103     }
 104 
 105     /**

 106      * Object constructor initialization helper.

 107      */
 108     void Init( int idfield );
 109 
 110     /**

 111      * Returns the field name.

 112      *

 113      * The first four field IDs are reserved and therefore always return their respective

 114      * names.

 115      *

 116      * The user definable fields will return FieldN where N is the ID of the field when the

 117      * m_name member is empyt unless false is passed to \a aUseDefaultName.

 118      */
 119     wxString GetName( bool aUseDefaultName = true ) const;
 120 
 121     /**

 122      * Get a non-language-specific name for a field which can be used for storage, variable

 123      * look-up, etc.

 124      */
 125     wxString GetCanonicalName() const;
 126 
 127     /**

 128      * Set a user definable field name to \a aName.

 129      *

 130      * Reserved fields such as value and reference are not renamed.  If the field name is

 131      * changed, the field modified flag is set.  If the field is the child of a component,

 132      * the parent component's modified flag is also set.

 133      *

 134      * @param aName - User defined field name.

 135      */
 136     void SetName( const wxString& aName );
 137 
 138     int GetId() const { return m_id; }
 139     void SetId( int aId ) { m_id = aId; }
 140 
 141     int GetPenWidth() const override;
 142 
 143     /**

 144      * Copy parameters of this field to another field. Pointers are not copied.

 145      *

 146      * @param aTarget Target field to copy values to.

 147      */
 148     void Copy( LIB_FIELD* aTarget ) const;
 149 
 150     void ViewGetLayers( int aLayers[], int& aCount ) const override;
 151 
 152     const EDA_RECT GetBoundingBox() const override;
 153 
 154     void GetMsgPanelInfo( EDA_DRAW_FRAME* aFrame, std::vector<MSG_PANEL_ITEM>& aList ) override;
 155 
 156     bool HitTest( const wxPoint& aPosition, int aAccuracy = 0 ) const override;
 157 
 158     LIB_FIELD& operator=( const LIB_FIELD& field );
 159 
 160     /**

 161      * Return the text of a field.

 162      *

 163      * If the field is the reference field, the unit number is used to

 164      * create a pseudo reference text.  If the base reference field is U,

 165      * the string U?A will be returned for unit = 1.

 166      *

 167      * @param unit - The package unit number.  Only effects reference field.

 168      * @return Field text.

 169      */
 170     wxString GetFullText( int unit = 1 ) const;
 171 
 172     SCH_LAYER_ID GetDefaultLayer();
 173 
 174     void BeginEdit( const wxPoint aStartPoint ) override;
 175 
 176     void Offset( const wxPoint& aOffset ) override;
 177 
 178     void MoveTo( const wxPoint& aPosition ) override;
 179 
 180     wxPoint GetPosition() const override { return EDA_TEXT::GetTextPos(); }
 181 
 182     void MirrorHorizontal( const wxPoint& aCenter ) override;
 183     void MirrorVertical( const wxPoint& aCenter ) override;
 184     void Rotate( const wxPoint& aCenter, bool aRotateCCW = true ) override;
 185 
 186     void Plot( PLOTTER* aPlotter, const wxPoint& aOffset, bool aFill,
 187                const TRANSFORM& aTransform ) override;
 188 
 189     int GetWidth() const override { return GetTextThickness(); }
 190     void SetWidth( int aWidth ) override { SetTextThickness( aWidth ); }
 191 
 192     wxString GetSelectMenuText( EDA_UNITS aUnits ) const override;
 193 
 194     BITMAP_DEF GetMenuImage() const override;
 195 
 196     EDA_ITEM* Clone() const override;
 197 
 198     bool IsMandatory() const;
 199 
 200 private:
 201 
 202     /**

 203      * @copydoc LIB_ITEM::compare()

 204      *

 205      * The field specific sort order is as follows:

 206      *

 207      *      - Field ID, REFERENCE, VALUE, etc.

 208      *      - Field string, case insensitive compare.

 209      *      - Field horizontal (X) position.

 210      *      - Field vertical (Y) position.

 211      *      - Field width.

 212      *      - Field height.

 213      */
 214     int compare( const LIB_ITEM& aOther,
 215             LIB_ITEM::COMPARE_FLAGS aCompareFlags = LIB_ITEM::COMPARE_FLAGS::NORMAL ) const override;
 216 };
