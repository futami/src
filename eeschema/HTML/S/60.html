 109 /**

 110  * DIALOG_LIB_EDIT_ONE_FIELD

 111  * is a the class to handle editing a single component field in the library editor.

 112  * <p>

 113  * @note Use ShowQuasiModal when calling this class!

 114  * </p>

 115  */
 116 class DIALOG_LIB_EDIT_ONE_FIELD : public DIALOG_EDIT_ONE_FIELD
 117 {
 118 public:
 119     DIALOG_LIB_EDIT_ONE_FIELD( SCH_BASE_FRAME* aParent, const wxString& aTitle,
 120                                const LIB_FIELD* aField );
 121 
 122     ~DIALOG_LIB_EDIT_ONE_FIELD() {}
 123 
 124     void UpdateField( LIB_FIELD* aField )
 125     {
 126         aField->SetText( m_text );
 127 
 128         // VALUE === component name, so update the parent component if it changes.

 129         if( aField->GetId() == VALUE && aField->GetParent() )
 130             aField->GetParent()->SetName( m_text );
 131 
 132         updateText( aField );
 133     }
 134 };
