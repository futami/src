 116 bool PANEL_EESCHEMA_COLOR_SETTINGS::TransferDataFromWindow()
 117 {
 118     m_currentSettings->SetOverrideSchItemColors( m_optOverrideColors->GetValue() );
 119 
 120     if( !saveCurrentTheme( true ) )
 121         return false;
 122 
 123     m_frame->GetCanvas()->GetView()->GetPainter()->GetSettings()->LoadColors( m_currentSettings );
 124 
 125     SETTINGS_MANAGER& settingsMgr = Pgm().GetSettingsManager();
 126     EESCHEMA_SETTINGS* app_settings = settingsMgr.GetAppSettings<EESCHEMA_SETTINGS>();
 127     app_settings->m_ColorTheme = m_currentSettings->GetFilename();
 128 
 129     return true;
 130 }

 140 bool PANEL_EESCHEMA_COLOR_SETTINGS::validateSave( bool aQuiet )
 141 {
 142     COLOR4D bgcolor = m_currentSettings->GetColor( LAYER_SCHEMATIC_BACKGROUND );
 143 
 144     for( SCH_LAYER_ID layer = SCH_LAYER_ID_START; layer < SCH_LAYER_ID_END; ++layer )
 145     {
 146         if( bgcolor == m_currentSettings->GetColor( layer )
 147             && layer != LAYER_SCHEMATIC_BACKGROUND && layer != LAYER_SHEET_BACKGROUND )
 148         {
 149             wxString msg = _( "Some items have the same color as the background\n"
 150                               "and they will not be seen on the screen.  Are you\n"
 151                               "sure you want to use these colors?" );
 152 
 153             if( wxMessageBox( msg, _( "Warning" ), wxYES_NO | wxICON_QUESTION, this ) == wxNO )
 154                 return false;
 155 
 156             break;
 157         }
 158     }
 159 
 160     return true;
 161 }

 164 bool PANEL_EESCHEMA_COLOR_SETTINGS::saveCurrentTheme( bool aValidate)
 165 {
 166     for( auto layer : m_validLayers )
 167     {
 168         COLOR4D color = m_currentSettings->GetColor( layer );
 169 
 170         // Do not allow non-background layers to be completely white.

 171         // This ensures the BW printing recognizes that the colors should be printed black.

 172         if( color == COLOR4D::WHITE && layer != LAYER_SCHEMATIC_BACKGROUND
 173                 && layer != LAYER_SHEET_BACKGROUND )
 174         {
 175             color.Darken( 0.01 );
 176         }
 177 
 178         m_currentSettings->SetColor( layer, color );
 179     }
 180 
 181     return PANEL_COLOR_SETTINGS::saveCurrentTheme( aValidate );
 182 }

 185 void PANEL_EESCHEMA_COLOR_SETTINGS::createSwatches()
 186 {
 187     std::vector<SCH_LAYER_ID> layers;
 188 
 189     for( SCH_LAYER_ID i = SCH_LAYER_ID_START; i < SCH_LAYER_ID_END; ++i )
 190         layers.push_back( i );
 191 
 192     std::sort( layers.begin(), layers.end(),
 193                []( SCH_LAYER_ID a, SCH_LAYER_ID b )
 194                {
 195                    return LayerName( a ) < LayerName( b );
 196                } );
 197 
 198     for( int layer : layers )
 199     {
 200         wxString name = LayerName( layer );
 201 
 202         if( layer == LAYER_SCHEMATIC_GRID_AXES )
 203             name += _( " (symbol editor only)" );
 204 
 205         createSwatch( layer, name );
 206     }
 207 
 208     // Give a minimal width to m_colorsListWindow, in order to always having

 209     // a full row shown

 210     int min_width = m_colorsGridSizer->GetMinSize().x;
 211     const int margin = 20;  // A margin around the sizer

 212     m_colorsListWindow->SetMinSize( wxSize( min_width + margin, -1 ) );
 213 }

 355 void PANEL_EESCHEMA_COLOR_SETTINGS::updatePreview()
 356 {
 357     KIGFX::VIEW* view = m_preview->GetView();
 358     auto settings = static_cast<KIGFX::SCH_RENDER_SETTINGS*>( view->GetPainter()->GetSettings() );
 359     settings->LoadColors( m_currentSettings );
 360 
 361     m_preview->GetGAL()->SetClearColor( settings->GetBackgroundColor() );
 362 
 363     view->UpdateAllItems( KIGFX::COLOR );
 364     auto rect = m_preview->GetScreenRect();
 365     m_preview->Refresh( true, &rect );
 366 }

 369 void PANEL_EESCHEMA_COLOR_SETTINGS::zoomFitPreview()
 370 {
 371     auto view = m_preview->GetView();
 372 
 373     view->SetScale( 1.0 );
 374     VECTOR2D screenSize = view->ToWorld( m_preview->GetClientSize(), false );
 375 
 376     VECTOR2I psize( m_page->GetWidthIU(), m_page->GetHeightIU() );
 377     double scale = view->GetScale() / std::max( fabs( psize.x / screenSize.x ),
 378                                                 fabs( psize.y / screenSize.y ) );
 379 
 380     view->SetScale( scale / 1.02 );
 381     view->SetCenter( m_ws->ViewBBox().Centre() );
 382     m_preview->ForceRefresh();
 383 }

 393 void PANEL_EESCHEMA_COLOR_SETTINGS::OnOverrideItemColorsClicked( wxCommandEvent& aEvent )
 394 {
 395     m_currentSettings->SetOverrideSchItemColors( m_optOverrideColors->GetValue() );
 396 
 397     // If the theme is not overriding individual item colors then don't show them so that

 398     // the user doesn't get seduced into thinking they'll have some effect.

 399     m_labels[ LAYER_SHEET ]->Show( m_currentSettings->GetOverrideSchItemColors() );
 400     m_swatches[ LAYER_SHEET ]->Show( m_currentSettings->GetOverrideSchItemColors() );
 401 
 402     m_labels[ LAYER_SHEET_BACKGROUND ]->Show( m_currentSettings->GetOverrideSchItemColors() );
 403     m_swatches[ LAYER_SHEET_BACKGROUND ]->Show( m_currentSettings->GetOverrideSchItemColors() );
 404 
 405     m_colorsGridSizer->Layout();
 406     m_colorsListWindow->Layout();
 407 }
