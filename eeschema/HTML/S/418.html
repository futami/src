  25 #ifndef SPICE_SIMULATOR_H
  26 #define SPICE_SIMULATOR_H
  27 
  28 #include "sim_types.h"
  29 
  30 #include <string>
  31 #include <vector>
  32 #include <complex>
  33 #include <memory>
  34 
  35 #include <wx/string.h>
  36 
  37 class SPICE_REPORTER;
  38 class SPICE_SIMULATOR;
  39 
  40 typedef std::complex<double> COMPLEX;
  41 
  42 class SPICE_SIMULATOR
  43 {
  44 public:
  45     SPICE_SIMULATOR() : m_reporter( NULL ) {}
  46     virtual ~SPICE_SIMULATOR() {}
  47 
  48     ///> Creates a simulator instance of particular type (currently only ngspice is handled)

  49     static std::shared_ptr<SPICE_SIMULATOR> CreateInstance( const std::string& aName );
  50 
  51     ///> Initializes the simulator

  52     virtual void Init() = 0;
  53 
  54     /*

  55      * @brief Loads a netlist for the simulation.

  56      * @return True in case of success, false otherwise.

  57      */
  58     virtual bool LoadNetlist( const std::string& aNetlist ) = 0;
  59 
  60     /**

  61      * @brief Executes the simulation with currently loaded netlist.

  62      * @return True in case of success, false otherwise.

  63      */
  64     virtual bool Run() = 0;
  65 
  66     /**

  67      * @brief Halts the simulation.

  68      * @return True in case of success, false otherwise.

  69      */
  70     virtual bool Stop() = 0;
  71 
  72     /**

  73      * @brief Checks if simulation is running at the moment.

  74      * @return True if simulation is currently executed.

  75      */
  76     virtual bool IsRunning() = 0;
  77 
  78     /**

  79      * @brief Executes a Spice command as if it was typed into console.

  80      * @param aCmd is the command to be issued.

  81      */
  82     virtual bool Command( const std::string& aCmd ) = 0;
  83 
  84     ///> Returns X axis name for a given simulation type

  85     virtual std::string GetXAxis( SIM_TYPE aType ) const = 0;
  86 
  87     ///> Sets a SPICE_REPORTER object to receive the simulation log.

  88     virtual void SetReporter( SPICE_REPORTER* aReporter )
  89     {
  90         m_reporter = aReporter;
  91     }
  92 
  93     /**

  94      * @brief Returns a list with all vectors generated in current simulation.

  95      * @param none

  96      * @return List of vector names. ?May not match to the net name elements.

  97      */
  98     virtual std::vector<std::string> AllPlots() = 0;
  99 
 100     /**

 101      * @brief Returns a requested vector with complex values. If the vector is real, then

 102      * the imaginary part is set to 0 in all values.

 103      * @param aName is the vector named in Spice convention (e.g. V(3), I(R1)).

 104      * @param aMaxLen is max count of returned values.

 105      * if -1 (default) all available values are returned.

 106      * @return Requested vector. It might be empty if there is no vector with requested name.

 107      */
 108     virtual std::vector<COMPLEX> GetPlot( const std::string& aName, int aMaxLen = -1 ) = 0;
 109 
 110     /**

 111      * @brief Returns a requested vector with real values. If the vector is complex, then

 112      * the real part is returned.

 113      * @param aName is the vector named in Spice convention (e.g. V(3), I(R1)).

 114      * @param aMaxLen is max count of returned values.

 115      * if -1 (default) all available values are returned.

 116      * @return Requested vector. It might be empty if there is no vector with requested name.

 117      */
 118     virtual std::vector<double> GetRealPlot( const std::string& aName, int aMaxLen = -1 ) = 0;
 119 
 120     /**

 121      * @brief Returns a requested vector with imaginary values. If the vector is complex, then

 122      * the imaginary part is returned. If the vector is reql, then only zeroes are returned.

 123      * @param aName is the vector named in Spice convention (e.g. V(3), I(R1)).

 124      * @param aMaxLen is max count of returned values.

 125      * if -1 (default) all available values are returned.

 126      * @return Requested vector. It might be empty if there is no vector with requested name.

 127      */
 128     virtual std::vector<double> GetImagPlot( const std::string& aName, int aMaxLen = -1 ) = 0;
 129 
 130     /**

 131      * @brief Returns a requested vector with magnitude values.

 132      * @param aName is the vector named in Spice convention (e.g. V(3), I(R1)).

 133      * @param aMaxLen is max count of returned values.

 134      * if -1 (default) all available values are returned.

 135      * @return Requested vector. It might be empty if there is no vector with requested name.

 136      */
 137     virtual std::vector<double> GetMagPlot( const std::string& aName, int aMaxLen = -1 ) = 0;
 138 
 139     /**

 140      * @brief Returns a requested vector with phase values.

 141      * @param aName is the vector named in Spice convention (e.g. V(3), I(R1)).

 142      * @param aMaxLen is max count of returned values.

 143      * if -1 (default) all available values are returned.

 144      * @return Requested vector. It might be empty if there is no vector with requested name.

 145      */
 146     virtual std::vector<double> GetPhasePlot( const std::string& aName, int aMaxLen = -1 ) = 0;
 147 
 148     /**

 149      * @brief Returns current SPICE netlist used by the simulator.

 150      * @return The netlist.

 151      */
 152     virtual const std::string GetNetlist() const = 0;
 153 
 154     /**

 155      * @brief Returns a string with simulation name based on enum.

 156      * @param aType is the enum describing simulation type

 157      * @param aShortName if true  - return is in format "TRAN", "OP".

 158      *                   if false - return is in format "Transient", "Operating Point".

 159      * @return String with requested name as described above.

 160      */
 161     static wxString TypeToName( SIM_TYPE aType, bool aShortName );
 162 
 163 protected:
 164     ///> Reporter object to receive simulation log

 165     SPICE_REPORTER* m_reporter;
 166 };
 167 
 168 #endif /* SPICE_SIMULATOR_H */
