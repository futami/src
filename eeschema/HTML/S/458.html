 174     /**

 175      * Read the footprint info from each line in the stuff file by reference designator.

 176      *

 177      * The footprint link file (.cmp) entries created by CvPcb:

 178      *

 179      *  BeginCmp

 180      *  TimeStamp = /32307DE2/AA450F67;

 181      *  Reference = C1;

 182      *  ValeurCmp = 47uF;

 183      *  IdModule  = CP6;

 184      *  EndCmp

 185      *

 186      * @param aFullFilename = the full filename to read

 187      * @param aForceVisibilityState = Set to true to change the footprint field visibility

 188      *                                state to \a aVisibilityState.  False retains the

 189      *                                current footprint field visibility state.

 190      * @param aVisibilityState True to show the footprint field or false to hide the footprint

 191      *                         field if \a aForceVisibilityState is true.

 192      * @return bool = true if success.

 193      */
 194     bool processCmpToFootprintLinkFile( const wxString& aFullFilename,
 195                                         bool            aForceVisibilityState,
 196                                         bool            aVisibilityState );
