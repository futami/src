 108 class SYMBOL_LIB_TABLE_GRID : public LIB_TABLE_GRID, public SYMBOL_LIB_TABLE
 109 {
 110     friend class SYMBOL_GRID_TRICKS;
 111 
 112 protected:
 113     LIB_TABLE_ROW* at( size_t aIndex ) override { return &rows.at( aIndex ); }
 114 
 115     size_t size() const override { return rows.size(); }
 116 
 117     LIB_TABLE_ROW* makeNewRow() override
 118     {
 119         return dynamic_cast< LIB_TABLE_ROW* >( new SYMBOL_LIB_TABLE_ROW );
 120     }
 121 
 122     LIB_TABLE_ROWS_ITER begin() override { return rows.begin(); }
 123 
 124     LIB_TABLE_ROWS_ITER insert( LIB_TABLE_ROWS_ITER aIterator, LIB_TABLE_ROW* aRow ) override

 131     LIB_TABLE_ROWS_ITER erase( LIB_TABLE_ROWS_ITER aFirst, LIB_TABLE_ROWS_ITER aLast ) override
 132     {
 133         return rows.erase( aFirst, aLast );
 134     }

 145 class SYMBOL_GRID_TRICKS : public GRID_TRICKS
 146 {
 147 public:
 148     SYMBOL_GRID_TRICKS( DIALOG_EDIT_LIBRARY_TABLES* aParent, WX_GRID* aGrid ) :
 149         GRID_TRICKS( aGrid ),
 150         m_dialog( aParent )
 151     {
 152     }
 153 
 154 protected:
 155     DIALOG_EDIT_LIBRARY_TABLES* m_dialog;
 156 
 157     /// handle specialized clipboard text, with leading "(sym_lib_table" or

 158     /// spreadsheet formatted text.

 159     virtual void paste_text( const wxString& cb_text ) override
 160     {
 161         SYMBOL_LIB_TABLE_GRID* tbl = (SYMBOL_LIB_TABLE_GRID*) m_grid->GetTable();
 162         size_t                 ndx = cb_text.find( "(sym_lib_table" );
 163 
 164         if( ndx != std::string::npos )
 165         {
 166             // paste the SYMBOL_LIB_TABLE_ROWs of s-expression (sym_lib_table), starting

 167             // at column 0 regardless of current cursor column.

 168 
 169             STRING_LINE_READER  slr( TO_UTF8( cb_text ), "Clipboard" );
 170             LIB_TABLE_LEXER     lexer( &slr );
 171             SYMBOL_LIB_TABLE    tmp_tbl;
 172             bool                parsed = true;
 173 
 174             try
 175             {
 176                 tmp_tbl.Parse( &lexer );
 177             }
 178             catch( PARSE_ERROR& pe )
 179             {
 180                 DisplayError( m_dialog, pe.What() );
 181                 parsed = false;
 182             }
 183 
 184             if( parsed )
 185             {
 186                 // make sure the table is big enough...

 187                 if( tmp_tbl.GetCount() > (unsigned) tbl->GetNumberRows() )
 188                     tbl->AppendRows( tmp_tbl.GetCount() - tbl->GetNumberRows() );
 189 
 190                 for( unsigned i = 0;  i < tmp_tbl.GetCount();  ++i )
 191                     tbl->rows.replace( i, tmp_tbl.At( i ).clone() );
 192             }
 193 
 194             m_grid->AutoSizeColumns( false );
 195         }
 196         else
 197         {
 198             // paste spreadsheet formatted text.

 199             GRID_TRICKS::paste_text( cb_text );
 200 
 201             m_grid->AutoSizeColumns( false );
 202         }
 203     }
 204 };

 273         g->AutoSizeColumn( COL_URI, false );
 274         g->AutoSizeColumn( COL_DESCR, false );
 275         g->AutoSizeColumn( COL_ENABLED, false );

 282     m_auinotebook->SetSelection( m_pageNdx );

 323 bool PANEL_SYM_LIB_TABLE::verifyTables()
 324 {
 325     for( SYMBOL_LIB_TABLE_GRID* model : { global_model(), project_model() } )
 326     {
 327         for( int r = 0; r < model->GetNumberRows(); )
 328         {
 329             wxString nick = model->GetValue( r, COL_NICKNAME ).Trim( false ).Trim();
 330             wxString uri  = model->GetValue( r, COL_URI ).Trim( false ).Trim();
 331             unsigned illegalCh = 0;
 332 
 333             if( !nick || !uri )
 334             {
 335                 // Delete the "empty" row, where empty means missing nick or uri.

 336                 // This also updates the UI which could be slow, but there should only be a few

 337                 // rows to delete, unless the user fell asleep on the Add Row

 338                 // button.

 339                 model->DeleteRows( r, 1 );
 340             }
 341             else if( ( illegalCh = LIB_ID::FindIllegalLibNicknameChar( nick, LIB_ID::ID_SCH ) ) )
 342             {
 343                 wxString msg = wxString::Format( _( "Illegal character '%c' in Nickname: \"%s\"" ),
 344                                                  illegalCh,
 345                                                  nick );
 346 
 347                 // show the tabbed panel holding the grid we have flunked:

 348                 if( model != cur_model() )
 349                     m_auinotebook->SetSelection( model == global_model() ? 0 : 1 );
 350 
 351                 m_cur_grid->MakeCellVisible( r, 0 );
 352                 m_cur_grid->SetGridCursor( r, 1 );
 353 
 354                 wxMessageDialog errdlg( this, msg, _( "No Colon in Nicknames" ) );
 355                 errdlg.ShowModal();
 356                 return false;
 357             }
 358             else
 359             {
 360                 // set the trimmed values back into the table so they get saved to disk.

 361                 model->SetValue( r, COL_NICKNAME, nick );
 362                 model->SetValue( r, COL_URI, uri );
 363                 ++r;        // this row was OK.

 364             }
 365         }
 366     }
 367 
 368     // check for duplicate nickNames, separately in each table.

 369     for( SYMBOL_LIB_TABLE_GRID* model : { global_model(), project_model() } )
 370     {
 371         for( int r1 = 0; r1 < model->GetNumberRows() - 1;  ++r1 )
 372         {
 373             wxString    nick1 = model->GetValue( r1, COL_NICKNAME );
 374 
 375             for( int r2=r1+1; r2 < model->GetNumberRows();  ++r2 )
 376             {
 377                 wxString    nick2 = model->GetValue( r2, COL_NICKNAME );
 378 
 379                 if( nick1 == nick2 )
 380                 {
 381                     wxString msg = wxString::Format( _( "Duplicate Nickname: \"%s\"." ), nick1 );
 382 
 383                     // show the tabbed panel holding the grid we have flunked:

 384                     if( model != cur_model() )
 385                         m_auinotebook->SetSelection( model == global_model() ? 0 : 1 );
 386 
 387                     // go to the lower of the two rows, it is technically the duplicate:

 388                     m_cur_grid->MakeCellVisible( r2, 0 );
 389                     m_cur_grid->SetGridCursor( r2, 1 );
 390 
 391                     wxMessageDialog errdlg( this, msg, _( "Please Delete or Modify One" ) );
 392                     errdlg.ShowModal();
 393 
 394                     return false;
 395                 }
 396             }
 397         }
 398     }
 399 
 400     for( SYMBOL_LIB_TABLE* table : { global_model(), project_model() } )
 401     {
 402         for( unsigned int r = 0; r < table->GetCount(); ++r )
 403         {
 404             SYMBOL_LIB_TABLE_ROW& row = dynamic_cast<SYMBOL_LIB_TABLE_ROW&>( table->At( r ) );
 405 
 406             if( !row.GetIsEnabled() )
 407                 continue;
 408 
 409             try
 410             {
 411                 if( row.Refresh() )
 412                 {
 413                     if( table == global_model() )
 414                         m_parent->m_GlobalTableChanged = true;
 415                     else
 416                         m_parent->m_ProjectTableChanged = true;
 417                 }
 418             }
 419             catch( const IO_ERROR& ioe )
 420             {
 421                 wxString msg = wxString::Format( _( "Symbol library \"%s\" failed to load.\n %s" ),
 422                                                  row.GetNickName(),
 423                                                  ioe.What() );
 424 
 425                 wxMessageDialog errdlg( this, msg, _( "Error Loading Library" ) );
 426                 errdlg.ShowModal();
 427 
 428                 return false;
 429             }
 430         }
 431     }
 432 
 433     return true;
 434 }

 444 void PANEL_SYM_LIB_TABLE::browseLibrariesHandler( wxCommandEvent& event )
 445 {
 446     wxString wildcards = AllSymbolLibFilesWildcard()
 447                             + "|" + KiCadSymbolLibFileWildcard()
 448                             + "|" + LegacySymbolLibFileWildcard();
 449 
 450     wxFileDialog dlg( this, _( "Select Library" ), m_lastBrowseDir, wxEmptyString, wildcards,
 451                       wxFD_OPEN | wxFD_FILE_MUST_EXIST | wxFD_MULTIPLE );
 452 
 453     auto result = dlg.ShowModal();
 454 
 455     if( result == wxID_CANCEL )
 456         return;
 457 
 458     m_lastBrowseDir = dlg.GetDirectory();
 459 
 460     const ENV_VAR_MAP& envVars       = Pgm().GetLocalEnvVariables();
 461     bool               addDuplicates = false;
 462     bool               applyToAll    = false;
 463     wxString           warning       = _( "Warning: Duplicate Nickname" );
 464     wxString           msg           = _( "A library nicknamed \"%s\" already exists." );
 465     wxString           detailedMsg   = _( "Please change the library nickname after adding this library." );
 466 
 467     wxArrayString files;
 468     dlg.GetFilenames( files );
 469 
 470     for( const auto& file : files )
 471     {
 472         wxString filePath = dlg.GetDirectory() + wxFileName::GetPathSeparator() + file;
 473         wxFileName fn( filePath );
 474         wxString nickname = LIB_ID::FixIllegalChars( fn.GetName(), LIB_ID::ID_SCH );
 475         bool doAdd = true;
 476 
 477         if( cur_model()->ContainsNickname( nickname ) )
 478         {
 479             if( !applyToAll )
 480             {
 481                 // The cancel button adds the library to the table anyway

 482                 addDuplicates = ( OKOrCancelDialog( this, warning, wxString::Format( msg, nickname ),
 483                         detailedMsg, _( "Skip" ), _( "Add Anyway" ), &applyToAll ) == wxID_CANCEL );
 484             }
 485 
 486             doAdd = addDuplicates;
 487         }
 488 
 489         if( doAdd && m_cur_grid->AppendRows( 1 ) )
 490         {
 491             int last_row = m_cur_grid->GetNumberRows() - 1;
 492 
 493             m_cur_grid->SetCellValue( last_row, COL_NICKNAME, nickname );
 494 
 495             // TODO the following code can detect only schematic types, not libs

 496             // SCH_IO_MGR needs to provide file extension information for libraries too

 497 
 498             // auto detect the plugin type

 499             for( auto pluginType : SCH_IO_MGR::SCH_FILE_T_vector )
 500             {
 501                 if( SCH_IO_MGR::GetLibraryFileExtension( pluginType ).Lower() == fn.GetExt().Lower() )
 502                 {
 503                     m_cur_grid->SetCellValue( last_row, COL_TYPE,
 504                             SCH_IO_MGR::ShowType( pluginType ) );
 505                     break;
 506                 }
 507             }
 508 
 509             // try to use path normalized to an environmental variable or project path

 510             wxString path = NormalizePath( filePath, &envVars, m_projectBasePath );
 511 
 512             // Do not use the project path in the global library table.  This will almost

 513             // assuredly be wrong for a different project.

 514             if( path.IsEmpty() || (m_pageNdx == 0 && path.Contains( "${KIPRJMOD}" )) )
 515                 path = fn.GetFullPath();
 516 
 517             m_cur_grid->SetCellValue( last_row, COL_URI, path );
 518         }
 519     }
 520 
 521     if( !files.IsEmpty() )
 522     {
 523         m_cur_grid->MakeCellVisible( m_cur_grid->GetNumberRows() - 1, 0 );
 524         m_cur_grid->SetGridCursor( m_cur_grid->GetNumberRows() - 1, 1 );
 525     }
 526 }

 547 void PANEL_SYM_LIB_TABLE::deleteRowHandler( wxCommandEvent& event )
 548 {
 549     if( !m_cur_grid->CommitPendingChanges() )
 550         return;
 551 
 552     int curRow = m_cur_grid->GetGridCursorRow();
 553     int curCol = m_cur_grid->GetGridCursorCol();
 554 
 555     // In a wxGrid, collect rows that have a selected cell, or are selected

 556     // It is not so easy: it depends on the way the selection was made.

 557     // Here, we collect rows selected by clicking on a row label, and rows that contain

 558     // previously-selected cells.

 559     // If no candidate, just delete the row with the grid cursor.

 560     wxArrayInt selectedRows = m_cur_grid->GetSelectedRows();
 561     wxGridCellCoordsArray cells = m_cur_grid->GetSelectedCells();
 562     wxGridCellCoordsArray blockTopLeft = m_cur_grid->GetSelectionBlockTopLeft();
 563     wxGridCellCoordsArray blockBotRight = m_cur_grid->GetSelectionBlockBottomRight();
 564 
 565     // Add all row having cell selected to list:

 566     for( unsigned ii = 0; ii < cells.GetCount(); ii++ )
 567         selectedRows.Add( cells[ii].GetRow() );
 568 
 569     // Handle block selection

 570     if( !blockTopLeft.IsEmpty() && !blockBotRight.IsEmpty() )
 571     {
 572         for( int i = blockTopLeft[0].GetRow(); i <= blockBotRight[0].GetRow(); ++i )
 573             selectedRows.Add( i );
 574     }
 575 
 576     // Use the row having the grid cursor only if we have no candidate:

 577     if( selectedRows.size() == 0 && m_cur_grid->GetGridCursorRow() >= 0 )
 578         selectedRows.Add( m_cur_grid->GetGridCursorRow() );
 579 
 580     if( selectedRows.size() == 0 )
 581     {
 582         wxBell();
 583         return;
 584     }
 585 
 586     std::sort( selectedRows.begin(), selectedRows.end() );
 587 
 588     // Remove selected rows (note: a row can be stored more than once in list)

 589     int last_row = -1;
 590 
 591     for( int ii = selectedRows.GetCount()-1; ii >= 0; ii-- )
 592     {
 593         int row = selectedRows[ii];
 594 
 595         if( row != last_row )
 596         {
 597             last_row = row;
 598             m_cur_grid->DeleteRows( row, 1 );
 599         }
 600     }
 601 
 602     if( m_cur_grid->GetNumberRows() > 0 && curRow >= 0 )
 603         m_cur_grid->SetGridCursor( std::min( curRow, m_cur_grid->GetNumberRows() - 1 ), curCol );
 604 }

 607 void PANEL_SYM_LIB_TABLE::moveUpHandler( wxCommandEvent& event )
 608 {
 609     if( !m_cur_grid->CommitPendingChanges() )
 610         return;
 611 
 612     SYMBOL_LIB_TABLE_GRID* tbl = cur_model();
 613     int curRow = m_cur_grid->GetGridCursorRow();
 614 
 615     // @todo: add multiple selection moves.

 616     if( curRow >= 1 )
 617     {
 618         boost::ptr_vector< LIB_TABLE_ROW >::auto_type move_me =
 619             tbl->rows.release( tbl->rows.begin() + curRow );
 620 
 621         --curRow;
 622         tbl->rows.insert( tbl->rows.begin() + curRow, move_me.release() );
 623 
 624         if( tbl->GetView() )
 625         {
 626             // Update the wxGrid

 627             wxGridTableMessage msg( tbl, wxGRIDTABLE_NOTIFY_ROWS_INSERTED, curRow, 0 );
 628             tbl->GetView()->ProcessTableMessage( msg );
 629         }
 630 
 631         m_cur_grid->MakeCellVisible( curRow, m_cur_grid->GetGridCursorCol() );
 632         m_cur_grid->SetGridCursor( curRow, m_cur_grid->GetGridCursorCol() );
 633     }
 634 }

 667 bool PANEL_SYM_LIB_TABLE::TransferDataFromWindow()
 668 {
 669     if( !m_cur_grid->CommitPendingChanges() )
 670         return false;
 671 
 672     if( !verifyTables() )
 673         return false;
 674 
 675     if( *global_model() != *m_globalTable )
 676     {
 677         m_parent->m_GlobalTableChanged = true;
 678 
 679         m_globalTable->Clear();
 680         m_globalTable->rows.transfer( m_globalTable->rows.end(), global_model()->rows.begin(),
 681                                       global_model()->rows.end(), global_model()->rows );
 682         m_globalTable->reindex();
 683     }
 684 
 685     if( *project_model() != *m_projectTable )
 686     {
 687         m_parent->m_ProjectTableChanged = true;
 688 
 689         m_projectTable->Clear();
 690         m_projectTable->rows.transfer( m_projectTable->rows.end(), project_model()->rows.begin(),
 691                                        project_model()->rows.end(), project_model()->rows );
 692         m_projectTable->reindex();
 693     }
 694 
 695     return true;
 696 }

 699 void PANEL_SYM_LIB_TABLE::populateEnvironReadOnlyTable()
 700 {
 701     wxRegEx re( ".*?(\\$\\{(.+?)\\})|(\\$\\((.+?)\\)).*?", wxRE_ADVANCED );
 702     wxASSERT( re.IsValid() );   // wxRE_ADVANCED is required.

 703 
 704     std::set< wxString > unique;
 705 
 706     // clear the table

 707     m_path_subs_grid->DeleteRows( 0, m_path_subs_grid->GetNumberRows() );
 708 
 709     for( SYMBOL_LIB_TABLE_GRID* tbl : { global_model(), project_model() } )
 710     {
 711         for( int row = 0; row < tbl->GetNumberRows(); ++row )
 712         {
 713             wxString uri = tbl->GetValue( row, COL_URI );
 714 
 715             while( re.Matches( uri ) )
 716             {
 717                 wxString envvar = re.GetMatch( uri, 2 );
 718 
 719                 // if not ${...} form then must be $(...)

 720                 if( envvar.IsEmpty() )
 721                     envvar = re.GetMatch( uri, 4 );
 722 
 723                 // ignore duplicates

 724                 unique.insert( envvar );
 725 
 726                 // delete the last match and search again

 727                 uri.Replace( re.GetMatch( uri, 0 ), wxEmptyString );
 728             }
 729         }
 730     }
 731 
 732     // Make sure this special environment variable shows up even if it was

 733     // not used yet.  It is automatically set by KiCad to the directory holding

 734     // the current project.

 735     unique.insert( PROJECT_VAR_NAME );
 736     unique.insert( SYMBOL_LIB_TABLE::GlobalPathEnvVariableName() );
 737 
 738     for( const wxString& evName : unique )
 739     {
 740         int row = m_path_subs_grid->GetNumberRows();
 741         m_path_subs_grid->AppendRows( 1 );
 742 
 743         m_path_subs_grid->SetCellValue( row, 0, wxT( "${" ) + evName + wxT( "}" ) );
 744         m_path_subs_grid->SetCellEditor( row, 0, new GRID_CELL_READONLY_TEXT_EDITOR() );
 745 
 746         wxString evValue;
 747         wxGetEnv( evName, &evValue );
 748         m_path_subs_grid->SetCellValue( row, 1, evValue );
 749         m_path_subs_grid->SetCellEditor( row, 1, new GRID_CELL_READONLY_TEXT_EDITOR() );
 750     }
 751 
 752     // No combobox editors here, but it looks better if its consistent with the other

 753     // grids in the dialog.

 754     m_path_subs_grid->SetDefaultRowSize( m_path_subs_grid->GetDefaultRowSize() + 2 );
 755 
 756     adjustPathSubsGridColumns( m_path_subs_grid->GetRect().GetWidth() );
 757 }

 799 void InvokeSchEditSymbolLibTable( KIWAY* aKiway, wxWindow *aParent )
 800 {
 801     SCH_EDIT_FRAME*   schEditor = (SCH_EDIT_FRAME*) aKiway->Player( FRAME_SCH, false );
 802     LIB_EDIT_FRAME*   libEditor = (LIB_EDIT_FRAME*) aKiway->Player( FRAME_SCH_LIB_EDITOR, false );
 803     LIB_VIEW_FRAME*   libViewer = (LIB_VIEW_FRAME*) aKiway->Player( FRAME_SCH_VIEWER, false );
 804 
 805     SYMBOL_LIB_TABLE* globalTable = &SYMBOL_LIB_TABLE::GetGlobalLibTable();
 806     wxString          globalTablePath = SYMBOL_LIB_TABLE::GetGlobalTableFileName();
 807     SYMBOL_LIB_TABLE* projectTable = aKiway->Prj().SchSymbolLibTable();
 808     wxString          projectPath = aKiway->Prj().GetProjectPath();
 809     wxFileName        projectTableFn( projectPath, SYMBOL_LIB_TABLE::GetSymbolLibTableFileName() );
 810     wxString          msg;
 811     wxString          currentLib;
 812 
 813     if( libEditor )
 814     {
 815         currentLib = libEditor->GetCurLib();
 816 
 817         // This prevents an ugly crash on OSX (https://bugs.launchpad.net/kicad/+bug/1765286)

 818         libEditor->FreezeSearchTree();
 819 
 820         if( libEditor->HasLibModifications() )
 821         {
 822             msg = _( "Modifications have been made to one or more symbol libraries.\n"
 823                      "Changes must be saved or discarded before the symbol library "
 824                      "table can be modified." );
 825 
 826             switch( UnsavedChangesDialog( aParent, msg ) )
 827             {
 828             case wxID_YES:    libEditor->SaveAll(); break;
 829             case wxID_NO:     libEditor->RevertAll(); break;
 830             default:
 831             case wxID_CANCEL: libEditor->ThawSearchTree(); return;
 832             }
 833         }
 834     }
 835 
 836     DIALOG_EDIT_LIBRARY_TABLES dlg( aParent, _( "Symbol Libraries" ) );
 837 
 838     dlg.InstallPanel( new PANEL_SYM_LIB_TABLE( &dlg, globalTable, globalTablePath,
 839                                                projectTable, projectTableFn.GetFullPath(),
 840                                                aKiway->Prj().GetProjectPath() ) );
 841 
 842     if( dlg.ShowModal() == wxID_CANCEL )
 843     {
 844         if( libEditor )
 845             libEditor->ThawSearchTree();
 846 
 847         return;
 848     }
 849 
 850     if( dlg.m_GlobalTableChanged )
 851     {
 852         try
 853         {
 854             globalTable->Save( globalTablePath );
 855         }
 856         catch( const IO_ERROR& ioe )
 857         {
 858             msg.Printf( _( "Error saving global library table:\n\n%s" ), ioe.What() );
 859             wxMessageBox( msg, _( "File Save Error" ), wxOK | wxICON_ERROR );
 860         }
 861     }
 862 
 863     if( dlg.m_ProjectTableChanged )
 864     {
 865         try
 866         {
 867             projectTable->Save( projectTableFn.GetFullPath() );
 868         }
 869         catch( const IO_ERROR& ioe )
 870         {
 871             msg.Printf( _( "Error saving project-specific library table:\n\n%s" ), ioe.What() );
 872             wxMessageBox( msg, _( "File Save Error" ), wxOK | wxICON_ERROR );
 873         }
 874     }
 875 
 876     if( schEditor )
 877         schEditor->SyncView();
 878 
 879     if( libEditor )
 880     {
 881         // Check if the currently selected symbol library been removed or disabled.

 882         if( !currentLib.empty() && !projectTable->HasLibrary( currentLib, true ) )
 883         {
 884             libEditor->SetCurLib( wxEmptyString );
 885             libEditor->emptyScreen();
 886         }
 887 
 888         libEditor->SyncLibraries( true );
 889         libEditor->ThawSearchTree();
 890     }
 891 
 892     if( libViewer )
 893         libViewer->ReCreateListLib();
 894 }
