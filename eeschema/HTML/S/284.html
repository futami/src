  32 /**

  33  * Define a symbol library graphical text item.

  34  * <p>

  35  * This is only a graphical text item.  Field text like the reference designator,

  36  * symbol value, etc. are not LIB_TEXT items.  See the #LIB_FIELD class for the

  37  * field item definition.

  38  * </p>

  39  */

  43                 const TRANSFORM& aTransform ) override;
  44 
  45 public:

  47 
  48     // Do not create a copy constructor.  The one generated by the compiler is adequate.

  49

  51 
  52     wxString GetClass() const override
  53     {
  54         return wxT( "LIB_TEXT" );

  56

  58     {
  59         return _( "Text" );
  60     }
  61 
  62     void ViewGetLayers( int aLayers[], int& aCount ) const override;
  63 
  64     bool HitTest( const wxPoint& aPosition, int aAccuracy = 0 ) const override;

  66     bool HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy = 0 ) const override

  71     int GetPenWidth() const override;

  86     void MirrorHorizontal( const wxPoint& aCenter ) override;
  87     void MirrorVertical( const wxPoint& aCenter ) override;
  88     void Rotate( const wxPoint& aCenter, bool aRotateCCW = true ) override;

 106     /**

 117             LIB_ITEM::COMPARE_FLAGS aCompareFlags = LIB_ITEM::COMPARE_FLAGS::NORMAL ) const override;
