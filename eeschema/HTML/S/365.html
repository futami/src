  62 void SCH_JUNCTION::SwapData( SCH_ITEM* aItem )
  63 {
  64     wxCHECK_RET( (aItem != NULL) && (aItem->Type() == SCH_JUNCTION_T),
  65                  wxT( "Cannot swap junction data with invalid item." ) );
  66 
  67     SCH_JUNCTION* item = (SCH_JUNCTION*) aItem;
  68     std::swap( m_pos, item->m_pos );
  69     std::swap( m_diameter, item->m_diameter );
  70     std::swap( m_color, item->m_color );
  71 }

  82 const EDA_RECT SCH_JUNCTION::GetBoundingBox() const
  83 {
  84     EDA_RECT rect;
  85 
  86     int size =
  87             Schematic() ? Schematic()->Settings().m_JunctionSize : Mils2iu( DEFAULT_JUNCTION_DIAM );
  88 
  89     if( m_diameter != 0 )
  90         size = m_diameter;
  91 
  92     rect.SetOrigin( m_pos );
  93     rect.Inflate( ( GetPenWidth() + size ) / 2 );
  94 
  95     return rect;
  96 }

  99 void SCH_JUNCTION::Print( RENDER_SETTINGS* aSettings, const wxPoint& aOffset )
 100 {
 101     wxDC*   DC    = aSettings->GetPrintDC();
 102     COLOR4D color = ( m_color == COLOR4D::UNSPECIFIED ) ? aSettings->GetLayerColor( GetLayer() ) :
 103             m_color ;
 104     int diameter =
 105             Schematic() ? Schematic()->Settings().m_JunctionSize : Mils2iu( DEFAULT_JUNCTION_DIAM );
 106 
 107     if( m_diameter != 0 )
 108         diameter = m_diameter;
 109 
 110     GRFilledCircle( nullptr, DC, m_pos.x + aOffset.x, m_pos.y + aOffset.y,
 111                     diameter / 2, 0, color, color );
 112 }

 173 bool SCH_JUNCTION::HitTest( const wxPoint& aPosition, int aAccuracy ) const
 174 {
 175     EDA_RECT rect = GetBoundingBox();
 176 
 177     rect.Inflate( aAccuracy );
 178 
 179     return rect.Contains( aPosition );
 180 }

 182 
 183 bool SCH_JUNCTION::HitTest( const EDA_RECT& aRect, bool aContained, int aAccuracy ) const
 184 {
 185     if( m_Flags & STRUCT_DELETED || m_Flags & SKIP_STRUCT )
 186         return false;
 187 
 188     EDA_RECT rect = aRect;
 189 
 190     rect.Inflate( aAccuracy );
 191 
 192     if( aContained )
 193         return rect.Contains( GetBoundingBox() );
 194 
 195     return rect.Intersects( GetBoundingBox() );
 196 }

 205 void SCH_JUNCTION::Plot( PLOTTER* aPlotter )
 206 {
 207     auto* settings = static_cast<KIGFX::SCH_RENDER_SETTINGS*>( aPlotter->RenderSettings() );
 208 
 209     COLOR4D color = ( m_color == COLOR4D::UNSPECIFIED ) ? settings->GetLayerColor( GetLayer() ) :
 210             m_color;
 211     int diameter =
 212             Schematic() ? Schematic()->Settings().m_JunctionSize : Mils2iu( DEFAULT_JUNCTION_DIAM );
 213 
 214     if( m_diameter != 0 )
 215         diameter = m_diameter;
 216 
 217     aPlotter->SetColor( color );
 218     aPlotter->Circle( m_pos, diameter, FILLED_SHAPE );
 219 }

 228 bool SCH_JUNCTION::operator <( const SCH_ITEM& aItem ) const
 229 {
 230     if( Type() != aItem.Type() )
 231         return Type() < aItem.Type();
 232 
 233     if( GetLayer() != aItem.GetLayer() )
 234         return GetLayer() < aItem.GetLayer();
 235 
 236     auto junction = static_cast<const SCH_JUNCTION*>( &aItem );
 237 
 238     if( GetPosition().x != junction->GetPosition().x )
 239         return GetPosition().x < junction->GetPosition().x;
 240 
 241     if( GetPosition().y != junction->GetPosition().y )
 242         return GetPosition().y < junction->GetPosition().y;
 243 
 244     if( GetDiameter() != junction->GetDiameter() )
 245         return GetDiameter() < junction->GetDiameter();
 246 
 247     return GetColor() < junction->GetColor();
 248 }
