  48     /**

  49      * Add libraries found in the legacy library list to \a aLibs that are not found in

  50      * the global symbol library table.

  51      *

  52      * This function is used to create a project symbol library table when converting

  53      * legacy projects over to the new symbol library table implementation.  This only

  56      *

  57      * @param aLibs is a vector container to add all of the libraries not found in the

  58      *              global symbol library table that were found in the legacy library

  59      *              list.

  60      * @return the number of libraries found.

  61      */

  63 
  64     void createProjectSymbolLibTable( REPORTER& aReporter );
  65 
  66     void remapSymbolsToLibTable( REPORTER& aReporter );
  67 
  68     bool remapSymbolToLibTable( SCH_COMPONENT* aSymbol );
  69 
  70     /**

  71      * Backup all of the files that could be modified by the remapping with a time stamp

  72      * appended to the file name into the "remap_backup" folder in case something goes wrong.

  73      *

  74      * Backup the following:

  75      * - All schematic (prj-name.sch -> remap_backup/prj-name-time-stamp.sch ) files.

  76      * - The project (prj-name.pro) -> remap_backup/prj-name-time-stamp.pro) file.

  77      * - The cache library (prj-name-cache.lib -> remap_backup/prj-name.-cache-time-stamp.lib)

  78      *   file.

  79      * - The rescue library (prj-name-rescue.lib -> remap_backup/prj-name.rescue-time-stamp.lib)

  80      *   file.

  81      * - The rescue library (prj-name-rescue.dcm -> remap_backup/prj-name.rescue-time-stamp.dcm)

  82      *   file.

  83      *

  84      * @param aReporter is the #REPORTER object in which to write information messages.

  85      * @return true to continue rescue or false to abort rescue.

  86      */
  87     bool backupProject( REPORTER& aReporter );
  88 
  89     bool m_remapped;

  91     SCH_EDIT_FRAME* m_frame;
