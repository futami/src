  44 /** TREE_PROJECT_FRAME

  45  * Window to display the tree files

  46  */
  47 class TREE_PROJECT_FRAME : public wxSashLayoutWindow
  48 {
  49     friend class TREEPROJECT_ITEM;
  50 public:
  51     KICAD_MANAGER_FRAME*    m_Parent;
  52     TREEPROJECTFILES*       m_TreeProject;
  53 
  54 private:
  55     bool                    m_isRenaming; // Are we in the process of renaming a file

  56     wxTreeItemId            m_root;
  57     std::vector<wxString>   m_filters;
  58     wxFileSystemWatcher*    m_watcher; // file system watcher

  59 
  60 public:
  61     TREE_PROJECT_FRAME( KICAD_MANAGER_FRAME* parent );
  62     ~TREE_PROJECT_FRAME();
  63 
  64     /**

  65      * Create or modify the tree showing project file names

  66      */
  67     void ReCreateTreePrj();

  69     /**

  70      * Reinit the watched paths

  71      * Should be called after opening a new project to

  72      * rebuild the list of watched paths.

  73      * Should be called *atfer* the main loop event handler is started

  74      */
  75     void FileWatcherReset();
  76 
  77 protected:
  78     static wxString GetFileExt( TreeFileType type );
  79 
  80     /**

  81      * Function GetSelectedData

  82      * return the item data from item currently selected (highlighted)

  83      * Note this is not necessary the "clicked" item,

  84      * because when expanding, collapsing an item this item is not selected

  85      */
  86     std::vector<TREEPROJECT_ITEM*> GetSelectedData();
  87 
  88     /**

  89      * Function GetItemIdData

  90      * return the item data corresponding to a wxTreeItemId identifier

  91      * @param  aId = the wxTreeItemId identifier.

  92      * @return a TREEPROJECT_ITEM pointer corresponding to item id aId

  93      */
  94     TREEPROJECT_ITEM* GetItemIdData( wxTreeItemId aId );
  95 
  96 private:
  97     /**

  98      * Called on a double click on an item

  99      */
 100     void OnSelect( wxTreeEvent& Event );
 101 
 102     /**

 103      * Called on a click on the + or - button of an item with children

 104      */
 105     void OnExpand( wxTreeEvent& Event );
 106 
 107     /**

 108      * Called on a right click on an item

 109      */
 110     void OnRight( wxTreeEvent& Event );
 111 
 112     /**

 113      * Function OnOpenSelectedFileWithTextEditor

 114      * Call the text editor to open the selected file in the tree project

 115      */
 116     void OnOpenSelectedFileWithTextEditor( wxCommandEvent& event );
 117 
 118     /**

 119      * Function OnDeleteFile

 120      * Delete the selected file or directory in the tree project

 121      */
 122     void OnDeleteFile( wxCommandEvent& event );
 123 
 124     /**

 125      * Function OnDeleteFile

 126      * Print the selected file or directory in the tree project

 127      */
 128     void OnPrintFile( wxCommandEvent& event );
 129 
 130     /**

 131      * Function OnRenameFile

 132      * Rename the selected file or directory in the tree project

 133      */
 134     void OnRenameFile( wxCommandEvent& event );
 135 
 136     /**

 137      * Function OnOpenDirectory

 138      * Handles the right-click menu for opening a directory in the current system file browser

 139      */
 140     void OnOpenDirectory( wxCommandEvent& event );
 141 
 142     /**

 143      * Function OnCreateNewDirectory

 144      * Creates a new subdirectory inside the current kicad project directory the user is

 145      * prompted to enter a directory name

 146      */
 147     void OnCreateNewDirectory( wxCommandEvent& event );
 148 
 149     /**

 150      * Switch to a other project selected from the tree project (by selecting an other .pro

 151      * file inside the current project folder)

 152      */
 153     void OnSwitchToSelectedProject( wxCommandEvent& event );
 154 
 155     /**

 156      * Function AddItemToTreeProject

 157      * @brief  Add the file or directory aName to the project tree

 158      * @param aName = the filename or the directory name to add in tree

 159      * @param aRoot = the wxTreeItemId item where to add sub tree items

 160      * @param aCanResetFileWatcher = true to reset file watcher if additional conditions are met

 161      * @param aRecurse = true to add file or subdir names to the current tree item

 162      *                   false to stop file add.

 163      * @return the Id for the new tree item

 164      */
 165     wxTreeItemId AddItemToTreeProject( const wxString& aName, wxTreeItemId& aRoot,
 166             bool aCanResetFileWatcher = true, bool aRecurse = true );
 167 
 168     /**

 169      * Function findSubdirTreeItem

 170      * searches for the item in tree project which is the node of the subdirectory aSubDir

 171      * @param aSubDir = the directory to find in tree

 172      * @return the opaque reference to the tree item; if not found, return an invalid tree item

 173      *         so that wxTreeItemId::IsOk() can be used to test the returned value

 174      */
 175     wxTreeItemId findSubdirTreeItem( const wxString& aSubDir );
 176 
 177     /**

 178      * called when a file or directory is modified/created/deleted

 179      * The tree project is modified when a file or directory is created/deleted/renamed to

 180      * reflect the file change

 181      */
 182     void OnFileSystemEvent( wxFileSystemWatcherEvent& event );
 183 
 184     DECLARE_EVENT_TABLE()
 185 };
