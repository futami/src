  88 KICAD_MANAGER_FRAME::KICAD_MANAGER_FRAME( wxWindow* parent, const wxString& title,
  89                                           const wxPoint& pos, const wxSize&   size ) :
  90         EDA_BASE_FRAME( parent, KICAD_MAIN_FRAME_T, title, pos, size,
  91                         KICAD_DEFAULT_DRAWFRAME_STYLE, KICAD_MANAGER_FRAME_NAME, &::Kiway ),
  92         m_leftWin( nullptr ),
  93         m_launcher( nullptr ),
  94         m_messagesBox( nullptr ),
  95         m_mainToolBar( nullptr )
  96 {
  97     m_active_project = false;
  98     m_leftWinWidth = 250;       // Default value

  99     m_AboutTitle = "KiCad";
 100 
 101     // Create the status line (bottom of the frame)

 102     static const int dims[3] = { -1, -1, 100 };
 103 
 104     CreateStatusBar( 3 );
 105     SetStatusWidths( 3, dims );
 106 
 107     // Give an icon

 108     wxIcon icon;
 109     icon.CopyFromBitmap( KiBitmap( icon_kicad_xpm ) );
 110     SetIcon( icon );
 111 
 112     // Load the settings

 113     LoadSettings( config() );
 114 
 115     // Left window: is the box which display tree project

 116     m_leftWin = new TREE_PROJECT_FRAME( this );
 117 
 118     // Add the wxTextCtrl showing all messages from KiCad:

 119     m_messagesBox = new wxTextCtrl( this, wxID_ANY, wxEmptyString,
 120                                     wxDefaultPosition, wxDefaultSize,
 121                                     wxTE_MULTILINE | wxTE_READONLY | wxBORDER_NONE );
 122 
 123     // Create the manager

 124     m_toolManager = new TOOL_MANAGER;
 125     m_toolManager->SetEnvironment( nullptr, nullptr, nullptr, config(), this );
 126     m_actions = new KICAD_MANAGER_ACTIONS();
 127 
 128     m_toolDispatcher = new TOOL_DISPATCHER( m_toolManager, m_actions );
 129 
 130     // Attach the events to the tool dispatcher

 131     Bind( wxEVT_TOOL, &TOOL_DISPATCHER::DispatchWxCommand, m_toolDispatcher );
 132     Bind( wxEVT_CHAR, &TOOL_DISPATCHER::DispatchWxEvent, m_toolDispatcher );
 133     Bind( wxEVT_CHAR_HOOK, &TOOL_DISPATCHER::DispatchWxEvent, m_toolDispatcher );
 134 
 135     // Register tools

 136     m_toolManager->RegisterTool( new COMMON_CONTROL );
 137     m_toolManager->RegisterTool( new KICAD_MANAGER_CONTROL );
 138     m_toolManager->InitTools();
 139 
 140     RecreateBaseHToolbar();
 141     RecreateLauncher();
 142     ReCreateMenuBar();
 143 
 144     m_auimgr.SetManagedWindow( this );
 145 
 146     m_auimgr.AddPane( m_mainToolBar, EDA_PANE().HToolbar().Name( "MainToolbar" ).Top().Layer(6) );
 147 
 148     // BestSize() does not always set the actual pane size of m_leftWin to the required value.

 149     // It happens when m_leftWin is too large (roughly > 1/3 of the kicad manager frame width.

 150     // (Well, BestSize() sets the best size... not the window size)

 151     // A trick is to use MinSize() to set the required pane width,

 152     // and after give a reasonable MinSize value

 153     m_auimgr.AddPane( m_leftWin, EDA_PANE().Palette().Name( "ProjectTree" ).Left().Layer(3)
 154                       .CaptionVisible( false ).PaneBorder( false )
 155                       .MinSize( m_leftWinWidth, -1 ).BestSize( m_leftWinWidth, -1 ) );
 156 
 157     m_auimgr.AddPane( m_launcher, EDA_PANE().HToolbar().Name( "Launcher" ).Top().Layer(1) );
 158 
 159     m_auimgr.AddPane( m_messagesBox, EDA_PANE().Messages().Name( "MsgPanel" ).Center() );
 160 
 161     m_auimgr.Update();
 162 
 163     // Now the actual m_leftWin size is set, give it a reasonable min width

 164     m_auimgr.GetPane( m_leftWin ).MinSize( 200, -1 );
 165 
 166     SetTitle( wxString( "KiCad " ) + GetBuildVersion() );
 167 
 168     // Do not let the messages window have initial focus

 169     m_leftWin->SetFocus();
 170 
 171     // Ensure the window is on top

 172     Raise();
 173 }

 297 void KICAD_MANAGER_FRAME::OnCloseWindow( wxCloseEvent& Event )
 298 {
 299 #ifdef _WINDOWS_
 300     // For some obscure reason, on Windows, when killing Kicad from the Windows task manager

 301     // if a editor frame (schematic, library, board editor or fp editor) is open and has

 302     // some edition to save, OnCloseWindow is run twice *at the same time*, creating race

 303     // conditions between OnCloseWindow() code.

 304     // Therefore I added (JPC) a ugly hack to discard the second call (unwanted) during

 305     // execution of the first call (only one call is right).

 306     // Note also if there is no change made in editors, this behavior does not happen.

 307     static std::atomic<unsigned int> lock_close_event( 0 );
 308 
 309     if( ++lock_close_event > 1 )    // Skip extra calls

 310     {
 311         return;
 312     }
 313 #endif
 314 
 315     if( Kiway().PlayersClose( false ) )
 316     {
 317         Event.SetCanVeto( true );
 318 
 319         m_leftWin->Show( false );
 320 
 321         Destroy();
 322     }
 323 
 324 #ifdef _WINDOWS_
 325     lock_close_event = 0;   // Reenable event management

 326 #endif
 327 }

 336 void KICAD_MANAGER_FRAME::LoadProject( const wxFileName& aProjectFileName )
 337 {
 338     // The project file should be valid by the time we get here or something has gone wrong.

 339     if( !aProjectFileName.Exists() )
 340         return;
 341 
 342     // Any open KIFACE's must be closed if they are not part of the new project.

 343     // (We never want a KIWAY_PLAYER open on a KIWAY that isn't in the same project.)

 344     // User is prompted here to close those KIWAY_PLAYERs:

 345     if( !Kiway().PlayersClose( false ) )
 346         return;
 347 
 348     // Save the project file for the currently loaded project.

 349     if( m_active_project )
 350     {
 351         SETTINGS_MANAGER& mgr = Pgm().GetSettingsManager();
 352 
 353         mgr.TriggerBackupIfNeeded( NULL_REPORTER::GetInstance() );
 354         mgr.SaveProject();
 355         mgr.UnloadProject( &Prj() );
 356     }
 357 
 358     m_active_project = true;
 359     ClearMsg();
 360 
 361     Pgm().GetSettingsManager().LoadProject( aProjectFileName.GetFullPath() );
 362     SetProjectFileName( Prj().GetProjectFullName() );
 363 
 364     if( aProjectFileName.IsDirWritable() )
 365         SetMruPath( Prj().GetProjectPath() ); // Only set MRU path if we have write access. Why?

 366 
 367     UpdateFileHistory( Prj().GetProjectFullName() );
 368 
 369     m_leftWin->ReCreateTreePrj();
 370 
 371     SyncToolbars();
 372 
 373     // Rebuild the list of watched paths.

 374     // however this is possible only when the main loop event handler is running,

 375     // so we use it to run the rebuild function.

 376     wxCommandEvent cmd( wxEVT_COMMAND_MENU_SELECTED, ID_INIT_WATCHED_PATHS );
 377 
 378     wxPostEvent( this, cmd );
 379 
 380     PrintPrjInfo();
 381 }

 384 void KICAD_MANAGER_FRAME::CreateNewProject( const wxFileName& aProjectFileName )
 385 {
 386     wxCHECK_RET( aProjectFileName.DirExists() && aProjectFileName.IsDirWritable(),
 387                  "Project folder must exist and be writable to create a new project." );
 388 
 389     // Init project filename.  This clears all elements from the project object.

 390     SetProjectFileName( aProjectFileName.GetFullPath() );
 391 
 392     // If the project is legacy, convert it

 393     if( !aProjectFileName.FileExists() )
 394     {
 395         wxFileName legacyPro( aProjectFileName );
 396         legacyPro.SetExt( LegacyProjectFileExtension );
 397 
 398         if( legacyPro.FileExists() )
 399         {
 400             GetSettingsManager()->LoadProject( legacyPro.GetFullPath() );
 401             GetSettingsManager()->SaveProject();
 402 
 403             wxRemoveFile( legacyPro.GetFullPath() );
 404         }
 405         else
 406         {
 407             // Copy template project file from template folder.

 408             wxString srcFileName = sys_search().FindValidPath( "kicad.kicad_pro" );
 409 
 410             wxFileName destFileName( aProjectFileName );
 411             destFileName.SetExt( ProjectFileExtension );
 412 
 413             // Create a minimal project file if the template project file could not be copied

 414             if( !wxFileName::FileExists( srcFileName )
 415                 || !wxCopyFile( srcFileName, destFileName.GetFullPath() ) )
 416             {
 417                 wxFile file( destFileName.GetFullPath(), wxFile::write );
 418 
 419                 if( file.IsOpened() )
 420                     file.Write( wxT( "{\n}\n") );
 421                 // wxFile dtor will close the file

 422             }
 423         }
 424     }
 425 
 426     // Ensure a "stub" for a schematic root sheet and a board exist.

 427     // It will avoid messages from the schematic editor or the board editor to create a new file

 428     // And forces the user to create main files under the right name for the project manager

 429     wxFileName fn( aProjectFileName.GetFullPath() );
 430     fn.SetExt( KiCadSchematicFileExtension );
 431 
 432     // If a <project>.kicad_sch file does not exist, create a "stub" file ( minimal schematic file )

 433     if( !fn.FileExists() )
 434     {
 435         wxFile file( fn.GetFullPath(), wxFile::write );
 436 
 437         if( file.IsOpened() )
 438             file.Write( wxT( "(kicad_sch (version 20200310) (host eeschema \"unknown\")\n"
 439                              "(  page \"A4\")\n  (lib_symbols)\n"
 440                              "  (symbol_instances)\n)\n" ) );
 441 
 442 
 443         // wxFile dtor will close the file

 444     }
 445 
 446     // If a <project>.kicad_pcb or <project>.brd file does not exist,

 447     // create a .kicad_pcb "stub" file

 448     fn.SetExt( KiCadPcbFileExtension );
 449     wxFileName leg_fn( fn );
 450     leg_fn.SetExt( LegacyPcbFileExtension );
 451 
 452     if( !fn.FileExists() && !leg_fn.FileExists() )
 453     {
 454         wxFile file( fn.GetFullPath(), wxFile::write );
 455 
 456         if( file.IsOpened() )
 457             file.Write( wxT( "(kicad_pcb (version 4) (host kicad \"dummy file\") )\n" ) );
 458 
 459         // wxFile dtor will close the file

 460     }
 461 
 462     UpdateFileHistory( aProjectFileName.GetFullPath() );
 463 }

 466 void KICAD_MANAGER_FRAME::OnOpenFileInTextEditor( wxCommandEvent& event )
 467 {
 468     // show all files in file dialog (in Kicad all files are editable texts):

 469     wxString wildcard = AllFilesWildcard();
 470 
 471     wxString default_dir = Prj().GetProjectPath();
 472 
 473     wxFileDialog dlg( this, _( "Load File to Edit" ), default_dir,
 474                       wxEmptyString, wildcard, wxFD_OPEN );
 475 
 476     if( dlg.ShowModal() == wxID_CANCEL )
 477         return;
 478 
 479     wxString filename = wxT( "\"" );
 480     filename += dlg.GetPath() + wxT( "\"" );
 481 
 482     if( !dlg.GetPath().IsEmpty() &&  !Pgm().GetEditorName().IsEmpty() )
 483         m_toolManager->RunAction( KICAD_MANAGER_ACTIONS::openTextEditor, true, &filename );
 484 }
